1וַיְהִ֗י כְּכַלֹּתוֹ֙ לְדַבֵּ֣ר אֶל־ שָׁא֔וּל וְנֶ֙פֶשׁ֙ יְה֣וֹנָתָ֔ן נִקְשְׁרָ֖ה בְּנֶ֣פֶשׁ דָּוִ֑ד [וַיֶּאֱהָבֹו כ] (וַיֶּאֱהָבֵ֥הוּ ק) יְהוֹנָתָ֖ן כְּנַפְשֽׁוֹ׃
2שָׁא֖וּל בַּיּ֣וֹם הַה֑וּא וְלֹ֣א נְתָנ֔וֹ לָשׁ֖וּב בֵּ֥ית אָבִֽיו׃
3יְהוֹנָתָ֛ן וְדָוִ֖ד בְּרִ֑ית בְּאַהֲבָת֥וֹ אֹת֖וֹ כְּנַפְשֽׁוֹ׃
4יְהוֹנָתָ֗ן אֶֽת־ הַמְּעִיל֙ אֲשֶׁ֣ר עָלָ֔יו וַֽיִּתְּנֵ֖הוּ לְדָוִ֑ד וּמַדָּ֕יו וְעַד־ חַרְבּ֥וֹ וְעַד־ קַשְׁתּ֖וֹ וְעַד־ חֲגֹרֽוֹ׃
5דָוִ֜ד בְּכֹל֩ אֲשֶׁ֨ר יִשְׁלָחֶ֤נּוּ שָׁאוּל֙ יַשְׂכִּ֔יל וַיְשִׂמֵ֣הוּ שָׁא֔וּל עַ֖ל אַנְשֵׁ֣י הַמִּלְחָמָ֑ה וַיִּיטַב֙ בְּעֵינֵ֣י כָל־ הָעָ֔ם וְגַ֕ם בְּעֵינֵ֖י עַבְדֵ֥י שָׁאֽוּל׃ פ
6בְּבוֹאָ֗ם בְּשׁ֤וּב דָּוִד֙ מֵהַכּ֣וֹת אֶת־ הַפְּלִשְׁתִּ֔י וַתֵּצֶ֨אנָה הַנָּשִׁ֜ים מִכָּל־ עָרֵ֤י יִשְׂרָאֵל֙ [לָשֹׁור כ] (לָשִׁ֣יר ק) וְהַמְּחֹל֔וֹת לִקְרַ֖את שָׁא֣וּל הַמֶּ֑לֶךְ בְּתֻפִּ֥ים בְּשִׂמְחָ֖ה וּבְשָׁלִשִֽׁים׃
7הַנָּשִׁ֥ים הַֽמְשַׂחֲק֖וֹת וַתֹּאמַ֑רְןָ הִכָּ֤ה שָׁאוּל֙ [בַּאֱלְפֹו כ] (בַּאֲלָפָ֔יו ק) וְדָוִ֖ד בְּרִבְבֹתָֽיו׃
8לְשָׁא֜וּל מְאֹ֗ד וַיֵּ֤רַע בְּעֵינָיו֙ הַדָּבָ֣ר הַזֶּ֔ה וַיֹּ֗אמֶר נָתְנ֤וּ לְדָוִד֙ רְבָב֔וֹת וְלִ֥י נָתְנ֖וּ הָאֲלָפִ֑ים וְע֥וֹד ל֖וֹ אַ֥ךְ הַמְּלוּכָֽה׃
9שָׁא֖וּל [עָוֹן כ] (עֹויֵ֣ן ק) אֶת־ דָּוִ֑ד מֵהַיּ֥וֹם הַה֖וּא וָהָֽלְאָה׃ ס
10מִֽמָּחֳרָ֗ת וַתִּצְלַ֣ח רוּחַ֩ אֱלֹהִ֨ים ׀ רָעָ֤ה ׀ אֶל־ שָׁאוּל֙ וַיִּתְנַבֵּ֣א בְתוֹךְ־ הַבַּ֔יִת וְדָוִ֛ד מְנַגֵּ֥ן בְּיָד֖וֹ כְּי֣וֹם ׀ בְּי֑וֹם וְהַחֲנִ֖ית בְּיַד־ שָׁאֽוּל׃
11שָׁאוּל֙ אֶֽת־ הַחֲנִ֔ית וַיֹּ֕אמֶר אַכֶּ֥ה בְדָוִ֖ד וּבַקִּ֑יר וַיִּסֹּ֥ב דָּוִ֛ד מִפָּנָ֖יו פַּעֲמָֽיִם׃
12שָׁא֖וּל מִלִּפְנֵ֣י דָוִ֑ד כִּֽי־ הָיָ֤ה יְהוָה֙ עִמּ֔וֹ וּמֵעִ֥ם שָׁא֖וּל סָֽר׃
13שָׁאוּל֙ מֵֽעִמּ֔וֹ וַיְשִׂמֵ֥הוּ ל֖וֹ שַׂר־ אָ֑לֶף וַיֵּצֵ֥א וַיָּבֹ֖א לִפְנֵ֥י הָעָֽם׃ פ
14דָוִ֛ד לְכָל־ דָּרְכָ֖ו מַשְׂכִּ֑יל וַֽיהוָ֖ה עִמּֽוֹ׃
15שָׁא֔וּל אֲשֶׁר־ ה֖וּא מַשְׂכִּ֣יל מְאֹ֑ד וַיָּ֖גָר מִפָּנָֽיו׃
16יִשְׂרָאֵל֙ וִיהוּדָ֔ה אֹהֵ֖ב אֶת־ דָּוִ֑ד כִּֽי־ ה֛וּא יוֹצֵ֥א וָבָ֖א לִפְנֵיהֶֽם׃ פ
17שָׁא֜וּל אֶל־ דָּוִ֗ד הִנֵּה֩ בִתִּ֨י הַגְּדוֹלָ֤ה מֵרַב֙ אֹתָהּ֙ אֶתֶּן־ לְךָ֣ לְאִשָּׁ֔ה אַ֚ךְ הֱיֵה־ לִּ֣י לְבֶן־ חַ֔יִל וְהִלָּחֵ֖ם מִלְחֲמ֣וֹת יְהוָ֑ה וְשָׁא֣וּל אָמַ֗ר אַל־ תְּהִ֤י יָדִי֙ בּ֔וֹ וּתְהִי־ ב֖וֹ יַד־ פְּלִשְׁתִּֽים׃ ס
18דָּוִ֜ד אֶל־ שָׁא֗וּל מִ֤י אָֽנֹכִי֙ וּמִ֣י חַיַּ֔י מִשְׁפַּ֥חַת אָבִ֖י בְּיִשְׂרָאֵ֑ל כִּֽי־ אֶהְיֶ֥ה חָתָ֖ן לַמֶּֽלֶךְ׃
19בְּעֵ֥ת תֵּ֛ת אֶת־ מֵרַ֥ב בַּת־ שָׁא֖וּל לְדָוִ֑ד וְהִ֧יא נִתְּנָ֛ה לְעַדְרִיאֵ֥ל הַמְּחֹלָתִ֖י לְאִשָּֽׁה׃
20מִיכַ֥ל בַּת־ שָׁא֖וּל אֶת־ דָּוִ֑ד וַיַּגִּ֣דוּ לְשָׁא֔וּל וַיִּשַׁ֥ר הַדָּבָ֖ר בְּעֵינָֽיו׃
21שָׁא֜וּל אֶתְּנֶ֤נָּה לּוֹ֙ וּתְהִי־ ל֣וֹ לְמוֹקֵ֔שׁ וּתְהִי־ ב֖וֹ יַד־ פְּלִשְׁתִּ֑ים וַיֹּ֤אמֶר שָׁאוּל֙ אֶל־ דָּוִ֔ד בִּשְׁתַּ֛יִם תִּתְחַתֵּ֥ן בִּ֖י הַיּֽוֹם׃
22שָׁא֜וּל אֶת־ עֲבָדָ֗ו דַּבְּר֨וּ אֶל־ דָּוִ֤ד בַּלָּט֙ לֵאמֹ֔ר הִנֵּ֨ה חָפֵ֤ץ בְּךָ֙ הַמֶּ֔לֶךְ וְכָל־ עֲבָדָ֖יו אֲהֵב֑וּךָ וְעַתָּ֖ה הִתְחַתֵּ֥ן בַּמֶּֽלֶךְ׃
23עַבְדֵ֤י שָׁאוּל֙ בְּאָזְנֵ֣י דָוִ֔ד אֶת־ הַדְּבָרִ֖ים הָאֵ֑לֶּה וַיֹּ֣אמֶר דָּוִ֗ד הַֽנְקַלָּ֤ה בְעֵֽינֵיכֶם֙ הִתְחַתֵּ֣ן בַּמֶּ֔לֶךְ וְאָנֹכִ֖י אִֽישׁ־ רָ֥שׁ וְנִקְלֶֽה׃
24עַבְדֵ֥י שָׁא֛וּל ל֖וֹ לֵאמֹ֑ר כַּדְּבָרִ֥ים הָאֵ֖לֶּה דִּבֶּ֥ר דָּוִֽד׃ פ
25שָׁא֜וּל כֹּֽה־ תֹאמְר֣וּ לְדָוִ֗ד אֵֽין־ חֵ֤פֶץ לַמֶּ֙לֶךְ֙ בְּמֹ֔הַר כִּ֗י בְּמֵאָה֙ עָרְל֣וֹת פְּלִשְׁתִּ֔ים לְהִנָּקֵ֖ם בְּאֹיְבֵ֣י הַמֶּ֑לֶךְ וְשָׁא֣וּל חָשַׁ֔ב לְהַפִּ֥יל אֶת־ דָּוִ֖ד בְּיַד־ פְּלִשְׁתִּֽים׃
26עֲבָדָ֤יו לְדָוִד֙ אֶת־ הַדְּבָרִ֣ים הָאֵ֔לֶּה וַיִּשַׁ֤ר הַדָּבָר֙ בְּעֵינֵ֣י דָוִ֔ד לְהִתְחַתֵּ֖ן בַּמֶּ֑לֶךְ וְלֹ֥א מָלְא֖וּ הַיָּמִֽים׃
27דָּוִ֜ד וַיֵּ֣לֶךְ ׀ ה֣וּא וַאֲנָשָׁ֗יו וַיַּ֣ךְ בַּפְּלִשְׁתִּים֮ מָאתַ֣יִם אִישׁ֒ וַיָּבֵ֤א דָוִד֙ אֶת־ עָרְלֹ֣תֵיהֶ֔ם וַיְמַלְא֣וּם לַמֶּ֔לֶךְ לְהִתְחַתֵּ֖ן בַּמֶּ֑לֶךְ וַיִּתֶּן־ ל֥וֹ שָׁא֛וּל אֶת־ מִיכַ֥ל בִּתּ֖וֹ לְאִשָּֽׁה׃ ס
28שָׁאוּל֙ וַיֵּ֔דַע כִּ֥י יְהוָ֖ה עִם־ דָּוִ֑ד וּמִיכַ֥ל בַּת־ שָׁא֖וּל אֲהֵבַֽתְהוּ׃
29שָׁא֗וּל לֵרֹ֛א מִפְּנֵ֥י דָוִ֖ד ע֑וֹד וַיְהִ֥י שָׁא֛וּל אֹיֵ֥ב אֶת־ דָּוִ֖ד כָּל־ הַיָּמִֽים׃ ס
