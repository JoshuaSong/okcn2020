1שִׁמְע֞וּ אֶת־ הַדָּבָ֣ר הַזֶּ֗ה אֲשֶׁ֨ר אָנֹכִ֜י נֹשֵׂ֧א עֲלֵיכֶ֛ם קִינָ֖ה בֵּ֥ית יִשְׂרָאֵֽל׃
2לֹֽא־ תוֹסִ֣יף ק֔וּם בְּתוּלַ֖ת יִשְׂרָאֵ֑ל נִטְּשָׁ֥ה עַל־ אַדְמָתָ֖הּ אֵ֥ין מְקִימָֽהּ׃
3כֹ֤ה אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה הָעִ֛יר הַיֹּצֵ֥את אֶ֖לֶף תַּשְׁאִ֣יר מֵאָ֑ה וְהַיּוֹצֵ֥את מֵאָ֛ה תַּשְׁאִ֥יר עֲשָׂרָ֖ה לְבֵ֥ית יִשְׂרָאֵֽל׃ ס
4כֹ֥ה אָמַ֛ר יְהוָ֖ה לְבֵ֣ית יִשְׂרָאֵ֑ל דִּרְשׁ֖וּנִי וִֽחְיֽוּ׃
5תִּדְרְשׁוּ֙ בֵּֽית־ אֵ֔ל וְהַגִּלְגָּל֙ לֹ֣א תָבֹ֔אוּ וּבְאֵ֥ר שֶׁ֖בַע לֹ֣א תַעֲבֹ֑רוּ כִּ֤י הַגִּלְגָּל֙ גָּלֹ֣ה יִגְלֶ֔ה וּבֵֽית־ אֵ֖ל יִהְיֶ֥ה לְאָֽוֶן׃
6אֶת־ יְהוָ֖ה וִֽחְי֑וּ פֶּן־ יִצְלַ֤ח כָּאֵשׁ֙ בֵּ֣ית יוֹסֵ֔ף וְאָכְלָ֥ה וְאֵין־ מְכַבֶּ֖ה לְבֵֽית־ אֵֽל׃
7לְלַעֲנָ֖ה מִשְׁפָּ֑ט וּצְדָקָ֖ה לָאָ֥רֶץ הִנִּֽיחוּ׃
8כִימָ֜ה וּכְסִ֗יל וְהֹפֵ֤ךְ לַבֹּ֙קֶר֙ צַלְמָ֔וֶת וְי֖וֹם לַ֣יְלָה הֶחְשִׁ֑יךְ הַקּוֹרֵ֣א לְמֵֽי־ הַיָּ֗ם וַֽיִּשְׁפְּכֵ֛ם עַל־ פְּנֵ֥י הָאָ֖רֶץ יְהוָ֥ה שְׁמֽוֹ׃ ס
9שֹׁ֖ד עַל־ עָ֑ז וְשֹׁ֖ד עַל־ מִבְצָ֥ר יָבֽוֹא׃
10בַשַּׁ֖עַר מוֹכִ֑יחַ וְדֹבֵ֥ר תָּמִ֖ים יְתָעֵֽבוּ׃
11יַ֣עַן בּוֹשַׁסְכֶ֞ם עַל־ דָּ֗ל וּמַשְׂאַת־ בַּר֙ תִּקְח֣וּ מִמֶּ֔נּוּ בָּתֵּ֥י גָזִ֛ית בְּנִיתֶ֖ם וְלֹא־ תֵ֣שְׁבוּ בָ֑ם כַּרְמֵי־ חֶ֣מֶד נְטַעְתֶּ֔ם וְלֹ֥א תִשְׁתּ֖וּ אֶת־ יֵינָֽם׃
12יָדַ֙עְתִּי֙ רַבִּ֣ים פִּשְׁעֵיכֶ֔ם וַעֲצֻמִ֖ים חַטֹּֽאתֵיכֶ֑ם צֹרְרֵ֤י צַדִּיק֙ לֹ֣קְחֵי כֹ֔פֶר וְאֶבְיוֹנִ֖ים בַּשַּׁ֥עַר הִטּֽוּ׃
13הַמַּשְׂכִּ֛יל בָּעֵ֥ת הַהִ֖יא יִדֹּ֑ם כִּ֛י עֵ֥ת רָעָ֖ה הִֽיא׃
14ט֥וֹב וְאַל־ רָ֖ע לְמַ֣עַן תִּֽחְי֑וּ וִיהִי־ כֵ֞ן יְהוָ֧ה אֱלֹהֵֽי־ צְבָא֛וֹת אִתְּכֶ֖ם כַּאֲשֶׁ֥ר אֲמַרְתֶּֽם׃
15רָע֙ וְאֶ֣הֱבוּ ט֔וֹב וְהַצִּ֥יגוּ בַשַּׁ֖עַר מִשְׁפָּ֑ט אוּלַ֗י יֶֽחֱנַ֛ן יְהוָ֥ה אֱלֹהֵֽי־ צְבָא֖וֹת שְׁאֵרִ֥ית יוֹסֵֽף׃ ס
16כֹּֽה־ אָמַ֨ר יְהוָ֜ה אֱלֹהֵ֤י צְבָאוֹת֙ אֲדֹנָ֔י בְּכָל־ רְחֹב֣וֹת מִסְפֵּ֔ד וּבְכָל־ חוּצ֖וֹת יֹאמְר֣וּ הוֹ־ ה֑וֹ וְקָרְא֤וּ אִכָּר֙ אֶל־ אֵ֔בֶל וּמִסְפֵּ֖ד אֶל־ י֥וֹדְעֵי נֶֽהִי׃
17כְּרָמִ֖ים מִסְפֵּ֑ד כִּֽי־ אֶעֱבֹ֥ר בְּקִרְבְּךָ֖ אָמַ֥ר יְהוָֽה׃ ס
18הַמִּתְאַוִּ֖ים אֶת־ י֣וֹם יְהוָ֑ה לָמָּה־ זֶּ֥ה לָכֶ֛ם י֥וֹם יְהוָ֖ה הוּא־ חֹ֥שֶׁךְ וְלֹא־ אֽוֹר׃
19יָנ֥וּס אִישׁ֙ מִפְּנֵ֣י הָאֲרִ֔י וּפְגָע֖וֹ הַדֹּ֑ב וּבָ֣א הַבַּ֔יִת וְסָמַ֤ךְ יָדוֹ֙ עַל־ הַקִּ֔יר וּנְשָׁכ֖וֹ הַנָּחָֽשׁ׃
20חֹ֛שֶׁךְ י֥וֹם יְהוָ֖ה וְלֹא־ א֑וֹר וְאָפֵ֖ל וְלֹא־ נֹ֥גַֽהּ לֽוֹ׃
21מָאַ֖סְתִּי חַגֵּיכֶ֑ם וְלֹ֥א אָרִ֖יחַ בְּעַצְּרֹֽתֵיכֶֽם׃
22אִם־ תַּעֲלוּ־ לִ֥י עֹל֛וֹת וּמִנְחֹתֵיכֶ֖ם לֹ֣א אֶרְצֶ֑ה וְשֶׁ֥לֶם מְרִיאֵיכֶ֖ם לֹ֥א אַבִּֽיט׃
23מֵעָלַ֖י הֲמ֣וֹן שִׁרֶ֑יךָ וְזִמְרַ֥ת נְבָלֶ֖יךָ לֹ֥א אֶשְׁמָֽע׃
24כַּמַּ֖יִם מִשְׁפָּ֑ט וּצְדָקָ֖ה כְּנַ֥חַל אֵיתָֽן׃
25וּמִנְחָ֜ה הִֽגַּשְׁתֶּם־ לִ֧י בַמִּדְבָּ֛ר אַרְבָּעִ֥ים שָׁנָ֖ה בֵּ֥ית יִשְׂרָאֵֽל׃
26אֵ֚ת סִכּ֣וּת מַלְכְּכֶ֔ם וְאֵ֖ת כִּיּ֣וּן צַלְמֵיכֶ֑ם כּוֹכַב֙ אֱלֹ֣הֵיכֶ֔ם אֲשֶׁ֥ר עֲשִׂיתֶ֖ם לָכֶֽם׃
