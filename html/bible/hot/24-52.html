1בֶּן־ עֶשְׂרִ֨ים וְאַחַ֤ת שָׁנָה֙ צִדְקִיָּ֣הוּ בְמָלְכ֔וֹ וְאַחַ֤ת עֶשְׂרֵה֙ שָׁנָ֔ה מָלַ֖ךְ בִּירֽוּשָׁלִָ֑ם וְשֵׁ֣ם אִמּ֔וֹ [חֲמִיטַל כ] (חֲמוּטַ֥ל ק) בַּֽת־ יִרְמְיָ֖הוּ מִלִּבְנָֽה׃
2הָרַ֖ע בְּעֵינֵ֣י יְהוָ֑ה כְּכֹ֥ל אֲשֶׁר־ עָשָׂ֖ה יְהוֹיָקִֽים׃
3עַל־ אַ֣ף יְהוָ֗ה הָֽיְתָה֙ בִּירוּשָׁלִַ֣ם וִֽיהוּדָ֔ה עַד־ הִשְׁלִיכ֥וֹ אוֹתָ֖ם מֵעַ֣ל פָּנָ֑יו וַיִּמְרֹ֥ד צִדְקִיָּ֖הוּ בְּמֶ֥לֶךְ בָּבֶֽל׃
4בַשָּׁנָ֨ה הַתְּשִׁעִ֜ית לְמָלְכ֗וֹ בַּחֹ֣דֶשׁ הָעֲשִׂירִי֮ בֶּעָשׂ֣וֹר לַחֹדֶשׁ֒ בָּ֠א נְבוּכַדְרֶאצַּ֨ר מֶֽלֶךְ־ בָּבֶ֜ל ה֤וּא וְכָל־ חֵילוֹ֙ עַל־ יְר֣וּשָׁלִַ֔ם וַֽיַּחֲנ֖וּ עָלֶ֑יהָ וַיִּבְנ֥וּ עָלֶ֛יהָ דָּיֵ֖ק סָבִֽיב׃
5הָעִ֖יר בַּמָּצ֑וֹר עַ֚ד עַשְׁתֵּ֣י עֶשְׂרֵ֣ה שָׁנָ֔ה לַמֶּ֖לֶךְ צִדְקִיָּֽהוּ׃
6הָֽרְבִיעִי֙ בְּתִשְׁעָ֣ה לַחֹ֔דֶשׁ וַיֶּחֱזַ֥ק הָרָעָ֖ב בָּעִ֑יר וְלֹא־ הָיָ֥ה לֶ֖חֶם לְעַ֥ם הָאָֽרֶץ׃
7הָעִ֗יר וְכָל־ אַנְשֵׁ֣י הַמִּלְחָמָ֡ה יִבְרְחוּ֩ וַיֵּצְא֨וּ מֵהָעִ֜יר לַ֗יְלָה דֶּ֜רֶךְ שַׁ֤עַר בֵּין־ הַחֹמֹתַ֙יִם֙ אֲשֶׁר֙ עַל־ גַּ֣ן הַמֶּ֔לֶךְ וְכַשְׂדִּ֥ים עַל־ הָעִ֖יר סָבִ֑יב וַיֵּלְכ֖וּ דֶּ֥רֶךְ הָעֲרָבָֽה׃
8חֵיל־ כַּשְׂדִּים֙ אַחֲרֵ֣י הַמֶּ֔לֶךְ וַיַּשִּׂ֥יגוּ אֶת־ צִדְקִיָּ֖הוּ בְּעַֽרְבֹ֣ת יְרֵח֑וֹ וְכָל־ חֵיל֔וֹ נָפֹ֖צוּ מֵעָלָֽיו׃
9אֶת־ הַמֶּ֔לֶךְ וַיַּעֲל֨וּ אֹת֜וֹ אֶל־ מֶ֧לֶךְ בָּבֶ֛ל רִבְלָ֖תָה בְּאֶ֣רֶץ חֲמָ֑ת וַיְדַבֵּ֥ר אִתּ֖וֹ מִשְׁפָּטִֽים׃
10מֶֽלֶךְ־ בָּבֶ֛ל אֶת־ בְּנֵ֥י צִדְקִיָּ֖הוּ לְעֵינָ֑יו וְגַ֛ם אֶת־ כָּל־ שָׂרֵ֥י יְהוּדָ֖ה שָׁחַ֥ט בְּרִבְלָֽתָה׃
11עֵינֵ֥י צִדְקִיָּ֖הוּ עִוֵּ֑ר וַיַּאַסְרֵ֣הוּ בַֽנְחֻשְׁתַּ֗יִם וַיְבִאֵ֤הוּ מֶֽלֶךְ־ בָּבֶל֙ בָּבֶ֔לָה וַיִּתְּנֵ֥הוּ [בְבֵית־ כ] (בֵֽית־ הַפְּקֻדֹּ֖ת ק) עַד־ י֥וֹם מוֹתֽוֹ׃
12הַֽחֲמִישִׁי֙ בֶּעָשׂ֣וֹר לַחֹ֔דֶשׁ הִ֗יא שְׁנַת֙ תְּשַֽׁע־ עֶשְׂרֵ֣ה שָׁנָ֔ה לַמֶּ֖לֶךְ נְבוּכַדְרֶאצַּ֣ר מֶֽלֶךְ־ בָּבֶ֑ל בָּ֗א נְבֽוּזַרְאֲדָן֙ רַב־ טַבָּחִ֔ים עָמַ֛ד לִפְנֵ֥י מֶֽלֶךְ־ בָּבֶ֖ל בִּירוּשָׁלִָֽם׃
13אֶת־ בֵּית־ יְהוָ֖ה וְאֶת־ בֵּ֣ית הַמֶּ֑לֶךְ וְאֵ֨ת כָּל־ בָּתֵּ֧י יְרוּשָׁלִַ֛ם וְאֶת־ כָּל־ בֵּ֥ית הַגָּד֖וֹל שָׂרַ֥ף בָּאֵֽשׁ׃ וְאֶת־ כָּל־
14יְרוּשָׁלִַ֖ם סָבִ֑יב נָֽתְצוּ֙ כָּל־ חֵ֣יל כַּשְׂדִּ֔ים אֲשֶׁ֖ר אֶת־ רַב־ טַבָּחִֽים׃
15הָעָ֜ם וְֽאֶת־ יֶ֥תֶר הָעָ֣ם ׀ הַנִּשְׁאָרִ֣ים בָּעִ֗יר וְאֶת־ הַנֹּֽפְלִים֙ אֲשֶׁ֤ר נָֽפְלוּ֙ אֶל־ מֶ֣לֶךְ בָּבֶ֔ל וְאֵ֖ת יֶ֣תֶר הָֽאָמ֑וֹן הֶגְלָ֕ה נְבוּזַרְאֲדָ֖ן רַב־ טַבָּחִֽים׃
16הָאָ֔רֶץ הִשְׁאִ֕יר נְבוּזַרְאֲדָ֖ן רַב־ טַבָּחִ֑ים לְכֹרְמִ֖ים וּלְיֹגְבִֽים׃
17עַמּוּדֵ֨י הַנְּחֹ֜שֶׁת אֲשֶׁ֣ר לְבֵית־ יְהוָ֗ה וְֽאֶת־ הַמְּכֹנ֞וֹת וְאֶת־ יָ֧ם הַנְּחֹ֛שֶׁת אֲשֶׁ֥ר בְּבֵית־ יְהוָ֖ה שִׁבְּר֣וּ כַשְׂדִּ֑ים וַיִּשְׂא֥וּ אֶת־ כָּל־ נְחֻשְׁתָּ֖ם בָּבֶֽלָה׃
18הַ֠סִּרוֹת וְאֶת־ הַיָּעִ֨ים וְאֶת־ הַֽמְזַמְּר֜וֹת וְאֶת־ הַמִּזְרָקֹ֣ת וְאֶת־ הַכַּפּ֗וֹת וְאֵ֨ת כָּל־ כְּלֵ֧י הַנְּחֹ֛שֶׁת אֲשֶׁר־ יְשָׁרְת֥וּ בָהֶ֖ם לָקָֽחוּ׃
19הַ֠סִּפִּים וְאֶת־ הַמַּחְתּ֨וֹת וְאֶת־ הַמִּזְרָק֜וֹת וְאֶת־ הַסִּיר֣וֹת וְאֶת־ הַמְּנֹר֗וֹת וְאֶת־ הַכַּפּוֹת֙ וְאֶת־ הַמְּנַקִי֔וֹת אֲשֶׁ֤ר זָהָב֙ זָהָ֔ב וַאֲשֶׁר־ כֶּ֖סֶף כָּ֑סֶף לָקַ֖ח רַב־ טַבָּחִֽים׃
20שְׁנַ֗יִם הַיָּ֤ם אֶחָד֙ וְהַבָּקָ֞ר שְׁנֵים־ עָשָׂ֤ר נְחֹ֙שֶׁת֙ אֲשֶׁר־ תַּ֣חַת הַמְּכֹנ֔וֹת אֲשֶׁ֥ר עָשָׂ֛ה הַמֶּ֥לֶךְ שְׁלֹמֹ֖ה לְבֵ֣ית יְהוָ֑ה לֹא־ הָיָ֣ה מִשְׁקָ֔ל לִנְחֻשְׁתָּ֖ם כָּל־ הַכֵּלִ֥ים הָאֵֽלֶּה׃
21שְׁמֹנֶ֨ה עֶשְׂרֵ֤ה אַמָּה֙ [קֹומָה כ] (קֹומַת֙ ק) הָעַמֻּ֣ד הָאֶחָ֔ד וְח֛וּט שְׁתֵּים־ עֶשְׂרֵ֥ה אַמָּ֖ה יְסֻבֶּ֑נּוּ וְעָבְי֛וֹ אַרְבַּ֥ע אַצְבָּע֖וֹת נָבֽוּב׃
22עָלָ֜יו נְחֹ֗שֶׁת וְקוֹמַ֨ת הַכֹּתֶ֥רֶת הָאַחַת֮ חָמֵ֣שׁ אַמּוֹת֒ וּשְׂבָכָ֨ה וְרִמּוֹנִ֧ים עַֽל־ הַכּוֹתֶ֛רֶת סָבִ֖יב הַכֹּ֣ל נְחֹ֑שֶׁת וְכָאֵ֛לֶּה לַֽעַמּ֥וּד הַשֵּׁנִ֖י וְרִמּוֹנִֽים׃
23הָֽרִמֹּנִ֔ים תִּשְׁעִ֥ים וְשִׁשָּׁ֖ה ר֑וּחָה כָּל־ הָרִמּוֹנִ֥ים מֵאָ֛ה עַל־ הַשְּׂבָכָ֖ה סָבִֽיב׃
24רַב־ טַבָּחִ֗ים אֶת־ שְׂרָיָה֙ כֹּהֵ֣ן הָרֹ֔אשׁ וְאֶת־ צְפַנְיָ֖ה כֹּהֵ֣ן הַמִּשְׁנֶ֑ה וְאֶת־ שְׁלֹ֖שֶׁת שֹׁמְרֵ֥י הַסַּֽף׃
25הָעִ֡יר לָקַח֩ סָרִ֨יס אֶחָ֜ד אֲ‍ֽשֶׁר־ הָיָ֥ה פָקִ֣יד ׀ עַל־ אַנְשֵׁ֣י הַמִּלְחָמָ֗ה וְשִׁבְעָ֨ה אֲנָשִׁ֜ים מֵרֹאֵ֤י פְנֵי־ הַמֶּ֙לֶךְ֙ אֲשֶׁ֣ר נִמְצְא֣וּ בָעִ֔יר וְאֵ֗ת סֹפֵר֙ שַׂ֣ר הַצָּבָ֔א הַמַּצְבִּ֖א אֶת־ עַ֣ם הָאָ֑רֶץ וְשִׁשִּׁ֥ים אִישׁ֙ מֵעַ֣ם הָאָ֔רֶץ הַֽנִּמְצְאִ֖ים בְּת֥וֹךְ הָעִֽיר׃
26אוֹתָ֔ם נְבוּזַרְאֲדָ֖ן רַב־ טַבָּחִ֑ים וַיֹּ֧לֶךְ אוֹתָ֛ם אֶל־ מֶ֥לֶךְ בָּבֶ֖ל רִבְלָֽתָה׃
27אוֹתָם֩ מֶ֨לֶךְ בָּבֶ֧ל וַיְמִתֵ֛ם בְּרִבְלָ֖ה בְּאֶ֣רֶץ חֲמָ֑ת וַיִּ֥גֶל יְהוּדָ֖ה מֵעַ֥ל אַדְמָתֽוֹ׃
28הָעָ֔ם אֲשֶׁ֥ר הֶגְלָ֖ה נְבֽוּכַדְרֶאצַּ֑ר בִּשְׁנַת־ שֶׁ֕בַע יְהוּדִ֕ים שְׁלֹ֥שֶׁת אֲלָפִ֖ים וְעֶשְׂרִ֥ים וּשְׁלֹשָֽׁה׃
29שְׁמוֹנֶ֥ה עֶשְׂרֵ֖ה לִנְבֽוּכַדְרֶאצַּ֑ר מִיר֣וּשָׁלִַ֔ם נֶ֕פֶשׁ שְׁמֹנֶ֥ה מֵא֖וֹת שְׁלֹשִׁ֥ים וּשְׁנָֽיִם׃
30שָׁלֹ֣שׁ וְעֶשְׂרִים֮ לִנְבֽוּכַדְרֶאצַּר֒ הֶגְלָ֗ה נְבֽוּזַרְאֲדָן֙ רַב־ טַבָּחִ֔ים יְהוּדִ֕ים נֶ֕פֶשׁ שְׁבַ֥ע מֵא֖וֹת אַרְבָּעִ֣ים וַחֲמִשָּׁ֑ה כָּל־ נֶ֕פֶשׁ אַרְבַּ֥עַת אֲלָפִ֖ים וְשֵׁ֥שׁ מֵאֽוֹת׃ פ
31בִשְׁלֹשִׁ֨ים וָשֶׁ֜בַע שָׁנָ֗ה לְגָלוּת֙ יְהוֹיָכִ֣ן מֶֽלֶךְ־ יְהוּדָ֔ה בִּשְׁנֵ֤ים עָשָׂר֙ חֹ֔דֶשׁ בְּעֶשְׂרִ֥ים וַחֲמִשָּׁ֖ה לַחֹ֑דֶשׁ נָשָׂ֡א אֱוִ֣יל מְרֹדַךְ֩ מֶ֨לֶךְ בָּבֶ֜ל בִּשְׁנַ֣ת מַלְכֻת֗וֹ אֶת־ רֹאשׁ֙ יְהוֹיָכִ֣ין מֶֽלֶךְ־ יְהוּדָ֔ה וַיֹּצֵ֥א אוֹת֖וֹ מִבֵּ֥ית [הַכְּלִיא כ] (הַכְּלֽוּא׃ ק)
32אִתּ֖וֹ טֹב֑וֹת וַיִּתֵּן֙ אֶת־ כִּסְא֔וֹ מִמַּ֗עַל לְכִסֵּ֧א [מְלָכִים כ] (הַמְּלָכִ֛ים ק) אֲשֶׁ֥ר אִתּ֖וֹ בְּבָבֶֽל׃
33אֵ֖ת בִּגְדֵ֣י כִלְא֑וֹ וְאָכַ֨ל לֶ֧חֶם לְפָנָ֛יו תָּמִ֖יד כָּל־ יְמֵ֥י חַיָּֽו׃
