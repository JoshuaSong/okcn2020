1וַ֭יַּעַן בִּלְדַּ֥ד הַשֻּׁחִ֗י וַיֹּאמַֽר׃
2אָ֤נָה ׀ תְּשִׂימ֣וּן קִנְצֵ֣י לְמִלִּ֑ין תָּ֝בִ֗ינוּ וְאַחַ֥ר נְדַבֵּֽר׃
3נֶחְשַׁ֣בְנוּ כַבְּהֵמָ֑ה נִ֝טְמִ֗ינוּ בְּעֵינֵיכֶֽם׃
4נַפְשׁ֗וֹ בְּאַ֫פּ֥וֹ הַ֭לְמַעַנְךָ תֵּעָ֣זַב אָ֑רֶץ וְיֶעְתַּק־ צ֝֗וּר מִמְּקֹמֽוֹ׃
5א֣וֹר רְשָׁעִ֣ים יִדְעָ֑ךְ וְלֹֽא־ יִ֝גַּ֗הּ שְׁבִ֣יב אִשּֽׁוֹ׃
6חָשַׁ֣ךְ בְּאָהֳל֑וֹ וְ֝נֵר֗וֹ עָלָ֥יו יִדְעָֽךְ׃
7צַעֲדֵ֣י אוֹנ֑וֹ וְֽתַשְׁלִיכֵ֥הוּ עֲצָתֽוֹ׃
8שֻׁלַּ֣ח בְּרֶ֣שֶׁת בְּרַגְלָ֑יו וְעַל־ שְׂ֝בָכָ֗ה יִתְהַלָּֽךְ׃
9בְּעָקֵ֣ב פָּ֑ח יַחֲזֵ֖ק עָלָ֣יו צַמִּֽים׃
10בָּאָ֣רֶץ חַבְל֑וֹ וּ֝מַלְכֻּדְתּ֗וֹ עֲלֵ֣י נָתִֽיב׃
11בִּֽעֲתֻ֣הוּ בַלָּה֑וֹת וֶהֱפִיצֻ֥הוּ לְרַגְלָֽיו׃
12רָעֵ֥ב אֹנ֑וֹ וְ֝אֵ֗יד נָכ֥וֹן לְצַלְעֽוֹ׃
13בַּדֵּ֣י עוֹר֑וֹ יֹאכַ֥ל בַּ֝דָּ֗יו בְּכ֣וֹר מָֽוֶת׃
14מֵ֭אָהֳלוֹ מִבְטַח֑וֹ וְ֝תַצְעִדֵ֗הוּ לְמֶ֣לֶךְ בַּלָּהֽוֹת׃
15בְּ֭אָהֳלוֹ מִבְּלִי־ ל֑וֹ יְזֹרֶ֖ה עַל־ נָוֵ֣הוּ גָפְרִֽית׃
16שָֽׁרָשָׁ֣יו יִבָ֑שׁוּ וּ֝מִמַּ֗עַל יִמַּ֥ל קְצִירֽוֹ׃
17אָ֭בַד מִנִּי־ אָ֑רֶץ וְלֹא־ שֵׁ֥ם ל֝֗וֹ עַל־ פְּנֵי־ חֽוּץ׃
18מֵא֣וֹר אֶל־ חֹ֑שֶׁךְ וּֽמִתֵּבֵ֥ל יְנִדֻּֽהוּ׃
19נִ֤ין ל֣וֹ וְלֹא־ נֶ֣כֶד בְּעַמּ֑וֹ וְאֵ֥ין שָׂ֝רִ֗יד בִּמְגוּרָֽיו׃
20י֭וֹמוֹ נָשַׁ֣מּוּ אַחֲרֹנִ֑ים וְ֝קַדְמֹנִ֗ים אָ֣חֲזוּ שָֽׂעַר׃
