1וַ֭יַּעַן אֱלִיפַ֥ז הַֽתֵּימָנִ֗י וַיֹּאמַֽר׃
2יַעֲנֶ֥ה דַֽעַת־ ר֑וּחַ וִֽימַלֵּ֖א קָדִ֣ים בִּטְנֽוֹ׃
3בְּ֭דָבָר לֹ֣א יִסְכּ֑וֹן וּ֝מִלִּ֗ים לֹא־ יוֹעִ֥יל בָּֽם׃
4אַ֭תָּה תָּפֵ֣ר יִרְאָ֑ה וְתִגְרַ֥ע שִׂ֝יחָ֗ה לִפְנֵי־ אֵֽל׃
5יְאַלֵּ֣ף עֲוֹנְךָ֣ פִ֑יךָ וְ֝תִבְחַ֗ר לְשׁ֣וֹן עֲרוּמִֽים׃
6פִ֣יךָ וְלֹא־ אָ֑נִי וּ֝שְׂפָתֶ֗יךָ יַעֲנוּ־ בָֽךְ׃
7אָ֭דָם תִּוָּלֵ֑ד וְלִפְנֵ֖י גְבָע֣וֹת חוֹלָֽלְתָּ׃
8אֱל֣וֹהַ תִּשְׁמָ֑ע וְתִגְרַ֖ע אֵלֶ֣יךָ חָכְמָֽה׃
9יָּ֭דַעְתָּ וְלֹ֣א נֵדָ֑ע תָּ֝בִ֗ין וְֽלֹא־ עִמָּ֥נוּ הֽוּא׃
10שָׂ֣ב גַּם־ יָשִׁ֣ישׁ בָּ֑נוּ כַּבִּ֖יר מֵאָבִ֣יךָ יָמִֽים׃
11מִ֭מְּךָ תַּנְחֻמ֣וֹת אֵ֑ל וְ֝דָבָ֗ר לָאַ֥ט עִמָּֽךְ׃
12יִּקָּחֲךָ֥ לִבֶּ֑ךָ וּֽמַה־ יִּרְזְמ֥וּן עֵינֶֽיךָ׃
13תָשִׁ֣יב אֶל־ אֵ֣ל רוּחֶ֑ךָ וְהֹצֵ֖אתָ מִפִּ֣יךָ מִלִּֽין׃
14אֱנ֥וֹשׁ כִּֽי־ יִזְכֶּ֑ה וְכִֽי־ יִ֝צְדַּ֗ק יְל֣וּד אִשָּֽׁה׃
15[בִּקְדֹשֹׁו כ] (בִּ֭קְדֹשָׁיו ק) לֹ֣א יַאֲמִ֑ין וְ֝שָׁמַ֗יִם לֹא־ זַכּ֥וּ בְעֵינָֽיו׃
16כִּֽי־ נִתְעָ֥ב וְֽנֶאֱלָ֑ח אִישׁ־ שֹׁתֶ֖ה כַמַּ֣יִם עַוְלָֽה׃
17שְֽׁמַֽע־ לִ֑י וְזֶֽה־ חָ֝זִ֗יתִי וַאֲסַפֵּֽרָה׃
18חֲכָמִ֥ים יַגִּ֑ידוּ וְלֹ֥א כִֽ֝חֲד֗וּ מֵאֲבוֹתָֽם׃
19לְ֭בַדָּם נִתְּנָ֣ה הָאָ֑רֶץ וְלֹא־ עָ֖בַר זָ֣ר בְּתוֹכָֽם׃
20יְמֵ֣י רָ֭שָׁע ה֣וּא מִתְחוֹלֵ֑ל וּמִסְפַּ֥ר שָׁ֝נִ֗ים נִצְפְּנ֥וּ לֶעָרִֽיץ׃
21פְּחָדִ֥ים בְּאָזְנָ֑יו בַּ֝שָּׁל֗וֹם שׁוֹדֵ֥ד יְבוֹאֶֽנּוּ׃
22יַאֲמִ֣ין שׁ֭וּב מִנִּי־ חֹ֑שֶׁךְ [וְצָפוּ כ] (וְצָפ֖וּי ק) ה֣וּא אֱלֵי־ חָֽרֶב׃
23ה֣וּא לַלֶּ֣חֶם אַיֵּ֑ה יָדַ֓ע ׀ כִּֽי־ נָכ֖וֹן בְּיָד֣וֹ יֽוֹם־ חֹֽשֶׁךְ׃
24צַ֣ר וּמְצוּקָ֑ה תִּ֝תְקְפֵ֗הוּ כְּמֶ֤לֶךְ ׀ עָתִ֬יד לַכִּידֽוֹר׃
25נָטָ֣ה אֶל־ אֵ֣ל יָד֑וֹ וְאֶל־ שַׁ֝דַּ֗י יִתְגַּבָּֽר׃
26אֵלָ֣יו בְּצַוָּ֑אר בַּ֝עֲבִ֗י גַּבֵּ֥י מָֽגִנָּֽיו׃
27כִסָּ֣ה פָנָ֣יו בְּחֶלְבּ֑וֹ וַיַּ֖עַשׂ פִּימָ֣ה עֲלֵי־ כָֽסֶל׃
28עָ֘רִ֤ים נִכְחָד֗וֹת בָּ֭תִּים לֹא־ יֵ֣שְׁבוּ לָ֑מוֹ אֲשֶׁ֖ר הִתְעַתְּד֣וּ לְגַלִּֽים׃
29יֶ֭עְשַׁר וְלֹא־ יָק֣וּם חֵיל֑וֹ וְלֹֽא־ יִטֶּ֖ה לָאָ֣רֶץ מִנְלָֽם׃
30יָס֨וּר ׀ מִנִּי־ חֹ֗שֶׁךְ יֹֽ֭נַקְתּוֹ תְּיַבֵּ֣שׁ שַׁלְהָ֑בֶת וְ֝יָס֗וּר בְּר֣וּחַ פִּֽיו׃
31יַאֲמֵ֣ן [בַּשֹּׁו כ] (בַּשָּׁ֣יו ק) נִתְעָ֑ה כִּי־ שָׁ֝֗וְא תִּהְיֶ֥ה תְמוּרָתֽוֹ׃
32י֭וֹמוֹ תִּמָּלֵ֑א וְ֝כִפָּת֗וֹ לֹ֣א רַעֲנָֽנָה׃
33כַּגֶּ֣פֶן בִּסְר֑וֹ וְיַשְׁלֵ֥ךְ כַּ֝זַּ֗יִת נִצָּתֽוֹ׃
34עֲדַ֣ת חָנֵ֣ף גַּלְמ֑וּד וְ֝אֵ֗שׁ אָכְלָ֥ה אָֽהֳלֵי־ שֹֽׁחַד׃
