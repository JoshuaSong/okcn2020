1תְּ֭פִלָּה לְעָנִ֣י כִֽי־ יַעֲטֹ֑ף וְלִפְנֵ֥י יְ֝הוָ֗ה יִשְׁפֹּ֥ךְ שִׂיחֽוֹ׃ יְ֭הוָה שִׁמְעָ֣ה תְפִלָּתִ֑י וְ֝שַׁוְעָתִ֗י אֵלֶ֥יךָ תָבֽוֹא׃
2תַּסְתֵּ֬ר פָּנֶ֨יךָ ׀ מִמֶּנִּי֮ בְּי֪וֹם צַ֫ר לִ֥י הַטֵּֽה־ אֵלַ֥י אָזְנֶ֑ךָ בְּי֥וֹם אֶ֝קְרָ֗א מַהֵ֥ר עֲנֵֽנִי׃
3כָל֣וּ בְעָשָׁ֣ן יָמָ֑י וְ֝עַצְמוֹתַ֗י כְּמוֹ־ קֵ֥ד נִחָֽרוּ׃
4כָ֭עֵשֶׂב וַיִּבַ֣שׁ לִבִּ֑י כִּֽי־ שָׁ֝כַ֗חְתִּי מֵאֲכֹ֥ל לַחְמִֽי׃
5אַנְחָתִ֑י דָּבְקָ֥ה עַ֝צְמִ֗י לִבְשָׂרִֽי׃
6לִקְאַ֣ת מִדְבָּ֑ר הָ֝יִ֗יתִי כְּכ֣וֹס חֳרָבֽוֹת׃
7וָאֶֽהְיֶ֑ה כְּ֝צִפּ֗וֹר בּוֹדֵ֥ד עַל־ גָּֽג׃
8הַ֭יּוֹם חֵרְפ֣וּנִי אוֹיְבָ֑י מְ֝הוֹלָלַ֗י בִּ֣י נִשְׁבָּֽעוּ׃
9אֵ֭פֶר כַּלֶּ֣חֶם אָכָ֑לְתִּי וְ֝שִׁקֻּוַ֗י בִּבְכִ֥י מָסָֽכְתִּי׃
10זַֽעַמְךָ֥ וְקִצְפֶּ֑ךָ כִּ֥י נְ֝שָׂאתַ֗נִי וַתַּשְׁלִיכֵֽנִי׃
11כְּצֵ֣ל נָט֑וּי וַ֝אֲנִ֗י כָּעֵ֥שֶׂב אִיבָֽשׁ׃
12יְ֭הוָה לְעוֹלָ֣ם תֵּשֵׁ֑ב וְ֝זִכְרְךָ֗ לְדֹ֣ר וָדֹֽר׃
13תָ֭קוּם תְּרַחֵ֣ם צִיּ֑וֹן כִּי־ עֵ֥ת לְ֝חֶֽנְנָ֗הּ כִּי־ בָ֥א מוֹעֵֽד׃
14רָצ֣וּ עֲ֭בָדֶיךָ אֶת־ אֲבָנֶ֑יהָ וְֽאֶת־ עֲפָרָ֥הּ יְחֹנֵֽנוּ׃
15ג֭וֹיִם אֶת־ שֵׁ֣ם יְהוָ֑ה וְֽכָל־ מַלְכֵ֥י הָ֝אָ֗רֶץ אֶת־ כְּבוֹדֶֽךָ׃
16בָנָ֣ה יְהוָ֣ה צִיּ֑וֹן נִ֝רְאָ֗ה בִּכְבוֹדֽוֹ׃
17אֶל־ תְּפִלַּ֣ת הָעַרְעָ֑ר וְלֹֽא־ בָ֝זָ֗ה אֶת־ תְּפִלָּתָֽם׃
18זֹ֭את לְד֣וֹר אַחֲר֑וֹן וְעַ֥ם נִ֝בְרָ֗א יְהַלֶּל־ יָֽהּ׃
19הִ֭שְׁקִיף מִמְּר֣וֹם קָדְשׁ֑וֹ יְ֝הוָ֗ה מִשָּׁמַ֤יִם ׀ אֶל־ אֶ֬רֶץ הִבִּֽיט׃
20אֶנְקַ֣ת אָסִ֑יר לְ֝פַתֵּ֗חַ בְּנֵ֣י תְמוּתָֽה׃
21בְּ֭צִיּוֹן שֵׁ֣ם יְהוָ֑ה וּ֝תְהִלָּת֗וֹ בִּירוּשָׁלִָֽם׃
22עַמִּ֣ים יַחְדָּ֑ו וּ֝מַמְלָכ֗וֹת לַעֲבֹ֥ד אֶת־ יְהוָֽה׃
23בַדֶּ֥רֶךְ [כֹּחֹו כ] (כֹּחִ֗י ק) קִצַּ֥ר יָמָֽי׃
24אֵלִ֗י אַֽל־ תַּ֭עֲלֵנִי בַּחֲצִ֣י יָמָ֑י בְּד֖וֹר דּוֹרִ֣ים שְׁנוֹתֶֽיךָ׃
25הָאָ֣רֶץ יָסַ֑דְתָּ וּֽמַעֲשֵׂ֖ה יָדֶ֣יךָ שָׁמָֽיִם׃
26יֹאבֵדוּ֮ וְאַתָּ֪ה תַ֫עֲמֹ֥ד וְ֭כֻלָּם כַּבֶּ֣גֶד יִבְל֑וּ כַּלְּב֖וּשׁ תַּחֲלִיפֵ֣ם וְֽיַחֲלֹֽפוּ׃
27ה֑וּא וּ֝שְׁנוֹתֶ֗יךָ לֹ֣א יִתָּֽמּוּ׃
