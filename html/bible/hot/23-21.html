1מַשָּׂ֖א מִדְבַּר־ יָ֑ם כְּסוּפ֤וֹת בַּנֶּ֙גֶב֙ לַֽחֲלֹ֔ף מִמִּדְבָּ֣ר בָּ֔א מֵאֶ֖רֶץ נוֹרָאָֽה׃
2קָשָׁ֖ה הֻגַּד־ לִ֑י הַבּוֹגֵ֤ד ׀ בּוֹגֵד֙ וְהַשּׁוֹדֵ֣ד ׀ שׁוֹדֵ֔ד עֲלִ֤י עֵילָם֙ צוּרִ֣י מָדַ֔י כָּל־ אַנְחָתָ֖ה הִשְׁבַּֽתִּי׃
3כֵּ֗ן מָלְא֤וּ מָתְנַי֙ חַלְחָלָ֔ה צִירִ֣ים אֲחָז֔וּנִי כְּצִירֵ֖י יֽוֹלֵדָ֑ה נַעֲוֵ֣יתִי מִשְּׁמֹ֔עַ נִבְהַ֖לְתִּי מֵרְאֽוֹת׃
4לְבָבִ֔י פַּלָּצ֖וּת בִּֽעֲתָ֑תְנִי אֵ֚ת נֶ֣שֶׁף חִשְׁקִ֔י שָׂ֥ם לִ֖י לַחֲרָדָֽה׃
5הַשֻּׁלְחָ֛ן צָפֹ֥ה הַצָּפִ֖ית אָכ֣וֹל שָׁתֹ֑ה ק֥וּמוּ הַשָּׂרִ֖ים מִשְׁח֥וּ מָגֵֽן׃ פ
6כֹ֥ה אָמַ֛ר אֵלַ֖י אֲדֹנָ֑י לֵ֚ךְ הַעֲמֵ֣ד הַֽמְצַפֶּ֔ה אֲשֶׁ֥ר יִרְאֶ֖ה יַגִּֽיד׃
7רֶ֗כֶב צֶ֚מֶד פָּֽרָשִׁ֔ים רֶ֥כֶב חֲמ֖וֹר רֶ֣כֶב גָּמָ֑ל וְהִקְשִׁ֥יב קֶ֖שֶׁב רַב־ קָֽשֶׁב׃
8אַרְיֵ֑ה עַל־ מִצְפֶּ֣ה ׀ אֲדֹנָ֗י אָנֹכִ֞י עֹמֵ֤ד תָּמִיד֙ יוֹמָ֔ם וְעַל־ מִ֨שְׁמַרְתִּ֔י אָנֹכִ֥י נִצָּ֖ב כָּל־ הַלֵּילֽוֹת׃
9זֶ֥ה בָא֙ רֶ֣כֶב אִ֔ישׁ צֶ֖מֶד פָּֽרָשִׁ֑ים וַיַּ֣עַן וַיֹּ֗אמֶר נָפְלָ֤ה נָֽפְלָה֙ בָּבֶ֔ל וְכָל־ פְּסִילֵ֥י אֱלֹהֶ֖יהָ שִׁבַּ֥ר לָאָֽרֶץ׃
10וּבֶן־ גָּרְנִ֑י אֲשֶׁ֣ר שָׁמַ֗עְתִּי מֵאֵ֨ת יְהוָ֧ה צְבָא֛וֹת אֱלֹהֵ֥י יִשְׂרָאֵ֖ל הִגַּ֥דְתִּי לָכֶֽם׃ ס
11דּוּמָ֑ה אֵלַי֙ קֹרֵ֣א מִשֵּׂעִ֔יר שֹׁמֵר֙ מַה־ מִלַּ֔יְלָה שֹׁמֵ֖ר מַה־ מִלֵּֽיל׃
12שֹׁמֵ֔ר אָתָ֥ה בֹ֖קֶר וְגַם־ לָ֑יְלָה אִם־ תִּבְעָי֥וּן בְּעָ֖יוּ שֻׁ֥בוּ אֵתָֽיוּ׃ ס
13בַּעְרָ֑ב בַּיַּ֤עַר בַּעְרַב֙ תָּלִ֔ינוּ אֹֽרְח֖וֹת דְּדָנִֽים׃
14צָמֵ֖א הֵתָ֣יוּ מָ֑יִם יֹשְׁבֵי֙ אֶ֣רֶץ תֵּימָ֔א בְּלַחְמ֖וֹ קִדְּמ֥וּ נֹדֵֽד׃
15מִפְּנֵ֥י חֲרָב֖וֹת נָדָ֑דוּ מִפְּנֵ֣י ׀ חֶ֣רֶב נְטוּשָׁ֗ה וּמִפְּנֵי֙ קֶ֣שֶׁת דְּרוּכָ֔ה וּמִפְּנֵ֖י כֹּ֥בֶד מִלְחָמָֽה׃ ס
16כֹ֛ה אָמַ֥ר אֲדֹנָ֖י אֵלָ֑י בְּע֤וֹד שָׁנָה֙ כִּשְׁנֵ֣י שָׂכִ֔יר וְכָלָ֖ה כָּל־ כְּב֥וֹד קֵדָֽר׃
