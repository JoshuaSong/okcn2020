1וְאַתָּה֙ שָׂ֣א קִינָ֔ה אֶל־ נְשִׂיאֵ֖י יִשְׂרָאֵֽל׃
2מָ֤ה אִמְּךָ֙ לְבִיָּ֔א בֵּ֥ין אֲרָי֖וֹת רָבָ֑צָה בְּת֥וֹךְ כְּפִרִ֖ים רִבְּתָ֥ה גוּרֶֽיהָ׃
3אֶחָ֥ד מִגֻּרֶ֖יהָ כְּפִ֣יר הָיָ֑ה וַיִּלְמַ֥ד לִטְרָף־ טֶ֖רֶף אָדָ֥ם אָכָֽל׃
4אֵלָ֛יו גּוֹיִ֖ם בְּשַׁחְתָּ֣ם נִתְפָּ֑שׂ וַיְבִאֻ֥הוּ בַֽחַחִ֖ים אֶל־ אֶ֥רֶץ מִצְרָֽיִם׃
5כִּ֣י נֽוֹחֲלָ֔ה אָבְדָ֖ה תִּקְוָתָ֑הּ וַתִּקַּ֛ח אֶחָ֥ד מִגֻּרֶ֖יהָ כְּפִ֥יר שָׂמָֽתְהוּ׃
6בְּתוֹךְ־ אֲרָי֖וֹת כְּפִ֣יר הָיָ֑ה וַיִּלְמַ֥ד לִטְרָף־ טֶ֖רֶף אָדָ֥ם אָכָֽל׃
7אַלְמְנוֹתָ֔יו וְעָרֵיהֶ֖ם הֶחֱרִ֑יב וַתֵּ֤שַׁם אֶ֙רֶץ֙ וּמְלֹאָ֔הּ מִקּ֖וֹל שַׁאֲגָתֽוֹ׃
8עָלָ֥יו גּוֹיִ֛ם סָבִ֖יב מִמְּדִינ֑וֹת וַֽיִּפְרְשׂ֥וּ עָלָ֛יו רִשְׁתָּ֖ם בְּשַׁחְתָּ֥ם נִתְפָּֽשׂ׃
9בַסּוּגַר֙ בַּֽחַחִ֔ים וַיְבִאֻ֖הוּ אֶל־ מֶ֣לֶךְ בָּבֶ֑ל יְבִאֻ֙הוּ֙ בַּמְּצֹד֔וֹת לְמַ֗עַן לֹא־ יִשָּׁמַ֥ע קוֹל֛וֹ ע֖וֹד אֶל־ הָרֵ֥י יִשְׂרָאֵֽל׃ פ
10כַגֶּ֛פֶן בְּדָמְךָ֖ עַל־ מַ֣יִם שְׁתוּלָ֑ה פֹּֽרִיָּה֙ וַֽעֲנֵפָ֔ה הָיְתָ֖ה מִמַּ֥יִם רַבִּֽים׃
11לָ֞הּ מַטּ֣וֹת עֹ֗ז אֶל־ שִׁבְטֵי֙ מֹֽשְׁלִ֔ים וַתִּגְבַּ֥הּ קֽוֹמָת֖וֹ עַל־ בֵּ֣ין עֲבֹתִ֑ים וַיֵּרָ֣א בְגָבְה֔וֹ בְּרֹ֖ב דָּלִיֹּתָֽיו׃
12בְּחֵמָה֙ לָאָ֣רֶץ הֻשְׁלָ֔כָה וְר֥וּחַ הַקָּדִ֖ים הוֹבִ֣ישׁ פִּרְיָ֑הּ הִתְפָּרְק֧וּ וְיָבֵ֛שׁוּ מַטֵּ֥ה עֻזָּ֖הּ אֵ֥שׁ אֲכָלָֽתְהוּ׃
13שְׁתוּלָ֣ה בַמִּדְבָּ֑ר בְּאֶ֖רֶץ צִיָּ֥ה וְצָמָֽא׃
