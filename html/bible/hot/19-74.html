1מַשְׂכִּ֗יל לְאָ֫סָ֥ף לָמָ֣ה אֱ֭לֹהִים זָנַ֣חְתָּ לָנֶ֑צַח יֶעְשַׁ֥ן אַ֝פְּךָ֗ בְּצֹ֣אן מַרְעִיתֶֽךָ׃
2עֲדָתְךָ֨ ׀ קָ֘נִ֤יתָ קֶּ֗דֶם גָּ֭אַלְתָּ שֵׁ֣בֶט נַחֲלָתֶ֑ךָ הַר־ צִ֝יּ֗וֹן זֶ֤ה ׀ שָׁכַ֬נְתָּ בּֽוֹ׃
3פְ֭עָמֶיךָ לְמַשֻּׁא֣וֹת נֶ֑צַח כָּל־ הֵרַ֖ע אוֹיֵ֣ב בַּקֹּֽדֶשׁ׃
4צֹ֭רְרֶיךָ בְּקֶ֣רֶב מוֹעֲדֶ֑ךָ שָׂ֖מוּ אוֹתֹתָ֣ם אֹתֽוֹת׃
5כְּמֵבִ֣יא לְמָ֑עְלָה בִּֽסֲבָךְ־ עֵ֝֗ץ קַרְדֻּמּֽוֹת׃
6כ] (וְ֭עַתָּה ק) פִּתּוּחֶ֣יהָ יָּ֑חַד בְּכַשִּׁ֥יל וְ֝כֵֽילַפֹּ֗ת יַהֲלֹמֽוּן׃
7בָ֭אֵשׁ מִקְדָּשֶׁ֑ךָ לָ֝אָ֗רֶץ חִלְּל֥וּ מִֽשְׁכַּן־ שְׁמֶֽךָ׃
8בְ֭לִבָּם נִינָ֣ם יָ֑חַד שָׂרְפ֖וּ כָל־ מוֹעֲדֵי־ אֵ֣ל בָּאָֽרֶץ׃
9לֹ֥א רָ֫אִ֥ינוּ אֵֽין־ ע֥וֹד נָבִ֑יא וְלֹֽא־ אִ֝תָּ֗נוּ יֹדֵ֥עַ עַד־ מָֽה׃
10מָתַ֣י אֱ֭לֹהִים יְחָ֣רֶף צָ֑ר יְנָ֘אֵ֤ץ אוֹיֵ֖ב שִׁמְךָ֣ לָנֶֽצַח׃
11תָשִׁ֣יב יָ֭דְךָ וִֽימִינֶ֑ךָ מִקֶּ֖רֶב [חֹוקְךָ כ] (חֵֽיקְךָ֣ ק) כַלֵּֽה׃
12מַלְכִּ֣י מִקֶּ֑דֶם פֹּעֵ֥ל יְ֝שׁוּע֗וֹת בְּקֶ֣רֶב הָאָֽרֶץ׃
13פוֹרַ֣רְתָּ בְעָזְּךָ֣ יָ֑ם שִׁבַּ֖רְתָּ רָאשֵׁ֥י תַ֝נִּינִ֗ים עַל־ הַמָּֽיִם׃
14רִ֭צַּצְתָּ רָאשֵׁ֣י לִוְיָתָ֑ן תִּתְּנֶ֥נּוּ מַ֝אֲכָ֗ל לְעָ֣ם לְצִיִּֽים׃
15בָ֭קַעְתָּ מַעְיָ֣ן וָנָ֑חַל אַתָּ֥ה ה֝וֹבַ֗שְׁתָּ נַהֲר֥וֹת אֵיתָֽן׃
16י֭וֹם אַף־ לְךָ֥ לָ֑יְלָה אַתָּ֥ה הֲ֝כִינ֗וֹתָ מָא֥וֹר וָשָֽׁמֶשׁ׃
17הִ֭צַּבְתָּ כָּל־ גְּבוּל֣וֹת אָ֑רֶץ קַ֥יִץ וָ֝חֹ֗רֶף אַתָּ֥ה יְצַרְתָּם׃
18זֹ֗את א֭וֹיֵב חֵרֵ֣ף ׀ יְהוָ֑ה וְעַ֥ם נָ֝בָ֗ל נִֽאֲצ֥וּ שְׁמֶֽךָ׃
19תִּתֵּ֣ן לְ֭חַיַּת נֶ֣פֶשׁ תּוֹרֶ֑ךָ חַיַּ֥ת עֲ֝נִיֶּ֗יךָ אַל־ תִּשְׁכַּ֥ח לָנֶֽצַח׃
20לַבְּרִ֑ית כִּ֥י מָלְא֥וּ מַחֲשַׁכֵּי־ אֶ֝֗רֶץ נְא֣וֹת חָמָֽס׃
21יָשֹׁ֣ב דַּ֣ךְ נִכְלָ֑ם עָנִ֥י וְ֝אֶבְי֗וֹן יְֽהַלְל֥וּ שְׁמֶֽךָ׃
22אֱ֭לֹהִים רִיבָ֣ה רִיבֶ֑ךָ זְכֹ֥ר חֶרְפָּתְךָ֥ מִנִּי־ נָ֝בָ֗ל כָּל־ הַיּֽוֹם׃
