1כִּ֤י יִגְנֹֽב־ אִישׁ֙ שׁ֣וֹר אוֹ־ שֶׂ֔ה וּטְבָח֖וֹ א֣וֹ מְכָר֑וֹ חֲמִשָּׁ֣ה בָקָ֗ר יְשַׁלֵּם֙ תַּ֣חַת הַשּׁ֔וֹר וְאַרְבַּע־ צֹ֖אן תַּ֥חַת הַשֶּֽׂה׃
2בַּמַּחְתֶּ֛רֶת יִמָּצֵ֥א הַגַּנָּ֖ב וְהֻכָּ֣ה וָמֵ֑ת אֵ֥ין ל֖וֹ דָּמִֽים׃
3זָרְחָ֥ה הַשֶּׁ֛מֶשׁ עָלָ֖יו דָּמִ֣ים ל֑וֹ שַׁלֵּ֣ם יְשַׁלֵּ֔ם אִם־ אֵ֣ין ל֔וֹ וְנִמְכַּ֖ר בִּגְנֵבָתֽוֹ׃
4הִמָּצֵא֩ תִמָּצֵ֨א בְיָד֜וֹ הַגְּנֵבָ֗ה מִשּׁ֧וֹר עַד־ חֲמ֛וֹר עַד־ שֶׂ֖ה חַיִּ֑ים שְׁנַ֖יִם יְשַׁלֵּֽם׃ ס
5יַבְעֶר־ אִישׁ֙ שָׂדֶ֣ה אוֹ־ כֶ֔רֶם וְשִׁלַּח֙ אֶת־ [בְּעִירָה כ] (בְּעִירֹ֔ו ק) וּבִעֵ֖ר בִּשְׂדֵ֣ה אַחֵ֑ר מֵיטַ֥ב שָׂדֵ֛הוּ וּמֵיטַ֥ב כַּרְמ֖וֹ יְשַׁלֵּֽם׃ ס
6תֵצֵ֨א אֵ֜שׁ וּמָצְאָ֤ה קֹצִים֙ וְנֶאֱכַ֣ל גָּדִ֔ישׁ א֥וֹ הַקָּמָ֖ה א֣וֹ הַשָּׂדֶ֑ה שַׁלֵּ֣ם יְשַׁלֵּ֔ם הַמַּבְעִ֖ר אֶת־ הַבְּעֵרָֽה׃ ס
7יִתֵּן֩ אִ֨ישׁ אֶל־ רֵעֵ֜הוּ כֶּ֤סֶף אֽוֹ־ כֵלִים֙ לִשְׁמֹ֔ר וְגֻנַּ֖ב מִבֵּ֣ית הָאִ֑ישׁ אִם־ יִמָּצֵ֥א הַגַּנָּ֖ב יְשַׁלֵּ֥ם שְׁנָֽיִם׃
8לֹ֤א יִמָּצֵא֙ הַגַּנָּ֔ב וְנִקְרַ֥ב בַּֽעַל־ הַבַּ֖יִת אֶל־ הָֽאֱלֹהִ֑ים אִם־ לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֥אכֶת רֵעֵֽהוּ׃
9כָּל־ דְּבַר־ פֶּ֡שַׁע עַל־ שׁ֡וֹר עַל־ חֲ֠מוֹר עַל־ שֶׂ֨ה עַל־ שַׂלְמָ֜ה עַל־ כָּל־ אֲבֵדָ֗ה אֲשֶׁ֤ר יֹאמַר֙ כִּי־ ה֣וּא זֶ֔ה עַ֚ד הָֽאֱלֹהִ֔ים יָבֹ֖א דְּבַר־ שְׁנֵיהֶ֑ם אֲשֶׁ֤ר יַרְשִׁיעֻן֙ אֱלֹהִ֔ים יְשַׁלֵּ֥ם שְׁנַ֖יִם לְרֵעֵֽהוּ׃ ס
10יִתֵּן֩ אִ֨ישׁ אֶל־ רֵעֵ֜הוּ חֲמ֨וֹר אוֹ־ שׁ֥וֹר אוֹ־ שֶׂ֛ה וְכָל־ בְּהֵמָ֖ה לִשְׁמֹ֑ר וּמֵ֛ת אוֹ־ נִשְׁבַּ֥ר אוֹ־ נִשְׁבָּ֖ה אֵ֥ין רֹאֶֽה׃
11יְהוָ֗ה תִּהְיֶה֙ בֵּ֣ין שְׁנֵיהֶ֔ם אִם־ לֹ֥א שָׁלַ֛ח יָד֖וֹ בִּמְלֶ֣אכֶת רֵעֵ֑הוּ וְלָקַ֥ח בְּעָלָ֖יו וְלֹ֥א יְשַׁלֵּֽם׃
12גָּנֹ֥ב יִגָּנֵ֖ב מֵעִמּ֑וֹ יְשַׁלֵּ֖ם לִבְעָלָֽיו׃
13טָרֹ֥ף יִטָּרֵ֖ף יְבִאֵ֣הוּ עֵ֑ד הַטְּרֵפָ֖ה לֹ֥א יְשַׁלֵּֽם׃ פ
14יִשְׁאַ֥ל אִ֛ישׁ מֵעִ֥ם רֵעֵ֖הוּ וְנִשְׁבַּ֣ר אוֹ־ מֵ֑ת בְּעָלָ֥יו אֵין־ עִמּ֖וֹ שַׁלֵּ֥ם יְשַׁלֵּֽם׃
15בְּעָלָ֥יו עִמּ֖וֹ לֹ֣א יְשַׁלֵּ֑ם אִם־ שָׂכִ֣יר ה֔וּא בָּ֖א בִּשְׂכָרֽוֹ׃ ס
16יְפַתֶּ֣ה אִ֗ישׁ בְּתוּלָ֛ה אֲשֶׁ֥ר לֹא־ אֹרָ֖שָׂה וְשָׁכַ֣ב עִמָּ֑הּ מָהֹ֛ר יִמְהָרֶ֥נָּה לּ֖וֹ לְאִשָּֽׁה׃
17מָאֵ֧ן יְמָאֵ֛ן אָבִ֖יהָ לְתִתָּ֣הּ ל֑וֹ כֶּ֣סֶף יִשְׁקֹ֔ל כְּמֹ֖הַר הַבְּתוּלֹֽת׃ ס
18לֹ֥א תְחַיֶּֽה׃ ס
19שֹׁכֵ֥ב עִם־ בְּהֵמָ֖ה מ֥וֹת יוּמָֽת׃ ס
20לָאֱלֹהִ֖ים יָֽחֳרָ֑ם בִּלְתִּ֥י לַיהוָ֖ה לְבַדּֽוֹ׃
21לֹא־ תוֹנֶ֖ה וְלֹ֣א תִלְחָצֶ֑נּוּ כִּֽי־ גֵרִ֥ים הֱיִיתֶ֖ם בְּאֶ֥רֶץ מִצְרָֽיִם׃
22אַלְמָנָ֥ה וְיָת֖וֹם לֹ֥א תְעַנּֽוּן׃
23עַנֵּ֥ה תְעַנֶּ֖ה אֹת֑וֹ כִּ֣י אִם־ צָעֹ֤ק יִצְעַק֙ אֵלַ֔י שָׁמֹ֥עַ אֶשְׁמַ֖ע צַעֲקָתֽוֹ׃
24אַפִּ֔י וְהָרַגְתִּ֥י אֶתְכֶ֖ם בֶּחָ֑רֶב וְהָי֤וּ נְשֵׁיכֶם֙ אַלְמָנ֔וֹת וּבְנֵיכֶ֖ם יְתֹמִֽים׃ פ
25כֶּ֣סֶף ׀ תַּלְוֶ֣ה אֶת־ עַמִּ֗י אֶת־ הֶֽעָנִי֙ עִמָּ֔ךְ לֹא־ תִהְיֶ֥ה ל֖וֹ כְּנֹשֶׁ֑ה לֹֽא־ תְשִׂימ֥וּן עָלָ֖יו נֶֽשֶׁךְ׃
26חָבֹ֥ל תַּחְבֹּ֖ל שַׂלְמַ֣ת רֵעֶ֑ךָ עַד־ בֹּ֥א הַשֶּׁ֖מֶשׁ תְּשִׁיבֶ֥נּוּ לֽוֹ׃
27הִ֤וא [כְסוּתָה כ] (כְסוּתֹו֙ ק) לְבַדָּ֔הּ הִ֥וא שִׂמְלָת֖וֹ לְעֹר֑וֹ בַּמֶּ֣ה יִשְׁכָּ֔ב וְהָיָה֙ כִּֽי־ יִצְעַ֣ק אֵלַ֔י וְשָׁמַעְתִּ֖י כִּֽי־ חַנּ֥וּן אָֽנִי׃ ס
28לֹ֣א תְקַלֵּ֑ל וְנָשִׂ֥יא בְעַמְּךָ֖ לֹ֥א תָאֹֽר׃
29וְדִמְעֲךָ֖ לֹ֣א תְאַחֵ֑ר בְּכ֥וֹר בָּנֶ֖יךָ תִּתֶּן־ לִּֽי׃
30תַּעֲשֶׂ֥ה לְשֹׁרְךָ֖ לְצֹאנֶ֑ךָ שִׁבְעַ֤ת יָמִים֙ יִהְיֶ֣ה עִם־ אִמּ֔וֹ בַּיּ֥וֹם הַשְּׁמִינִ֖י תִּתְּנוֹ־ לִֽי׃
