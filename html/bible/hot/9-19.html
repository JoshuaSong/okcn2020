1וַיְדַבֵּ֣ר שָׁא֗וּל אֶל־ יוֹנָתָ֤ן בְּנוֹ֙ וְאֶל־ כָּל־ עֲבָדָ֔יו לְהָמִ֖ית אֶת־ דָּוִ֑ד וִיהֽוֹנָתָן֙ בֶּן־ שָׁא֔וּל חָפֵ֥ץ בְּדָוִ֖ד מְאֹֽד׃
2יְהוֹנָתָן֙ לְדָוִ֣ד לֵאמֹ֔ר מְבַקֵּ֛שׁ שָׁא֥וּל אָבִ֖י לַהֲמִיתֶ֑ךָ וְעַתָּה֙ הִשָּֽׁמֶר־ נָ֣א בַבֹּ֔קֶר וְיָשַׁבְתָּ֥ בַסֵּ֖תֶר וְנַחְבֵּֽאתָ׃
3אֵצֵ֜א וְעָמַדְתִּ֣י לְיַד־ אָבִ֗י בַּשָּׂדֶה֙ אֲשֶׁ֣ר אַתָּ֣ה שָׁ֔ם וַאֲנִ֕י אֲדַבֵּ֥ר בְּךָ֖ אֶל־ אָבִ֑י וְרָאִ֥יתִי מָ֖ה וְהִגַּ֥דְתִּי לָֽךְ׃ ס
4יְהוֹנָתָ֤ן בְּדָוִד֙ ט֔וֹב אֶל־ שָׁא֖וּל אָבִ֑יו וַיֹּ֣אמֶר אֵ֠לָיו אַל־ יֶחֱטָ֨א הַמֶּ֜לֶךְ בְּעַבְדּ֣וֹ בְדָוִ֗ד כִּ֣י ל֤וֹא חָטָא֙ לָ֔ךְ וְכִ֥י מַעֲשָׂ֖יו טוֹב־ לְךָ֥ מְאֹֽד׃
5אֶת־ נַפְשׁ֨וֹ בְכַפּ֜וֹ וַיַּ֣ךְ אֶת־ הַפְּלִשְׁתִּ֗י וַיַּ֨עַשׂ יְהוָ֜ה תְּשׁוּעָ֤ה גְדוֹלָה֙ לְכָל־ יִשְׂרָאֵ֔ל רָאִ֖יתָ וַתִּשְׂמָ֑ח וְלָ֤מָּה תֶֽחֱטָא֙ בְּדָ֣ם נָקִ֔י לְהָמִ֥ית אֶת־ דָּוִ֖ד חִנָּֽם׃
6שָׁא֖וּל בְּק֣וֹל יְהוֹנָתָ֑ן וַיִּשָּׁבַ֣ע שָׁא֔וּל חַי־ יְהוָ֖ה אִם־ יוּמָֽת׃
7יְהוֹנָתָן֙ לְדָוִ֔ד וַיַּגֶּד־ לוֹ֙ יְה֣וֹנָתָ֔ן אֵ֥ת כָּל־ הַדְּבָרִ֖ים הָאֵ֑לֶּה וַיָּבֵ֨א יְהוֹנָתָ֤ן אֶת־ דָּוִד֙ אֶל־ שָׁא֔וּל וַיְהִ֥י לְפָנָ֖יו כְּאֶתְמ֥וֹל שִׁלְשֽׁוֹם׃ ס
8הַמִּלְחָמָ֖ה לִֽהְי֑וֹת וַיֵּצֵ֨א דָוִ֜ד וַיִּלָּ֣חֶם בַּפְּלִשְׁתִּ֗ים וַיַּ֤ךְ בָּהֶם֙ מַכָּ֣ה גְדוֹלָ֔ה וַיָּנֻ֖סוּ מִפָּנָֽיו׃
9ר֨וּחַ יְהוָ֤ה ׀ רָעָה֙ אֶל־ שָׁא֔וּל וְהוּא֙ בְּבֵית֣וֹ יוֹשֵׁ֔ב וַחֲנִית֖וֹ בְּיָד֑וֹ וְדָוִ֖ד מְנַגֵּ֥ן בְּיָֽד׃
10שָׁא֜וּל לְהַכּ֤וֹת בַּֽחֲנִית֙ בְּדָוִ֣ד וּבַקִּ֔יר וַיִּפְטַר֙ מִפְּנֵ֣י שָׁא֔וּל וַיַּ֥ךְ אֶֽת־ הַחֲנִ֖ית בַּקִּ֑יר וְדָוִ֛ד נָ֥ס וַיִּמָּלֵ֖ט בַּלַּ֥יְלָה הֽוּא׃ פ
11שָׁא֨וּל מַלְאָכִ֜ים אֶל־ בֵּ֤ית דָּוִד֙ לְשָׁמְר֔וֹ וְלַהֲמִית֖וֹ בַּבֹּ֑קֶר וַתַּגֵּ֣ד לְדָוִ֗ד מִיכַ֤ל אִשְׁתּוֹ֙ לֵאמֹ֔ר אִם־ אֵ֨ינְךָ֜ מְמַלֵּ֤ט אֶֽת־ נַפְשְׁךָ֙ הַלַּ֔יְלָה מָחָ֖ר אַתָּ֥ה מוּמָֽת׃
12מִיכַ֛ל אֶת־ דָּוִ֖ד בְּעַ֣ד הַחַלּ֑וֹן וַיֵּ֥לֶךְ וַיִּבְרַ֖ח וַיִּמָּלֵֽט׃
13מִיכַ֜ל אֶת־ הַתְּרָפִ֗ים וַתָּ֙שֶׂם֙ אֶל־ הַמִּטָּ֔ה וְאֵת֙ כְּבִ֣יר הָֽעִזִּ֔ים שָׂ֖מָה מְרַֽאֲשֹׁתָ֑יו וַתְּכַ֖ס בַּבָּֽגֶד׃ ס
14שָׁא֛וּל מַלְאָכִ֖ים לָקַ֣חַת אֶת־ דָּוִ֑ד וַתֹּ֖אמֶר חֹלֶ֥ה הֽוּא׃ פ
15שָׁאוּל֙ אֶת־ הַמַּלְאָכִ֔ים לִרְא֥וֹת אֶת־ דָּוִ֖ד לֵאמֹ֑ר הַעֲל֨וּ אֹת֧וֹ בַמִּטָּ֛ה אֵלַ֖י לַהֲמִתֽוֹ׃
16הַמַּלְאָכִ֔ים וְהִנֵּ֥ה הַתְּרָפִ֖ים אֶל־ הַמִּטָּ֑ה וּכְבִ֥יר הָעִזִּ֖ים מְרַאֲשֹׁתָֽיו׃
17שָׁא֜וּל אֶל־ מִיכַ֗ל לָ֤מָּה כָּ֙כָה֙ רִמִּיתִ֔נִי וַתְּשַׁלְּחִ֥י אֶת־ אֹיְבִ֖י וַיִּמָּלֵ֑ט וַתֹּ֤אמֶר מִיכַל֙ אֶל־ שָׁא֔וּל הוּא־ אָמַ֥ר אֵלַ֛י שַׁלְּחִ֖נִי לָמָ֥ה אֲמִיתֵֽךְ׃
18בָּרַ֜ח וַיִּמָּלֵ֗ט וַיָּבֹ֤א אֶל־ שְׁמוּאֵל֙ הָרָמָ֔תָה וַיַּ֨גֶּד־ ל֔וֹ אֵ֛ת כָּל־ אֲשֶׁ֥ר עָֽשָׂה־ ל֖וֹ שָׁא֑וּל וַיֵּ֤לֶךְ הוּא֙ וּשְׁמוּאֵ֔ל וַיֵּשְׁב֖וּ [בְּנֹוִית כ] (בְּנָֽיֹות׃ ק)
19לְשָׁא֖וּל לֵאמֹ֑ר הִנֵּ֣ה דָוִ֔ד [בְּנֹוִית כ] (בְּנָיֹ֖ות ק) בָּרָמָֽה׃
20שָׁא֣וּל מַלְאָכִים֮ לָקַ֣חַת אֶת־ דָּוִד֒ וַיַּ֗רְא אֶֽת־ לַהֲקַ֤ת הַנְּבִיאִים֙ נִבְּאִ֔ים וּשְׁמוּאֵ֕ל עֹמֵ֥ד נִצָּ֖ב עֲלֵיהֶ֑ם וַתְּהִ֞י עַֽל־ מַלְאֲכֵ֤י שָׁאוּל֙ ר֣וּחַ אֱלֹהִ֔ים וַיִּֽתְנַבְּא֖וּ גַּם־ הֵֽמָּה׃
21לְשָׁא֗וּל וַיִּשְׁלַח֙ מַלְאָכִ֣ים אֲחֵרִ֔ים וַיִּֽתְנַבְּא֖וּ גַּם־ הֵ֑מָּה וַיֹּ֣סֶף שָׁא֗וּל וַיִּשְׁלַח֙ מַלְאָכִ֣ים שְׁלִשִׁ֔ים וַיִּֽתְנַבְּא֖וּ גַּם־ הֵֽמָּה׃
22גַּם־ ה֜וּא הָרָמָ֗תָה וַיָּבֹא֙ עַד־ בּ֤וֹר הַגָּדוֹל֙ אֲשֶׁ֣ר בַּשֶּׂ֔כוּ וַיִּשְׁאַ֣ל וַיֹּ֔אמֶר אֵיפֹ֥ה שְׁמוּאֵ֖ל וְדָוִ֑ד וַיֹּ֕אמֶר הִנֵּ֖ה [בְּנֹוִית כ] (בְּנָיֹ֥ות ק) בָּרָמָֽה׃
23שָׁ֔ם אֶל־ [נֹוִית כ] (נָיֹ֖ות ק) בָּרָמָ֑ה וַתְּהִי֩ עָלָ֨יו גַּם־ ה֜וּא ר֣וּחַ אֱלֹהִ֗ים וַיֵּ֤לֶךְ הָלוֹךְ֙ וַיִּתְנַבֵּ֔א עַד־ בֹּא֖וֹ [בְּנֹוִית כ] (בְּנָיֹ֥ות ק) בָּרָמָֽה׃
