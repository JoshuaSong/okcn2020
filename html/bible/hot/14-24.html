1בֶּן־ שֶׁ֤בַע שָׁנִים֙ יֹאָ֣שׁ בְּמָלְכ֔וֹ וְאַרְבָּעִ֣ים שָׁנָ֔ה מָלַ֖ךְ בִּֽירוּשָׁלִָ֑ם וְשֵׁ֣ם אִמּ֔וֹ צִבְיָ֖ה מִבְּאֵ֥ר שָֽׁבַע׃
2יוֹאָ֛שׁ הַיָּשָׁ֖ר בְּעֵינֵ֣י יְהוָ֑ה כָּל־ יְמֵ֖י יְהוֹיָדָ֥ע הַכֹּהֵֽן׃
3ל֥וֹ יְהוֹיָדָ֖ע נָשִׁ֣ים שְׁתָּ֑יִם וַיּ֖וֹלֶד בָּנִ֥ים וּבָנֽוֹת׃
4אַחֲרֵיכֵ֑ן הָיָה֙ עִם־ לֵ֣ב יוֹאָ֔שׁ לְחַדֵּ֖שׁ אֶת־ בֵּ֥ית יְהוָֽה׃
5אֶת־ הַכֹּהֲנִ֣ים וְהַלְוִיִּם֒ וַיֹּ֣אמֶר לָהֶ֡ם צְא֣וּ לְעָרֵ֪י יְהוּדָ֟ה וְקִבְצוּ֩ מִכָּל־ יִשְׂרָאֵ֨ל כֶּ֜סֶף לְחַזֵּ֣ק ׀ אֶת־ בֵּ֣ית אֱלֹֽהֵיכֶ֗ם מִדֵּ֤י שָׁנָה֙ בְּשָׁנָ֔ה וְאַתֶּ֖ם תְּמַהֲר֣וּ לַדָּבָ֑ר וְלֹ֥א מִֽהֲר֖וּ הַֽלְוִיִּֽם׃
6הַמֶּלֶךְ֮ לִֽיהוֹיָדָ֣ע הָרֹאשׁ֒ וַיֹּ֣אמֶר ל֗וֹ מַדּ֙וּעַ֙ לֹֽא־ דָרַ֣שְׁתָּ עַל־ הַלְוִיִּ֔ם לְהָבִ֞יא מִֽיהוּדָ֣ה וּמִֽירוּשָׁלִַ֗ם אֶת־ מַשְׂאַת֙ מֹשֶׁ֣ה עֶֽבֶד־ יְהוָ֔ה וְהַקָּהָ֖ל לְיִשְׂרָאֵ֑ל לְאֹ֖הֶל הָעֵדֽוּת׃
7עֲתַלְיָ֙הוּ֙ הַמִּרְשַׁ֔עַת בָּנֶ֥יהָ פָרְצ֖וּ אֶת־ בֵּ֣ית הָאֱלֹהִ֑ים וְגַם֙ כָּל־ קָדְשֵׁ֣י בֵית־ יְהוָ֔ה עָשׂ֖וּ לַבְּעָלִֽים׃
8הַמֶּ֔לֶךְ וַֽיַּעֲשׂ֖וּ אֲר֣וֹן אֶחָ֑ד וַֽיִּתְּנֻ֛הוּ בְּשַׁ֥עַר בֵּית־ יְהוָ֖ה חֽוּצָה׃
9ק֞וֹל בִּֽיהוּדָ֣ה וּבִֽירוּשָׁלִַ֗ם לְהָבִ֤יא לַֽיהוָה֙ מַשְׂאַ֞ת מֹשֶׁ֧ה עֶֽבֶד־ הָאֱלֹהִ֛ים עַל־ יִשְׂרָאֵ֖ל בַּמִּדְבָּֽר׃
10כָל־ הַשָּׂרִ֖ים וְכָל־ הָעָ֑ם וַיָּבִ֛יאוּ וַיַּשְׁלִ֥יכוּ לָאָר֖וֹן עַד־ לְכַלֵּֽה׃
11בְּעֵת֩ יָבִ֨יא אֶת־ הָֽאָר֜וֹן אֶל־ פְּקֻדַּ֣ת הַמֶּלֶךְ֮ בְּיַ֣ד הַלְוִיִּם֒ וְכִרְאוֹתָ֞ם כִּי־ רַ֣ב הַכֶּ֗סֶף וּבָ֨א סוֹפֵ֤ר הַמֶּ֙לֶךְ֙ וּפְקִיד֙ כֹּהֵ֣ן הָרֹ֔אשׁ וִיעָ֙רוּ֙ אֶת־ הָ֣אָר֔וֹן וְיִשָּׂאֻ֖הוּ וִֽישִׁיבֻ֣הוּ אֶל־ מְקֹמ֑וֹ כֹּ֤ה עָשׂוּ֙ לְי֣וֹם ׀ בְּי֔וֹם וַיַּֽאַסְפוּ־ כֶ֖סֶף לָרֹֽב׃
12הַמֶּ֜לֶךְ וִֽיהוֹיָדָ֗ע אֶל־ עוֹשֵׂה֙ מְלֶ֙אכֶת֙ עֲבוֹדַ֣ת בֵּית־ יְהוָ֔ה וַיִּֽהְי֤וּ שֹׂכְרִים֙ חֹצְבִ֣ים וְחָרָשִׁ֔ים לְחַדֵּ֖שׁ בֵּ֣ית יְהוָ֑ה וְ֠גַם לְחָרָשֵׁ֤י בַרְזֶל֙ וּנְחֹ֔שֶׁת לְחַזֵּ֖ק אֶת־ בֵּ֥ית יְהוָֽה׃
13עֹשֵׂ֣י הַמְּלָאכָ֔ה וַתַּ֧עַל אֲרוּכָ֛ה לַמְּלָאכָ֖ה בְּיָדָ֑ם וַֽיַּעֲמִ֜ידוּ אֶת־ בֵּ֧ית הָֽאֱלֹהִ֛ים עַל־ מַתְכֻּנְתּ֖וֹ וַֽיְאַמְּצֻֽהוּ׃
14הֵבִ֣יאוּ לִפְנֵי֩ הַמֶּ֨לֶךְ וִֽיהוֹיָדָ֜ע אֶת־ שְׁאָ֣ר הַכֶּ֗סֶף וַיַּעֲשֵׂ֨הוּ כֵלִ֤ים לְבֵית־ יְהוָה֙ כְּלֵ֣י שָׁרֵ֔ת וְהַעֲל֣וֹת וְכַפּ֔וֹת וּכְלֵ֥י זָהָ֖ב וָכָ֑סֶף וַ֠יִּֽהְיוּ מַעֲלִ֨ים עֹל֤וֹת בְּבֵית־ יְהוָה֙ תָּמִ֔יד כֹּ֖ל יְמֵ֥י יְהוֹיָדָֽע׃ פ
15יְהוֹיָדָ֛ע וַיִּשְׂבַּ֥ע יָמִ֖ים וַיָּמֹ֑ת בֶּן־ מֵאָ֧ה וּשְׁלֹשִׁ֛ים שָׁנָ֖ה בְּמוֹתֽוֹ׃
16בְעִיר־ דָּוִ֖יד עִם־ הַמְּלָכִ֑ים כִּֽי־ עָשָׂ֤ה טוֹבָה֙ בְּיִשְׂרָאֵ֔ל וְעִ֥ם הָאֱלֹהִ֖ים וּבֵיתֽוֹ׃ ס
17מוֹת֙ יְה֣וֹיָדָ֔ע בָּ֚אוּ שָׂרֵ֣י יְהוּדָ֔ה וַיִּֽשְׁתַּחֲו֖וּ לַמֶּ֑לֶךְ אָ֛ז שָׁמַ֥ע הַמֶּ֖לֶךְ אֲלֵיהֶֽם׃
18אֶת־ בֵּ֤ית יְהוָה֙ אֱלֹהֵ֣י אֲבוֹתֵיהֶ֔ם וַיַּֽעַבְד֥וּ אֶת־ הָאֲשֵׁרִ֖ים וְאֶת־ הָֽעֲצַבִּ֑ים וַֽיְהִי־ קֶ֗צֶף עַל־ יְהוּדָה֙ וִיר֣וּשָׁלִַ֔ם בְּאַשְׁמָתָ֖ם זֹֽאת׃
19בָּהֶם֙ נְבִאִ֔ים לַהֲשִׁיבָ֖ם אֶל־ יְהוָ֑ה וַיָּעִ֥ידוּ בָ֖ם וְלֹ֥א הֶאֱזִֽינוּ׃ ס
20אֱלֹהִ֗ים לָֽבְשָׁה֙ אֶת־ זְכַרְיָה֙ בֶּן־ יְהוֹיָדָ֣ע הַכֹּהֵ֔ן וַֽיַּעֲמֹ֖ד מֵעַ֣ל לָעָ֑ם וַיֹּ֨אמֶר לָהֶ֜ם כֹּ֣ה ׀ אָמַ֣ר הָאֱלֹהִ֗ים לָמָה֩ אַתֶּ֨ם עֹבְרִ֜ים אֶת־ מִצְוֹ֤ת יְהוָה֙ וְלֹ֣א תַצְלִ֔יחוּ כִּֽי־ עֲזַבְתֶּ֥ם אֶת־ יְהוָ֖ה וַיַּֽעֲזֹ֥ב אֶתְכֶֽם׃
21עָלָ֔יו וַיִּרְגְּמֻ֥הוּ אֶ֖בֶן בְּמִצְוַ֣ת הַמֶּ֑לֶךְ בַּחֲצַ֖ר בֵּ֥ית יְהוָֽה׃
22זָכַ֞ר יוֹאָ֣שׁ הַמֶּ֗לֶךְ הַחֶ֙סֶד֙ אֲשֶׁ֨ר עָשָׂ֜ה יְהוֹיָדָ֤ע אָבִיו֙ עִמּ֔וֹ וַֽיַּהֲרֹ֖ג אֶת־ בְּנ֑וֹ וּכְמוֹת֣וֹ אָמַ֔ר יֵ֥רֶא יְהוָ֖ה וְיִדְרֹֽשׁ׃ פ
23לִתְקוּפַ֣ת הַשָּׁנָ֗ה עָלָ֣ה עָלָיו֮ חֵ֣יל אֲרָם֒ וַיָּבֹ֗אוּ אֶל־ יְהוּדָה֙ וִיר֣וּשָׁלִַ֔ם וַיַּשְׁחִ֛יתוּ אֶת־ כָּל־ שָׂרֵ֥י הָעָ֖ם מֵעָ֑ם וְכָל־ שְׁלָלָ֥ם שִׁלְּח֖וּ לְמֶ֥לֶךְ דַּרְמָֽשֶׂק׃
24בְמִצְעַ֨ר אֲנָשִׁ֜ים בָּ֣אוּ ׀ חֵ֣יל אֲרָ֗ם וַֽיהוָה֙ נָתַ֨ן בְּיָדָ֥ם חַ֙יִל֙ לָרֹ֣ב מְאֹ֔ד כִּ֣י עָֽזְב֔וּ אֶת־ יְהוָ֖ה אֱלֹהֵ֣י אֲבוֹתֵיהֶ֑ם וְאֶת־ יוֹאָ֖שׁ עָשׂ֥וּ שְׁפָטִֽים׃
25מִמֶּ֗נּוּ כִּֽי־ עָזְב֣וּ אֹתוֹ֮ [בְּמַחֲלִיִּים כ] (בְּמַחֲלוּיִ֣ם ק) רַבִּים֒ הִתְקַשְּׁר֨וּ עָלָ֜יו עֲבָדָ֗יו בִּדְמֵי֙ בְּנֵי֙ יְהוֹיָדָ֣ע הַכֹּהֵ֔ן וַיַּֽהַרְגֻ֥הוּ עַל־ מִטָּת֖וֹ וַיָּמֹ֑ת וַֽיִּקְבְּרֻ֙הוּ֙ בְּעִ֣יר דָּוִ֔יד וְלֹ֥א קְבָרֻ֖הוּ בְּקִבְר֥וֹת הַמְּלָכִֽים׃ ס
26הַמִּתְקַשְּׁרִ֣ים עָלָ֑יו זָבָ֗ד בֶּן־ שִׁמְעָת֙ הָֽעַמּוֹנִ֔ית וִיה֣וֹזָבָ֔ד בֶּן־ שִׁמְרִ֖ית הַמּוֹאָבִֽית׃
