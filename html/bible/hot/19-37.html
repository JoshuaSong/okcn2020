1לְדָוִ֨ד ׀ אַל־ תִּתְחַ֥ר בַּמְּרֵעִ֑ים אַל־ תְּ֝קַנֵּ֗א בְּעֹשֵׂ֥י עַוְלָֽה׃
2כֶ֭חָצִיר מְהֵרָ֣ה יִמָּ֑לוּ וּכְיֶ֥רֶק דֶּ֝֗שֶׁא יִבּוֹלֽוּן׃
3בַּֽ֭יהוָה וַעֲשֵׂה־ ט֑וֹב שְׁכָן־ אֶ֝֗רֶץ וּרְעֵ֥ה אֱמוּנָֽה׃
4עַל־ יְהוָ֑ה וְיִֽתֶּן־ לְ֝ךָ֗ מִשְׁאֲלֹ֥ת לִבֶּֽךָ׃
5עַל־ יְהוָ֣ה דַּרְכֶּ֑ךָ וּבְטַ֥ח עָ֝לָ֗יו וְה֣וּא יַעֲשֶֽׂה׃
6כָא֣וֹר צִדְקֶ֑ךָ וּ֝מִשְׁפָּטֶ֗ךָ כַּֽצָּהֳרָֽיִם׃
7לַיהוָה֮ וְהִתְח֪וֹלֵ֫ל ל֥וֹ אַל־ תִּ֭תְחַר בְּמַצְלִ֣יחַ דַּרְכּ֑וֹ בְּ֝אִ֗ישׁ עֹשֶׂ֥ה מְזִמּֽוֹת׃
8מֵ֭אַף וַעֲזֹ֣ב חֵמָ֑ה אַל־ תִּ֝תְחַ֗ר אַךְ־ לְהָרֵֽעַ׃
9מְ֭רֵעִים יִכָּרֵת֑וּן וְקֹוֵ֥י יְ֝הוָ֗ה הֵ֣מָּה יִֽירְשׁוּ־ אָֽרֶץ׃
10מְ֭עַט וְאֵ֣ין רָשָׁ֑ע וְהִתְבּוֹנַ֖נְתָּ עַל־ מְקוֹמ֣וֹ וְאֵינֶֽנּוּ׃
11יִֽירְשׁוּ־ אָ֑רֶץ וְ֝הִתְעַנְּג֗וּ עַל־ רֹ֥ב שָׁלֽוֹם׃
12רָ֭שָׁע לַצַּדִּ֑יק וְחֹרֵ֖ק עָלָ֣יו שִׁנָּֽיו׃
13יִשְׂחַק־ ל֑וֹ כִּֽי־ רָ֝אָ֗ה כִּֽי־ יָבֹ֥א יוֹמֽוֹ׃
14פָּֽתְח֣וּ רְשָׁעִים֮ וְדָרְכ֪וּ קַ֫שְׁתָּ֥ם לְ֭הַפִּיל עָנִ֣י וְאֶבְי֑וֹן לִ֝טְב֗וֹחַ יִשְׁרֵי־ דָֽרֶךְ׃
15תָּב֣וֹא בְלִבָּ֑ם וְ֝קַשְּׁתוֹתָ֗ם תִּשָּׁבַֽרְנָה׃
16מְ֭עַט לַצַּדִּ֑יק מֵ֝הֲמ֗וֹן רְשָׁעִ֥ים רַבִּֽים׃
17זְרוֹע֣וֹת רְ֭שָׁעִים תִּשָּׁבַ֑רְנָה וְסוֹמֵ֖ךְ צַדִּיקִ֣ים יְהוָֽה׃
18יְ֭הוָה יְמֵ֣י תְמִימִ֑ם וְ֝נַחֲלָתָ֗ם לְעוֹלָ֥ם תִּהְיֶֽה׃
19יֵ֭בֹשׁוּ בְּעֵ֣ת רָעָ֑ה וּבִימֵ֖י רְעָב֣וֹן יִשְׂבָּֽעוּ׃
20רְשָׁעִ֨ים ׀ יֹאבֵ֗דוּ וְאֹיְבֵ֣י יְ֭הוָה כִּיקַ֣ר כָּרִ֑ים כָּל֖וּ בֶעָשָׁ֣ן כָּֽלוּ׃
21רָ֭שָׁע וְלֹ֣א יְשַׁלֵּ֑ם וְ֝צַדִּ֗יק חוֹנֵ֥ן וְנוֹתֵֽן׃
22מְ֭בֹרָכָיו יִ֣ירְשׁוּ אָ֑רֶץ וּ֝מְקֻלָּלָ֗יו יִכָּרֵֽתוּ׃
23מִֽצְעֲדֵי־ גֶ֥בֶר כּוֹנָ֗נוּ וְדַרְכּ֥וֹ יֶחְפָּֽץ׃
24יִפֹּ֥ל לֹֽא־ יוּטָ֑ל כִּֽי־ יְ֝הוָ֗ה סוֹמֵ֥ךְ יָדֽוֹ׃
25הָיִ֗יתִי גַּם־ זָ֫קַ֥נְתִּי וְֽלֹא־ רָ֭אִיתִי צַדִּ֣יק נֶעֱזָ֑ב וְ֝זַרְע֗וֹ מְבַקֶּשׁ־ לָֽחֶם׃
26הַ֭יּוֹם חוֹנֵ֣ן וּמַלְוֶ֑ה וְ֝זַרְע֗וֹ לִבְרָכָֽה׃
27מֵ֭רָע וַעֲשֵׂה־ ט֗וֹב וּשְׁכֹ֥ן לְעוֹלָֽם׃
28יְהוָ֨ה ׀ אֹ֘הֵ֤ב מִשְׁפָּ֗ט וְלֹא־ יַעֲזֹ֣ב אֶת־ חֲ֭סִידָיו לְעוֹלָ֣ם נִשְׁמָ֑רוּ וְזֶ֖רַע רְשָׁעִ֣ים נִכְרָֽת׃
29יִֽירְשׁוּ־ אָ֑רֶץ וְיִשְׁכְּנ֖וּ לָעַ֣ד עָלֶֽיהָ׃
30צַ֭דִּיק יֶהְגֶּ֣ה חָכְמָ֑ה וּ֝לְשׁוֹנ֗וֹ תְּדַבֵּ֥ר מִשְׁפָּֽט׃
31אֱלֹהָ֣יו בְּלִבּ֑וֹ לֹ֖א תִמְעַ֣ד אֲשֻׁרָֽיו׃
32רָ֭שָׁע לַצַּדִּ֑יק וּ֝מְבַקֵּ֗שׁ לַהֲמִיתוֹ׃
33לֹא־ יַעַזְבֶ֣נּוּ בְיָד֑וֹ וְלֹ֥א יַ֝רְשִׁיעֶ֗נּוּ בְּהִשָּׁפְטֽוֹ׃
34אֶל־ יְהוָ֨ה ׀ וּשְׁמֹ֬ר דַּרְכּ֗וֹ וִֽ֭ירוֹמִמְךָ לָרֶ֣שֶׁת אָ֑רֶץ בְּהִכָּרֵ֖ת רְשָׁעִ֣ים תִּרְאֶֽה׃
35רָשָׁ֣ע עָרִ֑יץ וּ֝מִתְעָרֶ֗ה כְּאֶזְרָ֥ח רַעֲנָֽן׃
36וְהִנֵּ֣ה אֵינֶ֑נּוּ וָֽ֝אֲבַקְשֵׁ֗הוּ וְלֹ֣א נִמְצָֽא׃
37תָּ֭ם וּרְאֵ֣ה יָשָׁ֑ר כִּֽי־ אַחֲרִ֖ית לְאִ֣ישׁ שָׁלֽוֹם׃
38נִשְׁמְד֣וּ יַחְדָּ֑ו אַחֲרִ֖ית רְשָׁעִ֣ים נִכְרָֽתָה׃
39צַ֭דִּיקִים מֵיְהוָ֑ה מָֽ֝עוּזָּ֗ם בְּעֵ֣ת צָרָֽה׃
