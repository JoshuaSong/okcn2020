1וּבָעֵ֣ת הַהִיא֩ יַעֲמֹ֨ד מִֽיכָאֵ֜ל הַשַּׂ֣ר הַגָּד֗וֹל הָעֹמֵד֮ עַל־ בְּנֵ֣י עַמֶּךָ֒ וְהָיְתָה֙ עֵ֣ת צָרָ֔ה אֲשֶׁ֤ר לֹֽא־ נִהְיְתָה֙ מִֽהְי֣וֹת גּ֔וֹי עַ֖ד הָעֵ֣ת הַהִ֑יא וּבָעֵ֤ת הַהִיא֙ יִמָּלֵ֣ט עַמְּךָ֔ כָּל־ הַנִּמְצָ֖א כָּת֥וּב בַּסֵּֽפֶר׃
2מִיְּשֵׁנֵ֥י אַדְמַת־ עָפָ֖ר יָקִ֑יצוּ אֵ֚לֶּה לְחַיֵּ֣י עוֹלָ֔ם וְאֵ֥לֶּה לַחֲרָפ֖וֹת לְדִרְא֥וֹן עוֹלָֽם׃ ס
3יַזְהִ֖רוּ כְּזֹ֣הַר הָרָקִ֑יעַ וּמַצְדִּיקֵי֙ הָֽרַבִּ֔ים כַּכּוֹכָבִ֖ים לְעוֹלָ֥ם וָעֶֽד׃ פ
4דָֽנִיֵּ֗אל סְתֹ֧ם הַדְּבָרִ֛ים וַחֲתֹ֥ם הַסֵּ֖פֶר עַד־ עֵ֣ת קֵ֑ץ יְשֹׁטְט֥וּ רַבִּ֖ים וְתִרְבֶּ֥ה הַדָּֽעַת׃
5אֲנִ֣י דָנִיֵּ֔אל וְהִנֵּ֛ה שְׁנַ֥יִם אֲחֵרִ֖ים עֹמְדִ֑ים אֶחָ֥ד הֵ֙נָּה֙ לִשְׂפַ֣ת הַיְאֹ֔ר וְאֶחָ֥ד הֵ֖נָּה לִשְׂפַ֥ת הַיְאֹֽר׃
6לָאִישׁ֙ לְב֣וּשׁ הַבַּדִּ֔ים אֲשֶׁ֥ר מִמַּ֖עַל לְמֵימֵ֣י הַיְאֹ֑ר עַד־ מָתַ֖י קֵ֥ץ הַפְּלָאֽוֹת׃
7אֶת־ הָאִ֣ישׁ ׀ לְב֣וּשׁ הַבַּדִּ֗ים אֲשֶׁ֣ר מִמַּעַל֮ לְמֵימֵ֣י הַיְאֹר֒ וַיָּ֨רֶם יְמִינ֤וֹ וּשְׂמֹאלוֹ֙ אֶל־ הַשָּׁמַ֔יִם וַיִּשָּׁבַ֖ע בְּחֵ֣י הָעוֹלָ֑ם כִּי֩ לְמוֹעֵ֨ד מֽוֹעֲדִ֜ים וָחֵ֗צִי וּכְכַלּ֛וֹת נַפֵּ֥ץ יַד־ עַם־ קֹ֖דֶשׁ תִּכְלֶ֥ינָה כָל־ אֵֽלֶּה׃
8שָׁמַ֖עְתִּי וְלֹ֣א אָבִ֑ין וָאֹ֣מְרָ֔ה אֲדֹנִ֕י מָ֥ה אַחֲרִ֖ית אֵֽלֶּה׃ פ
9לֵ֣ךְ דָּנִיֵּ֑אל כִּֽי־ סְתֻמִ֧ים וַחֲתֻמִ֛ים הַדְּבָרִ֖ים עַד־ עֵ֥ת קֵֽץ׃
10וְיִֽתְלַבְּנ֤וּ וְיִצָּֽרְפוּ֙ רַבִּ֔ים וְהִרְשִׁ֣יעוּ רְשָׁעִ֔ים וְלֹ֥א יָבִ֖ינוּ כָּל־ רְשָׁעִ֑ים וְהַמַּשְׂכִּלִ֖ים יָבִֽינוּ׃
11הוּסַ֣ר הַתָּמִ֔יד וְלָתֵ֖ת שִׁקּ֣וּץ שֹׁמֵ֑ם יָמִ֕ים אֶ֖לֶף מָאתַ֥יִם וְתִשְׁעִֽים׃
12הַֽמְחַכֶּ֖ה וְיַגִּ֑יעַ לְיָמִ֕ים אֶ֕לֶף שְׁלֹ֥שׁ מֵא֖וֹת שְׁלֹשִׁ֥ים וַחֲמִשָּֽׁה׃
