1וּכְהִתְפַּלֵּ֤ל עֶזְרָא֙ וּכְ֨הִתְוַדֹּת֔וֹ בֹּכֶה֙ וּמִתְנַפֵּ֔ל לִפְנֵ֖י בֵּ֣ית הָאֱלֹהִ֑ים נִקְבְּצוּ֩ אֵלָ֨יו מִיִּשְׂרָאֵ֜ל קָהָ֣ל רַב־ מְאֹ֗ד אֲנָשִׁ֤ים וְנָשִׁים֙ וִֽילָדִ֔ים כִּֽי־ בָכ֥וּ הָעָ֖ם הַרְבֵּה־ בֶֽכֶה׃ ס
2שְׁכַנְיָ֨ה בֶן־ יְחִיאֵ֜ל מִבְּנֵ֤י [עֹולָם כ] (עֵילָם֙ ק) וַיֹּ֣אמֶר לְעֶזְרָ֔א אֲנַ֙חְנוּ֙ מָעַ֣לְנוּ בֵאלֹהֵ֔ינוּ וַנֹּ֛שֶׁב נָשִׁ֥ים נָכְרִיּ֖וֹת מֵעַמֵּ֣י הָאָ֑רֶץ וְעַתָּ֛ה יֵשׁ־ מִקְוֶ֥ה לְיִשְׂרָאֵ֖ל עַל־ זֹֽאת׃
3נִֽכְרָת־ בְּרִ֣ית לֵ֠אלֹהֵינוּ לְהוֹצִ֨יא כָל־ נָשִׁ֜ים וְהַנּוֹלָ֤ד מֵהֶם֙ בַּעֲצַ֣ת אֲדֹנָ֔י וְהַחֲרֵדִ֖ים בְּמִצְוַ֣ת אֱלֹהֵ֑ינוּ וְכַתּוֹרָ֖ה יֵעָשֶֽׂה׃
4כִּֽי־ עָלֶ֥יךָ הַדָּבָ֖ר וַאֲנַ֣חְנוּ עִמָּ֑ךְ חֲזַ֖ק וַעֲשֵֽׂה׃ פ
5עֶזְרָ֡א וַיַּשְׁבַּ֣ע אֶת־ שָׂרֵי֩ הַכֹּהֲנִ֨ים הַלְוִיִּ֜ם וְכָל־ יִשְׂרָאֵ֗ל לַעֲשׂ֛וֹת כַּדָּבָ֥ר הַזֶּ֖ה וַיִּשָּׁבֵֽעוּ׃
6עֶזְרָ֗א מִלִּפְנֵי֙ בֵּ֣ית הָֽאֱלֹהִ֔ים וַיֵּ֕לֶךְ אֶל־ לִשְׁכַּ֖ת יְהוֹחָנָ֣ן בֶּן־ אֶלְיָשִׁ֑יב וַיֵּ֣לֶךְ שָׁ֗ם לֶ֤חֶם לֹֽא־ אָכַל֙ וּמַ֣יִם לֹֽא־ שָׁתָ֔ה כִּ֥י מִתְאַבֵּ֖ל עַל־ מַ֥עַל הַגּוֹלָֽה׃ ס
7ק֜וֹל בִּיהוּדָ֣ה וִירֽוּשָׁלִַ֗ם לְכֹל֙ בְּנֵ֣י הַגּוֹלָ֔ה לְהִקָּבֵ֖ץ יְרוּשָׁלִָֽם׃
8אֲשֶׁ֨ר לֹֽא־ יָב֜וֹא לִשְׁלֹ֣שֶׁת הַיָּמִ֗ים כַּעֲצַ֤ת הַשָּׂרִים֙ וְהַזְּקֵנִ֔ים יָחֳרַ֖ם כָּל־ רְכוּשׁ֑וֹ וְה֥וּא יִבָּדֵ֖ל מִקְּהַ֥ל הַגּוֹלָֽה׃ ס
9כָל־ אַנְשֵֽׁי־ יְהוּדָה֩ וּבִנְיָמִ֨ן ׀ יְרוּשָׁלִַ֜ם לִשְׁלֹ֣שֶׁת הַיָּמִ֗ים ה֛וּא חֹ֥דֶשׁ הַתְּשִׁיעִ֖י בְּעֶשְׂרִ֣ים בַּחֹ֑דֶשׁ וַיֵּשְׁב֣וּ כָל־ הָעָ֗ם בִּרְחוֹב֙ בֵּ֣ית הָאֱלֹהִ֔ים מַרְעִידִ֥ים עַל־ הַדָּבָ֖ר וּמֵהַגְּשָׁמִֽים׃ פ
10עֶזְרָ֤א הַכֹּהֵן֙ וַיֹּ֣אמֶר אֲלֵהֶ֔ם אַתֶּ֣ם מְעַלְתֶּ֔ם וַתֹּשִׁ֖יבוּ נָשִׁ֣ים נָכְרִיּ֑וֹת לְהוֹסִ֖יף עַל־ אַשְׁמַ֥ת יִשְׂרָאֵֽל׃
11תְּנ֥וּ תוֹדָ֛ה לַיהוָ֥ה אֱלֹהֵֽי־ אֲבֹתֵיכֶ֖ם וַעֲשׂ֣וּ רְצוֹנ֑וֹ וְהִבָּֽדְלוּ֙ מֵעַמֵּ֣י הָאָ֔רֶץ וּמִן־ הַנָּשִׁ֖ים הַנָּכְרִיּֽוֹת׃
12כָֽל־ הַקָּהָ֛ל וַיֹּאמְר֖וּ ק֣וֹל גָּד֑וֹל כֵּ֛ן [כְּדִבְרֶיךָ כ] (כִּדְבָרְךָ֥ ק) עָלֵ֖ינוּ לַעֲשֽׂוֹת׃
13הָעָ֥ם רָב֙ וְהָעֵ֣ת גְּשָׁמִ֔ים וְאֵ֥ין כֹּ֖חַ לַעֲמ֣וֹד בַּח֑וּץ וְהַמְּלָאכָ֗ה לֹֽא־ לְי֤וֹם אֶחָד֙ וְלֹ֣א לִשְׁנַ֔יִם כִּֽי־ הִרְבִּ֥ינוּ לִפְשֹׁ֖עַ בַּדָּבָ֥ר הַזֶּֽה׃
14נָ֣א שָׂ֠רֵינוּ לְֽכָל־ הַקָּהָ֞ל וְכֹ֣ל ׀ אֲשֶׁ֣ר בֶּעָרֵ֗ינוּ הַהֹשִׁ֞יב נָשִׁ֤ים נָכְרִיּוֹת֙ יָבֹא֙ לְעִתִּ֣ים מְזֻמָּנִ֔ים וְעִמָּהֶ֛ם זִקְנֵי־ עִ֥יר וָעִ֖יר וְשֹׁפְטֶ֑יהָ עַ֠ד לְהָשִׁ֞יב חֲר֤וֹן אַף־ אֱלֹהֵ֙ינוּ֙ מִמֶּ֔נּוּ עַ֖ד לַדָּבָ֥ר הַזֶּֽה׃ פ
15יוֹנָתָ֧ן בֶּן־ עֲשָׂהאֵ֛ל וְיַחְזְיָ֥ה בֶן־ תִּקְוָ֖ה עָמְד֣וּ עַל־ זֹ֑את וּמְשֻׁלָּ֛ם וְשַׁבְּתַ֥י הַלֵּוִ֖י עֲזָרֻֽם׃
16כֵן֮ בְּנֵ֣י הַגּוֹלָה֒ וַיִּבָּדְלוּ֩ עֶזְרָ֨א הַכֹּהֵ֜ן אֲנָשִׁ֨ים רָאשֵׁ֧י הָאָב֛וֹת לְבֵ֥ית אֲבֹתָ֖ם וְכֻלָּ֣ם בְּשֵׁמ֑וֹת וַיֵּשְׁב֗וּ בְּי֤וֹם אֶחָד֙ לַחֹ֣דֶשׁ הָעֲשִׂירִ֔י לְדַרְי֖וֹשׁ הַדָּבָֽר׃
17בַכֹּ֔ל אֲנָשִׁ֕ים הַהֹשִׁ֖יבוּ נָשִׁ֣ים נָכְרִיּ֑וֹת עַ֛ד י֥וֹם אֶחָ֖ד לַחֹ֥דֶשׁ הָרִאשֽׁוֹן׃ פ
18מִבְּנֵ֣י הַכֹּהֲנִ֔ים אֲשֶׁ֥ר הֹשִׁ֖יבוּ נָשִׁ֣ים נָכְרִיּ֑וֹת מִבְּנֵ֨י יֵשׁ֤וּעַ בֶּן־ יֽוֹצָדָק֙ וְאֶחָ֔יו מַֽעֲשֵׂיָה֙ וֶֽאֱלִיעֶ֔זֶר וְיָרִ֖יב וּגְדַלְיָֽה׃
19יָדָ֖ם לְהוֹצִ֣יא נְשֵׁיהֶ֑ם וַאֲשֵׁמִ֥ים אֵֽיל־ צֹ֖אן עַל־ אַשְׁמָתָֽם׃ ס
20אִמֵּ֔ר חֲנָ֖נִי וּזְבַדְיָֽה׃ ס
21חָרִ֑ם מַעֲשֵׂיָ֤ה וְאֵֽלִיָּה֙ וּֽשְׁמַֽעְיָ֔ה וִיחִיאֵ֖ל וְעֻזִיָּֽה׃
22פַּשְׁח֑וּר אֶלְיוֹעֵינַ֤י מַֽעֲשֵׂיָה֙ יִשְׁמָעֵ֣אל נְתַנְאֵ֔ל יוֹזָבָ֖ד וְאֶלְעָשָֽׂה׃ ס
23הַלְוִיִּ֑ם יוֹזָבָ֣ד וְשִׁמְעִ֗י וְקֵֽלָיָה֙ ה֣וּא קְלִיטָ֔א פְּתַֽחְיָ֥ה יְהוּדָ֖ה וֶאֱלִיעֶֽזֶר׃ ס
24הַמְשֹׁרְרִ֖ים אֶלְיָשִׁ֑יב וּמִן־ הַשֹּׁ֣עֲרִ֔ים שַׁלֻּ֥ם וָטֶ֖לֶם וְאוּרִֽי׃ ס
25מִבְּנֵ֣י פַרְעֹ֡שׁ רַ֠מְיָה וְיִזִּיָּ֤ה וּמַלְכִּיָּה֙ וּמִיָּמִ֣ן וְאֶלְעָזָ֔ר וּמַלְכִּיָּ֖ה וּבְנָיָֽה׃ ס
26עֵילָ֑ם מַתַּנְיָ֤ה זְכַרְיָה֙ וִיחִיאֵ֣ל וְעַבְדִּ֔י וִירֵמ֖וֹת וְאֵלִיָּֽה׃ ס
27זַתּ֑וּא אֶלְיוֹעֵנַ֤י אֶלְיָשִׁיב֙ מַתַּנְיָ֣ה וִֽירֵמ֔וֹת וְזָבָ֖ד וַעֲזִיזָֽא׃ ס
28בֵּבָ֑י יְהוֹחָנָ֥ן חֲנַנְיָ֖ה זַבַּ֥י עַתְלָֽי׃ ס
29בָּנִ֑י מְשֻׁלָּ֤ם מַלּוּךְ֙ וַעֲדָיָ֔ה יָשׁ֖וּב וּשְׁאָ֥ל [יְרֵמֹות כ] (וְרָמֹֽות׃ ק) ס
30פַּחַ֥ת מוֹאָ֖ב עַדְנָ֣א וּכְלָ֑ל בְּנָיָ֤ה מַעֲשֵׂיָה֙ מַתַּנְיָ֣ה בְצַלְאֵ֔ל וּבִנּ֖וּי וּמְנַשֶּֽׁה׃ ס
31חָרִ֑ם אֱלִיעֶ֧זֶר יִשִּׁיָּ֛ה מַלְכִּיָּ֖ה שְׁמַֽעְיָ֥ה שִׁמְעֽוֹן׃
32מַלּ֖וּךְ שְׁמַרְיָֽה׃ ס
33חָשֻׁ֑ם מַתְּנַ֤י מַתַּתָּה֙ זָבָ֣ד אֱלִיפֶ֔לֶט יְרֵמַ֥י מְנַשֶּׁ֖ה שִׁמְעִֽי׃ ס
34בָנִ֔י מַעֲדַ֥י עַמְרָ֖ם וְאוּאֵֽל׃ ס

36מְרֵמ֖וֹת אֶלְיָשִֽׁיב׃
37מַתְּנַ֖י [וְיַעֲשֹׂו כ] (וְיַעֲשָֽׂי׃ ק)
38וּבִנּ֖וּי שִׁמְעִֽי׃
39וְנָתָ֖ן וַעֲדָיָֽה׃
40שָׁשַׁ֖י שָׁרָֽי׃
