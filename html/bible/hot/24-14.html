1אֲשֶׁ֨ר הָיָ֤ה דְבַר־ יְהוָה֙ אֶֽל־ יִרְמְיָ֔הוּ עַל־ דִּבְרֵ֖י הַבַּצָּרֽוֹת׃
2יְהוּדָ֔ה וּשְׁעָרֶ֥יהָ אֻמְלְל֖וּ קָדְר֣וּ לָאָ֑רֶץ וְצִוְחַ֥ת יְרוּשָׁלִַ֖ם עָלָֽתָה׃
3שָׁלְח֥וּ [צְעֹורֵיהֶם כ] (צְעִירֵיהֶ֖ם ק) לַמָּ֑יִם בָּ֣אוּ עַל־ גֵּבִ֞ים לֹא־ מָ֣צְאוּ מַ֗יִם שָׁ֤בוּ כְלֵיהֶם֙ רֵיקָ֔ם בֹּ֥שׁוּ וְהָכְלְמ֖וּ וְחָפ֥וּ רֹאשָֽׁם׃
4הָאֲדָמָה֙ חַ֔תָּה כִּ֛י לֹא־ הָיָ֥ה גֶ֖שֶׁם בָּאָ֑רֶץ בֹּ֥שׁוּ אִכָּרִ֖ים חָפ֥וּ רֹאשָֽׁם׃
5גַם־ אַיֶּ֙לֶת֙ בַּשָּׂדֶ֔ה יָלְדָ֖ה וְעָז֑וֹב כִּ֥י לֹֽא־ הָיָ֖ה דֶּֽשֶׁא׃
6עָמְד֣וּ עַל־ שְׁפָיִ֔ם שָׁאֲפ֥וּ ר֖וּחַ כַּתַּנִּ֑ים כָּל֥וּ עֵינֵיהֶ֖ם כִּי־ אֵ֥ין עֵֽשֶׂב׃
7עֲוֹנֵ֙ינוּ֙ עָ֣נוּ בָ֔נוּ יְהוָ֕ה עֲשֵׂ֖ה לְמַ֣עַן שְׁמֶ֑ךָ כִּֽי־ רַבּ֥וּ מְשׁוּבֹתֵ֖ינוּ לְךָ֥ חָטָֽאנוּ׃
8יִשְׂרָאֵ֔ל מֽוֹשִׁיע֖וֹ בְּעֵ֣ת צָרָ֑ה לָ֤מָּה תִֽהְיֶה֙ כְּגֵ֣ר בָּאָ֔רֶץ וּכְאֹרֵ֖חַ נָטָ֥ה לָלֽוּן׃
9תִֽהְיֶה֙ כְּאִ֣ישׁ נִדְהָ֔ם כְּגִבּ֖וֹר לֹא־ יוּכַ֣ל לְהוֹשִׁ֑יעַ וְאַתָּ֧ה בְקִרְבֵּ֣נוּ יְהוָ֗ה וְשִׁמְךָ֛ עָלֵ֥ינוּ נִקְרָ֖א אַל־ תַּנִּחֵֽנוּ׃ ס
10אָמַ֨ר יְהוָ֜ה לָעָ֣ם הַזֶּ֗ה כֵּ֤ן אָֽהֲבוּ֙ לָנ֔וּעַ רַגְלֵיהֶ֖ם לֹ֣א חָשָׂ֑כוּ וַיהוָה֙ לֹ֣א רָצָ֔ם עַתָּה֙ יִזְכֹּ֣ר עֲוֹנָ֔ם וְיִפְקֹ֖ד חַטֹּאתָֽם׃ ס
11יְהוָ֖ה אֵלָ֑י אַל־ תִּתְפַּלֵּ֛ל בְּעַד־ הָעָ֥ם הַזֶּ֖ה לְטוֹבָֽה׃
12יָצֻ֗מוּ אֵינֶ֤נִּי שֹׁמֵ֙עַ֙ אֶל־ רִנָּתָ֔ם וְכִ֧י יַעֲל֛וּ עֹלָ֥ה וּמִנְחָ֖ה אֵינֶ֣נִּי רֹצָ֑ם כִּ֗י בַּחֶ֙רֶב֙ וּבָרָעָ֣ב וּבַדֶּ֔בֶר אָנֹכִ֖י מְכַלֶּ֥ה אוֹתָֽם׃ ס
13אֲהָ֣הּ ׀ אֲדֹנָ֣י יְהוִ֗ה הִנֵּ֨ה הַנְּבִאִ֜ים אֹמְרִ֤ים לָהֶם֙ לֹֽא־ תִרְא֣וּ חֶ֔רֶב וְרָעָ֖ב לֹֽא־ יִהְיֶ֣ה לָכֶ֑ם כִּֽי־ שְׁל֤וֹם אֱמֶת֙ אֶתֵּ֣ן לָכֶ֔ם בַּמָּק֖וֹם הַזֶּֽה׃ ס
14יְהוָ֜ה אֵלַ֗י שֶׁ֚קֶר הַנְּבִאִים֙ נִבְּאִ֣ים בִּשְׁמִ֔י לֹ֤א שְׁלַחְתִּים֙ וְלֹ֣א צִוִּיתִ֔ים וְלֹ֥א דִבַּ֖רְתִּי אֲלֵיהֶ֑ם חֲז֨וֹן שֶׁ֜קֶר וְקֶ֤סֶם [וֶאֱלוּל כ] (וֶֽאֱלִיל֙ ק) [וְתַרְמוּת כ] (וְתַרְמִ֣ית ק) לִבָּ֔ם הֵ֖מָּה מִֽתְנַבְּאִ֥ים לָכֶֽם׃ ס
15כֹּֽה־ אָמַ֣ר יְהוָ֗ה עַֽל־ הַנְּבִאִ֞ים הַנִּבְּאִ֣ים בִּשְׁמִי֮ וַאֲנִ֣י לֹֽא־ שְׁלַחְתִּים֒ וְהֵ֙מָּה֙ אֹֽמְרִ֔ים חֶ֣רֶב וְרָעָ֔ב לֹ֥א יִהְיֶ֖ה בָּאָ֣רֶץ הַזֹּ֑את בַּחֶ֤רֶב וּבָֽרָעָב֙ יִתַּ֔מּוּ הַנְּבִאִ֖ים הָהֵֽמָּה׃
16אֲשֶׁר־ הֵ֣מָּה נִבְּאִ֣ים לָהֶ֡ם יִֽהְי֣וּ מֻשְׁלָכִים֩ בְּחֻצ֨וֹת יְרוּשָׁלִַ֜ם מִפְּנֵ֣י ׀ הָרָעָ֣ב וְהַחֶ֗רֶב וְאֵ֤ין מְקַבֵּר֙ לָהֵ֔מָּה הֵ֣מָּה נְשֵׁיהֶ֔ם וּבְנֵיהֶ֖ם וּבְנֹֽתֵיהֶ֑ם וְשָׁפַכְתִּ֥י עֲלֵיהֶ֖ם אֶת־ רָעָתָֽם׃
17אֲלֵיהֶם֙ אֶת־ הַדָּבָ֣ר הַזֶּ֔ה תֵּרַ֨דְנָה עֵינַ֥י דִּמְעָ֛ה לַ֥יְלָה וְיוֹמָ֖ם וְאַל־ תִּדְמֶ֑ינָה כִּי֩ שֶׁ֨בֶר גָּד֜וֹל נִשְׁבְּרָ֗ה בְּתוּלַת֙ בַּת־ עַמִּ֔י מַכָּ֖ה נַחְלָ֥ה מְאֹֽד׃
18יָצָ֣אתִי הַשָּׂדֶ֗ה וְהִנֵּה֙ חַלְלֵי־ חֶ֔רֶב וְאִם֙ בָּ֣אתִי הָעִ֔יר וְהִנֵּ֖ה תַּחֲלוּאֵ֣י רָעָ֑ב כִּֽי־ גַם־ נָבִ֧יא גַם־ כֹּהֵ֛ן סָחֲר֥וּ אֶל־ אֶ֖רֶץ וְלֹ֥א יָדָֽעוּ׃ ס
19מָאַ֜סְתָּ אֶת־ יְהוּדָ֗ה אִם־ בְּצִיּוֹן֙ גָּעֲלָ֣ה נַפְשֶׁ֔ךָ מַדּ֙וּעַ֙ הִכִּיתָ֔נוּ וְאֵ֥ין לָ֖נוּ מַרְפֵּ֑א קַוֵּ֤ה לְשָׁלוֹם֙ וְאֵ֣ין ט֔וֹב וּלְעֵ֥ת מַרְפֵּ֖א וְהִנֵּ֥ה בְעָתָֽה׃
20יְהוָ֛ה רִשְׁעֵ֖נוּ עֲוֹ֣ן אֲבוֹתֵ֑ינוּ כִּ֥י חָטָ֖אנוּ לָֽךְ׃
21תִּנְאַץ֙ לְמַ֣עַן שִׁמְךָ֔ אַל־ תְּנַבֵּ֖ל כִּסֵּ֣א כְבוֹדֶ֑ךָ זְכֹ֕ר אַל־ תָּפֵ֥ר בְּרִֽיתְךָ֖ אִתָּֽנוּ׃
