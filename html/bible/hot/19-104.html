1בָּרֲכִ֥י נַפְשִׁ֗י אֶת־ יְה֫וָ֥ה יְהוָ֣ה אֱ֭לֹהַי גָּדַ֣לְתָּ מְּאֹ֑ד ה֭וֹד וְהָדָ֣ר לָבָֽשְׁתָּ׃
2א֭וֹר כַּשַּׂלְמָ֑ה נוֹטֶ֥ה שָׁ֝מַ֗יִם כַּיְרִיעָֽה׃
3בַמַּ֗יִם עֲ‍ֽלִיּ֫וֹתָ֥יו הַשָּׂם־ עָבִ֥ים רְכוּב֑וֹ הַֽ֝מְהַלֵּ֗ךְ עַל־ כַּנְפֵי־ רֽוּחַ׃
4מַלְאָכָ֣יו רוּח֑וֹת מְ֝שָׁרְתָ֗יו אֵ֣שׁ לֹהֵֽט׃
5אֶ֭רֶץ עַל־ מְכוֹנֶ֑יהָ בַּל־ תִּ֝מּ֗וֹט עוֹלָ֥ם וָעֶֽד׃
6כַּלְּב֣וּשׁ כִּסִּית֑וֹ עַל־ הָ֝רִ֗ים יַֽעַמְדוּ־ מָֽיִם׃
7גַּעֲרָ֣תְךָ֣ יְנוּס֑וּן מִן־ ק֥וֹל רַֽ֝עַמְךָ֗ יֵחָפֵזֽוּן׃
8הָ֭רִים יֵרְד֣וּ בְקָע֑וֹת אֶל־ מְ֝ק֗וֹם זֶ֤ה ׀ יָסַ֬דְתָּ לָהֶֽם׃
9שַׂ֭מְתָּ בַּל־ יַֽעֲבֹר֑וּן בַּל־ יְ֝שׁוּב֗וּן לְכַסּ֥וֹת הָאָֽרֶץ׃
10מַ֭עְיָנִים בַּנְּחָלִ֑ים בֵּ֥ין הָ֝רִ֗ים יְהַלֵּכֽוּן׃
11כָּל־ חַיְת֣וֹ שָׂדָ֑י יִשְׁבְּר֖וּ פְרָאִ֣ים צְמָאָֽם׃
12עוֹף־ הַשָּׁמַ֣יִם יִשְׁכּ֑וֹן מִבֵּ֥ין עֳ֝פָאיִ֗ם יִתְּנוּ־ קֽוֹל׃
13הָ֭רִים מֵעֲלִיּוֹתָ֑יו מִפְּרִ֥י מַ֝עֲשֶׂ֗יךָ תִּשְׂבַּ֥ע הָאָֽרֶץ׃
14חָצִ֨יר ׀ לַבְּהֵמָ֗ה וְ֭עֵשֶׂב לַעֲבֹדַ֣ת הָאָדָ֑ם לְה֥וֹצִיא לֶ֝֗חֶם מִן־ הָאָֽרֶץ׃
15יְשַׂמַּ֬ח לְֽבַב־ אֱנ֗וֹשׁ לְהַצְהִ֣יל פָּנִ֣ים מִשָּׁ֑מֶן וְ֝לֶ֗חֶם לְֽבַב־ אֱנ֥וֹשׁ יִסְעָֽד׃
16עֲצֵ֣י יְהוָ֑ה אַֽרְזֵ֥י לְ֝בָנ֗וֹן אֲשֶׁ֣ר נָטָֽע׃
17שָׁ֭ם צִפֳּרִ֣ים יְקַנֵּ֑נוּ חֲ֝סִידָ֗ה בְּרוֹשִׁ֥ים בֵּיתָֽהּ׃
18הַ֭גְּבֹהִים לַיְּעֵלִ֑ים סְ֝לָעִ֗ים מַחְסֶ֥ה לַֽשְׁפַנִּֽים׃
19יָ֭רֵחַ לְמוֹעֲדִ֑ים שֶׁ֝֗מֶשׁ יָדַ֥ע מְבוֹאֽוֹ׃
20חֹ֭שֶׁךְ וִ֣יהִי לָ֑יְלָה בּֽוֹ־ תִ֝רְמֹ֗שׂ כָּל־ חַיְתוֹ־ יָֽעַר׃
21שֹׁאֲגִ֣ים לַטָּ֑רֶף וּלְבַקֵּ֖שׁ מֵאֵ֣ל אָכְלָֽם׃
22הַ֭שֶּׁמֶשׁ יֵאָסֵפ֑וּן וְאֶל־ מְ֝עוֹנֹתָ֗ם יִרְבָּצֽוּן׃
23אָדָ֣ם לְפָעֳל֑וֹ וְֽלַעֲבֹ֖דָת֣וֹ עֲדֵי־ עָֽרֶב׃
24רַבּ֬וּ מַעֲשֶׂ֨יךָ ׀ יְֽהוָ֗ה כֻּ֭לָּם בְּחָכְמָ֣ה עָשִׂ֑יתָ מָלְאָ֥ה הָ֝אָ֗רֶץ קִנְיָנֶֽךָ׃
25הַיָּ֥ם גָּדוֹל֮ וּרְחַ֪ב יָ֫דָ֥יִם שָֽׁם־ רֶ֭מֶשׂ וְאֵ֣ין מִסְפָּ֑ר חַיּ֥וֹת קְ֝טַנּ֗וֹת עִם־ גְּדֹלֽוֹת׃
26אֳנִיּ֣וֹת יְהַלֵּכ֑וּן לִ֝וְיָתָ֗ן זֶֽה־ יָצַ֥רְתָּ לְשַֽׂחֶק־ בּֽוֹ׃
27אֵלֶ֣יךָ יְשַׂבֵּר֑וּן לָתֵ֖ת אָכְלָ֣ם בְּעִתּֽוֹ׃
28לָ֭הֶם יִלְקֹט֑וּן תִּפְתַּ֥ח יָֽ֝דְךָ֗ יִשְׂבְּע֥וּן טֽוֹב׃
29פָּנֶיךָ֮ יִֽבָּהֵ֫ל֥וּן תֹּסֵ֣ף ר֭וּחָם יִגְוָע֑וּן וְֽאֶל־ עֲפָרָ֥ם יְשׁוּבֽוּן׃
30ר֭וּחֲךָ יִבָּרֵא֑וּן וּ֝תְחַדֵּ֗שׁ פְּנֵ֣י אֲדָמָֽה׃
31כְב֣וֹד יְהוָ֣ה לְעוֹלָ֑ם יִשְׂמַ֖ח יְהוָ֣ה בְּמַעֲשָֽׂיו׃
32לָ֭אָרֶץ וַתִּרְעָ֑ד יִגַּ֖ע בֶּהָרִ֣ים וְֽיֶעֱשָֽׁנוּ׃
33לַיהוָ֣ה בְּחַיָּ֑י אֲזַמְּרָ֖ה לֵאלֹהַ֣י בְּעוֹדִֽי׃
34עָלָ֣יו שִׂיחִ֑י אָ֝נֹכִ֗י אֶשְׂמַ֥ח בַּיהוָֽה׃
