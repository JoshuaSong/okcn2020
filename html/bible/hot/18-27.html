1וַיֹּ֣סֶף אִ֭יּוֹב שְׂאֵ֥ת מְשָׁל֗וֹ וַיֹּאמַֽר׃
2אֵ֭ל הֵסִ֣יר מִשְׁפָּטִ֑י וְ֝שַׁדַּ֗י הֵמַ֥ר נַפְשִֽׁי׃ כִּֽי־ כָל־
3נִשְׁמָתִ֣י בִ֑י וְר֖וּחַ אֱל֣וֹהַּ בְּאַפִּֽי׃
4תְּדַבֵּ֣רְנָה שְׂפָתַ֣י עַוְלָ֑ה וּ֝לְשׁוֹנִ֗י אִם־ יֶהְגֶּ֥ה רְמִיָּֽה׃
5לִּי֮ אִם־ אַצְדִּ֪יק אֶ֫תְכֶ֥ם עַד־ אֶגְוָ֑ע לֹא־ אָסִ֖יר תֻּמָּתִ֣י מִמֶּֽנִּי׃
6הֶ֭חֱזַקְתִּי וְלֹ֣א אַרְפֶּ֑הָ לֹֽא־ יֶחֱרַ֥ף לְ֝בָבִ֗י מִיָּמָֽי׃
7כְ֭רָשָׁע אֹ֣יְבִ֑י וּמִתְקוֹמְמִ֥י כְעַוָּֽל׃
8מַה־ תִּקְוַ֣ת חָ֭נֵף כִּ֣י יִבְצָ֑ע כִּ֤י יֵ֖שֶׁל אֱל֣וֹהַּ נַפְשֽׁוֹ׃
9יִשְׁמַ֥ע ׀ אֵ֑ל כִּֽי־ תָב֖וֹא עָלָ֣יו צָרָֽה׃ אִם־ עַל־
10יִתְעַנָּ֑ג יִקְרָ֖א אֱל֣וֹהַּ בְּכָל־ עֵֽת׃
11אֶתְכֶ֣ם בְּיַד־ אֵ֑ל אֲשֶׁ֥ר עִם־ שַׁ֝דַּ֗י לֹ֣א אֲכַחֵֽד׃
12אַתֶּ֣ם כֻּלְּכֶ֣ם חֲזִיתֶ֑ם וְלָמָּה־ זֶּ֝֗ה הֶ֣בֶל תֶּהְבָּֽלוּ׃
13חֵֽלֶק־ אָדָ֖ם רָשָׁ֥ע ׀ עִם־ אֵ֑ל וְֽנַחֲלַ֥ת עָ֝רִיצִ֗ים מִשַּׁדַּ֥י יִקָּֽחוּ׃
14יִרְבּ֣וּ בָנָ֣יו לְמוֹ־ חָ֑רֶב וְ֝צֶאֱצָאָ֗יו לֹ֣א יִשְׂבְּעוּ־ לָֽחֶם׃
15כ] (רִידָיו ק) בַּמָּ֣וֶת יִקָּבֵ֑רוּ וְ֝אַלְמְנֹתָ֗יו לֹ֣א תִבְכֶּֽינָה׃
16יִצְבֹּ֣ר כֶּעָפָ֣ר כָּ֑סֶף וְ֝כַחֹ֗מֶר יָכִ֥ין מַלְבּֽוּשׁ׃
17וְצַדִּ֣יק יִלְבָּ֑שׁ וְ֝כֶ֗סֶף נָקִ֥י יַחֲלֹֽק׃
18כָעָ֣שׁ בֵּית֑וֹ וּ֝כְסֻכָּ֗ה עָשָׂ֥ה נֹצֵֽר׃
19יִ֭שְׁכַּב וְלֹ֣א יֵאָסֵ֑ף עֵינָ֖יו פָּקַ֣ח וְאֵינֶֽנּוּ׃
20כַ֭מַּיִם בַּלָּה֑וֹת לַ֝֗יְלָה גְּנָבַ֥תּוּ סוּפָֽה׃
21קָדִ֣ים וְיֵלַ֑ךְ וִֽ֝ישָׂעֲרֵ֗הוּ מִמְּקֹמֽוֹ׃
22עָ֭לָיו וְלֹ֣א יַחְמֹ֑ל מִ֝יָּד֗וֹ בָּר֥וֹחַ יִבְרָֽח׃
