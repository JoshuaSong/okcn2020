1וַֽיְהִי־ אִ֥ישׁ מֵֽהַר־ אֶפְרָ֖יִם וּשְׁמ֥וֹ מִיכָֽיְהוּ׃
2לְאִמּ֡וֹ אֶלֶף֩ וּמֵאָ֨ה הַכֶּ֜סֶף אֲשֶׁ֣ר לֻֽקַּֽח־ לָ֗ךְ [וְאַתִּי כ] (וְאַ֤תְּ ק) אָלִית֙ וְגַם֙ אָמַ֣רְתְּ בְּאָזְנַ֔י הִנֵּֽה־ הַכֶּ֥סֶף אִתִּ֖י אֲנִ֣י לְקַחְתִּ֑יו וַתֹּ֣אמֶר אִמּ֔וֹ בָּר֥וּךְ בְּנִ֖י לַיהוָֽה׃
3אֶת־ אֶֽלֶף־ וּמֵאָ֥ה הַכֶּ֖סֶף לְאִמּ֑וֹ וַתֹּ֣אמֶר אִמּ֡וֹ הַקְדֵּ֣שׁ הִקְדַּ֣שְׁתִּי אֶת־ הַכֶּסֶף֩ לַיהוָ֨ה מִיָּדִ֜י לִבְנִ֗י לַֽעֲשׂוֹת֙ פֶּ֣סֶל וּמַסֵּכָ֔ה וְעַתָּ֖ה אֲשִׁיבֶ֥נּוּ לָֽךְ׃
4אֶת־ הַכֶּ֖סֶף לְאִמּ֑וֹ וַתִּקַּ֣ח אִמּוֹ֩ מָאתַ֨יִם כֶּ֜סֶף וַתִּתְּנֵ֣הוּ לַצּוֹרֵ֗ף וַֽיַּעֲשֵׂ֙הוּ֙ פֶּ֣סֶל וּמַסֵּכָ֔ה וַיְהִ֖י בְּבֵ֥ית מִיכָֽיְהוּ׃
5מִיכָ֔ה ל֖וֹ בֵּ֣ית אֱלֹהִ֑ים וַיַּ֤עַשׂ אֵפוֹד֙ וּתְרָפִ֔ים וַיְמַלֵּ֗א אֶת־ יַ֤ד אַחַד֙ מִבָּנָ֔יו וַיְהִי־ ל֖וֹ לְכֹהֵֽן׃
6הָהֵ֔ם אֵ֥ין מֶ֖לֶךְ בְּיִשְׂרָאֵ֑ל אִ֛ישׁ הַיָּשָׁ֥ר בְּעֵינָ֖יו יַעֲשֶֽׂה׃ פ
7נַ֗עַר מִבֵּ֥ית לֶ֙חֶם֙ יְהוּדָ֔ה מִמִּשְׁפַּ֖חַת יְהוּדָ֑ה וְה֥וּא לֵוִ֖י וְה֥וּא גָֽר־ שָֽׁם׃
8הָאִ֜ישׁ מֵהָעִ֗יר מִבֵּ֥ית לֶ֙חֶם֙ יְהוּדָ֔ה לָג֖וּר בַּאֲשֶׁ֣ר יִמְצָ֑א וַיָּבֹ֧א הַר־ אֶפְרַ֛יִם עַד־ בֵּ֥ית מִיכָ֖ה לַעֲשׂ֥וֹת דַּרְכּֽוֹ׃
9ל֥וֹ מִיכָ֖ה מֵאַ֣יִן תָּב֑וֹא וַיֹּ֨אמֶר אֵלָ֜יו לֵוִ֣י אָנֹ֗כִי מִבֵּ֥ית לֶ֙חֶם֙ יְהוּדָ֔ה וְאָנֹכִ֣י הֹלֵ֔ךְ לָג֖וּר בַּאֲשֶׁ֥ר אֶמְצָֽא׃
10ל֨וֹ מִיכָ֜ה שְׁבָ֣ה עִמָּדִ֗י וֶֽהְיֵה־ לִי֮ לְאָ֣ב וּלְכֹהֵן֒ וְאָנֹכִ֨י אֶֽתֶּן־ לְךָ֜ עֲשֶׂ֤רֶת כֶּ֙סֶף֙ לַיָּמִ֔ים וְעֵ֥רֶךְ בְּגָדִ֖ים וּמִחְיָתֶ֑ךָ וַיֵּ֖לֶךְ הַלֵּוִֽי׃
11הַלֵּוִ֖י לָשֶׁ֣בֶת אֶת־ הָאִ֑ישׁ וַיְהִ֤י הַנַּ֙עַר֙ ל֔וֹ כְּאַחַ֖ד מִבָּנָֽיו׃
12מִיכָה֙ אֶת־ יַ֣ד הַלֵּוִ֔י וַיְהִי־ ל֥וֹ הַנַּ֖עַר לְכֹהֵ֑ן וַיְהִ֖י בְּבֵ֥ית מִיכָֽה׃
