1וַיֹּ֖אמֶר אֵלָ֑י בֶּן־ אָדָם֙ עֲמֹ֣ד עַל־ רַגְלֶ֔יךָ וַאֲדַבֵּ֖ר אֹתָֽךְ׃
2בִ֣י ר֗וּחַ כַּֽאֲשֶׁר֙ דִּבֶּ֣ר אֵלַ֔י וַתַּעֲמִדֵ֖נִי עַל־ רַגְלָ֑י וָאֶשְׁמַ֕ע אֵ֖ת מִדַּבֵּ֥ר אֵלָֽי׃ פ
3אֵלַ֗י בֶּן־ אָדָם֙ שׁוֹלֵ֨חַ אֲנִ֤י אֽוֹתְךָ֙ אֶל־ בְּנֵ֣י יִשְׂרָאֵ֔ל אֶל־ גּוֹיִ֥ם הַמּוֹרְדִ֖ים אֲשֶׁ֣ר מָרְדוּ־ בִ֑י הֵ֤מָּה וַאֲבוֹתָם֙ פָּ֣שְׁעוּ בִ֔י עַד־ עֶ֖צֶם הַיּ֥וֹם הַזֶּֽה׃
4קְשֵׁ֤י פָנִים֙ וְחִזְקֵי־ לֵ֔ב אֲנִ֛י שׁוֹלֵ֥חַ אוֹתְךָ֖ אֲלֵיהֶ֑ם וְאָמַרְתָּ֣ אֲלֵיהֶ֔ם כֹּ֥ה אָמַ֖ר אֲדֹנָ֥י יְהֹוִֽה׃
5אִם־ יִשְׁמְע֣וּ וְאִם־ יֶחְדָּ֔לוּ כִּ֛י בֵּ֥ית מְרִ֖י הֵ֑מָּה וְיָ֣דְע֔וּ כִּ֥י נָבִ֖יא הָיָ֥ה בְתוֹכָֽם׃ פ
6בֶן־ אָ֠דָם אַל־ תִּירָ֨א מֵהֶ֜ם וּמִדִּבְרֵיהֶ֣ם אַל־ תִּירָ֗א כִּ֣י סָרָבִ֤ים וְסַלּוֹנִים֙ אוֹתָ֔ךְ וְאֶל־ עַקְרַבִּ֖ים אַתָּ֣ה יוֹשֵׁ֑ב מִדִּבְרֵיהֶ֤ם אַל־ תִּירָא֙ וּמִפְּנֵיהֶ֣ם אַל־ תֵּחָ֔ת כִּ֛י בֵּ֥ית מְרִ֖י הֵֽמָּה׃
7אֶת־ דְּבָרַי֙ אֲלֵיהֶ֔ם אִֽם־ יִשְׁמְע֖וּ וְאִם־ יֶחְדָּ֑לוּ כִּ֥י מְרִ֖י הֵֽמָּה׃ פ
8בֶן־ אָדָ֗ם שְׁמַע֙ אֵ֤ת אֲשֶׁר־ אֲנִי֙ מְדַבֵּ֣ר אֵלֶ֔יךָ אַל־ תְּהִי־ מֶ֖רִי כְּבֵ֣ית הַמֶּ֑רִי פְּצֵ֣ה פִ֔יךָ וֶאֱכֹ֕ל אֵ֥ת אֲשֶׁר־ אֲנִ֖י נֹתֵ֥ן אֵלֶֽיךָ׃
9וְהִנֵּה־ יָ֖ד שְׁלוּחָ֣ה אֵלָ֑י וְהִנֵּה־ ב֖וֹ מְגִלַּת־ סֵֽפֶר׃
