1אַף־ לְ֭זֹאת יֶחֱרַ֣ד לִבִּ֑י וְ֝יִתַּ֗ר מִמְּקוֹמֽוֹ׃
2שָׁמ֣וֹעַ בְּרֹ֣גֶז קֹל֑וֹ וְ֝הֶ֗גֶה מִפִּ֥יו יֵצֵֽא׃ תַּֽחַת־ כָּל־
3יִשְׁרֵ֑הוּ וְ֝אוֹר֗וֹ עַל־ כַּנְפ֥וֹת הָאָֽרֶץ׃
4יִשְׁאַג־ ק֗וֹל יַ֭רְעֵם בְּק֣וֹל גְּאוֹנ֑וֹ וְלֹ֥א יְ֝עַקְּבֵ֗ם כִּֽי־ יִשָּׁמַ֥ע קוֹלֽוֹ׃
5אֵ֣ל בְּ֭קוֹלוֹ נִפְלָא֑וֹת עֹשֶׂ֥ה גְ֝דֹל֗וֹת וְלֹ֣א נֵדָֽע׃
6לַשֶּׁ֨לַג ׀ יֹאמַ֗ר הֱוֵ֫א אָ֥רֶץ וְגֶ֥שֶׁם מָטָ֑ר וְ֝גֶ֗שֶׁם מִטְר֥וֹת עֻזּֽוֹ׃ בְּיַד־ כָּל־
7יַחְתּ֑וֹם לָ֝דַ֗עַת כָּל־ אַנְשֵׁ֥י מַעֲשֵֽׂהוּ׃
8חַיָּ֣ה בְמוֹ־ אָ֑רֶב וּבִמְע֖וֹנֹתֶ֣יהָ תִשְׁכֹּֽן׃
9הַ֭חֶדֶר תָּב֣וֹא סוּפָ֑ה וּֽמִמְּזָרִ֥ים קָרָֽה׃
10אֵ֥ל יִתֶּן־ קָ֑רַח וְרֹ֖חַב מַ֣יִם בְּמוּצָֽק׃
11בְּ֭רִי יַטְרִ֣יחַ עָ֑ב יָ֝פִ֗יץ עֲנַ֣ן אוֹרֽוֹ׃
12מְסִבּ֨וֹת ׀ מִתְהַפֵּ֣ךְ [בְּתַחְבּוּלָתֹו כ] (בְּתַחְבּוּלֹתָ֣יו ק) לְפָעֳלָ֑ם כֹּ֖ל אֲשֶׁ֥ר יְצַוֵּ֓ם ׀ עַל־ פְּנֵ֖י תֵבֵ֣ל אָֽרְצָה׃
13לְשֵׁ֥בֶט אִם־ לְאַרְצ֑וֹ אִם־ לְ֝חֶ֗סֶד יַמְצִאֵֽהוּ׃
14זֹּ֣את אִיּ֑וֹב עֲ֝מֹ֗ד וְהִתְבּוֹנֵ֤ן ׀ נִפְלְא֬וֹת אֵֽל׃
15בְּשׂוּם־ אֱל֣וֹהַּ עֲלֵיהֶ֑ם וְ֝הוֹפִ֗יעַ א֣וֹר עֲנָנֽוֹ׃
16עַל־ מִפְלְשֵׂי־ עָ֑ב מִ֝פְלְא֗וֹת תְּמִ֣ים דֵּעִֽים׃
17בְּגָדֶ֥יךָ חַמִּ֑ים בְּהַשְׁקִ֥ט אֶ֝֗רֶץ מִדָּרֽוֹם׃
18עִ֭מּוֹ לִשְׁחָקִ֑ים חֲ֝זָקִ֗ים כִּרְאִ֥י מוּצָֽק׃
19מַה־ נֹּ֣אמַר ל֑וֹ לֹ֥א־ נַ֝עֲרֹ֗ךְ מִפְּנֵי־ חֹֽשֶׁךְ׃
20ל֭וֹ כִּ֣י אֲדַבֵּ֑ר אִֽם־ אָ֥מַר אִ֝֗ישׁ כִּ֣י יְבֻלָּֽע׃
21לֹ֘א רָ֤אוּ א֗וֹר בָּהִ֣יר ה֭וּא בַּשְּׁחָקִ֑ים וְר֥וּחַ עָ֝בְרָ֗ה וַֽתְּטַהֲרֵֽם׃
22זָהָ֣ב יֶֽאֱתֶ֑ה עַל־ אֱ֝ל֗וֹהַּ נ֣וֹרָא הֽוֹד׃
23לֹֽא־ מְ֭צָאנֻהוּ שַׂגִּיא־ כֹ֑חַ וּמִשְׁפָּ֥ט וְרֹב־ צְ֝דָקָ֗ה לֹ֣א יְעַנֶּֽה׃
