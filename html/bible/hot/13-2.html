1אֵ֖לֶּה בְּנֵ֣י יִשְׂרָאֵ֑ל רְאוּבֵ֤ן שִׁמְעוֹן֙ לֵוִ֣י וִיהוּדָ֔ה יִשָׂשכָ֖ר וּזְבֻלֽוּן׃
2יוֹסֵ֣ף וּבִנְיָמִ֔ן נַפְתָּלִ֖י גָּ֥ד וְאָשֵֽׁר׃ ס
3יְהוּדָ֗ה עֵ֤ר וְאוֹנָן֙ וְשֵׁלָ֔ה שְׁלוֹשָׁה֙ נ֣וֹלַד ל֔וֹ מִבַּת־ שׁ֖וּעַ הַֽכְּנַעֲנִ֑ית וַיְהִ֞י עֵ֣ר ׀ בְּכ֣וֹר יְהוּדָ֗ה רַ֛ע בְּעֵינֵ֥י יְהוָ֖ה וַיְמִיתֵֽהוּ׃ ס
4כַּלָּת֔וֹ יָ֥לְדָה לּ֖וֹ אֶת־ פֶּ֣רֶץ וְאֶת־ זָ֑רַח כָּל־ בְּנֵ֥י יְהוּדָ֖ה חֲמִשָּֽׁה׃ ס
5פֶ֖רֶץ חֶצְר֥וֹן וְחָמֽוּל׃ ס
6זֶ֗רַח זִ֠מְרִי וְאֵיתָ֧ן וְהֵימָ֛ן וְכַלְכֹּ֥ל וָדָ֖רַע כֻּלָּ֥ם חֲמִשָּֽׁה׃ ס
7כַּרְמִ֑י עָכָר֙ עוֹכֵ֣ר יִשְׂרָאֵ֔ל אֲשֶׁ֥ר מָעַ֖ל בַּחֵֽרֶם׃ ס
8אֵיתָ֖ן עֲזַרְיָֽה׃
9חֶצְר֖וֹן אֲשֶׁ֣ר נוֹלַד־ ל֑וֹ אֶת־ יְרַחְמְאֵ֥ל וְאֶת־ רָ֖ם וְאֶת־ כְּלוּבָֽי׃
10הוֹלִ֣יד אֶת־ עַמִּינָדָ֑ב וְעַמִּינָדָב֙ הוֹלִ֣יד אֶת־ נַחְשׁ֔וֹן נְשִׂ֖יא בְּנֵ֥י יְהוּדָֽה׃
11הוֹלִ֣יד אֶת־ שַׂלְמָ֔א וְשַׂלְמָ֖א הוֹלִ֥יד אֶת־ בֹּֽעַז׃
12הוֹלִ֣יד אֶת־ עוֹבֵ֔ד וְעוֹבֵ֖ד הוֹלִ֥יד אֶת־ יִשָֽׁי׃
13הוֹלִ֥יד אֶת־ בְּכֹר֖וֹ אֶת־ אֱלִיאָ֑ב וַאֲבִינָדָב֙ הַשֵּׁנִ֔י וְשִׁמְעָ֖א הַשְּׁלִישִֽׁי׃
14הָֽרְבִיעִ֔י רַדַּ֖י הַחֲמִישִֽׁי׃
15הַשִּׁשִּׁ֔י דָּוִ֖יד הַשְּׁבִעִֽי׃
16כ] (וְאַחְיֹותֵיהֶ֖ם ק) צְרוּיָ֣ה וַאֲבִיגָ֑יִל וּבְנֵ֣י צְרוּיָ֗ה אַבְשַׁ֛י וְיוֹאָ֥ב וַעֲשָׂה־ אֵ֖ל שְׁלֹשָֽׁה׃
17יָלְדָ֖ה אֶת־ עֲמָשָׂ֑א וַאֲבִ֣י עֲמָשָׂ֔א יֶ֖תֶר הַיִּשְׁמְעֵאלִֽי׃
18בֶּן־ חֶצְר֗וֹן הוֹלִ֛יד אֶת־ עֲזוּבָ֥ה אִשָּׁ֖ה וְאֶת־ יְרִיע֑וֹת וְאֵ֣לֶּה בָנֶ֔יהָ יֵ֥שֶׁר וְשׁוֹבָ֖ב וְאַרְדּֽוֹן׃
19עֲזוּבָ֑ה וַיִּֽקַּֽח־ ל֤וֹ כָלֵב֙ אֶת־ אֶפְרָ֔ת וַתֵּ֥לֶד ל֖וֹ אֶת־ חֽוּר׃
20הוֹלִ֣יד אֶת־ אוּרִ֔י וְאוּרִ֖י הוֹלִ֥יד אֶת־ בְּצַלְאֵֽל׃ ס
21בָּ֤א חֶצְרוֹן֙ אֶל־ בַּת־ מָכִיר֙ אֲבִ֣י גִלְעָ֔ד וְה֣וּא לְקָחָ֔הּ וְה֖וּא בֶּן־ שִׁשִּׁ֣ים שָׁנָ֑ה וַתֵּ֥לֶד ל֖וֹ אֶת־ שְׂגֽוּב׃
22הוֹלִ֣יד אֶת־ יָאִ֑יר וַֽיְהִי־ ל֗וֹ עֶשְׂרִ֤ים וְשָׁלוֹשׁ֙ עָרִ֔ים בְּאֶ֖רֶץ הַגִּלְעָֽד׃
23גְּשֽׁוּר־ וַ֠אֲרָם אֶת־ חַוֹּ֨ת יָאִ֧יר מֵאִתָּ֛ם אֶת־ קְנָ֥ת וְאֶת־ בְּנֹתֶ֖יהָ שִׁשִּׁ֣ים עִ֑יר כָּל־ אֵ֕לֶּה בְּנֵ֖י מָכִ֥יר אֲבִי־ גִלְעָֽד׃
24מוֹת־ חֶצְר֖וֹן בְּכָלֵ֣ב אֶפְרָ֑תָה וְאֵ֤שֶׁת חֶצְרוֹן֙ אֲבִיָּ֔ה וַתֵּ֣לֶד ל֔וֹ אֶת־ אַשְׁח֖וּר אֲבִ֥י תְקֽוֹעַ׃
25בְנֵי־ יְרַחְמְאֵ֛ל בְּכ֥וֹר חֶצְר֖וֹן הַבְּכ֣וֹר ׀ רָ֑ם וּבוּנָ֥ה וָאֹ֛רֶן וָאֹ֖צֶם אֲחִיָּֽה׃
26אִשָּׁ֥ה אַחֶ֛רֶת לִֽירַחְמְאֵ֖ל וּשְׁמָ֣הּ עֲטָרָ֑ה הִ֖יא אֵ֥ם אוֹנָֽם׃ ס
27בְנֵי־ רָ֖ם בְּכ֣וֹר יְרַחְמְאֵ֑ל מַ֥עַץ וְיָמִ֖ין וָעֵֽקֶר׃
28בְנֵי־ אוֹנָ֖ם שַׁמַּ֣י וְיָדָ֑ע וּבְנֵ֣י שַׁמַּ֔י נָדָ֖ב וַאֲבִישֽׁוּר׃
29אֵ֥שֶׁת אֲבִישׁ֖וּר אֲבִיהָ֑יִל וַתֵּ֣לֶד ל֔וֹ אֶת־ אַחְבָּ֖ן וְאֶת־ מוֹלִֽיד׃
30נָדָ֖ב סֶ֣לֶד וְאַפָּ֑יִם וַיָּ֥מָת סֶ֖לֶד לֹ֥א בָנִֽים׃ ס
31אַפַּ֖יִם יִשְׁעִ֑י וּבְנֵ֤י יִשְׁעִי֙ שֵׁשָׁ֔ן וּבְנֵ֥י שֵׁשָׁ֖ן אַחְלָֽי׃
32יָדָע֙ אֲחִ֣י שַׁמַּ֔י יֶ֖תֶר וְיוֹנָתָ֑ן וַיָּ֥מָת יֶ֖תֶר לֹ֥א בָנִֽים׃ ס
33יוֹנָתָ֖ן פֶּ֣לֶת וְזָזָ֑א אֵ֥לֶּה הָי֖וּ בְּנֵ֥י יְרַחְמְאֵֽל׃
34הָיָ֧ה לְשֵׁשָׁ֛ן בָּנִ֖ים כִּ֣י אִם־ בָּנ֑וֹת וּלְשֵׁשָׁ֛ן עֶ֥בֶד מִצְרִ֖י וּשְׁמ֥וֹ יַרְחָֽע׃
35שֵׁשָׁ֧ן אֶת־ בִּתּ֛וֹ לְיַרְחָ֥ע עַבְדּ֖וֹ לְאִשָּׁ֑ה וַתֵּ֥לֶד ל֖וֹ אֶת־ עַתָּֽי׃
36הֹלִ֣יד אֶת־ נָתָ֔ן וְנָתָ֖ן הוֹלִ֥יד אֶת־ זָבָֽד׃
37הוֹלִ֣יד אֶת־ אֶפְלָ֔ל וְאֶפְלָ֖ל הוֹלִ֥יד אֶת־ עוֹבֵֽד׃
38הוֹלִ֣יד אֶת־ יֵה֔וּא וְיֵה֖וּא הוֹלִ֥יד אֶת־ עֲזַרְיָֽה׃
39הֹלִ֣יד אֶת־ חָ֔לֶץ וְחֶ֖לֶץ הֹלִ֥יד אֶת־ אֶלְעָשָֽׂה׃
40הֹלִ֣יד אֶת־ סִֽסְמָ֔י וְסִסְמַ֖י הֹלִ֥יד אֶת־ שַׁלּֽוּם׃
41הוֹלִ֣יד אֶת־ יְקַמְיָ֔ה וִֽיקַמְיָ֖ה הֹלִ֥יד אֶת־ אֱלִישָׁמָֽע׃
42כָלֵב֙ אֲחִ֣י יְרַחְמְאֵ֔ל מֵישָׁ֥ע בְּכֹר֖וֹ ה֣וּא אֲבִי־ זִ֑יף וּבְנֵ֥י מָרֵשָׁ֖ה אֲבִ֥י חֶבְרֽוֹן׃
43חֶבְר֑וֹן קֹ֥רַח וְתַפֻּ֖חַ וְרֶ֥קֶם וָשָֽׁמַע׃
44הוֹלִ֔יד אֶת־ רַ֖חַם אֲבִ֣י יָרְקֳעָ֑ם וְרֶ֖קֶם הוֹלִ֥יד אֶת־ שַׁמָּֽי׃
45שַׁמַּ֖י מָע֑וֹן וּמָע֖וֹן אֲבִ֥י בֵֽית־ צֽוּר׃
46פִּילֶ֣גֶשׁ כָּלֵ֔ב יָֽלְדָ֛ה אֶת־ חָרָ֥ן וְאֶת־ מוֹצָ֖א וְאֶת־ גָּזֵ֑ז וְחָרָ֖ן הֹלִ֥יד אֶת־ גָּזֵֽז׃ ס
47יָהְדָּ֑י רֶ֧גֶם וְיוֹתָ֛ם וְגֵישָׁ֥ן וָפֶ֖לֶט וְעֵיפָ֥ה וָשָֽׁעַף׃
48כָּלֵב֙ מַעֲכָ֔ה יָ֥לַד שֶׁ֖בֶר וְאֶֽת־ תִּרְחֲנָֽה׃
49שַׁ֚עַף אֲבִ֣י מַדְמַנָּ֔ה אֶת־ שְׁוָ֛א אֲבִ֥י מַכְבֵּנָ֖ה וַאֲבִ֣י גִבְעָ֑א וּבַת־ כָּלֵ֖ב עַכְסָֽה׃ ס
50הָיוּ֙ בְּנֵ֣י כָלֵ֔ב בֶּן־ ח֖וּר בְּכ֣וֹר אֶפְרָ֑תָה שׁוֹבָ֕ל אֲבִ֖י קִרְיַ֥ת יְעָרִֽים׃
51אֲבִ֣י בֵֽית־ לָ֔חֶם חָרֵ֖ף אֲבִ֥י בֵית־ גָּדֵֽר׃
52בָנִים֙ לְשׁוֹבָ֔ל אֲבִ֖י קִרְיַ֣ת יְעָרִ֑ים הָרֹאֶ֖ה חֲצִ֥י הַמְּנֻחֽוֹת׃
53קִרְיַ֣ת יְעָרִ֔ים הַיִּתְרִי֙ וְהַפּוּתִ֔י וְהַשֻּׁמָתִ֖י וְהַמִּשְׁרָעִ֑י מֵאֵ֗לֶּה יָצְאוּ֙ הַצָּ֣רְעָתִ֔י וְהָאֶשְׁתָּ֖אֻֽלִֽי׃ ס
