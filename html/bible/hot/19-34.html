1לְדָוִ֗ד בְּשַׁנּוֹת֣וֹ אֶת־ טַ֭עְמוֹ לִפְנֵ֣י אֲבִימֶ֑לֶךְ וַֽ֝יְגָרֲשֵׁ֗הוּ וַיֵּלַֽךְ׃ אֲבָרֲכָ֣ה אֶת־ יְהוָ֣ה בְּכָל־ עֵ֑ת תָּ֝מִ֗יד תְּֽהִלָּת֥וֹ בְּפִֽי׃
2תִּתְהַלֵּ֣ל נַפְשִׁ֑י יִשְׁמְע֖וּ עֲנָוִ֣ים וְיִשְׂמָֽחוּ׃
3לַיהוָ֣ה אִתִּ֑י וּנְרוֹמְמָ֖ה שְׁמ֣וֹ יַחְדָּֽו׃
4אֶת־ יְהוָ֣ה וְעָנָ֑נִי וּמִכָּל־ מְ֝גוּרוֹתַ֗י הִצִּילָֽנִי׃
5אֵלָ֣יו וְנָהָ֑רוּ וּ֝פְנֵיהֶ֗ם אַל־ יֶחְפָּֽרוּ׃
6עָנִ֣י קָ֭רָא וַיהוָ֣ה שָׁמֵ֑עַ וּמִכָּל־ צָ֝רוֹתָ֗יו הוֹשִׁיעֽוֹ׃
7מַלְאַךְ־ יְהוָ֓ה סָ֘בִ֤יב לִֽירֵאָ֗יו וַֽיְחַלְּצֵֽם׃
8וּ֭רְאוּ כִּי־ ט֣וֹב יְהוָ֑ה אַֽשְׁרֵ֥י הַ֝גֶּ֗בֶר יֶחֱסֶה־ בּֽוֹ׃
9אֶת־ יְהוָ֣ה קְדֹשָׁ֑יו כִּי־ אֵ֥ין מַ֝חְס֗וֹר לִירֵאָֽיו׃
10רָשׁ֣וּ וְרָעֵ֑בוּ וְדֹרְשֵׁ֥י יְ֝הוָ֗ה לֹא־ יַחְסְר֥וּ כָל־ טֽוֹב׃
11בָ֭נִים שִׁמְעוּ־ לִ֑י יִֽרְאַ֥ת יְ֝הוָ֗ה אֲלַמֶּדְכֶֽם׃
12הָ֭אִישׁ הֶחָפֵ֣ץ חַיִּ֑ים אֹהֵ֥ב יָ֝מִ֗ים לִרְא֥וֹת טֽוֹב׃
13לְשׁוֹנְךָ֣ מֵרָ֑ע וּ֝שְׂפָתֶ֗יךָ מִדַּבֵּ֥ר מִרְמָֽה׃
14מֵ֭רָע וַעֲשֵׂה־ ט֑וֹב בַּקֵּ֖שׁ שָׁל֣וֹם וְרָדְפֵֽהוּ׃
15יְ֭הוָה אֶל־ צַדִּיקִ֑ים וְ֝אָזְנָ֗יו אֶל־ שַׁוְעָתָֽם׃
16יְ֭הוָה בְּעֹ֣שֵׂי רָ֑ע לְהַכְרִ֖ית מֵאֶ֣רֶץ זִכְרָֽם׃
17וַיהוָ֣ה שָׁמֵ֑עַ וּמִכָּל־ צָ֝רוֹתָ֗ם הִצִּילָֽם׃
18יְ֭הוָה לְנִשְׁבְּרֵי־ לֵ֑ב וְֽאֶת־ דַּכְּאֵי־ ר֥וּחַ יוֹשִֽׁיעַ׃
19רָע֣וֹת צַדִּ֑יק וּ֝מִכֻּלָּ֗ם יַצִּילֶ֥נּוּ יְהוָֽה׃
20כָּל־ עַצְמוֹתָ֑יו אַחַ֥ת מֵ֝הֵ֗נָּה לֹ֣א נִשְׁבָּֽרָה׃
21רָשָׁ֣ע רָעָ֑ה וְשֹׂנְאֵ֖י צַדִּ֣יק יֶאְשָֽׁמוּ׃
