1וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־ מֹשֶׁ֥ה לֵּאמֹֽר׃
2תִּֽהְיֶה֙ תּוֹרַ֣ת הַמְּצֹרָ֔ע בְּי֖וֹם טָהֳרָת֑וֹ וְהוּבָ֖א אֶל־ הַכֹּהֵֽן׃
3הַכֹּהֵ֔ן אֶל־ מִח֖וּץ לַֽמַּחֲנֶ֑ה וְרָאָה֙ הַכֹּהֵ֔ן וְהִנֵּ֛ה נִרְפָּ֥א נֶֽגַע־ הַצָּרַ֖עַת מִן־ הַצָּרֽוּעַ׃
4הַכֹּהֵ֔ן וְלָקַ֧ח לַמִּטַּהֵ֛ר שְׁתֵּֽי־ צִפֳּרִ֥ים חַיּ֖וֹת טְהֹר֑וֹת וְעֵ֣ץ אֶ֔רֶז וּשְׁנִ֥י תוֹלַ֖עַת וְאֵזֹֽב׃
5הַכֹּהֵ֔ן וְשָׁחַ֖ט אֶת־ הַצִּפּ֣וֹר הָאֶחָ֑ת אֶל־ כְּלִי־ חֶ֖רֶשׂ עַל־ מַ֥יִם חַיִּֽים׃
6הַצִּפֹּ֤ר הַֽחַיָּה֙ יִקַּ֣ח אֹתָ֔הּ וְאֶת־ עֵ֥ץ הָאֶ֛רֶז וְאֶת־ שְׁנִ֥י הַתּוֹלַ֖עַת וְאֶת־ הָאֵזֹ֑ב וְטָבַ֨ל אוֹתָ֜ם וְאֵ֣ת ׀ הַצִּפֹּ֣ר הַֽחַיָּ֗ה בְּדַם֙ הַצִּפֹּ֣ר הַשְּׁחֻטָ֔ה עַ֖ל הַמַּ֥יִם הַֽחַיִּֽים׃
7עַ֧ל הַמִּטַּהֵ֛ר מִן־ הַצָּרַ֖עַת שֶׁ֣בַע פְּעָמִ֑ים וְטִ֣הֲר֔וֹ וְשִׁלַּ֛ח אֶת־ הַצִּפֹּ֥ר הַֽחַיָּ֖ה עַל־ פְּנֵ֥י הַשָּׂדֶֽה׃
8הַמִּטַּהֵ֨ר אֶת־ בְּגָדָ֜יו וְגִלַּ֣ח אֶת־ כָּל־ שְׂעָר֗וֹ וְרָחַ֤ץ בַּמַּ֙יִם֙ וְטָהֵ֔ר וְאַחַ֖ר יָב֣וֹא אֶל־ הַֽמַּחֲנֶ֑ה וְיָשַׁ֛ב מִח֥וּץ לְאָהֳל֖וֹ שִׁבְעַ֥ת יָמִֽים׃
9בַיּ֨וֹם הַשְּׁבִיעִ֜י יְגַלַּ֣ח אֶת־ כָּל־ שְׂעָר֗וֹ אֶת־ רֹאשׁ֤וֹ וְאֶת־ זְקָנוֹ֙ וְאֵת֙ גַּבֹּ֣ת עֵינָ֔יו וְאֶת־ כָּל־ שְׂעָר֖וֹ יְגַלֵּ֑חַ וְכִבֶּ֣ס אֶת־ בְּגָדָ֗יו וְרָחַ֧ץ אֶת־ בְּשָׂר֛וֹ בַּמַּ֖יִם וְטָהֵֽר׃
10הַשְּׁמִינִ֗י יִקַּ֤ח שְׁנֵֽי־ כְבָשִׂים֙ תְּמִימִ֔ים וְכַבְשָׂ֥ה אַחַ֛ת בַּת־ שְׁנָתָ֖הּ תְּמִימָ֑ה וּשְׁלֹשָׁ֣ה עֶשְׂרֹנִ֗ים סֹ֤לֶת מִנְחָה֙ בְּלוּלָ֣ה בַשֶּׁ֔מֶן וְלֹ֥ג אֶחָ֖ד שָֽׁמֶן׃
11הַכֹּהֵ֣ן הַֽמְטַהֵ֗ר אֵ֛ת הָאִ֥ישׁ הַמִּטַּהֵ֖ר וְאֹתָ֑ם לִפְנֵ֣י יְהוָ֔ה פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃
12הַכֹּהֵ֜ן אֶת־ הַכֶּ֣בֶשׂ הָאֶחָ֗ד וְהִקְרִ֥יב אֹת֛וֹ לְאָשָׁ֖ם וְאֶת־ לֹ֣ג הַשָּׁ֑מֶן וְהֵנִ֥יף אֹתָ֛ם תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֽה׃
13אֶת־ הַכֶּ֗בֶשׂ בִּ֠מְקוֹם אֲשֶׁ֨ר יִשְׁחַ֧ט אֶת־ הַֽחַטָּ֛את וְאֶת־ הָעֹלָ֖ה בִּמְק֣וֹם הַקֹּ֑דֶשׁ כִּ֡י כַּ֠חַטָּאת הָאָשָׁ֥ם הוּא֙ לַכֹּהֵ֔ן קֹ֥דֶשׁ קָֽדָשִׁ֖ים הֽוּא׃
14הַכֹּהֵן֮ מִדַּ֣ם הָאָשָׁם֒ וְנָתַן֙ הַכֹּהֵ֔ן עַל־ תְּנ֛וּךְ אֹ֥זֶן הַמִּטַּהֵ֖ר הַיְמָנִ֑ית וְעַל־ בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־ בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִֽית׃
15הַכֹּהֵ֖ן מִלֹּ֣ג הַשָּׁ֑מֶן וְיָצַ֛ק עַל־ כַּ֥ף הַכֹּהֵ֖ן הַשְּׂמָאלִֽית׃
16הַכֹּהֵן֙ אֶת־ אֶצְבָּע֣וֹ הַיְמָנִ֔ית מִן־ הַשֶּׁ֕מֶן אֲשֶׁ֥ר עַל־ כַּפּ֖וֹ הַשְּׂמָאלִ֑ית וְהִזָּ֨ה מִן־ הַשֶּׁ֧מֶן בְּאֶצְבָּע֛וֹ שֶׁ֥בַע פְּעָמִ֖ים לִפְנֵ֥י יְהוָֽה׃
17הַשֶּׁ֜מֶן אֲשֶׁ֣ר עַל־ כַּפּ֗וֹ יִתֵּ֤ן הַכֹּהֵן֙ עַל־ תְּנ֞וּךְ אֹ֤זֶן הַמִּטַּהֵר֙ הַיְמָנִ֔ית וְעַל־ בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־ בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִ֑ית עַ֖ל דַּ֥ם הָאָשָֽׁם׃
18בַּשֶּׁ֙מֶן֙ אֲשֶׁר֙ עַל־ כַּ֣ף הַכֹּהֵ֔ן יִתֵּ֖ן עַל־ רֹ֣אשׁ הַמִּטַּהֵ֑ר וְכִפֶּ֥ר עָלָ֛יו הַכֹּהֵ֖ן לִפְנֵ֥י יְהוָֽה׃
19הַכֹּהֵן֙ אֶת־ הַ֣חַטָּ֔את וְכִפֶּ֕ר עַל־ הַמִּטַּהֵ֖ר מִטֻּמְאָת֑וֹ וְאַחַ֖ר יִשְׁחַ֥ט אֶת־ הָעֹלָֽה׃
20הַכֹּהֵ֛ן אֶת־ הָעֹלָ֥ה וְאֶת־ הַמִּנְחָ֖ה הַמִּזְבֵּ֑חָה וְכִפֶּ֥ר עָלָ֛יו הַכֹּהֵ֖ן וְטָהֵֽר׃ ס
21דַּ֣ל ה֗וּא וְאֵ֣ין יָדוֹ֮ מַשֶּׂגֶת֒ וְ֠לָקַח כֶּ֣בֶשׂ אֶחָ֥ד אָשָׁ֛ם לִתְנוּפָ֖ה לְכַפֵּ֣ר עָלָ֑יו וְעִשָּׂר֨וֹן סֹ֜לֶת אֶחָ֨ד בָּל֥וּל בַּשֶּׁ֛מֶן לְמִנְחָ֖ה וְלֹ֥ג שָֽׁמֶן׃
22תֹרִ֗ים א֤וֹ שְׁנֵי֙ בְּנֵ֣י יוֹנָ֔ה אֲשֶׁ֥ר תַּשִּׂ֖יג יָד֑וֹ וְהָיָ֤ה אֶחָד֙ חַטָּ֔את וְהָאֶחָ֖ד עֹלָֽה׃
23אֹתָ֜ם בַּיּ֧וֹם הַשְּׁמִינִ֛י לְטָהֳרָת֖וֹ אֶל־ הַכֹּהֵ֑ן אֶל־ פֶּ֥תַח אֹֽהֶל־ מוֹעֵ֖ד לִפְנֵ֥י יְהוָֽה׃
24הַכֹּהֵ֛ן אֶת־ כֶּ֥בֶשׂ הָאָשָׁ֖ם וְאֶת־ לֹ֣ג הַשָּׁ֑מֶן וְהֵנִ֨יף אֹתָ֧ם הַכֹּהֵ֛ן תְּנוּפָ֖ה לִפְנֵ֥י יְהוָֽה׃
25אֶת־ כֶּ֣בֶשׂ הָֽאָשָׁם֒ וְלָקַ֤ח הַכֹּהֵן֙ מִדַּ֣ם הָֽאָשָׁ֔ם וְנָתַ֛ן עַל־ תְּנ֥וּךְ אֹֽזֶן־ הַמִּטַּהֵ֖ר הַיְמָנִ֑ית וְעַל־ בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־ בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִֽית׃
26הַשֶּׁ֖מֶן יִצֹ֣ק הַכֹּהֵ֑ן עַל־ כַּ֥ף הַכֹּהֵ֖ן הַשְּׂמָאלִֽית׃
27הַכֹּהֵן֙ בְּאֶצְבָּע֣וֹ הַיְמָנִ֔ית מִן־ הַשֶּׁ֕מֶן אֲשֶׁ֥ר עַל־ כַּפּ֖וֹ הַשְּׂמָאלִ֑ית שֶׁ֥בַע פְּעָמִ֖ים לִפְנֵ֥י יְהוָֽה׃
28הַכֹּהֵ֜ן מִן־ הַשֶּׁ֣מֶן ׀ אֲשֶׁ֣ר עַל־ כַּפּ֗וֹ עַל־ תְּנ֞וּךְ אֹ֤זֶן הַמִּטַּהֵר֙ הַיְמָנִ֔ית וְעַל־ בֹּ֤הֶן יָדוֹ֙ הַיְמָנִ֔ית וְעַל־ בֹּ֥הֶן רַגְל֖וֹ הַיְמָנִ֑ית עַל־ מְק֖וֹם דַּ֥ם הָאָשָֽׁם׃
29מִן־ הַשֶּׁ֙מֶן֙ אֲשֶׁר֙ עַל־ כַּ֣ף הַכֹּהֵ֔ן יִתֵּ֖ן עַל־ רֹ֣אשׁ הַמִּטַּהֵ֑ר לְכַפֵּ֥ר עָלָ֖יו לִפְנֵ֥י יְהוָֽה׃
30אֶת־ הָֽאֶחָד֙ מִן־ הַתֹּרִ֔ים א֖וֹ מִן־ בְּנֵ֣י הַיּוֹנָ֑ה מֵאֲשֶׁ֥ר תַּשִּׂ֖יג יָדֽוֹ׃
31אֲשֶׁר־ תַּשִּׂ֞יג יָד֗וֹ אֶת־ הָאֶחָ֥ד חַטָּ֛את וְאֶת־ הָאֶחָ֥ד עֹלָ֖ה עַל־ הַמִּנְחָ֑ה וְכִפֶּ֧ר הַכֹּהֵ֛ן עַ֥ל הַמִּטַּהֵ֖ר לִפְנֵ֥י יְהוָֽה׃
32תּוֹרַ֔ת אֲשֶׁר־ בּ֖וֹ נֶ֣גַע צָרָ֑עַת אֲשֶׁ֛ר לֹֽא־ תַשִּׂ֥יג יָד֖וֹ בְּטָהֳרָתֽוֹ׃ פ
33יְהוָ֔ה אֶל־ מֹשֶׁ֥ה וְאֶֽל־ אַהֲרֹ֖ן לֵאמֹֽר׃
34תָבֹ֙אוּ֙ אֶל־ אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁ֥ר אֲנִ֛י נֹתֵ֥ן לָכֶ֖ם לַאֲחֻזָּ֑ה וְנָתַתִּי֙ נֶ֣גַע צָרַ֔עַת בְּבֵ֖ית אֶ֥רֶץ אֲחֻזַּתְכֶֽם׃
35אֲשֶׁר־ ל֣וֹ הַבַּ֔יִת וְהִגִּ֥יד לַכֹּהֵ֖ן לֵאמֹ֑ר כְּנֶ֕גַע נִרְאָ֥ה לִ֖י בַּבָּֽיִת׃
36הַכֹּהֵ֜ן וּפִנּ֣וּ אֶת־ הַבַּ֗יִת בְּטֶ֨רֶם יָבֹ֤א הַכֹּהֵן֙ לִרְא֣וֹת אֶת־ הַנֶּ֔גַע וְלֹ֥א יִטְמָ֖א כָּל־ אֲשֶׁ֣ר בַּבָּ֑יִת וְאַ֥חַר כֵּ֛ן יָבֹ֥א הַכֹּהֵ֖ן לִרְא֥וֹת אֶת־ הַבָּֽיִת׃
37אֶת־ הַנֶּ֗גַע וְהִנֵּ֤ה הַנֶּ֙גַע֙ בְּקִירֹ֣ת הַבַּ֔יִת שְׁקַֽעֲרוּרֹת֙ יְרַקְרַקֹּ֔ת א֖וֹ אֲדַמְדַּמֹּ֑ת וּמַרְאֵיהֶ֥ן שָׁפָ֖ל מִן־ הַקִּֽיר׃
38הַכֹּהֵ֛ן מִן־ הַבַּ֖יִת אֶל־ פֶּ֣תַח הַבָּ֑יִת וְהִסְגִּ֥יר אֶת־ הַבַּ֖יִת שִׁבְעַ֥ת יָמִֽים׃
39הַכֹּהֵ֖ן בַּיּ֣וֹם הַשְּׁבִיעִ֑י וְרָאָ֕ה וְהִנֵּ֛ה פָּשָׂ֥ה הַנֶּ֖גַע בְּקִירֹ֥ת הַבָּֽיִת׃
40הַכֹּהֵ֔ן וְחִלְּצוּ֙ אֶת־ הָ֣אֲבָנִ֔ים אֲשֶׁ֥ר בָּהֵ֖ן הַנָּ֑גַע וְהִשְׁלִ֤יכוּ אֶתְהֶן֙ אֶל־ מִח֣וּץ לָעִ֔יר אֶל־ מָק֖וֹם טָמֵֽא׃
41הַבַּ֛יִת יַקְצִ֥עַ מִבַּ֖יִת סָבִ֑יב וְשָׁפְכ֗וּ אֶת־ הֶֽעָפָר֙ אֲשֶׁ֣ר הִקְצ֔וּ אֶל־ מִח֣וּץ לָעִ֔יר אֶל־ מָק֖וֹם טָמֵֽא׃
42אֲבָנִ֣ים אֲחֵר֔וֹת וְהֵבִ֖יאוּ אֶל־ תַּ֣חַת הָאֲבָנִ֑ים וְעָפָ֥ר אַחֵ֛ר יִקַּ֖ח וְטָ֥ח אֶת־ הַבָּֽיִת׃
43יָשׁ֤וּב הַנֶּ֙גַע֙ וּפָרַ֣ח בַּבַּ֔יִת אַחַ֖ר חִלֵּ֣ץ אֶת־ הָאֲבָנִ֑ים וְאַחֲרֵ֛י הִקְצ֥וֹת אֶת־ הַבַּ֖יִת וְאַחֲרֵ֥י הִטּֽוֹחַ׃
44הַכֹּהֵ֔ן וְרָאָ֕ה וְהִנֵּ֛ה פָּשָׂ֥ה הַנֶּ֖גַע בַּבָּ֑יִת צָרַ֨עַת מַמְאֶ֥רֶת הִ֛וא בַּבַּ֖יִת טָמֵ֥א הֽוּא׃
45אֶת־ הַבַּ֗יִת אֶת־ אֲבָנָיו֙ וְאֶת־ עֵצָ֔יו וְאֵ֖ת כָּל־ עֲפַ֣ר הַבָּ֑יִת וְהוֹצִיא֙ אֶל־ מִח֣וּץ לָעִ֔יר אֶל־ מָק֖וֹם טָמֵֽא׃
46אֶל־ הַבַּ֔יִת כָּל־ יְמֵ֖י הִסְגִּ֣יר אֹת֑וֹ יִטְמָ֖א עַד־ הָעָֽרֶב׃
47בַּבַּ֔יִת יְכַבֵּ֖ס אֶת־ בְּגָדָ֑יו וְהָאֹכֵ֣ל בַּבַּ֔יִת יְכַבֵּ֖ס אֶת־ בְּגָדָֽיו׃
48בֹּ֨א יָבֹ֜א הַכֹּהֵ֗ן וְרָאָה֙ וְ֠הִנֵּה לֹא־ פָשָׂ֤ה הַנֶּ֙גַע֙ בַּבַּ֔יִת אַחֲרֵ֖י הִטֹּ֣חַ אֶת־ הַבָּ֑יִת וְטִהַ֤ר הַכֹּהֵן֙ אֶת־ הַבַּ֔יִת כִּ֥י נִרְפָּ֖א הַנָּֽגַע׃
49לְחַטֵּ֥א אֶת־ הַבַּ֖יִת שְׁתֵּ֣י צִפֳּרִ֑ים וְעֵ֣ץ אֶ֔רֶז וּשְׁנִ֥י תוֹלַ֖עַת וְאֵזֹֽב׃
50אֶת־ הַצִּפֹּ֣ר הָאֶחָ֑ת אֶל־ כְּלִי־ חֶ֖רֶשׂ עַל־ מַ֥יִם חַיִּֽים׃
51אֶת־ עֵֽץ־ הָ֠אֶרֶז וְאֶת־ הָ֨אֵזֹ֜ב וְאֵ֣ת ׀ שְׁנִ֣י הַתּוֹלַ֗עַת וְאֵת֮ הַצִּפֹּ֣ר הַֽחַיָּה֒ וְטָבַ֣ל אֹתָ֗ם בְּדַם֙ הַצִּפֹּ֣ר הַשְּׁחוּטָ֔ה וּבַמַּ֖יִם הַֽחַיִּ֑ים וְהִזָּ֥ה אֶל־ הַבַּ֖יִת שֶׁ֥בַע פְּעָמִֽים׃
52אֶת־ הַבַּ֔יִת בְּדַם֙ הַצִּפּ֔וֹר וּבַמַּ֖יִם הַֽחַיִּ֑ים וּבַצִּפֹּ֣ר הַחַיָּ֗ה וּבְעֵ֥ץ הָאֶ֛רֶז וּבָאֵזֹ֖ב וּבִשְׁנִ֥י הַתּוֹלָֽעַת׃
53אֶת־ הַצִּפֹּ֧ר הַֽחַיָּ֛ה אֶל־ מִח֥וּץ לָעִ֖יר אֶל־ פְּנֵ֣י הַשָּׂדֶ֑ה וְכִפֶּ֥ר עַל־ הַבַּ֖יִת וְטָהֵֽר׃
54הַתּוֹרָ֑ה לְכָל־ נֶ֥גַע הַצָּרַ֖עַת וְלַנָּֽתֶק׃
55הַבֶּ֖גֶד וְלַבָּֽיִת׃
56וְלַסַּפַּ֖חַת וְלַבֶּהָֽרֶת׃
