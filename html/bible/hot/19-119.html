1אַשְׁרֵ֥י תְמִֽימֵי־ דָ֑רֶךְ הַֽ֝הֹלְכִ֗ים בְּתוֹרַ֥ת יְהוָֽה׃
2נֹצְרֵ֥י עֵדֹתָ֗יו בְּכָל־ לֵ֥ב יִדְרְשֽׁוּהוּ׃
3לֹֽא־ פָעֲל֣וּ עַוְלָ֑ה בִּדְרָכָ֥יו הָלָֽכוּ׃
4צִוִּ֥יתָה פִקֻּדֶ֗יךָ לִשְׁמֹ֥ר מְאֹֽד׃
5יִכֹּ֥נוּ דְרָכָ֗י לִשְׁמֹ֥ר חֻקֶּֽיךָ׃
6לֹא־ אֵב֑וֹשׁ בְּ֝הַבִּיטִ֗י אֶל־ כָּל־ מִצְוֹתֶֽיךָ׃
7בְּיֹ֣שֶׁר לֵבָ֑ב בְּ֝לָמְדִ֗י מִשְׁפְּטֵ֥י צִדְקֶֽךָ׃
8חֻקֶּ֥יךָ אֶשְׁמֹ֑ר אַֽל־ תַּעַזְבֵ֥נִי עַד־ מְאֹֽד׃
9יְזַכֶּה־ נַּ֭עַר אֶת־ אָרְח֑וֹ לִ֝שְׁמֹ֗ר כִּדְבָרֶֽךָ׃
10לִבִּ֥י דְרַשְׁתִּ֑יךָ אַל־ תַּ֝שְׁגֵּ֗נִי מִמִּצְוֹתֶֽיךָ׃
11צָפַ֣נְתִּי אִמְרָתֶ֑ךָ לְ֝מַ֗עַן לֹ֣א אֶֽחֱטָא־ לָֽךְ׃
12אַתָּ֥ה יְהוָ֗ה לַמְּדֵ֥נִי חֻקֶּֽיךָ׃
13סִפַּ֑רְתִּי כֹּ֝֗ל מִשְׁפְּטֵי־ פִֽיךָ׃
14עֵדְוֹתֶ֥יךָ שַׂ֗שְׂתִּי כְּעַ֣ל כָּל־ הֽוֹן׃
15אָשִׂ֑יחָה וְ֝אַבִּ֗יטָה אֹרְחֹתֶֽיךָ׃
16אֶֽשְׁתַּעֲשָׁ֑ע לֹ֭א אֶשְׁכַּ֣ח דְּבָרֶֽךָ׃
17עַֽל־ עַבְדְּךָ֥ אֶֽחְיֶ֗ה וְאֶשְׁמְרָ֥ה דְבָרֶֽךָ׃
18עֵינַ֥י וְאַבִּ֑יטָה נִ֝פְלָא֗וֹת מִתּוֹרָתֶֽךָ׃
19אָנֹכִ֣י בָאָ֑רֶץ אַל־ תַּסְתֵּ֥ר מִ֝מֶּ֗נִּי מִצְוֹתֶֽיךָ׃
20נַפְשִׁ֣י לְתַאֲבָ֑ה אֶֽל־ מִשְׁפָּטֶ֥יךָ בְכָל־ עֵֽת׃
21זֵדִ֣ים אֲרוּרִ֑ים הַ֝שֹּׁגִים מִמִּצְוֹתֶֽיךָ׃
22מֵֽ֭עָלַי חֶרְפָּ֣ה וָב֑וּז כִּ֖י עֵדֹתֶ֣יךָ נָצָֽרְתִּי׃
23יָֽשְׁב֣וּ שָׂ֭רִים בִּ֣י נִדְבָּ֑רוּ עַ֝בְדְּךָ֗ יָשִׂ֥יחַ בְּחֻקֶּֽיךָ׃
24עֵ֭דֹתֶיךָ שַׁעֲשֻׁעָ֗י אַנְשֵׁ֥י עֲצָתִֽי׃
25לֶעָפָ֣ר נַפְשִׁ֑י חַ֝יֵּ֗נִי כִּדְבָרֶֽךָ׃
26סִ֭פַּרְתִּי וַֽתַּעֲנֵ֗נִי לַמְּדֵ֥נִי חֻקֶּֽיךָ׃
27פִּקּוּדֶ֥יךָ הֲבִינֵ֑נִי וְ֝אָשִׂ֗יחָה בְּנִפְלְאוֹתֶֽיךָ׃
28נַ֭פְשִׁי מִתּוּגָ֑ה קַ֝יְּמֵ֗נִי כִּדְבָרֶֽךָ׃
29שֶׁ֭קֶר הָסֵ֣ר מִמֶּ֑נִּי וְֽתוֹרָתְךָ֥ חָנֵּֽנִי׃
30אֱמוּנָ֥ה בָחָ֑רְתִּי מִשְׁפָּטֶ֥יךָ שִׁוִּֽיתִי׃
31בְעֵֽדְוֹתֶ֑יךָ יְ֝הוָ֗ה אַל־ תְּבִישֵֽׁנִי׃
32מִצְוֹתֶ֥יךָ אָר֑וּץ כִּ֖י תַרְחִ֣יב לִבִּֽי׃
33יְ֭הוָה דֶּ֥רֶךְ חֻקֶּ֗יךָ וְאֶצְּרֶ֥נָּה עֵֽקֶב׃
34וְאֶצְּרָ֥ה תֽוֹרָתֶ֗ךָ וְאֶשְׁמְרֶ֥נָּה בְכָל־ לֵֽב׃
35בִּנְתִ֣יב מִצְוֹתֶ֑יךָ כִּי־ ב֥וֹ חָפָֽצְתִּי׃
36לִ֭בִּי אֶל־ עֵדְוֹתֶ֗יךָ וְאַ֣ל אֶל־ בָּֽצַע׃
37עֵ֭ינַי מֵרְא֣וֹת שָׁ֑וְא בִּדְרָכֶ֥ךָ חַיֵּֽנִי׃
38לְ֭עַבְדְּךָ אִמְרָתֶ֑ךָ אֲ֝שֶׁ֗ר לְיִרְאָתֶֽךָ׃
39חֶ֭רְפָּתִי אֲשֶׁ֣ר יָגֹ֑רְתִּי כִּ֖י מִשְׁפָּטֶ֣יךָ טוֹבִֽים׃
40תָּאַ֣בְתִּי לְפִקֻּדֶ֑יךָ בְּצִדְקָתְךָ֥ חַיֵּֽנִי׃
41חֲסָדֶ֣ךָ יְהוָ֑ה תְּ֝שֽׁוּעָתְךָ֗ כְּאִמְרָתֶֽךָ׃
42חֹרְפִ֣י דָבָ֑ר כִּֽי־ בָ֝טַחְתִּי בִּדְבָרֶֽךָ׃
43תַּצֵּ֬ל מִפִּ֣י דְבַר־ אֱמֶ֣ת עַד־ מְאֹ֑ד כִּ֖י לְמִשְׁפָּטֶ֣ךָ יִחָֽלְתִּי׃
44תוֹרָתְךָ֥ תָמִ֗יד לְעוֹלָ֥ם וָעֶֽד׃
45בָרְחָבָ֑ה כִּ֖י פִקֻּדֶ֣יךָ דָרָֽשְׁתִּי׃
46בְ֭עֵדֹתֶיךָ נֶ֥גֶד מְלָכִ֗ים וְלֹ֣א אֵבֽוֹשׁ׃
47בְּמִצְוֹתֶ֗יךָ אֲשֶׁ֣ר אָהָֽבְתִּי׃
48כַפַּ֗י אֶֽל־ מִ֭צְוֹתֶיךָ אֲשֶׁ֥ר אָהָ֗בְתִּי וְאָשִׂ֥יחָה בְחֻקֶּֽיךָ׃
49דָּבָ֥ר לְעַבְדֶּ֑ךָ עַ֝֗ל אֲשֶׁ֣ר יִֽחַלְתָּֽנִי׃
50נֶחָמָתִ֣י בְעָנְיִ֑י כִּ֖י אִמְרָתְךָ֣ חִיָּֽתְנִי׃
51הֱלִיצֻ֣נִי עַד־ מְאֹ֑ד מִ֝תּֽוֹרָתְךָ֗ לֹ֣א נָטִֽיתִי׃
52מִשְׁפָּטֶ֖יךָ מֵעוֹלָ֥ם ׀ יְהוָ֗ה וָֽאֶתְנֶחָֽם׃
53אֲ֭חָזַתְנִי מֵרְשָׁעִ֑ים עֹ֝זְבֵ֗י תּוֹרָתֶֽךָ׃
54הָֽיוּ־ לִ֥י חֻקֶּ֗יךָ בְּבֵ֣ית מְגוּרָֽי׃
55בַלַּ֣יְלָה שִׁמְךָ֣ יְהוָ֑ה וָֽ֝אֶשְׁמְרָ֗ה תּוֹרָתֶֽךָ׃
56הָֽיְתָה־ לִּ֑י כִּ֖י פִקֻּדֶ֣יךָ נָצָֽרְתִּי׃
57יְהוָ֥ה אָמַ֗רְתִּי לִשְׁמֹ֥ר דְּבָרֶֽיךָ׃
58פָנֶ֣יךָ בְכָל־ לֵ֑ב חָ֝נֵּ֗נִי כְּאִמְרָתֶֽךָ׃
59דְרָכָ֑י וָאָשִׁ֥יבָה רַ֝גְלַ֗י אֶל־ עֵדֹתֶֽיךָ׃
60וְלֹ֣א הִתְמַהְמָ֑הְתִּי לִ֝שְׁמֹ֗ר מִצְוֹתֶֽיךָ׃
61רְשָׁעִ֣ים עִוְּדֻ֑נִי תּֽ֝וֹרָתְךָ֗ לֹ֣א שָׁכָֽחְתִּי׃
62לַ֗יְלָה אָ֭קוּם לְהוֹד֣וֹת לָ֑ךְ עַ֝֗ל מִשְׁפְּטֵ֥י צִדְקֶֽךָ׃
63אָ֭נִי לְכָל־ אֲשֶׁ֣ר יְרֵא֑וּךָ וּ֝לְשֹׁמְרֵ֗י פִּקּוּדֶֽיךָ׃
64יְ֭הוָה מָלְאָ֥ה הָאָ֗רֶץ חֻקֶּ֥יךָ לַמְּדֵֽנִי׃
65עָשִׂ֣יתָ עִֽם־ עַבְדְּךָ֑ יְ֝הוָ֗ה כִּדְבָרֶֽךָ׃
66טַ֣עַם וָדַ֣עַת לַמְּדֵ֑נִי כִּ֖י בְמִצְוֹתֶ֣יךָ הֶאֱמָֽנְתִּי׃
67אֶ֭עֱנֶה אֲנִ֣י שֹׁגֵ֑ג וְ֝עַתָּ֗ה אִמְרָתְךָ֥ שָׁמָֽרְתִּי׃
68אַתָּ֥ה וּמֵטִ֗יב לַמְּדֵ֥נִי חֻקֶּֽיךָ׃
69עָלַ֣י שֶׁ֣קֶר זֵדִ֑ים אֲ֝נִ֗י בְּכָל־ לֵ֤ב ׀ אֱצֹּ֬ר פִּקּוּדֶֽיךָ׃
70כַּחֵ֣לֶב לִבָּ֑ם אֲ֝נִ֗י תּוֹרָתְךָ֥ שִֽׁעֲשָֽׁעְתִּי׃
71לִ֥י כִֽי־ עֻנֵּ֑יתִי לְ֝מַ֗עַן אֶלְמַ֥ד חֻקֶּֽיךָ׃
72לִ֥י תֽוֹרַת־ פִּ֑יךָ מֵ֝אַלְפֵ֗י זָהָ֥ב וָכָֽסֶף׃
73עָ֭שׂוּנִי וַֽיְכוֹנְנ֑וּנִי הֲ֝בִינֵ֗נִי וְאֶלְמְדָ֥ה מִצְוֹתֶֽיךָ׃
74יִרְא֣וּנִי וְיִשְׂמָ֑חוּ כִּ֖י לִדְבָרְךָ֣ יִחָֽלְתִּי׃
75יְ֭הוָה כִּי־ צֶ֣דֶק מִשְׁפָּטֶ֑יךָ וֶ֝אֱמוּנָ֗ה עִנִּיתָֽנִי׃
76נָ֣א חַסְדְּךָ֣ לְנַחֲמֵ֑נִי כְּאִמְרָתְךָ֥ לְעַבְדֶּֽךָ׃
77רַחֲמֶ֣יךָ וְאֶֽחְיֶ֑ה כִּי־ תֽ֝וֹרָתְךָ֗ שַֽׁעֲשֻׁעָֽי׃
78זֵ֭דִים כִּי־ שֶׁ֣קֶר עִוְּת֑וּנִי אֲ֝נִ֗י אָשִׂ֥יחַ בְּפִקּוּדֶֽיךָ׃
79לִ֣י יְרֵאֶ֑יךָ [וְיָדְעוּ כ] (וְ֝יֹדְעֵ֗י ק) עֵדֹתֶֽיךָ׃
80לִבִּ֣י תָמִ֣ים בְּחֻקֶּ֑יךָ לְ֝מַ֗עַן לֹ֣א אֵבֽוֹשׁ׃
81לִתְשׁוּעָתְךָ֣ נַפְשִׁ֑י לִדְבָרְךָ֥ יִחָֽלְתִּי׃
82עֵ֭ינַי לְאִמְרָתֶ֑ךָ לֵ֝אמֹ֗ר מָתַ֥י תְּֽנַחֲמֵֽנִי׃
83הָ֭יִיתִי כְּנֹ֣אד בְּקִיט֑וֹר חֻ֝קֶּ֗יךָ לֹ֣א שָׁכָֽחְתִּי׃
84יְמֵֽי־ עַבְדֶּ֑ךָ מָתַ֬י תַּעֲשֶׂ֖ה בְרֹדְפַ֣י מִשְׁפָּֽט׃
85לִ֣י זֵדִ֣ים שִׁיח֑וֹת אֲ֝שֶׁ֗ר לֹ֣א כְתוֹרָתֶֽךָ׃
86מִצְוֹתֶ֥יךָ אֱמוּנָ֑ה שֶׁ֖קֶר רְדָפ֣וּנִי עָזְרֵֽנִי׃
87כִּלּ֣וּנִי בָאָ֑רֶץ וַ֝אֲנִ֗י לֹא־ עָזַ֥בְתִּי פִקֻּודֶֽיךָ׃
88חַיֵּ֑נִי וְ֝אֶשְׁמְרָ֗ה עֵד֥וּת פִּֽיךָ׃
89יְהוָ֑ה דְּ֝בָרְךָ֗ נִצָּ֥ב בַּשָּׁמָֽיִם׃
90וָ֭דֹר אֱמֽוּנָתֶ֑ךָ כּוֹנַ֥נְתָּ אֶ֝֗רֶץ וַֽתַּעֲמֹֽד׃
91עָמְד֣וּ הַיּ֑וֹם כִּ֖י הַכֹּ֣ל עֲבָדֶֽיךָ׃
92ת֭וֹרָתְךָ שַׁעֲשֻׁעָ֑י אָ֝֗ז אָבַ֥דְתִּי בְעָנְיִֽי׃
93לֹא־ אֶשְׁכַּ֣ח פִּקּוּדֶ֑יךָ כִּ֥י בָ֝֗ם חִיִּיתָֽנִי׃
94אֲ֭נִי הוֹשִׁיעֵ֑נִי כִּ֖י פִקּוּדֶ֣יךָ דָרָֽשְׁתִּי׃
95קִוּ֣וּ רְשָׁעִ֣ים לְאַבְּדֵ֑נִי עֵ֝דֹתֶ֗יךָ אֶתְבּוֹנָֽן׃
96תִּ֭כְלָה רָאִ֣יתִי קֵ֑ץ רְחָבָ֖ה מִצְוָתְךָ֣ מְאֹֽד׃
97אָהַ֥בְתִּי תוֹרָתֶ֑ךָ כָּל־ הַ֝יּ֗וֹם הִ֣יא שִׂיחָתִֽי׃
98תְּחַכְּמֵ֣נִי מִצְוֹתֶ֑ךָ כִּ֖י לְעוֹלָ֣ם הִיא־ לִֽי׃
99מְלַמְּדַ֥י הִשְׂכַּ֑לְתִּי כִּ֥י עֵ֝דְוֹתֶ֗יךָ שִׂ֣יחָה לִֽֿי׃
100אֶתְבּוֹנָ֑ן כִּ֖י פִקּוּדֶ֣יךָ נָצָֽרְתִּי׃
101אֹ֣רַח רָ֭ע כָּלִ֣אתִי רַגְלָ֑י לְ֝מַ֗עַן אֶשְׁמֹ֥ר דְּבָרֶֽךָ׃
102לֹא־ סָ֑רְתִּי כִּֽי־ אַ֝תָּ֗ה הוֹרֵתָֽנִי׃
103נִּמְלְצ֣וּ לְ֭חִכִּי אִמְרָתֶ֗ךָ מִדְּבַ֥שׁ לְפִֽי׃
104אֶתְבּוֹנָ֑ן עַל־ כֵּ֝֗ן שָׂנֵ֤אתִי ׀ כָּל־ אֹ֬רַח שָֽׁקֶר׃
105לְרַגְלִ֥י דְבָרֶ֑ךָ וְ֝א֗וֹר לִנְתִיבָתִֽי׃
106וָאֲקַיֵּ֑מָה לִ֝שְׁמֹ֗ר מִשְׁפְּטֵ֥י צִדְקֶֽךָ׃
107עַד־ מְאֹ֑ד יְ֝הוָ֗ה חַיֵּ֥נִי כִדְבָרֶֽךָ׃
108פִּ֭י רְצֵה־ נָ֣א יְהוָ֑ה וּֽמִשְׁפָּטֶ֥יךָ לַמְּדֵֽנִי׃
109בְכַפִּ֣י תָמִ֑יד וְ֝תֽוֹרָתְךָ֗ לֹ֣א שָׁכָֽחְתִּי׃
110רְשָׁעִ֣ים פַּ֣ח לִ֑י וּ֝מִפִּקּוּדֶ֗יךָ לֹ֣א תָעִֽיתִי׃
111עֵדְוֹתֶ֣יךָ לְעוֹלָ֑ם כִּֽי־ שְׂשׂ֖וֹן לִבִּ֣י הֵֽמָּה׃
112לִ֭בִּי לַעֲשׂ֥וֹת חֻקֶּ֗יךָ לְעוֹלָ֥ם עֵֽקֶב׃
113שָׂנֵ֑אתִי וְֽתוֹרָתְךָ֥ אָהָֽבְתִּי׃
114וּמָגִנִּ֣י אָ֑תָּה לִדְבָרְךָ֥ יִחָֽלְתִּי׃
115מִמֶּ֥נִּי מְרֵעִ֑ים וְ֝אֶצְּרָ֗ה מִצְוֹ֥ת אֱלֹהָֽי׃
116כְאִמְרָתְךָ֣ וְאֶֽחְיֶ֑ה וְאַל־ תְּ֝בִישֵׁ֗נִי מִשִּׂבְרִֽי׃
117וְאִוָּשֵׁ֑עָה וְאֶשְׁעָ֖ה בְחֻקֶּ֣יךָ תָמִֽיד׃
118כָּל־ שׁוֹגִ֣ים מֵחֻקֶּ֑יךָ כִּי־ שֶׁ֝֗קֶר תַּרְמִיתָֽם׃
119הִשְׁבַּ֥תָּ כָל־ רִשְׁעֵי־ אָ֑רֶץ לָ֝כֵ֗ן אָהַ֥בְתִּי עֵדֹתֶֽיךָ׃
120מִפַּחְדְּךָ֣ בְשָׂרִ֑י וּֽמִמִּשְׁפָּטֶ֥יךָ יָרֵֽאתִי׃
121מִשְׁפָּ֣ט וָצֶ֑דֶק בַּל־ תַּ֝נִּיחֵ֗נִי לְעֹֽשְׁקָֽי׃
122עַבְדְּךָ֣ לְט֑וֹב אַֽל־ יַעַשְׁקֻ֥נִי זֵדִֽים׃
123כָּל֣וּ לִֽישׁוּעָתֶ֑ךָ וּלְאִמְרַ֥ת צִדְקֶֽךָ׃
124עִם־ עַבְדְּךָ֥ כְחַסְדֶּ֗ךָ וְחֻקֶּ֥יךָ לַמְּדֵֽנִי׃
125אָ֥נִי הֲבִינֵ֑נִי וְ֝אֵדְעָ֗ה עֵדֹתֶֽיךָ׃
126לַעֲשׂ֣וֹת לַיהוָ֑ה הֵ֝פֵ֗רוּ תּוֹרָתֶֽךָ׃
127כֵּ֭ן אָהַ֣בְתִּי מִצְוֹתֶ֑יךָ מִזָּהָ֥ב וּמִפָּֽז׃
128כֵּ֤ן ׀ כָּל־ פִּקּ֣וּדֵי כֹ֣ל יִשָּׁ֑רְתִּי כָּל־ אֹ֖רַח שֶׁ֣קֶר שָׂנֵֽאתִי׃
129עֵדְוֹתֶ֑יךָ עַל־ כֵּ֝֗ן נְצָרָ֥תַם נַפְשִֽׁי׃
130דְּבָרֶ֥יךָ יָאִ֗יר מֵבִ֥ין פְּתָיִֽים׃
131פָ֭עַרְתִּי וָאֶשְׁאָ֑פָה כִּ֖י לְמִצְוֹתֶ֣יךָ יָאָֽבְתִּי׃
132אֵלַ֥י וְחָנֵּ֑נִי כְּ֝מִשְׁפָּ֗ט לְאֹהֲבֵ֥י שְׁמֶֽךָ׃
133הָכֵ֣ן בְּאִמְרָתֶ֑ךָ וְֽאַל־ תַּשְׁלֶט־ בִּ֥י כָל־ אָֽוֶן׃
134מֵעֹ֣שֶׁק אָדָ֑ם וְ֝אֶשְׁמְרָ֗ה פִּקּוּדֶֽיךָ׃
135הָאֵ֣ר בְּעַבְדֶּ֑ךָ וְ֝לַמְּדֵ֗נִי אֶת־ חֻקֶּֽיךָ׃
136מַ֭יִם יָרְד֣וּ עֵינָ֑י עַ֝֗ל לֹא־ שָׁמְר֥וּ תוֹרָתֶֽךָ׃
137אַתָּ֣ה יְהוָ֑ה וְ֝יָשָׁ֗ר מִשְׁפָּטֶֽיךָ׃
138צֶ֣דֶק עֵדֹתֶ֑יךָ וֶֽאֱמוּנָ֥ה מְאֹֽד׃
139קִנְאָתִ֑י כִּֽי־ שָׁכְח֖וּ דְבָרֶ֣יךָ צָרָֽי׃
140אִמְרָתְךָ֥ מְאֹ֗ד וְֽעַבְדְּךָ֥ אֲהֵבָֽהּ׃
141אָנֹכִ֣י וְנִבְזֶ֑ה פִּ֝קֻּדֶ֗יךָ לֹ֣א שָׁכָֽחְתִּי׃
142צֶ֣דֶק לְעוֹלָ֑ם וְֽתוֹרָתְךָ֥ אֱמֶֽת׃
143וּמָצ֥וֹק מְצָא֑וּנִי מִ֝צְוֹתֶ֗יךָ שַׁעֲשֻׁעָֽי׃
144עֵדְוֹתֶ֥יךָ לְעוֹלָ֗ם הֲבִינֵ֥נִי וְאֶחְיֶֽה׃
145בְכָל־ לֵ֭ב עֲנֵ֥נִי יְהוָ֗ה חֻקֶּ֥יךָ אֶצֹּֽרָה׃
146הוֹשִׁיעֵ֑נִי וְ֝אֶשְׁמְרָ֗ה עֵדֹתֶֽיךָ׃
147בַ֭נֶּשֶׁף וָאֲשַׁוֵּ֑עָה [לִדְבָרֶיךָ כ] (לִדְבָרְךָ֥ ק) יִחָֽלְתִּי׃
148עֵ֭ינַי אַשְׁמֻר֑וֹת לָ֝שִׂ֗יחַ בְּאִמְרָתֶֽךָ׃
149שִׁמְעָ֣ה כְחַסְדֶּ֑ךָ יְ֝הוָ֗ה כְּֽמִשְׁפָּטֶ֥ךָ חַיֵּֽנִי׃
150רֹדְפֵ֣י זִמָּ֑ה מִתּוֹרָתְךָ֥ רָחָֽקוּ׃
151אַתָּ֣ה יְהוָ֑ה וְֽכָל־ מִצְוֹתֶ֥יךָ אֱמֶֽת׃
152יָ֭דַעְתִּי מֵעֵדֹתֶ֑יךָ כִּ֖י לְעוֹלָ֣ם יְסַדְתָּֽם׃
153עָנְיִ֥י וְחַלְּצֵ֑נִי כִּי־ תֽ֝וֹרָתְךָ֗ לֹ֣א שָׁכָֽחְתִּי׃
154רִ֭יבִי וּגְאָלֵ֑נִי לְאִמְרָתְךָ֥ חַיֵּֽנִי׃
155מֵרְשָׁעִ֣ים יְשׁוּעָ֑ה כִּֽי־ חֻ֝קֶּיךָ לֹ֣א דָרָֽשׁוּ׃
156רַבִּ֥ים ׀ יְהוָ֑ה כְּֽמִשְׁפָּטֶ֥יךָ חַיֵּֽנִי׃
157רֹדְפַ֣י וְצָרָ֑י מֵ֝עֵדְוֹתֶ֗יךָ לֹ֣א נָטִֽיתִי׃
158בֹ֭גְדִים וָֽאֶתְקוֹטָ֑טָה אֲשֶׁ֥ר אִ֝מְרָתְךָ֗ לֹ֣א שָׁמָֽרוּ׃
159כִּי־ פִקּוּדֶ֣יךָ אָהָ֑בְתִּי יְ֝הוָ֗ה כְּֽחַסְדְּךָ֥ חַיֵּֽנִי׃
160דְּבָרְךָ֥ אֱמֶ֑ת וּ֝לְעוֹלָ֗ם כָּל־ מִשְׁפַּ֥ט צִדְקֶֽךָ׃
161רְדָפ֣וּנִי חִנָּ֑ם [וּמִדְּבָרֶיךָ כ] (וּ֝מִדְּבָרְךָ֗ ק) פָּחַ֥ד לִבִּֽי׃
162אָ֭נֹכִֽי עַל־ אִמְרָתֶ֑ךָ כְּ֝מוֹצֵ֗א שָׁלָ֥ל רָֽב׃
163שָׂ֭נֵאתִי וַאֲתַעֵ֑בָה תּוֹרָתְךָ֥ אָהָֽבְתִּי׃
164בַּ֭יּוֹם הִלַּלְתִּ֑יךָ עַ֝֗ל מִשְׁפְּטֵ֥י צִדְקֶֽךָ׃
165רָ֭ב לְאֹהֲבֵ֣י תוֹרָתֶ֑ךָ וְאֵֽין־ לָ֥מוֹ מִכְשֽׁוֹל׃
166לִֽישׁוּעָתְךָ֣ יְהוָ֑ה וּֽמִצְוֹתֶ֥יךָ עָשִֽׂיתִי׃
167נַ֭פְשִׁי עֵדֹתֶ֑יךָ וָאֹהֲבֵ֥ם מְאֹֽד׃
168פִ֭קּוּדֶיךָ וְעֵדֹתֶ֑יךָ כִּ֖י כָל־ דְּרָכַ֣י נֶגְדֶּֽךָ׃
169רִנָּתִ֣י לְפָנֶ֣יךָ יְהוָ֑ה כִּדְבָרְךָ֥ הֲבִינֵֽנִי׃
170תְּחִנָּתִ֣י לְפָנֶ֑יךָ כְּ֝אִמְרָתְךָ֗ הַצִּילֵֽנִי׃
171שְׂפָתַ֣י תְּהִלָּ֑ה כִּ֖י תְלַמְּדֵ֣נִי חֻקֶּֽיךָ׃
172לְ֭שׁוֹנִי אִמְרָתֶ֑ךָ כִּ֖י כָל־ מִצְוֹתֶ֣יךָ צֶּֽדֶק׃
173יָדְךָ֥ לְעָזְרֵ֑נִי כִּ֖י פִקּוּדֶ֣יךָ בָחָֽרְתִּי׃
174לִֽישׁוּעָתְךָ֣ יְהוָ֑ה וְ֝תֽוֹרָתְךָ֗ שַׁעֲשֻׁעָֽי׃
