1וַיַּעַל֩ אַבְרָ֨ם מִמִּצְרַ֜יִם ה֠וּא וְאִשְׁתּ֧וֹ וְכָל־ אֲשֶׁר־ ל֛וֹ וְל֥וֹט עִמּ֖וֹ הַנֶּֽגְבָּה׃
2כָּבֵ֣ד מְאֹ֑ד בַּמִּקְנֶ֕ה בַּכֶּ֖סֶף וּבַזָּהָֽב׃
3לְמַסָּעָ֔יו מִנֶּ֖גֶב וְעַד־ בֵּֽית־ אֵ֑ל עַד־ הַמָּק֗וֹם אֲשֶׁר־ הָ֨יָה שָׁ֤ם [אָהֳלֹה כ] (אָֽהֳלֹו֙ ק) בַּתְּחִלָּ֔ה בֵּ֥ין בֵּֽית־ אֵ֖ל וּבֵ֥ין הָעָֽי׃
4מְקוֹם֙ הַמִּזְבֵּ֔חַ אֲשֶׁר־ עָ֥שָׂה שָׁ֖ם בָּרִאשֹׁנָ֑ה וַיִּקְרָ֥א שָׁ֛ם אַבְרָ֖ם בְּשֵׁ֥ם יְהוָֽה׃
5לְל֔וֹט הַהֹלֵ֖ךְ אֶת־ אַבְרָ֑ם הָיָ֥ה צֹאן־ וּבָקָ֖ר וְאֹהָלִֽים׃
6נָשָׂ֥א אֹתָ֛ם הָאָ֖רֶץ לָשֶׁ֣בֶת יַחְדָּ֑ו כִּֽי־ הָיָ֤ה רְכוּשָׁם֙ רָ֔ב וְלֹ֥א יָֽכְל֖וּ לָשֶׁ֥בֶת יַחְדָּֽו׃
7רִ֗יב בֵּ֚ין רֹעֵ֣י מִקְנֵֽה־ אַבְרָ֔ם וּבֵ֖ין רֹעֵ֣י מִקְנֵה־ ל֑וֹט וְהַֽכְּנַעֲנִי֙ וְהַפְּרִזִּ֔י אָ֖ז יֹשֵׁ֥ב בָּאָֽרֶץ׃
8אַבְרָ֜ם אֶל־ ל֗וֹט אַל־ נָ֨א תְהִ֤י מְרִיבָה֙ בֵּינִ֣י וּבֵינֶ֔יךָ וּבֵ֥ין רֹעַ֖י וּבֵ֣ין רֹעֶ֑יךָ כִּֽי־ אֲנָשִׁ֥ים אַחִ֖ים אֲנָֽחְנוּ׃
9כָל־ הָאָ֙רֶץ֙ לְפָנֶ֔יךָ הִפָּ֥רֶד נָ֖א מֵעָלָ֑י אִם־ הַשְּׂמֹ֣אל וְאֵימִ֔נָה וְאִם־ הַיָּמִ֖ין וְאַשְׂמְאִֽילָה׃
10ל֣וֹט אֶת־ עֵינָ֗יו וַיַּרְא֙ אֶת־ כָּל־ כִּכַּ֣ר הַיַּרְדֵּ֔ן כִּ֥י כֻלָּ֖הּ מַשְׁקֶ֑ה לִפְנֵ֣י ׀ שַׁחֵ֣ת יְהוָ֗ה אֶת־ סְדֹם֙ וְאֶת־ עֲמֹרָ֔ה כְּגַן־ יְהוָה֙ כְּאֶ֣רֶץ מִצְרַ֔יִם בֹּאֲכָ֖ה צֹֽעַר׃
11ל֣וֹ ל֗וֹט אֵ֚ת כָּל־ כִּכַּ֣ר הַיַּרְדֵּ֔ן וַיִּסַּ֥ע ל֖וֹט מִקֶּ֑דֶם וַיִּפָּ֣רְד֔וּ אִ֖ישׁ מֵעַ֥ל אָחִֽיו׃
12יָשַׁ֣ב בְּאֶֽרֶץ־ כְּנָ֑עַן וְל֗וֹט יָשַׁב֙ בְּעָרֵ֣י הַכִּכָּ֔ר וַיֶּאֱהַ֖ל עַד־ סְדֹֽם׃
13סְדֹ֔ם רָעִ֖ים וְחַטָּאִ֑ים לַיהוָ֖ה מְאֹֽד׃
14אָמַ֣ר אֶל־ אַבְרָ֗ם אַחֲרֵי֙ הִפָּֽרֶד־ ל֣וֹט מֵֽעִמּ֔וֹ שָׂ֣א נָ֤א עֵינֶ֙יךָ֙ וּרְאֵ֔ה מִן־ הַמָּק֖וֹם אֲשֶׁר־ אַתָּ֣ה שָׁ֑ם צָפֹ֥נָה וָנֶ֖גְבָּה וָקֵ֥דְמָה וָיָֽמָּה׃
15אֶת־ כָּל־ הָאָ֛רֶץ אֲשֶׁר־ אַתָּ֥ה רֹאֶ֖ה לְךָ֣ אֶתְּנֶ֑נָּה וּֽלְזַרְעֲךָ֖ עַד־ עוֹלָֽם׃
16אֶֽת־ זַרְעֲךָ֖ כַּעֲפַ֣ר הָאָ֑רֶץ אֲשֶׁ֣ר ׀ אִם־ יוּכַ֣ל אִ֗ישׁ לִמְנוֹת֙ אֶת־ עֲפַ֣ר הָאָ֔רֶץ גַּֽם־ זַרְעֲךָ֖ יִמָּנֶֽה׃
17הִתְהַלֵּ֣ךְ בָּאָ֔רֶץ לְאָרְכָּ֖הּ וּלְרָחְבָּ֑הּ כִּ֥י לְךָ֖ אֶתְּנֶֽנָּה׃
