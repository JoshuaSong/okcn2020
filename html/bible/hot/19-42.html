1לַמְנַצֵּ֗חַ מַשְׂכִּ֥יל לִבְנֵי־ קֹֽרַח׃ כְּאַיָּ֗ל תַּעֲרֹ֥ג עַל־ אֲפִֽיקֵי־ מָ֑יִם כֵּ֤ן נַפְשִׁ֨י תַעֲרֹ֖ג אֵלֶ֣יךָ אֱלֹהִֽים׃
2נַפְשִׁ֨י ׀ לֵאלֹהִים֮ לְאֵ֪ל חָ֥י מָתַ֥י אָב֑וֹא וְ֝אֵרָאֶ֗ה פְּנֵ֣י אֱלֹהִֽים׃
3לִּ֬י דִמְעָתִ֣י לֶ֭חֶם יוֹמָ֣ם וָלָ֑יְלָה בֶּאֱמֹ֥ר אֵלַ֥י כָּל־ הַ֝יּ֗וֹם אַיֵּ֥ה אֱלֹהֶֽיךָ׃
4אֶזְכְּרָ֨ה ׀ וְאֶשְׁפְּכָ֬ה עָלַ֨י ׀ נַפְשִׁ֗י כִּ֤י אֶֽעֱבֹ֨ר ׀ בַּסָּךְ֮ אֶדַּדֵּ֗ם עַד־ בֵּ֥ית אֱלֹ֫הִ֥ים בְּקוֹל־ רִנָּ֥ה וְתוֹדָ֗ה הָמ֥וֹן חוֹגֵֽג׃
5תִּשְׁתּ֬וֹחֲחִ֨י ׀ נַפְשִׁי֮ וַתֶּהֱמִ֪י עָ֫לָ֥י הוֹחִ֣ילִי לֵֽ֭אלֹהִים כִּי־ ע֥וֹד אוֹדֶ֗נּוּ יְשׁוּע֥וֹת פָּנָֽיו׃
6עָלַי֮ נַפְשִׁ֪י תִשְׁתּ֫וֹחָ֥ח עַל־ כֵּ֗ן אֶ֭זְכָּרְךָ מֵאֶ֣רֶץ יַרְדֵּ֑ן וְ֝חֶרְמוֹנִ֗ים מֵהַ֥ר מִצְעָֽר׃ תְּהֽוֹם־ אֶל־
7ק֭וֹרֵא לְק֣וֹל צִנּוֹרֶ֑יךָ כָּֽל־ מִשְׁבָּרֶ֥יךָ וְ֝גַלֶּ֗יךָ עָלַ֥י עָבָֽרוּ׃
8יְצַוֶּ֬ה יְהוָ֨ה ׀ חַסְדּ֗וֹ וּ֭בַלַּיְלָה [שִׁירָהּ כ] (שִׁירֹ֣ו ק) עִמִּ֑י תְּ֝פִלָּ֗ה לְאֵ֣ל חַיָּֽי׃
9לְאֵ֥ל סַלְעִי֮ לָמָ֪ה שְׁכַ֫חְתָּ֥נִי לָֽמָּה־ קֹדֵ֥ר אֵלֵ֗ךְ בְּלַ֣חַץ אוֹיֵֽב׃
10בְּֽעַצְמוֹתַ֗י חֵרְפ֥וּנִי צוֹרְרָ֑י בְּאָמְרָ֥ם אֵלַ֥י כָּל־ הַ֝יּ֗וֹם אַיֵּ֥ה אֱלֹהֶֽיךָ׃
