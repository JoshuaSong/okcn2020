1לַמְנַצֵּ֬חַ ׀ לִבְנֵי־ קֹ֬רַח מִזְמֽוֹר׃ רָצִ֣יתָ יְהוָ֣ה אַרְצֶ֑ךָ שַׁ֝֗בְתָּ [שְׁבוּת כ] (שְׁבִ֣ית ק) יַעֲקֹֽב׃
2עֲוֹ֣ן עַמֶּ֑ךָ כִּסִּ֖יתָ כָל־ חַטָּאתָ֣ם סֶֽלָה׃
3כָל־ עֶבְרָתֶ֑ךָ הֱ֝שִׁיב֗וֹתָ מֵחֲר֥וֹן אַפֶּֽךָ׃
4אֱלֹהֵ֣י יִשְׁעֵ֑נוּ וְהָפֵ֖ר כַּֽעַסְךָ֣ עִמָּֽנוּ׃
5תֶּֽאֱנַף־ בָּ֑נוּ תִּמְשֹׁ֥ךְ אַ֝פְּךָ֗ לְדֹ֣ר וָדֹֽר׃
6אַ֭תָּה תָּשׁ֣וּב תְּחַיֵּ֑נוּ וְ֝עַמְּךָ֗ יִשְׂמְחוּ־ בָֽךְ׃
7יְהוָ֣ה חַסְדֶּ֑ךָ וְ֝יֶשְׁעֲךָ֗ תִּתֶּן־ לָֽנוּ׃
8מַה־ יְדַבֵּר֮ הָאֵ֪ל ׀ יְה֫וָ֥ה כִּ֤י ׀ יְדַבֵּ֬ר שָׁל֗וֹם אֶל־ עַמּ֥וֹ וְאֶל־ חֲסִידָ֑יו וְֽאַל־ יָשׁ֥וּבוּ לְכִסְלָֽה׃
9קָר֣וֹב לִירֵאָ֣יו יִשְׁע֑וֹ לִשְׁכֹּ֖ן כָּב֣וֹד בְּאַרְצֵֽנוּ׃
10וֶאֱמֶ֥ת נִפְגָּ֑שׁוּ צֶ֖דֶק וְשָׁל֣וֹם נָשָֽׁקוּ׃
11מֵאֶ֣רֶץ תִּצְמָ֑ח וְ֝צֶ֗דֶק מִשָּׁמַ֥יִם נִשְׁקָֽף׃
12יְ֭הוָה יִתֵּ֣ן הַטּ֑וֹב וְ֝אַרְצֵ֗נוּ תִּתֵּ֥ן יְבוּלָֽהּ׃
