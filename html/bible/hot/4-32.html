1וּמִקְנֶ֣ה ׀ רַ֗ב הָיָ֞ה לִבְנֵ֧י רְאוּבֵ֛ן וְלִבְנֵי־ גָ֖ד עָצ֣וּם מְאֹ֑ד וַיִּרְא֞וּ אֶת־ אֶ֤רֶץ יַעְזֵר֙ וְאֶת־ אֶ֣רֶץ גִּלְעָ֔ד וְהִנֵּ֥ה הַמָּק֖וֹם מְק֥וֹם מִקְנֶֽה׃
2בְנֵֽי־ גָ֖ד וּבְנֵ֣י רְאוּבֵ֑ן וַיֹּאמְר֤וּ אֶל־ מֹשֶׁה֙ וְאֶל־ אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְאֶל־ נְשִׂיאֵ֥י הָעֵדָ֖ה לֵאמֹֽר׃
3וְדִיבֹן֙ וְיַעְזֵ֣ר וְנִמְרָ֔ה וְחֶשְׁבּ֖וֹן וְאֶלְעָלֵ֑ה וּשְׂבָ֥ם וּנְב֖וֹ וּבְעֹֽן׃
4אֲשֶׁ֨ר הִכָּ֤ה יְהוָה֙ לִפְנֵי֙ עֲדַ֣ת יִשְׂרָאֵ֔ל אֶ֥רֶץ מִקְנֶ֖ה הִ֑וא וְלַֽעֲבָדֶ֖יךָ מִקְנֶֽה׃ ס
5אִם־ מָצָ֤אנוּ חֵן֙ בְּעֵינֶ֔יךָ יֻתַּ֞ן אֶת־ הָאָ֧רֶץ הַזֹּ֛את לַעֲבָדֶ֖יךָ לַאֲחֻזָּ֑ה אַל־ תַּעֲבִרֵ֖נוּ אֶת־ הַיַּרְדֵּֽן׃
6מֹשֶׁ֔ה לִבְנֵי־ גָ֖ד וְלִבְנֵ֣י רְאוּבֵ֑ן הַאַֽחֵיכֶ֗ם יָבֹ֙אוּ֙ לַמִּלְחָמָ֔ה וְאַתֶּ֖ם תֵּ֥שְׁבוּ פֹֽה׃
7[תְנוּאוּן כ] (תְנִיא֔וּן ק) אֶת־ לֵ֖ב בְּנֵ֣י יִשְׂרָאֵ֑ל מֵֽעֲבֹר֙ אֶל־ הָאָ֔רֶץ אֲשֶׁר־ נָתַ֥ן לָהֶ֖ם יְהוָֽה׃
8עָשׂ֖וּ אֲבֹתֵיכֶ֑ם בְּשָׁלְחִ֥י אֹתָ֛ם מִקָּדֵ֥שׁ בַּרְנֵ֖עַ לִרְא֥וֹת אֶת־ הָאָֽרֶץ׃
9עַד־ נַ֣חַל אֶשְׁכּ֗וֹל וַיִּרְאוּ֙ אֶת־ הָאָ֔רֶץ וַיָּנִ֕יאוּ אֶת־ לֵ֖ב בְּנֵ֣י יִשְׂרָאֵ֑ל לְבִלְתִּי־ בֹא֙ אֶל־ הָאָ֔רֶץ אֲשֶׁר־ נָתַ֥ן לָהֶ֖ם יְהוָֽה׃
10אַ֥ף יְהוָ֖ה בַּיּ֣וֹם הַה֑וּא וַיִּשָּׁבַ֖ע לֵאמֹֽר׃
11יִרְא֨וּ הָאֲנָשִׁ֜ים הָעֹלִ֣ים מִמִּצְרַ֗יִם מִבֶּ֨ן עֶשְׂרִ֤ים שָׁנָה֙ וָמַ֔עְלָה אֵ֚ת הָאֲדָמָ֔ה אֲשֶׁ֥ר נִשְׁבַּ֛עְתִּי לְאַבְרָהָ֥ם לְיִצְחָ֖ק וּֽלְיַעֲקֹ֑ב כִּ֥י לֹא־ מִלְא֖וּ אַחֲרָֽי׃
12כָּלֵ֤ב בֶּן־ יְפֻנֶּה֙ הַקְּנִזִּ֔י וִיהוֹשֻׁ֖עַ בִּן־ נ֑וּן כִּ֥י מִלְא֖וּ אַחֲרֵ֥י יְהוָֽה׃
13אַ֤ף יְהוָה֙ בְּיִשְׂרָאֵ֔ל וַיְנִעֵם֙ בַּמִּדְבָּ֔ר אַרְבָּעִ֖ים שָׁנָ֑ה עַד־ תֹּם֙ כָּל־ הַדּ֔וֹר הָעֹשֶׂ֥ה הָרַ֖ע בְּעֵינֵ֥י יְהוָֽה׃
14קַמְתֶּ֗ם תַּ֚חַת אֲבֹ֣תֵיכֶ֔ם תַּרְבּ֖וּת אֲנָשִׁ֣ים חַטָּאִ֑ים לִסְפּ֣וֹת ע֗וֹד עַ֛ל חֲר֥וֹן אַף־ יְהוָ֖ה אֶל־ יִשְׂרָאֵֽל׃
15תְשׁוּבֻן֙ מֵֽאַחֲרָ֔יו וְיָסַ֣ף ע֔וֹד לְהַנִּיח֖וֹ בַּמִּדְבָּ֑ר וְשִֽׁחַתֶּ֖ם לְכָל־ הָעָ֥ם הַזֶּֽה׃ ס
16אֵלָיו֙ וַ֣יֹּאמְר֔וּ גִּדְרֹ֥ת צֹ֛אן נִבְנֶ֥ה לְמִקְנֵ֖נוּ פֹּ֑ה וְעָרִ֖ים לְטַפֵּֽנוּ׃
17נֵחָלֵ֣ץ חֻשִׁ֗ים לִפְנֵי֙ בְּנֵ֣י יִשְׂרָאֵ֔ל עַ֛ד אֲשֶׁ֥ר אִם־ הֲבִֽיאֹנֻ֖ם אֶל־ מְקוֹמָ֑ם וְיָשַׁ֤ב טַפֵּ֙נוּ֙ בְּעָרֵ֣י הַמִּבְצָ֔ר מִפְּנֵ֖י יֹשְׁבֵ֥י הָאָֽרֶץ׃
18נָשׁ֖וּב אֶל־ בָּתֵּ֑ינוּ עַ֗ד הִתְנַחֵל֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אִ֖ישׁ נַחֲלָתֽוֹ׃
19לֹ֤א נִנְחַל֙ אִתָּ֔ם מֵעֵ֥בֶר לַיַּרְדֵּ֖ן וָהָ֑לְאָה כִּ֣י בָ֤אָה נַחֲלָתֵ֙נוּ֙ אֵלֵ֔ינוּ מֵעֵ֥בֶר הַיַּרְדֵּ֖ן מִזְרָֽחָה׃ פ
20אֲלֵיהֶם֙ מֹשֶׁ֔ה אִֽם־ תַּעֲשׂ֖וּן אֶת־ הַדָּבָ֣ר הַזֶּ֑ה אִם־ תֵּחָ֥לְצ֛וּ לִפְנֵ֥י יְהוָ֖ה לַמִּלְחָמָֽה׃
21לָכֶ֧ם כָּל־ חָל֛וּץ אֶת־ הַיַּרְדֵּ֖ן לִפְנֵ֣י יְהוָ֑ה עַ֧ד הוֹרִישׁ֛וֹ אֶת־ אֹיְבָ֖יו מִפָּנָֽיו׃
22הָאָ֜רֶץ לִפְנֵ֤י יְהוָה֙ וְאַחַ֣ר תָּשֻׁ֔בוּ וִהְיִיתֶ֧ם נְקִיִּ֛ים מֵיְהוָ֖ה וּמִיִּשְׂרָאֵ֑ל וְ֠הָיְתָה הָאָ֨רֶץ הַזֹּ֥את לָכֶ֛ם לַאֲחֻזָּ֖ה לִפְנֵ֥י יְהוָֽה׃
23לֹ֤א תַעֲשׂוּן֙ כֵּ֔ן הִנֵּ֥ה חֲטָאתֶ֖ם לַיהוָ֑ה וּדְעוּ֙ חַטַּאתְכֶ֔ם אֲשֶׁ֥ר תִּמְצָ֖א אֶתְכֶֽם׃
24לָכֶ֤ם עָרִים֙ לְטַפְּכֶ֔ם וּגְדֵרֹ֖ת לְצֹנַאֲכֶ֑ם וְהַיֹּצֵ֥א מִפִּיכֶ֖ם תַּעֲשֽׂוּ׃
25בְּנֵי־ גָד֙ וּבְנֵ֣י רְאוּבֵ֔ן אֶל־ מֹשֶׁ֖ה לֵאמֹ֑ר עֲבָדֶ֣יךָ יַעֲשׂ֔וּ כַּאֲשֶׁ֥ר אֲדֹנִ֖י מְצַוֶּֽה׃
26נָשֵׁ֔ינוּ מִקְנֵ֖נוּ וְכָל־ בְּהֶמְתֵּ֑נוּ יִֽהְיוּ־ שָׁ֖ם בְּעָרֵ֥י הַגִּלְעָֽד׃
27יַֽעַבְר֜וּ כָּל־ חֲל֥וּץ צָבָ֛א לִפְנֵ֥י יְהוָ֖ה לַמִּלְחָמָ֑ה כַּאֲשֶׁ֥ר אֲדֹנִ֖י דֹּבֵֽר׃
28לָהֶם֙ מֹשֶׁ֔ה אֵ֚ת אֶלְעָזָ֣ר הַכֹּהֵ֔ן וְאֵ֖ת יְהוֹשֻׁ֣עַ בִּן־ נ֑וּן וְאֶת־ רָאשֵׁ֛י אֲב֥וֹת הַמַּטּ֖וֹת לִבְנֵ֥י יִשְׂרָאֵֽל׃
29מֹשֶׁ֜ה אֲלֵהֶ֗ם אִם־ יַעַבְר֣וּ בְנֵי־ גָ֣ד וּבְנֵי־ רְאוּבֵ֣ן ׀ אִ֠תְּכֶם אֶֽת־ הַיַּרְדֵּ֞ן כָּל־ חָל֤וּץ לַמִּלְחָמָה֙ לִפְנֵ֣י יְהוָ֔ה וְנִכְבְּשָׁ֥ה הָאָ֖רֶץ לִפְנֵיכֶ֑ם וּנְתַתֶּ֥ם לָהֶ֛ם אֶת־ אֶ֥רֶץ הַגִּלְעָ֖ד לַאֲחֻזָּֽה׃
30לֹ֧א יַֽעַבְר֛וּ חֲלוּצִ֖ים אִתְּכֶ֑ם וְנֹֽאחֲז֥וּ בְתֹכְכֶ֖ם בְּאֶ֥רֶץ כְּנָֽעַן׃
31בְנֵי־ גָ֛ד וּבְנֵ֥י רְאוּבֵ֖ן לֵאמֹ֑ר אֵת֩ אֲשֶׁ֨ר דִּבֶּ֧ר יְהוָ֛ה אֶל־ עֲבָדֶ֖יךָ כֵּ֥ן נַעֲשֶֽׂה׃
32נַעֲבֹ֧ר חֲלוּצִ֛ים לִפְנֵ֥י יְהוָ֖ה אֶ֣רֶץ כְּנָ֑עַן וְאִתָּ֙נוּ֙ אֲחֻזַּ֣ת נַחֲלָתֵ֔נוּ מֵעֵ֖בֶר לַיַּרְדֵּֽן׃
33לָהֶ֣ם ׀ מֹשֶׁ֡ה לִבְנֵי־ גָד֩ וְלִבְנֵ֨י רְאוּבֵ֜ן וְלַחֲצִ֣י ׀ שֵׁ֣בֶט ׀ מְנַשֶּׁ֣ה בֶן־ יוֹסֵ֗ף אֶת־ מַמְלֶ֙כֶת֙ סִיחֹן֙ מֶ֣לֶךְ הָֽאֱמֹרִ֔י וְאֶת־ מַמְלֶ֔כֶת ע֖וֹג מֶ֣לֶךְ הַבָּשָׁ֑ן הָאָ֗רֶץ לְעָרֶ֙יהָ֙ בִּגְבֻלֹ֔ת עָרֵ֥י הָאָ֖רֶץ סָבִֽיב׃
34בְנֵי־ גָ֔ד אֶת־ דִּיבֹ֖ן וְאֶת־ עֲטָרֹ֑ת וְאֵ֖ת עֲרֹעֵֽר׃
35עַטְרֹ֥ת שׁוֹפָ֛ן וְאֶת־ יַעְזֵ֖ר וְיָגְבֳּהָֽה׃
36בֵּ֥ית נִמְרָ֖ה וְאֶת־ בֵּ֣ית הָרָ֑ן עָרֵ֥י מִבְצָ֖ר וְגִדְרֹ֥ת צֹֽאן׃
37רְאוּבֵן֙ בָּנ֔וּ אֶת־ חֶשְׁבּ֖וֹן וְאֶת־ אֶלְעָלֵ֑א וְאֵ֖ת קִרְיָתָֽיִם׃
38נְב֞וֹ וְאֶת־ בַּ֧עַל מְע֛וֹן מֽוּסַבֹּ֥ת שֵׁ֖ם וְאֶת־ שִׂבְמָ֑ה וַיִּקְרְא֣וּ בְשֵׁמֹ֔ת אֶת־ שְׁמ֥וֹת הֶעָרִ֖ים אֲשֶׁ֥ר בָּנֽוּ׃
39בְּנֵ֨י מָכִ֧יר בֶּן־ מְנַשֶּׁ֛ה גִּלְעָ֖דָה וַֽיִּלְכְּדֻ֑הָ וַיּ֖וֹרֶשׁ אֶת־ הָאֱמֹרִ֥י אֲשֶׁר־ בָּֽהּ׃
40מֹשֶׁה֙ אֶת־ הַגִּלְעָ֔ד לְמָכִ֖יר בֶּן־ מְנַשֶּׁ֑ה וַיֵּ֖שֶׁב בָּֽהּ׃
41בֶּן־ מְנַשֶּׁה֙ הָלַ֔ךְ וַיִּלְכֹּ֖ד אֶת־ חַוֹּתֵיהֶ֑ם וַיִּקְרָ֥א אֶתְהֶ֖ן חַוֹּ֥ת יָאִֽיר׃
