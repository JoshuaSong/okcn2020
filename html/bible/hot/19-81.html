1לַמְנַצֵּ֬חַ ׀ עַֽל־ הַגִּתִּ֬ית לְאָסָֽף׃ הַ֭רְנִינוּ לֵאלֹהִ֣ים עוּזֵּ֑נוּ הָ֝רִ֗יעוּ לֵאלֹהֵ֥י יַעֲקֹֽב׃
2זִ֭מְרָה וּתְנוּ־ תֹ֑ף כִּנּ֖וֹר נָעִ֣ים עִם־ נָֽבֶל׃
3בַחֹ֣דֶשׁ שׁוֹפָ֑ר בַּ֝כֵּ֗סֶה לְי֣וֹם חַגֵּֽנוּ׃
4חֹ֣ק לְיִשְׂרָאֵ֣ל ה֑וּא מִ֝שְׁפָּ֗ט לֵאלֹהֵ֥י יַעֲקֹֽב׃
5בִּֽיה֘וֹסֵ֤ף שָׂמ֗וֹ בְּ֭צֵאתוֹ עַל־ אֶ֣רֶץ מִצְרָ֑יִם שְׂפַ֖ת לֹא־ יָדַ֣עְתִּי אֶשְׁמָֽע׃
6מִסֵּ֣בֶל שִׁכְמ֑וֹ כַּ֝פָּ֗יו מִדּ֥וּד תַּעֲבֹֽרְנָה׃
7קָרָ֗אתָ וָאֲחַ֫לְּצֶ֥ךָּ אֶ֭עֶנְךָ בְּסֵ֣תֶר רַ֑עַם אֶבְחָֽנְךָ֨ עַל־ מֵ֖י מְרִיבָ֣ה סֶֽלָה׃
8עַ֭מִּי וְאָעִ֣ידָה בָּ֑ךְ יִ֝שְׂרָאֵ֗ל אִם־ תִּֽשְׁמַֽע־ לִֽי׃
9יִהְיֶ֣ה בְ֭ךָ אֵ֣ל זָ֑ר וְלֹ֥א תִ֝שְׁתַּחֲוֶ֗ה לְאֵ֣ל נֵכָֽר׃
10יְה֘וָ֤ה אֱלֹהֶ֗יךָ הַֽ֭מַּעַלְךָ מֵאֶ֣רֶץ מִצְרָ֑יִם הַרְחֶב־ פִּ֝֗יךָ וַאֲמַלְאֵֽהוּ׃
11שָׁמַ֣ע עַמִּ֣י לְקוֹלִ֑י וְ֝יִשְׂרָאֵ֗ל לֹא־ אָ֥בָה לִֽי׃
12בִּשְׁרִיר֣וּת לִבָּ֑ם יֵ֝לְכ֗וּ בְּֽמוֹעֲצוֹתֵיהֶֽם׃
13עַ֭מִּי שֹׁמֵ֣עַֽ לִ֑י יִ֝שְׂרָאֵ֗ל בִּדְרָכַ֥י יְהַלֵּֽכוּ׃
14אוֹיְבֵיהֶ֣ם אַכְנִ֑יעַ וְעַ֥ל צָ֝רֵיהֶ֗ם אָשִׁ֥יב יָדִֽי׃
15יְ֭הוָה יְכַֽחֲשׁוּ־ ל֑וֹ וִיהִ֖י עִתָּ֣ם לְעוֹלָֽם׃
