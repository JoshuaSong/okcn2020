1לַמְנַצֵּ֥חַ אֶֽל־ הַנְּחִיל֗וֹת מִזְמ֥וֹר לְדָוִֽד׃ אֲמָרַ֖י הַאֲזִ֥ינָה ׀ יְהוָ֗ה בִּ֣ינָה הֲגִֽיגִי׃
2לְק֬וֹל שַׁוְעִ֗י מַלְכִּ֥י וֵאלֹהָ֑י כִּֽי־ אֵ֝לֶ֗יךָ אֶתְפַּלָּֽל׃
3בֹּ֭קֶר תִּשְׁמַ֣ע קוֹלִ֑י בֹּ֥קֶר אֶֽעֱרָךְ־ לְ֝ךָ֗ וַאֲצַפֶּֽה׃
4לֹ֤א אֵֽל־ חָפֵ֘ץ רֶ֥שַׁע ׀ אָ֑תָּה לֹ֖א יְגֻרְךָ֣ רָֽע׃
5יִתְיַצְּב֣וּ הֽ֭וֹלְלִים לְנֶ֣גֶד עֵינֶ֑יךָ שָׂ֝נֵ֗אתָ כָּל־ פֹּ֥עֲלֵי אָֽוֶן׃
6דֹּבְרֵ֪י כָ֫זָ֥ב אִישׁ־ דָּמִ֥ים וּמִרְמָ֗ה יְתָ֘עֵ֥ב ׀ יְהוָֽה׃
7בְּרֹ֣ב חַ֭סְדְּךָ אָב֣וֹא בֵיתֶ֑ךָ אֶשְׁתַּחֲוֶ֥ה אֶל־ הֵֽיכַל־ קָ֝דְשְׁךָ֗ בְּיִרְאָתֶֽךָ׃
8נְחֵ֬נִי בְצִדְקָתֶ֗ךָ לְמַ֥עַן שׁוֹרְרָ֑י [הֹושַׁר כ] (הַיְשַׁ֖ר ק) לְפָנַ֣י דַּרְכֶּֽךָ׃
9אֵ֪ין בְּפִ֡יהוּ נְכוֹנָה֮ קִרְבָּ֪ם הַ֫וּ֥וֹת קֶֽבֶר־ פָּת֥וּחַ גְּרוֹנָ֑ם לְ֝שׁוֹנָ֗ם יַחֲלִֽיקוּן׃
10אֱ‍ֽלֹהִ֗ים יִפְּלוּ֮ מִֽמֹּעֲצ֪וֹתֵ֫יהֶ֥ם בְּרֹ֣ב פִּ֭שְׁעֵיהֶם הַדִּיחֵ֑מוֹ כִּי־ מָ֥רוּ בָֽךְ׃
11כָל־ ח֪וֹסֵי בָ֡ךְ לְעוֹלָ֣ם יְ֭רַנֵּנוּ וְתָסֵ֣ךְ עָלֵ֑ימוֹ וְֽיַעְלְצ֥וּ בְ֝ךָ֗ אֹהֲבֵ֥י שְׁמֶֽךָ׃
