1וְאֶת־ בֵּיתוֹ֙ בָּנָ֣ה שְׁלֹמֹ֔ה שְׁלֹ֥שׁ עֶשְׂרֵ֖ה שָׁנָ֑ה וַיְכַ֖ל אֶת־ כָּל־ בֵּיתֽוֹ׃
2אֶת־ בֵּ֣ית ׀ יַ֣עַר הַלְּבָנ֗וֹן מֵאָ֨ה אַמָּ֤ה אָרְכּוֹ֙ וַחֲמִשִּׁ֤ים אַמָּה֙ רָחְבּ֔וֹ וּשְׁלֹשִׁ֥ים אַמָּ֖ה קוֹמָת֑וֹ עַ֗ל אַרְבָּעָה֙ טוּרֵי֙ עַמּוּדֵ֣י אֲרָזִ֔ים וּכְרֻת֥וֹת אֲרָזִ֖ים עַל־ הָעַמּוּדִֽים׃
3בָּאֶ֗רֶז מִמַּ֙עַל֙ עַל־ הַצְּלָעֹת֙ אֲשֶׁ֣ר עַל־ הָֽעַמּוּדִ֔ים אַרְבָּעִ֖ים וַחֲמִשָּׁ֑ה חֲמִשָּׁ֥ה עָשָׂ֖ר הַטּֽוּר׃
4שְׁלֹשָׁ֣ה טוּרִ֑ים וּמֶחֱזָ֥ה אֶל־ מֶחֱזָ֖ה שָׁלֹ֥שׁ פְּעָמִֽים׃
5הַפְּתָחִ֥ים וְהַמְּזוּז֖וֹת רְבֻעִ֣ים שָׁ֑קֶף וּמ֧וּל מֶחֱזָ֛ה אֶל־ מֶחֱזָ֖ה שָׁלֹ֥שׁ פְּעָמִֽים׃
6אוּלָ֤ם הָֽעַמּוּדִים֙ עָשָׂ֔ה חֲמִשִּׁ֤ים אַמָּה֙ אָרְכּ֔וֹ וּשְׁלֹשִׁ֥ים אַמָּ֖ה רָחְבּ֑וֹ וְאוּלָם֙ עַל־ פְּנֵיהֶ֔ם וְעַמֻּדִ֥ים וְעָ֖ב עַל־ פְּנֵיהֶֽם׃
7הַכִּסֵּא֙ אֲשֶׁ֣ר יִשְׁפָּט־ שָׁ֔ם אֻלָ֥ם הַמִּשְׁפָּ֖ט עָשָׂ֑ה וְסָפ֣וּן בָּאֶ֔רֶז מֵהַקַּרְקַ֖ע עַד־ הַקַּרְקָֽע׃
8אֲשֶׁר־ יֵ֨שֶׁב שָׁ֜ם חָצֵ֣ר הָאַחֶ֗רֶת מִבֵּית֙ לָֽאוּלָ֔ם כַּמַּֽעֲשֶׂ֥ה הַזֶּ֖ה הָיָ֑ה וּבַ֜יִת יַעֲשֶׂ֤ה לְבַת־ פַּרְעֹה֙ אֲשֶׁ֣ר לָקַ֣ח שְׁלֹמֹ֔ה כָּאוּלָ֖ם הַזֶּֽה׃
9אֵ֜לֶּה אֲבָנִ֤ים יְקָרֹת֙ כְּמִדֹּ֣ת גָּזִ֔ית מְגֹרָר֥וֹת בַּמְּגֵרָ֖ה מִבַּ֣יִת וּמִח֑וּץ וּמִמַּסָּד֙ עַד־ הַטְּפָח֔וֹת וּמִח֖וּץ עַד־ הֶחָצֵ֥ר הַגְּדוֹלָֽה׃
10אֲבָנִ֥ים יְקָר֖וֹת אֲבָנִ֣ים גְּדֹל֑וֹת אַבְנֵי֙ עֶ֣שֶׂר אַמּ֔וֹת וְאַבְנֵ֖י שְׁמֹנֶ֥ה אַמּֽוֹת׃
11אֲבָנִ֧ים יְקָר֛וֹת כְּמִדּ֥וֹת גָּזִ֖ית וָאָֽרֶז׃
12הַגְּדוֹלָ֜ה סָבִ֗יב שְׁלֹשָׁה֙ טוּרִ֣ים גָּזִ֔ית וְט֖וּר כְּרֻתֹ֣ת אֲרָזִ֑ים וְלַחֲצַ֧ר בֵּית־ יְהוָ֛ה הַפְּנִימִ֖ית וּלְאֻלָ֥ם הַבָּֽיִת׃ פ
13הַמֶּ֣לֶךְ שְׁלֹמֹ֔ה וַיִּקַּ֥ח אֶת־ חִירָ֖ם מִצֹּֽר׃
14אִשָּׁה֩ אַלְמָנָ֨ה ה֜וּא מִמַּטֵּ֣ה נַפְתָּלִ֗י וְאָבִ֣יו אִישׁ־ צֹרִי֮ חֹרֵ֣שׁ נְחֹשֶׁת֒ וַ֠יִּמָּלֵא אֶת־ הַחָכְמָ֤ה וְאֶת־ הַתְּבוּנָה֙ וְאֶת־ הַדַּ֔עַת לַעֲשׂ֥וֹת כָּל־ מְלָאכָ֖ה בַּנְּחֹ֑שֶׁת וַיָּבוֹא֙ אֶל־ הַמֶּ֣לֶךְ שְׁלֹמֹ֔ה וַיַּ֖עַשׂ אֶת־ כָּל־ מְלַאכְתּֽוֹ׃
15אֶת־ שְׁנֵ֥י הָעַמּוּדִ֖ים נְחֹ֑שֶׁת שְׁמֹנֶ֨ה עֶשְׂרֵ֜ה אַמָּ֗ה קוֹמַת֙ הָעַמּ֣וּד הָאֶחָ֔ד וְחוּט֙ שְׁתֵּים־ עֶשְׂרֵ֣ה אַמָּ֔ה יָסֹ֖ב אֶת־ הָעַמּ֥וּד הַשֵּׁנִֽי׃
16כֹתָרֹ֜ת עָשָׂ֗ה לָתֵ֛ת עַל־ רָאשֵׁ֥י הָֽעַמּוּדִ֖ים מֻצַ֣ק נְחֹ֑שֶׁת חָמֵ֣שׁ אַמּ֗וֹת קוֹמַת֙ הַכֹּתֶ֣רֶת הָאֶחָ֔ת וְחָמֵ֣שׁ אַמּ֔וֹת קוֹמַ֖ת הַכֹּתֶ֥רֶת הַשֵּׁנִֽית׃
17מַעֲשֵׂ֣ה שְׂבָכָ֗ה גְּדִלִים֙ מַעֲשֵׂ֣ה שַׁרְשְׁר֔וֹת לַכֹּ֣תָרֹ֔ת אֲשֶׁ֖ר עַל־ רֹ֣אשׁ הָעַמּוּדִ֑ים שִׁבְעָה֙ לַכֹּתֶ֣רֶת הָאֶחָ֔ת וְשִׁבְעָ֖ה לַכֹּתֶ֥רֶת הַשֵּׁנִֽית׃
18אֶת־ הָעַמּוּדִ֑ים וּשְׁנֵי֩ טוּרִ֨ים סָבִ֜יב עַל־ הַשְּׂבָכָ֣ה הָאֶחָ֗ת לְכַסּ֤וֹת אֶת־ הַכֹּֽתָרֹת֙ אֲשֶׁר֙ עַל־ רֹ֣אשׁ הָֽרִמֹּנִ֔ים וְכֵ֣ן עָשָׂ֔ה לַכֹּתֶ֖רֶת הַשֵּׁנִֽית׃
19אֲשֶׁר֙ עַל־ רֹ֣אשׁ הָעַמּוּדִ֔ים מַעֲשֵׂ֖ה שׁוּשַׁ֣ן בָּֽאוּלָ֑ם אַרְבַּ֖ע אַמּֽוֹת׃
20עַל־ שְׁנֵי֙ הָֽעַמּוּדִ֔ים גַּם־ מִמַּ֙עַל֙ מִלְּעֻמַּ֣ת הַבֶּ֔טֶן אֲשֶׁ֖ר לְעֵ֣בֶר [שְׂבָכָה כ] (הַשְּׂבָכָ֑ה ק) וְהָרִמּוֹנִ֤ים מָאתַ֙יִם֙ טֻרִ֣ים סָבִ֔יב עַ֖ל הַכֹּתֶ֥רֶת הַשֵּׁנִֽית׃
21אֶת־ הָֽעַמֻּדִ֔ים לְאֻלָ֖ם הַֽהֵיכָ֑ל וַיָּ֜קֶם אֶת־ הָעַמּ֣וּד הַיְמָנִ֗י וַיִּקְרָ֤א אֶת־ שְׁמוֹ֙ יָכִ֔ין וַיָּ֙קֶם֙ אֶת־ הָעַמּ֣וּד הַשְּׂמָאלִ֔י וַיִּקְרָ֥א אֶת־ שְׁמ֖וֹ בֹּֽעַז׃
22רֹ֥אשׁ הָעַמּוּדִ֖ים מַעֲשֵׂ֣ה שׁוֹשָׁ֑ן וַתִּתֹּ֖ם מְלֶ֥אכֶת הָעַמּוּדִֽים׃
23אֶת־ הַיָּ֖ם מוּצָ֑ק עֶ֣שֶׂר בָּ֠אַמָּה מִשְּׂפָת֨וֹ עַד־ שְׂפָת֜וֹ עָגֹ֣ל ׀ סָבִ֗יב וְחָמֵ֤שׁ בָּֽאַמָּה֙ קוֹמָת֔וֹ [וּקְוֵה כ] (וְקָו֙ ק) שְׁלֹשִׁ֣ים בָּֽאַמָּ֔ה יָסֹ֥ב אֹת֖וֹ סָבִֽיב׃
24מִתַּ֨חַת לִשְׂפָת֤וֹ ׀ סָבִיב֙ סֹבְבִ֣ים אֹת֔וֹ עֶ֚שֶׂר בָּֽאַמָּ֔ה מַקִּפִ֥ים אֶת־ הַיָּ֖ם סָבִ֑יב שְׁנֵ֤י טוּרִים֙ הַפְּקָעִ֔ים יְצֻקִ֖ים בִּיצֻקָתֽוֹ׃
25עַל־ שְׁנֵ֧י עָשָׂ֣ר בָּקָ֗ר שְׁלֹשָׁ֣ה פֹנִ֣ים ׀ צָפ֡וֹנָה וּשְׁלֹשָׁה֩ פֹנִ֨ים ׀ יָ֜מָּה וּשְׁלֹשָׁ֣ה ׀ פֹּנִ֣ים נֶ֗גְבָּה וּשְׁלֹשָׁה֙ פֹּנִ֣ים מִזְרָ֔חָה וְהַיָּ֥ם עֲלֵיהֶ֖ם מִלְמָ֑עְלָה וְכָל־ אֲחֹֽרֵיהֶ֖ם בָּֽיְתָה׃
26טֶ֔פַח וּשְׂפָת֛וֹ כְּמַעֲשֵׂ֥ה שְׂפַת־ כּ֖וֹס פֶּ֣רַח שׁוֹשָׁ֑ן אַלְפַּ֥יִם בַּ֖ת יָכִֽיל׃ פ
27אֶת־ הַמְּכֹנ֛וֹת עֶ֖שֶׂר נְחֹ֑שֶׁת אַרְבַּ֣ע בָּאַמָּ֗ה אֹ֚רֶךְ הַמְּכוֹנָ֣ה הָֽאֶחָ֔ת וְאַרְבַּ֤ע בָּֽאַמָּה֙ רָחְבָּ֔הּ וְשָׁלֹ֥שׁ בָּאַמָּ֖ה קוֹמָתָֽהּ׃
28מַעֲשֵׂ֥ה הַמְּכוֹנָ֖ה מִסְגְּרֹ֣ת לָהֶ֑ם וּמִסְגְּרֹ֖ת בֵּ֥ין הַשְׁלַבִּֽים׃
29הַמִּסְגְּר֞וֹת אֲשֶׁ֣ר ׀ בֵּ֣ין הַשְׁלַבִּ֗ים אֲרָי֤וֹת ׀ בָּקָר֙ וּכְרוּבִ֔ים וְעַל־ הַשְׁלַבִּ֖ים כֵּ֣ן מִמָּ֑עַל וּמִתַּ֙חַת֙ לַאֲרָי֣וֹת וְלַבָּקָ֔ר לֹי֖וֹת מַעֲשֵׂ֥ה מוֹרָֽד׃
30אוֹפַנֵּ֨י נְחֹ֜שֶׁת לַמְּכוֹנָ֤ה הָֽאַחַת֙ וְסַרְנֵ֣י נְחֹ֔שֶׁת וְאַרְבָּעָ֥ה פַעֲמֹתָ֖יו כְּתֵפֹ֣ת לָהֶ֑ם מִתַּ֤חַת לַכִּיֹּר֙ הַכְּתֵפֹ֣ת יְצֻק֔וֹת מֵעֵ֥בֶר אִ֖ישׁ לֹיֽוֹת׃
31מִבֵּ֨ית לַכֹּתֶ֤רֶת וָמַ֙עְלָה֙ בָּֽאַמָּ֔ה וּפִ֙יהָ֙ עָגֹ֣ל מַעֲשֵׂה־ כֵ֔ן אַמָּ֖ה וַחֲצִ֣י הָֽאַמָּ֑ה וְגַם־ עַל־ פִּ֙יהָ֙ מִקְלָע֔וֹת וּמִסְגְּרֹתֵיהֶ֥ם מְרֻבָּע֖וֹת לֹ֥א עֲגֻלּֽוֹת׃
32הָאֽוֹפַנִּ֗ים לְמִתַּ֙חַת֙ לַֽמִּסְגְּר֔וֹת וִיד֥וֹת הָאֽוֹפַנִּ֖ים בַּמְּכוֹנָ֑ה וְקוֹמַת֙ הָאוֹפַ֣ן הָאֶחָ֔ד אַמָּ֖ה וַחֲצִ֥י הָאַמָּֽה׃
33הָא֣וֹפַנִּ֔ים כְּמַעֲשֵׂ֖ה אוֹפַ֣ן הַמֶּרְכָּבָ֑ה יְדוֹתָ֣ם וְגַבֵּיהֶ֗ם וְחִשֻּׁקֵיהֶ֛ם וְחִשֻּׁרֵיהֶ֖ם הַכֹּ֥ל מוּצָֽק׃
34כְּתֵפ֔וֹת אֶ֚ל אַרְבַּ֣ע פִּנּ֔וֹת הַמְּכֹנָ֖ה הָֽאֶחָ֑ת מִן־ הַמְּכֹנָ֖ה כְּתֵפֶֽיהָ׃
35הַמְּכוֹנָ֗ה חֲצִ֧י הָאַמָּ֛ה קוֹמָ֖ה עָגֹ֣ל ׀ סָבִ֑יב וְעַ֨ל רֹ֤אשׁ הַמְּכֹנָה֙ יְדֹתֶ֔יהָ וּמִסְגְּרֹתֶ֖יהָ מִמֶּֽנָּה׃
36עַל־ הַלֻּחֹת֙ יְדֹתֶ֔יהָ וְעַל֙ [וַמִסְגְּרֹתֶיהָ כ] (מִסְגְּרֹתֶ֔יהָ ק) כְּרוּבִ֖ים אֲרָי֣וֹת וְתִמֹרֹ֑ת כְּמַֽעַר־ אִ֥ישׁ וְלֹי֖וֹת סָבִֽיב׃
37עָשָׂ֔ה אֵ֖ת עֶ֣שֶׂר הַמְּכֹנ֑וֹת מוּצָ֨ק אֶחָ֜ד מִדָּ֥ה אַחַ֛ת קֶ֥צֶב אֶחָ֖ד לְכֻלָּֽהְנָה׃ ס
38עֲשָׂרָ֥ה כִיֹּר֖וֹת נְחֹ֑שֶׁת אַרְבָּעִ֨ים בַּ֜ת יָכִ֣יל ׀ הַכִּיּ֣וֹר הָאֶחָ֗ד אַרְבַּ֤ע בָּֽאַמָּה֙ הַכִּיּ֣וֹר הָאֶחָ֔ד כִּיּ֤וֹר אֶחָד֙ עַל־ הַמְּכוֹנָ֣ה הָאַחַ֔ת לְעֶ֖שֶׂר הַמְּכֹנֽוֹת׃
39אֶת־ הַמְּכֹנ֔וֹת חָמֵ֞שׁ עַל־ כֶּ֤תֶף הַבַּ֙יִת֙ מִיָּמִ֔ין וְחָמֵ֛שׁ עַל־ כֶּ֥תֶף הַבַּ֖יִת מִשְּׂמֹאל֑וֹ וְאֶת־ הַיָּ֗ם נָתַ֞ן מִכֶּ֨תֶף הַבַּ֧יִת הַיְמָנִ֛ית קֵ֖דְמָה מִמּ֥וּל נֶֽגֶב׃ ס
40חִיר֔וֹם אֶת־ הַ֨כִּיֹּר֔וֹת וְאֶת־ הַיָּעִ֖ים וְאֶת־ הַמִּזְרָק֑וֹת וַיְכַ֣ל חִירָ֗ם לַֽעֲשׂוֹת֙ אֶת־ כָּל־ הַמְּלָאכָ֔ה אֲשֶׁ֥ר עָשָׂ֛ה לַמֶּ֥לֶךְ שְׁלֹמֹ֖ה בֵּ֥ית יְהוָֽה׃
41שְׁנַ֔יִם וְגֻלֹּ֧ת הַכֹּתָרֹ֛ת אֲשֶׁר־ עַל־ רֹ֥אשׁ הָֽעַמֻּדִ֖ים שְׁתָּ֑יִם וְהַשְּׂבָכ֣וֹת שְׁתַּ֔יִם לְכַסּ֗וֹת אֶת־ שְׁתֵּי֙ גֻּלֹּ֣ת הַכֹּֽתָרֹ֔ת אֲשֶׁ֖ר עַל־ רֹ֥אשׁ הָעַמּוּדִֽים׃
42הָרִמֹּנִ֛ים אַרְבַּ֥ע מֵא֖וֹת לִשְׁתֵּ֣י הַשְּׂבָכ֑וֹת שְׁנֵֽי־ טוּרִ֤ים רִמֹּנִים֙ לַשְּׂבָכָ֣ה הָֽאֶחָ֔ת לְכַסּ֗וֹת אֶת־ שְׁתֵּי֙ גֻּלֹּ֣ת הַכֹּֽתָרֹ֔ת אֲשֶׁ֖ר עַל־ פְּנֵ֥י הָעַמּוּדִֽים׃
43הַמְּכֹנ֖וֹת עָ֑שֶׂר וְאֶת־ הַכִּיֹּרֹ֥ת עֲשָׂרָ֖ה עַל־ הַמְּכֹנֽוֹת׃
44הַיָּ֖ם הָאֶחָ֑ד וְאֶת־ הַבָּקָ֥ר שְׁנֵים־ עָשָׂ֖ר תַּ֥חַת הַיָּֽם׃
45הַסִּיר֨וֹת וְאֶת־ הַיָּעִ֜ים וְאֶת־ הַמִּזְרָק֗וֹת וְאֵת֙ כָּל־ הַכֵּלִ֣ים [הָאֹהֶל כ] (הָאֵ֔לֶּה ק) אֲשֶׁ֨ר עָשָׂ֥ה חִירָ֛ם לַמֶּ֥לֶךְ שְׁלֹמֹ֖ה בֵּ֣ית יְהוָ֑ה נְחֹ֖שֶׁת מְמֹרָֽט׃
46הַיַּרְדֵּן֙ יְצָקָ֣ם הַמֶּ֔לֶךְ בְּמַעֲבֵ֖ה הָאֲדָמָ֑ה בֵּ֥ין סֻכּ֖וֹת וּבֵ֥ין צָרְתָֽן׃
47שְׁלֹמֹה֙ אֶת־ כָּל־ הַכֵּלִ֔ים מֵרֹ֖ב מְאֹ֣ד מְאֹ֑ד לֹ֥א נֶחְקַ֖ר מִשְׁקַ֥ל הַנְּחֹֽשֶׁת׃
48שְׁלֹמֹ֔ה אֵ֚ת כָּל־ הַכֵּלִ֔ים אֲשֶׁ֖ר בֵּ֣ית יְהוָ֑ה אֵ֚ת מִזְבַּ֣ח הַזָּהָ֔ב וְאֶת־ הַשֻּׁלְחָ֗ן אֲשֶׁ֥ר עָלָ֛יו לֶ֥חֶם הַפָּנִ֖ים זָהָֽב׃
49הַ֠מְּנֹרוֹת חָמֵ֨שׁ מִיָּמִ֜ין וְחָמֵ֧שׁ מִשְּׂמֹ֛אול לִפְנֵ֥י הַדְּבִ֖יר זָהָ֣ב סָג֑וּר וְהַפֶּ֧רַח וְהַנֵּרֹ֛ת וְהַמֶּלְקַחַ֖יִם זָהָֽב׃
50וְהַֽמְזַמְּר֧וֹת וְהַמִּזְרָק֛וֹת וְהַכַּפּ֥וֹת וְהַמַּחְתּ֖וֹת זָהָ֣ב סָג֑וּר וְהַפֹּת֡וֹת לְדַלְתוֹת֩ הַבַּ֨יִת הַפְּנִימִ֜י לְקֹ֣דֶשׁ הַקֳּדָשִׁ֗ים לְדַלְתֵ֥י הַבַּ֛יִת לַהֵיכָ֖ל זָהָֽב׃ פ
