1לְדָוִ֨ד ׀ בָּ֘ר֤וּךְ יְהוָ֨ה ׀ צוּרִ֗י הַֽמְלַמֵּ֣ד יָדַ֣י לַקְרָ֑ב אֶ֝צְבְּעוֹתַ֗י לַמִּלְחָמָֽה׃
2וּמְצוּדָתִי֮ מִשְׂגַּבִּ֪י וּֽמְפַלְטִ֫י לִ֥י מָ֭גִנִּי וּב֣וֹ חָסִ֑יתִי הָרוֹדֵ֖ד עַמִּ֣י תַחְתָּֽי׃
3מָה־ אָ֭דָם וַתֵּדָעֵ֑הוּ בֶּן־ אֱ֝נ֗וֹשׁ וַֽתְּחַשְּׁבֵֽהוּ׃
4לַהֶ֣בֶל דָּמָ֑ה יָ֝מָ֗יו כְּצֵ֣ל עוֹבֵֽר׃
5הַט־ שָׁמֶ֣יךָ וְתֵרֵ֑ד גַּ֖ע בֶּהָרִ֣ים וְֽיֶעֱשָֽׁנוּ׃
6בָּ֭רָק וּתְפִיצֵ֑ם שְׁלַ֥ח חִ֝צֶּ֗יךָ וּתְהֻמֵּֽם׃
7יָדֶ֗יךָ מִמָּ֫ר֥וֹם פְּצֵ֣נִי וְ֭הַצִּילֵנִי מִמַּ֣יִם רַבִּ֑ים מִ֝יַּ֗ד בְּנֵ֣י נֵכָֽר׃
8פִּ֭יהֶם דִּבֶּר־ שָׁ֑וְא וִֽ֝ימִינָ֗ם יְמִ֣ין שָֽׁקֶר׃
9שִׁ֣יר חָ֭דָשׁ אָשִׁ֣ירָה לָּ֑ךְ בְּנֵ֥בֶל עָ֝שׂ֗וֹר אֲזַמְּרָה־ לָּֽךְ׃
10תְּשׁוּעָ֗ה לַמְּלָ֫כִ֥ים הַ֭פּוֹצֶה אֶת־ דָּוִ֥ד עַבְדּ֗וֹ מֵחֶ֥רֶב רָעָֽה׃
11וְהַצִּילֵנִי֮ מִיַּ֪ד בְּֽנֵי־ נֵ֫כָ֥ר אֲשֶׁ֣ר פִּ֭יהֶם דִּבֶּר־ שָׁ֑וְא וִֽ֝ימִינָ֗ם יְמִ֣ין שָֽׁקֶר׃
12בָּנֵ֨ינוּ ׀ כִּנְטִעִים֮ מְגֻדָּלִ֪ים בִּֽנְעוּרֵ֫יהֶ֥ם בְּנוֹתֵ֥ינוּ כְזָוִיֹּ֑ת מְ֝חֻטָּב֗וֹת תַּבְנִ֥ית הֵיכָֽל׃
13מְלֵאִים֮ מְפִיקִ֥ים מִזַּ֗ן אֶ֫ל־ זַ֥ן צֹאונֵ֣נוּ מַ֭אֲלִיפוֹת מְרֻבָּב֗וֹת בְּחוּצוֹתֵֽינוּ׃
14מְֽסֻבָּ֫לִ֥ים אֵֽין־ פֶּ֭רֶץ וְאֵ֣ין יוֹצֵ֑את וְאֵ֥ין צְ֝וָחָ֗ה בִּרְחֹבֹתֵֽינוּ׃
