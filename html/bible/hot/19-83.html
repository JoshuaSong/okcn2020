1שִׁ֖יר מִזְמ֣וֹר לְאָסָֽף׃ אֱלֹהִ֥ים אַל־ דֳּמִי־ לָ֑ךְ אַל־ תֶּחֱרַ֖שׁ וְאַל־ תִּשְׁקֹ֣ט אֵֽל׃
2הִנֵּ֣ה א֭וֹיְבֶיךָ יֶהֱמָי֑וּן וּ֝מְשַׂנְאֶ֗יךָ נָ֣שְׂאוּ רֹֽאשׁ׃
3עַ֭מְּךָ יַעֲרִ֣ימוּ ס֑וֹד וְ֝יִתְיָעֲצ֗וּ עַל־ צְפוּנֶֽיךָ׃
4לְ֭כוּ וְנַכְחִידֵ֣ם מִגּ֑וֹי וְלֹֽא־ יִזָּכֵ֖ר שֵֽׁם־ יִשְׂרָאֵ֣ל עֽוֹד׃
5נוֹעֲצ֣וּ לֵ֣ב יַחְדָּ֑ו עָ֝לֶ֗יךָ בְּרִ֣ית יִכְרֹֽתוּ׃
6אֱ֭דוֹם וְיִשְׁמְעֵאלִ֗ים מוֹאָ֥ב וְהַגְרִֽים׃
7וְ֭עַמּוֹן וַעֲמָלֵ֑ק פְּ֝לֶ֗שֶׁת עִם־ יֹ֥שְׁבֵי צֽוֹר׃
8אַ֭שּׁוּר נִלְוָ֣ה עִמָּ֑ם הָ֤י֥וּ זְר֖וֹעַ לִבְנֵי־ ל֣וֹט סֶֽלָה׃
9לָהֶ֥ם כְּמִדְיָ֑ן כְּֽסִֽיסְרָ֥א כְ֝יָבִ֗ין בְּנַ֣חַל קִישֽׁוֹן׃
10בְֽעֵין־ דֹּ֑אר הָ֥יוּ דֹּ֝֗מֶן לָאֲדָמָֽה׃
11נְ֭דִיבֵמוֹ כְּעֹרֵ֣ב וְכִזְאֵ֑ב וּֽכְזֶ֥בַח וּ֝כְצַלְמֻנָּ֗ע כָּל־ נְסִיכֵֽמוֹ׃
12אָ֭מְרוּ נִ֣ירֲשָׁה לָּ֑נוּ אֵ֝֗ת נְא֣וֹת אֱלֹהִֽים׃
13שִׁיתֵ֥מוֹ כַגַּלְגַּ֑ל כְּ֝קַ֗שׁ לִפְנֵי־ רֽוּחַ׃
14תִּבְעַר־ יָ֑עַר וּ֝כְלֶהָבָ֗ה תְּלַהֵ֥ט הָרִֽים׃
15תִּרְדְּפֵ֣ם בְּסַעֲרֶ֑ךָ וּבְסוּפָתְךָ֥ תְבַהֲלֵֽם׃
16פְנֵיהֶ֣ם קָל֑וֹן וִֽיבַקְשׁ֖וּ שִׁמְךָ֣ יְהוָֽה׃
17וְיִבָּהֲל֥וּ עֲדֵי־ עַ֗ד וְֽיַחְפְּר֥וּ וְיֹאבֵֽדוּ׃
