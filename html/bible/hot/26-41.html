1וַיְבִיאֵ֖נִי אֶל־ הַהֵיכָ֑ל וַיָּ֣מָד אֶת־ הָאֵילִ֗ים שֵׁשׁ־ אַמּ֨וֹת רֹ֧חַב־ מִפּ֛וֹ וְשֵׁשׁ־ אַמּֽוֹת־ רֹ֥חַב מִפּ֖וֹ רֹ֥חַב הָאֹֽהֶל׃
2הַפֶּתַח֮ עֶ֣שֶׂר אַמּוֹת֒ וְכִתְפ֣וֹת הַפֶּ֔תַח חָמֵ֤שׁ אַמּוֹת֙ מִפּ֔וֹ וְחָמֵ֥שׁ אַמּ֖וֹת מִפּ֑וֹ וַיָּ֤מָד אָרְכּוֹ֙ אַרְבָּעִ֣ים אַמָּ֔ה וְרֹ֖חַב עֶשְׂרִ֥ים אַמָּֽה׃
3לִפְנִ֔ימָה וַיָּ֥מָד אֵֽיל־ הַפֶּ֖תַח שְׁתַּ֣יִם אַמּ֑וֹת וְהַפֶּ֙תַח֙ שֵׁ֣שׁ אַמּ֔וֹת וְרֹ֥חַב הַפֶּ֖תַח שֶׁ֥בַע אַמּֽוֹת׃
4אֶת־ אָרְכּ֜וֹ עֶשְׂרִ֣ים אַמָּ֗ה וְרֹ֛חַב עֶשְׂרִ֥ים אַמָּ֖ה אֶל־ פְּנֵ֣י הַֽהֵיכָ֑ל וַיֹּ֣אמֶר אֵלַ֔י זֶ֖ה קֹ֥דֶשׁ הַקֳּדָשִֽׁים׃
5קִֽיר־ הַבַּ֖יִת שֵׁ֣שׁ אַמּ֑וֹת וְרֹ֣חַב הַצֵּלָע֩ אַרְבַּ֨ע אַמּ֜וֹת סָבִ֧יב ׀ סָבִ֛יב לַבַּ֖יִת סָבִֽיב׃
6צֵלָ֨ע אֶל־ צֵלָ֜ע שָׁל֧וֹשׁ וּשְׁלֹשִׁ֣ים פְּעָמִ֗ים וּ֠בָאוֹת בַּקִּ֨יר אֲשֶׁר־ לַבַּ֧יִת לַצְּלָע֛וֹת סָבִ֥יב ׀ סָבִ֖יב לִהְי֣וֹת אֲחוּזִ֑ים וְלֹֽא־ יִהְי֥וּ אֲחוּזִ֖ים בְּקִ֥יר הַבָּֽיִת׃
7וְֽנָסְבָה֩ לְמַ֨עְלָה לְמַ֜עְלָה לַצְּלָע֗וֹת כִּ֣י מֽוּסַב־ הַ֠בַּיִת לְמַ֨עְלָה לְמַ֜עְלָה סָבִ֤יב ׀ סָבִיב֙ לַבַּ֔יִת עַל־ כֵּ֥ן רֹֽחַב־ לַבַּ֖יִת לְמָ֑עְלָה וְכֵ֧ן הַתַּחְתּוֹנָ֛ה יַעֲלֶ֥ה עַל־ הָעֶלְיוֹנָ֖ה לַתִּיכוֹנָֽה׃
8לַבַּ֛יִת גֹּ֖בַהּ סָבִ֣יב ׀ סָבִ֑יב [מְיֻסְּדֹות כ] (מוּסְדֹ֤ות ק) הַצְּלָעוֹת֙ מְל֣וֹ הַקָּנֶ֔ה שֵׁ֥שׁ אַמּ֖וֹת אַצִּֽילָה׃
9הַקִּ֧יר אֲ‍ֽשֶׁר־ לַצֵּלָ֛ע אֶל־ הַח֖וּץ חָמֵ֣שׁ אַמּ֑וֹת וַאֲשֶׁ֣ר מֻנָּ֔ח בֵּ֥ית צְלָע֖וֹת אֲשֶׁ֥ר לַבָּֽיִת׃
10הַלְּשָׁכ֜וֹת רֹ֣חַב עֶשְׂרִ֥ים אַמָּ֛ה סָבִ֥יב לַבַּ֖יִת סָבִ֥יב ׀ סָבִֽיב׃
11הַצֵּלָע֙ לַמֻּנָּ֔ח פֶּ֤תַח אֶחָד֙ דֶּ֣רֶךְ הַצָּפ֔וֹן וּפֶ֥תַח אֶחָ֖ד לַדָּר֑וֹם וְרֹ֙חַב֙ מְק֣וֹם הַמֻּנָּ֔ח חָמֵ֥שׁ אַמּ֖וֹת סָבִ֥יב ׀ סָבִֽיב׃
12אֲשֶׁר֩ אֶל־ פְּנֵ֨י הַגִּזְרָ֜ה פְּאַ֣ת דֶּֽרֶךְ־ הַיָּ֗ם רֹ֚חַב שִׁבְעִ֣ים אַמָּ֔ה וְקִ֧יר הַבִּנְיָ֛ן חָֽמֵשׁ־ אַמּ֥וֹת רֹ֖חַב סָבִ֣יב ׀ סָבִ֑יב וְאָרְכּ֖וֹ תִּשְׁעִ֥ים אַמָּֽה׃
13אֶת־ הַבַּ֔יִת אֹ֖רֶךְ מֵאָ֣ה אַמָּ֑ה וְהַגִּזְרָ֤ה וְהַבִּנְיָה֙ וְקִ֣ירוֹתֶ֔יהָ אֹ֖רֶךְ מֵאָ֥ה אַמָּֽה׃
14פְּנֵ֨י הַבַּ֧יִת וְהַגִּזְרָ֛ה לַקָּדִ֖ים מֵאָ֥ה אַמָּֽה׃
15אֹֽרֶךְ־ הַ֠בִּנְיָן אֶל־ פְּנֵ֨י הַגִּזְרָ֜ה אֲשֶׁ֨ר עַל־ אַחֲרֶ֧יהָ [וְאַתּוּקֵיהָא כ] (וְאַתִּיקֶ֛יהָא ק) מִפּ֥וֹ וּמִפּ֖וֹ מֵאָ֣ה אַמָּ֑ה וְהַֽהֵיכָל֙ הַפְּנִימִ֔י וְאֻֽלַמֵּ֖י הֶחָצֵֽר׃
16וְהַחַלּוֹנִ֣ים הָ֠אֲטֻמוֹת וְהָאַתִּיקִ֤ים ׀ סָבִיב֙ לִשְׁלָשְׁתָּ֔ם נֶ֧גֶד הַסַּ֛ף שְׂחִ֥יף עֵ֖ץ סָבִ֣יב ׀ סָבִ֑יב וְהָאָ֙רֶץ֙ עַד־ הַֽחַלֹּנ֔וֹת וְהַֽחַלֹּנ֖וֹת מְכֻסּֽוֹת׃
17מֵעַ֣ל הַפֶּ֡תַח וְעַד־ הַבַּיִת֩ הַפְּנִימִ֨י וְלַח֜וּץ וְאֶל־ כָּל־ הַקִּ֨יר סָבִ֧יב ׀ סָבִ֛יב בַּפְּנִימִ֥י וּבַחִיצ֖וֹן מִדּֽוֹת׃
18כְּרוּבִ֖ים וְתִֽמֹרִ֑ים וְתִֽמֹרָה֙ בֵּין־ כְּר֣וּב לִכְר֔וּב וּשְׁנַ֥יִם פָּנִ֖ים לַכְּרֽוּב׃
19אָדָ֤ם אֶל־ הַתִּֽמֹרָה֙ מִפּ֔וֹ וּפְנֵֽי־ כְפִ֥יר אֶל־ הַתִּֽמֹרָ֖ה מִפּ֑וֹ עָשׂ֥וּי אֶל־ כָּל־ הַבַּ֖יִת סָבִ֥יב ׀ סָבִֽיב׃
20עַד־ מֵעַ֣ל הַפֶּ֔תַח הַכְּרוּבִ֥ים וְהַתִּֽמֹרִ֖ים עֲשׂוּיִ֑ם וְקִ֖יר הַׅהֵׅיׅכָֽׅלׅ׃
21מְזוּזַ֣ת רְבֻעָ֑ה וּפְנֵ֣י הַקֹּ֔דֶשׁ הַמַּרְאֶ֖ה כַּמַּרְאֶֽה׃
22עֵ֣ץ שָׁלוֹשׁ֩ אַמּ֨וֹת גָּבֹ֜הַּ וְאָרְכּ֣וֹ שְׁתַּֽיִם־ אַמּ֗וֹת וּמִקְצֹֽעוֹתָיו֙ ל֔וֹ וְאָרְכּ֥וֹ וְקִֽירֹתָ֖יו עֵ֑ץ וַיְדַבֵּ֣ר אֵלַ֔י זֶ֚ה הַשֻּׁלְחָ֔ן אֲשֶׁ֖ר לִפְנֵ֥י יְהוָֽה׃
23דְּלָת֛וֹת לַֽהֵיכָ֖ל וְלַקֹּֽדֶשׁ׃
24דְּלָת֖וֹת לַדְּלָת֑וֹת שְׁ֚תַּיִם מוּסַבּ֣וֹת דְּלָת֔וֹת שְׁ֚תַּיִם לְדֶ֣לֶת אֶחָ֔ת וּשְׁתֵּ֥י דְלָת֖וֹת לָאַחֶֽרֶת׃
25אֲלֵיהֶ֜ן אֶל־ דַּלְת֤וֹת הַֽהֵיכָל֙ כְּרוּבִ֣ים וְתִֽמֹרִ֔ים כַּאֲשֶׁ֥ר עֲשׂוּיִ֖ם לַקִּיר֑וֹת וְעָ֥ב עֵ֛ץ אֶל־ פְּנֵ֥י הָאוּלָ֖ם מֵהַחֽוּץ׃
