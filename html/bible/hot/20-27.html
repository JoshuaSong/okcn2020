1אַֽל־ תִּ֭תְהַלֵּל בְּי֣וֹם מָחָ֑ר כִּ֤י לֹא־ תֵ֝דַ֗ע מַה־ יֵּ֥לֶד יֽוֹם׃
2זָ֣ר וְלֹא־ פִ֑יךָ נָ֝כְרִ֗י וְאַל־ שְׂפָתֶֽיךָ׃
3אֶ֭בֶן וְנֵ֣טֶל הַח֑וֹל וְכַ֥עַס אֱ֝וִ֗יל כָּבֵ֥ד מִשְּׁנֵיהֶֽם׃
4חֵ֭מָה וְשֶׁ֣טֶף אָ֑ף וּמִ֥י יַ֝עֲמֹד לִפְנֵ֥י קִנְאָֽה׃
5תּוֹכַ֣חַת מְגֻלָּ֑ה מֵֽאַהֲבָ֥ה מְסֻתָּֽרֶת׃
6פִּצְעֵ֣י אוֹהֵ֑ב וְ֝נַעְתָּר֗וֹת נְשִׁיק֥וֹת שׂוֹנֵֽא׃
7שְׂ֭בֵעָה תָּב֣וּס נֹ֑פֶת וְנֶ֥פֶשׁ רְ֝עֵבָ֗ה כָּל־ מַ֥ר מָתֽוֹק׃
8נוֹדֶ֣דֶת מִן־ קִנָּ֑הּ כֵּֽן־ אִ֝֗ישׁ נוֹדֵ֥ד מִמְּקוֹמֽוֹ׃
9וּ֭קְטֹרֶת יְשַׂמַּֽח־ לֵ֑ב וּמֶ֥תֶק רֵ֝עֵ֗הוּ מֵֽעֲצַת־ נָֽפֶשׁ׃
10[וְרֵעֶה כ] (וְרֵ֪עַ ק) אָבִ֡יךָ אַֽל־ תַּעֲזֹ֗ב וּבֵ֥ית אָחִ֗יךָ אַל־ תָּ֭בוֹא בְּי֣וֹם אֵידֶ֑ךָ ט֥וֹב שָׁכֵ֥ן קָ֝ר֗וֹב מֵאָ֥ח רָחֽוֹק׃
11בְּ֭נִי וְשַׂמַּ֣ח לִבִּ֑י וְאָשִׁ֖יבָה חֹרְפִ֣י דָבָֽר׃
12רָאָ֣ה רָעָ֣ה נִסְתָּ֑ר פְּ֝תָאיִ֗ם עָבְר֥וּ נֶעֱנָֽשׁוּ׃
13בִּ֭גְדוֹ כִּי־ עָ֣רַב זָ֑ר וּבְעַ֖ד נָכְרִיָּ֣ה חַבְלֵֽהוּ׃
14רֵעֵ֨הוּ ׀ בְּק֣וֹל גָּ֭דוֹל בַּבֹּ֣קֶר הַשְׁכֵּ֑ים קְ֝לָלָ֗ה תֵּחָ֥שֶׁב לֽוֹ׃
15ט֭וֹרֵד בְּי֣וֹם סַגְרִ֑יר וְאֵ֥שֶׁת [מִדֹונִים כ] (מִ֝דְיָנִ֗ים ק) נִשְׁתָּוָֽה׃
16צָֽפַן־ ר֑וּחַ וְשֶׁ֖מֶן יְמִינ֣וֹ יִקְרָֽא׃
17בְּבַרְזֶ֣ל יָ֑חַד וְ֝אִ֗ישׁ יַ֣חַד פְּנֵֽי־ רֵעֵֽהוּ׃
18תְּ֭אֵנָה יֹאכַ֣ל פִּרְיָ֑הּ וְשֹׁמֵ֖ר אֲדֹנָ֣יו יְכֻבָּֽד׃
19הַפָּנִ֣ים לַפָּנִ֑ים כֵּ֤ן לֵֽב־ הָ֝אָדָ֗ם לָאָדָֽם׃
20[וַאֲבַדֹּה כ] (וַ֭אֲבַדֹּו ק) לֹ֣א תִשְׂבַּ֑עְנָה וְעֵינֵ֥י הָ֝אָדָ֗ם לֹ֣א תִשְׂבַּֽעְנָה׃
21לַ֭כֶּסֶף וְכ֣וּר לַזָּהָ֑ב וְ֝אִ֗ישׁ לְפִ֣י מַהֲלָלֽוֹ׃
22תִּכְתּֽוֹשׁ־ אֶת־ הָאֱוִ֨יל ׀ בַּֽמַּכְתֵּ֡שׁ בְּת֣וֹךְ הָ֭רִיפוֹת בַּֽעֱלִ֑י לֹא־ תָס֥וּר מֵ֝עָלָ֗יו אִוַּלְתּֽוֹ׃ פ
23תֵּ֭דַע פְּנֵ֣י צֹאנֶ֑ךָ שִׁ֥ית לִ֝בְּךָ֗ לַעֲדָרִֽים׃
24לֹ֣א לְעוֹלָ֣ם חֹ֑סֶן וְאִם־ נֵ֝֗זֶר לְד֣וֹר [דֹור כ] (וָדֹֽור׃ ק)
25חָ֭צִיר וְנִרְאָה־ דֶ֑שֶׁא וְ֝נֶאֶסְפ֗וּ עִשְּׂב֥וֹת הָרִֽים׃
26לִלְבוּשֶׁ֑ךָ וּמְחִ֥יר שָׂ֝דֶ֗ה עַתּוּדִֽים׃
