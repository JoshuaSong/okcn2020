1אָ֭הַבְתִּי כִּֽי־ יִשְׁמַ֥ע ׀ יְהוָ֑ה אֶת־ ק֝וֹלִ֗י תַּחֲנוּנָֽי׃
2הִטָּ֣ה אָזְנ֣וֹ לִ֑י וּבְיָמַ֥י אֶקְרָֽא׃
3חֶבְלֵי־ מָ֗וֶת וּמְצָרֵ֣י שְׁא֣וֹל מְצָא֑וּנִי צָרָ֖ה וְיָג֣וֹן אֶמְצָֽא׃
4יְהוָ֥ה אֶקְרָ֑א אָנָּ֥ה יְ֝הוָ֗ה מַלְּטָ֥ה נַפְשִֽׁי׃
5יְהֹוָ֣ה וְצַדִּ֑יק וֵ֖אלֹהֵ֣ינוּ מְרַחֵֽם׃
6פְּתָאיִ֣ם יְהֹוָ֑ה דַּ֝לּוֹתִ֗י וְלִ֣י יְהוֹשִֽׁיעַ׃
7נַ֭פְשִׁי לִמְנוּחָ֑יְכִי כִּֽי־ יְ֝הוָ֗ה גָּמַ֥ל עָלָֽיְכִי׃
8חִלַּ֥צְתָּ נַפְשִׁ֗י מִ֫מָּ֥וֶת אֶת־ עֵינִ֥י מִן־ דִּמְעָ֑ה אֶת־ רַגְלִ֥י מִדֶּֽחִי׃
9לִפְנֵ֣י יְהוָ֑ה בְּ֝אַרְצ֗וֹת הַֽחַיִּֽים׃
10כִּ֣י אֲדַבֵּ֑ר אֲ֝נִ֗י עָנִ֥יתִי מְאֹֽד׃
11אָמַ֣רְתִּי בְחָפְזִ֑י כָּֽל־ הָאָדָ֥ם כֹּזֵֽב׃
12אָשִׁ֥יב לַיהוָ֑ה כָּֽל־ תַּגְמוּל֥וֹהִי עָלָֽי׃
13יְשׁוּע֥וֹת אֶשָּׂ֑א וּבְשֵׁ֖ם יְהוָ֣ה אֶקְרָֽא׃
14לַיהוָ֣ה אֲשַׁלֵּ֑ם נֶגְדָה־ נָּ֝֗א לְכָל־ עַמּֽוֹ׃
15בְּעֵינֵ֣י יְהוָ֑ה הַ֝מָּ֗וְתָה לַחֲסִידָֽיו׃
16יְהוָה֮ כִּֽי־ אֲנִ֪י עַ֫בְדֶּ֥ךָ אֲ‍ֽנִי־ עַ֭בְדְּךָ בֶּן־ אֲמָתֶ֑ךָ פִּ֝תַּ֗חְתָּ לְמוֹסֵרָֽי׃
17אֶ֭זְבַּח זֶ֣בַח תּוֹדָ֑ה וּבְשֵׁ֖ם יְהוָ֣ה אֶקְרָֽא׃
18לַיהוָ֣ה אֲשַׁלֵּ֑ם נֶגְדָה־ נָּ֝֗א לְכָל־ עַמּֽוֹ׃
