1כִּי־ יִמָּצֵ֣א חָלָ֗ל בָּאֲדָמָה֙ אֲשֶׁר֩ יְהוָ֨ה אֱלֹהֶ֜יךָ נֹתֵ֤ן לְךָ֙ לְרִשְׁתָּ֔הּ נֹפֵ֖ל בַּשָּׂדֶ֑ה לֹ֥א נוֹדַ֖ע מִ֥י הִכָּֽהוּ׃
2זְקֵנֶ֖יךָ וְשֹׁפְטֶ֑יךָ וּמָדְדוּ֙ אֶל־ הֶ֣עָרִ֔ים אֲשֶׁ֖ר סְבִיבֹ֥ת הֶחָלָֽל׃
3הָעִ֔יר הַקְּרֹבָ֖ה אֶל־ הֶחָלָ֑ל וְלָֽקְח֡וּ זִקְנֵי֩ הָעִ֨יר הַהִ֜וא עֶגְלַ֣ת בָּקָ֗ר אֲשֶׁ֤ר לֹֽא־ עֻבַּד֙ בָּ֔הּ אֲשֶׁ֥ר לֹא־ מָשְׁכָ֖ה בְּעֹֽל׃
4זִקְנֵי֩ הָעִ֨יר הַהִ֤וא אֶת־ הָֽעֶגְלָה֙ אֶל־ נַ֣חַל אֵיתָ֔ן אֲשֶׁ֛ר לֹא־ יֵעָבֵ֥ד בּ֖וֹ וְלֹ֣א יִזָּרֵ֑עַ וְעָֽרְפוּ־ שָׁ֥ם אֶת־ הָעֶגְלָ֖ה בַּנָּֽחַל׃
5הַכֹּהֲנִים֮ בְּנֵ֣י לֵוִי֒ כִּ֣י בָ֗ם בָּחַ֞ר יְהוָ֤ה אֱלֹהֶ֙יךָ֙ לְשָׁ֣רְת֔וֹ וּלְבָרֵ֖ךְ בְּשֵׁ֣ם יְהוָ֑ה וְעַל־ פִּיהֶ֥ם יִהְיֶ֖ה כָּל־ רִ֥יב וְכָל־ נָֽגַע׃
6זִקְנֵי֙ הָעִ֣יר הַהִ֔וא הַקְּרֹבִ֖ים אֶל־ הֶחָלָ֑ל יִרְחֲצוּ֙ אֶת־ יְדֵיהֶ֔ם עַל־ הָעֶגְלָ֖ה הָעֲרוּפָ֥ה בַנָּֽחַל׃
7וְאָמְר֑וּ יָדֵ֗ינוּ לֹ֤א [שָׁפְכָה כ] (שָֽׁפְכוּ֙ ק) אֶת־ הַדָּ֣ם הַזֶּ֔ה וְעֵינֵ֖ינוּ לֹ֥א רָאֽוּ׃
8לְעַמְּךָ֨ יִשְׂרָאֵ֤ל אֲשֶׁר־ פָּדִ֙יתָ֙ יְהוָ֔ה וְאַל־ תִּתֵּן֙ דָּ֣ם נָקִ֔י בְּקֶ֖רֶב עַמְּךָ֣ יִשְׂרָאֵ֑ל וְנִכַּפֵּ֥ר לָהֶ֖ם הַדָּֽם׃
9תְּבַעֵ֛ר הַדָּ֥ם הַנָּקִ֖י מִקִּרְבֶּ֑ךָ כִּֽי־ תַעֲשֶׂ֥ה הַיָּשָׁ֖ר בְּעֵינֵ֥י יְהוָֽה׃ ס
10תֵצֵ֥א לַמִּלְחָמָ֖ה עַל־ אֹיְבֶ֑יךָ וּנְתָנ֞וֹ יְהוָ֧ה אֱלֹהֶ֛יךָ בְּיָדֶ֖ךָ וְשָׁבִ֥יתָ שִׁבְיֽוֹ׃
11בַּשִּׁבְיָ֔ה אֵ֖שֶׁת יְפַת־ תֹּ֑אַר וְחָשַׁקְתָּ֣ בָ֔הּ וְלָקַחְתָּ֥ לְךָ֖ לְאִשָּֽׁה׃
12אֶל־ תּ֣וֹךְ בֵּיתֶ֑ךָ וְגִלְּחָה֙ אֶת־ רֹאשָׁ֔הּ וְעָשְׂתָ֖ה אֶת־ צִפָּרְנֶֽיהָ׃
13אֶת־ שִׂמְלַ֨ת שִׁבְיָ֜הּ מֵעָלֶ֗יהָ וְיָֽשְׁבָה֙ בְּבֵיתֶ֔ךָ וּבָֽכְתָ֛ה אֶת־ אָבִ֥יהָ וְאֶת־ אִמָּ֖הּ יֶ֣רַח יָמִ֑ים וְאַ֨חַר כֵּ֜ן תָּב֤וֹא אֵלֶ֙יהָ֙ וּבְעַלְתָּ֔הּ וְהָיְתָ֥ה לְךָ֖ לְאִשָּֽׁה׃
14אִם־ לֹ֧א חָפַ֣צְתָּ בָּ֗הּ וְשִׁלַּחְתָּהּ֙ לְנַפְשָׁ֔הּ וּמָכֹ֥ר לֹא־ תִמְכְּרֶ֖נָּה בַּכָּ֑סֶף לֹא־ תִתְעַמֵּ֣ר בָּ֔הּ תַּ֖חַת אֲשֶׁ֥ר עִנִּיתָֽהּ׃ ס
15תִהְיֶ֨יןָ לְאִ֜ישׁ שְׁתֵּ֣י נָשִׁ֗ים הָאַחַ֤ת אֲהוּבָה֙ וְהָאַחַ֣ת שְׂנוּאָ֔ה וְיָֽלְדוּ־ ל֣וֹ בָנִ֔ים הָאֲהוּבָ֖ה וְהַשְּׂנוּאָ֑ה וְהָיָ֛ה הַבֵּ֥ן הַבְּכ֖וֹר לַשְּׂנִיאָֽה׃
16בְּיוֹם֙ הַנְחִיל֣וֹ אֶת־ בָּנָ֔יו אֵ֥ת אֲשֶׁר־ יִהְיֶ֖ה ל֑וֹ לֹ֣א יוּכַ֗ל לְבַכֵּר֙ אֶת־ בֶּן־ הָ֣אֲהוּבָ֔ה עַל־ פְּנֵ֥י בֶן־ הַשְּׂנוּאָ֖ה הַבְּכֹֽר׃
17אֶת־ הַבְּכֹ֨ר בֶּן־ הַשְּׂנוּאָ֜ה יַכִּ֗יר לָ֤תֶת לוֹ֙ פִּ֣י שְׁנַ֔יִם בְּכֹ֥ל אֲשֶׁר־ יִמָּצֵ֖א ל֑וֹ כִּי־ הוּא֙ רֵאשִׁ֣ית אֹנ֔וֹ ל֖וֹ מִשְׁפַּ֥ט הַבְּכֹרָֽה׃ ס
18יִהְיֶ֣ה לְאִ֗ישׁ בֵּ֚ן סוֹרֵ֣ר וּמוֹרֶ֔ה אֵינֶ֣נּוּ שֹׁמֵ֔עַ בְּק֥וֹל אָבִ֖יו וּבְק֣וֹל אִמּ֑וֹ וְיִסְּר֣וּ אֹת֔וֹ וְלֹ֥א יִשְׁמַ֖ע אֲלֵיהֶֽם׃
19ב֖וֹ אָבִ֣יו וְאִמּ֑וֹ וְהוֹצִ֧יאוּ אֹת֛וֹ אֶל־ זִקְנֵ֥י עִיר֖וֹ וְאֶל־ שַׁ֥עַר מְקֹמֽוֹ׃
20אֶל־ זִקְנֵ֣י עִיר֗וֹ בְּנֵ֤נוּ זֶה֙ סוֹרֵ֣ר וּמֹרֶ֔ה אֵינֶ֥נּוּ שֹׁמֵ֖עַ בְּקֹלֵ֑נוּ זוֹלֵ֖ל וְסֹבֵֽא׃
21כָּל־ אַנְשֵׁ֨י עִיר֤וֹ בָֽאֲבָנִים֙ וָמֵ֔ת וּבִֽעַרְתָּ֥ הָרָ֖ע מִקִּרְבֶּ֑ךָ וְכָל־ יִשְׂרָאֵ֖ל יִשְׁמְע֥וּ וְיִרָֽאוּ׃ ס
22יִהְיֶ֣ה בְאִ֗ישׁ חֵ֛טְא מִשְׁפַּט־ מָ֖וֶת וְהוּמָ֑ת וְתָלִ֥יתָ אֹת֖וֹ עַל־ עֵֽץ׃
