1וַ֭יַּעַן צֹפַ֥ר הַנַּֽעֲמָתִ֗י וַיֹּאמַֽר׃
2שְׂעִפַּ֣י יְשִׁיב֑וּנִי וּ֝בַעֲב֗וּר ח֣וּשִׁי בִֽי׃
3כְּלִמָּתִ֣י אֶשְׁמָ֑ע וְ֝ר֗וּחַ מִֽבִּינָתִ֥י יַעֲנֵֽנִי׃
4יָ֭דַעְתָּ מִנִּי־ עַ֑ד מִנִּ֤י שִׂ֖ים אָדָ֣ם עֲלֵי־ אָֽרֶץ׃
5רִנְנַ֣ת רְ֭שָׁעִים מִקָּר֑וֹב וְשִׂמְחַ֖ת חָנֵ֣ף עֲדֵי־ רָֽגַע׃
6יַעֲלֶ֣ה לַשָּׁמַ֣יִם שִׂיא֑וֹ וְ֝רֹאשׁ֗וֹ לָעָ֥ב יַגִּֽיעַ׃
7לָנֶ֣צַח יֹאבֵ֑ד רֹ֝אָ֗יו יֹאמְר֥וּ אַיּֽוֹ׃
8יָ֭עוּף וְלֹ֣א יִמְצָא֑וּהוּ וְ֝יֻדַּ֗ד כְּחֶזְי֥וֹן לָֽיְלָה׃
9שְׁ֭זָפַתּוּ וְלֹ֣א תוֹסִ֑יף וְלֹא־ ע֝֗וֹד תְּשׁוּרֶ֥נּוּ מְקוֹמֽוֹ׃
10יְרַצּ֣וּ דַלִּ֑ים וְ֝יָדָ֗יו תָּשֵׁ֥בְנָה אוֹנֽוֹ׃
11מָלְא֣וּ [עֲלוּמֹו כ] (עֲלוּמָ֑יו ק) וְ֝עִמּ֗וֹ עַל־ עָפָ֥ר תִּשְׁכָּֽב׃
12תַּמְתִּ֣יק בְּפִ֣יו רָעָ֑ה יַ֝כְחִידֶ֗נָּה תַּ֣חַת לְשׁוֹנֽוֹ׃
13עָ֭לֶיהָ וְלֹ֣א יַֽעַזְבֶ֑נָּה וְ֝יִמְנָעֶ֗נָּה בְּת֣וֹךְ חִכּֽוֹ׃
14בְּמֵעָ֣יו נֶהְפָּ֑ךְ מְרוֹרַ֖ת פְּתָנִ֣ים בְּקִרְבּֽוֹ׃
15בָּ֭לַע וַיְקִאֶ֑נּוּ מִ֝בִּטְנ֗וֹ יֽוֹרִשֶׁ֥נּוּ אֵֽל׃
16פְּתָנִ֥ים יִינָ֑ק תַּֽ֝הַרְגֵ֗הוּ לְשׁ֣וֹן אֶפְעֶֽה׃
17יֵ֥רֶא בִפְלַגּ֑וֹת נַהֲרֵ֥י נַ֝חֲלֵ֗י דְּבַ֣שׁ וְחֶמְאָֽה׃
18יָ֭גָע וְלֹ֣א יִבְלָ֑ע כְּחֵ֥יל תְּ֝מוּרָת֗וֹ וְלֹ֣א יַעֲלֹֽס׃
19רִ֭צַּץ עָזַ֣ב דַּלִּ֑ים בַּ֥יִת גָּ֝זַ֗ל וְלֹ֣א יִבֶנֵֽהוּ׃
20לֹא־ יָדַ֣ע שָׁלֵ֣ו בְּבִטְנ֑וֹ בַּ֝חֲמוּד֗וֹ לֹ֣א יְמַלֵּֽט׃
21שָׂרִ֥יד לְאָכְל֑וֹ עַל־ כֵּ֝֗ן לֹא־ יָחִ֥יל טוּבֽוֹ׃
22שִׂ֭פְקוֹ יֵ֣צֶר ל֑וֹ כָּל־ יַ֖ד עָמֵ֣ל תְּבוֹאֶֽנּוּ׃
23לְמַלֵּ֬א בִטְנ֗וֹ יְֽשַׁלַּח־ בּ֭וֹ חֲר֣וֹן אַפּ֑וֹ וְיַמְטֵ֥ר עָ֝לֵ֗ימוֹ בִּלְחוּמֽוֹ׃
24מִנֵּ֣שֶׁק בַּרְזֶ֑ל תַּ֝חְלְפֵ֗הוּ קֶ֣שֶׁת נְחוּשָֽׁה׃
25וַיֵּצֵ֪א מִגֵּ֫וָ֥ה וּ֭בָרָק מִֽמְּרֹרָת֥וֹ יַהֲלֹ֗ךְ עָלָ֥יו אֵמִֽים׃
26חֹשֶׁךְ֮ טָמ֪וּן לִצְפּ֫וּנָ֥יו תְּ֭אָכְלֵהוּ אֵ֣שׁ לֹֽא־ נֻפָּ֑ח יֵ֖רַע שָׂרִ֣יד בְּאָהֳלֽוֹ׃
27שָׁמַ֣יִם עֲוֹנ֑וֹ וְ֝אֶ֗רֶץ מִתְקוֹמָ֘מָ֥ה לֽוֹ׃
28יְב֣וּל בֵּית֑וֹ נִ֝גָּר֗וֹת בְּי֣וֹם אַפּֽוֹ׃
