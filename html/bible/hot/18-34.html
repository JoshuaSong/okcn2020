1וַיַּ֥עַן אֱלִיה֗וּא וַיֹּאמַֽר׃
2חֲכָמִ֣ים מִלָּ֑י וְ֝יֹדְעִ֗ים הַאֲזִ֥ינוּ לִֽי׃
3אֹ֭זֶן מִלִּ֣ין תִּבְחָ֑ן וְ֝חֵ֗ךְ יִטְעַ֥ם לֶאֱכֹֽל׃
4נִבְחֲרָה־ לָּ֑נוּ נֵדְעָ֖ה בֵינֵ֣ינוּ מַה־ טּֽוֹב׃
5אָ֭מַר אִיּ֣וֹב צָדַ֑קְתִּי וְ֝אֵ֗ל הֵסִ֥יר מִשְׁפָּטִֽי׃
6מִשְׁפָּטִ֥י אֲכַזֵּ֑ב אָנ֖וּשׁ חִצִּ֣י בְלִי־ פָֽשַׁע׃
7גֶ֥בֶר כְּאִיּ֑וֹב יִֽשְׁתֶּה־ לַּ֥עַג כַּמָּֽיִם׃
8לְ֭חֶבְרָה עִם־ פֹּ֣עֲלֵי אָ֑וֶן וְ֝לָלֶ֗כֶת עִם־ אַנְשֵׁי־ רֶֽשַׁע׃
9אָ֭מַר לֹ֣א יִסְכָּן־ גָּ֑בֶר בִּ֝רְצֹת֗וֹ עִם־ אֱלֹהִֽים׃
10אַ֥נֲשֵׁ֥י לֵבָ֗ב שִׁמְע֫וּ לִ֥י חָלִ֖לָה לָאֵ֥ל מֵרֶ֗שַׁע וְשַׁדַּ֥י מֵעָֽוֶל׃
11פֹ֣עַל אָ֭דָם יְשַׁלֶּם־ ל֑וֹ וּֽכְאֹ֥רַח אִ֝֗ישׁ יַמְצִאֶֽנּוּ׃
12אָמְנָ֗ם אֵ֥ל לֹֽא־ יַרְשִׁ֑יעַ וְ֝שַׁדַּ֗י לֹֽא־ יְעַוֵּ֥ת מִשְׁפָּֽט׃
13פָקַ֣ד עָלָ֣יו אָ֑רְצָה וּמִ֥י שָׂ֝֗ם תֵּבֵ֥ל כֻּלָּֽהּ׃
14יָשִׂ֣ים אֵלָ֣יו לִבּ֑וֹ רוּח֥וֹ וְ֝נִשְׁמָת֗וֹ אֵלָ֥יו יֶאֱסֹֽף׃
15כָּל־ בָּשָׂ֣ר יָ֑חַד וְ֝אָדָ֗ם עַל־ עָפָ֥ר יָשֽׁוּב׃
16בִּ֥ינָה שִׁמְעָה־ זֹּ֑את הַ֝אֲזִ֗ינָה לְק֣וֹל מִלָּֽי׃
17שׂוֹנֵ֣א מִשְׁפָּ֣ט יַחֲב֑וֹשׁ וְאִם־ צַדִּ֖יק כַּבִּ֣יר תַּרְשִֽׁיעַ׃
18לְמֶ֣לֶךְ בְּלִיָּ֑עַל רָ֝שָׁ֗ע אֶל־ נְדִיבִֽים׃
19לֹֽא־ נָשָׂ֨א ׀ פְּנֵ֥י שָׂרִ֗ים וְלֹ֣א נִכַּר־ שׁ֭וֹעַ לִפְנֵי־ דָ֑ל כִּֽי־ מַעֲשֵׂ֖ה יָדָ֣יו כֻּלָּֽם׃
20יָמֻתוּ֮ וַחֲצ֪וֹת לָ֥יְלָה יְגֹעֲשׁ֣וּ עָ֣ם וְיַעֲבֹ֑רוּ וְיָסִ֥ירוּ אַ֝בִּ֗יר לֹ֣א בְיָֽד׃
21עֵ֭ינָיו עַל־ דַּרְכֵי־ אִ֑ישׁ וְֽכָל־ צְעָדָ֥יו יִרְאֶֽה׃
22חֹ֭שֶׁךְ וְאֵ֣ין צַלְמָ֑וֶת לְהִסָּ֥תֶר שָׁ֝֗ם פֹּ֣עֲלֵי אָֽוֶן׃
23לֹ֣א עַל־ אִ֭ישׁ יָשִׂ֣ים ע֑וֹד לַהֲלֹ֥ךְ אֶל־ אֵ֝֗ל בַּמִּשְׁפָּֽט׃
24כַּבִּירִ֣ים לֹא־ חֵ֑קֶר וַיַּעֲמֵ֖ד אֲחֵרִ֣ים תַּחְתָּֽם׃
25יַ֭כִּיר מַעְבָּֽדֵיהֶ֑ם וְהָ֥פַךְ לַ֝֗יְלָה וְיִדַּכָּֽאוּ׃
26רְשָׁעִ֥ים סְפָקָ֗ם בִּמְק֥וֹם רֹאִֽים׃
27עַל־ כֵּ֭ן סָ֣רוּ מֵֽאַחֲרָ֑יו וְכָל־ דְּ֝רָכָ֗יו לֹ֣א הִשְׂכִּֽילוּ׃
28עָ֭לָיו צַֽעֲקַת־ דָּ֑ל וְצַעֲקַ֖ת עֲנִיִּ֣ים יִשְׁמָֽע׃
29יַשְׁקִ֨ט ׀ וּמִ֥י יַרְשִׁ֗עַ וְיַסְתֵּ֣ר פָּ֭נִים וּמִ֣י יְשׁוּרֶ֑נּוּ וְעַל־ גּ֖וֹי וְעַל־ אָדָ֣ם יָֽחַד׃
30אָדָ֥ם חָנֵ֗ף מִמֹּ֥קְשֵׁי עָֽם׃ כִּֽי־ אֶל־
31הֶאָמַ֥ר נָשָׂ֗אתִי לֹ֣א אֶחְבֹּֽל׃
32אֶ֭חֱזֶה אַתָּ֣ה הֹרֵ֑נִי אִֽם־ עָ֥וֶל פָּ֝עַ֗לְתִּי לֹ֣א אֹסִֽיף׃
33יְשַׁלְמֶ֨נָּה כִּֽי־ מָאַ֗סְתָּ כִּי־ אַתָּ֣ה תִבְחַ֣ר וְלֹא־ אָ֑נִי וּֽמַה־ יָדַ֥עְתָּ דַבֵּֽר׃
34לֵ֭בָב יֹ֣אמְרוּ לִ֑י וְגֶ֥בֶר חָ֝כָ֗ם שֹׁמֵ֥עַֽ לִֽי׃
35לֹא־ בְדַ֣עַת יְדַבֵּ֑ר וּ֝דְבָרָ֗יו לֹ֣א בְהַשְׂכֵּֽיל׃
36יִבָּחֵ֣ן אִיּ֣וֹב עַד־ נֶ֑צַח עַל־ תְּ֝שֻׁבֹ֗ת בְּאַנְשֵׁי־ אָֽוֶן׃
