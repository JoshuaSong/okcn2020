1בְּיָמָ֣יו עָלָ֔ה נְבֻכַדְנֶאצַּ֖ר מֶ֣לֶךְ בָּבֶ֑ל וַיְהִי־ ל֨וֹ יְהוֹיָקִ֥ים עֶ֙בֶד֙ שָׁלֹ֣שׁ שָׁנִ֔ים וַיָּ֖שָׁב וַיִּמְרָד־ בּֽוֹ׃
2יְהוָ֣ה ׀ בּ֡וֹ אֶת־ גְּדוּדֵ֣י כַשְׂדִּים֩ וְאֶת־ גְּדוּדֵ֨י אֲרָ֜ם וְאֵ֣ת ׀ גְּדוּדֵ֣י מוֹאָ֗ב וְאֵת֙ גְּדוּדֵ֣י בְנֵֽי־ עַמּ֔וֹן וַיְשַׁלְּחֵ֥ם בִּֽיהוּדָ֖ה לְהַֽאֲבִיד֑וֹ כִּדְבַ֣ר יְהוָ֔ה אֲשֶׁ֣ר דִּבֶּ֔ר בְּיַ֖ד עֲבָדָ֥יו הַנְּבִיאִֽים׃
3עַל־ פִּ֣י יְהוָ֗ה הָֽיְתָה֙ בִּֽיהוּדָ֔ה לְהָסִ֖יר מֵעַ֣ל פָּנָ֑יו בְּחַטֹּ֣את מְנַשֶּׁ֔ה כְּכֹ֖ל אֲשֶׁ֥ר עָשָֽׂה׃
4דַּֽם־ הַנָּקִי֙ אֲשֶׁ֣ר שָׁפָ֔ךְ וַיְמַלֵּ֥א אֶת־ יְרוּשָׁלִַ֖ם דָּ֣ם נָקִ֑י וְלֹֽא־ אָבָ֥ה יְהוָ֖ה לִסְלֹֽחַ׃
5דִּבְרֵ֥י יְהוֹיָקִ֖ים וְכָל־ אֲשֶׁ֣ר עָשָׂ֑ה הֲלֹא־ הֵ֣ם כְּתוּבִ֗ים עַל־ סֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לְמַלְכֵ֥י יְהוּדָֽה׃
6יְהוֹיָקִ֖ים עִם־ אֲבֹתָ֑יו וַיִּמְלֹ֛ךְ יְהוֹיָכִ֥ין בְּנ֖וֹ תַּחְתָּֽיו׃
7הֹסִ֥יף עוֹד֙ מֶ֣לֶךְ מִצְרַ֔יִם לָצֵ֖את מֵֽאַרְצ֑וֹ כִּֽי־ לָקַ֞ח מֶ֣לֶךְ בָּבֶ֗ל מִנַּ֤חַל מִצְרַ֙יִם֙ עַד־ נְהַר־ פְּרָ֔ת כֹּ֛ל אֲשֶׁ֥ר הָיְתָ֖ה לְמֶ֥לֶךְ מִצְרָֽיִם׃ פ
8שְׁמֹנֶ֨ה עֶשְׂרֵ֤ה שָׁנָה֙ יְהוֹיָכִ֣ין בְּמָלְכ֔וֹ וּשְׁלֹשָׁ֣ה חֳדָשִׁ֔ים מָלַ֖ךְ בִּירוּשָׁלִָ֑ם וְשֵׁ֣ם אִמּ֔וֹ נְחֻשְׁתָּ֥א בַת־ אֶלְנָתָ֖ן מִירוּשָׁלִָֽם׃
9הָרַ֖ע בְּעֵינֵ֣י יְהוָ֑ה כְּכֹ֥ל אֲשֶׁר־ עָשָׂ֖ה אָבִֽיו׃
10הַהִ֔יא [עָלָה כ] (עָל֗וּ ק) עַבְדֵ֛י נְבֻכַדְנֶאצַּ֥ר מֶֽלֶךְ־ בָּבֶ֖ל יְרוּשָׁלִָ֑ם וַתָּבֹ֥א הָעִ֖יר בַּמָּצֽוֹר׃
11נְבוּכַדְנֶאצַּ֥ר מֶֽלֶךְ־ בָּבֶ֖ל עַל־ הָעִ֑יר וַעֲבָדָ֖יו צָרִ֥ים עָלֶֽיהָ׃
12יְהוֹיָכִ֤ין מֶֽלֶךְ־ יְהוּדָה֙ עַל־ מֶ֣לֶךְ בָּבֶ֔ל ה֣וּא וְאִמּ֔וֹ וַעֲבָדָ֖יו וְשָׂרָ֣יו וְסָֽרִיסָ֑יו וַיִּקַּ֤ח אֹתוֹ֙ מֶ֣לֶךְ בָּבֶ֔ל בִּשְׁנַ֥ת שְׁמֹנֶ֖ה לְמָלְכֽוֹ׃
13מִשָּׁ֗ם אֶת־ כָּל־ אוֹצְרוֹת֙ בֵּ֣ית יְהוָ֔ה וְאֽוֹצְר֖וֹת בֵּ֣ית הַמֶּ֑לֶךְ וַיְקַצֵּ֞ץ אֶת־ כָּל־ כְּלֵ֣י הַזָּהָ֗ב אֲשֶׁ֨ר עָשָׂ֜ה שְׁלֹמֹ֤ה מֶֽלֶךְ־ יִשְׂרָאֵל֙ בְּהֵיכַ֣ל יְהוָ֔ה כַּֽאֲשֶׁ֖ר דִּבֶּ֥ר יְהוָֽה׃
14אֶת־ כָּל־ יְ֠רוּשָׁלִַם וְֽאֶת־ כָּל־ הַשָּׂרִ֞ים וְאֵ֣ת ׀ כָּל־ גִּבּוֹרֵ֣י הַחַ֗יִל [עֲשָׂרָה כ] (עֲשֶׂ֤רֶת ק) אֲלָפִים֙ גּוֹלֶ֔ה וְכָל־ הֶחָרָ֖שׁ וְהַמַּסְגֵּ֑ר לֹ֣א נִשְׁאַ֔ר זוּלַ֖ת דַּלַּ֥ת עַם־ הָאָֽרֶץ׃
15אֶת־ יְהוֹיָכִ֖ין בָּבֶ֑לָה וְאֶת־ אֵ֣ם הַ֠מֶּלֶךְ וְאֶת־ נְשֵׁ֨י הַמֶּ֜לֶךְ וְאֶת־ סָרִיסָ֗יו וְאֵת֙ [אֱוִלֵי כ] (אֵילֵ֣י ק) הָאָ֔רֶץ הוֹלִ֛יךְ גּוֹלָ֥ה מִירוּשָׁלִַ֖ם בָּבֶֽלָה׃
16כָּל־ אַנְשֵׁ֨י הַחַ֜יִל שִׁבְעַ֣ת אֲלָפִ֗ים וְהֶחָרָ֤שׁ וְהַמַּסְגֵּר֙ אֶ֔לֶף הַכֹּ֕ל גִּבּוֹרִ֖ים עֹשֵׂ֣י מִלְחָמָ֑ה וַיְבִיאֵ֧ם מֶֽלֶךְ־ בָּבֶ֛ל גּוֹלָ֖ה בָּבֶֽלָה׃
17מֶֽלֶךְ־ בָּבֶ֛ל אֶת־ מַתַּנְיָ֥ה דֹד֖וֹ תַּחְתָּ֑יו וַיַּסֵּ֥ב אֶת־ שְׁמ֖וֹ צִדְקִיָּֽהוּ׃ פ
18עֶשְׂרִ֨ים וְאַחַ֤ת שָׁנָה֙ צִדְקִיָּ֣הוּ בְמָלְכ֔וֹ וְאַחַ֤ת עֶשְׂרֵה֙ שָׁנָ֔ה מָלַ֖ךְ בִּירוּשָׁלִָ֑ם וְשֵׁ֣ם אִמּ֔וֹ [חֲמִיטַל כ] (חֲמוּטַ֥ל ק) בַּֽת־ יִרְמְיָ֖הוּ מִלִּבְנָֽה׃
19הָרַ֖ע בְּעֵינֵ֣י יְהוָ֑ה כְּכֹ֥ל אֲשֶׁר־ עָשָׂ֖ה יְהוֹיָקִֽים׃
