1וְאֵ֙לֶּה֙ תּוֹלְדֹ֣ת בְּנֵי־ נֹ֔חַ שֵׁ֖ם חָ֣ם וָיָ֑פֶת וַיִּוָּלְד֥וּ לָהֶ֛ם בָּנִ֖ים אַחַ֥ר הַמַּבּֽוּל׃
2יֶ֔פֶת גֹּ֣מֶר וּמָג֔וֹג וּמָדַ֖י וְיָוָ֣ן וְתֻבָ֑ל וּמֶ֖שֶׁךְ וְתִירָֽס׃
3גֹּ֑מֶר אַשְׁכֲּנַ֥ז וְרִיפַ֖ת וְתֹגַרְמָֽה׃
4יָוָ֖ן אֱלִישָׁ֣ה וְתַרְשִׁ֑ישׁ כִּתִּ֖ים וְדֹדָנִֽים׃
5נִפְרְד֞וּ אִיֵּ֤י הַגּוֹיִם֙ בְּאַרְצֹתָ֔ם אִ֖ישׁ לִלְשֹׁנ֑וֹ לְמִשְׁפְּחֹתָ֖ם בְּגוֹיֵהֶֽם׃
6חָ֑ם כּ֥וּשׁ וּמִצְרַ֖יִם וּפ֥וּט וּכְנָֽעַן׃
7כ֔וּשׁ סְבָא֙ וַֽחֲוִילָ֔ה וְסַבְתָּ֥ה וְרַעְמָ֖ה וְסַבְתְּכָ֑א וּבְנֵ֥י רַעְמָ֖ה שְׁבָ֥א וּדְדָֽן׃
8יָלַ֣ד אֶת־ נִמְרֹ֑ד ה֣וּא הֵחֵ֔ל לִֽהְי֥וֹת גִּבֹּ֖ר בָּאָֽרֶץ׃
9הָיָ֥ה גִבֹּֽר־ צַ֖יִד לִפְנֵ֣י יְהוָ֑ה עַל־ כֵּן֙ יֵֽאָמַ֔ר כְּנִמְרֹ֛ד גִּבּ֥וֹר צַ֖יִד לִפְנֵ֥י יְהוָֽה׃
10רֵאשִׁ֤ית מַמְלַכְתּוֹ֙ בָּבֶ֔ל וְאֶ֖רֶךְ וְאַכַּ֣ד וְכַלְנֵ֑ה בְּאֶ֖רֶץ שִׁנְעָֽר׃
11הָאָ֥רֶץ הַהִ֖וא יָצָ֣א אַשּׁ֑וּר וַיִּ֙בֶן֙ אֶת־ נִ֣ינְוֵ֔ה וְאֶת־ רְחֹבֹ֥ת עִ֖יר וְאֶת־ כָּֽלַח׃
12רֶ֔סֶן בֵּ֥ין נִֽינְוֵ֖ה וּבֵ֣ין כָּ֑לַח הִ֖וא הָעִ֥יר הַגְּדֹלָֽה׃
13יָלַ֞ד אֶת־ לוּדִ֧ים וְאֶת־ עֲנָמִ֛ים וְאֶת־ לְהָבִ֖ים וְאֶת־ נַפְתֻּחִֽים׃
14פַּתְרֻסִ֞ים וְאֶת־ כַּסְלֻחִ֗ים אֲשֶׁ֨ר יָצְא֥וּ מִשָּׁ֛ם פְּלִשְׁתִּ֖ים וְאֶת־ כַּפְתֹּרִֽים׃ ס
15יָלַ֛ד אֶת־ צִידֹ֥ן בְּכֹר֖וֹ וְאֶת־ חֵֽת׃
16הַיְבוּסִי֙ וְאֶת־ הָ֣אֱמֹרִ֔י וְאֵ֖ת הַגִּרְגָּשִֽׁי׃
17הַֽחִוִּ֥י וְאֶת־ הַֽעַרְקִ֖י וְאֶת־ הַסִּינִֽי׃
18הָֽאַרְוָדִ֥י וְאֶת־ הַצְּמָרִ֖י וְאֶת־ הַֽחֲמָתִ֑י וְאַחַ֣ר נָפֹ֔צוּ מִשְׁפְּח֖וֹת הַֽכְּנַעֲנִֽי׃
19גְּב֤וּל הַֽכְּנַעֲנִי֙ מִצִּידֹ֔ן בֹּאֲכָ֥ה גְרָ֖רָה עַד־ עַזָּ֑ה בֹּאֲכָ֞ה סְדֹ֧מָה וַעֲמֹרָ֛ה וְאַדְמָ֥ה וּצְבֹיִ֖ם עַד־ לָֽשַׁע׃
20בְנֵי־ חָ֔ם לְמִשְׁפְּחֹתָ֖ם לִלְשֹֽׁנֹתָ֑ם בְּאַרְצֹתָ֖ם בְּגוֹיֵהֶֽם׃ ס
21יֻלַּ֖ד גַּם־ ה֑וּא אֲבִי֙ כָּל־ בְּנֵי־ עֵ֔בֶר אֲחִ֖י יֶ֥פֶת הַגָּדֽוֹל׃
22שֵׁ֖ם עֵילָ֣ם וְאַשּׁ֑וּר וְאַרְפַּכְשַׁ֖ד וְל֥וּד וַֽאֲרָֽם׃
23אֲרָ֑ם ע֥וּץ וְח֖וּל וְגֶ֥תֶר וָמַֽשׁ׃
24יָלַ֣ד אֶת־ שָׁ֑לַח וְשֶׁ֖לַח יָלַ֥ד אֶת־ עֵֽבֶר׃
25יֻלַּ֖ד שְׁנֵ֣י בָנִ֑ים שֵׁ֣ם הָֽאֶחָ֞ד פֶּ֗לֶג כִּ֤י בְיָמָיו֙ נִפְלְגָ֣ה הָאָ֔רֶץ וְשֵׁ֥ם אָחִ֖יו יָקְטָֽן׃
26יָלַ֔ד אֶת־ אַלְמוֹדָ֖ד וְאֶת־ שָׁ֑לֶף וְאֶת־ חֲצַרְמָ֖וֶת וְאֶת־ יָֽרַח׃
27הֲדוֹרָ֥ם וְאֶת־ אוּזָ֖ל וְאֶת־ דִּקְלָֽה׃
28עוֹבָ֥ל וְאֶת־ אֲבִֽימָאֵ֖ל וְאֶת־ שְׁבָֽא׃
29אוֹפִ֥ר וְאֶת־ חֲוִילָ֖ה וְאֶת־ יוֹבָ֑ב כָּל־ אֵ֖לֶּה בְּנֵ֥י יָקְטָֽן׃
30מוֹשָׁבָ֖ם מִמֵּשָׁ֑א בֹּאֲכָ֥ה סְפָ֖רָה הַ֥ר הַקֶּֽדֶם׃
31בְנֵי־ שֵׁ֔ם לְמִשְׁפְּחֹתָ֖ם לִלְשֹׁנֹתָ֑ם בְּאַרְצֹתָ֖ם לְגוֹיֵהֶֽם׃
