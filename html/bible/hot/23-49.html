1שִׁמְע֤וּ אִיִּים֙ אֵלַ֔י וְהַקְשִׁ֥יבוּ לְאֻמִּ֖ים מֵרָח֑וֹק יְהוָה֙ מִבֶּ֣טֶן קְרָאָ֔נִי מִמְּעֵ֥י אִמִּ֖י הִזְכִּ֥יר שְׁמִֽי׃
2פִּי֙ כְּחֶ֣רֶב חַדָּ֔ה בְּצֵ֥ל יָד֖וֹ הֶחְבִּיאָ֑נִי וַיְשִׂימֵ֙נִי֙ לְחֵ֣ץ בָּר֔וּר בְּאַשְׁפָּת֖וֹ הִסְתִּירָֽנִי׃
3לִ֖י עַבְדִּי־ אָ֑תָּה יִשְׂרָאֵ֕ל אֲשֶׁר־ בְּךָ֖ אֶתְפָּאָֽר׃
4אָמַ֙רְתִּי֙ לְרִ֣יק יָגַ֔עְתִּי לְתֹ֥הוּ וְהֶ֖בֶל כֹּחִ֣י כִלֵּ֑יתִי אָכֵן֙ מִשְׁפָּטִ֣י אֶת־ יְהוָ֔ה וּפְעֻלָּתִ֖י אֶת־ אֱלֹהָֽי׃
5אָמַ֣ר יְהוָ֗ה יֹצְרִ֤י מִבֶּ֙טֶן֙ לְעֶ֣בֶד ל֔וֹ לְשׁוֹבֵ֤ב יַֽעֲקֹב֙ אֵלָ֔יו וְיִשְׂרָאֵ֖ל [לֹא כ] (לֹ֣ו ק) יֵאָסֵ֑ף וְאֶכָּבֵד֙ בְּעֵינֵ֣י יְהוָ֔ה וֵאלֹהַ֖י הָיָ֥ה עֻזִּֽי׃
6נָקֵ֨ל מִֽהְיוֹתְךָ֥ לִי֙ עֶ֔בֶד לְהָקִים֙ אֶת־ שִׁבְטֵ֣י יַעֲקֹ֔ב [וּנְצִירֵי כ] (וּנְצוּרֵ֥י ק) יִשְׂרָאֵ֖ל לְהָשִׁ֑יב וּנְתַתִּ֙יךָ֙ לְא֣וֹר גּוֹיִ֔ם לִֽהְי֥וֹת יְשׁוּעָתִ֖י עַד־ קְצֵ֥ה הָאָֽרֶץ׃ ס
7אָֽמַר־ יְהוָה֩ גֹּאֵ֨ל יִשְׂרָאֵ֜ל קְדוֹשׁ֗וֹ לִבְזֹה־ נֶ֜פֶשׁ לִמְתָ֤עֵֽב גּוֹי֙ לְעֶ֣בֶד מֹשְׁלִ֔ים מְלָכִים֙ יִרְא֣וּ וָקָ֔מוּ שָׂרִ֖ים וְיִֽשְׁתַּחֲוּ֑וּ לְמַ֤עַן יְהוָה֙ אֲשֶׁ֣ר נֶאֱמָ֔ן קְדֹ֥שׁ יִשְׂרָאֵ֖ל וַיִּבְחָרֶֽךָּ׃
8אָמַ֣ר יְהוָ֗ה בְּעֵ֤ת רָצוֹן֙ עֲנִיתִ֔יךָ וּבְי֥וֹם יְשׁוּעָ֖ה עֲזַרְתִּ֑יךָ וְאֶצָּרְךָ֗ וְאֶתֶּנְךָ֙ לִבְרִ֣ית עָ֔ם לְהָקִ֣ים אֶ֔רֶץ לְהַנְחִ֖יל נְחָל֥וֹת שֹׁמֵמֽוֹת׃
9לַֽאֲסוּרִים֙ צֵ֔אוּ לַאֲשֶׁ֥ר בַּחֹ֖שֶׁךְ הִגָּל֑וּ עַל־ דְּרָכִ֣ים יִרְע֔וּ וּבְכָל־ שְׁפָיִ֖ים מַרְעִיתָֽם׃
10יִרְעָ֙בוּ֙ וְלֹ֣א יִצְמָ֔אוּ וְלֹא־ יַכֵּ֥ם שָׁרָ֖ב וָשָׁ֑מֶשׁ כִּי־ מְרַחֲמָ֣ם יְנַהֲגֵ֔ם וְעַל־ מַבּ֥וּעֵי מַ֖יִם יְנַהֲלֵֽם׃
11כָל־ הָרַ֖י לַדָּ֑רֶךְ וּמְסִלֹּתַ֖י יְרֻמֽוּן׃
12אֵ֕לֶּה מֵרָח֖וֹק יָבֹ֑אוּ וְהִֽנֵּה־ אֵ֙לֶּה֙ מִצָּפ֣וֹן וּמִיָּ֔ם וְאֵ֖לֶּה מֵאֶ֥רֶץ סִינִֽים׃
13שָׁמַ֙יִם֙ וְגִ֣ילִי אָ֔רֶץ [יִפְצְחוּ כ] (וּפִצְח֥וּ ק) הָרִ֖ים רִנָּ֑ה כִּֽי־ נִחַ֤ם יְהוָה֙ עַמּ֔וֹ וַעֲנִיָּ֖ו יְרַחֵֽם׃ ס
14צִיּ֖וֹן עֲזָבַ֣נִי יְהוָ֑ה וַאדֹנָ֖י שְׁכֵחָֽנִי׃
15אִשָּׁה֙ עוּלָ֔הּ מֵרַחֵ֖ם בֶּן־ בִּטְנָ֑הּ גַּם־ אֵ֣לֶּה תִשְׁכַּ֔חְנָה וְאָנֹכִ֖י לֹ֥א אֶשְׁכָּחֵֽךְ׃
16עַל־ כַּפַּ֖יִם חַקֹּתִ֑יךְ חוֹמֹתַ֥יִךְ נֶגְדִּ֖י תָּמִֽיד׃
17בָּנָ֑יִךְ מְהָֽרְסַ֥יִךְ וּמַחֲרִבַ֖יִךְ מִמֵּ֥ךְ יֵצֵֽאוּ׃
18סָבִ֤יב עֵינַ֙יִךְ֙ וּרְאִ֔י כֻּלָּ֖ם נִקְבְּצ֣וּ בָֽאוּ־ לָ֑ךְ חַי־ אָ֣נִי נְאֻם־ יְהוָ֗ה כִּ֤י כֻלָּם֙ כָּעֲדִ֣י תִלְבָּ֔שִׁי וּֽתְקַשְּׁרִ֖ים כַּכַּלָּֽה׃
19חָרְבֹתַ֙יִךְ֙ וְשֹׁ֣מְמֹתַ֔יִךְ וְאֶ֖רֶץ הֲרִֽסֻתֵ֑יךְ כִּ֤י עַתָּה֙ תֵּצְרִ֣י מִיּוֹשֵׁ֔ב וְרָחֲק֖וּ מְבַלְּעָֽיִךְ׃
20יֹאמְר֣וּ בְאָזְנַ֔יִךְ בְּנֵ֖י שִׁכֻּלָ֑יִךְ צַר־ לִ֥י הַמָּק֖וֹם גְּשָׁה־ לִּ֥י וְאֵשֵֽׁבָה׃
21בִּלְבָבֵ֗ךְ מִ֤י יָֽלַד־ לִי֙ אֶת־ אֵ֔לֶּה וַאֲנִ֥י שְׁכוּלָ֖ה וְגַלְמוּדָ֑ה גֹּלָ֣ה ׀ וְסוּרָ֗ה וְאֵ֙לֶּה֙ מִ֣י גִדֵּ֔ל הֵ֤ן אֲנִי֙ נִשְׁאַ֣רְתִּי לְבַדִּ֔י אֵ֖לֶּה אֵיפֹ֥ה הֵֽם׃ פ
22אָמַ֞ר אֲדֹנָ֣י יְהוִ֗ה הִנֵּ֨ה אֶשָּׂ֤א אֶל־ גּוֹיִם֙ יָדִ֔י וְאֶל־ עַמִּ֖ים אָרִ֣ים נִסִּ֑י וְהֵבִ֤יאוּ בָנַ֙יִךְ֙ בְּחֹ֔צֶן וּבְנֹתַ֖יִךְ עַל־ כָּתֵ֥ף תִּנָּשֶֽׂאנָה׃
23מְלָכִ֜ים אֹֽמְנַ֗יִךְ וְשָׂרֽוֹתֵיהֶם֙ מֵינִ֣יקֹתַ֔יִךְ אַפַּ֗יִם אֶ֚רֶץ יִשְׁתַּ֣חֲווּ לָ֔ךְ וַעֲפַ֥ר רַגְלַ֖יִךְ יְלַחֵ֑כוּ וְיָדַ֙עַתְּ֙ כִּֽי־ אֲנִ֣י יְהוָ֔ה אֲשֶׁ֥ר לֹֽא־ יֵבֹ֖שׁוּ קוָֹֽי׃ ס
24מִגִּבּ֖וֹר מַלְק֑וֹחַ וְאִם־ שְׁבִ֥י צַדִּ֖יק יִמָּלֵֽט׃
25כֹ֣ה ׀ אָמַ֣ר יְהוָ֗ה גַּם־ שְׁבִ֤י גִבּוֹר֙ יֻקָּ֔ח וּמַלְק֥וֹחַ עָרִ֖יץ יִמָּלֵ֑ט וְאֶת־ יְרִיבֵךְ֙ אָנֹכִ֣י אָרִ֔יב וְאֶת־ בָּנַ֖יִךְ אָנֹכִ֥י אוֹשִֽׁיעַ׃
