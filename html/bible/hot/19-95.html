1לְ֭כוּ נְרַנְּנָ֣ה לַיהוָ֑ה נָ֝רִ֗יעָה לְצ֣וּר יִשְׁעֵֽנוּ׃
2פָנָ֣יו בְּתוֹדָ֑ה בִּ֝זְמִר֗וֹת נָרִ֥יעַֽ לֽוֹ׃
3אֵ֣ל גָּד֣וֹל יְהוָ֑ה וּמֶ֥לֶךְ גָּ֝ד֗וֹל עַל־ כָּל־ אֱלֹהִֽים׃
4בְּ֭יָדוֹ מֶחְקְרֵי־ אָ֑רֶץ וְתוֹעֲפ֖וֹת הָרִ֣ים לֽוֹ׃
5ל֣וֹ הַ֭יָּם וְה֣וּא עָשָׂ֑הוּ וְ֝יַבֶּ֗שֶׁת יָדָ֥יו יָצָֽרוּ׃
6נִשְׁתַּחֲוֶ֣ה וְנִכְרָ֑עָה נִ֝בְרְכָ֗ה לִֽפְנֵי־ יְהוָ֥ה עֹשֵֽׂנוּ׃
7ה֤וּא אֱלֹהֵ֗ינוּ וַאֲנַ֤חְנוּ עַ֣ם מַ֭רְעִיתוֹ וְצֹ֣אן יָד֑וֹ הַ֝יּ֗וֹם אִֽם־ בְּקֹל֥וֹ תִשְׁמָֽעוּ׃
8תַּקְשׁ֣וּ לְ֭בַבְכֶם כִּמְרִיבָ֑ה כְּי֥וֹם מַ֝סָּ֗ה בַּמִּדְבָּֽר׃
9נִ֭סּוּנִי אֲבוֹתֵיכֶ֑ם בְּ֝חָנ֗וּנִי גַּם־ רָא֥וּ פָעֳלִֽי׃
10שָׁנָ֨ה ׀ אָ֘ק֤וּט בְּד֗וֹר וָאֹמַ֗ר עַ֤ם תֹּעֵ֣י לֵבָ֣ב הֵ֑ם וְ֝הֵ֗ם לֹא־ יָדְע֥וּ דְרָכָֽי׃
