1הֲלֹֽא־ חָכְמָ֥ה תִקְרָ֑א וּ֝תְבוּנָ֗ה תִּתֵּ֥ן קוֹלָֽהּ׃
2מְרוֹמִ֥ים עֲלֵי־ דָ֑רֶךְ בֵּ֖ית נְתִיב֣וֹת נִצָּֽבָה׃
3שְׁעָרִ֥ים לְפִי־ קָ֑רֶת מְב֖וֹא פְתָחִ֣ים תָּרֹֽנָּה׃
4אִישִׁ֣ים אֶקְרָ֑א וְ֝קוֹלִ֗י אֶל־ בְּנֵ֥י אָדָֽם׃
5פְתָאיִ֣ם עָרְמָ֑ה וּ֝כְסִילִ֗ים הָבִ֥ינוּ לֵֽב׃
6כִּֽי־ נְגִידִ֣ים אֲדַבֵּ֑ר וּמִפְתַּ֥ח שְׂ֝פָתַ֗י מֵישָׁרִֽים׃
7אֱ֭מֶת יֶהְגֶּ֣ה חִכִּ֑י וְתוֹעֲבַ֖ת שְׂפָתַ֣י רֶֽשַׁע׃
8כָּל־ אִמְרֵי־ פִ֑י אֵ֥ין בָּ֝הֶ֗ם נִפְתָּ֥ל וְעִקֵּֽשׁ׃
9נְ֭כֹחִים לַמֵּבִ֑ין וִֽ֝ישָׁרִ֗ים לְמֹ֣צְאֵי דָֽעַת׃
10מוּסָרִ֥י וְאַל־ כָּ֑סֶף וְ֝דַ֗עַת מֵחָר֥וּץ נִבְחָֽר׃
11טוֹבָ֣ה חָ֭כְמָה מִפְּנִינִ֑ים וְכָל־ חֲ֝פָצִ֗ים לֹ֣א יִֽשְׁווּ־ בָֽהּ׃
12חָ֭כְמָה שָׁכַ֣נְתִּי עָרְמָ֑ה וְדַ֖עַת מְזִמּ֣וֹת אֶמְצָֽא׃
13יְהוָה֮ שְֽׂנֹ֫את רָ֥ע גֵּ֘אָ֤ה וְגָא֨וֹן ׀ וְדֶ֣רֶךְ רָ֭ע וּפִ֨י תַהְפֻּכ֬וֹת שָׂנֵֽאתִי׃
14עֵ֭צָה וְתוּשִׁיָּ֑ה אֲנִ֥י בִ֝ינָ֗ה לִ֣י גְבוּרָֽה׃
15מְלָכִ֣ים יִמְלֹ֑כוּ וְ֝רוֹזְנִ֗ים יְחֹ֣קְקוּ צֶֽדֶק׃
16שָׂרִ֣ים יָשֹׂ֑רוּ וּ֝נְדִיבִ֗ים כָּל־ שֹׁ֥פְטֵי צֶֽדֶק׃
17[אֹהֲבֶיהָ כ] (אֹהֲבַ֣י ק) אֵהָ֑ב וּ֝מְשַׁחֲרַ֗י יִמְצָאֻֽנְנִי׃
18וְכָב֥וֹד אִתִּ֑י ה֥וֹן עָ֝תֵ֗ק וּצְדָקָֽה׃
19פִּ֭רְיִי מֵחָר֣וּץ וּמִפָּ֑ז וּ֝תְבוּאָתִ֗י מִכֶּ֥סֶף נִבְחָֽר׃
20צְדָקָ֥ה אֲהַלֵּ֑ך בְּ֝ת֗וֹךְ נְתִיב֥וֹת מִשְׁפָּֽט׃
21אֹהֲבַ֥י ׀ יֵ֑שׁ וְאֹצְרֹ֖תֵיהֶ֣ם אֲמַלֵּֽא׃ פ
22קָ֭נָנִי רֵאשִׁ֣ית דַּרְכּ֑וֹ קֶ֖דֶם מִפְעָלָ֣יו מֵאָֽז׃
23נִסַּ֥כְתִּי מֵרֹ֗אשׁ מִקַּדְמֵי־ אָֽרֶץ׃
24תְּהֹמ֥וֹת חוֹלָ֑לְתִּי בְּאֵ֥ין מַ֝עְיָנ֗וֹת נִכְבַּדֵּי־ מָֽיִם׃
25הָרִ֣ים הָטְבָּ֑עוּ לִפְנֵ֖י גְבָע֣וֹת חוֹלָֽלְתִּי׃
26לֹ֣א עָ֭שָׂה אֶ֣רֶץ וְחוּצ֑וֹת וְ֝רֹ֗אשׁ עָפְר֥וֹת תֵּבֵֽל׃
27שָׁ֭מַיִם שָׁ֣ם אָ֑נִי בְּח֥וּקוֹ ח֝֗וּג עַל־ פְּנֵ֥י תְהֽוֹם׃
28שְׁחָקִ֣ים מִמָּ֑עַל בַּ֝עֲז֗וֹז עִינ֥וֹת תְּהוֹם׃
29לַיָּ֨ם ׀ חֻקּ֗וֹ וּ֭מַיִם לֹ֣א יַֽעַבְרוּ־ פִ֑יו בְּ֝חוּק֗וֹ מ֣וֹסְדֵי אָֽרֶץ׃
30אֶצְל֗וֹ אָ֫מ֥וֹן וָֽאֶהְיֶ֣ה שַׁ֭עֲשֻׁעִים י֤וֹם ׀ י֑וֹם מְשַׂחֶ֖קֶת לְפָנָ֣יו בְּכָל־ עֵֽת׃
31בְּתֵבֵ֣ל אַרְצ֑וֹ וְ֝שַׁעֲשֻׁעַ֗י אֶת־ בְּנֵ֥י אָדָֽם׃ פ
32בָ֭נִים שִׁמְעוּ־ לִ֑י וְ֝אַשְׁרֵ֗י דְּרָכַ֥י יִשְׁמֹֽרוּ׃
33מוּסָ֥ר וַחֲכָ֗מוּ וְאַל־ תִּפְרָֽעוּ׃
34אָדָם֮ שֹׁמֵ֪עַֽ֫ לִ֥י לִשְׁקֹ֣ד עַל־ דַּ֭לְתֹתַי י֤וֹם ׀ י֑וֹם לִ֝שְׁמֹ֗ר מְזוּזֹ֥ת פְּתָחָֽי׃
35מֹ֭צְאִי [מֹצְאֵי כ] (מָצָ֣א ק) חַיִּ֑ים וַיָּ֥פֶק רָ֝צ֗וֹן מֵיְהוָֽה׃
