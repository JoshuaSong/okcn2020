1וַיַּ֖עַן יְהוָ֥ה אֶת־ אִיּ֗וֹב וַיֹּאמַֽר׃
2עִם־ שַׁדַּ֣י יִסּ֑וֹר מוֹכִ֖יחַ אֱל֣וֹהַּ יַעֲנֶֽנָּה׃ פ
3אִיּ֥וֹב אֶת־ יְהוָ֗ה וַיֹּאמַֽר׃
4קַ֭לֹּתִי מָ֣ה אֲשִׁיבֶ֑ךָּ יָ֝דִ֗י שַׂ֣מְתִּי לְמוֹ־ פִֽי׃
5דִּ֭בַּרְתִּי וְלֹ֣א אֶֽעֱנֶ֑ה וּ֝שְׁתַּ֗יִם וְלֹ֣א אוֹסִֽיף׃ פ
6יְהוָ֣ה אֶת־ אִ֭יּוֹב [מִן כ] [סְעָרָה כ] (מִ֥ן ק) (סְעָרָ֗ה ק) וַיֹּאמַֽר׃
7נָ֣א כְגֶ֣בֶר חֲלָצֶ֑יךָ אֶ֝שְׁאָלְךָ֗ וְהוֹדִיעֵֽנִי׃
8תָּפֵ֣ר מִשְׁפָּטִ֑י תַּ֝רְשִׁיעֵ֗נִי לְמַ֣עַן תִּצְדָּֽק׃
9זְר֖וֹעַ כָּאֵ֥ל ׀ לָ֑ךְ וּ֝בְק֗וֹל כָּמֹ֥הוּ תַרְעֵֽם׃
10נָ֣א גָֽא֣וֹן וָגֹ֑בַהּ וְה֖וֹד וְהָדָ֣ר תִּלְבָּֽשׁ׃
11עֶבְר֣וֹת אַפֶּ֑ךָ וּרְאֵ֥ה כָל־ גֵּ֝אֶ֗ה וְהַשְׁפִּילֵֽהוּ׃
12כָל־ גֵּ֭אֶה הַכְנִיעֵ֑הוּ וַהֲדֹ֖ךְ רְשָׁעִ֣ים תַּחְתָּֽם׃
13בֶּעָפָ֣ר יָ֑חַד פְּ֝נֵיהֶ֗ם חֲבֹ֣שׁ בַּטָּמֽוּן׃
14אֲנִ֥י אוֹדֶ֑ךָּ כִּֽי־ תוֹשִׁ֖עַ לְךָ֣ יְמִינֶֽךָ׃
15נָ֣א בְ֭הֵמוֹת אֲשֶׁר־ עָשִׂ֣יתִי עִמָּ֑ךְ חָ֝צִ֗יר כַּבָּקָ֥ר יֹאכֵֽל׃
16נָ֣א כֹח֣וֹ בְמָתְנָ֑יו וְ֝אֹנ֗וֹ בִּשְׁרִירֵ֥י בִטְנֽוֹ׃
17זְנָב֣וֹ כְמוֹ־ אָ֑רֶז גִּידֵ֖י [פַחֲדֹו כ] (פַחֲדָ֣יו ק) יְשֹׂרָֽגוּ׃
18אֲפִיקֵ֣י נְחוּשָׁ֑ה גְּ֝רָמָ֗יו כִּמְטִ֥יל בַּרְזֶֽל׃
19רֵאשִׁ֣ית דַּרְכֵי־ אֵ֑ל הָ֝עֹשׂוֹ יַגֵּ֥שׁ חַרְבּֽוֹ׃
20ב֭וּל הָרִ֣ים יִשְׂאוּ־ ל֑וֹ וְֽכָל־ חַיַּ֥ת הַ֝שָּׂדֶ֗ה יְשַֽׂחֲקוּ־ שָֽׁם׃
21צֶאֱלִ֥ים יִשְׁכָּ֑ב בְּסֵ֖תֶר קָנֶ֣ה וּבִצָּֽה׃
22צֶאֱלִ֣ים צִֽלֲל֑וֹ יְ֝סֻבּ֗וּהוּ עַרְבֵי־ נָֽחַל׃
23יַעֲשֹׁ֣ק נָ֭הָר לֹ֣א יַחְפּ֑וֹז יִבְטַ֓ח ׀ כִּֽי־ יָגִ֖יחַ יַרְדֵּ֣ן אֶל־ פִּֽיהוּ׃
