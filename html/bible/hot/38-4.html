1וַיָּ֕שָׁב הַמַּלְאָ֖ךְ הַדֹּבֵ֣ר בִּ֑י וַיְעִירֵ֕נִי כְּאִ֖ישׁ אֲשֶׁר־ יֵע֥וֹר מִשְּׁנָתֽוֹ׃
2אֵלַ֔י מָ֥ה אַתָּ֖ה רֹאֶ֑ה [וַיֹּאמֶר כ] (וָאֹמַ֡ר ק) רָאִ֣יתִי ׀ וְהִנֵּ֣ה מְנוֹרַת֩ זָהָ֨ב כֻּלָּ֜הּ וְגֻלָּ֣הּ עַל־ רֹאשָׁ֗הּ וְשִׁבְעָ֤ה נֵרֹתֶ֙יהָ֙ עָלֶ֔יהָ שִׁבְעָ֤ה וְשִׁבְעָה֙ מֽוּצָק֔וֹת לַנֵּר֖וֹת אֲשֶׁ֥ר עַל־ רֹאשָֽׁהּ׃
3זֵיתִ֖ים עָלֶ֑יהָ אֶחָד֙ מִימִ֣ין הַגֻּלָּ֔ה וְאֶחָ֖ד עַל־ שְׂמֹאלָֽהּ׃
4וָֽאֹמַ֔ר אֶל־ הַמַּלְאָ֛ךְ הַדֹּבֵ֥ר בִּ֖י לֵאמֹ֑ר מָה־ אֵ֖לֶּה אֲדֹנִֽי׃
5הַמַּלְאָ֞ךְ הַדֹּבֵ֥ר בִּי֙ וַיֹּ֣אמֶר אֵלַ֔י הֲל֥וֹא יָדַ֖עְתָּ מָה־ הֵ֣מָּה אֵ֑לֶּה וָאֹמַ֖ר לֹ֥א אֲדֹנִֽי׃
6וַיֹּ֤אמֶר אֵלַי֙ לֵאמֹ֔ר זֶ֚ה דְּבַר־ יְהוָ֔ה אֶל־ זְרֻבָּבֶ֖ל לֵאמֹ֑ר לֹ֤א בְחַ֙יִל֙ וְלֹ֣א בְכֹ֔חַ כִּ֣י אִם־ בְּרוּחִ֔י אָמַ֖ר יְהוָ֥ה צְבָאֽוֹת׃
7אַתָּ֧ה הַֽר־ הַגָּד֛וֹל לִפְנֵ֥י זְרֻבָּבֶ֖ל לְמִישֹׁ֑ר וְהוֹצִיא֙ אֶת־ הָאֶ֣בֶן הָרֹאשָׁ֔ה תְּשֻׁא֕וֹת חֵ֥ן חֵ֖ן לָֽהּ׃ פ
8דְבַר־ יְהוָ֖ה אֵלַ֥י לֵאמֹֽר׃
9זְרֻבָּבֶ֗ל יִסְּד֛וּ הַבַּ֥יִת הַזֶּ֖ה וְיָדָ֣יו תְּבַצַּ֑עְנָה וְיָ֣דַעְתָּ֔ כִּֽי־ יְהוָ֥ה צְבָא֖וֹת שְׁלָחַ֥נִי אֲלֵיכֶֽם׃
10מִ֣י בַז֮ לְי֣וֹם קְטַנּוֹת֒ וְשָׂמְח֗וּ וְרָא֞וּ אֶת־ הָאֶ֧בֶן הַבְּדִ֛יל בְּיַ֥ד זְרֻבָּבֶ֖ל שִׁבְעָה־ אֵ֑לֶּה עֵינֵ֣י יְהוָ֔ה הֵ֥מָּה מְשׁוֹטְטִ֖ים בְּכָל־ הָאָֽרֶץ׃
11וָאֹמַ֣ר אֵלָ֑יו מַה־ שְּׁנֵ֤י הַזֵּיתִים֙ הָאֵ֔לֶה עַל־ יְמִ֥ין הַמְּנוֹרָ֖ה וְעַל־ שְׂמֹאולָֽהּ׃
12שֵׁנִ֔ית וָאֹמַ֖ר אֵלָ֑יו מַה־ שְׁתֵּ֞י שִׁבֲּלֵ֣י הַזֵּיתִ֗ים אֲשֶׁר֙ בְּיַ֗ד שְׁנֵי֙ צַנְתְּר֣וֹת הַזָּהָ֔ב הַֽמְרִיקִ֥ים מֵעֲלֵיהֶ֖ם הַזָּהָֽב׃
13אֵלַי֙ לֵאמֹ֔ר הֲל֥וֹא יָדַ֖עְתָּ מָה־ אֵ֑לֶּה וָאֹמַ֖ר לֹ֥א אֲדֹנִֽי׃
