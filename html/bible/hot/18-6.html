1וַיַּ֥עַן אִיּ֗וֹב וַיֹּאמַֽר׃
2שָׁק֣וֹל יִשָּׁקֵ֣ל כַּעְשִׂ֑י [וְהַיָּתִי כ] (וְ֝הַוָּתִ֗י ק) בְּֽמֹאזְנַ֥יִם יִשְׂאוּ־ יָֽחַד׃
3עַתָּ֗ה מֵח֣וֹל יַמִּ֣ים יִכְבָּ֑ד עַל־ כֵּ֝֗ן דְּבָרַ֥י לָֽעוּ׃
4חִצֵּ֪י שַׁדַּ֡י עִמָּדִ֗י אֲשֶׁ֣ר חֲ֭מָתָם שֹׁתָ֣ה רוּחִ֑י בִּעוּתֵ֖י אֱל֣וֹהַּ יַֽעַרְכֽוּנִי׃
5פֶּ֥רֶא עֲלֵי־ דֶ֑שֶׁא אִ֥ם יִגְעֶה־ שּׁ֝֗וֹר עַל־ בְּלִילֽוֹ׃
6תָּ֭פֵל מִבְּלִי־ מֶ֑לַח אִם־ יֶשׁ־ טַ֝֗עַם בְּרִ֣יר חַלָּמֽוּת׃
7לִנְגּ֣וֹעַ נַפְשִׁ֑י הֵ֝֗מָּה כִּדְוֵ֥י לַחְמִֽי׃
8יִ֭תֵּן תָּב֣וֹא שֶֽׁאֱלָתִ֑י וְ֝תִקְוָתִ֗י יִתֵּ֥ן אֱלֽוֹהַּ׃
9אֱ֭לוֹהַּ וִֽידַכְּאֵ֑נִי יַתֵּ֥ר יָ֝ד֗וֹ וִֽיבַצְּעֵֽנִי׃
10ע֨וֹד ׀ נֶ֘חָ֤מָתִ֗י וַאֲסַלְּדָ֣ה בְ֭חִילָה לֹ֣א יַחְמ֑וֹל כִּי־ לֹ֥א כִ֝חַ֗דְתִּי אִמְרֵ֥י קָדֽוֹשׁ׃
11כֹּחִ֥י כִֽי־ אֲיַחֵ֑ל וּמַה־ קִּ֝צִּ֗י כִּֽי־ אַאֲרִ֥יךְ נַפְשִֽׁי׃
12כֹּ֣חַ אֲבָנִ֣ים כֹּחִ֑י אִֽם־ בְּשָׂרִ֥י נָחֽוּשׁ׃
13אֵ֣ין עֶזְרָתִ֣י בִ֑י וְ֝תֻשִׁיָּ֗ה נִדְּחָ֥ה מִמֶּֽנִּי׃
14מֵרֵעֵ֣הוּ חָ֑סֶד וְיִרְאַ֖ת שַׁדַּ֣י יַעֲזֽוֹב׃
15בָּגְד֣וּ כְמוֹ־ נָ֑חַל כַּאֲפִ֖יק נְחָלִ֣ים יַעֲבֹֽרוּ׃
16מִנִּי־ קָ֑רַח עָ֝לֵ֗ימוֹ יִתְעַלֶּם־ שָֽׁלֶג׃
17יְזֹרְב֣וּ נִצְמָ֑תוּ בְּ֝חֻמּ֗וֹ נִדְעֲכ֥וּ מִמְּקוֹמָֽם׃
18אָרְח֣וֹת דַּרְכָּ֑ם יַעֲל֖וּ בַתֹּ֣הוּ וְיֹאבֵֽדוּ׃
19אָרְח֣וֹת תֵּמָ֑א הֲלִיכֹ֥ת שְׁ֝בָ֗א קִוּוּ־ לָֽמוֹ׃
20כִּֽי־ בָטָ֑ח בָּ֥אוּ עָ֝דֶ֗יהָ וַיֶּחְפָּֽרוּ׃
21עַ֭תָּה הֱיִ֣יתֶם [לֹא כ] (לֹ֑ו ק) תִּֽרְא֥וּ חֲ֝תַ֗ת וַתִּירָֽאוּ׃
22אָ֭מַרְתִּי הָ֣בוּ לִ֑י וּ֝מִכֹּחֲכֶ֗ם שִׁחֲד֥וּ בַעֲדִֽי׃
23מִיַּד־ צָ֑ר וּמִיַּ֖ד עָרִיצִ֣ים תִּפְדּֽוּנִי׃
24וַאֲנִ֣י אַחֲרִ֑ישׁ וּמַה־ שָּׁ֝גִ֗יתִי הָבִ֥ינוּ לִֽי׃
25נִּמְרְצ֥וּ אִמְרֵי־ יֹ֑שֶׁר וּמַה־ יּוֹכִ֖יחַ הוֹכֵ֣חַ מִכֶּֽם׃
26מִלִּ֣ים תַּחְשֹׁ֑בוּ וּ֝לְר֗וּחַ אִמְרֵ֥י נֹאָֽשׁ׃ אַף־ עַל־
27תַּפִּ֑ילוּ וְ֝תִכְר֗וּ עַל־ רֵֽיעֲכֶֽם׃
28הוֹאִ֣ילוּ פְנוּ־ בִ֑י וְעַל־ פְּ֝נֵיכֶ֗ם אִם־ אֲכַזֵּֽב׃
29נָ֭א אַל־ תְּהִ֣י עַוְלָ֑ה [וְשֻׁבִי כ] (וְשׁ֥וּבוּ ק) ע֝וֹד צִדְקִי־ בָֽהּ׃
