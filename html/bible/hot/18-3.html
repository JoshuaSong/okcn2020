1אַחֲרֵי־ כֵ֗ן פָּתַ֤ח אִיּוֹב֙ אֶת־ פִּ֔יהוּ וַיְקַלֵּ֖ל אֶת־ יוֹמֽוֹ׃ פ
2אִיּ֗וֹב וַיֹּאמַֽר׃
3י֭וֹם אִוָּ֣לֶד בּ֑וֹ וְהַלַּ֥יְלָה אָ֝מַ֗ר הֹ֣רָה גָֽבֶר׃
4הַה֗וּא יְֽהִ֫י חֹ֥שֶׁךְ אַֽל־ יִדְרְשֵׁ֣הוּ אֱל֣וֹהַּ מִמָּ֑עַל וְאַל־ תּוֹפַ֖ע עָלָ֣יו נְהָרָֽה׃
5חֹ֣שֶׁךְ וְ֭צַלְמָוֶת תִּשְׁכָּן־ עָלָ֣יו עֲנָנָ֑ה יְ֝בַעֲתֻ֗הוּ כִּֽמְרִ֥ירֵי יֽוֹם׃
6הַהוּא֮ יִקָּחֵ֪ה֫וּ אֹ֥פֶל אַל־ יִ֭חַדְּ בִּימֵ֣י שָׁנָ֑ה בְּמִסְפַּ֥ר יְ֝רָחִ֗ים אַל־ יָבֹֽא׃
7הַלַּ֣יְלָה הַ֭הוּא יְהִ֣י גַלְמ֑וּד אַל־ תָּבֹ֖א רְנָנָ֣ה בֽוֹ׃
8אֹרְרֵי־ י֑וֹם הָ֝עֲתִידִ֗ים עֹרֵ֥ר לִוְיָתָֽן׃
9כּוֹכְבֵ֪י נִ֫שְׁפּ֥וֹ יְקַו־ לְא֥וֹר וָאַ֑יִן וְאַל־ יִ֝רְאֶ֗ה בְּעַפְעַפֵּי־ שָֽׁחַר׃
10לֹ֣א סָ֭גַר דַּלְתֵ֣י בִטְנִ֑י וַיַּסְתֵּ֥ר עָ֝מָ֗ל מֵעֵינָֽי׃
11לֹּ֣א מֵרֶ֣חֶם אָמ֑וּת מִבֶּ֖טֶן יָצָ֣אתִי וְאֶגְוָֽע׃
12קִדְּמ֣וּנִי בִרְכָּ֑יִם וּמַה־ שָּׁ֝דַ֗יִם כִּ֣י אִינָֽק׃
13עַ֭תָּה שָׁכַ֣בְתִּי וְאֶשְׁק֑וֹט יָ֝שַׁ֗נְתִּי אָ֤ז ׀ יָנ֬וּחַֽ לִֽי׃
14מְ֭לָכִים וְיֹ֣עֲצֵי אָ֑רֶץ הַבֹּנִ֖ים חֳרָב֣וֹת לָֽמוֹ׃
15עִם־ שָׂ֭רִים זָהָ֣ב לָהֶ֑ם הַֽמְמַלְאִ֖ים בָּתֵּיהֶ֣ם כָּֽסֶף׃
16כְנֵ֣פֶל טָ֭מוּן לֹ֣א אֶהְיֶ֑ה כְּ֝עֹלְלִ֗ים לֹא־ רָ֥אוּ אֽוֹר׃
17רְ֭שָׁעִים חָ֣דְלוּ רֹ֑גֶז וְשָׁ֥ם יָ֝נ֗וּחוּ יְגִ֣יעֵי כֹֽחַ׃
18אֲסִירִ֣ים שַׁאֲנָ֑נוּ לֹ֥א שָׁ֝מְע֗וּ ק֣וֹל נֹגֵֽשׂ׃
19וְ֭גָדוֹל שָׁ֣ם ה֑וּא וְ֝עֶ֗בֶד חָפְשִׁ֥י מֵאֲדֹנָֽיו׃
20יִתֵּ֣ן לְעָמֵ֣ל א֑וֹר וְ֝חַיִּ֗ים לְמָ֣רֵי נָֽפֶשׁ׃
21לַמָּ֣וֶת וְאֵינֶ֑נּוּ וַֽ֝יַּחְפְּרֻ֗הוּ מִמַּטְמוֹנִֽים׃
22אֱלֵי־ גִ֑יל יָ֝שִׂ֗ישׂוּ כִּ֣י יִמְצְאוּ־ קָֽבֶר׃
23אֲשֶׁר־ דַּרְכּ֣וֹ נִסְתָּ֑רָה וַיָּ֖סֶךְ אֱל֣וֹהַּ בַּעֲדֽוֹ׃
24לִפְנֵ֣י לַ֭חְמִי אַנְחָתִ֣י תָבֹ֑א וַֽיִּתְּכ֥וּ כַ֝מַּ֗יִם שַׁאֲגֹתָֽי׃
25פַ֣חַד פָּ֭חַדְתִּי וַיֶּאֱתָיֵ֑נִי וַאֲשֶׁ֥ר יָ֝גֹ֗רְתִּי יָ֣בֹא לִֽי׃
