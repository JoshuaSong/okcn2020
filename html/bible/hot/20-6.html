1בְּ֭נִי אִם־ עָרַ֣בְתָּ לְרֵעֶ֑ךָ תָּקַ֖עְתָּ לַזָּ֣ר כַּפֶּֽיךָ׃
2בְאִמְרֵי־ פִ֑יךָ נִ֝לְכַּ֗דְתָּ בְּאִמְרֵי־ פִֽיךָ׃
3זֹ֥את אֵפ֪וֹא ׀ בְּנִ֡י וְֽהִנָּצֵ֗ל כִּ֘י בָ֤אתָ בְכַף־ רֵעֶ֑ךָ לֵ֥ךְ הִ֝תְרַפֵּ֗ס וּרְהַ֥ב רֵעֶֽיךָ׃
4תִּתֵּ֣ן שֵׁנָ֣ה לְעֵינֶ֑יךָ וּ֝תְנוּמָ֗ה לְעַפְעַפֶּֽיךָ׃
5כִּצְבִ֣י מִיָּ֑ד וּ֝כְצִפּ֗וֹר מִיַּ֥ד יָקֽוּשׁ׃ פ לֵֽךְ־ אֶל־
6עָצֵ֑ל רְאֵ֖ה דְרָכֶ֣יהָ וַחֲכָֽם׃
7אֵֽין־ לָ֥הּ קָצִ֗ין שֹׁטֵ֥ר וּמֹשֵֽׁל׃
8בַּקַּ֣יִץ לַחְמָ֑הּ אָגְרָ֥ה בַ֝קָּצִ֗יר מַאֲכָלָֽהּ׃
9מָתַ֖י עָצֵ֥ל ׀ תִּשְׁכָּ֑ב מָ֝תַ֗י תָּק֥וּם מִשְּׁנָתֶֽךָ׃
10שֵׁ֭נוֹת מְעַ֣ט תְּנוּמ֑וֹת מְעַ֓ט ׀ חִבֻּ֖ק יָדַ֣יִם לִשְׁכָּֽב׃
11כִמְהַלֵּ֥ךְ רֵאשֶׁ֑ךָ וּ֝מַחְסֹֽרְךָ֗ כְּאִ֣ישׁ מָגֵֽן׃ פ
12בְּ֭לִיַּעַל אִ֣ישׁ אָ֑וֶן ה֝וֹלֵ֗ךְ עִקְּשׁ֥וּת פֶּֽה׃
13בְּ֭עֵינָיו מֹלֵ֣ל בְּרַגְלָ֑ו מֹ֝רֶ֗ה בְּאֶצְבְּעֹתָֽיו׃
14בְּלִבּ֗וֹ חֹרֵ֣שׁ רָ֣ע בְּכָל־ עֵ֑ת [מְדָנִים כ] (מִדְיָנִ֥ים ק) יְשַׁלֵּֽחַ׃
15כֵּ֗ן פִּ֭תְאֹם יָב֣וֹא אֵיד֑וֹ פֶּ֥תַע יִ֝שָּׁבֵ֗ר וְאֵ֣ין מַרְפֵּֽא׃ פ
16הֵ֭נָּה שָׂנֵ֣א יְהוָ֑ה וְ֝שֶׁ֗בַע [תֹּועֲבֹות כ] (תֹּועֲבַ֥ת ק) נַפְשֽׁוֹ׃
17רָ֭מוֹת לְשׁ֣וֹן שָׁ֑קֶר וְ֝יָדַ֗יִם שֹׁפְכ֥וֹת דָּם־ נָקִֽי׃
18חֹ֭רֵשׁ מַחְשְׁב֣וֹת אָ֑וֶן רַגְלַ֥יִם מְ֝מַהֲר֗וֹת לָר֥וּץ לָֽרָעָה׃
19כְּ֭זָבִים עֵ֣ד שָׁ֑קֶר וּמְשַׁלֵּ֥חַ מְ֝דָנִ֗ים בֵּ֣ין אַחִֽים׃ פ
20בְּ֭נִי מִצְוַ֣ת אָבִ֑יךָ וְאַל־ תִּ֝טֹּ֗שׁ תּוֹרַ֥ת אִמֶּֽךָ׃
21עַל־ לִבְּךָ֣ תָמִ֑יד עָ֝נְדֵ֗ם עַל־ גַּרְגְּרֹתֶֽךָ׃
22תַּנְחֶ֬ה אֹתָ֗ךְ בְּֽ֭שָׁכְבְּךָ תִּשְׁמֹ֣ר עָלֶ֑יךָ וַ֝הֲקִיצ֗וֹתָ הִ֣יא תְשִׂיחֶֽךָ׃
23נֵ֣ר מִ֭צְוָה וְת֣וֹרָה א֑וֹר וְדֶ֥רֶךְ חַ֝יִּ֗ים תּוֹכְח֥וֹת מוּסָֽר׃
24מֵאֵ֣שֶׁת רָ֑ע מֵֽ֝חֶלְקַ֗ת לָשׁ֥וֹן נָכְרִיָּֽה׃
25תַּחְמֹ֣ד יָ֭פְיָהּ בִּלְבָבֶ֑ךָ וְאַל־ תִּ֝קָּֽחֲךָ֗ בְּעַפְעַפֶּֽיהָ׃
26בְעַד־ אִשָּׁ֥ה זוֹנָ֗ה עַֽד־ כִּכַּ֫ר לָ֥חֶם וְאֵ֥שֶׁת אִ֑ישׁ נֶ֖פֶשׁ יְקָרָ֣ה תָצֽוּד׃ פ
27אִ֓ישׁ אֵ֬שׁ בְּחֵיק֑וֹ וּ֝בְגָדָ֗יו לֹ֣א תִשָּׂרַֽפְנָה׃
28יְהַלֵּ֣ךְ אִ֭ישׁ עַל־ הַגֶּחָלִ֑ים וְ֝רַגְלָ֗יו לֹ֣א תִכָּוֶֽינָה׃
29הַ֭בָּא אֶל־ אֵ֣שֶׁת רֵעֵ֑הוּ לֹ֥א יִ֝נָּקֶ֗ה כָּֽל־ הַנֹּגֵ֥עַ בָּֽהּ׃
30יָב֣וּזוּ לַ֭גַּנָּב כִּ֣י יִגְנ֑וֹב לְמַלֵּ֥א נַ֝פְשׁ֗וֹ כִּ֣י יִרְעָֽב׃
31יְשַׁלֵּ֣ם שִׁבְעָתָ֑יִם אֶת־ כָּל־ ה֖וֹן בֵּית֣וֹ יִתֵּֽן׃
32אִשָּׁ֣ה חֲסַר־ לֵ֑ב מַֽשְׁחִ֥ית נַ֝פְשׁ֗וֹ ה֣וּא יַעֲשֶֽׂנָּה׃
33וְקָל֥וֹן יִמְצָ֑א וְ֝חֶרְפָּת֗וֹ לֹ֣א תִמָּחֶֽה׃
34קִנְאָ֥ה חֲמַת־ גָּ֑בֶר וְלֹֽא־ יַ֝חְמ֗וֹל בְּי֣וֹם נָקָֽם׃
