1וַיְדַבֵּ֥ר יְהֹוָ֖ה אֶל־ מֹשֶׁ֥ה לֵּאמֹֽר׃
2אֶל־ בְּנֵ֣י יִשְׂרָאֵל֒ וְיָשֻׁ֗בוּ וְיַחֲנוּ֙ לִפְנֵי֙ פִּ֣י הַחִירֹ֔ת בֵּ֥ין מִגְדֹּ֖ל וּבֵ֣ין הַיָּ֑ם לִפְנֵי֙ בַּ֣עַל צְפֹ֔ן נִכְח֥וֹ תַחֲנ֖וּ עַל־ הַיָּֽם׃
3פַּרְעֹה֙ לִבְנֵ֣י יִשְׂרָאֵ֔ל נְבֻכִ֥ים הֵ֖ם בָּאָ֑רֶץ סָגַ֥ר עֲלֵיהֶ֖ם הַמִּדְבָּֽר׃
4אֶת־ לֵב־ פַּרְעֹה֮ וְרָדַ֣ף אַחֲרֵיהֶם֒ וְאִכָּבְדָ֤ה בְּפַרְעֹה֙ וּבְכָל־ חֵיל֔וֹ וְיָדְע֥וּ מִצְרַ֖יִם כִּֽי־ אֲנִ֣י יְהוָ֑ה וַיַּֽעֲשׂוּ־ כֵֽן׃
5לְמֶ֣לֶךְ מִצְרַ֔יִם כִּ֥י בָרַ֖ח הָעָ֑ם וַ֠יֵּהָפֵךְ לְבַ֨ב פַּרְעֹ֤ה וַעֲבָדָיו֙ אֶל־ הָעָ֔ם וַיֹּֽאמרוּ֙ מַה־ זֹּ֣את עָשִׂ֔ינוּ כִּֽי־ שִׁלַּ֥חְנוּ אֶת־ יִשְׂרָאֵ֖ל מֵעָבְדֵֽנוּ׃
6אֶת־ רִכְבּ֑וֹ וְאֶת־ עַמּ֖וֹ לָקַ֥ח עִמּֽוֹ׃
7שֵׁשׁ־ מֵא֥וֹת רֶ֙כֶב֙ בָּח֔וּר וְכֹ֖ל רֶ֣כֶב מִצְרָ֑יִם וְשָׁלִשִׁ֖ם עַל־ כֻּלּֽוֹ׃
8יְהֹוָ֗ה אֶת־ לֵ֤ב פַּרְעֹה֙ מֶ֣לֶךְ מִצְרַ֔יִם וַיִּרְדֹּ֕ף אַחֲרֵ֖י בְּנֵ֣י יִשְׂרָאֵ֑ל וּבְנֵ֣י יִשְׂרָאֵ֔ל יֹצְאִ֖ים בְּיָ֥ד רָמָֽה׃
9מִצְרַ֜יִם אַחֲרֵיהֶ֗ם וַיַּשִּׂ֤יגוּ אוֹתָם֙ חֹנִ֣ים עַל־ הַיָּ֔ם כָּל־ סוּס֙ רֶ֣כֶב פַּרְעֹ֔ה וּפָרָשָׁ֖יו וְחֵיל֑וֹ עַל־ פִּי֙ הַֽחִירֹ֔ת לִפְנֵ֖י בַּ֥עַל צְפֹֽן׃
10הִקְרִ֑יב וַיִּשְׂאוּ֩ בְנֵֽי־ יִשְׂרָאֵ֨ל אֶת־ עֵינֵיהֶ֜ם וְהִנֵּ֥ה מִצְרַ֣יִם ׀ נֹסֵ֣עַ אַחֲרֵיהֶ֗ם וַיִּֽירְאוּ֙ מְאֹ֔ד וַיִּצְעֲק֥וּ בְנֵֽי־ יִשְׂרָאֵ֖ל אֶל־ יְהוָֽה׃
11אֶל־ מֹשֶׁה֒ הַֽמִבְּלִ֤י אֵין־ קְבָרִים֙ בְּמִצְרַ֔יִם לְקַחְתָּ֖נוּ לָמ֣וּת בַּמִּדְבָּ֑ר מַה־ זֹּאת֙ עָשִׂ֣יתָ לָּ֔נוּ לְהוֹצִיאָ֖נוּ מִמִּצְרָֽיִם׃
12זֶ֣ה הַדָּבָ֗ר אֲשֶׁר֩ דִּבַּ֨רְנוּ אֵלֶ֤יךָ בְמִצְרַ֙יִם֙ לֵאמֹ֔ר חֲדַ֥ל מִמֶּ֖נּוּ וְנַֽעַבְדָ֣ה אֶת־ מִצְרָ֑יִם כִּ֣י ט֥וֹב לָ֙נוּ֙ עֲבֹ֣ד אֶת־ מִצְרַ֔יִם מִמֻּתֵ֖נוּ בַּמִּדְבָּֽר׃
13מֹשֶׁ֣ה אֶל־ הָעָם֮ אַל־ תִּירָאוּ֒ הִֽתְיַצְב֗וּ וּרְאוּ֙ אֶת־ יְשׁוּעַ֣ת יְהוָ֔ה אֲשֶׁר־ יַעֲשֶׂ֥ה לָכֶ֖ם הַיּ֑וֹם כִּ֗י אֲשֶׁ֨ר רְאִיתֶ֤ם אֶת־ מִצְרַ֙יִם֙ הַיּ֔וֹם לֹ֥א תֹסִ֛יפוּ לִרְאֹתָ֥ם ע֖וֹד עַד־ עוֹלָֽם׃
14יִלָּחֵ֣ם לָכֶ֑ם וְאַתֶּ֖ם תַּחֲרִישֽׁוּן׃ פ
15יְהוָה֙ אֶל־ מֹשֶׁ֔ה מַה־ תִּצְעַ֖ק אֵלָ֑י דַּבֵּ֥ר אֶל־ בְּנֵי־ יִשְׂרָאֵ֖ל וְיִסָּֽעוּ׃
16הָרֵ֣ם אֶֽת־ מַטְּךָ֗ וּנְטֵ֧ה אֶת־ יָדְךָ֛ עַל־ הַיָּ֖ם וּבְקָעֵ֑הוּ וְיָבֹ֧אוּ בְנֵֽי־ יִשְׂרָאֵ֛ל בְּת֥וֹךְ הַיָּ֖ם בַּיַּבָּשָֽׁה׃
17הִנְנִ֤י מְחַזֵּק֙ אֶת־ לֵ֣ב מִצְרַ֔יִם וְיָבֹ֖אוּ אַחֲרֵיהֶ֑ם וְאִכָּבְדָ֤ה בְּפַרְעֹה֙ וּבְכָל־ חֵיל֔וֹ בְּרִכְבּ֖וֹ וּבְפָרָשָֽׁיו׃
18מִצְרַ֖יִם כִּי־ אֲנִ֣י יְהוָ֑ה בְּהִכָּבְדִ֣י בְּפַרְעֹ֔ה בְּרִכְבּ֖וֹ וּבְפָרָשָֽׁיו׃
19מַלְאַ֣ךְ הָאֱלֹהִ֗ים הַהֹלֵךְ֙ לִפְנֵי֙ מַחֲנֵ֣ה יִשְׂרָאֵ֔ל וַיֵּ֖לֶךְ מֵאַחֲרֵיהֶ֑ם וַיִּסַּ֞ע עַמּ֤וּד הֶֽעָנָן֙ מִפְּנֵיהֶ֔ם וַיַּֽעֲמֹ֖ד מֵאַחֲרֵיהֶֽם׃
20בֵּ֣ין ׀ מַחֲנֵ֣ה מִצְרַ֗יִם וּבֵין֙ מַחֲנֵ֣ה יִשְׂרָאֵ֔ל וַיְהִ֤י הֶֽעָנָן֙ וְהַחֹ֔שֶׁךְ וַיָּ֖אֶר אֶת־ הַלָּ֑יְלָה וְלֹא־ קָרַ֥ב זֶ֛ה אֶל־ זֶ֖ה כָּל־ הַלָּֽיְלָה׃
21מֹשֶׁ֣ה אֶת־ יָדוֹ֮ עַל־ הַיָּם֒ וַיּ֣וֹלֶךְ יְהוָ֣ה ׀ אֶת־ הַ֠יָּם בְּר֨וּחַ קָדִ֤ים עַזָּה֙ כָּל־ הַלַּ֔יְלָה וַיָּ֥שֶׂם אֶת־ הַיָּ֖ם לֶחָרָבָ֑ה וַיִּבָּקְע֖וּ הַמָּֽיִם׃
22בְנֵֽי־ יִשְׂרָאֵ֛ל בְּת֥וֹךְ הַיָּ֖ם בַּיַּבָּשָׁ֑ה וְהַמַּ֤יִם לָהֶם֙ חֹמָ֔ה מִֽימִינָ֖ם וּמִשְּׂמֹאלָֽם׃
23מִצְרַ֙יִם֙ וַיָּבֹ֣אוּ אַחֲרֵיהֶ֔ם כֹּ֚ל ס֣וּס פַּרְעֹ֔ה רִכְבּ֖וֹ וּפָרָשָׁ֑יו אֶל־ תּ֖וֹךְ הַיָּֽם׃
24בְּאַשְׁמֹ֣רֶת הַבֹּ֔קֶר וַיַּשְׁקֵ֤ף יְהוָה֙ אֶל־ מַחֲנֵ֣ה מִצְרַ֔יִם בְּעַמּ֥וּד אֵ֖שׁ וְעָנָ֑ן וַיָּ֕הָם אֵ֖ת מַחֲנֵ֥ה מִצְרָֽיִם׃
25אֵ֚ת אֹפַ֣ן מַרְכְּבֹתָ֔יו וַֽיְנַהֲגֵ֖הוּ בִּכְבֵדֻ֑ת וַיֹּ֣אמֶר מִצְרַ֗יִם אָנ֙וּסָה֙ מִפְּנֵ֣י יִשְׂרָאֵ֔ל כִּ֣י יְהוָ֔ה נִלְחָ֥ם לָהֶ֖ם בְּמִצְרָֽיִם פ
26יְהוָה֙ אֶל־ מֹשֶׁ֔ה נְטֵ֥ה אֶת־ יָדְךָ֖ עַל־ הַיָּ֑ם וְיָשֻׁ֤בוּ הַמַּ֙יִם֙ עַל־ מִצְרַ֔יִם עַל־ רִכְבּ֖וֹ וְעַל־ פָּרָשָֽׁיו׃
27מֹשֶׁ֨ה אֶת־ יָד֜וֹ עַל־ הַיָּ֗ם וַיָּ֨שָׁב הַיָּ֜ם לִפְנ֥וֹת בֹּ֙קֶר֙ לְאֵ֣יתָנ֔וֹ וּמִצְרַ֖יִם נָסִ֣ים לִקְרָאת֑וֹ וַיְנַעֵ֧ר יְהוָ֛ה אֶת־ מִצְרַ֖יִם בְּת֥וֹךְ הַיָּֽם׃
28הַמַּ֗יִם וַיְכַסּ֤וּ אֶת־ הָרֶ֙כֶב֙ וְאֶת־ הַפָּ֣רָשִׁ֔ים לְכֹל֙ חֵ֣יל פַּרְעֹ֔ה הַבָּאִ֥ים אַחֲרֵיהֶ֖ם בַּיָּ֑ם לֹֽא־ נִשְׁאַ֥ר בָּהֶ֖ם עַד־ אֶחָֽד׃
29יִשְׂרָאֵ֛ל הָלְכ֥וּ בַיַּבָּשָׁ֖ה בְּת֣וֹךְ הַיָּ֑ם וְהַמַּ֤יִם לָהֶם֙ חֹמָ֔ה מִֽימִינָ֖ם וּמִשְּׂמֹאלָֽם
30יְהוָ֜ה בַּיּ֥וֹם הַה֛וּא אֶת־ יִשְׂרָאֵ֖ל מִיַּ֣ד מִצְרָ֑יִם וַיַּ֤רְא יִשְׂרָאֵל֙ אֶת־ מִצְרַ֔יִם מֵ֖ת עַל־ שְׂפַ֥ת הַיָּֽם׃
