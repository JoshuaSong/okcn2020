1כָּרַ֥ע בֵּל֙ קֹרֵ֣ס נְב֔וֹ הָיוּ֙ עֲצַבֵּיהֶ֔ם לַחַיָּ֖ה וְלַבְּהֵמָ֑ה נְשֻׂאֹתֵיכֶ֣ם עֲמוּס֔וֹת מַשָּׂ֖א לַעֲיֵפָֽה׃
2כָֽרְעוּ֙ יַחְדָּ֔ו לֹ֥א יָכְל֖וּ מַלֵּ֣ט מַשָּׂ֑א וְנַפְשָׁ֖ם בַּשְּׁבִ֥י הָלָֽכָה׃ ס
3אֵלַי֙ בֵּ֣ית יַעֲקֹ֔ב וְכָל־ שְׁאֵרִ֖ית בֵּ֣ית יִשְׂרָאֵ֑ל הַֽעֲמֻסִים֙ מִנִּי־ בֶ֔טֶן הַנְּשֻׂאִ֖ים מִנִּי־ רָֽחַם׃
4זִקְנָה֙ אֲנִ֣י ה֔וּא וְעַד־ שֵיבָ֖ה אֲנִ֣י אֶסְבֹּ֑ל אֲנִ֤י עָשִׂ֙יתִי֙ וַאֲנִ֣י אֶשָּׂ֔א וַאֲנִ֥י אֶסְבֹּ֖ל וַאֲמַלֵּֽט׃ ס
5תְדַמְי֖וּנִי וְתַשְׁו֑וּ וְתַמְשִׁל֖וּנִי וְנִדְמֶֽה׃
6זָהָב֙ מִכִּ֔יס וְכֶ֖סֶף בַּקָּנֶ֣ה יִשְׁקֹ֑לוּ יִשְׂכְּר֤וּ צוֹרֵף֙ וְיַעֲשֵׂ֣הוּ אֵ֔ל יִסְגְּד֖וּ אַף־ יִֽשְׁתַּחֲוּֽוּ׃
7עַל־ כָּתֵ֨ף יִסְבְּלֻ֜הוּ וְיַנִּיחֻ֤הוּ תַחְתָּיו֙ וְיַֽעֲמֹ֔ד מִמְּקוֹמ֖וֹ לֹ֣א יָמִ֑ישׁ אַף־ יִצְעַ֤ק אֵלָיו֙ וְלֹ֣א יַעֲנֶ֔ה מִצָּרָת֖וֹ לֹ֥א יוֹשִׁיעֶֽנּוּ׃ ס
8זֹ֖את וְהִתְאֹשָׁ֑שׁוּ הָשִׁ֥יבוּ פוֹשְׁעִ֖ים עַל־ לֵֽב׃
9רִאשֹׁנ֖וֹת מֵעוֹלָ֑ם כִּ֣י אָנֹכִ֥י אֵל֙ וְאֵ֣ין ע֔וֹד אֱלֹהִ֖ים וְאֶ֥פֶס כָּמֽוֹנִי׃
10מֵֽרֵאשִׁית֙ אַחֲרִ֔ית וּמִקֶּ֖דֶם אֲשֶׁ֣ר לֹא־ נַעֲשׂ֑וּ אֹמֵר֙ עֲצָתִ֣י תָק֔וּם וְכָל־ חֶפְצִ֖י אֶעֱשֶֽׂה׃
11מִמִּזְרָח֙ עַ֔יִט מֵאֶ֥רֶץ מֶרְחָ֖ק אִ֣ישׁ [עֲצָתֹו כ] (עֲצָתִ֑י ק) אַף־ דִּבַּ֙רְתִּי֙ אַף־ אֲבִיאֶ֔נָּה יָצַ֖רְתִּי אַף־ אֶעֱשֶֽׂנָּה׃ ס
12אֵלַ֖י אַבִּ֣ירֵי לֵ֑ב הָרְחוֹקִ֖ים מִצְּדָקָֽה׃
