1לַ֭מְנַצֵּחַ עַלְמ֥וּת לַבֵּ֗ן מִזְמ֥וֹר לְדָוִֽד׃ אוֹדֶ֣ה יְ֭הוָה בְּכָל־ לִבִּ֑י אֲ֝סַפְּרָ֗ה כָּל־ נִפְלְאוֹתֶֽיךָ׃
2וְאֶעֶלְצָ֣ה בָ֑ךְ אֲזַמְּרָ֖ה שִׁמְךָ֣ עֶלְיֽוֹן׃
3אוֹיְבַ֥י אָח֑וֹר יִכָּשְׁל֥וּ וְ֝יֹאבְד֗וּ מִפָּנֶֽיךָ׃
4עָ֭שִׂיתָ מִשְׁפָּטִ֣י וְדִינִ֑י יָשַׁ֥בְתָּ לְ֝כִסֵּ֗א שׁוֹפֵ֥ט צֶֽדֶק׃
5ג֭וֹיִם אִבַּ֣דְתָּ רָשָׁ֑ע שְׁמָ֥ם מָ֝חִ֗יתָ לְעוֹלָ֥ם וָעֶֽד׃
6תַּ֥מּוּ חֳרָב֗וֹת לָ֫נֶ֥צַח וְעָרִ֥ים נָתַ֑שְׁתָּ אָבַ֖ד זִכְרָ֣ם הֵֽמָּה׃
7לְעוֹלָ֣ם יֵשֵׁ֑ב כּוֹנֵ֖ן לַמִּשְׁפָּ֣ט כִּסְאֽוֹ׃
8יִשְׁפֹּֽט־ תֵּבֵ֥ל בְּצֶ֑דֶק יָדִ֥ין לְ֝אֻמִּ֗ים בְּמֵישָׁרִֽים׃
9יְהוָ֣ה מִשְׂגָּ֣ב לַדָּ֑ךְ מִ֝שְׂגָּ֗ב לְעִתּ֥וֹת בַּצָּרָֽה׃
10בְ֭ךָ יוֹדְעֵ֣י שְׁמֶ֑ךָ כִּ֤י לֹֽא־ עָזַ֖בְתָּ דֹרְשֶׁ֣יךָ יְהוָֽה׃
11לַ֭יהוָה יֹשֵׁ֣ב צִיּ֑וֹן הַגִּ֥ידוּ בָ֝עַמִּ֗ים עֲלִֽילוֹתָֽיו׃
12דֹרֵ֣שׁ דָּ֭מִים אוֹתָ֣ם זָכָ֑ר לֹֽא־ שָׁ֝כַ֗ח צַעֲקַ֥ת [עֲנִיִּים כ] (עֲנָוִֽים׃ ק)
13יְהוָ֗ה רְאֵ֣ה עָ֭נְיִי מִשֹּׂנְאָ֑י מְ֝רוֹמְמִ֗י מִשַּׁ֥עֲרֵי מָֽוֶת׃
14אֲסַפְּרָ֗ה כָּֽל־ תְּהִלָּ֫תֶ֥יךָ בְּשַֽׁעֲרֵ֥י בַת־ צִיּ֑וֹן אָ֝גִ֗ילָה בִּישׁוּעָתֶֽךָ׃
15ג֭וֹיִם בְּשַׁ֣חַת עָשׂ֑וּ בְּרֶֽשֶׁת־ ז֥וּ טָ֝מָ֗נוּ נִלְכְּדָ֥ה רַגְלָֽם׃
16יְהוָה֮ מִשְׁפָּ֪ט עָ֫שָׂ֥ה בְּפֹ֣עַל כַּ֭פָּיו נוֹקֵ֣שׁ רָשָׁ֑ע הִגָּי֥וֹן סֶֽלָה׃
17רְשָׁעִ֣ים לִשְׁא֑וֹלָה כָּל־ גּ֝וֹיִ֗ם שְׁכֵחֵ֥י אֱלֹהִֽים׃
18לֹ֣א לָ֭נֶצַח יִשָּׁכַ֣ח אֶבְי֑וֹן תִּקְוַ֥ת [עֲנָוִים כ] (עֲ֝נִיִּ֗ים ק) תֹּאבַ֥ד לָעַֽד׃
19יְ֭הוָה אַל־ יָעֹ֣ז אֱנ֑וֹשׁ יִשָּׁפְט֥וּ ג֝וֹיִ֗ם עַל־ פָּנֶֽיךָ׃
