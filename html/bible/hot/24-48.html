1לְמוֹאָ֡ב כֹּֽה־ אָמַר֩ יְהוָ֨ה צְבָא֜וֹת אֱלֹהֵ֣י יִשְׂרָאֵ֗ל ה֤וֹי אֶל־ נְבוֹ֙ כִּ֣י שֻׁדָּ֔דָה הֹבִ֥ישָׁה נִלְכְּדָ֖ה קִרְיָתָ֑יִם הֹבִ֥ישָׁה הַמִּשְׂגָּ֖ב וָחָֽתָּה׃
2עוֹד֮ תְּהִלַּ֣ת מוֹאָב֒ בְּחֶשְׁבּ֗וֹן חָשְׁב֤וּ עָלֶ֙יהָ֙ רָעָ֔ה לְכ֖וּ וְנַכְרִיתֶ֣נָּה מִגּ֑וֹי גַּם־ מַדְמֵ֣ן תִּדֹּ֔מִּי אַחֲרַ֖יִךְ תֵּ֥לֶךְ חָֽרֶב׃
3צְעָקָ֖ה מֵחֹֽרוֹנָ֑יִם שֹׁ֖ד וָשֶׁ֥בֶר גָּדֽוֹל׃
4מוֹאָ֑ב הִשְׁמִ֥יעוּ זְּעָקָ֖ה [צְעֹורֶיהָ כ] (צְעִירֶֽיהָ׃ ק)
5מַעֲלֵ֣ה [הַלֻּחֹות כ] (הַלּוּחִ֔ית ק) בִּבְכִ֖י יַֽעֲלֶה־ בֶּ֑כִי כִּ֚י בְּמוֹרַ֣ד חוֹרֹנַ֔יִם צָרֵ֥י צַֽעֲקַת־ שֶׁ֖בֶר שָׁמֵֽעוּ׃
6מַלְּט֣וּ נַפְשְׁכֶ֑ם וְתִֽהְיֶ֕ינָה כַּעֲרוֹעֵ֖ר בַּמִּדְבָּֽר׃
7יַ֣עַן בִּטְחֵ֤ךְ בְּמַעֲשַׂ֙יִךְ֙ וּבְא֣וֹצְרוֹתַ֔יִךְ גַּם־ אַ֖תְּ תִּלָּכֵ֑דִי וְיָצָ֤א [כְמִישׁ כ] (כְמֹושׁ֙ ק) בַּגּוֹלָ֔ה כֹּהֲנָ֥יו וְשָׂרָ֖יו [יַחַד כ] (יַחְדָּֽיו׃ ק)
8שֹׁדֵ֜ד אֶל־ כָּל־ עִ֗יר וְעִיר֙ לֹ֣א תִמָּלֵ֔ט וְאָבַ֥ד הָעֵ֖מֶק וְנִשְׁמַ֣ד הַמִּישֹׁ֑ר אֲשֶׁ֖ר אָמַ֥ר יְהוָֽה׃
9צִ֣יץ לְמוֹאָ֔ב כִּ֥י נָצֹ֖א תֵּצֵ֑א וְעָרֶ֙יהָ֙ לְשַׁמָּ֣ה תִֽהְיֶ֔ינָה מֵאֵ֥ין יוֹשֵׁ֖ב בָּהֵֽן׃
10עֹשֶׂ֛ה מְלֶ֥אכֶת יְהוָ֖ה רְמִיָּ֑ה וְאָר֕וּר מֹנֵ֥עַ חַרְבּ֖וֹ מִדָּֽם׃
11מוֹאָ֜ב מִנְּעוּרָ֗יו וְשֹׁקֵ֥ט הוּא֙ אֶל־ שְׁמָרָ֔יו וְלֹֽא־ הוּרַ֤ק מִכְּלִי֙ אֶל־ כֶּ֔לִי וּבַגּוֹלָ֖ה לֹ֣א הָלָ֑ךְ עַל־ כֵּ֗ן עָמַ֤ד טַעְמוֹ֙ בּ֔וֹ וְרֵיח֖וֹ לֹ֥א נָמָֽר׃ ס
12הִנֵּ֖ה־ יָמִ֤ים בָּאִים֙ נְאֻם־ יְהוָ֔ה וְשִׁלַּחְתִּי־ ל֥וֹ צֹעִ֖ים וְצֵעֻ֑הוּ וְכֵלָ֣יו יָרִ֔יקוּ וְנִבְלֵיהֶ֖ם יְנַפֵּֽצוּ׃
13מוֹאָ֖ב מִכְּמ֑וֹשׁ כַּאֲשֶׁר־ בֹּ֙שׁוּ֙ בֵּ֣ית יִשְׂרָאֵ֔ל מִבֵּ֥ית אֵ֖ל מִבְטֶחָֽם׃
14תֹּֽאמְר֔וּ גִּבּוֹרִ֖ים אֲנָ֑חְנוּ וְאַנְשֵׁי־ חַ֖יִל לַמִּלְחָמָֽה׃
15מוֹאָב֙ וְעָרֶ֣יהָ עָלָ֔ה וּמִבְחַ֥ר בַּֽחוּרָ֖יו יָרְד֣וּ לַטָּ֑בַח נְאֻ֨ם־ הַמֶּ֔לֶךְ יְהוָ֥ה צְבָא֖וֹת שְׁמֽוֹ׃
16אֵיד־ מוֹאָ֖ב לָב֑וֹא וְרָ֣עָת֔וֹ מִהֲרָ֖ה מְאֹֽד׃
17לוֹ֙ כָּל־ סְבִיבָ֔יו וְכֹ֖ל יֹדְעֵ֣י שְׁמ֑וֹ אִמְר֗וּ אֵיכָ֤ה נִשְׁבַּר֙ מַטֵּה־ עֹ֔ז מַקֵּ֖ל תִּפְאָרָֽה׃
18מִכָּבוֹד֙ [יֹשְׁבֶי כ] (וּשְׁבִ֣י ק) בַצָּמָ֔א יֹשֶׁ֖בֶת בַּת־ דִּיב֑וֹן כִּֽי־ שֹׁדֵ֤ד מוֹאָב֙ עָ֣לָה בָ֔ךְ שִׁחֵ֖ת מִבְצָרָֽיִךְ׃
19דֶּ֛רֶךְ עִמְדִ֥י וְצַפִּ֖י יוֹשֶׁ֣בֶת עֲרוֹעֵ֑ר שַׁאֲלִי־ נָ֣ס וְנִמְלָ֔טָה אִמְרִ֖י מַה־ נִּֽהְיָֽתָה׃
20מוֹאָ֛ב כִּֽי־ חַ֖תָּה [הֵילִילִי כ] (הֵילִ֣ילוּ ק) [וּזְעָקִי כ] (וּֽזְעָ֑קוּ ק) הַגִּ֣ידוּ בְאַרְנ֔וֹן כִּ֥י שֻׁדַּ֖ד מוֹאָֽב׃
21בָּ֖א אֶל־ אֶ֣רֶץ הַמִּישֹׁ֑ר אֶל־ חֹל֥וֹן וְאֶל־ יַ֖הְצָה וְעַל־ [מֹופָעַת כ] (מֵיפָֽעַת׃ ק)
22דִּיב֣וֹן וְעַל־ נְב֔וֹ וְעַל־ בֵּ֖ית דִּבְלָתָֽיִם׃
23קִרְיָתַ֛יִם וְעַל־ בֵּ֥ית גָּמ֖וּל וְעַל־ בֵּ֥ית מְעֽוֹן׃
24קְרִיּ֖וֹת וְעַל־ בָּצְרָ֑ה וְעַ֗ל כָּל־ עָרֵי֙ אֶ֣רֶץ מוֹאָ֔ב הָרְחֹק֖וֹת וְהַקְּרֹבֽוֹת׃
25קֶ֣רֶן מוֹאָ֔ב וּזְרֹע֖וֹ נִשְׁבָּ֑רָה נְאֻ֖ם יְהוָֽה׃
26כִּ֥י עַל־ יְהוָ֖ה הִגְדִּ֑יל וְסָפַ֤ק מוֹאָב֙ בְּקִיא֔וֹ וְהָיָ֥ה לִשְׂחֹ֖ק גַּם־ הֽוּא׃
27ל֣וֹא הַשְּׂחֹ֗ק הָיָ֤ה לְךָ֙ יִשְׂרָאֵ֔ל אִם־ בְּגַנָּבִ֖ים [נִמְצָאָה כ] (נִמְצָ֑א ק) כִּֽי־ מִדֵּ֧י דְבָרֶ֥יךָ בּ֖וֹ תִּתְנוֹדָֽד׃
28עָרִים֙ וְשִׁכְנ֣וּ בַּסֶּ֔לַע יֹשְׁבֵ֖י מוֹאָ֑ב וִֽהְי֣וּ כְיוֹנָ֔ה תְּקַנֵּ֖ן בְּעֶבְרֵ֥י פִי־ פָֽחַת׃
29גְאוֹן־ מוֹאָ֖ב גֵּאֶ֣ה מְאֹ֑ד גָּבְה֧וֹ וּגְאוֹנ֛וֹ וְגַאֲוָת֖וֹ וְרֻ֥ם לִבּֽוֹ׃
30יָדַ֙עְתִּי֙ נְאֻם־ יְהוָ֔ה עֶבְרָת֖וֹ וְלֹא־ כֵ֑ן בַּדָּ֖יו לֹא־ כֵ֥ן עָשֽׂוּ׃
31כֵּן֙ עַל־ מוֹאָ֣ב אֲיֵלִ֔יל וּלְמוֹאָ֥ב כֻּלֹּ֖ה אֶזְעָ֑ק אֶל־ אַנְשֵׁ֥י קִֽיר־ חֶ֖רֶשׂ יֶהְגֶּֽה׃
32יַעְזֵ֤ר אֶבְכֶּה־ לָּךְ֙ הַגֶּ֣פֶן שִׂבְמָ֔ה נְטִֽישֹׁתַ֙יִךְ֙ עָ֣בְרוּ יָ֔ם עַ֛ד יָ֥ם יַעְזֵ֖ר נָגָ֑עוּ עַל־ קֵיצֵ֥ךְ וְעַל־ בְּצִירֵ֖ךְ שֹׁדֵ֥ד נָפָֽל׃
33שִׂמְחָ֥ה וָגִ֛יל מִכַּרְמֶ֖ל וּמֵאֶ֣רֶץ מוֹאָ֑ב וְיַ֙יִן֙ מִיקָבִ֣ים הִשְׁבַּ֔תִּי לֹֽא־ יִדְרֹ֣ךְ הֵידָ֔ד הֵידָ֖ד לֹ֥א הֵידָֽד׃
34חֶשְׁבּ֜וֹן עַד־ אֶלְעָלֵ֗ה עַד־ יַ֙הַץ֙ נָתְנ֣וּ קוֹלָ֔ם מִצֹּ֙עַר֙ עַד־ חֹ֣רֹנַ֔יִם עֶגְלַ֖ת שְׁלִֽשִׁיָּ֑ה כִּ֚י גַּם־ מֵ֣י נִמְרִ֔ים לִמְשַׁמּ֖וֹת יִהְיֽוּ׃
35לְמוֹאָ֖ב נְאֻם־ יְהוָ֑ה מַעֲלֶ֣ה בָמָ֔ה וּמַקְטִ֖יר לֵאלֹהָֽיו׃
36כֵּ֞ן לִבִּ֤י לְמוֹאָב֙ כַּחֲלִלִ֣ים יֶהֱמֶ֔ה וְלִבִּי֙ אֶל־ אַנְשֵׁ֣י קִֽיר־ חֶ֔רֶשׂ כַּחֲלִילִ֖ים יֶהֱמֶ֑ה עַל־ כֵּ֛ן יִתְרַ֥ת עָשָׂ֖ה אָבָֽדוּ׃
37כָל־ רֹאשׁ֙ קָרְחָ֔ה וְכָל־ זָקָ֖ן גְּרֻעָ֑ה עַ֤ל כָּל־ יָדַ֙יִם֙ גְּדֻדֹ֔ת וְעַל־ מָתְנַ֖יִם שָֽׂק׃
38כָּל־ גַּגּ֥וֹת מוֹאָ֛ב וּבִרְחֹבֹתֶ֖יהָ כֻּלֹּ֣ה מִסְפֵּ֑ד כִּֽי־ שָׁבַ֣רְתִּי אֶת־ מוֹאָ֗ב כִּכְלִ֛י אֵֽין־ חֵ֥פֶץ בּ֖וֹ נְאֻם־ יְהוָֽה׃
39חַ֙תָּה֙ הֵילִ֔ילוּ אֵ֛יךְ הִפְנָה־ עֹ֥רֶף מוֹאָ֖ב בּ֑וֹשׁ וְהָיָ֥ה מוֹאָ֛ב לִשְׂחֹ֥ק וְלִמְחִתָּ֖ה לְכָל־ סְבִיבָֽיו׃ ס
40כֹה֙ אָמַ֣ר יְהוָ֔ה הִנֵּ֥ה כַנֶּ֖שֶׁר יִדְאֶ֑ה וּפָרַ֥שׂ כְּנָפָ֖יו אֶל־ מוֹאָֽב׃
41הַקְּרִיּ֔וֹת וְהַמְּצָד֖וֹת נִתְפָּ֑שָׂה וְֽ֠הָיָה לֵ֞ב גִּבּוֹרֵ֤י מוֹאָב֙ בַּיּ֣וֹם הַה֔וּא כְּלֵ֖ב אִשָּׁ֥ה מְצֵרָֽה׃
42מוֹאָ֖ב מֵעָ֑ם כִּ֥י עַל־ יְהוָ֖ה הִגְדִּֽיל׃
43וָפַ֖חַת וָפָ֑ח עָלֶ֛יךָ יוֹשֵׁ֥ב מוֹאָ֖ב נְאֻם־ יְהוָֽה׃
44כ] (הַנָּ֞ס ק) מִפְּנֵ֤י הַפַּ֙חַד֙ יִפֹּ֣ל אֶל־ הַפַּ֔חַת וְהָֽעֹלֶה֙ מִן־ הַפַּ֔חַת יִלָּכֵ֖ד בַּפָּ֑ח כִּֽי־ אָבִ֨יא אֵלֶ֧יהָ אֶל־ מוֹאָ֛ב שְׁנַ֥ת פְּקֻדָּתָ֖ם נְאֻם־ יְהוָֽה׃
45חֶשְׁבּ֛וֹן עָמְד֖וּ מִכֹּ֣חַ נָסִ֑ים כִּֽי־ אֵ֞שׁ יָצָ֣א מֵחֶשְׁבּ֗וֹן וְלֶֽהָבָה֙ מִבֵּ֣ין סִיח֔וֹן וַתֹּ֙אכַל֙ פְּאַ֣ת מוֹאָ֔ב וְקָדְקֹ֖ד בְּנֵ֥י שָׁאֽוֹן׃
46לְךָ֣ מוֹאָ֔ב אָבַ֖ד עַם־ כְּמ֑וֹשׁ כִּֽי־ לֻקְּח֤וּ בָנֶ֙יךָ֙ בַּשֶּׁ֔בִי וּבְנֹתֶ֖יךָ בַּשִּׁבְיָֽה׃
