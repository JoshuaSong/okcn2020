1לַמְנַצֵּ֣חַ עַל־ שֹׁ֭שַׁנִּים לִבְנֵי־ קֹ֑רַח מַ֝שְׂכִּ֗יל שִׁ֣יר יְדִידֹֽת׃ רָ֘חַ֤שׁ לִבִּ֨י ׀ דָּ֘בָ֤ר ט֗וֹב אֹמֵ֣ר אָ֭נִי מַעֲשַׂ֣י לְמֶ֑לֶךְ לְ֝שׁוֹנִ֗י עֵ֤ט ׀ סוֹפֵ֬ר מָהִֽיר׃
2מִבְּנֵ֬י אָדָ֗ם ה֣וּצַק חֵ֭ן בְּשְׂפְתוֹתֶ֑יךָ עַל־ כֵּ֤ן בֵּֽרַכְךָ֖ אֱלֹהִ֣ים לְעוֹלָֽם׃
3חַרְבְּךָ֣ עַל־ יָרֵ֣ךְ גִּבּ֑וֹר ה֝וֹדְךָ֗ וַהֲדָרֶֽךָ׃
4צְלַ֬ח רְכַ֗ב עַֽל־ דְּבַר־ אֱ֭מֶת וְעַנְוָה־ צֶ֑דֶק וְתוֹרְךָ֖ נוֹרָא֣וֹת יְמִינֶֽךָ׃
5שְׁנ֫וּנִ֥ים עַ֭מִּים תַּחְתֶּ֣יךָ יִפְּל֑וּ בְּ֝לֵ֗ב אוֹיְבֵ֥י הַמֶּֽלֶךְ׃
6אֱ֭לֹהִים עוֹלָ֣ם וָעֶ֑ד שֵׁ֥בֶט מִ֝ישֹׁ֗ר שֵׁ֣בֶט מַלְכוּתֶֽךָ׃
7צֶּדֶק֮ וַתִּשְׂנָ֫א רֶ֥שַׁע עַל־ כֵּ֤ן ׀ מְשָׁחֲךָ֡ אֱלֹהִ֣ים אֱ֭לֹהֶיךָ שֶׁ֥מֶן שָׂשׂ֗וֹן מֵֽחֲבֵרֶֽיךָ׃
8וַאֲהָל֣וֹת קְ֭צִיעוֹת כָּל־ בִּגְדֹתֶ֑יךָ מִֽן־ הֵ֥יכְלֵי שֵׁ֝֗ן מִנִּ֥י שִׂמְּחֽוּךָ׃
9מְ֭לָכִים בְּיִקְּרוֹתֶ֑יךָ נִצְּבָ֥ה שֵׁגַ֥ל לִֽ֝ימִינְךָ֗ בְּכֶ֣תֶם אוֹפִֽיר׃
10בַ֣ת וּ֭רְאִי וְהַטִּ֣י אָזְנֵ֑ךְ וְשִׁכְחִ֥י עַ֝מֵּ֗ךְ וּבֵ֥ית אָבִֽיךְ׃
11הַמֶּ֣לֶךְ יָפְיֵ֑ךְ כִּי־ ה֥וּא אֲ֝דֹנַ֗יִךְ וְהִשְׁתַּֽחֲוִי־ לֽוֹ׃
12צֹ֨ר ׀ בְּ֭מִנְחָה פָּנַ֥יִךְ יְחַלּ֗וּ עֲשִׁ֣ירֵי עָֽם׃
13כְּבוּדָּ֣ה בַת־ מֶ֣לֶךְ פְּנִ֑ימָה מִֽמִּשְׁבְּצ֖וֹת זָהָ֣ב לְבוּשָֽׁהּ׃
14תּוּבַ֪ל לַ֫מֶּ֥לֶךְ בְּתוּל֣וֹת אַ֭חֲרֶיהָ רֵעוֹתֶ֑יהָ מ֖וּבָא֣וֹת לָֽךְ׃
15בִּשְׂמָחֹ֣ת וָגִ֑יל תְּ֝בֹאֶ֗ינָה בְּהֵ֣יכַל מֶֽלֶךְ׃
16אֲ֭בֹתֶיךָ יִהְי֣וּ בָנֶ֑יךָ תְּשִׁיתֵ֥מוֹ לְ֝שָׂרִ֗ים בְּכָל־ הָאָֽרֶץ׃
