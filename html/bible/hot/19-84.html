1לַמְנַצֵּ֥חַ עַֽל־ הַגִּתִּ֑ית לִבְנֵי־ קֹ֥רַח מִזְמֽוֹר׃ מַה־ יְּדִיד֥וֹת מִשְׁכְּנוֹתֶ֗יךָ יְהוָ֥ה צְבָאֽוֹת׃
2וְגַם־ כָּלְתָ֨ה ׀ נַפְשִׁי֮ לְחַצְר֪וֹת יְה֫וָ֥ה לִבִּ֥י וּבְשָׂרִ֑י יְ֝רַנְּנ֗וּ אֶ֣ל אֵֽל־ חָֽי׃
3צִפּ֨וֹר ׀ מָ֪צְאָה בַ֡יִת וּדְר֤וֹר ׀ קֵ֥ן לָהּ֮ אֲשֶׁר־ שָׁ֪תָה אֶפְרֹ֫חֶ֥יהָ אֶֽת־ מִ֭זְבְּחוֹתֶיךָ יְהוָ֣ה צְבָא֑וֹת מַ֝לְכִּ֗י וֵאלֹהָֽי׃
4יוֹשְׁבֵ֣י בֵיתֶ֑ךָ ע֝֗וֹד יְֽהַלְל֥וּךָ סֶּֽלָה׃
5אָ֭דָם עֽוֹז־ ל֥וֹ בָ֑ךְ מְ֝סִלּ֗וֹת בִּלְבָבָֽם׃
6בְּעֵ֣מֶק הַ֭בָּכָא מַעְיָ֣ן יְשִׁית֑וּהוּ גַּם־ בְּ֝רָכ֗וֹת יַעְטֶ֥ה מוֹרֶֽה׃
7מֵחַ֣יִל אֶל־ חָ֑יִל יֵרָאֶ֖ה אֶל־ אֱלֹהִ֣ים בְּצִיּֽוֹן׃
8אֱלֹהִ֣ים צְ֭בָאוֹת שִׁמְעָ֣ה תְפִלָּתִ֑י הַאֲזִ֨ינָה אֱלֹהֵ֖י יַעֲקֹ֣ב סֶֽלָה׃
9רְאֵ֣ה אֱלֹהִ֑ים וְ֝הַבֵּ֗ט פְּנֵ֣י מְשִׁיחֶֽךָ׃
10טֽוֹב־ י֥וֹם בַּחֲצֵרֶ֗יךָ מֵ֫אָ֥לֶף בָּחַ֗רְתִּי הִ֭סְתּוֹפֵף בְּבֵ֣ית אֱלֹהַ֑י מִ֝דּ֗וּר בְּאָהֳלֵי־ רֶֽשַׁע׃
11שֶׁ֨מֶשׁ ׀ וּמָגֵן֮ יְהוָ֪ה אֱלֹ֫הִ֥ים חֵ֣ן וְ֭כָבוֹד יִתֵּ֣ן יְהוָ֑ה לֹ֥א יִמְנַע־ ט֝֗וֹב לַֽהֹלְכִ֥ים בְּתָמִֽים׃
