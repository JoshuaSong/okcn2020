1הַצַּדִּ֣יק אָבָ֔ד וְאֵ֥ין אִ֖ישׁ שָׂ֣ם עַל־ לֵ֑ב וְאַנְשֵׁי־ חֶ֤סֶד נֶֽאֱסָפִים֙ בְּאֵ֣ין מֵבִ֔ין כִּֽי־ מִפְּנֵ֥י הָרָעָ֖ה נֶאֱסַ֥ף הַצַּדִּֽיק׃
2שָׁל֔וֹם יָנ֖וּחוּ עַל־ מִשְׁכְּבוֹתָ֑ם הֹלֵ֖ךְ נְכֹחֽוֹ׃
3קִרְבוּ־ הֵ֖נָּה בְּנֵ֣י עֹנְנָ֑ה זֶ֥רַע מְנָאֵ֖ף וַתִּזְנֶֽה׃
4מִי֙ תִּתְעַנָּ֔גוּ עַל־ מִ֛י תַּרְחִ֥יבוּ פֶ֖ה תַּאֲרִ֣יכוּ לָשׁ֑וֹן הֲלֽוֹא־ אַתֶּ֥ם יִלְדֵי־ פֶ֖שַׁע זֶ֥רַע שָֽׁקֶר׃
5בָּֽאֵלִ֔ים תַּ֖חַת כָּל־ עֵ֣ץ רַעֲנָ֑ן שֹׁחֲטֵ֤י הַיְלָדִים֙ בַּנְּחָלִ֔ים תַּ֖חַת סְעִפֵ֥י הַסְּלָעִֽים׃
6נַ֣חַל חֶלְקֵ֔ךְ הֵ֥ם הֵ֖ם גּוֹרָלֵ֑ךְ גַּם־ לָהֶ֞ם שָׁפַ֥כְתְּ נֶ֙סֶךְ֙ הֶעֱלִ֣ית מִנְחָ֔ה הַ֥עַל אֵ֖לֶּה אֶנָּחֵֽם׃
7הַר־ גָּבֹ֙הַּ֙ וְנִשָּׂ֔א שַׂ֖מְתְּ מִשְׁכָּבֵ֑ךְ גַּם־ שָׁ֥ם עָלִ֖ית לִזְבֹּ֥חַ זָֽבַח׃
8הַדֶּ֙לֶת֙ וְהַמְּזוּזָ֔ה שַׂ֖מְתְּ זִכְרוֹנֵ֑ךְ כִּ֣י מֵאִתִּ֞י גִּלִּ֣ית וַֽתַּעֲלִ֗י הִרְחַ֤בְתְּ מִשְׁכָּבֵךְ֙ וַתִּכְרָת־ לָ֣ךְ מֵהֶ֔ם אָהַ֥בְתְּ מִשְׁכָּבָ֖ם יָ֥ד חָזִֽית׃
9לַמֶּ֙לֶךְ֙ בַּשֶּׁ֔מֶן וַתַּרְבִּ֖י רִקֻּחָ֑יִךְ וַתְּשַׁלְּחִ֤י צִרַ֙יִךְ֙ עַד־ מֵ֣רָחֹ֔ק וַתַּשְׁפִּ֖ילִי עַד־ שְׁאֽוֹל׃
10דַּרְכֵּךְ֙ יָגַ֔עַתְּ לֹ֥א אָמַ֖רְתְּ נוֹאָ֑שׁ חַיַּ֤ת יָדֵךְ֙ מָצָ֔את עַל־ כֵּ֖ן לֹ֥א חָלִֽית׃
11מִ֞י דָּאַ֤גְתְּ וַתִּֽירְאִי֙ כִּ֣י תְכַזֵּ֔בִי וְאוֹתִי֙ לֹ֣א זָכַ֔רְתְּ לֹא־ שַׂ֖מְתְּ עַל־ לִבֵּ֑ךְ הֲלֹ֨א אֲנִ֤י מַחְשֶׁה֙ וּמֵ֣עֹלָ֔ם וְאוֹתִ֖י לֹ֥א תִירָֽאִי׃
12אַגִּ֖יד צִדְקָתֵ֑ךְ וְאֶֽת־ מַעֲשַׂ֖יִךְ וְלֹ֥א יוֹעִילֽוּךְ׃
13יַצִּילֻ֣ךְ קִבּוּצַ֔יִךְ וְאֶת־ כֻּלָּ֥ם יִשָּׂא־ ר֖וּחַ יִקַּח־ הָ֑בֶל וְהַחוֹסֶ֥ה בִי֙ יִנְחַל־ אֶ֔רֶץ וְיִירַ֖שׁ הַר־ קָדְשִֽׁי׃
14סֹֽלּוּ־ סֹ֖לּוּ פַּנּוּ־ דָ֑רֶךְ הָרִ֥ימוּ מִכְשׁ֖וֹל מִדֶּ֥רֶךְ עַמִּֽי׃ ס
15כֹ֨ה אָמַ֜ר רָ֣ם וְנִשָּׂ֗א שֹׁכֵ֥ן עַד֙ וְקָד֣וֹשׁ שְׁמ֔וֹ מָר֥וֹם וְקָד֖וֹשׁ אֶשְׁכּ֑וֹן וְאֶת־ דַּכָּא֙ וּשְׁפַל־ ר֔וּחַ לְהַחֲיוֹת֙ ר֣וּחַ שְׁפָלִ֔ים וּֽלְהַחֲי֖וֹת לֵ֥ב נִדְכָּאִֽים׃
16לֹ֤א לְעוֹלָם֙ אָרִ֔יב וְלֹ֥א לָנֶ֖צַח אֶקְּצ֑וֹף כִּי־ ר֙וּחַ֙ מִלְּפָנַ֣י יַֽעֲט֔וֹף וּנְשָׁמ֖וֹת אֲנִ֥י עָשִֽׂיתִי׃
17בִּצְע֛וֹ קָצַ֥פְתִּי וְאַכֵּ֖הוּ הַסְתֵּ֣ר וְאֶקְצֹ֑ף וַיֵּ֥לֶךְ שׁוֹבָ֖ב בְּדֶ֥רֶךְ לִבּֽוֹ׃
18רָאִ֖יתִי וְאֶרְפָּאֵ֑הוּ וְאַנְחֵ֕הוּ וַאֲשַׁלֵּ֧ם נִֽחֻמִ֛ים ל֖וֹ וְלַאֲבֵלָֽיו׃
19[נוּב כ] (נִ֣יב ק) שְׂפָתָ֑יִם שָׁל֨וֹם ׀ שָׁל֜וֹם לָרָח֧וֹק וְלַקָּר֛וֹב אָמַ֥ר יְהוָ֖ה וּרְפָאתִֽיו׃
20כַּיָּ֣ם נִגְרָ֑שׁ כִּ֤י הַשְׁקֵט֙ לֹ֣א יוּכָ֔ל וַיִּגְרְשׁ֥וּ מֵימָ֖יו רֶ֥פֶשׁ וָטִֽיט׃
