1הֲיָדַ֗עְתָּ עֵ֭ת לֶ֣דֶת יַעֲלֵי־ סָ֑לַע חֹלֵ֖ל אַיָּל֣וֹת תִּשְׁמֹֽר׃
2יְרָחִ֣ים תְּמַלֶּ֑אנָה וְ֝יָדַ֗עְתָּ עֵ֣ת לִדְתָּֽנָה׃
3יַלְדֵיהֶ֣ן תְּפַלַּ֑חְנָה חֶבְלֵיהֶ֥ם תְּשַׁלַּֽחְנָה׃
4בְ֭נֵיהֶם יִרְבּ֣וּ בַבָּ֑ר יָ֝צְא֗וּ וְלֹא־ שָׁ֥בוּ לָֽמוֹ׃
5שִׁלַּ֣ח פֶּ֣רֶא חָפְשִׁ֑י וּמֹסְר֥וֹת עָ֝ר֗וֹד מִ֣י פִתֵּֽחַ׃
6שַׂ֣מְתִּי עֲרָבָ֣ה בֵית֑וֹ וּֽמִשְׁכְּנוֹתָ֥יו מְלֵֽחָה׃
7לַהֲמ֣וֹן קִרְיָ֑ה תְּשֻׁא֥וֹת נ֝וֹגֵ֗שׂ לֹ֣א יִשְׁמָֽע׃
8הָרִ֣ים מִרְעֵ֑הוּ וְאַחַ֖ר כָּל־ יָר֣וֹק יִדְרֽוֹשׁ׃
9רֵּ֣ים עָבְדֶ֑ךָ אִם־ יָ֝לִ֗ין עַל־ אֲבוּסֶֽךָ׃
10רֵ֭ים בְּתֶ֣לֶם עֲבֹת֑וֹ אִם־ יְשַׂדֵּ֖ד עֲמָקִ֣ים אַחֲרֶֽיךָ׃
11בּ֭וֹ כִּי־ רַ֣ב כֹּח֑וֹ וְתַעֲזֹ֖ב אֵלָ֣יו יְגִיעֶֽךָ׃
12בּ֭וֹ כִּי־ [יָשׁוּב כ] (יָשִׁ֣יב ק) זַרְעֶ֑ךָ וְֽגָרְנְךָ֥ יֶאֱסֹֽף׃
13רְנָנִ֥ים נֶעֱלָ֑סָה אִם־ אֶ֝בְרָ֗ה חֲסִידָ֥ה וְנֹצָֽה׃
14תַעֲזֹ֣ב לָאָ֣רֶץ בֵּצֶ֑יהָ וְֽעַל־ עָפָ֥ר תְּחַמֵּֽם׃
15כִּי־ רֶ֣גֶל תְּזוּרֶ֑הָ וְחַיַּ֖ת הַשָּׂדֶ֣ה תְּדוּשֶֽׁהָ׃
16בָּנֶ֣יהָ לְּלֹא־ לָ֑הּ לְרִ֖יק יְגִיעָ֣הּ בְּלִי־ פָֽחַד׃
17הִשָּׁ֣הּ אֱל֣וֹהַּ חָכְמָ֑ה וְלֹא־ חָ֥לַק לָ֝֗הּ בַּבִּינָֽה׃
18בַּמָּר֣וֹם תַּמְרִ֑יא תִּֽשְׂחַ֥ק לַ֝סּ֗וּס וּלְרֹֽכְבֽוֹ׃
19לַסּ֣וּס גְּבוּרָ֑ה הֲתַלְבִּ֖ישׁ צַוָּאר֣וֹ רַעְמָֽה׃
20כָּאַרְבֶּ֑ה ה֖וֹד נַחְר֣וֹ אֵימָֽה׃
21בָ֭עֵמֶק וְיָשִׂ֣ישׂ בְּכֹ֑חַ יֵ֝צֵ֗א לִקְרַאת־ נָֽשֶׁק׃
22לְ֭פַחַד וְלֹ֣א יֵחָ֑ת וְלֹֽא־ יָ֝שׁ֗וּב מִפְּנֵי־ חָֽרֶב׃
23תִּרְנֶ֣ה אַשְׁפָּ֑ה לַ֖הַב חֲנִ֣ית וְכִידֽוֹן׃
24וְ֭רֹגֶז יְגַמֶּא־ אָ֑רֶץ וְלֹֽא־ יַ֝אֲמִ֗ין כִּי־ ק֥וֹל שׁוֹפָֽר׃
25שֹׁפָ֨ר ׀ יֹ֘אמַ֤ר הֶאָ֗ח וּֽ֭מֵרָחוֹק יָרִ֣יחַ מִלְחָמָ֑ה רַ֥עַם שָׂ֝רִים וּתְרוּעָֽה׃
26יַֽאֲבֶר־ נֵ֑ץ יִפְרֹ֖שׂ [כְּנָפֹו כ] (כְּנָפָ֣יו ק) לְתֵימָֽן׃ אִם־ עַל־
27יַגְבִּ֣יהַּ נָ֑שֶׁר וְ֝כִ֗י יָרִ֥ים קִנּֽוֹ׃
28יִ֭שְׁכֹּן וְיִתְלֹנָ֑ן עַֽל־ שֶׁן־ סֶ֝֗לַע וּמְצוּדָֽה׃
29חָֽפַר־ אֹ֑כֶל לְ֝מֵרָח֗וֹק עֵינָ֥יו יַבִּֽיטוּ׃
