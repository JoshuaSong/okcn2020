1וַיְהִ֗י בִּימֵי֙ אַמְרָפֶ֣ל מֶֽלֶךְ־ שִׁנְעָ֔ר אַרְי֖וֹךְ מֶ֣לֶךְ אֶלָּסָ֑ר כְּדָרְלָעֹ֙מֶר֙ מֶ֣לֶךְ עֵילָ֔ם וְתִדְעָ֖ל מֶ֥לֶךְ גּוֹיִֽם׃
2מִלְחָמָ֗ה אֶת־ בֶּ֙רַע֙ מֶ֣לֶךְ סְדֹ֔ם וְאֶת־ בִּרְשַׁ֖ע מֶ֣לֶךְ עֲמֹרָ֑ה שִׁנְאָ֣ב ׀ מֶ֣לֶךְ אַדְמָ֗ה וְשֶׁמְאֵ֙בֶר֙ מֶ֣לֶךְ [צְבֹיִים כ] (צְבֹויִ֔ים ק) וּמֶ֥לֶךְ בֶּ֖לַע הִיא־ צֹֽעַר׃
3אֵ֙לֶּה֙ חָֽבְר֔וּ אֶל־ עֵ֖מֶק הַשִּׂדִּ֑ים ה֖וּא יָ֥ם הַמֶּֽלַח׃
4עֶשְׂרֵה֙ שָׁנָ֔ה עָבְד֖וּ אֶת־ כְּדָרְלָעֹ֑מֶר וּשְׁלֹשׁ־ עֶשְׂרֵ֥ה שָׁנָ֖ה מָרָֽדוּ׃
5עֶשְׂרֵ֨ה שָׁנָ֜ה בָּ֣א כְדָרְלָעֹ֗מֶר וְהַמְּלָכִים֙ אֲשֶׁ֣ר אִתּ֔וֹ וַיַּכּ֤וּ אֶת־ רְפָאִים֙ בְּעַשְׁתְּרֹ֣ת קַרְנַ֔יִם וְאֶת־ הַזּוּזִ֖ים בְּהָ֑ם וְאֵת֙ הָֽאֵימִ֔ים בְּשָׁוֵ֖ה קִרְיָתָֽיִם׃
6הַחֹרִ֖י בְּהַרְרָ֣ם שֵׂעִ֑יר עַ֚ד אֵ֣יל פָּארָ֔ן אֲשֶׁ֖ר עַל־ הַמִּדְבָּֽר׃
7וַיָּבֹ֜אוּ אֶל־ עֵ֤ין מִשְׁפָּט֙ הִ֣וא קָדֵ֔שׁ וַיַּכּ֕וּ אֶֽת־ כָּל־ שְׂדֵ֖ה הָעֲמָלֵקִ֑י וְגַם֙ אֶת־ הָ֣אֱמֹרִ֔י הַיֹּשֵׁ֖ב בְּחַֽצְצֹ֥ן תָּמָֽר׃
8מֶֽלֶךְ־ סְדֹ֜ם וּמֶ֣לֶךְ עֲמֹרָ֗ה וּמֶ֤לֶךְ אַדְמָה֙ וּמֶ֣לֶךְ [צְבֹיִים כ] (צְבֹויִ֔ם ק) וּמֶ֥לֶךְ בֶּ֖לַע הִוא־ צֹ֑עַר וַיַּֽעַרְכ֤וּ אִתָּם֙ מִלְחָמָ֔ה בְּעֵ֖מֶק הַשִּׂדִּֽים׃
9כְּדָרְלָעֹ֜מֶר מֶ֣לֶךְ עֵילָ֗ם וְתִדְעָל֙ מֶ֣לֶךְ גּוֹיִ֔ם וְאַמְרָפֶל֙ מֶ֣לֶךְ שִׁנְעָ֔ר וְאַרְי֖וֹךְ מֶ֣לֶךְ אֶלָּסָ֑ר אַרְבָּעָ֥ה מְלָכִ֖ים אֶת־ הַחֲמִשָּֽׁה׃
10הַשִׂדִּ֗ים בֶּֽאֱרֹ֤ת בֶּאֱרֹת֙ חֵמָ֔ר וַיָּנֻ֛סוּ מֶֽלֶךְ־ סְדֹ֥ם וַעֲמֹרָ֖ה וַיִּפְּלוּ־ שָׁ֑מָּה וְהַנִּשְׁאָרִ֖ים הֶ֥רָה נָּֽסוּ׃
11אֶת־ כָּל־ רְכֻ֨שׁ סְדֹ֧ם וַעֲמֹרָ֛ה וְאֶת־ כָּל־ אָכְלָ֖ם וַיֵּלֵֽכוּ׃
12אֶת־ ל֧וֹט וְאֶת־ רְכֻשׁ֛וֹ בֶּן־ אֲחִ֥י אַבְרָ֖ם וַיֵּלֵ֑כוּ וְה֥וּא יֹשֵׁ֖ב בִּסְדֹֽם׃
13הַפָּלִ֔יט וַיַּגֵּ֖ד לְאַבְרָ֣ם הָעִבְרִ֑י וְהוּא֩ שֹׁכֵ֨ן בְּאֵֽלֹנֵ֜י מַמְרֵ֣א הָאֱמֹרִ֗י אֲחִ֤י אֶשְׁכֹּל֙ וַאֲחִ֣י עָנֵ֔ר וְהֵ֖ם בַּעֲלֵ֥י בְרִית־ אַבְרָֽם׃
14אַבְרָ֔ם כִּ֥י נִשְׁבָּ֖ה אָחִ֑יו וַיָּ֨רֶק אֶת־ חֲנִיכָ֜יו יְלִידֵ֣י בֵית֗וֹ שְׁמֹנָ֤ה עָשָׂר֙ וּשְׁלֹ֣שׁ מֵא֔וֹת וַיִּרְדֹּ֖ף עַד־ דָּֽן׃
15עֲלֵיהֶ֧ם ׀ לַ֛יְלָה ה֥וּא וַעֲבָדָ֖יו וַיַּכֵּ֑ם וַֽיִּרְדְּפֵם֙ עַד־ חוֹבָ֔ה אֲשֶׁ֥ר מִשְּׂמֹ֖אל לְדַמָּֽשֶׂק׃
16אֵ֖ת כָּל־ הָרְכֻ֑שׁ וְגַם֩ אֶת־ ל֨וֹט אָחִ֤יו וּרְכֻשׁוֹ֙ הֵשִׁ֔יב וְגַ֥ם אֶת־ הַנָּשִׁ֖ים וְאֶת־ הָעָֽם׃
17מֶֽלֶךְ־ סְדֹם֮ לִקְרָאתוֹ֒ אַחֲרֵ֣י שׁוּב֗וֹ מֵֽהַכּוֹת֙ אֶת־ כְּדָר־ לָעֹ֔מֶר וְאֶת־ הַמְּלָכִ֖ים אֲשֶׁ֣ר אִתּ֑וֹ אֶל־ עֵ֣מֶק שָׁוֵ֔ה ה֖וּא עֵ֥מֶק הַמֶּֽלֶךְ׃
18צֶ֙דֶק֙ מֶ֣לֶךְ שָׁלֵ֔ם הוֹצִ֖יא לֶ֣חֶם וָיָ֑יִן וְה֥וּא כֹהֵ֖ן לְאֵ֥ל עֶלְיֽוֹן׃
19וַיֹּאמַ֑ר בָּר֤וּךְ אַבְרָם֙ לְאֵ֣ל עֶלְי֔וֹן קֹנֵ֖ה שָׁמַ֥יִם וָאָֽרֶץ׃
20אֵ֣ל עֶלְי֔וֹן אֲשֶׁר־ מִגֵּ֥ן צָרֶ֖יךָ בְּיָדֶ֑ךָ וַיִּתֶּן־ ל֥וֹ מַעֲשֵׂ֖ר מִכֹּֽל׃
21מֶֽלֶךְ־ סְדֹ֖ם אֶל־ אַבְרָ֑ם תֶּן־ לִ֣י הַנֶּ֔פֶשׁ וְהָרְכֻ֖שׁ קַֽח־ לָֽךְ׃
22אַבְרָ֖ם אֶל־ מֶ֣לֶךְ סְדֹ֑ם הֲרִימֹ֨תִי יָדִ֤י אֶל־ יְהוָה֙ אֵ֣ל עֶלְי֔וֹן קֹנֵ֖ה שָׁמַ֥יִם וָאָֽרֶץ׃
23מִחוּט֙ וְעַ֣ד שְׂרֽוֹךְ־ נַ֔עַל וְאִם־ אֶקַּ֖ח מִכָּל־ אֲשֶׁר־ לָ֑ךְ וְלֹ֣א תֹאמַ֔ר אֲנִ֖י הֶעֱשַׁ֥רְתִּי אֶת־ אַבְרָֽם׃
