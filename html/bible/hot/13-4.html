1בְּנֵ֖י יְהוּדָ֑ה פֶּ֧רֶץ חֶצְר֛וֹן וְכַרְמִ֖י וְח֥וּר וְשׁוֹבָֽל׃
2בֶן־ שׁוֹבָל֙ הוֹלִ֣יד אֶת־ יַ֔חַת וְיַ֣חַת הֹלִ֔יד אֶת־ אֲחוּמַ֖י וְאֶת־ לָ֑הַד אֵ֖לֶּה מִשְׁפְּח֥וֹת הַצָּֽרְעָתִֽי׃ ס
3אֲבִ֣י עֵיטָ֔ם יִזְרְעֶ֥אל וְיִשְׁמָ֖א וְיִדְבָּ֑שׁ וְשֵׁ֥ם אֲחוֹתָ֖ם הַצְלֶלְפּֽוֹנִי׃
4אֲבִ֣י גְדֹ֔ר וְעֵ֖זֶר אֲבִ֣י חוּשָׁ֑ה אֵ֤לֶּה בְנֵי־ חוּר֙ בְּכ֣וֹר אֶפְרָ֔תָה אֲבִ֖י בֵּ֥ית לָֽחֶם׃
5אֲבִ֣י תְק֔וֹעַ הָי֖וּ שְׁתֵּ֣י נָשִׁ֑ים חֶלְאָ֖ה וְנַעֲרָֽה׃
6ל֤וֹ נַעֲרָה֙ אֶת־ אֲחֻזָּ֣ם וְאֶת־ חֵ֔פֶר וְאֶת־ תֵּימְנִ֖י וְאֶת־ הָאֲחַשְׁתָּרִ֑י אֵ֖לֶּה בְּנֵ֥י נַעֲרָֽה׃
7חֶלְאָ֑ה צֶ֥רֶת [יִצְחָר כ] (וְצֹ֖חַר ק) וְאֶתְנָֽן׃
8הוֹלִ֔יד אֶת־ עָנ֖וּב וְאֶת־ הַצֹּבֵבָ֑ה וּמִשְׁפְּח֥וֹת אֲחַרְחֵ֖ל בֶּן־ הָרֽוּם׃
9יַעְבֵּ֔ץ נִכְבָּ֖ד מֵאֶחָ֑יו וְאִמּ֗וֹ קָרְאָ֨ה שְׁמ֤וֹ יַעְבֵּץ֙ לֵאמֹ֔ר כִּ֥י יָלַ֖דְתִּי בְּעֹֽצֶב׃
10יַ֠עְבֵּץ לֵאלֹהֵ֨י יִשְׂרָאֵ֜ל לֵאמֹ֗ר אִם־ בָּרֵ֨ךְ תְּבָרֲכֵ֜נִי וְהִרְבִּ֤יתָ אֶת־ גְּבוּלִי֙ וְהָיְתָ֤ה יָדְךָ֙ עִמִּ֔י וְעָשִׂ֥יתָ מֵּרָעָ֖ה לְבִלְתִּ֣י עָצְבִּ֑י וַיָּבֵ֥א אֱלֹהִ֖ים אֵ֥ת אֲשֶׁר־ שָׁאָֽל׃
11אֲחִֽי־ שׁוּחָ֖ה הוֹלִ֣יד אֶת־ מְחִ֑יר ה֖וּא אֲבִ֥י אֶשְׁתּֽוֹן׃
12הוֹלִ֞יד אֶת־ בֵּ֤ית רָפָא֙ וְאֶת־ פָּסֵ֔חַ וְאֶת־ תְּחִנָּ֖ה אֲבִ֣י עִ֣יר נָחָ֑שׁ אֵ֖לֶּה אַנְשֵׁ֥י רֵכָֽה׃ ס
13קְנַ֔ז עָתְנִיאֵ֖ל וּשְׂרָיָ֑ה וּבְנֵ֥י עָתְנִיאֵ֖ל חֲתַֽת׃
14הוֹלִ֣יד אֶת־ עָפְרָ֑ה וּשְׂרָיָ֗ה הוֹלִ֤יד אֶת־ יוֹאָב֙ אֲבִי֙ גֵּ֣יא חֲרָשִׁ֔ים כִּ֥י חֲרָשִׁ֖ים הָיֽוּ׃ פ
15כָּלֵ֣ב בֶּן־ יְפֻנֶּ֔ה עִ֥ירוּ אֵלָ֖ה וָנָ֑עַם וּבְנֵ֥י אֵלָ֖ה וּקְנַֽז׃
16יְהַלֶּלְאֵ֑ל זִ֣יף וְזִיפָ֔ה תִּירְיָ֖א וַאֲשַׂרְאֵֽל׃
17עֶזְרָ֔ה יֶ֥תֶר וּמֶ֖רֶד וְעֵ֣פֶר וְיָל֑וֹן וַתַּ֙הַר֙ אֶת־ מִרְיָ֣ם וְאֶת־ שַׁמַּ֔י וְאֶת־ יִשְׁבָּ֖ח אֲבִ֥י אֶשְׁתְּמֹֽעַ׃
18הַיְהֻדִיָּ֗ה יָלְדָ֞ה אֶת־ יֶ֨רֶד אֲבִ֤י גְדוֹר֙ וְאֶת־ חֶ֙בֶר֙ אֲבִ֣י שׂוֹכ֔וֹ וְאֶת־ יְקֽוּתִיאֵ֖ל אֲבִ֣י זָנ֑וֹחַ וְאֵ֗לֶּה בְּנֵי֙ בִּתְיָ֣ה בַת־ פַּרְעֹ֔ה אֲשֶׁ֥ר לָקַ֖ח מָֽרֶד׃ ס
19אֵ֣שֶׁת הֽוֹדִיָּ֔ה אֲח֣וֹת נַ֔חַם אֲבִ֥י קְעִילָ֖ה הַגַּרְמִ֑י וְאֶשְׁתְּמֹ֖עַ הַמַּעֲכָתִֽי׃
20שִׁימ֔וֹן אַמְנ֣וֹן וְרִנָּ֔ה בֶּן־ חָנָ֖ן [וְתֹולֹון כ] (וְתִילֹ֑ון ק) וּבְנֵ֣י יִשְׁעִ֔י זוֹחֵ֖ת וּבֶן־ זוֹחֵֽת׃
21שֵׁלָ֣ה בֶן־ יְהוּדָ֔ה עֵ֚ר אֲבִ֣י לֵכָ֔ה וְלַעְדָּ֖ה אֲבִ֣י מָרֵשָׁ֑ה וּמִשְׁפְּח֛וֹת בֵּית־ עֲבֹדַ֥ת הַבֻּ֖ץ לְבֵ֥ית אַשְׁבֵּֽעַ׃
22וְאַנְשֵׁ֣י כֹזֵבָ֗א וְיוֹאָ֧שׁ וְשָׂרָ֛ף אֲשֶׁר־ בָּעֲל֥וּ לְמוֹאָ֖ב וְיָשֻׁ֣בִי לָ֑חֶם וְהַדְּבָרִ֖ים עַתִּיקִֽים׃
23הַיּ֣וֹצְרִ֔ים וְיֹשְׁבֵ֥י נְטָעִ֖ים וּגְדֵרָ֑ה עִם־ הַמֶּ֥לֶךְ בִּמְלַאכְתּ֖וֹ יָ֥שְׁבוּ שָֽׁם׃ ס
24שִׁמְע֑וֹן נְמוּאֵ֣ל וְיָמִ֔ין יָרִ֖יב זֶ֥רַח שָׁאֽוּל׃
25בְּנ֛וֹ מִבְשָׂ֥ם בְּנ֖וֹ מִשְׁמָ֥ע בְּנֽוֹ׃
26מִשְׁמָ֑ע חַמּוּאֵ֥ל בְּנ֛וֹ זַכּ֥וּר בְּנ֖וֹ שִׁמְעִ֥י בְנֽוֹ׃
27בָּנִ֨ים שִׁשָּׁ֤ה עָשָׂר֙ וּבָנ֣וֹת שֵׁ֔שׁ וּלְאֶחָ֕יו אֵ֖ין בָּנִ֣ים רַבִּ֑ים וְכֹל֙ מִשְׁפַּחְתָּ֔ם לֹ֥א הִרְבּ֖וּ עַד־ בְּנֵ֥י יְהוּדָֽה׃ ס
28בִּבְאֵֽר־ שֶׁ֥בַע וּמוֹלָדָ֖ה וַחֲצַ֥ר שׁוּעָֽל׃
29וּבְעֶ֖צֶם וּבְתוֹלָֽד׃
30וּבְחָרְמָ֖ה וּבְצִֽיקְלָֽג׃
31מַרְכָּבוֹת֙ וּבַחֲצַ֣ר סוּסִ֔ים וּבְבֵ֥ית בִּרְאִ֖י וּֽבְשַׁעֲרָ֑יִם אֵ֥לֶּה עָרֵיהֶ֖ם עַד־ מְלֹ֥ךְ דָּוִֽיד׃
32עֵיטָ֣ם וָעַ֔יִן רִמּ֥וֹן וְתֹ֖כֶן וְעָשָׁ֑ן עָרִ֖ים חָמֵֽשׁ׃
33חַצְרֵיהֶ֗ם אֲשֶׁ֧ר סְבִיב֛וֹת הֶעָרִ֥ים הָאֵ֖לֶּה עַד־ בָּ֑עַל זֹ֚את מוֹשְׁבֹתָ֔ם וְהִתְיַחְשָׂ֖ם לָהֶֽם׃
34וְיַמְלֵ֔ךְ וְיוֹשָׁ֖ה בֶּן־ אֲמַצְיָֽה׃
35וְיֵהוּא֙ בֶּן־ י֣וֹשִׁבְיָ֔ה בֶּן־ שְׂרָיָ֖ה בֶּן־ עֲשִׂיאֵֽל׃
36וְֽיַעֲקֹ֡בָה וִ֠ישׁוֹחָיָה וַעֲשָׂיָ֧ה וַעֲדִיאֵ֛ל וִישִׂימִאֵ֖ל וּבְנָיָֽה׃
37בֶן־ שִׁפְעִ֧י בֶן־ אַלּ֛וֹן בֶּן־ יְדָיָ֥ה בֶן־ שִׁמְרִ֖י בֶּן־ שְׁמַֽעְיָֽה׃
38הַבָּאִ֣ים בְּשֵׁמ֔וֹת נְשִׂיאִ֖ים בְּמִשְׁפְּחוֹתָ֑ם וּבֵית֙ אֲב֣וֹתֵיהֶ֔ם פָּרְצ֖וּ לָרֽוֹב׃
39לִמְב֣וֹא גְדֹ֔ר עַ֖ד לְמִזְרַ֣ח הַגָּ֑יְא לְבַקֵּ֥שׁ מִרְעֶ֖ה לְצֹאנָֽם׃
40מִרְעֶה֙ שָׁמֵ֣ן וָט֔וֹב וְהָאָ֙רֶץ֙ רַחֲבַ֣ת יָדַ֔יִם וְשֹׁקֶ֖טֶת וּשְׁלֵוָ֑ה כִּ֣י מִן־ חָ֔ם הַיֹּשְׁבִ֥ים שָׁ֖ם לְפָנִֽים׃
41אֵלֶּה֩ הַכְּתוּבִ֨ים בְּשֵׁמ֜וֹת בִּימֵ֣י ׀ יְחִזְקִיָּ֣הוּ מֶֽלֶךְ־ יְהוּדָ֗ה וַיַּכּ֨וּ אֶת־ אָהֳלֵיהֶ֜ם וְאֶת־ [הַמְּעִינִים כ] (הַמְּעוּנִ֨ים ק) אֲשֶׁ֤ר נִמְצְאוּ־ שָׁ֙מָּה֙ וַיַּחֲרִימֻם֙ עַד־ הַיּ֣וֹם הַזֶּ֔ה וַיֵּשְׁב֖וּ תַּחְתֵּיהֶ֑ם כִּֽי־ מִרְעֶ֥ה לְצֹאנָ֖ם שָֽׁם׃
42מִן־ בְּנֵ֣י שִׁמְע֗וֹן הָלְכוּ֙ לְהַ֣ר שֵׂעִ֔יר אֲנָשִׁ֖ים חֲמֵ֣שׁ מֵא֑וֹת וּפְלַטְיָ֡ה וּ֠נְעַרְיָה וּרְפָיָ֧ה וְעֻזִּיאֵ֛ל בְּנֵ֥י יִשְׁעִ֖י בְּרֹאשָֽׁם׃
