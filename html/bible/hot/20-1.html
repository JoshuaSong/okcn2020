1מִ֭שְׁלֵי שְׁלֹמֹ֣ה בֶן־ דָּוִ֑ד מֶ֝֗לֶךְ יִשְׂרָאֵֽל׃
2חָכְמָ֣ה וּמוּסָ֑ר לְ֝הָבִ֗ין אִמְרֵ֥י בִינָֽה׃
3מוּסַ֣ר הַשְׂכֵּ֑ל צֶ֥דֶק וּ֝מִשְׁפָּ֗ט וּמֵישָׁרִֽים׃
4לִפְתָאיִ֣ם עָרְמָ֑ה לְ֝נַ֗עַר דַּ֣עַת וּמְזִמָּֽה׃
5חָ֭כָם וְי֣וֹסֶף לֶ֑קַח וְ֝נָב֗וֹן תַּחְבֻּל֥וֹת יִקְנֶֽה׃
6מָ֭שָׁל וּמְלִיצָ֑ה דִּבְרֵ֥י חֲ֝כָמִ֗ים וְחִידֹתָֽם׃
7יְ֭הוָה רֵאשִׁ֣ית דָּ֑עַת חָכְמָ֥ה וּ֝מוּסָ֗ר אֱוִילִ֥ים בָּֽזוּ׃ פ
8בְּ֭נִי מוּסַ֣ר אָבִ֑יךָ וְאַל־ תִּ֝טֹּ֗שׁ תּוֹרַ֥ת אִמֶּֽךָ׃
9לִוְיַ֤ת חֵ֓ן הֵ֬ם לְרֹאשֶׁ֑ךָ וַ֝עֲנָקִ֗ים לְגַרְגְּרֹתֶֽיךָ׃
10אִם־ יְפַתּ֥וּךָ חַ֝טָּאִ֗ים אַל־ תֹּבֵֽא׃
11יֹאמְרוּ֮ לְכָ֪ה אִ֫תָּ֥נוּ נֶאֶרְבָ֥ה לְדָ֑ם נִצְפְּנָ֖ה לְנָקִ֣י חִנָּֽם׃
12כִּשְׁא֣וֹל חַיִּ֑ים וּ֝תְמִימִ֗ים כְּי֣וֹרְדֵי בֽוֹר׃
13ה֣וֹן יָקָ֣ר נִמְצָ֑א נְמַלֵּ֖א בָתֵּ֣ינוּ שָׁלָֽל׃
14תַּפִּ֣יל בְּתוֹכֵ֑נוּ כִּ֥יס אֶ֝חָ֗ד יִהְיֶ֥ה לְכֻלָּֽנוּ׃
15אַל־ תֵּלֵ֣ךְ בְּדֶ֣רֶךְ אִתָּ֑ם מְנַ֥ע רַ֝גְלְךָ֗ מִנְּתִיבָתָֽם׃
16רַ֭גְלֵיהֶם לָרַ֣ע יָר֑וּצוּ וִֽ֝ימַהֲר֗וּ לִשְׁפָּךְ־ דָּֽם׃
17חִ֭נָּם מְזֹרָ֣ה הָרָ֑שֶׁת בְּ֝עֵינֵ֗י כָל־ בַּ֥עַל כָּנָֽף׃
18לְדָמָ֣ם יֶאֱרֹ֑בוּ יִ֝צְפְּנ֗וּ לְנַפְשֹׁתָֽם׃
19אָ֭רְחוֹת כָּל־ בֹּ֣צֵֽעַ בָּ֑צַע אֶת־ נֶ֖פֶשׁ בְּעָלָ֣יו יִקָּֽח׃ פ
20בַּח֣וּץ תָּרֹ֑נָּה בָּ֝רְחֹב֗וֹת תִּתֵּ֥ן קוֹלָֽהּ׃
21הֹמִיּ֗וֹת תִּ֫קְרָ֥א בְּפִתְחֵ֖י שְׁעָרִ֥ים בָּעִ֗יר אֲמָרֶ֥יהָ תֹאמֵֽר׃
22מָתַ֣י ׀ פְּתָיִם֮ תְּֽאֵהֲב֫וּ פֶ֥תִי וְלֵצִ֗ים לָ֭צוֹן חָמְד֣וּ לָהֶ֑ם וּ֝כְסִילִ֗ים יִשְׂנְאוּ־ דָֽעַת׃
23לְֽת֫וֹכַחְתִּ֥י הִנֵּ֤ה אַבִּ֣יעָה לָכֶ֣ם רוּחִ֑י אוֹדִ֖יעָה דְבָרַ֣י אֶתְכֶֽם׃
24קָ֭רָאתִי וַתְּמָאֵ֑נוּ נָטִ֥יתִי יָ֝דִ֗י וְאֵ֣ין מַקְשִֽׁיב׃
25כָל־ עֲצָתִ֑י וְ֝תוֹכַחְתִּ֗י לֹ֣א אֲבִיתֶֽם׃
26אֲ֭נִי בְּאֵידְכֶ֣ם אֶשְׂחָ֑ק אֶ֝לְעַ֗ג בְּבֹ֣א פַחְדְּכֶֽם׃
27[כְשַׁאֲוָה כ] (כְשֹׁואָ֨ה ק) פַּחְדְּכֶ֗ם וְֽ֭אֵידְכֶם כְּסוּפָ֣ה יֶאֱתֶ֑ה בְּבֹ֥א עֲ֝לֵיכֶ֗ם צָרָ֥ה וְצוּקָֽה׃
28יִ֭קְרָאֻנְנִי וְלֹ֣א אֶֽעֱנֶ֑ה יְ֝שַׁחֲרֻ֗נְנִי וְלֹ֣א יִמְצָאֻֽנְנִי׃
29כִּי־ שָׂ֣נְאוּ דָ֑עַת וְיִרְאַ֥ת יְ֝הֹוָ֗ה לֹ֣א בָחָֽרוּ׃
30אָב֥וּ לַעֲצָתִ֑י נָ֝אֲצ֗וּ כָּל־ תּוֹכַחְתִּֽי׃
31מִפְּרִ֣י דַרְכָּ֑ם וּֽמִמֹּעֲצֹ֖תֵיהֶ֣ם יִשְׂבָּֽעוּ׃
32מְשׁוּבַ֣ת פְּתָיִ֣ם תַּֽהַרְגֵ֑ם וְשַׁלְוַ֖ת כְּסִילִ֣ים תְּאַבְּדֵֽם׃
