1ט֤וֹב פַּ֣ת חֲ֭רֵבָה וְשַׁלְוָה־ בָ֑הּ מִ֝בַּ֗יִת מָלֵ֥א זִבְחֵי־ רִֽיב׃
2מַשְׂכִּ֗יל יִ֭מְשֹׁל בְּבֵ֣ן מֵבִ֑ישׁ וּבְת֥וֹךְ אַ֝חִ֗ים יַחֲלֹ֥ק נַחֲלָֽה׃
3לַ֭כֶּסֶף וְכ֣וּר לַזָּהָ֑ב וּבֹחֵ֖ן לִבּ֣וֹת יְהוָֽה׃
4מַקְשִׁ֣יב עַל־ שְׂפַת־ אָ֑וֶן שֶׁ֥קֶר מֵ֝זִין עַל־ לְשׁ֥וֹן הַוֹּֽת׃
5לָ֭רָשׁ חֵרֵ֣ף עֹשֵׂ֑הוּ שָׂמֵ֥חַ לְ֝אֵ֗יד לֹ֣א יִנָּקֶֽה׃
6זְ֭קֵנִים בְּנֵ֣י בָנִ֑ים וְתִפְאֶ֖רֶת בָּנִ֣ים אֲבוֹתָֽם׃
7נָאוָ֣ה לְנָבָ֣ל שְׂפַת־ יֶ֑תֶר אַ֝֗ף כִּֽי־ לְנָדִ֥יב שְׂפַת־ שָֽׁקֶר׃
8חֵ֣ן הַ֭שֹּׁחַד בְּעֵינֵ֣י בְעָלָ֑יו אֶֽל־ כָּל־ אֲשֶׁ֖ר יִפְנֶ֣ה יַשְׂכִּֽיל׃
9פֶּ֭שַׁע מְבַקֵּ֣שׁ אַהֲבָ֑ה וְשֹׁנֶ֥ה בְ֝דָבָ֗ר מַפְרִ֥יד אַלּֽוּף׃
10גְּעָרָ֣ה בְמֵבִ֑ין מֵהַכּ֖וֹת כְּסִ֣יל מֵאָֽה׃
11מְרִ֥י יְבַקֶּשׁ־ רָ֑ע וּמַלְאָ֥ךְ אַ֝כְזָרִ֗י יְשֻׁלַּח־ בּֽוֹ׃
12דֹּ֣ב שַׁכּ֣וּל בְּאִ֑ישׁ וְאַל־ כְּ֝סִ֗יל בְּאִוַּלְתּֽוֹ׃
13רָ֭עָה תַּ֣חַת טוֹבָ֑ה לֹא־ [תָמִישׁ כ] (תָמ֥וּשׁ ק) רָ֝עָ֗ה מִבֵּיתֽוֹ׃
14מַ֭יִם רֵאשִׁ֣ית מָד֑וֹן וְלִפְנֵ֥י הִ֝תְגַּלַּ֗ע הָרִ֥יב נְטֽוֹשׁ׃
15רָ֭שָׁע וּמַרְשִׁ֣יעַ צַדִּ֑יק תּוֹעֲבַ֥ת יְ֝הוָ֗ה גַּם־ שְׁנֵיהֶֽם׃
16זֶּ֣ה מְחִ֣יר בְּיַד־ כְּסִ֑יל לִקְנ֖וֹת חָכְמָ֣ה וְלֶב־ אָֽיִן׃
17עֵ֭ת אֹהֵ֣ב הָרֵ֑עַ וְאָ֥ח לְ֝צָרָ֗ה יִוָּלֵֽד׃
18חֲסַר־ לֵ֭ב תּוֹקֵ֣עַ כָּ֑ף עֹרֵ֥ב עֲ֝רֻבָּ֗ה לִפְנֵ֥י רֵעֵֽהוּ׃
19פֶּ֭שַׁע אֹהֵ֣ב מַצָּ֑ה מַגְבִּ֥יהַּ פִּ֝תְח֗וֹ מְבַקֶּשׁ־ שָֽׁבֶר׃
20לֵ֭ב לֹ֣א יִמְצָא־ ט֑וֹב וְנֶהְפָּ֥ךְ בִּ֝לְשׁוֹנ֗וֹ יִפּ֥וֹל בְּרָעָֽה׃
21כְּ֭סִיל לְת֣וּגָה ל֑וֹ וְלֹֽא־ יִ֝שְׂמַ֗ח אֲבִ֣י נָבָֽל׃
22שָׂ֭מֵחַ יֵיטִ֣ב גֵּהָ֑ה וְר֥וּחַ נְ֝כֵאָ֗ה תְּיַבֶּשׁ־ גָּֽרֶם׃
23מֵ֭חֵיק רָשָׁ֣ע יִקָּ֑ח לְ֝הַטּ֗וֹת אָרְח֥וֹת מִשְׁפָּֽט׃
24פְּנֵ֣י מֵבִ֣ין חָכְמָ֑ה וְעֵינֵ֥י כְ֝סִ֗יל בִּקְצֵה־ אָֽרֶץ׃
25לְ֭אָבִיו בֵּ֣ן כְּסִ֑יל וּ֝מֶ֗מֶר לְיוֹלַדְתּֽוֹ׃
26עֲנ֣וֹשׁ לַצַּדִּ֣יק לֹא־ ט֑וֹב לְהַכּ֖וֹת נְדִיבִ֣ים עַל־ יֹֽשֶׁר׃
27אֲ֭מָרָיו יוֹדֵ֣עַ דָּ֑עַת [וְקַר־ כ] (יְקַר־ ק) אִ֣ישׁ תְּבוּנָֽה׃
