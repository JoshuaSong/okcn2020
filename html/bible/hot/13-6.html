1בְּנֵ֖י לֵוִ֑י גֵּרְשׁ֕וֹן קְהָ֖ת וּמְרָרִֽי׃
2קְהָ֑ת עַמְרָ֣ם יִצְהָ֔ר וְחֶבְר֖וֹן וְעֻזִּיאֵֽל׃ ס
3עַמְרָ֔ם אַהֲרֹ֥ן וּמֹשֶׁ֖ה וּמִרְיָ֑ם ס וּבְנֵ֣י אַהֲרֹ֔ן נָדָב֙ וַאֲבִיה֔וּא אֶלְעָזָ֖ר וְאִיתָמָֽר׃ ס
4הוֹלִ֣יד אֶת־ פִּֽינְחָ֔ס פִּֽינְחָ֖ס הֹלִ֥יד אֶת־ אֲבִישֽׁוּעַ׃
5הוֹלִ֣יד אֶת־ בֻּקִּ֔י וּבֻקִּ֖י הוֹלִ֥יד אֶת־ עֻזִּֽי׃
6הוֹלִ֣יד אֶת־ זְרַֽחְיָ֔ה וּֽזְרַֽחְיָ֖ה הוֹלִ֥יד אֶת־ מְרָיֽוֹת׃
7הוֹלִ֣יד אֶת־ אֲמַרְיָ֔ה וַאֲמַרְיָ֖ה הוֹלִ֥יד אֶת־ אֲחִיטֽוּב׃
8הוֹלִ֣יד אֶת־ צָד֔וֹק וְצָד֖וֹק הוֹלִ֥יד אֶת־ אֲחִימָֽעַץ׃
9הוֹלִ֣יד אֶת־ עֲזַרְיָ֔ה וַעֲזַרְיָ֖ה הוֹלִ֥יד אֶת־ יוֹחָנָֽן׃
10הוֹלִ֣יד אֶת־ עֲזַרְיָ֑ה ה֚וּא אֲשֶׁ֣ר כִּהֵ֔ן בַּבַּ֕יִת אֲשֶׁר־ בָּנָ֥ה שְׁלֹמֹ֖ה בִּירוּשָׁלִָֽם׃
11עֲזַרְיָ֖ה אֶת־ אֲמַרְיָ֑ה וַאֲמַרְיָ֖ה הוֹלִ֥יד אֶת־ אֲחִיטֽוּב׃
12הוֹלִ֣יד אֶת־ צָד֔וֹק וְצָד֖וֹק הוֹלִ֥יד אֶת־ שַׁלּֽוּם׃
13הוֹלִ֣יד אֶת־ חִלְקִיָּ֔ה וְחִלְקִיָּ֖ה הוֹלִ֥יד אֶת־ עֲזַרְיָֽה׃
14הוֹלִ֣יד אֶת־ שְׂרָיָ֔ה וּשְׂרָיָ֖ה הוֹלִ֥יד אֶת־ יְהוֹצָדָֽק׃
15הָלַ֔ךְ בְּהַגְל֣וֹת יְהוָ֔ה אֶת־ יְהוּדָ֖ה וִירוּשָׁלִָ֑ם בְּיַ֖ד נְבֻכַדְנֶאצַּֽר׃ ס
16לֵוִ֑י גֵּרְשֹׁ֕ם קְהָ֖ת וּמְרָרִֽי׃
17שְׁמ֥וֹת בְּֽנֵי־ גֵרְשׁ֖וֹם לִבְנִ֥י וְשִׁמְעִֽי׃
18קְהָ֑ת עַמְרָ֣ם וְיִצְהָ֔ר וְחֶבְר֖וֹן וְעֻזִּיאֵֽל׃
19מְרָרִ֖י מַחְלִ֣י וּמֻשִׁ֑י וְאֵ֛לֶּה מִשְׁפְּח֥וֹת הַלֵּוִ֖י לַאֲבוֹתֵיהֶֽם׃
20לִבְנִ֥י בְנ֛וֹ יַ֥חַת בְּנ֖וֹ זִמָּ֥ה בְנֽוֹ׃
21בְּנוֹ֙ עִדּ֣וֹ בְנ֔וֹ זֶ֥רַח בְּנ֖וֹ יְאָתְרַ֥י בְּנֽוֹ׃
22קְהָ֑ת עַמִּינָדָ֣ב בְּנ֔וֹ קֹ֥רַח בְּנ֖וֹ אַסִּ֥יר בְּנֽוֹ׃
23בְנ֛וֹ וְאֶבְיָסָ֥ף בְּנ֖וֹ וְאַסִּ֥יר בְּנֽוֹ׃
24בְּנוֹ֙ אוּרִיאֵ֣ל בְּנ֔וֹ עֻזִּיָּ֥ה בְנ֖וֹ וְשָׁא֥וּל בְּנֽוֹ׃
25אֶלְקָנָ֔ה עֲמָשַׂ֖י וַאֲחִימֽוֹת׃
26[בְּנֹו כ] (בְּנֵי֙ ק) אֶלְקָנָ֔ה צוֹפַ֥י בְּנ֖וֹ וְנַ֥חַת בְּנֽוֹ׃
27בְּנ֛וֹ יְרֹחָ֥ם בְּנ֖וֹ אֶלְקָנָ֥ה בְנֽוֹ׃
28שְׁמוּאֵ֛ל הַבְּכֹ֥ר וַשְׁנִ֖י וַאֲבִיָּֽה׃ ס
29מְרָרִ֖י מַחְלִ֑י לִבְנִ֥י בְנ֛וֹ שִׁמְעִ֥י בְנ֖וֹ עֻזָּ֥ה בְנֽוֹ׃
30בְנ֛וֹ חַגִּיָּ֥ה בְנ֖וֹ עֲשָׂיָ֥ה בְנֽוֹ׃ פ
31אֲשֶׁ֨ר הֶעֱמִ֥יד דָּוִ֛יד עַל־ יְדֵי־ שִׁ֖יר בֵּ֣ית יְהוָ֑ה מִמְּנ֖וֹחַ הָאָרֽוֹן׃
32מְשָׁרְתִ֜ים לִפְנֵ֨י מִשְׁכַּ֤ן אֹֽהֶל־ מוֹעֵד֙ בַּשִּׁ֔יר עַד־ בְּנ֧וֹת שְׁלֹמֹ֛ה אֶת־ בֵּ֥ית יְהוָ֖ה בִּירוּשָׁלִָ֑ם וַיַּעַמְד֥וּ כְמִשְׁפָּטָ֖ם עַל־ עֲבוֹדָתָֽם׃
33הָעֹמְדִ֖ים וּבְנֵיהֶ֑ם מִבְּנֵי֙ הַקְּהָתִ֔י הֵימָן֙ הַמְשׁוֹרֵ֔ר בֶּן־ יוֹאֵ֖ל בֶּן־ שְׁמוּאֵֽל׃
34אֶלְקָנָה֙ בֶּן־ יְרֹחָ֔ם בֶּן־ אֱלִיאֵ֖ל בֶּן־ תּֽוֹחַ׃
35[צִיף כ] (צוּף֙ ק) בֶּן־ אֶלְקָנָ֔ה בֶּן־ מַ֖חַת בֶּן־ עֲמָשָֽׂי׃
36אֶלְקָנָה֙ בֶּן־ יוֹאֵ֔ל בֶּן־ עֲזַרְיָ֖ה בֶּן־ צְפַנְיָֽה׃
37תַּ֙חַת֙ בֶּן־ אַסִּ֔יר בֶּן־ אֶבְיָסָ֖ף בֶּן־ קֹֽרַח׃
38יִצְהָ֣ר בֶּן־ קְהָ֔ת בֶּן־ לֵוִ֖י בֶּן־ יִשְׂרָאֵֽל׃
39אָסָ֔ף הָעֹמֵ֖ד עַל־ יְמִינ֑וֹ אָסָ֥ף בֶּן־ בֶּרֶכְיָ֖הוּ בֶּן־ שִׁמְעָֽא׃
40מִיכָאֵ֥ל בֶּן־ בַּעֲשֵׂיָ֖ה בֶּן־ מַלְכִּיָּֽה׃
41אֶתְנִ֥י בֶן־ זֶ֖רַח בֶּן־ עֲדָיָֽה׃
42אֵיתָ֥ן בֶּן־ זִמָּ֖ה בֶּן־ שִׁמְעִֽי׃
43יַ֥חַת בֶּן־ גֵּרְשֹׁ֖ם בֶּן־ לֵוִֽי׃ ס
44מְרָרִ֛י אֲחֵיהֶ֖ם עַֽל־ הַשְּׂמֹ֑אול אֵיתָן֙ בֶּן־ קִישִׁ֔י בֶּן־ עַבְדִּ֖י בֶּן־ מַלּֽוּךְ׃
45חֲשַׁבְיָ֥ה בֶן־ אֲמַצְיָ֖ה בֶּן־ חִלְקִיָּֽה׃
46אַמְצִ֥י בֶן־ בָּנִ֖י בֶּן־ שָֽׁמֶר׃
47מַחְלִי֙ בֶּן־ מוּשִׁ֔י בֶּן־ מְרָרִ֖י בֶּן־ לֵוִֽי׃ ס
48הַלְוִיִּ֑ם נְתוּנִ֕ים לְכָ֨ל־ עֲבוֹדַ֔ת מִשְׁכַּ֖ן בֵּ֥ית הָאֱלֹהִֽים׃
49וּבָנָ֜יו מַקְטִירִ֨ים עַל־ מִזְבַּ֤ח הָֽעוֹלָה֙ וְעַל־ מִזְבַּ֣ח הַקְּטֹ֔רֶת לְכֹ֕ל מְלֶ֖אכֶת קֹ֣דֶשׁ הַקֳּדָשִׁ֑ים וּלְכַפֵּר֙ עַל־ יִשְׂרָאֵ֔ל כְּכֹל֙ אֲשֶׁ֣ר צִוָּ֔ה מֹשֶׁ֖ה עֶ֥בֶד הָאֱלֹהִֽים׃ פ
50בְּנֵ֣י אַהֲרֹ֑ן אֶלְעָזָ֥ר בְּנ֛וֹ פִּֽינְחָ֥ס בְּנ֖וֹ אֲבִישׁ֥וּעַ בְּנֽוֹ׃
51בְנ֛וֹ עֻזִּ֥י בְנ֖וֹ זְרַֽחְיָ֥ה בְנֽוֹ׃
52בְּנ֛וֹ אֲמַרְיָ֥ה בְנ֖וֹ אֲחִיט֥וּב בְּנֽוֹ׃
53בְּנ֖וֹ אֲחִימַ֥עַץ בְּנֽוֹ׃ ס
54מוֹשְׁבוֹתָ֔ם לְטִירוֹתָ֖ם בִּגְבוּלָ֑ם לִבְנֵ֤י אַהֲרֹן֙ לְמִשְׁפַּחַ֣ת הַקְּהָתִ֔י כִּ֥י לָהֶ֖ם הָיָ֥ה הַגּוֹרָֽל׃
55לָהֶ֛ם אֶת־ חֶבְר֖וֹן בְּאֶ֣רֶץ יְהוּדָ֑ה וְאֶת־ מִגְרָשֶׁ֖יהָ סְבִיבֹתֶֽיהָ׃
56שְׂדֵ֥ה הָעִ֖יר וְאֶת־ חֲצֵרֶ֑יהָ נָתְנ֖וּ לְכָלֵ֥ב בֶּן־ יְפֻנֶּֽה׃ ס
57אַהֲרֹ֗ן נָתְנוּ֙ אֶת־ עָרֵ֣י הַמִּקְלָ֔ט אֶת־ חֶבְר֥וֹן וְאֶת־ לִבְנָ֖ה וְאֶת־ מִגְרָשֶׁ֑יהָ וְאֶת־ יַתִּ֥ר וְאֶֽת־ אֶשְׁתְּמֹ֖עַ וְאֶת־ מִגְרָשֶֽׁיהָ׃
58חִילֵז֙ וְאֶת־ מִגְרָשֶׁ֔יהָ אֶת־ דְּבִ֖יר וְאֶת־ מִגְרָשֶֽׁיהָ׃
59עָשָׁן֙ וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ בֵּ֥ית שֶׁ֖מֶשׁ וְאֶת־ מִגְרָשֶֽׁיהָ׃ ס
60בִנְיָמִ֗ן אֶת־ גֶּ֤בַע וְאֶת־ מִגְרָשֶׁ֙יהָ֙ וְאֶת־ עָלֶ֣מֶת וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ עֲנָת֖וֹת וְאֶת־ מִגְרָשֶׁ֑יהָ כָּל־ עָרֵיהֶ֛ם שְׁלֹשׁ־ עֶשְׂרֵ֥ה עִ֖יר בְּמִשְׁפְּחוֹתֵיהֶֽם׃ ס
61קְהָ֜ת הַנּוֹתָרִ֗ים מִמִּשְׁפַּ֣חַת הַמַּטֶּ֡ה מִֽ֠מַּחֲצִית מַטֵּ֨ה חֲצִ֧י מְנַשֶּׁ֛ה בַּגּוֹרָ֖ל עָרִ֥ים עָֽשֶׂר׃ ס
62גֵרְשׁ֜וֹם לְמִשְׁפְּחוֹתָ֗ם מִמַּטֵּ֣ה יִ֠שָׂשכָר וּמִמַּטֵּ֨ה אָשֵׁ֜ר וּמִמַּטֵּ֣ה נַפְתָּלִ֗י וּמִמַּטֵּ֤ה מְנַשֶּׁה֙ בַּבָּשָׁ֔ן עָרִ֖ים שְׁלֹ֥שׁ עֶשְׂרֵֽה׃ ס
63מְרָרִ֜י לְמִשְׁפְּחוֹתָ֗ם מִמַּטֵּ֣ה רְ֠אוּבֵן וּֽמִמַּטֵּה־ גָ֞ד וּמִמַּטֵּ֤ה זְבוּלֻן֙ בַּגּוֹרָ֔ל עָרִ֖ים שְׁתֵּ֥ים עֶשְׂרֵֽה׃
64בְנֵי־ יִשְׂרָאֵ֖ל לַלְוִיִּ֑ם אֶת־ הֶעָרִ֖ים וְאֶת־ מִגְרְשֵׁיהֶֽם׃
65בַגּוֹרָ֗ל מִמַּטֵּ֤ה בְנֵי־ יְהוּדָה֙ וּמִמַּטֵּ֣ה בְנֵי־ שִׁמְע֔וֹן וּמִמַּטֵּ֖ה בְּנֵ֣י בִנְיָמִ֑ן אֵ֚ת הֶעָרִ֣ים הָאֵ֔לֶּה אֲשֶׁר־ יִקְרְא֥וּ אֶתְהֶ֖ם בְּשֵׁמֽוֹת׃ ס
66בְּנֵ֣י קְהָ֑ת וַיְהִי֙ עָרֵ֣י גְבוּלָ֔ם מִמַּטֵּ֖ה אֶפְרָֽיִם׃
67לָהֶ֜ם אֶת־ עָרֵ֧י הַמִּקְלָ֛ט אֶת־ שְׁכֶ֥ם וְאֶת־ מִגְרָשֶׁ֖יהָ בְּהַ֣ר אֶפְרָ֑יִם וְאֶת־ גֶּ֖זֶר וְאֶת־ מִגְרָשֶֽׁיהָ׃
68יָקְמְעָם֙ וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ בֵּ֥ית חוֹר֖וֹן וְאֶת־ מִגְרָשֶֽׁיהָ׃
69אַיָּלוֹן֙ וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ גַּת־ רִמּ֖וֹן וְאֶת־ מִגְרָשֶֽׁיהָ׃ פ
70מַטֵּ֣ה מְנַשֶּׁ֔ה אֶת־ עָנֵר֙ וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ בִּלְעָ֖ם וְאֶת־ מִגְרָשֶׁ֑יהָ לְמִשְׁפַּ֥חַת לִבְנֵי־ קְהָ֖ת הַנּוֹתָרִֽים׃ פ
71גֵּרְשׁוֹם֒ מִמִּשְׁפַּ֗חַת חֲצִי֙ מַטֵּ֣ה מְנַשֶּׁ֔ה אֶת־ גּוֹלָ֥ן בַּבָּשָׁ֖ן וְאֶת־ מִגְרָשֶׁ֑יהָ וְאֶת־ עַשְׁתָּר֖וֹת וְאֶת־ מִגְרָשֶֽׁיהָ׃ ס
72יִשָׂשכָ֔ר אֶת־ קֶ֖דֶשׁ וְאֶת־ מִגְרָשֶׁ֑יהָ אֶת־ דָּבְרַ֖ת וְאֶת־ מִגְרָשֶֽׁיהָ׃
73רָאמוֹת֙ וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ עָנֵ֖ם וְאֶת־ מִגְרָשֶֽׁיהָ׃ ס
74אָשֵׁ֔ר אֶת־ מָשָׁ֖ל וְאֶת־ מִגְרָשֶׁ֑יהָ וְאֶת־ עַבְדּ֖וֹן וְאֶת־ מִגְרָשֶֽׁיהָ׃
75חוּקֹק֙ וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ רְחֹ֖ב וְאֶת־ מִגְרָשֶֽׁיהָ׃
76נַפְתָּלִ֗י אֶת־ קֶ֤דֶשׁ בַּגָּלִיל֙ וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ חַמּ֖וֹן וְאֶת־ מִגְרָשֶׁ֑יהָ וְאֶת־ קִרְיָתַ֖יִם וְאֶת־ מִגְרָשֶֽׁיהָ׃ ס
77מְרָרִי֮ הַנּוֹתָרִים֒ מִמַּטֵּ֣ה זְבוּלֻ֔ן אֶת־ רִמּוֹנ֖וֹ וְאֶת־ מִגְרָשֶׁ֑יהָ אֶת־ תָּב֖וֹר וְאֶת־ מִגְרָשֶֽׁיהָ׃
78לְיַרְדֵּ֣ן יְרֵחוֹ֮ לְמִזְרַ֣ח הַיַּרְדֵּן֒ מִמַּטֵּ֣ה רְאוּבֵ֔ן אֶת־ בֶּ֥צֶר בַּמִּדְבָּ֖ר וְאֶת־ מִגְרָשֶׁ֑יהָ וְאֶת־ יַ֖הְצָה וְאֶת־ מִגְרָשֶֽׁיהָ׃
79קְדֵמוֹת֙ וְאֶת־ מִגְרָשֶׁ֔יהָ וְאֶת־ מֵיפַ֖עַת וְאֶת־ מִגְרָשֶֽׁיהָ׃
80גָ֔ד אֶת־ רָאמ֥וֹת בַּגִּלְעָ֖ד וְאֶת־ מִגְרָשֶׁ֑יהָ וְאֶֽת־ מַחֲנַ֖יִם וְאֶת־ מִגְרָשֶֽׁיהָ׃
