1וְאֵ֛לֶּה דִּבְרֵ֥י דָוִ֖ד הָאַֽחֲרֹנִ֑ים נְאֻ֧ם דָּוִ֣ד בֶּן־ יִשַׁ֗י וּנְאֻ֤ם הַגֶּ֙בֶר֙ הֻ֣קַם עָ֔ל מְשִׁ֙יחַ֙ אֱלֹהֵ֣י יַֽעֲקֹ֔ב וּנְעִ֖ים זְמִר֥וֹת יִשְׂרָאֵֽל׃
2יְהוָ֖ה דִּבֶּר־ בִּ֑י וּמִלָּת֖וֹ עַל־ לְשׁוֹנִֽי׃
3אֱלֹהֵ֣י יִשְׂרָאֵ֔ל לִ֥י דִבֶּ֖ר צ֣וּר יִשְׂרָאֵ֑ל מוֹשֵׁל֙ בָּאָדָ֔ם צַדִּ֕יק מוֹשֵׁ֖ל יִרְאַ֥ת אֱלֹהִֽים׃
4בֹּ֖קֶר יִזְרַח־ שָׁ֑מֶשׁ בֹּ֚קֶר לֹ֣א עָב֔וֹת מִנֹּ֥גַהּ מִמָּטָ֖ר דֶּ֥שֶׁא מֵאָֽרֶץ׃
5לֹא־ כֵ֥ן בֵּיתִ֖י עִם־ אֵ֑ל כִּי֩ בְרִ֨ית עוֹלָ֜ם שָׂ֣ם לִ֗י עֲרוּכָ֤ה בַכֹּל֙ וּשְׁמֻרָ֔ה כִּֽי־ כָל־ יִשְׁעִ֥י וְכָל־ חֵ֖פֶץ כִּֽי־ לֹ֥א יַצְמִֽיחַ׃
6כְּק֥וֹץ מֻנָ֖ד כֻּלָּ֑הַם כִּֽי־ לֹ֥א בְיָ֖ד יִקָּֽחוּ׃
7יִגַּ֣ע בָּהֶ֔ם יִמָּלֵ֥א בַרְזֶ֖ל וְעֵ֣ץ חֲנִ֑ית וּבָאֵ֕שׁ שָׂר֥וֹף יִשָּׂרְפ֖וּ בַּשָּֽׁבֶת׃ פ
8שְׁמ֥וֹת הַגִּבֹּרִ֖ים אֲשֶׁ֣ר לְדָוִ֑ד יֹשֵׁ֨ב בַּשֶּׁ֜בֶת תַּחְכְּמֹנִ֣י ׀ רֹ֣אשׁ הַשָּׁלִשִׁ֗י ה֚וּא עֲדִינ֣וֹ [הָעֶצְנֹו כ] (הָֽעֶצְנִ֔י ק) עַל־ שְׁמֹנֶ֥ה מֵא֛וֹת חָלָ֖ל בְּפַ֥עַם [אֶחָד כ] (אֶחָֽת׃ ק) ס
9כ] (וְאַחֲרָ֛יו ק) אֶלְעָזָ֥ר בֶּן־ [דֹּדִי כ] (דֹּדֹ֖ו ק) בֶּן־ אֲחֹחִ֑י בִּשְׁלֹשָׁ֨ה [גִּבֹּרִים כ] (הַגִּבֹּרִ֜ים ק) עִם־ דָּוִ֗ד בְּחָֽרְפָ֤ם בַּפְּלִשְׁתִּים נֶאֶסְפוּ־ שָׁ֣ם לַמִּלְחָמָ֔ה וַֽיַּעֲל֖וּ אִ֥ישׁ יִשְׂרָאֵֽל׃
10קָם֩ וַיַּ֨ךְ בַּפְּלִשְׁתִּ֜ים עַ֣ד ׀ כִּֽי־ יָגְעָ֣ה יָד֗וֹ וַתִּדְבַּ֤ק יָדוֹ֙ אֶל־ הַחֶ֔רֶב וַיַּ֧עַשׂ יְהוָ֛ה תְּשׁוּעָ֥ה גְדוֹלָ֖ה בַּיּ֣וֹם הַה֑וּא וְהָעָ֛ם יָשֻׁ֥בוּ אַחֲרָ֖יו אַךְ־ לְפַשֵּֽׁט׃ ס
11שַׁמָּ֥א בֶן־ אָגֵ֖א הָרָרִ֑י וַיֵּאָסְפ֨וּ פְלִשְׁתִּ֜ים לַחַיָּ֗ה וַתְּהִי־ שָׁ֞ם חֶלְקַ֤ת הַשָּׂדֶה֙ מְלֵאָ֣ה עֲדָשִׁ֔ים וְהָעָ֥ם נָ֖ס מִפְּנֵ֥י פְלִשְׁתִּֽים׃
12בְּתוֹךְ־ הַֽחֶלְקָה֙ וַיַּצִּילֶ֔הָ וַיַּ֖ךְ אֶת־ פְּלִשְׁתִּ֑ים וַיַּ֥עַשׂ יְהוָ֖ה תְּשׁוּעָ֥ה גְדוֹלָֽה׃ ס
13[שְׁלֹשִׁים כ] (שְׁלֹשָׁ֜ה ק) מֵהַשְּׁלֹשִׁ֣ים רֹ֗אשׁ וַיָּבֹ֤אוּ אֶל־ קָצִיר֙ אֶל־ דָּוִ֔ד אֶל־ מְעָרַ֖ת עֲדֻלָּ֑ם וְחַיַּ֣ת פְּלִשְׁתִּ֔ים חֹנָ֖ה בְּעֵ֥מֶק רְפָאִֽים׃
14אָ֣ז בַּמְּצוּדָ֑ה וּמַצַּ֣ב פְּלִשְׁתִּ֔ים אָ֖ז בֵּ֥ית לָֽחֶם׃
15דָוִ֖ד וַיֹּאמַ֑ר מִ֚י יַשְׁקֵ֣נִי מַ֔יִם מִבֹּ֥אר בֵּֽית־ לֶ֖חֶם אֲשֶׁ֥ר בַּשָּֽׁעַר׃
16שְׁלֹ֨שֶׁת הַגִּבֹּרִ֜ים בְּמַחֲנֵ֣ה פְלִשְׁתִּ֗ים וַיִּֽשְׁאֲבוּ־ מַ֙יִם֙ מִבֹּ֤אר בֵּֽית־ לֶ֙חֶם֙ אֲשֶׁ֣ר בַּשַּׁ֔עַר וַיִּשְׂא֖וּ וַיָּבִ֣אוּ אֶל־ דָּוִ֑ד וְלֹ֤א אָבָה֙ לִשְׁתּוֹתָ֔ם וַיַּסֵּ֥ךְ אֹתָ֖ם לַֽיהוָֽה׃
17חָלִילָה֩ לִּ֨י יְהוָ֜ה מֵעֲשֹׂ֣תִי זֹ֗את הֲדַ֤ם הָֽאֲנָשִׁים֙ הַהֹלְכִ֣ים בְּנַפְשׁוֹתָ֔ם וְלֹ֥א אָבָ֖ה לִשְׁתּוֹתָ֑ם אֵ֣לֶּה עָשׂ֔וּ שְׁלֹ֖שֶׁת הַגִּבֹּרִֽים׃ ס
18אֲחִ֣י ׀ יוֹאָ֣ב בֶּן־ צְרוּיָ֗ה ה֚וּא רֹ֣אשׁ [הַשְּׁלֹשִׁי כ] (הַשְּׁלֹשָׁ֔ה ק) וְהוּא֙ עוֹרֵ֣ר אֶת־ חֲנִית֔וֹ עַל־ שְׁלֹ֥שׁ מֵא֖וֹת חָלָ֑ל וְלוֹ־ שֵׁ֖ם בַּשְּׁלֹשָֽׁה׃
19הַשְּׁלֹשָׁה֙ הֲכִ֣י נִכְבָּ֔ד וַיְהִ֥י לָהֶ֖ם לְשָׂ֑ר וְעַד־ הַשְּׁלֹשָׁ֖ה לֹא־ בָֽא׃ ס
20בֶן־ יְהוֹיָדָ֧ע בֶּן־ אִֽישׁ־ [חַי כ] (חַ֛יִל ק) רַב־ פְּעָלִ֖ים מִֽקַּבְצְאֵ֑ל ה֣וּא הִכָּ֗ה אֵ֣ת שְׁנֵ֤י אֲרִאֵל֙ מוֹאָ֔ב וְ֠הוּא יָרַ֞ד וְהִכָּ֧ה אֶֽת־ [הָאֲרִיַּה כ] (הָאֲרִ֛י ק) בְּת֥וֹךְ הַבֹּ֖אר בְּי֥וֹם הַשָּֽׁלֶג׃
21הִכָּה֩ אֶת־ אִ֨ישׁ מִצְרִ֜י [אֲשֶׁר כ] (אִ֣ישׁ ק) מַרְאֶ֗ה וּבְיַ֤ד הַמִּצְרִי֙ חֲנִ֔ית וַיֵּ֥רֶד אֵלָ֖יו בַּשָּׁ֑בֶט וַיִּגְזֹ֤ל אֶֽת־ הַחֲנִית֙ מִיַּ֣ד הַמִּצְרִ֔י וַיַּהַרְגֵ֖הוּ בַּחֲנִיתֽוֹ׃
22עָשָׂ֔ה בְּנָיָ֖הוּ בֶּן־ יְהוֹיָדָ֑ע וְלוֹ־ שֵׁ֖ם בִּשְׁלֹשָׁ֥ה הַגִּבֹּרִֽים׃
23הַשְּׁלֹשִׁ֣ים נִכְבָּ֔ד וְאֶל־ הַשְּׁלֹשָׁ֖ה לֹא־ בָ֑א וַיְשִׂמֵ֥הוּ דָוִ֖ד אֶל־ מִשְׁמַעְתּֽוֹ׃ ס
24אֵ֥ל אֲחִֽי־ יוֹאָ֖ב בַּשְּׁלֹשִׁ֑ים אֶלְחָנָ֥ן בֶּן־ דֹּד֖וֹ בֵּ֥ית לָֽחֶם׃
25הַֽחֲרֹדִ֔י אֱלִיקָ֖א הַחֲרֹדִֽי׃ ס
26הַפַּלְטִ֔י עִירָ֥א בֶן־ עִקֵּ֖שׁ הַתְּקוֹעִֽי׃ ס
27הָֽעַנְּתֹתִ֔י מְבֻנַּ֖י הַחֻשָׁתִֽי׃ ס
28הָֽאֲחֹחִ֔י מַהְרַ֖י הַנְּטֹפָתִֽי׃ ס
29בֶּֽן־ בַּעֲנָ֖ה הַנְּטֹפָתִ֑י ס אִתַּי֙ בֶּן־ רִיבַ֔י מִגִּבְעַ֖ת בְּנֵ֥י בִנְיָמִֽן׃ ס
30פִּרְעָ֣תֹנִ֔י הִדַּ֖י מִנַּ֥חֲלֵי גָֽעַשׁ׃ ס
31עַלְבוֹן֙ הָֽעַרְבָתִ֔י עַזְמָ֖וֶת הַבַּרְחֻמִֽי׃ ס
32הַשַּׁ֣עַלְבֹנִ֔י בְּנֵ֥י יָשֵׁ֖ן יְהוֹנָתָֽן׃ ס
33הַֽהֲרָרִ֔י אֲחִיאָ֥ם בֶּן־ שָׁרָ֖ר הָארָרִֽי׃ ס
34בֶּן־ אֲחַסְבַּ֖י בֶּן־ הַמַּֽעֲכָתִ֑י ס אֱלִיעָ֥ם בֶּן־ אֲחִיתֹ֖פֶל הַגִּלֹנִֽי׃ ס
35כ] (חֶצְרַי֙ ק) הַֽכַּרְמְלִ֔י פַּעֲרַ֖י הָאַרְבִּֽי׃ ס
36בֶּן־ נָתָן֙ מִצֹּבָ֔ה ס בָּנִ֖י הַגָּדִֽי׃ ס
37הָעַמֹּנִ֑י ס נַחְרַי֙ הַבְּאֵ֣רֹתִ֔י [נֹשְׂאֵי כ] (נֹשֵׂ֕א ק) כְּלֵ֖י יוֹאָ֥ב בֶּן־ צְרֻיָֽה׃ ס
38הַיִּתְרִ֔י גָּרֵ֖ב הַיִּתְרִֽי׃ ס
