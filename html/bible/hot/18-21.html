1וַיַּ֥עַן אִיּ֗וֹב וַיֹּאמַֽר׃
2שָׁ֭מוֹעַ מִלָּתִ֑י וּתְהִי־ זֹ֝֗את תַּנְח֥וּמֹֽתֵיכֶֽם׃
3וְאָנֹכִ֣י אֲדַבֵּ֑ר וְאַחַ֖ר דַּבְּרִ֣י תַלְעִֽיג׃
4לְאָדָ֣ם שִׂיחִ֑י וְאִם־ מַ֝דּ֗וּעַ לֹא־ תִקְצַ֥ר רוּחִֽי׃
5אֵלַ֥י וְהָשַׁ֑מּוּ וְשִׂ֖ימוּ יָ֣ד עַל־ פֶּֽה׃
6זָכַ֥רְתִּי וְנִבְהָ֑לְתִּי וְאָחַ֥ז בְּ֝שָׂרִ֗י פַּלָּצֽוּת׃
7רְשָׁעִ֣ים יִחְי֑וּ עָ֝תְק֗וּ גַּם־ גָּ֥בְרוּ חָֽיִל׃
8נָכ֣וֹן לִפְנֵיהֶ֣ם עִמָּ֑ם וְ֝צֶאֱצָאֵיהֶ֗ם לְעֵינֵיהֶֽם׃
9שָׁל֣וֹם מִפָּ֑חַד וְלֹ֤א שֵׁ֖בֶט אֱל֣וֹהַּ עֲלֵיהֶֽם׃
10עִ֭בַּר וְלֹ֣א יַגְעִ֑ל תְּפַלֵּ֥ט פָּ֝רָת֗וֹ וְלֹ֣א תְשַׁכֵּֽל׃
11כַ֭צֹּאן עֲוִילֵיהֶ֑ם וְ֝יַלְדֵיהֶ֗ם יְרַקֵּדֽוּן׃
12כְּתֹ֣ף וְכִנּ֑וֹר וְ֝יִשְׂמְח֗וּ לְק֣וֹל עוּגָֽב׃
13כ] (יְכַלּ֣וּ ק) בַטּ֣וֹב יְמֵיהֶ֑ם וּ֝בְרֶ֗גַע שְׁא֣וֹל יֵחָֽתּוּ׃
14לָ֭אֵל ס֣וּר מִמֶּ֑נּוּ וְדַ֥עַת דְּ֝רָכֶ֗יךָ לֹ֣א חָפָֽצְנוּ׃
15שַׁדַּ֥י כִּֽי־ נַֽעַבְדֶ֑נּוּ וּמַה־ נּ֝וֹעִ֗יל כִּ֣י נִפְגַּע־ בּֽוֹ׃
16לֹ֣א בְיָדָ֣ם טוּבָ֑ם עֲצַ֥ת רְ֝שָׁעִ֗ים רָ֣חֲקָה מֶֽנִּי׃
17נֵר־ רְשָׁ֘עִ֤ים יִדְעָ֗ךְ וְיָבֹ֣א עָלֵ֣ימוֹ אֵידָ֑ם חֲ֝בָלִ֗ים יְחַלֵּ֥ק בְּאַפּֽוֹ׃
18כְּתֶ֥בֶן לִפְנֵי־ ר֑וּחַ וּ֝כְמֹ֗ץ גְּנָבַ֥תּוּ סוּפָֽה׃
19יִצְפֹּן־ לְבָנָ֥יו אוֹנ֑וֹ יְשַׁלֵּ֖ם אֵלָ֣יו וְיֵדָֽע׃
20[עֵינֹו כ] (עֵינָ֣יו ק) כִּיד֑וֹ וּמֵחֲמַ֖ת שַׁדַּ֣י יִשְׁתֶּֽה׃
21מַה־ חֶפְצ֣וֹ בְּבֵית֣וֹ אַחֲרָ֑יו וּמִסְפַּ֖ר חֳדָשָׁ֣יו חֻצָּֽצוּ׃
22יְלַמֶּד־ דָּ֑עַת וְ֝ה֗וּא רָמִ֥ים יִשְׁפּֽוֹט׃
23יָ֭מוּת בְּעֶ֣צֶם תֻּמּ֑וֹ כֻּ֝לּ֗וֹ שַׁלְאֲנַ֥ן וְשָׁלֵֽיו׃
24מָלְא֣וּ חָלָ֑ב וּמֹ֖חַ עַצְמוֹתָ֣יו יְשֻׁקֶּֽה׃
25יָ֭מוּת בְּנֶ֣פֶשׁ מָרָ֑ה וְלֹֽא־ אָ֝כַ֗ל בַּטּוֹבָֽה׃
26עַל־ עָפָ֣ר יִשְׁכָּ֑בוּ וְ֝רִמָּ֗ה תְּכַסֶּ֥ה עֲלֵיהֶֽם׃
27יָ֭דַעְתִּי מַחְשְׁבֽוֹתֵיכֶ֑ם וּ֝מְזִמּ֗וֹת עָלַ֥י תַּחְמֹֽסוּ׃
28תֹֽאמְר֗וּ אַיֵּ֥ה בֵית־ נָדִ֑יב וְ֝אַיֵּ֗ה אֹ֤הֶל ׀ מִשְׁכְּנ֬וֹת רְשָׁעִֽים׃
29שְׁ֭אֶלְתֶּם ע֣וֹבְרֵי דָ֑רֶךְ וְ֝אֹתֹתָ֗ם לֹ֣א תְנַכֵּֽרוּ׃
30לְי֣וֹם אֵ֭יד יֵחָ֣שֶׂךְ רָ֑ע לְי֖וֹם עֲבָר֣וֹת יוּבָֽלוּ׃
31יַגִּ֣יד עַל־ פָּנָ֣יו דַּרְכּ֑וֹ וְהֽוּא־ עָ֝שָׂ֗ה מִ֣י יְשַׁלֶּם־ לֽוֹ׃
32לִקְבָר֣וֹת יוּבָ֑ל וְֽעַל־ גָּדִ֥ישׁ יִשְׁקֽוֹד׃
33ל֗וֹ רִגְבֵ֫י נָ֥חַל וְ֭אַחֲרָיו כָּל־ אָדָ֣ם יִמְשׁ֑וֹךְ וּ֝לְפָנָ֗יו אֵ֣ין מִסְפָּֽר׃
