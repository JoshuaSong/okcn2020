1הַ֥לְלוּ יָ֨הּ ׀ כִּי־ ט֭וֹב זַמְּרָ֣ה אֱלֹהֵ֑ינוּ כִּֽי־ נָ֝עִים נָאוָ֥ה תְהִלָּֽה׃
2יְרוּשָׁלִַ֣ם יְהוָ֑ה נִדְחֵ֖י יִשְׂרָאֵ֣ל יְכַנֵּֽס׃
3לִשְׁב֣וּרֵי לֵ֑ב וּ֝מְחַבֵּ֗שׁ לְעַצְּבוֹתָֽם׃
4מִ֭סְפָּר לַכּוֹכָבִ֑ים לְ֝כֻלָּ֗ם שֵׁמ֥וֹת יִקְרָֽא׃
5אֲדוֹנֵ֣ינוּ וְרַב־ כֹּ֑חַ לִ֝תְבוּנָת֗וֹ אֵ֣ין מִסְפָּֽר׃
6עֲנָוִ֣ים יְהוָ֑ה מַשְׁפִּ֖יל רְשָׁעִ֣ים עֲדֵי־ אָֽרֶץ׃
7לַיהוָ֣ה בְּתוֹדָ֑ה זַמְּר֖וּ לֵאלֹהֵ֣ינוּ בְכִנּֽוֹר׃
8שָׁמַ֨יִם ׀ בְּעָבִ֗ים הַמֵּכִ֣ין לָאָ֣רֶץ מָטָ֑ר הַמַּצְמִ֖יחַ הָרִ֣ים חָצִֽיר׃
9לִבְהֵמָ֣ה לַחְמָ֑הּ לִבְנֵ֥י עֹ֝רֵ֗ב אֲשֶׁ֣ר יִקְרָֽאוּ׃
10בִגְבוּרַ֣ת הַסּ֣וּס יֶחְפָּ֑ץ לֹֽא־ בְשׁוֹקֵ֖י הָאִ֣ישׁ יִרְצֶֽה׃
11יְ֭הוָה אֶת־ יְרֵאָ֑יו אֶת־ הַֽמְיַחֲלִ֥ים לְחַסְדּֽוֹ׃
12יְ֭רוּשָׁלִַם אֶת־ יְהוָ֑ה הַֽלְלִ֖י אֱלֹהַ֣יִךְ צִיּֽוֹן׃
13חִ֭זַּק בְּרִיחֵ֣י שְׁעָרָ֑יִךְ בֵּרַ֖ךְ בָּנַ֣יִךְ בְּקִרְבֵּֽךְ׃
14גְּבוּלֵ֥ךְ שָׁל֑וֹם חֵ֥לֶב חִ֝טִּ֗ים יַשְׂבִּיעֵֽךְ׃
15אִמְרָת֣וֹ אָ֑רֶץ עַד־ מְ֝הֵרָ֗ה יָר֥וּץ דְּבָרֽוֹ׃
16שֶׁ֣לֶג כַּצָּ֑מֶר כְּ֝פ֗וֹר כָּאֵ֥פֶר יְפַזֵּֽר׃
17קַֽרְח֣וֹ כְפִתִּ֑ים לִפְנֵ֥י קָ֝רָת֗וֹ מִ֣י יַעֲמֹֽד׃
18דְּבָר֣וֹ וְיַמְסֵ֑ם יַשֵּׁ֥ב ר֝וּח֗וֹ יִזְּלוּ־ מָֽיִם׃
19[דְּבָרֹו כ] (דְּבָרָ֣יו ק) לְיַעֲקֹ֑ב חֻקָּ֥יו וּ֝מִשְׁפָּטָ֗יו לְיִשְׂרָאֵֽל׃
