1נִבְחָ֣ר שֵׁ֭ם מֵעֹ֣שֶׁר רָ֑ב מִכֶּ֥סֶף וּ֝מִזָּהָ֗ב חֵ֣ן טֽוֹב׃
2וָרָ֣שׁ נִפְגָּ֑שׁוּ עֹשֵׂ֖ה כֻלָּ֣ם יְהוָֽה׃
3רָאָ֣ה רָעָ֣ה [וְיִסָּתֵר כ] (וְנִסְתָּ֑ר ק) וּ֝פְתָיִ֗ים עָבְר֥וּ וְֽנֶעֱנָֽשׁוּ׃
4עֲ֭נָוָה יִרְאַ֣ת יְהוָ֑ה עֹ֖שֶׁר וְכָב֣וֹד וְחַיִּֽים׃
5פַּ֭חִים בְּדֶ֣רֶךְ עִקֵּ֑שׁ שׁוֹמֵ֥ר נַ֝פְשׁ֗וֹ יִרְחַ֥ק מֵהֶֽם׃
6לַ֭נַּעַר עַל־ פִּ֣י דַרְכּ֑וֹ גַּ֥ם כִּֽי־ יַ֝זְקִ֗ין לֹֽא־ יָס֥וּר מִמֶּֽנָּה׃
7בְּרָשִׁ֣ים יִמְשׁ֑וֹל וְעֶ֥בֶד לֹ֝וֶ֗ה לְאִ֣ישׁ מַלְוֶֽה׃
8עַ֭וְלָה [יִקְצֹור־ כ] (יִקְצָר־ אָ֑וֶן ק) וְשֵׁ֖בֶט עֶבְרָת֣וֹ יִכְלֶֽה׃
9עַ֭יִן ה֣וּא יְבֹרָ֑ךְ כִּֽי־ נָתַ֖ן מִלַּחְמ֣וֹ לַדָּֽל׃
10לֵ֭ץ וְיֵצֵ֣א מָד֑וֹן וְ֝יִשְׁבֹּ֗ת דִּ֣ין וְקָלֽוֹן׃
11[טְהֹור־ כ] (טְהָר־ לֵ֑ב ק) חֵ֥ן שְׂ֝פָתָ֗יו רֵעֵ֥הוּ מֶֽלֶךְ׃
12יְ֭הוָה נָ֣צְרוּ דָ֑עַת וַ֝יְסַלֵּ֗ף דִּבְרֵ֥י בֹגֵֽד׃
13עָ֭צֵל אֲרִ֣י בַח֑וּץ בְּת֥וֹךְ רְ֝חֹב֗וֹת אֵֽרָצֵֽחַ׃
14עֲ֭מֻקָּה פִּ֣י זָר֑וֹת זְע֥וּם יְ֝הוָ֗ה [יִפֹּול־ כ] (יִפָּל־ שָֽׁם׃ ק)
15קְשׁוּרָ֣ה בְלֶב־ נָ֑עַר שֵׁ֥בֶט מ֝וּסָ֗ר יַרְחִיקֶ֥נָּה מִמֶּֽנּוּ׃
16דָּ֭ל לְהַרְבּ֣וֹת ל֑וֹ נֹתֵ֥ן לְ֝עָשִׁ֗יר אַךְ־ לְמַחְסֽוֹר׃
17אָזְנְךָ֗ וּ֭שְׁמַע דִּבְרֵ֣י חֲכָמִ֑ים וְ֝לִבְּךָ֗ תָּשִׁ֥ית לְדַעְתִּֽי׃
18נָ֭עִים כִּֽי־ תִשְׁמְרֵ֣ם בְּבִטְנֶ֑ךָ יִכֹּ֥נוּ יַ֝חְדָּ֗ו עַל־ שְׂפָתֶֽיךָ׃
19בַּ֭יהוָה מִבְטַחֶ֑ךָ הוֹדַעְתִּ֖יךָ הַיּ֣וֹם אַף־ אָֽתָּה׃
20כָתַ֣בְתִּי לְ֭ךָ [שִׁלְשֹׁום כ] (שָׁלִישִׁ֑ים ק) בְּמ֖וֹעֵצֹ֣ת וָדָֽעַת׃
21קֹ֭שְׁטְ אִמְרֵ֣י אֱמֶ֑ת לְהָשִׁ֥יב אֲמָרִ֥ים אֱ֝מֶ֗ת לְשֹׁלְחֶֽיךָ׃ פ אַֽל־ תִּגְזָל־
22כִּ֣י דַל־ ה֑וּא וְאַל־ תְּדַכֵּ֖א עָנִ֣י בַשָּֽׁעַר׃
23יְ֭הוָה יָרִ֣יב רִיבָ֑ם וְקָבַ֖ע אֶת־ קֹבְעֵיהֶ֣ם נָֽפֶשׁ׃
24תִּ֭תְרַע אֶת־ בַּ֣עַל אָ֑ף וְאֶת־ אִ֥ישׁ חֵ֝מוֹת לֹ֣א תָבֽוֹא׃
25תֶּאֱלַ֥ף [אָרְחָתֹו כ] (אֹֽרְחֹתָ֑יו ק) וְלָקַחְתָּ֖ מוֹקֵ֣שׁ לְנַפְשֶֽׁךָ׃
26תְּהִ֥י בְתֹֽקְעֵי־ כָ֑ף בַּ֝עֹרְבִ֗ים מַשָּׁאֽוֹת׃ אִם־ אֵֽין־
27לְשַׁלֵּ֑ם לָ֥מָּה יִקַּ֥ח מִ֝שְׁכָּבְךָ֗ מִתַּחְתֶּֽיךָ׃
28תַּ֭סֵּג גְּב֣וּל עוֹלָ֑ם אֲשֶׁ֖ר עָשׂ֣וּ אֲבוֹתֶֽיךָ׃
