1וְאַתָּ֤ה בֶן־ אָדָם֙ קַח־ לְךָ֣ לְבֵנָ֔ה וְנָתַתָּ֥ה אוֹתָ֖הּ לְפָנֶ֑יךָ וְחַקּוֹתָ֥ עָלֶ֛יהָ עִ֖יר אֶת־ יְרוּשָׁלִָֽם׃
2עָלֶ֜יהָ מָצ֗וֹר וּבָנִ֤יתָ עָלֶ֙יהָ֙ דָּיֵ֔ק וְשָׁפַכְתָּ֥ עָלֶ֖יהָ סֹֽלְלָ֑ה וְנָתַתָּ֨ה עָלֶ֧יהָ מַחֲנ֛וֹת וְשִׂים־ עָלֶ֥יהָ כָּרִ֖ים סָבִֽיב׃
3קַח־ לְךָ֙ מַחֲבַ֣ת בַּרְזֶ֔ל וְנָתַתָּ֤ה אוֹתָהּ֙ קִ֣יר בַּרְזֶ֔ל בֵּינְךָ֖ וּבֵ֣ין הָעִ֑יר וַהֲכִינֹתָה֩ אֶת־ פָּנֶ֨יךָ אֵלֶ֜יהָ וְהָיְתָ֤ה בַמָּצוֹר֙ וְצַרְתָּ֣ עָלֶ֔יהָ א֥וֹת הִ֖יא לְבֵ֥ית יִשְׂרָאֵֽל׃ ס
4שְׁכַב֙ עַל־ צִדְּךָ֣ הַשְּׂמָאלִ֔י וְשַׂמְתָּ֛ אֶת־ עֲוֹ֥ן בֵּֽית־ יִשְׂרָאֵ֖ל עָלָ֑יו מִסְפַּ֤ר הַיָּמִים֙ אֲשֶׁ֣ר תִּשְׁכַּ֣ב עָלָ֔יו תִּשָּׂ֖א אֶת־ עֲוֹנָֽם׃
5נָתַ֤תִּֽי לְךָ֙ אֶת־ שְׁנֵ֣י עֲוֹנָ֔ם לְמִסְפַּ֣ר יָמִ֔ים שְׁלֹשׁ־ מֵא֥וֹת וְתִשְׁעִ֖ים י֑וֹם וְנָשָׂ֖אתָ עֲוֹ֥ן בֵּֽית־ יִשְׂרָאֵֽל׃
6אֶת־ אֵ֗לֶּה וְשָׁ֨כַבְתָּ֜ עַל־ צִדְּךָ֤ [הַיְמֹונִי כ] (הַיְמָנִי֙ ק) שֵׁנִ֔ית וְנָשָׂ֖אתָ אֶת־ עֲוֹ֣ן בֵּית־ יְהוּדָ֑ה אַרְבָּעִ֣ים י֔וֹם י֧וֹם לַשָּׁנָ֛ה י֥וֹם לַשָּׁנָ֖ה נְתַתִּ֥יו לָֽךְ׃
7מְצ֤וֹר יְרוּשָׁלִַ֙ם֙ תָּכִ֣ין פָּנֶ֔יךָ וּֽזְרֹעֲךָ֖ חֲשׂוּפָ֑ה וְנִבֵּאתָ֖ עָלֶֽיהָ׃
8נָתַ֥תִּי עָלֶ֖יךָ עֲבוֹתִ֑ים וְלֹֽא־ תֵהָפֵ֤ךְ מִֽצִּדְּךָ֙ אֶל־ צִדֶּ֔ךָ עַד־ כַּלּוֹתְךָ֖ יְמֵ֥י מְצוּרֶֽךָ׃
9קַח־ לְךָ֡ חִטִּ֡ין וּ֠שְׂעֹרִים וּפ֨וֹל וַעֲדָשִׁ֜ים וְדֹ֣חַן וְכֻסְּמִ֗ים וְנָתַתָּ֤ה אוֹתָם֙ בִּכְלִ֣י אֶחָ֔ד וְעָשִׂ֧יתָ אוֹתָ֛ם לְךָ֖ לְלָ֑חֶם מִסְפַּ֨ר הַיָּמִ֜ים אֲשֶׁר־ אַתָּ֣ה ׀ שׁוֹכֵ֣ב עַֽל־ צִדְּךָ֗ שְׁלֹשׁ־ מֵא֧וֹת וְתִשְׁעִ֛ים י֖וֹם תֹּאכֲלֶֽנּוּ׃
10אֲשֶׁ֣ר תֹּאכֲלֶ֔נּוּ בְּמִשְׁק֕וֹל עֶשְׂרִ֥ים שֶׁ֖קֶל לַיּ֑וֹם מֵעֵ֥ת עַד־ עֵ֖ת תֹּאכֲלֶֽנּוּ׃
11בִּמְשׂוּרָ֥ה תִשְׁתֶּ֖ה שִׁשִּׁ֣ית הַהִ֑ין מֵעֵ֥ת עַד־ עֵ֖ת תִּשְׁתֶּֽה׃
12שְׂעֹרִ֖ים תֹּֽאכֲלֶ֑נָּה וְהִ֗יא בְּגֶֽלְלֵי֙ צֵאַ֣ת הָֽאָדָ֔ם תְּעֻגֶ֖נָה לְעֵינֵיהֶֽם׃ ס
13יְהוָ֔ה כָּ֣כָה יֹאכְל֧וּ בְנֵֽי־ יִשְׂרָאֵ֛ל אֶת־ לַחְמָ֖ם טָמֵ֑א בַּגּוֹיִ֕ם אֲשֶׁ֥ר אַדִּיחֵ֖ם שָֽׁם׃
14אֲהָהּ֙ אֲדֹנָ֣י יְהוִ֔ה הִנֵּ֥ה נַפְשִׁ֖י לֹ֣א מְטֻמָּאָ֑ה וּנְבֵלָ֨ה וּטְרֵפָ֤ה לֹֽא־ אָכַ֙לְתִּי֙ מִנְּעוּרַ֣י וְעַד־ עַ֔תָּה וְלֹא־ בָ֥א בְּפִ֖י בְּשַׂ֥ר פִּגּֽוּל׃ ס
15אֵלַ֔י רְאֵ֗ה נָתַ֤תִּֽי לְךָ֙ אֶת־ [צְפוּעֵי כ] (צְפִיעֵ֣י ק) הַבָּקָ֔ר תַּ֖חַת גֶּלְלֵ֣י הָֽאָדָ֑ם וְעָשִׂ֥יתָ אֶֽת־ לַחְמְךָ֖ עֲלֵיהֶֽם׃ ס
16אֵלַ֗י בֶּן־ אָדָם֙ הִנְנִ֨י שֹׁבֵ֤ר מַטֵּה־ לֶ֙חֶם֙ בִּיר֣וּשָׁלִַ֔ם וְאָכְלוּ־ לֶ֥חֶם בְּמִשְׁקָ֖ל וּבִדְאָגָ֑ה וּמַ֕יִם בִּמְשׂוּרָ֥ה וּבְשִׁמָּמ֖וֹן יִשְׁתּֽוּ׃
