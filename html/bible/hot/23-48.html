1שִׁמְעוּ־ זֹ֣את בֵּֽית־ יַעֲקֹ֗ב הַנִּקְרָאִים֙ בְּשֵׁ֣ם יִשְׂרָאֵ֔ל וּמִמֵּ֥י יְהוּדָ֖ה יָצָ֑אוּ הַֽנִּשְׁבָּעִ֣ים ׀ בְּשֵׁ֣ם יְהוָ֗ה וּבֵאלֹהֵ֤י יִשְׂרָאֵל֙ יַזְכִּ֔ירוּ לֹ֥א בֶאֱמֶ֖ת וְלֹ֥א בִצְדָקָֽה׃
2מֵעִ֤יר הַקֹּ֙דֶשׁ֙ נִקְרָ֔אוּ וְעַל־ אֱלֹהֵ֥י יִשְׂרָאֵ֖ל נִסְמָ֑כוּ יְהוָ֥ה צְבָא֖וֹת שְׁמֽוֹ׃ ס
3מֵאָ֣ז הִגַּ֔דְתִּי וּמִפִּ֥י יָצְא֖וּ וְאַשְׁמִיעֵ֑ם פִּתְאֹ֥ם עָשִׂ֖יתִי וַתָּבֹֽאנָה׃
4כִּ֥י קָשֶׁ֖ה אָ֑תָּה וְגִ֤יד בַּרְזֶל֙ עָרְפֶּ֔ךָ וּמִצְחֲךָ֖ נְחוּשָֽׁה׃
5לְךָ֙ מֵאָ֔ז בְּטֶ֥רֶם תָּב֖וֹא הִשְׁמַעְתִּ֑יךָ פֶּן־ תֹּאמַר֙ עָצְבִּ֣י עָשָׂ֔ם וּפִסְלִ֥י וְנִסְכִּ֖י צִוָּֽם׃
6חֲזֵה֙ כֻּלָּ֔הּ וְאַתֶּ֖ם הֲל֣וֹא תַגִּ֑ידוּ הִשְׁמַעְתִּ֤יךָ חֲדָשׁוֹת֙ מֵעַ֔תָּה וּנְצֻר֖וֹת וְלֹ֥א יְדַעְתָּֽם׃
7נִבְרְאוּ֙ וְלֹ֣א מֵאָ֔ז וְלִפְנֵי־ י֖וֹם וְלֹ֣א שְׁמַעְתָּ֑ם פֶּן־ תֹּאמַ֖ר הִנֵּ֥ה יְדַעְתִּֽין׃
8לֹֽא־ שָׁמַ֗עְתָּ גַּ֚ם לֹ֣א יָדַ֔עְתָּ גַּ֕ם מֵאָ֖ז לֹא־ פִתְּחָ֣ה אָזְנֶ֑ךָ כִּ֤י יָדַ֙עְתִּי֙ בָּג֣וֹד תִּבְגּ֔וֹד וּפֹשֵׁ֥עַ מִבֶּ֖טֶן קֹ֥רָא לָֽךְ׃
9שְׁמִי֙ אַאֲרִ֣יךְ אַפִּ֔י וּתְהִלָּתִ֖י אֶחֱטָם־ לָ֑ךְ לְבִלְתִּ֖י הַכְרִיתֶֽךָ׃
10צְרַפְתִּ֖יךָ וְלֹ֣א בְכָ֑סֶף בְּחַרְתִּ֖יךָ בְּכ֥וּר עֹֽנִי׃
11לְמַעֲנִ֛י אֶעֱשֶׂ֖ה כִּ֣י אֵ֣יךְ יֵחָ֑ל וּכְבוֹדִ֖י לְאַחֵ֥ר לֹֽא־ אֶתֵּֽן׃ ס
12אֵלַי֙ יַֽעֲקֹ֔ב וְיִשְׂרָאֵ֖ל מְקֹרָאִ֑י אֲנִי־ הוּא֙ אֲנִ֣י רִאשׁ֔וֹן אַ֖ף אֲנִ֥י אַחֲרֽוֹן׃
13יָדִי֙ יָ֣סְדָה אֶ֔רֶץ וִֽימִינִ֖י טִפְּחָ֣ה שָׁמָ֑יִם קֹרֵ֥א אֲנִ֛י אֲלֵיהֶ֖ם יַעַמְד֥וּ יַחְדָּֽו׃
14כֻלְּכֶם֙ וּֽשֲׁמָ֔עוּ מִ֥י בָהֶ֖ם הִגִּ֣יד אֶת־ אֵ֑לֶּה יְהוָ֣ה אֲהֵב֔וֹ יַעֲשֶׂ֤ה חֶפְצוֹ֙ בְּבָבֶ֔ל וּזְרֹע֖וֹ כַּשְׂדִּֽים׃
15אֲנִ֛י דִּבַּ֖רְתִּי אַף־ קְרָאתִ֑יו הֲבִיאֹתִ֖יו וְהִצְלִ֥יחַ דַּרְכּֽוֹ׃
16אֵלַ֣י שִׁמְעוּ־ זֹ֗את לֹ֤א מֵרֹאשׁ֙ בַּסֵּ֣תֶר דִּבַּ֔רְתִּי מֵעֵ֥ת הֱיוֹתָ֖הּ שָׁ֣ם אָ֑נִי וְעַתָּ֗ה אֲדֹנָ֧י יְהוִ֛ה שְׁלָחַ֖נִי וְרוּחֽוֹ׃ פ
17אָמַ֧ר יְהוָ֛ה גֹּאַלְךָ֖ קְד֣וֹשׁ יִשְׂרָאֵ֑ל אֲנִ֨י יְהוָ֤ה אֱלֹהֶ֙יךָ֙ מְלַמֶּדְךָ֣ לְהוֹעִ֔יל מַדְרִֽיכֲךָ֖ בְּדֶ֥רֶךְ תֵּלֵֽךְ׃
18הִקְשַׁ֖בְתָּ לְמִצְוֹתָ֑י וַיְהִ֤י כַנָּהָר֙ שְׁלוֹמֶ֔ךָ וְצִדְקָתְךָ֖ כְּגַלֵּ֥י הַיָּֽם׃
19כַחוֹל֙ זַרְעֶ֔ךָ וְצֶאֱצָאֵ֥י מֵעֶ֖יךָ כִּמְעֹתָ֑יו לֹֽא־ יִכָּרֵ֧ת וְֽלֹא־ יִשָּׁמֵ֛ד שְׁמ֖וֹ מִלְּפָנָֽי׃
20מִבָּבֶל֮ בִּרְח֣וּ מִכַּשְׂדִּים֒ בְּק֣וֹל רִנָּ֗ה הַגִּ֤ידוּ הַשְׁמִ֙יעוּ֙ זֹ֔את הוֹצִיא֖וּהָ עַד־ קְצֵ֣ה הָאָ֑רֶץ אִמְר֕וּ גָּאַ֥ל יְהוָ֖ה עַבְדּ֥וֹ יַעֲקֹֽב׃
21צָמְא֗וּ בָּחֳרָבוֹת֙ הֽוֹלִיכָ֔ם מַ֥יִם מִצּ֖וּר הִזִּ֣יל לָ֑מוֹ וַיִּ֨בְקַע־ צ֔וּר וַיָּזֻ֖בוּ מָֽיִם׃
