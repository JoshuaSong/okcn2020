1הוֹד֣וּ לַ֭יהוָה קִרְא֣וּ בִּשְׁמ֑וֹ הוֹדִ֥יעוּ בָ֝עַמִּ֗ים עֲלִילוֹתָֽיו׃
2ל֭וֹ זַמְּרוּ־ ל֑וֹ שִׂ֝֗יחוּ בְּכָל־ נִפְלְאוֹתָֽיו׃
3בְּשֵׁ֣ם קָדְשׁ֑וֹ יִ֝שְׂמַ֗ח לֵ֤ב ׀ מְבַקְשֵׁ֬י יְהוָֽה׃
4יְהוָ֣ה וְעֻזּ֑וֹ בַּקְּשׁ֖וּ פָנָ֣יו תָּמִֽיד׃
5נִפְלְאוֹתָ֥יו אֲשֶׁר־ עָשָׂ֑ה מֹ֝פְתָ֗יו וּמִשְׁפְּטֵי־ פִֽיו׃
6אַבְרָהָ֣ם עַבְדּ֑וֹ בְּנֵ֖י יַעֲקֹ֣ב בְּחִירָֽיו׃
7יְהוָ֣ה אֱלֹהֵ֑ינוּ בְּכָל־ הָ֝אָ֗רֶץ מִשְׁפָּטָֽיו׃
8לְעוֹלָ֣ם בְּרִית֑וֹ דָּבָ֥ר צִ֝וָּ֗ה לְאֶ֣לֶף דּֽוֹר׃
9כָּ֭רַת אֶת־ אַבְרָהָ֑ם וּשְׁב֖וּעָת֣וֹ לְיִשְׂחָֽק׃
10לְיַעֲקֹ֣ב לְחֹ֑ק לְ֝יִשְׂרָאֵ֗ל בְּרִ֣ית עוֹלָֽם׃
11לְךָ֗ אֶתֵּ֥ן אֶת־ אֶֽרֶץ־ כְּנָ֑עַן חֶ֝֗בֶל נַחֲלַתְכֶֽם׃
12מְתֵ֣י מִסְפָּ֑ר כִּ֝מְעַ֗ט וְגָרִ֥ים בָּֽהּ׃
13מִגּ֣וֹי אֶל־ גּ֑וֹי מִ֝מַּמְלָכָ֗ה אֶל־ עַ֥ם אַחֵֽר׃
14הִנִּ֣יחַ אָדָ֣ם לְעָשְׁקָ֑ם וַיּ֖וֹכַח עֲלֵיהֶ֣ם מְלָכִֽים׃
15תִּגְּע֥וּ בִמְשִׁיחָ֑י וְ֝לִנְבִיאַי אַל־ תָּרֵֽעוּ׃
16רָ֭עָב עַל־ הָאָ֑רֶץ כָּֽל־ מַטֵּה־ לֶ֥חֶם שָׁבָֽר׃
17לִפְנֵיהֶ֣ם אִ֑ישׁ לְ֝עֶ֗בֶד נִמְכַּ֥ר יוֹסֵֽף׃
18בַכֶּ֣בֶל [רַגְלָיו כ] (רַגְלֹ֑ו ק) בַּ֝רְזֶ֗ל בָּ֣אָה נַפְשֽׁוֹ׃
19עֵ֥ת בֹּֽא־ דְבָר֑וֹ אִמְרַ֖ת יְהוָ֣ה צְרָפָֽתְהוּ׃
20מֶ֭לֶךְ וַיַּתִּירֵ֑הוּ מֹשֵׁ֥ל עַ֝מִּ֗ים וַֽיְפַתְּחֵֽהוּ׃
21אָד֣וֹן לְבֵית֑וֹ וּ֝מֹשֵׁ֗ל בְּכָל־ קִנְיָנֽוֹ׃
22שָׂרָ֣יו בְּנַפְשׁ֑וֹ וּזְקֵנָ֥יו יְחַכֵּֽם׃
23יִשְׂרָאֵ֣ל מִצְרָ֑יִם וְ֝יַעֲקֹ֗ב גָּ֣ר בְּאֶֽרֶץ־ חָֽם׃
24אֶת־ עַמּ֣וֹ מְאֹ֑ד וַ֝יַּֽעֲצִמֵהוּ מִצָּרָֽיו׃
25לִ֭בָּם לִשְׂנֹ֣א עַמּ֑וֹ לְ֝הִתְנַכֵּ֗ל בַּעֲבָדָֽיו׃
26מֹשֶׁ֣ה עַבְדּ֑וֹ אַ֝הֲרֹ֗ן אֲשֶׁ֣ר בָּֽחַר־ בּֽוֹ׃
27בָ֭ם דִּבְרֵ֣י אֹתוֹתָ֑יו וּ֝מֹפְתִ֗ים בְּאֶ֣רֶץ חָֽם׃
28חֹ֭שֶׁךְ וַיַּחְשִׁ֑ךְ וְלֹֽא־ מָ֝ר֗וּ אֶת־ [דְּבָרָוו כ] (דְּבָרֹֽו׃ ק)
29אֶת־ מֵימֵיהֶ֣ם לְדָ֑ם וַ֝יָּ֗מֶת אֶת־ דְּגָתָֽם׃
30אַרְצָ֣ם צְפַרְדְּעִ֑ים בְּ֝חַדְרֵ֗י מַלְכֵיהֶֽם׃
31וַיָּבֹ֣א עָרֹ֑ב כִּ֝נִּ֗ים בְּכָל־ גְּבוּלָֽם׃
32גִּשְׁמֵיהֶ֣ם בָּרָ֑ד אֵ֖שׁ לֶהָב֣וֹת בְּאַרְצָֽם׃
33גַּ֭פְנָם וּתְאֵנָתָ֑ם וַ֝יְשַׁבֵּ֗ר עֵ֣ץ גְּבוּלָֽם׃
34וַיָּבֹ֣א אַרְבֶּ֑ה וְ֝יֶ֗לֶק וְאֵ֣ין מִסְפָּֽר׃
35כָּל־ עֵ֣שֶׂב בְּאַרְצָ֑ם וַ֝יֹּ֗אכַל פְּרִ֣י אַדְמָתָֽם׃
36כָּל־ בְּכ֣וֹר בְּאַרְצָ֑ם רֵ֝אשִׁ֗ית לְכָל־ אוֹנָֽם׃
37בְּכֶ֣סֶף וְזָהָ֑ב וְאֵ֖ין בִּשְׁבָטָ֣יו כּוֹשֵֽׁל׃
38מִצְרַ֣יִם בְּצֵאתָ֑ם כִּֽי־ נָפַ֖ל פַּחְדָּ֣ם עֲלֵיהֶֽם׃
39עָנָ֣ן לְמָסָ֑ךְ וְ֝אֵ֗שׁ לְהָאִ֥יר לָֽיְלָה׃
40וַיָּבֵ֣א שְׂלָ֑ו וְלֶ֥חֶם שָׁ֝מַ֗יִם יַשְׂבִּיעֵֽם׃
41צ֭וּר וַיָּז֣וּבוּ מָ֑יִם הָ֝לְכ֗וּ בַּצִּיּ֥וֹת נָהָֽר׃
42זָ֭כַר אֶת־ דְּבַ֣ר קָדְשׁ֑וֹ אֶֽת־ אַבְרָהָ֥ם עַבְדּֽוֹ׃
43עַמּ֣וֹ בְשָׂשׂ֑וֹן בְּ֝רִנָּ֗ה אֶת־ בְּחִירָֽיו׃
44לָ֭הֶם אַרְצ֣וֹת גּוֹיִ֑ם וַעֲמַ֖ל לְאֻמִּ֣ים יִירָֽשׁוּ׃
