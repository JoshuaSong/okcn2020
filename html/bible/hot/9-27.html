1וַיֹּ֤אמֶר דָּוִד֙ אֶל־ לִבּ֔וֹ עַתָּ֛ה אֶסָּפֶ֥ה יוֹם־ אֶחָ֖ד בְּיַד־ שָׁא֑וּל אֵֽין־ לִ֨י ט֜וֹב כִּ֣י הִמָּלֵ֥ט אִמָּלֵ֣ט ׀ אֶל־ אֶ֣רֶץ פְּלִשְׁתִּ֗ים וְנוֹאַ֨שׁ מִמֶּ֤נִּי שָׁאוּל֙ לְבַקְשֵׁ֤נִי עוֹד֙ בְּכָל־ גְּב֣וּל יִשְׂרָאֵ֔ל וְנִמְלַטְתִּ֖י מִיָּדֽוֹ׃
2דָּוִ֔ד וַיַּעֲבֹ֣ר ה֔וּא וְשֵׁשׁ־ מֵא֥וֹת אִ֖ישׁ אֲשֶׁ֣ר עִמּ֑וֹ אֶל־ אָכִ֥ישׁ בֶּן־ מָע֖וֹךְ מֶ֥לֶךְ גַּֽת׃
3דָּוִ֨ד עִם־ אָכִ֥ישׁ בְּגַ֛ת ה֥וּא וַאֲנָשָׁ֖יו אִ֣ישׁ וּבֵית֑וֹ דָּוִד֙ וּשְׁתֵּ֣י נָשָׁ֔יו אֲחִינֹ֙עַם֙ הַיִּזְרְעֵאלִ֔ית וַאֲבִיגַ֥יִל אֵֽשֶׁת־ נָבָ֖ל הַֽכַּרְמְלִֽית׃
4לְשָׁא֔וּל כִּֽי־ בָרַ֥ח דָּוִ֖ד גַּ֑ת וְלֹֽא־ [יֹוסַף כ] (יָסַ֥ף ק) ע֖וֹד לְבַקְשֽׁוֹ׃ ס
5דָּוִ֜ד אֶל־ אָכִ֗ישׁ אִם־ נָא֩ מָצָ֨אתִי חֵ֤ן בְּעֵינֶ֙יךָ֙ יִתְּנוּ־ לִ֣י מָק֗וֹם בְּאַחַ֛ת עָרֵ֥י הַשָּׂדֶ֖ה וְאֵ֣שְׁבָה שָּׁ֑ם וְלָ֨מָּה יֵשֵׁ֧ב עַבְדְּךָ֛ בְּעִ֥יר הַמַּמְלָכָ֖ה עִמָּֽךְ׃
6ל֥וֹ אָכִ֛ישׁ בַּיּ֥וֹם הַה֖וּא אֶת־ צִֽקְלָ֑ג לָכֵ֞ן הָיְתָ֤ה צִֽקְלַג֙ לְמַלְכֵ֣י יְהוּדָ֔ה עַ֖ד הַיּ֥וֹם הַזֶּֽה׃ פ
7מִסְפַּ֣ר הַיָּמִ֔ים אֲשֶׁר־ יָשַׁ֥ב דָּוִ֖ד בִּשְׂדֵ֣ה פְלִשְׁתִּ֑ים יָמִ֖ים וְאַרְבָּעָ֥ה חֳדָשִֽׁים׃
8דָּוִד֙ וַֽאֲנָשָׁ֔יו וַֽיִּפְשְׁט֛וּ אֶל־ הַגְּשׁוּרִ֥י [וְהַגִּרְזִי כ] (וְהַגִּזְרִ֖י ק) וְהָעֲמָלֵקִ֑י כִּ֣י הֵ֜נָּה יֹשְׁב֤וֹת הָאָ֙רֶץ֙ אֲשֶׁ֣ר מֵֽעוֹלָ֔ם בּוֹאֲךָ֥ שׁ֖וּרָה וְעַד־ אֶ֥רֶץ מִצְרָֽיִם׃
9דָוִד֙ אֶת־ הָאָ֔רֶץ וְלֹ֥א יְחַיֶּ֖ה אִ֣ישׁ וְאִשָּׁ֑ה וְלָקַח֩ צֹ֨אן וּבָקָ֜ר וַחֲמֹרִ֤ים וּגְמַלִּים֙ וּבְגָדִ֔ים וַיָּ֖שָׁב וַיָּבֹ֥א אֶל־ אָכִֽישׁ׃
10אָכִ֔ישׁ אַל־ פְּשַׁטְתֶּ֖ם הַיּ֑וֹם וַיֹּ֣אמֶר דָּוִ֗ד עַל־ נֶ֤גֶב יְהוּדָה֙ וְעַל־ נֶ֣גֶב הַיַּרְחְמְאֵלִ֔י וְאֶל־ נֶ֖גֶב הַקֵּינִֽי׃
11וְאִשָּׁ֜ה לֹֽא־ יְחַיֶּ֣ה דָוִ֗ד לְהָבִ֥יא גַת֙ לֵאמֹ֔ר פֶּן־ יַגִּ֥דוּ עָלֵ֖ינוּ לֵאמֹ֑ר כֹּֽה־ עָשָׂ֤ה דָוִד֙ וְכֹ֣ה מִשְׁפָּט֔וֹ כָּל־ הַ֨יָּמִ֔ים אֲשֶׁ֥ר יָשַׁ֖ב בִּשְׂדֵ֥ה פְלִשְׁתִּֽים׃
