1כִּֽי־ תֵ֭שֵׁב לִלְח֣וֹם אֶת־ מוֹשֵׁ֑ל בִּ֥ין תָּ֝בִ֗ין אֶת־ אֲשֶׁ֥ר לְפָנֶֽיךָ׃
2שַׂכִּ֣ין בְּלֹעֶ֑ךָ אִם־ בַּ֖עַל נֶ֣פֶשׁ אָֽתָּה׃
3תִּ֭תְאָו לְמַטְעַמּוֹתָ֑יו וְ֝ה֗וּא לֶ֣חֶם כְּזָבִֽים׃
4תִּיגַ֥ע לְֽהַעֲשִׁ֑יר מִֽבִּינָתְךָ֥ חֲדָֽל׃
5כ] (הֲתָ֤עִיף ק) עֵינֶ֥יךָ בּ֗וֹ וְֽאֵ֫ינֶ֥נּוּ כִּ֤י עָשֹׂ֣ה יַעֲשֶׂה־ לּ֣וֹ כְנָפַ֑יִם כְּ֝נֶ֗שֶׁר [וְעָיֵף כ] (יָע֥וּף ק) הַשָּׁמָֽיִם׃ פ
6תִּלְחַ֗ם אֶת־ לֶ֭חֶם רַ֣ע עָ֑יִן וְאַל־ [תִּתְאָו כ] (תִּ֝תְאָ֗יו ק) לְמַטְעַמֹּתָֽיו׃
7כְּמוֹ־ שָׁעַ֥ר בְּנַפְשׁ֗וֹ כֶּ֫ן־ ה֥וּא אֱכֹ֣ל וּ֭שְׁתֵה יֹ֣אמַר לָ֑ךְ וְ֝לִבּ֗וֹ בַּל־ עִמָּֽךְ׃
8אָכַ֥לְתָּ תְקִיאֶ֑נָּה וְ֝שִׁחַ֗תָּ דְּבָרֶ֥יךָ הַנְּעִימִֽים׃
9כְ֭סִיל אַל־ תְּדַבֵּ֑ר כִּֽי־ יָ֝ב֗וּז לְשֵׂ֣כֶל מִלֶּֽיךָ׃
10תַּ֭סֵּג גְּב֣וּל עוֹלָ֑ם וּבִשְׂדֵ֥י יְ֝תוֹמִ֗ים אַל־ תָּבֹֽא׃
11גֹאֲלָ֥ם חָזָ֑ק הֽוּא־ יָרִ֖יב אֶת־ רִיבָ֣ם אִתָּֽךְ׃
12לַמּוּסָ֣ר לִבֶּ֑ךָ וְ֝אָזְנֶ֗ךָ לְאִמְרֵי־ דָֽעַת׃
13תִּמְנַ֣ע מִנַּ֣עַר מוּסָ֑ר כִּֽי־ תַכֶּ֥נּוּ בַ֝שֵּׁ֗בֶט לֹ֣א יָמֽוּת׃
14בַּשֵּׁ֣בֶט תַּכֶּ֑נּוּ וְ֝נַפְשׁ֗וֹ מִשְּׁא֥וֹל תַּצִּֽיל׃
15אִם־ חָכַ֣ם לִבֶּ֑ךָ יִשְׂמַ֖ח לִבִּ֣י גַם־ אָֽנִי׃
16כִלְיוֹתָ֑י בְּדַבֵּ֥ר שְׂ֝פָתֶ֗יךָ מֵישָׁרִֽים׃
17יְקַנֵּ֣א לִ֭בְּךָ בַּֽחַטָּאִ֑ים כִּ֥י אִם־ בְּיִרְאַת־ יְ֝הוָ֗ה כָּל־ הַיּֽוֹם׃
18אִם־ יֵ֣שׁ אַחֲרִ֑ית וְ֝תִקְוָתְךָ֗ לֹ֣א תִכָּרֵֽת׃
19אַתָּ֣ה בְנִ֣י וַחֲכָ֑ם וְאַשֵּׁ֖ר בַּדֶּ֣רֶךְ לִבֶּֽךָ׃
20תְּהִ֥י בְסֹֽבְאֵי־ יָ֑יִן בְּזֹלֲלֵ֖י בָשָׂ֣ר לָֽמוֹ׃
21סֹבֵ֣א וְ֭זוֹלֵל יִוָּרֵ֑שׁ וּ֝קְרָעִ֗ים תַּלְבִּ֥ישׁ נוּמָֽה׃
22לְ֭אָבִיךָ זֶ֣ה יְלָדֶ֑ךָ וְאַל־ תָּ֝ב֗וּז כִּֽי־ זָקְנָ֥ה אִמֶּֽךָ׃
23קְ֭נֵה וְאַל־ תִּמְכֹּ֑ר חָכְמָ֖ה וּמוּסָ֣ר וּבִינָֽה׃
24כ] (גִּ֣יל ק) [יָגֹול כ] (יָ֭גִיל ק) אֲבִ֣י צַדִּ֑יק [יֹולֵד כ] (וְיֹולֵ֥ד ק) חָ֝כָ֗ם [וְיִשְׂמַח־ כ] (יִשְׂמַח־ בּֽוֹ׃ ק)
25אָבִ֥יךָ וְאִמֶּ֑ךָ וְ֝תָגֵ֗ל יֽוֹלַדְתֶּֽךָ׃
26בְנִ֣י לִבְּךָ֣ לִ֑י וְ֝עֵינֶ֗יךָ דְּרָכַ֥י [תִּרְצֶנָה כ] (תִּצֹּֽרְנָה ק)
27שׁוּחָ֣ה עֲמֻקָּ֣ה זוֹנָ֑ה וּבְאֵ֥ר צָ֝רָ֗ה נָכְרִיָּֽה׃
28הִ֭יא כְּחֶ֣תֶף תֶּֽאֱרֹ֑ב וּ֝בוֹגְדִ֗ים בְּאָדָ֥ם תּוֹסִֽף׃
29א֥וֹי לְמִ֪י אֲב֡וֹי לְמִ֤י [מִדֹונִים כ] (מִדְיָנִ֨ים ק) לְמִ֥י שִׂ֗יחַ לְ֭מִי פְּצָעִ֣ים חִנָּ֑ם לְ֝מִ֗י חַכְלִל֥וּת עֵינָֽיִם׃
30עַל־ הַיָּ֑יִן לַ֝בָּאִ֗ים לַחְקֹ֥ר מִמְסָֽךְ׃
31תֵּ֥רֶא יַיִן֮ כִּ֪י יִתְאַ֫דָּ֥ם כִּֽי־ יִתֵּ֣ן [בַּכִּיס כ] (בַּכֹּ֣וס ק) עֵינ֑וֹ יִ֝תְהַלֵּ֗ךְ בְּמֵישָׁרִֽים׃
32כְּנָחָ֣שׁ יִשָּׁ֑ךְ וּֽכְצִפְעֹנִ֥י יַפְרִֽשׁ׃
33יִרְא֣וּ זָר֑וֹת וְ֝לִבְּךָ֗ יְדַבֵּ֥ר תַּהְפֻּכֽוֹת׃
34כְּשֹׁכֵ֣ב בְּלֶב־ יָ֑ם וּ֝כְשֹׁכֵ֗ב בְּרֹ֣אשׁ חִבֵּֽל׃
