1וְלִבְנֵ֣י יִשָׂשכָ֗ר תּוֹלָ֧ע וּפוּאָ֛ה [יָשִׁיב כ] (יָשׁ֥וּב ק) וְשִׁמְר֖וֹן אַרְבָּעָֽה׃ ס
2תוֹלָ֗ע עֻזִּ֡י וּרְפָיָ֡ה וִֽ֠ירִיאֵל וְיַחְמַ֨י וְיִבְשָׂ֜ם וּשְׁמוּאֵ֗ל רָאשִׁ֤ים לְבֵית־ אֲבוֹתָם֙ לְתוֹלָ֔ע גִּבּ֥וֹרֵי חַ֖יִל לְתֹלְדוֹתָ֑ם מִסְפָּרָם֙ בִּימֵ֣י דָוִ֔יד עֶשְׂרִֽים־ וּשְׁנַ֥יִם אֶ֖לֶף וְשֵׁ֥שׁ מֵאֽוֹת׃ ס
3עֻזִּ֖י יִֽזְרַֽחְיָ֑ה וּבְנֵ֣י יִֽזְרַֽחְיָ֗ה מִֽיכָאֵ֡ל וְ֠עֹבַדְיָה וְיוֹאֵ֧ל יִשִּׁיָּ֛ה חֲמִשָּׁ֖ה רָאשִׁ֥ים כֻּלָּֽם׃
4לְתֹלְדוֹתָ֜ם לְבֵ֣ית אֲבוֹתָ֗ם גְּדוּדֵי֙ צְבָ֣א מִלְחָמָ֔ה שְׁלֹשִׁ֥ים וְשִׁשָּׁ֖ה אָ֑לֶף כִּֽי־ הִרְבּ֥וּ נָשִׁ֖ים וּבָנִֽים׃
5לְכֹל֙ מִשְׁפְּח֣וֹת יִשָׂשכָ֔ר גִּבּוֹרֵ֖י חֲיָלִ֑ים שְׁמוֹנִ֤ים וְשִׁבְעָה֙ אֶ֔לֶף הִתְיַחְשָׂ֖ם לַכֹּֽל׃ פ
6בֶּ֧לַע וָבֶ֛כֶר וִידִֽיעֲאֵ֖ל שְׁלֹשָֽׁה׃
7בֶ֗לַע אֶצְבּ֡וֹן וְעֻזִּ֡י וְ֠עֻזִּיאֵל וִירִימ֨וֹת וְעִירִ֜י חֲמִשָּׁ֗ה רָאשֵׁי֙ בֵּ֣ית אָב֔וֹת גִּבּוֹרֵ֖י חֲיָלִ֑ים וְהִתְיַחְשָׂ֗ם עֶשְׂרִ֤ים וּשְׁנַ֙יִם֙ אֶ֔לֶף וּשְׁלֹשִׁ֖ים וְאַרְבָּעָֽה׃ ס
8בֶ֗כֶר זְמִירָ֡ה וְיוֹעָ֡שׁ וֶ֠אֱלִיעֶזֶר וְאֶלְיוֹעֵינַ֤י וְעָמְרִי֙ וִירֵמ֣וֹת וַאֲבִיָּ֔ה וַעֲנָת֖וֹת וְעָלָ֑מֶת כָּל־ אֵ֖לֶּה בְּנֵי־ בָֽכֶר׃
9לְתֹלְדוֹתָ֗ם רָאשֵׁי֙ בֵּ֣ית אֲבוֹתָ֔ם גִּבּוֹרֵ֖י חָ֑יִל עֶשְׂרִ֥ים אֶ֖לֶף וּמָאתָֽיִם׃ ס
10יְדִיעֲאֵ֖ל בִּלְהָ֑ן וּבְנֵ֣י בִלְהָ֗ן [יְעִישׁ כ] (יְע֡וּשׁ ק) וּ֠בִנְיָמִן וְאֵה֤וּד וּֽכְנַעֲנָה֙ וְזֵיתָ֔ן וְתַרְשִׁ֖ישׁ וַאֲחִישָֽׁחַר׃
11אֵ֜לֶּה בְּנֵ֤י יְדִֽיעֲאֵל֙ לְרָאשֵׁ֣י הָאָב֔וֹת גִּבּוֹרֵ֖י חֲיָלִ֑ים שִׁבְעָֽה־ עָשָׂ֥ר אֶ֙לֶף֙ וּמָאתַ֔יִם יֹצְאֵ֥י צָבָ֖א לַמִּלְחָמָֽה׃
12וְחֻפִּם֙ בְּנֵ֣י עִ֔יר חֻשִׁ֖ם בְּנֵ֥י אַחֵֽר׃
13נַפְתָּלִ֗י יַחֲצִיאֵ֧ל וְגוּנִ֛י וְיֵ֥צֶר וְשַׁלּ֖וּם בְּנֵ֥י בִלְהָֽה׃ פ
14מְנַשֶּׁ֔ה אַשְׂרִיאֵ֖ל אֲשֶׁ֣ר יָלָ֑דָה פִּֽילַגְשׁוֹ֙ הָֽאֲרַמִּיָּ֔ה יָלְדָ֕ה אֶת־ מָכִ֖יר אֲבִ֥י גִלְעָֽד׃
15לָקַ֤ח אִשָּׁה֙ לְחֻפִּ֣ים וּלְשֻׁפִּ֔ים וְשֵׁ֤ם אֲחֹתוֹ֙ מַעֲכָ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖י צְלָפְחָ֑ד וַתִּהְיֶ֥נָה לִצְלָפְחָ֖ד בָּנֽוֹת׃
16מַעֲכָ֤ה אֵֽשֶׁת־ מָכִיר֙ בֵּ֔ן וַתִּקְרָ֤א שְׁמוֹ֙ פֶּ֔רֶשׁ וְשֵׁ֥ם אָחִ֖יו שָׁ֑רֶשׁ וּבָנָ֖יו אוּלָ֥ם וָרָֽקֶם׃
17אוּלָ֖ם בְּדָ֑ן אֵ֚לֶּה בְּנֵ֣י גִלְעָ֔ד בֶּן־ מָכִ֖יר בֶּן־ מְנַשֶּֽׁה׃
18הַמֹּלֶ֑כֶת יָלְדָה֙ אֶת־ אִישְׁה֔וֹד וְאֶת־ אֲבִיעֶ֖זֶר וְאֶת־ מַחְלָֽה׃
19בְּנֵ֣י שְׁמִידָ֑ע אַחְיָ֣ן וָשֶׁ֔כֶם וְלִקְחִ֖י וַאֲנִיעָֽם׃ פ
20אֶפְרַ֖יִם שׁוּתָ֑לַח וּבֶ֤רֶד בְּנוֹ֙ וְתַ֣חַת בְּנ֔וֹ וְאֶלְעָדָ֥ה בְנ֖וֹ וְתַ֥חַת בְּנֽוֹ׃
21בְּנ֛וֹ וְשׁוּתֶ֥לַח בְּנ֖וֹ וְעֵ֣זֶר וְאֶלְעָ֑ד וַהֲרָג֗וּם אַנְשֵׁי־ גַת֙ הַנּוֹלָדִ֣ים בָּאָ֔רֶץ כִּ֣י יָרְד֔וּ לָקַ֖חַת אֶת־ מִקְנֵיהֶֽם׃
22אֶפְרַ֥יִם אֲבִיהֶ֖ם יָמִ֣ים רַבִּ֑ים וַיָּבֹ֥אוּ אֶחָ֖יו לְנַחֲמֽוֹ׃
23אֶל־ אִשְׁתּ֔וֹ וַתַּ֖הַר וַתֵּ֣לֶד בֵּ֑ן וַיִּקְרָ֤א אֶת־ שְׁמוֹ֙ בְּרִיעָ֔ה כִּ֥י בְרָעָ֖ה הָיְתָ֥ה בְּבֵיתֽוֹ׃
24שֶׁאֱרָ֔ה וַתִּ֧בֶן אֶת־ בֵּית־ חוֹר֛וֹן הַתַּחְתּ֖וֹן וְאֶת־ הָעֶלְי֑וֹן וְאֵ֖ת אֻזֵּ֥ן שֶׁאֱרָֽה׃
25בְּנ֗וֹ וְרֶ֧שֶׁף וְתֶ֛לַח בְּנ֖וֹ וְתַ֥חַן בְּנֽוֹ׃
26בְּנ֛וֹ עַמִּיה֥וּד בְּנ֖וֹ אֱלִישָׁמָ֥ע בְּנֽוֹ׃
27בְּנ֖וֹ יְהוֹשֻׁ֥עַ בְּנֽוֹ׃
28וּמֹ֣שְׁבוֹתָ֔ם בֵּֽית־ אֵ֖ל וּבְנֹתֶ֑יהָ וְלַמִּזְרָ֣ח נַעֲרָ֔ן וְלַֽמַּעֲרָ֗ב גֶּ֤זֶר וּבְנֹתֶ֙יהָ֙ וּשְׁכֶ֣ם וּבְנֹתֶ֔יהָ עַד־ עַיָּ֖ה וּבְנֹתֶֽיהָ׃
29יְדֵ֣י בְנֵי־ מְנַשֶּׁ֗ה בֵּית־ שְׁאָ֤ן וּבְנֹתֶ֙יהָ֙ תַּעְנַ֣ךְ וּבְנֹתֶ֔יהָ מְגִדּ֥וֹ וּבְנוֹתֶ֖יהָ דּ֣וֹר וּבְנוֹתֶ֑יהָ בְּאֵ֙לֶּה֙ יָשְׁב֔וּ בְּנֵ֥י יוֹסֵ֖ף בֶּן־ יִשְׂרָאֵֽל׃ פ
30אָשֵׁ֗ר יִמְנָ֧ה וְיִשְׁוָ֛ה וְיִשְׁוִ֥י וּבְרִיעָ֖ה וְשֶׂ֥רַח אֲחוֹתָֽם׃
31בְרִיעָ֔ה חֶ֖בֶר וּמַלְכִּיאֵ֑ל ה֖וּא אֲבִ֥י [בִרְזֹות כ] (בִרְזָֽיִת׃ ק)
32הוֹלִ֣יד אֶת־ יַפְלֵ֔ט וְאֶת־ שׁוֹמֵ֖ר וְאֶת־ חוֹתָ֑ם וְאֵ֖ת שׁוּעָ֥א אֲחוֹתָֽם׃
33יַפְלֵ֔ט פָּסַ֥ךְ וּבִמְהָ֖ל וְעַשְׁוָ֑ת אֵ֖לֶּה בְּנֵ֥י יַפְלֵֽט׃
34שָׁ֑מֶר אֲחִ֥י [וְרֹוהֲגָה כ] (וְרָהְגָּ֖ה ק) [יַחְבָּה כ] (וְחֻבָּ֥ה ק) וַאֲרָֽם׃
35הֵ֖לֶם אָחִ֑יו צוֹפַ֥ח וְיִמְנָ֖ע וְשֵׁ֥לֶשׁ וְעָמָֽל׃
36צוֹפָ֑ח ס֧וּחַ וְחַרְנֶ֛פֶר וְשׁוּעָ֖ל וּבֵרִ֥י וְיִמְרָֽה׃
37וָה֗וֹד וְשַׁמָּ֧א וְשִׁלְשָׁ֛ה וְיִתְרָ֖ן וּבְאֵרָֽא׃
38יֶ֑תֶר יְפֻנֶּ֥ה וּפִסְפָּ֖ה וַאְרָֽא׃
39עֻלָּ֑א אָרַ֥ח וְחַנִּיאֵ֖ל וְרִצְיָֽא׃
