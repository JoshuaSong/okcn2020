1עַ֥ל נַהֲר֨וֹת ׀ בָּבֶ֗ל שָׁ֣ם יָ֭שַׁבְנוּ גַּם־ בָּכִ֑ינוּ בְּ֝זָכְרֵ֗נוּ אֶת־ צִיּֽוֹן׃
2עֲרָבִ֥ים בְּתוֹכָ֑הּ תָּ֝לִ֗ינוּ כִּנֹּרוֹתֵֽינוּ׃
3שָׁ֨ם שְֽׁאֵל֪וּנוּ שׁוֹבֵ֡ינוּ דִּבְרֵי־ שִׁ֭יר וְתוֹלָלֵ֣ינוּ שִׂמְחָ֑ה שִׁ֥ירוּ לָ֝֗נוּ מִשִּׁ֥יר צִיּֽוֹן׃
4נָשִׁ֥יר אֶת־ שִׁיר־ יְהוָ֑ה עַ֝֗ל אַדְמַ֥ת נֵכָֽר׃
5אֶשְׁכָּחֵ֥ךְ יְֽרוּשָׁלִָ֗ם תִּשְׁכַּ֥ח יְמִינִֽי׃
6לְשׁוֹנִ֨י ׀ לְחִכִּי֮ אִם־ לֹ֪א אֶ֫זְכְּרֵ֥כִי אִם־ לֹ֣א אַ֭עֲלֶה אֶת־ יְרוּשָׁלִַ֑ם עַ֝֗ל רֹ֣אשׁ שִׂמְחָתִֽי׃
7יְהוָ֨ה ׀ לִבְנֵ֬י אֱד֗וֹם אֵת֮ י֤וֹם יְֽרוּשָׁ֫לִָ֥ם הָ֭אֹ֣מְרִים עָ֤רוּ ׀ עָ֑רוּ עַ֝֗ד הַיְס֥וֹד בָּֽהּ׃
8בָּבֶ֗ל הַשְּׁד֫וּדָ֥ה אַשְׁרֵ֥י שֶׁיְשַׁלֶּם־ לָ֑ךְ אֶת־ גְּ֝מוּלֵ֗ךְ שֶׁגָּמַ֥לְתְּ לָֽנוּ׃
