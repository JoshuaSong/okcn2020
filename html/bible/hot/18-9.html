1וַיַּ֥עַן אִיּ֗וֹב וַיֹּאמַֽר׃
2יָדַ֣עְתִּי כִי־ כֵ֑ן וּמַה־ יִּצְדַּ֖ק אֱנ֣וֹשׁ עִם־ אֵֽל׃
3יַ֭חְפֹּץ לָרִ֣יב עִמּ֑וֹ לֹֽא־ יַ֝עֲנֶ֗נּוּ אַחַ֥ת מִנִּי־ אָֽלֶף׃
4לֵ֭בָב וְאַמִּ֣יץ כֹּ֑חַ מִֽי־ הִקְשָׁ֥ה אֵ֝לָ֗יו וַיִּשְׁלָֽם׃
5הָ֭רִים וְלֹ֣א יָדָ֑עוּ אֲשֶׁ֖ר הֲפָכָ֣ם בְּאַפּֽוֹ׃
6אֶ֭רֶץ מִמְּקוֹמָ֑הּ וְ֝עַמּוּדֶ֗יהָ יִתְפַלָּצֽוּן׃
7לַ֭חֶרֶס וְלֹ֣א יִזְרָ֑ח וּבְעַ֖ד כּוֹכָבִ֣ים יַחְתֹּֽם׃
8שָׁמַ֣יִם לְבַדּ֑וֹ וְ֝דוֹרֵ֗ךְ עַל־ בָּ֥מֳתֵי יָֽם׃
9עָ֭שׁ כְּסִ֥יל וְכִימָ֗ה וְחַדְרֵ֥י תֵמָֽן׃
10גְ֭דֹלוֹת עַד־ אֵ֣ין חֵ֑קֶר וְנִפְלָא֗וֹת עַד־ אֵ֥ין מִסְפָּֽר׃
11יַעֲבֹ֣ר עָ֭לַי וְלֹ֣א אֶרְאֶ֑ה וְ֝יַחֲלֹ֗ף וְֽלֹא־ אָבִ֥ין לֽוֹ׃
12יַ֭חְתֹּף מִ֣י יְשִׁיבֶ֑נּוּ מִֽי־ יֹאמַ֥ר אֵ֝לָ֗יו מַֽה־ תַּעֲשֶֽׂה׃
13לֹא־ יָשִׁ֣יב אַפּ֑וֹ [תַּחַתֹו כ] (תַּחְתָּ֥יו ק) שָׁ֝חֲח֗וּ עֹ֣זְרֵי רָֽהַב׃
14כִּֽי־ אָנֹכִ֣י אֶֽעֱנֶ֑נּוּ אֶבְחֲרָ֖ה דְבָרַ֣י עִמּֽוֹ׃
15אִם־ צָ֭דַקְתִּי לֹ֣א אֶעֱנֶ֑ה לִ֝מְשֹׁפְטִ֗י אֶתְחַנָּֽן׃
16קָרָ֥אתִי וַֽיַּעֲנֵ֑נִי לֹֽא־ אַ֝אֲמִ֗ין כִּֽי־ יַאֲזִ֥ין קוֹלִֽי׃
17בִּשְׂעָרָ֥ה יְשׁוּפֵ֑נִי וְהִרְבָּ֖ה פְצָעַ֣י חִנָּֽם׃
18יִ֭תְּנֵנִי הָשֵׁ֣ב רוּחִ֑י כִּ֥י יַ֝שְׂבִּעַ֗נִי מַמְּרֹרִֽים׃
19לְכֹ֣חַ אַמִּ֣יץ הִנֵּ֑ה וְאִם־ לְ֝מִשְׁפָּ֗ט מִ֣י יוֹעִידֵֽנִי׃
20אֶ֭צְדָּק פִּ֣י יַרְשִׁיעֵ֑נִי תָּֽם־ אָ֝֗נִי וַֽיַּעְקְשֵֽׁנִי׃
21אָ֭נִי לֹֽא־ אֵדַ֥ע נַפְשִׁ֗י אֶמְאַ֥ס חַיָּֽי׃
22הִ֥יא עַל־ כֵּ֥ן אָמַ֑רְתִּי תָּ֥ם וְ֝רָשָׁ֗ע ה֣וּא מְכַלֶּֽה׃
23שׁ֭וֹט יָמִ֣ית פִּתְאֹ֑ם לְמַסַּ֖ת נְקִיִּ֣ם יִלְעָֽג׃
24נִתְּנָ֬ה בְֽיַד־ רָשָׁ֗ע פְּנֵֽי־ שֹׁפְטֶ֥יהָ יְכַסֶּ֑ה אִם־ לֹ֖א אֵפ֣וֹא מִי־ הֽוּא׃
25קַ֭לּוּ מִנִּי־ רָ֑ץ בָּֽ֝רְח֗וּ לֹא־ רָא֥וּ טוֹבָֽה׃
26עִם־ אֳנִיּ֣וֹת אֵבֶ֑ה כְּ֝נֶ֗שֶׁר יָט֥וּשׂ עֲלֵי־ אֹֽכֶל׃
27אָ֭מְרִי אֶשְׁכְּחָ֣ה שִׂיחִ֑י אֶעֶזְבָ֖ה פָנַ֣י וְאַבְלִֽיגָה׃
28כָל־ עַצְּבֹתָ֑י יָ֝דַ֗עְתִּי כִּי־ לֹ֥א תְנַקֵּֽנִי׃
29אֶרְשָׁ֑ע לָמָּה־ זֶּ֝֗ה הֶ֣בֶל אִיגָֽע׃
30הִתְרָחַ֥צְתִּי [בְמֹו כ] (בְמֵי־ שָׁ֑לֶג ק) וַ֝הֲזִכּ֗וֹתִי בְּבֹ֣ר כַּפָּֽי׃
31בַּשַּׁ֣חַת תִּטְבְּלֵ֑נִי וְ֝תִֽעֲב֗וּנִי שַׂלְמוֹתָֽי׃ כִּי־ לֹא־
32כָּמֹ֣נִי אֶֽעֱנֶ֑נּוּ נָב֥וֹא יַ֝חְדָּ֗ו בַּמִּשְׁפָּֽט׃
33יֵשׁ־ בֵּינֵ֣ינוּ מוֹכִ֑יחַ יָשֵׁ֖ת יָד֣וֹ עַל־ שְׁנֵֽינוּ׃
34מֵעָלַ֣י שִׁבְט֑וֹ וְ֝אֵמָת֗וֹ אַֽל־ תְּבַעֲתַֽנִּי׃
