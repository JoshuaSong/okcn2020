1אֲנִ֤י הַגֶּ֙בֶר֙ רָאָ֣ה עֳנִ֔י בְּשֵׁ֖בֶט עֶבְרָתֽוֹ׃
2נָהַ֛ג וַיֹּלַ֖ךְ חֹ֥שֶׁךְ וְלֹא־ אֽוֹר׃
3בִּ֥י יָשֻׁ֛ב יַהֲפֹ֥ךְ יָד֖וֹ כָּל־ הַיּֽוֹם׃ ס
4בְשָׂרִי֙ וְעוֹרִ֔י שִׁבַּ֖ר עַצְמוֹתָֽי׃
5עָלַ֛י וַיַּקַּ֖ף רֹ֥אשׁ וּתְלָאָֽה׃
6הוֹשִׁיבַ֖נִי כְּמֵתֵ֥י עוֹלָֽם׃ ס
7בַּעֲדִ֛י וְלֹ֥א אֵצֵ֖א הִכְבִּ֥יד נְחָשְׁתִּֽי׃
8כִּ֤י אֶזְעַק֙ וַאֲשַׁוֵּ֔עַ שָׂתַ֖ם תְּפִלָּתִֽי׃
9דְּרָכַי֙ בְּגָזִ֔ית נְתִיבֹתַ֖י עִוָּֽה׃ ס
10אֹרֵ֥ב הוּא֙ לִ֔י [אַרְיֵה כ] (אֲרִ֖י ק) בְּמִסְתָּרִֽים׃
11סוֹרֵ֛ר וַֽיְפַשְּׁחֵ֖נִי שָׂמַ֥נִי שֹׁמֵֽם׃
12קַשְׁתּוֹ֙ וַיַּצִּיבֵ֔נִי כַּמַּטָּרָ֖א לַחֵֽץ׃ ס
13בְּכִלְיוֹתָ֔י בְּנֵ֖י אַשְׁפָּתֽוֹ׃
14שְּׂחֹק֙ לְכָל־ עַמִּ֔י נְגִינָתָ֖ם כָּל־ הַיּֽוֹם׃
15בַמְּרוֹרִ֖ים הִרְוַ֥נִי לַעֲנָֽה׃ ס
16בֶּֽחָצָץ֙ שִׁנָּ֔י הִכְפִּישַׁ֖נִי בָּאֵֽפֶר׃
17מִשָּׁל֛וֹם נַפְשִׁ֖י נָשִׁ֥יתִי טוֹבָֽה׃
18אָבַ֣ד נִצְחִ֔י וְתוֹחַלְתִּ֖י מֵיְהוָֽה׃ ס
19עָנְיִ֥י וּמְרוּדִ֖י לַעֲנָ֥ה וָרֹֽאשׁ׃
20תִּזְכּ֔וֹר [וְתָשִׁיחַ כ] (וְתָשֹׁ֥וחַ ק) עָלַ֖י נַפְשִֽׁי׃
21אָשִׁ֥יב אֶל־ לִבִּ֖י עַל־ כֵּ֥ן אוֹחִֽיל׃ ס
22יְהוָה֙ כִּ֣י לֹא־ תָ֔מְנוּ כִּ֥י לֹא־ כָל֖וּ רַחֲמָֽיו׃
23לַבְּקָרִ֔ים רַבָּ֖ה אֱמוּנָתֶֽךָ׃
24יְהוָה֙ אָמְרָ֣ה נַפְשִׁ֔י עַל־ כֵּ֖ן אוֹחִ֥יל לֽוֹ׃ ס
25יְהוָה֙ לְקֹוָ֔ו לְנֶ֖פֶשׁ תִּדְרְשֶֽׁנּוּ׃
26וְיָחִיל֙ וְדוּמָ֔ם לִתְשׁוּעַ֖ת יְהוָֽה׃
27לַגֶּ֔בֶר כִּֽי־ יִשָּׂ֥א עֹ֖ל בִּנְעוּרָֽיו׃ ס
28בָּדָד֙ וְיִדֹּ֔ם כִּ֥י נָטַ֖ל עָלָֽיו׃
29בֶּֽעָפָר֙ פִּ֔יהוּ אוּלַ֖י יֵ֥שׁ תִּקְוָֽה׃
30לְמַכֵּ֛הוּ לֶ֖חִי יִשְׂבַּ֥ע בְּחֶרְפָּֽה׃ ס
31לֹ֥א יִזְנַ֛ח לְעוֹלָ֖ם אֲדֹנָֽי׃
32אִם־ הוֹגָ֔ה וְרִחַ֖ם כְּרֹ֥ב [חַסְדֹּו כ] (חֲסָדָֽיו׃ ק)
33לֹ֤א עִנָּה֙ מִלִּבּ֔וֹ וַיַּגֶּ֖ה בְּנֵי־ אִֽישׁ׃ ס
34תַּ֣חַת רַגְלָ֔יו כֹּ֖ל אֲסִ֥ירֵי אָֽרֶץ׃
35מִשְׁפַּט־ גָּ֔בֶר נֶ֖גֶד פְּנֵ֥י עֶלְיֽוֹן׃
36אָדָם֙ בְּרִיב֔וֹ אֲדֹנָ֖י לֹ֥א רָאָֽה׃ ס
37זֶ֤ה אָמַר֙ וַתֶּ֔הִי אֲדֹנָ֖י לֹ֥א צִוָּֽה׃
38עֶלְיוֹן֙ לֹ֣א תֵצֵ֔א הָרָע֖וֹת וְהַטּֽוֹב׃
39יִּתְאוֹנֵן֙ אָדָ֣ם חָ֔י גֶּ֖בֶר עַל־ [חֶטְאֹו כ] (חֲטָאָֽיו׃ ק) ס
40דְרָכֵ֙ינוּ֙ וְֽנַחְקֹ֔רָה וְנָשׁ֖וּבָה עַד־ יְהוָֽה׃
41לְבָבֵ֙נוּ֙ אֶל־ כַּפָּ֔יִם אֶל־ אֵ֖ל בַּשָּׁמָֽיִם׃
42פָשַׁ֙עְנוּ֙ וּמָרִ֔ינוּ אַתָּ֖ה לֹ֥א סָלָֽחְתָּ׃ ס
43בָאַף֙ וַֽתִּרְדְּפֵ֔נוּ הָרַ֖גְתָּ לֹ֥א חָמָֽלְתָּ׃
44בֶֽעָנָן֙ לָ֔ךְ מֵעֲב֖וֹר תְּפִלָּֽה׃
45וּמָא֛וֹס תְּשִׂימֵ֖נוּ בְּקֶ֥רֶב הָעַמִּֽים׃ ס
46עָלֵ֛ינוּ פִּיהֶ֖ם כָּל־ אֹיְבֵֽינוּ׃
47וָפַ֛חַת הָ֥יָה לָ֖נוּ הַשֵּׁ֥את וְהַשָּֽׁבֶר׃
48מַ֙יִם֙ תֵּרַ֣ד עֵינִ֔י עַל־ שֶׁ֖בֶר בַּת־ עַמִּֽי׃ ס
49נִגְּרָ֛ה וְלֹ֥א תִדְמֶ֖ה מֵאֵ֥ין הֲפֻגֽוֹת׃
50יַשְׁקִ֣יף וְיֵ֔רֶא יְהוָ֖ה מִשָּׁמָֽיִם׃
51עֽוֹלְלָ֣ה לְנַפְשִׁ֔י מִכֹּ֖ל בְּנ֥וֹת עִירִֽי׃ ס
52צָד֛וּנִי כַּצִּפּ֖וֹר אֹיְבַ֥י חִנָּֽם׃
53בַבּוֹר֙ חַיָּ֔י וַיַּדּוּ־ אֶ֖בֶן בִּֽי׃
54מַ֥יִם עַל־ רֹאשִׁ֖י אָמַ֥רְתִּי נִגְזָֽרְתִּי׃ ס
55שִׁמְךָ֙ יְהוָ֔ה מִבּ֖וֹר תַּחְתִּיּֽוֹת׃
56שָׁמָ֑עְתָּ אַל־ תַּעְלֵ֧ם אָזְנְךָ֛ לְרַוְחָתִ֖י לְשַׁוְעָתִֽי׃
57בְּי֣וֹם אֶקְרָאֶ֔ךָּ אָמַ֖רְתָּ אַל־ תִּירָֽא׃ ס
58אֲדֹנָ֛י רִיבֵ֥י נַפְשִׁ֖י גָּאַ֥לְתָּ חַיָּֽי׃
59יְהוָה֙ עַוָּ֣תָתִ֔י שָׁפְטָ֖ה מִשְׁפָּטִֽי׃
60כָּל־ נִקְמָתָ֔ם כָּל־ מַחְשְׁבֹתָ֖ם לִֽי׃ ס
61חֶרְפָּתָם֙ יְהוָ֔ה כָּל־ מַחְשְׁבֹתָ֖ם עָלָֽי׃
62קָמַי֙ וְהֶגְיוֹנָ֔ם עָלַ֖י כָּל־ הַיּֽוֹם׃
63וְקִֽימָתָם֙ הַבִּ֔יטָה אֲנִ֖י מַנְגִּינָתָֽם׃ ס
64לָהֶ֥ם גְּמ֛וּל יְהוָ֖ה כְּמַעֲשֵׂ֥ה יְדֵיהֶֽם׃
65לָהֶם֙ מְגִנַּת־ לֵ֔ב תַּאֲלָֽתְךָ֖ לָהֶֽם׃
