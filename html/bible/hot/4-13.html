1וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־ מֹשֶׁ֥ה לֵּאמֹֽר׃
2לְךָ֣ אֲנָשִׁ֗ים וְיָתֻ֙רוּ֙ אֶת־ אֶ֣רֶץ כְּנַ֔עַן אֲשֶׁר־ אֲנִ֥י נֹתֵ֖ן לִבְנֵ֣י יִשְׂרָאֵ֑ל אִ֣ישׁ אֶחָד֩ אִ֨ישׁ אֶחָ֜ד לְמַטֵּ֤ה אֲבֹתָיו֙ תִּשְׁלָ֔חוּ כֹּ֖ל נָשִׂ֥יא בָהֶֽם׃
3אֹתָ֥ם מֹשֶׁ֛ה מִמִּדְבַּ֥ר פָּארָ֖ן עַל־ פִּ֣י יְהוָ֑ה כֻּלָּ֣ם אֲנָשִׁ֔ים רָאשֵׁ֥י בְנֵֽי־ יִשְׂרָאֵ֖ל הֵֽמָּה׃
4שְׁמוֹתָ֑ם לְמַטֵּ֣ה רְאוּבֵ֔ן שַׁמּ֖וּעַ בֶּן־ זַכּֽוּר׃
5שִׁמְע֔וֹן שָׁפָ֖ט בֶּן־ חוֹרִֽי׃
6יְהוּדָ֔ה כָּלֵ֖ב בֶּן־ יְפֻנֶּֽה׃
7יִשָּׂשכָ֔ר יִגְאָ֖ל בֶּן־ יוֹסֵֽף׃
8אֶפְרָ֖יִם הוֹשֵׁ֥עַ בִּן־ נֽוּן׃
9בִנְיָמִ֔ן פַּלְטִ֖י בֶּן־ רָפֽוּא׃
10זְבוּלֻ֔ן גַּדִּיאֵ֖ל בֶּן־ סוֹדִֽי׃
11יוֹסֵ֖ף לְמַטֵּ֣ה מְנַשֶּׁ֑ה גַּדִּ֖י בֶּן־ סוּסִֽי׃
12דָ֔ן עַמִּיאֵ֖ל בֶּן־ גְּמַלִּֽי׃
13אָשֵׁ֔ר סְת֖וּר בֶּן־ מִיכָאֵֽל׃
14נַפְתָּלִ֔י נַחְבִּ֖י בֶּן־ וָפְסִֽי׃
15גָ֔ד גְּאוּאֵ֖ל בֶּן־ מָכִֽי׃
16שְׁמ֣וֹת הָֽאֲנָשִׁ֔ים אֲשֶׁר־ שָׁלַ֥ח מֹשֶׁ֖ה לָת֣וּר אֶת־ הָאָ֑רֶץ וַיִּקְרָ֥א מֹשֶׁ֛ה לְהוֹשֵׁ֥עַ בִּן־ נ֖וּן יְהוֹשֻֽׁעַ׃
17אֹתָם֙ מֹשֶׁ֔ה לָת֖וּר אֶת־ אֶ֣רֶץ כְּנָ֑עַן וַיֹּ֣אמֶר אֲלֵהֶ֗ם עֲל֥וּ זֶה֙ בַּנֶּ֔גֶב וַעֲלִיתֶ֖ם אֶת־ הָהָֽר׃
18אֶת־ הָאָ֖רֶץ מַה־ הִ֑וא וְאֶת־ הָעָם֙ הַיֹּשֵׁ֣ב עָלֶ֔יהָ הֶחָזָ֥ק הוּא֙ הֲרָפֶ֔ה הַמְעַ֥ט ה֖וּא אִם־ רָֽב׃
19הָאָ֗רֶץ אֲשֶׁר־ הוּא֙ יֹשֵׁ֣ב בָּ֔הּ הֲטוֹבָ֥ה הִ֖וא אִם־ רָעָ֑ה וּמָ֣ה הֶֽעָרִ֗ים אֲשֶׁר־ הוּא֙ יוֹשֵׁ֣ב בָּהֵ֔נָּה הַבְּמַֽחֲנִ֖ים אִ֥ם בְּמִבְצָרִֽים׃
20הָ֠אָרֶץ הַשְּׁמֵנָ֨ה הִ֜וא אִם־ רָזָ֗ה הֲיֵֽשׁ־ בָּ֥הּ עֵץ֙ אִם־ אַ֔יִן וְהִ֨תְחַזַּקְתֶּ֔ם וּלְקַחְתֶּ֖ם מִפְּרִ֣י הָאָ֑רֶץ וְהַ֨יָּמִ֔ים יְמֵ֖י בִּכּוּרֵ֥י עֲנָבִֽים׃
21וַיָּתֻ֣רוּ אֶת־ הָאָ֑רֶץ מִמִּדְבַּר־ צִ֥ן עַד־ רְחֹ֖ב לְבֹ֥א חֲמָֽת׃
22בַנֶּגֶב֮ וַיָּבֹ֣א עַד־ חֶבְרוֹן֒ וְשָׁ֤ם אֲחִימַן֙ שֵׁשַׁ֣י וְתַלְמַ֔י יְלִידֵ֖י הָעֲנָ֑ק וְחֶבְר֗וֹן שֶׁ֤בַע שָׁנִים֙ נִבְנְתָ֔ה לִפְנֵ֖י צֹ֥עַן מִצְרָֽיִם׃
23עַד־ נַ֣חַל אֶשְׁכֹּ֗ל וַיִּכְרְת֨וּ מִשָּׁ֤ם זְמוֹרָה֙ וְאֶשְׁכּ֤וֹל עֲנָבִים֙ אֶחָ֔ד וַיִּשָּׂאֻ֥הוּ בַמּ֖וֹט בִּשְׁנָ֑יִם וּמִן־ הָרִמֹּנִ֖ים וּמִן־ הַתְּאֵנִֽים׃
24הַה֔וּא קָרָ֖א נַ֣חַל אֶשְׁכּ֑וֹל עַ֚ל אֹד֣וֹת הָֽאֶשְׁכּ֔וֹל אֲשֶׁר־ כָּרְת֥וּ מִשָּׁ֖ם בְּנֵ֥י יִשְׂרָאֵֽל׃
25מִתּ֣וּר הָאָ֑רֶץ מִקֵּ֖ץ אַרְבָּעִ֥ים יֽוֹם׃
26וַיָּבֹאוּ֩ אֶל־ מֹשֶׁ֨ה וְאֶֽל־ אַהֲרֹ֜ן וְאֶל־ כָּל־ עֲדַ֧ת בְּנֵֽי־ יִשְׂרָאֵ֛ל אֶל־ מִדְבַּ֥ר פָּארָ֖ן קָדֵ֑שָׁה וַיָּשִׁ֨יבוּ אוֹתָ֤ם דָּבָר֙ וְאֶת־ כָּל־ הָ֣עֵדָ֔ה וַיַּרְא֖וּם אֶת־ פְּרִ֥י הָאָֽרֶץ׃
27לוֹ֙ וַיֹּ֣אמְר֔וּ בָּ֕אנוּ אֶל־ הָאָ֖רֶץ אֲשֶׁ֣ר שְׁלַחְתָּ֑נוּ וְ֠גַם זָבַ֨ת חָלָ֥ב וּדְבַ֛שׁ הִ֖וא וְזֶה־ פִּרְיָֽהּ׃
28כִּֽי־ עַ֣ז הָעָ֔ם הַיֹּשֵׁ֖ב בָּאָ֑רֶץ וְהֶֽעָרִ֗ים בְּצֻר֤וֹת גְּדֹלֹת֙ מְאֹ֔ד וְגַם־ יְלִדֵ֥י הָֽעֲנָ֖ק רָאִ֥ינוּ שָֽׁם׃
29יוֹשֵׁ֖ב בְּאֶ֣רֶץ הַנֶּ֑גֶב וְ֠הַֽחִתִּי וְהַיְבוּסִ֤י וְהָֽאֱמֹרִי֙ יוֹשֵׁ֣ב בָּהָ֔ר וְהַֽכְּנַעֲנִי֙ יֹשֵׁ֣ב עַל־ הַיָּ֔ם וְעַ֖ל יַ֥ד הַיַּרְדֵּֽן׃
30כָּלֵ֛ב אֶת־ הָעָ֖ם אֶל־ מֹשֶׁ֑ה וַיֹּ֗אמֶר עָלֹ֤ה נַעֲלֶה֙ וְיָרַ֣שְׁנוּ אֹתָ֔הּ כִּֽי־ יָכ֥וֹל נוּכַ֖ל לָֽהּ׃
31אֲשֶׁר־ עָל֤וּ עִמּוֹ֙ אָֽמְר֔וּ לֹ֥א נוּכַ֖ל לַעֲל֣וֹת אֶל־ הָעָ֑ם כִּֽי־ חָזָ֥ק ה֖וּא מִמֶּֽנּוּ׃
32דִּבַּ֤ת הָאָ֙רֶץ֙ אֲשֶׁ֣ר תָּר֣וּ אֹתָ֔הּ אֶל־ בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר הָאָ֡רֶץ אֲשֶׁר֩ עָבַ֨רְנוּ בָ֜הּ לָת֣וּר אֹתָ֗הּ אֶ֣רֶץ אֹכֶ֤לֶת יוֹשְׁבֶ֙יהָ֙ הִ֔וא וְכָל־ הָעָ֛ם אֲשֶׁר־ רָאִ֥ינוּ בְתוֹכָ֖הּ אַנְשֵׁ֥י מִדּֽוֹת׃
