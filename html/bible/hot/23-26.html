1בַּיּ֣וֹם הַה֔וּא יוּשַׁ֥ר הַשִּׁיר־ הַזֶּ֖ה בְּאֶ֣רֶץ יְהוּדָ֑ה עִ֣יר עָז־ לָ֔נוּ יְשׁוּעָ֥ה יָשִׁ֖ית חוֹמ֥וֹת וָחֵֽל׃
2שְׁעָרִ֑ים וְיָבֹ֥א גוֹי־ צַדִּ֖יק שֹׁמֵ֥ר אֱמֻנִֽים׃
3סָמ֔וּךְ תִּצֹּ֖ר שָׁל֣וֹם ׀ שָׁל֑וֹם כִּ֥י בְךָ֖ בָּטֽוּחַ׃
4בַֽיהוָ֖ה עֲדֵי־ עַ֑ד כִּ֚י בְּיָ֣הּ יְהוָ֔ה צ֖וּר עוֹלָמִֽים׃
5הֵשַׁח֙ יֹשְׁבֵ֣י מָר֔וֹם קִרְיָ֖ה נִשְׂגָּבָ֑ה יַשְׁפִּילֶ֤נָּה יַשְׁפִּילָהּ֙ עַד־ אֶ֔רֶץ יַגִּיעֶ֖נָּה עַד־ עָפָֽר׃
6רָ֑גֶל רַגְלֵ֥י עָנִ֖י פַּעֲמֵ֥י דַלִּֽים׃
7לַצַּדִּ֖יק מֵֽישָׁרִ֑ים יָשָׁ֕ר מַעְגַּ֥ל צַדִּ֖יק תְּפַלֵּֽס׃
8אֹ֧רַח מִשְׁפָּטֶ֛יךָ יְהוָ֖ה קִוִּינ֑וּךָ לְשִׁמְךָ֥ וּֽלְזִכְרְךָ֖ תַּאֲוַת־ נָֽפֶשׁ׃
9אִוִּיתִ֙יךָ֙ בַּלַּ֔יְלָה אַף־ רוּחִ֥י בְקִרְבִּ֖י אֲשַֽׁחֲרֶ֑ךָּ כִּ֞י כַּאֲשֶׁ֤ר מִשְׁפָּטֶ֙יךָ֙ לָאָ֔רֶץ צֶ֥דֶק לָמְד֖וּ יֹשְׁבֵ֥י תֵבֵֽל׃
10רָשָׁע֙ בַּל־ לָמַ֣ד צֶ֔דֶק בְּאֶ֥רֶץ נְכֹח֖וֹת יְעַוֵּ֑ל וּבַל־ יִרְאֶ֖ה גֵּא֥וּת יְהוָֽה׃ ס
11רָ֥מָה יָדְךָ֖ בַּל־ יֶחֱזָי֑וּן יֶחֱז֤וּ וְיֵבֹ֙שׁוּ֙ קִנְאַת־ עָ֔ם אַף־ אֵ֖שׁ צָרֶ֥יךָ תֹאכְלֵֽם׃ ס
12תִּשְׁפֹּ֥ת שָׁל֖וֹם לָ֑נוּ כִּ֛י גַּ֥ם כָּֽל־ מַעֲשֵׂ֖ינוּ פָּעַ֥לְתָּ לָּֽנוּ׃
13אֱלֹהֵ֔ינוּ בְּעָל֥וּנוּ אֲדֹנִ֖ים זֽוּלָתֶ֑ךָ לְבַד־ בְּךָ֖ נַזְכִּ֥יר שְׁמֶֽךָ׃
14בַּל־ יִחְי֔וּ רְפָאִ֖ים בַּל־ יָקֻ֑מוּ לָכֵ֤ן פָּקַ֙דְתָּ֙ וַתַּשְׁמִידֵ֔ם וַתְּאַבֵּ֥ד כָּל־ זֵ֖כֶר לָֽמוֹ׃
15לַגּוֹי֙ יְהוָ֔ה יָסַ֥פְתָּ לַגּ֖וֹי נִכְבָּ֑דְתָּ רִחַ֖קְתָּ כָּל־ קַצְוֵי־ אָֽרֶץ׃
16בַּצַּ֣ר פְּקָד֑וּךָ צָק֣וּן לַ֔חַשׁ מוּסָרְךָ֖ לָֽמוֹ׃
17הָרָה֙ תַּקְרִ֣יב לָלֶ֔דֶת תָּחִ֥יל תִּזְעַ֖ק בַּחֲבָלֶ֑יהָ כֵּ֛ן הָיִ֥ינוּ מִפָּנֶ֖יךָ יְהוָֽה׃
18חַ֔לְנוּ כְּמ֖וֹ יָלַ֣דְנוּ ר֑וּחַ יְשׁוּעֹת֙ בַּל־ נַ֣עֲשֶׂה אֶ֔רֶץ וּבַֽל־ יִפְּל֖וּ יֹשְׁבֵ֥י תֵבֵֽל׃
19מֵתֶ֔יךָ נְבֵלָתִ֖י יְקוּמ֑וּן הָקִ֨יצוּ וְרַנְּנ֜וּ שֹׁכְנֵ֣י עָפָ֗ר כִּ֣י טַ֤ל אוֹרֹת֙ טַלֶּ֔ךָ וָאָ֖רֶץ רְפָאִ֥ים תַּפִּֽיל׃ ס
20עַמִּי֙ בֹּ֣א בַחֲדָרֶ֔יךָ וּֽסְגֹ֥ר [דְּלָתֶיךָ כ] (דְּלָתְךָ֖ ק) בַּעֲדֶ֑ךָ חֲבִ֥י כִמְעַט־ רֶ֖גַע עַד־ [יַעֲבֹור כ] (יַעֲבָר־ זָֽעַם׃ ק)
