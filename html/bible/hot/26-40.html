1בְּעֶשְׂרִ֣ים וְחָמֵ֣שׁ שָׁנָ֣ה לְ֠גָלוּתֵנוּ בְּרֹ֨אשׁ הַשָּׁנָ֜ה בֶּעָשׂ֣וֹר לַחֹ֗דֶשׁ בְּאַרְבַּ֤ע עֶשְׂרֵה֙ שָׁנָ֔ה אַחַ֕ר אֲשֶׁ֥ר הֻכְּתָ֖ה הָעִ֑יר בְּעֶ֣צֶם ׀ הַיּ֣וֹם הַזֶּ֗ה הָיְתָ֤ה עָלַי֙ יַד־ יְהוָ֔ה וַיָּבֵ֥א אֹתִ֖י שָֽׁמָּה׃
2אֱלֹהִ֔ים הֱבִיאַ֖נִי אֶל־ אֶ֣רֶץ יִשְׂרָאֵ֑ל וַיְנִיחֵ֗נִי אֶל־ הַ֤ר גָּבֹ֙הַּ֙ מְאֹ֔ד וְעָלָ֥יו כְּמִבְנֵה־ עִ֖יר מִנֶּֽגֶב׃
3אוֹתִ֜י שָׁ֗מָּה וְהִנֵּה־ אִישׁ֙ מַרְאֵ֙הוּ֙ כְּמַרְאֵ֣ה נְחֹ֔שֶׁת וּפְתִיל־ פִּשְׁתִּ֥ים בְּיָד֖וֹ וּקְנֵ֣ה הַמִּדָּ֑ה וְה֥וּא עֹמֵ֖ד בַּשָּֽׁעַר׃
4אֵלַ֜י הָאִ֗ישׁ בֶּן־ אָדָ֡ם רְאֵ֣ה בְעֵינֶיךָ֩ וּבְאָזְנֶ֨יךָ שְּׁמָ֜ע וְשִׂ֣ים לִבְּךָ֗ לְכֹ֤ל אֲשֶׁר־ אֲנִי֙ מַרְאֶ֣ה אוֹתָ֔ךְ כִּ֛י לְמַ֥עַן הַרְאוֹתְכָ֖ה הֻבָ֣אתָה הֵ֑נָּה הַגֵּ֛ד אֶת־ כָּל־ אֲשֶׁר־ אַתָּ֥ה רֹאֶ֖ה לְבֵ֥ית יִשְׂרָאֵֽל׃
5חוֹמָ֛ה מִח֥וּץ לַבַּ֖יִת סָבִ֣יב ׀ סָבִ֑יב וּבְיַ֨ד הָאִ֜ישׁ קְנֵ֣ה הַמִּדָּ֗ה שֵׁשׁ־ אַמּ֤וֹת בָּֽאַמָּה֙ וָטֹ֔פַח וַיָּ֜מָד אֶת־ רֹ֤חַב הַבִּנְיָן֙ קָנֶ֣ה אֶחָ֔ד וְקוֹמָ֖ה קָנֶ֥ה אֶחָֽד׃
6אֶל־ שַׁ֙עַר֙ אֲשֶׁ֤ר פָּנָיו֙ דֶּ֣רֶךְ הַקָּדִ֔ימָה וַיַּ֖עַל [בְּמַעֲלֹותֹו כ] (בְּמַֽעֲלֹותָ֑יו ק) וַיָּ֣מָד ׀ אֶת־ סַ֣ף הַשַּׁ֗עַר קָנֶ֤ה אֶחָד֙ רֹ֔חַב וְאֵת֙ סַ֣ף אֶחָ֔ד קָנֶ֥ה אֶחָ֖ד רֹֽחַב׃
7קָנֶ֨ה אֶחָ֥ד אֹ֙רֶךְ֙ וְקָנֶ֤ה אֶחָד֙ רֹ֔חַב וּבֵ֥ין הַתָּאִ֖ים חָמֵ֣שׁ אַמּ֑וֹת וְסַ֣ף הַ֠שַּׁעַר מֵאֵ֨צֶל אוּלָ֥ם הַשַּׁ֛עַר מֵֽהַבַּ֖יִת קָנֶ֥ה אֶחָֽד׃
8אֶת־ אֻלָ֥ם הַשַּׁ֛עַר מֵהַבַּ֖יִת קָנֶ֥ה אֶחָֽד׃
9אֶת־ אֻלָ֤ם הַשַּׁ֙עַר֙ שְׁמֹנֶ֣ה אַמּ֔וֹת [וְאֵילֹו כ] (וְאֵילָ֖יו ק) שְׁתַּ֣יִם אַמּ֑וֹת וְאֻלָ֥ם הַשַּׁ֖עַר מֵהַבָּֽיִת׃
10הַשַּׁ֜עַר דֶּ֣רֶךְ הַקָּדִ֗ים שְׁלֹשָׁ֤ה מִפֹּה֙ וּשְׁלֹשָׁ֣ה מִפֹּ֔ה מִדָּ֥ה אַחַ֖ת לִשְׁלָשְׁתָּ֑ם וּמִדָּ֥ה אַחַ֛ת לָאֵילִ֖ם מִפֹּ֥ה וּמִפּֽוֹ׃
11אֶת־ רֹ֥חַב פֶּֽתַח־ הַשַּׁ֖עַר עֶ֣שֶׂר אַמּ֑וֹת אֹ֣רֶךְ הַשַּׁ֔עַר שְׁל֥וֹשׁ עֶשְׂרֵ֖ה אַמּֽוֹת׃
12לִפְנֵ֤י הַתָּאוֹת֙ אַמָּ֣ה אֶחָ֔ת וְאַמָּה־ אַחַ֥ת גְּב֖וּל מִפֹּ֑ה וְהַתָּ֕א שֵׁשׁ־ אַמּ֣וֹת מִפּ֔וֹ וְשֵׁ֥שׁ אַמּ֖וֹת מִפּֽוֹ׃
13אֶת־ הַשַּׁ֗עַר מִגַּ֤ג הַתָּא֙ לְגַגּ֔וֹ רֹ֕חַב עֶשְׂרִ֥ים וְחָמֵ֖שׁ אַמּ֑וֹת פֶּ֖תַח נֶ֥גֶד פָּֽתַח׃
14אֶת־ אֵילִ֖ים שִׁשִּׁ֣ים אַמָּ֑ה וְאֶל־ אֵיל֙ הֶֽחָצֵ֔ר הַשַּׁ֖עַר סָבִ֥יב ׀ סָבִֽיב׃
15פְּנֵי֙ הַשַּׁ֣עַר [הַיֹּאתֹון כ] (הָֽאִיתֹ֔ון ק) עַל־ לִפְנֵ֕י אֻלָ֥ם הַשַּׁ֖עַר הַפְּנִימִ֑י חֲמִשִּׁ֖ים אַמָּֽה׃
16אֲטֻמ֣וֹת אֶֽל־ הַתָּאִ֡ים וְאֶל֩ אֵלֵיהֵ֨מָה לִפְנִ֤ימָה לַשַּׁ֙עַר֙ סָבִ֣יב ׀ סָבִ֔יב וְכֵ֖ן לָאֵֽלַמּ֑וֹת וְחַלּוֹנ֞וֹת סָבִ֤יב ׀ סָבִיב֙ לִפְנִ֔ימָה וְאֶל־ אַ֖יִל תִּמֹרִֽים׃
17אֶל־ הֶֽחָצֵר֙ הַחִ֣יצוֹנָ֔ה וְהִנֵּ֤ה לְשָׁכוֹת֙ וְרִֽצְפָ֔ה עָשׂ֥וּי לֶחָצֵ֖ר סָבִ֣יב ׀ סָבִ֑יב שְׁלֹשִׁ֥ים לְשָׁכ֖וֹת אֶל־ הָרִֽצְפָֽה׃
18אֶל־ כֶּ֣תֶף הַשְּׁעָרִ֔ים לְעֻמַּ֖ת אֹ֣רֶךְ הַשְּׁעָרִ֑ים הָרִֽצְפָ֖ה הַתַּחְתּוֹנָֽה׃
19רֹ֡חַב מִלִּפְנֵי֩ הַשַּׁ֨עַר הַתַּחְתּ֜וֹנָה לִפְנֵ֨י הֶחָצֵ֧ר הַפְּנִימִ֛י מִח֖וּץ מֵאָ֣ה אַמָּ֑ה הַקָּדִ֖ים וְהַצָּפֽוֹן׃
20אֲשֶׁ֤ר פָּנָיו֙ דֶּ֣רֶךְ הַצָּפ֔וֹן לֶחָצֵ֖ר הַחִֽיצוֹנָ֑ה מָדַ֥ד אָרְכּ֖וֹ וְרָחְבּֽוֹ׃
21כ] (וְתָאָ֗יו ק) שְׁלוֹשָׁ֤ה מִפּוֹ֙ וּשְׁלֹשָׁ֣ה מִפּ֔וֹ [וְאֵילֹו כ] (וְאֵילָ֤יו ק) [וְאֵלַמֹּו כ] (וְאֵֽלַמָּיו֙ ק) הָיָ֔ה כְּמִדַּ֖ת הַשַּׁ֣עַר הָרִאשׁ֑וֹן חֲמִשִּׁ֤ים אַמָּה֙ אָרְכּ֔וֹ וְרֹ֕חַב חָמֵ֥שׁ וְעֶשְׂרִ֖ים בָּאַמָּֽה׃
22כ] (וְחַלֹּונָ֤יו ק) [וְאֵלַמֹּו כ] (וְאֵֽלַמָּיו֙ ק) [וְתִמֹרֹו כ] (וְתִ֣מֹרָ֔יו ק) כְּמִדַּ֣ת הַשַּׁ֔עַר אֲשֶׁ֥ר פָּנָ֖יו דֶּ֣רֶךְ הַקָּדִ֑ים וּבְמַעֲל֥וֹת שֶׁ֙בַע֙ יַֽעֲלוּ־ ב֔וֹ [וְאֵילַמֹּו כ] (וְאֵֽילַמָּ֖יו ק) לִפְנֵיהֶֽם׃
23לֶחָצֵ֣ר הַפְּנִימִ֔י נֶ֣גֶד הַשַּׁ֔עַר לַצָּפ֖וֹן וְלַקָּדִ֑ים וַיָּ֧מָד מִשַּׁ֛עַר אֶל־ שַׁ֖עַר מֵאָ֥ה אַמָּֽה׃
24דֶּ֣רֶךְ הַדָּר֔וֹם וְהִנֵּה־ שַׁ֖עַר דֶּ֣רֶךְ הַדָּר֑וֹם וּמָדַ֤ד [אֵילֹו כ] (אֵילָיו֙ ק) [וְאֵילַמֹּו כ] (וְאֵ֣ילַמָּ֔יו ק) כַּמִּדּ֖וֹת הָאֵֽלֶּה׃
25ל֤וֹ [וּלְאֵילַמֹּו כ] (וּלְאֵֽילַמָּיו֙ ק) סָבִ֣יב ׀ סָבִ֔יב כְּהַחֲלֹּנ֖וֹת הָאֵ֑לֶּה חֲמִשִּׁ֤ים אַמָּה֙ אֹ֔רֶךְ וְרֹ֕חַב חָמֵ֥שׁ וְעֶשְׂרִ֖ים אַמָּֽה׃
26שִׁבְעָה֙ [עֹלֹותֹו כ] (עֹֽלֹותָ֔יו ק) [וְאֵלַמֹּו כ] (וְאֵֽלַמָּ֖יו ק) לִפְנֵיהֶ֑ם וְתִמֹרִ֣ים ל֗וֹ אֶחָ֥ד מִפּ֛וֹ וְאֶחָ֥ד מִפּ֖וֹ אֶל־ [אֵילֹו כ] (אֵילָֽיו׃ ק)
27לֶחָצֵ֥ר הַפְּנִימִ֖י דֶּ֣רֶךְ הַדָּר֑וֹם וַיָּ֨מָד מִשַּׁ֧עַר אֶל־ הַשַּׁ֛עַר דֶּ֥רֶךְ הַדָּר֖וֹם מֵאָ֥ה אַמּֽוֹת׃
28אֶל־ חָצֵ֥ר הַפְּנִימִ֖י בְּשַׁ֣עַר הַדָּר֑וֹם וַיָּ֙מָד֙ אֶת־ הַשַּׁ֣עַר הַדָּר֔וֹם כַּמִּדּ֖וֹת הָאֵֽלֶּה׃
29כ] (וְתָאָ֞יו ק) [וְאֵילֹו כ] (וְאֵילָ֤יו ק) [וְאֵלַמֹּו כ] (וְאֵֽלַמָּיו֙ ק) כַּמִּדּ֣וֹת הָאֵ֔לֶּה וְחַלּוֹנ֥וֹת ל֛וֹ [וּלְאֵלַמֹּו כ] (וּלְאֵלַמָּ֖יו ק) סָבִ֣יב ׀ סָבִ֑יב חֲמִשִּׁ֤ים אַמָּה֙ אֹ֔רֶךְ וְרֹ֕חַב עֶשְׂרִ֥ים וְחָמֵ֖שׁ אַמּֽוֹת׃
30סָבִ֣יב ׀ סָבִ֑יב אֹ֗רֶךְ חָמֵ֤שׁ וְעֶשְׂרִים֙ אַמָּ֔ה וְרֹ֖חַב חָמֵ֥שׁ אַמּֽוֹת׃
31אֶל־ חָצֵר֙ הַחִ֣צוֹנָ֔ה וְתִמֹרִ֖ים אֶל־ [אֵילֹו כ] (אֵילָ֑יו ק) וּמַעֲל֥וֹת שְׁמוֹנֶ֖ה [מַעֲלֹו כ] (מַעֲלָֽיו׃ ק)
32אֶל־ הֶחָצֵ֥ר הַפְּנִימִ֖י דֶּ֣רֶךְ הַקָּדִ֑ים וַיָּ֣מָד אֶת־ הַשַּׁ֔עַר כַּמִּדּ֖וֹת הָאֵֽלֶּה׃
33כ] (וְתָאָ֞יו ק) [וְאֵלֹו כ] (וְאֵלָ֤יו ק) [וְאֵלַמֹּו כ] (וְאֵֽלַמָּיו֙ ק) כַּמִּדּ֣וֹת הָאֵ֔לֶּה וְחַלּוֹנ֥וֹת ל֛וֹ [וּלְאֵלַמֹּו כ] (וּלְאֵלַמָּ֖יו ק) סָבִ֣יב ׀ סָבִ֑יב אֹ֚רֶךְ חֲמִשִּׁ֣ים אַמָּ֔ה וְרֹ֕חַב חָמֵ֥שׁ וְעֶשְׂרִ֖ים אַמָּֽה׃
34כ] (וְאֵלַמָּ֗יו ק) לֶֽחָצֵר֙ הַחִ֣יצוֹנָ֔ה וְתִמֹרִ֥ים אֶל־ [אֵלֹו כ] (אֵלָ֖יו ק) מִפּ֣וֹ וּמִפּ֑וֹ וּשְׁמֹנֶ֥ה מַעֲל֖וֹת [מַעֲלֹו כ] (מַעֲלָֽיו׃ ק)
35אֶל־ שַׁ֣עַר הַצָּפ֑וֹן וּמָדַ֖ד כַּמִּדּ֥וֹת הָאֵֽלֶּה׃
36כ] (תָּאָיו֙ ק) [אֵלֹו כ] (אֵלָ֣יו ק) [וְאֵלַמֹּו כ] (וְאֵֽלַמָּ֔יו ק) וְחַלּוֹנ֥וֹת ל֖וֹ סָבִ֣יב ׀ סָבִ֑יב אֹ֚רֶךְ חֲמִשִּׁ֣ים אַמָּ֔ה וְרֹ֕חַב חָמֵ֥שׁ וְעֶשְׂרִ֖ים אַמָּֽה׃
37כ] (וְאֵילָ֗יו ק) לֶֽחָצֵר֙ הַחִ֣יצוֹנָ֔ה וְתִמֹרִ֥ים אֶל־ [אֵילֹו כ] (אֵילָ֖יו ק) מִפּ֣וֹ וּמִפּ֑וֹ וּשְׁמֹנֶ֥ה מַעֲל֖וֹת [מַעֲלֹו כ] (מַעֲלָֽיו׃ ק)
38וּפִתְחָ֔הּ בְּאֵילִ֖ים הַשְּׁעָרִ֑ים שָׁ֖ם יָדִ֥יחוּ אֶת־ הָעֹלָֽה׃
39הַשַּׁ֗עַר שְׁנַ֤יִם שֻׁלְחָנוֹת֙ מִפּ֔וֹ וּשְׁנַ֥יִם שֻׁלְחָנ֖וֹת מִפֹּ֑ה לִשְׁח֤וֹט אֲלֵיהֶם֙ הָעוֹלָ֔ה וְהַחַטָּ֖את וְהָאָשָֽׁם׃
40הַכָּתֵ֣ף מִח֗וּצָה לָעוֹלֶה֙ לְפֶ֙תַח֙ הַשַּׁ֣עַר הַצָּפ֔וֹנָה שְׁנַ֖יִם שֻׁלְחָנ֑וֹת וְאֶל־ הַכָּתֵ֣ף הָאַחֶ֗רֶת אֲשֶׁר֙ לְאֻלָ֣ם הַשַּׁ֔עַר שְׁנַ֖יִם שֻׁלְחָנֽוֹת׃
41שֻׁלְחָנ֜וֹת מִפֹּ֗ה וְאַרְבָּעָ֧ה שֻׁלְחָנ֛וֹת מִפֹּ֖ה לְכֶ֣תֶף הַשָּׁ֑עַר שְׁמוֹנָ֥ה שֻׁלְחָנ֖וֹת אֲלֵיהֶ֥ם יִשְׁחָֽטוּ׃
42שֻׁלְחָנ֨וֹת לָעוֹלָ֜ה אַבְנֵ֣י גָזִ֗ית אֹרֶךְ֩ אַמָּ֨ה אַחַ֤ת וָחֵ֙צִי֙ וְרֹ֨חַב אַמָּ֤ה אַחַת֙ וָחֵ֔צִי וְגֹ֖בַהּ אַמָּ֣ה אֶחָ֑ת אֲלֵיהֶ֗ם וְיַנִּ֤יחוּ אֶת־ הַכֵּלִים֙ אֲשֶׁ֨ר יִשְׁחֲט֧וּ אֶת־ הָעוֹלָ֛ה בָּ֖ם וְהַזָּֽבַח׃
43טֹ֧פַח אֶחָ֛ד מוּכָנִ֥ים בַּבַּ֖יִת סָבִ֣יב ׀ סָבִ֑יב וְאֶל־ הַשֻּׁלְחָנ֖וֹת בְּשַׂ֥ר הַקָּרְבָֽן׃
44לַשַּׁ֨עַר הַפְּנִימִ֜י לִֽשְׁכ֣וֹת שָׁרִ֗ים בֶּחָצֵ֤ר הַפְּנִימִי֙ אֲשֶׁ֗ר אֶל־ כֶּ֙תֶף֙ שַׁ֣עַר הַצָּפ֔וֹן וּפְנֵיהֶ֖ם דֶּ֣רֶךְ הַדָּר֑וֹם אֶחָ֗ד אֶל־ כֶּ֙תֶף֙ שַׁ֣עַר הַקָּדִ֔ים פְּנֵ֖י דֶּ֥רֶךְ הַצָּפֹֽן׃
45אֵלָ֑י זֹ֣ה הַלִּשְׁכָּ֗ה אֲשֶׁ֤ר פָּנֶ֙יהָ֙ דֶּ֣רֶךְ הַדָּר֔וֹם לַכֹּ֣הֲנִ֔ים שֹׁמְרֵ֖י מִשְׁמֶ֥רֶת הַבָּֽיִת׃
46אֲשֶׁ֤ר פָּנֶ֙יהָ֙ דֶּ֣רֶךְ הַצָּפ֔וֹן לַכֹּ֣הֲנִ֔ים שֹׁמְרֵ֖י מִשְׁמֶ֣רֶת הַמִּזְבֵּ֑חַ הֵ֣מָּה בְנֵֽי־ צָד֗וֹק הַקְּרֵבִ֧ים מִבְּנֵֽי־ לֵוִ֛י אֶל־ יְהוָ֖ה לְשָׁרְתֽוֹ׃
47אֶת־ הֶחָצֵ֜ר אֹ֣רֶךְ ׀ מֵאָ֣ה אַמָּ֗ה וְרֹ֛חַב מֵאָ֥ה אַמָּ֖ה מְרֻבָּ֑עַת וְהַמִּזְבֵּ֖חַ לִפְנֵ֥י הַבָּֽיִת׃
48אֶל־ אֻלָ֣ם הַבַּיִת֒ וַיָּ֙מָד֙ אֵ֣ל אֻלָ֔ם חָמֵ֤שׁ אַמּוֹת֙ מִפֹּ֔ה וְחָמֵ֥שׁ אַמּ֖וֹת מִפֹּ֑ה וְרֹ֣חַב הַשַּׁ֔עַר שָׁלֹ֤שׁ אַמּוֹת֙ מִפּ֔וֹ וְשָׁלֹ֥שׁ אַמּ֖וֹת מִפּֽוֹ׃
