1אָ֭דָם יְל֣וּד אִשָּׁ֑ה קְצַ֥ר יָ֝מִ֗ים וּֽשְׂבַֽע־ רֹֽגֶז׃
2יָ֭צָא וַיִּמָּ֑ל וַיִּבְרַ֥ח כַּ֝צֵּ֗ל וְלֹ֣א יַעֲמֽוֹד׃ אַף־ עַל־
3פָּקַ֣חְתָּ עֵינֶ֑ךָ וְאֹ֘תִ֤י תָבִ֖יא בְמִשְׁפָּ֣ט עִמָּֽךְ׃
4יִתֵּ֣ן טָ֭הוֹר מִטָּמֵ֗א לֹ֣א אֶחָֽד׃
5חֲרוּצִ֨ים ׀ יָמָ֗יו מִֽסְפַּר־ חֳדָשָׁ֥יו אִתָּ֑ךְ [חֻקֹּו כ] (חֻקָּ֥יו ק) עָ֝שִׂ֗יתָ וְלֹ֣א יַעֲבֽוֹר׃
6מֵעָלָ֣יו וְיֶחְדָּ֑ל עַד־ יִ֝רְצֶ֗ה כְּשָׂכִ֥יר יוֹמֽוֹ׃
7יֵ֥שׁ לָעֵ֗ץ תִּ֫קְוָ֥ה אִֽם־ יִ֭כָּרֵת וְע֣וֹד יַחֲלִ֑יף וְ֝יֹֽנַקְתּ֗וֹ לֹ֣א תֶחְדָּֽל׃
8יַזְקִ֣ין בָּאָ֣רֶץ שָׁרְשׁ֑וֹ וּ֝בֶעָפָ֗ר יָמ֥וּת גִּזְעֽוֹ׃
9מַ֣יִם יַפְרִ֑חַ וְעָשָׂ֖ה קָצִ֣יר כְּמוֹ־ נָֽטַע׃
10יָ֭מוּת וַֽיֶּחֱלָ֑שׁ וַיִּגְוַ֖ע אָדָ֣ם וְאַיּֽוֹ׃
11מַ֭יִם מִנִּי־ יָ֑ם וְ֝נָהָ֗ר יֶחֱרַ֥ב וְיָבֵֽשׁ׃
12שָׁכַ֗ב וְֽלֹא־ יָ֫ק֥וּם עַד־ בִּלְתִּ֣י שָׁ֭מַיִם לֹ֣א יָקִ֑יצוּ וְלֹֽא־ יֵ֝עֹ֗רוּ מִשְּׁנָתָֽם׃
13יִתֵּ֨ן ׀ בִּשְׁא֬וֹל תַּצְפִּנֵ֗נִי תַּ֭סְתִּירֵנִי עַד־ שׁ֣וּב אַפֶּ֑ךָ תָּ֤שִׁ֥ית לִ֖י חֹ֣ק וְתִזְכְּרֵֽנִי׃
14יָמ֥וּת גֶּ֗בֶר הֲיִ֫חְיֶ֥ה כָּל־ יְמֵ֣י צְבָאִ֣י אֲיַחֵ֑ל עַד־ בּ֝֗וֹא חֲלִיפָתִֽי׃
15וְאָנֹכִ֣י אֶֽעֱנֶ֑ךָּ לְֽמַעֲשֵׂ֖ה יָדֶ֣יךָ תִכְסֹֽף׃
16עַ֭תָּה צְעָדַ֣י תִּסְפּ֑וֹר לֹֽא־ תִ֝שְׁמ֗וֹר עַל־ חַטָּאתִֽי׃
17בִּצְר֣וֹר פִּשְׁעִ֑י וַ֝תִּטְפֹּ֗ל עַל־ עֲוֹנִֽי׃
18הַר־ נוֹפֵ֣ל יִבּ֑וֹל וְ֝צ֗וּר יֶעְתַּ֥ק מִמְּקֹמֽוֹ׃
19שָׁ֥חֲקוּ מַ֗יִם תִּשְׁטֹֽף־ סְפִיחֶ֥יהָ עֲפַר־ אָ֑רֶץ וְתִקְוַ֖ת אֱנ֣וֹשׁ הֶאֱבַֽדְתָּ׃
20לָ֭נֶצַח וַֽיַּהֲלֹ֑ךְ מְשַׁנֶּ֥ה פָ֝נָ֗יו וַֽתְּשַׁלְּחֵֽהוּ׃
21בָ֭נָיו וְלֹ֣א יֵדָ֑ע וְ֝יִצְעֲר֗וּ וְֽלֹא־ יָבִ֥ין לָֽמוֹ׃
