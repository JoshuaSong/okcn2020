1לַ֝מְנַצֵּ֗חַ לְדָוִ֥ד מִזְמֽוֹר׃ קַוֹּ֣ה קִוִּ֣יתִי יְהוָ֑ה וַיֵּ֥ט אֵ֝לַ֗י וַיִּשְׁמַ֥ע שַׁוְעָתִֽי׃
2מִבּ֥וֹר שָׁאוֹן֮ מִטִּ֪יט הַיָּ֫וֵ֥ן וַיָּ֖קֶם עַל־ סֶ֥לַע רַגְלַ֗י כּוֹנֵ֥ן אֲשֻׁרָֽי׃
3בְּפִ֨י ׀ שִׁ֥יר חָדָשׁ֮ תְּהִלָּ֪ה לֵֽאלֹ֫הֵ֥ינוּ יִרְא֣וּ רַבִּ֣ים וְיִירָ֑אוּ וְ֝יִבְטְח֗וּ בַּיהוָֽה׃
4הַגֶּ֗בֶר אֲשֶׁר־ שָׂ֣ם יְ֭הֹוָה מִבְטַח֑וֹ וְֽלֹא־ פָנָ֥ה אֶל־ רְ֝הָבִ֗ים וְשָׂטֵ֥י כָזָֽב׃
5עָשִׂ֨יתָ ׀ אַתָּ֤ה ׀ יְהוָ֣ה אֱלֹהַי֮ נִֽפְלְאֹתֶ֥יךָ וּמַחְשְׁבֹתֶ֗יךָ אֵ֫לֵ֥ינוּ אֵ֤ין ׀ עֲרֹ֬ךְ אֵלֶ֗יךָ אַגִּ֥ידָה וַאֲדַבֵּ֑רָה עָ֝צְמ֗וּ מִסַּפֵּֽר׃
6וּמִנְחָ֨ה ׀ לֹֽא־ חָפַ֗צְתָּ אָ֭זְנַיִם כָּרִ֣יתָ לִּ֑י עוֹלָ֥ה וַ֝חֲטָאָ֗ה לֹ֣א שָׁאָֽלְתָּ׃
7אָ֭מַרְתִּי הִנֵּה־ בָ֑אתִי בִּמְגִלַּת־ סֵ֝֗פֶר כָּת֥וּב עָלָֽי׃
8רְצוֹנְךָ֣ אֱלֹהַ֣י חָפָ֑צְתִּי וְ֝ת֥וֹרָתְךָ֗ בְּת֣וֹךְ מֵעָֽי׃
9צֶ֨דֶק ׀ בְּקָ֘הָ֤ל רָ֗ב הִנֵּ֣ה שְׂ֭פָתַי לֹ֣א אֶכְלָ֑א יְ֝הוָ֗ה אַתָּ֥ה יָדָֽעְתָּ׃
10לֹא־ כִסִּ֨יתִי ׀ בְּת֬וֹךְ לִבִּ֗י אֱמוּנָתְךָ֣ וּתְשׁוּעָתְךָ֣ אָמָ֑רְתִּי לֹא־ כִחַ֥דְתִּי חַסְדְּךָ֥ וַ֝אֲמִתְּךָ֗ לְקָהָ֥ל רָֽב׃
11יְהוָ֗ה לֹא־ תִכְלָ֣א רַחֲמֶ֣יךָ מִמֶּ֑נִּי חַסְדְּךָ֥ וַ֝אֲמִתְּךָ֗ תָּמִ֥יד יִצְּרֽוּנִי׃
12אָפְפ֥וּ־ עָלַ֨י ׀ רָע֡וֹת עַד־ אֵ֬ין מִסְפָּ֗ר הִשִּׂיג֣וּנִי עֲ֭וֹנֹתַי וְלֹא־ יָכֹ֣לְתִּי לִרְא֑וֹת עָצְמ֥וּ מִשַּֽׂעֲר֥וֹת רֹ֝אשִׁ֗י וְלִבִּ֥י עֲזָבָֽנִי׃
13יְ֭הוָה לְהַצִּילֵ֑נִי יְ֝הוָ֗ה לְעֶזְרָ֥תִי חֽוּשָׁה׃
14וְיַחְפְּר֨וּ ׀ יַחַד֮ מְבַקְשֵׁ֥י נַפְשִׁ֗י לִסְפּ֫וֹתָ֥הּ יִסֹּ֣גוּ אָ֭חוֹר וְיִכָּלְמ֑וּ חֲ֝פֵצֵ֗י רָעָתִֽי׃
15עַל־ עֵ֣קֶב בָּשְׁתָּ֑ם הָאֹמְרִ֥ים לִ֝֗י הֶאָ֥ח ׀ הֶאָֽח׃
16וְיִשְׂמְח֨וּ ׀ בְּךָ֗ כָּֽל־ מְבַ֫קְשֶׁ֥יךָ יֹאמְר֣וּ תָ֭מִיד יִגְדַּ֣ל יְהוָ֑ה אֹֽ֝הֲבֵ֗י תְּשׁוּעָתֶֽךָ׃
