1וְהֶחֱזִיקוּ֩ שֶׁ֨בַע נָשִׁ֜ים בְּאִ֣ישׁ אֶחָ֗ד בַּיּ֤וֹם הַהוּא֙ לֵאמֹ֔ר לַחְמֵ֣נוּ נֹאכֵ֔ל וְשִׂמְלָתֵ֖נוּ נִלְבָּ֑שׁ רַ֗ק יִקָּרֵ֤א שִׁמְךָ֙ עָלֵ֔ינוּ אֱסֹ֖ף חֶרְפָּתֵֽנוּ׃ ס
2הַה֗וּא יִֽהְיֶה֙ צֶ֣מַח יְהוָ֔ה לִצְבִ֖י וּלְכָב֑וֹד וּפְרִ֤י הָאָ֙רֶץ֙ לְגָא֣וֹן וּלְתִפְאֶ֔רֶת לִפְלֵיטַ֖ת יִשְׂרָאֵֽל׃
3הַנִּשְׁאָ֣ר בְּצִיּ֗וֹן וְהַנּוֹתָר֙ בִּיר֣וּשָׁלִַ֔ם קָד֖וֹשׁ יֵאָ֣מֶר ל֑וֹ כָּל־ הַכָּת֥וּב לַחַיִּ֖ים בִּירוּשָׁלִָֽם׃
4רָחַ֣ץ אֲדֹנָ֗י אֵ֚ת צֹאַ֣ת בְּנוֹת־ צִיּ֔וֹן וְאֶת־ דְּמֵ֥י יְרוּשָׁלִַ֖ם יָדִ֣יחַ מִקִּרְבָּ֑הּ בְּר֥וּחַ מִשְׁפָּ֖ט וּבְר֥וּחַ בָּעֵֽר׃
5יְהוָ֡ה עַל֩ כָּל־ מְכ֨וֹן הַר־ צִיּ֜וֹן וְעַל־ מִקְרָאֶ֗הָ עָנָ֤ן ׀ יוֹמָם֙ וְעָשָׁ֔ן וְנֹ֛גַהּ אֵ֥שׁ לֶהָבָ֖ה לָ֑יְלָה כִּ֥י עַל־ כָּל־ כָּב֖וֹד חֻפָּֽה׃
