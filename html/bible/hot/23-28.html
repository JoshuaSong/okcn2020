1ה֗וֹי עֲטֶ֤רֶת גֵּאוּת֙ שִׁכֹּרֵ֣י אֶפְרַ֔יִם וְצִ֥יץ נֹבֵ֖ל צְבִ֣י תִפְאַרְתּ֑וֹ אֲשֶׁ֛ר עַל־ רֹ֥אשׁ גֵּֽיא־ שְׁמָנִ֖ים הֲל֥וּמֵי יָֽיִן׃
2חָזָ֤ק וְאַמִּץ֙ לַֽאדֹנָ֔י כְּזֶ֥רֶם בָּרָ֖ד שַׂ֣עַר קָ֑טֶב כְּ֠זֶרֶם מַ֣יִם כַּבִּירִ֥ים שֹׁטְפִ֛ים הִנִּ֥יחַ לָאָ֖רֶץ בְּיָֽד׃
3תֵּֽרָמַ֑סְנָה עֲטֶ֥רֶת גֵּא֖וּת שִׁכּוֹרֵ֥י אֶפְרָֽיִם׃
4צִיצַ֤ת נֹבֵל֙ צְבִ֣י תִפְאַרְתּ֔וֹ אֲשֶׁ֥ר עַל־ רֹ֖אשׁ גֵּ֣יא שְׁמָנִ֑ים כְּבִכּוּרָהּ֙ בְּטֶ֣רֶם קַ֔יִץ אֲשֶׁ֨ר יִרְאֶ֤ה הָֽרֹאֶה֙ אוֹתָ֔הּ בְּעוֹדָ֥הּ בְּכַפּ֖וֹ יִבְלָעֶֽנָּה׃ ס
5הַה֗וּא יִֽהְיֶה֙ יְהוָ֣ה צְבָא֔וֹת לַעֲטֶ֣רֶת צְבִ֔י וְלִצְפִירַ֖ת תִּפְאָרָ֑ה לִשְׁאָ֖ר עַמּֽוֹ׃
6מִשְׁפָּ֑ט לַיּוֹשֵׁב֙ עַל־ הַמִּשְׁפָּ֔ט וְלִ֨גְבוּרָ֔ה מְשִׁיבֵ֥י מִלְחָמָ֖ה שָֽׁעְרָה׃ ס
7אֵ֙לֶּה֙ בַּיַּ֣יִן שָׁג֔וּ וּבַשֵּׁכָ֖ר תָּע֑וּ כֹּהֵ֣ן וְנָבִיא֩ שָׁג֨וּ בַשֵּׁכָ֜ר נִבְלְע֣וּ מִן־ הַיַּ֗יִן תָּעוּ֙ מִן־ הַשֵּׁכָ֔ר שָׁגוּ֙ בָּֽרֹאֶ֔ה פָּק֖וּ פְּלִילִיָּֽה׃
8כָּל־ שֻׁלְחָנ֔וֹת מָלְא֖וּ קִ֣יא צֹאָ֑ה בְּלִ֖י מָקֽוֹם׃ ס
9מִי֙ יוֹרֶ֣ה דֵעָ֔ה וְאֶת־ מִ֖י יָבִ֣ין שְׁמוּעָ֑ה גְּמוּלֵי֙ מֵֽחָלָ֔ב עַתִּיקֵ֖י מִשָּׁדָֽיִם׃
10צַ֤ו לָצָו֙ צַ֣ו לָצָ֔ו קַ֥ו לָקָ֖ו קַ֣ו לָקָ֑ו זְעֵ֥יר שָׁ֖ם זְעֵ֥יר שָֽׁם׃
11בְּלַעֲגֵ֣י שָׂפָ֔ה וּבְלָשׁ֖וֹן אַחֶ֑רֶת יְדַבֵּ֖ר אֶל־ הָעָ֥ם הַזֶּֽה׃
12אָמַ֣ר אֲלֵיהֶ֗ם זֹ֤את הַמְּנוּחָה֙ הָנִ֣יחוּ לֶֽעָיֵ֔ף וְזֹ֖את הַמַּרְגֵּעָ֑ה וְלֹ֥א אָב֖וּא שְׁמֽוֹעַ׃
13לָהֶ֜ם דְּבַר־ יְהוָ֗ה צַ֣ו לָצָ֞ו צַ֤ו לָצָו֙ קַ֤ו לָקָו֙ קַ֣ו לָקָ֔ו זְעֵ֥יר שָׁ֖ם זְעֵ֣יר שָׁ֑ם לְמַ֨עַן יֵלְכ֜וּ וְכָשְׁל֤וּ אָחוֹר֙ וְנִשְׁבָּ֔רוּ וְנוֹקְשׁ֖וּ וְנִלְכָּֽדוּ׃ פ
14שִׁמְע֥וּ דְבַר־ יְהוָ֖ה אַנְשֵׁ֣י לָצ֑וֹן מֹֽשְׁלֵי֙ הָעָ֣ם הַזֶּ֔ה אֲשֶׁ֖ר בִּירוּשָׁלִָֽם׃
15אֲמַרְתֶּ֗ם כָּרַ֤תְנֽוּ בְרִית֙ אֶת־ מָ֔וֶת וְעִם־ שְׁא֖וֹל עָשִׂ֣ינוּ חֹזֶ֑ה [שִׁיט כ] (שֹׁ֣וט ק) שׁוֹטֵ֤ף כִּֽי־ [עָבַר כ] (יַֽעֲבֹר֙ ק) לֹ֣א יְבוֹאֵ֔נוּ כִּ֣י שַׂ֧מְנוּ כָזָ֛ב מַחְסֵ֖נוּ וּבַשֶּׁ֥קֶר נִסְתָּֽרְנוּ׃ ס
16כֹּ֤ה אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה הִנְנִ֛י יִסַּ֥ד בְּצִיּ֖וֹן אָ֑בֶן אֶ֣בֶן בֹּ֜חַן פִּנַּ֤ת יִקְרַת֙ מוּסָ֣ד מוּסָּ֔ד הַֽמַּאֲמִ֖ין לֹ֥א יָחִֽישׁ׃
17מִשְׁפָּט֙ לְקָ֔ו וּצְדָקָ֖ה לְמִשְׁקָ֑לֶת וְיָעָ֤ה בָרָד֙ מַחְסֵ֣ה כָזָ֔ב וְסֵ֥תֶר מַ֖יִם יִשְׁטֹֽפוּ׃
18בְּרִֽיתְכֶם֙ אֶת־ מָ֔וֶת וְחָזוּתְכֶ֥ם אֶת־ שְׁא֖וֹל לֹ֣א תָק֑וּם שׁ֤וֹט שׁוֹטֵף֙ כִּ֣י יַֽעֲבֹ֔ר וִהְיִ֥יתֶם ל֖וֹ לְמִרְמָֽס׃
19עָבְרוֹ֙ יִקַּ֣ח אֶתְכֶ֔ם כִּֽי־ בַבֹּ֧קֶר בַּבֹּ֛קֶר יַעֲבֹ֖ר בַּיּ֣וֹם וּבַלָּ֑יְלָה וְהָיָ֥ה רַק־ זְוָעָ֖ה הָבִ֥ין שְׁמוּעָֽה׃
20קָצַ֥ר הַמַּצָּ֖ע מֵֽהִשְׂתָּרֵ֑עַ וְהַמַּסֵּכָ֥ה צָ֖רָה כְּהִתְכַּנֵּֽס׃
21כְהַר־ פְּרָצִים֙ יָק֣וּם יְהוָ֔ה כְּעֵ֖מֶק בְּגִבְע֣וֹן יִרְגָּ֑ז לַעֲשׂ֤וֹת מַעֲשֵׂ֙הוּ֙ זָ֣ר מַעֲשֵׂ֔הוּ וְלַֽעֲבֹד֙ עֲבֹ֣דָת֔וֹ נָכְרִיָּ֖ה עֲבֹדָתֽוֹ׃
22אַל־ תִּתְלוֹצָ֔צוּ פֶּֽן־ יֶחְזְק֖וּ מֽוֹסְרֵיכֶ֑ם כִּֽי־ כָלָ֨ה וְנֶחֱרָצָ֜ה שָׁמַ֗עְתִּי מֵאֵ֨ת אֲדֹנָ֧י יְהוִ֛ה צְבָא֖וֹת עַל־ כָּל־ הָאָֽרֶץ׃
23וְשִׁמְע֖וּ קוֹלִ֑י הַקְשִׁ֥יבוּ וְשִׁמְע֖וּ אִמְרָתִֽי׃
24הַיּ֔וֹם יַחֲרֹ֥שׁ הַחֹרֵ֖שׁ לִזְרֹ֑עַ יְפַתַּ֥ח וִֽישַׂדֵּ֖ד אַדְמָתֽוֹ׃
25אִם־ שִׁוָּ֣ה פָנֶ֔יהָ וְהֵפִ֥יץ קֶ֖צַח וְכַמֹּ֣ן יִזְרֹ֑ק וְשָׂ֨ם חִטָּ֤ה שׂוֹרָה֙ וּשְׂעֹרָ֣ה נִסְמָ֔ן וְכֻסֶּ֖מֶת גְּבֻלָתֽוֹ׃
26לַמִּשְׁפָּ֖ט אֱלֹהָ֥יו יוֹרֶֽנּוּ׃
27לֹ֤א בֶֽחָרוּץ֙ י֣וּדַשׁ קֶ֔צַח וְאוֹפַ֣ן עֲגָלָ֔ה עַל־ כַּמֹּ֖ן יוּסָּ֑ב כִּ֧י בַמַּטֶּ֛ה יֵחָ֥בֶט קֶ֖צַח וְכַמֹּ֥ן בַּשָּֽׁבֶט׃
28יוּדָ֔ק כִּ֛י לֹ֥א לָנֶ֖צַח אָד֣וֹשׁ יְדוּשֶׁ֑נּוּ וְ֠הָמַם גִּלְגַּ֧ל עֶגְלָת֛וֹ וּפָרָשָׁ֖יו לֹֽא־ יְדֻקֶּֽנּוּ׃
