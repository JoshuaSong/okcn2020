1הַדָּבָר֙ אֲשֶׁ֣ר הָיָ֣ה אֶֽל־ יִרְמְיָ֔הוּ אֶ֚ל כָּל־ הַיְּהוּדִ֔ים הַיֹּשְׁבִ֖ים בְּאֶ֣רֶץ מִצְרָ֑יִם הַיֹּשְׁבִ֤ים בְּמִגְדֹּל֙ וּבְתַחְפַּנְחֵ֣ס וּבְנֹ֔ף וּבְאֶ֥רֶץ פַּתְר֖וֹס לֵאמֹֽר׃ ס
2אָמַ֞ר יְהוָ֤ה צְבָאוֹת֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל אַתֶּ֣ם רְאִיתֶ֗ם אֵ֤ת כָּל־ הָֽרָעָה֙ אֲשֶׁ֤ר הֵבֵ֙אתִי֙ עַל־ יְר֣וּשָׁלִַ֔ם וְעַ֖ל כָּל־ עָרֵ֣י יְהוּדָ֑ה וְהִנָּ֤ם חָרְבָּה֙ הַיּ֣וֹם הַזֶּ֔ה וְאֵ֥ין בָּהֶ֖ם יוֹשֵֽׁב׃
3רָעָתָ֗ם אֲשֶׁ֤ר עָשׂוּ֙ לְהַכְעִסֵ֔נִי לָלֶ֣כֶת לְקַטֵּ֔ר לַעֲבֹ֖ד לֵאלֹהִ֣ים אֲחֵרִ֑ים אֲשֶׁר֙ לֹ֣א יְדָע֔וּם הֵ֖מָּה אַתֶּ֥ם וַאֲבֹתֵיכֶֽם׃
4אֲלֵיכֶם֙ אֶת־ כָּל־ עֲבָדַ֣י הַנְּבִיאִ֔ים הַשְׁכֵּ֥ים וְשָׁלֹ֖חַ לֵאמֹ֑ר אַל־ נָ֣א תַעֲשׂ֗וּ אֵ֛ת דְּבַֽר־ הַתֹּעֵבָ֥ה הַזֹּ֖את אֲשֶׁ֥ר שָׂנֵֽאתִי׃
5שָֽׁמְעוּ֙ וְלֹא־ הִטּ֣וּ אֶת־ אָזְנָ֔ם לָשׁ֖וּב מֵרָֽעָתָ֑ם לְבִלְתִּ֥י קַטֵּ֖ר לֵאלֹהִ֥ים אֲחֵרִֽים׃
6חֲמָתִי֙ וְאַפִּ֔י וַתִּבְעַר֙ בְּעָרֵ֣י יְהוּדָ֔ה וּבְחֻצ֖וֹת יְרֽוּשָׁלִָ֑ם וַתִּהְיֶ֛ינָה לְחָרְבָּ֥ה לִשְׁמָמָ֖ה כַּיּ֥וֹם הַזֶּֽה׃ ס
7כֹּֽה־ אָמַ֣ר יְהוָה֩ אֱלֹהֵ֨י צְבָא֜וֹת אֱלֹהֵ֣י יִשְׂרָאֵ֗ל לָמָה֩ אַתֶּ֨ם עֹשִׂ֜ים רָעָ֤ה גְדוֹלָה֙ אֶל־ נַפְשֹׁ֣תֵכֶ֔ם לְהַכְרִ֨ית לָכֶ֧ם אִישׁ־ וְאִשָּׁ֛ה עוֹלֵ֥ל וְיוֹנֵ֖ק מִתּ֣וֹךְ יְהוּדָ֑ה לְבִלְתִּ֛י הוֹתִ֥יר לָכֶ֖ם שְׁאֵרִֽית׃
8בְּמַעֲשֵׂ֣י יְדֵיכֶ֔ם לְקַטֵּ֞ר לֵאלֹהִ֤ים אֲחֵרִים֙ בְּאֶ֣רֶץ מִצְרַ֔יִם אֲשֶׁר־ אַתֶּ֥ם בָּאִ֖ים לָג֣וּר שָׁ֑ם לְמַ֙עַן֙ הַכְרִ֣ית לָכֶ֔ם וּלְמַ֤עַן הֱיֽוֹתְכֶם֙ לִקְלָלָ֣ה וּלְחֶרְפָּ֔ה בְּכֹ֖ל גּוֹיֵ֥י הָאָֽרֶץ׃
9אֶת־ רָע֨וֹת אֲבוֹתֵיכֶ֜ם וְאֶת־ רָע֣וֹת ׀ מַלְכֵ֣י יְהוּדָ֗ה וְאֵת֙ רָע֣וֹת נָשָׁ֔יו וְאֵת֙ רָעֹ֣תֵכֶ֔ם וְאֵ֖ת רָעֹ֣ת נְשֵׁיכֶ֑ם אֲשֶׁ֤ר עָשׂוּ֙ בְּאֶ֣רֶץ יְהוּדָ֔ה וּבְחֻצ֖וֹת יְרוּשָׁלִָֽם׃
10דֻכְּא֔וּ עַ֖ד הַיּ֣וֹם הַזֶּ֑ה וְלֹ֣א יָרְא֗וּ וְלֹֽא־ הָלְכ֤וּ בְתֽוֹרָתִי֙ וּבְחֻקֹּתַ֔י אֲשֶׁר־ נָתַ֥תִּי לִפְנֵיכֶ֖ם וְלִפְנֵ֥י אֲבוֹתֵיכֶֽם׃ ס
11כֹּֽה־ אָמַ֞ר יְהוָ֤ה צְבָאוֹת֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל הִנְנִ֨י שָׂ֥ם פָּנַ֛י בָּכֶ֖ם לְרָעָ֑ה וּלְהַכְרִ֖ית אֶת־ כָּל־ יְהוּדָֽה׃
12אֶת־ שְׁאֵרִ֣ית יְהוּדָ֗ה אֲשֶׁר־ שָׂ֨מוּ פְנֵיהֶ֜ם לָב֣וֹא אֶֽרֶץ־ מִצְרַיִם֮ לָג֣וּר שָׁם֒ וְתַ֨מּוּ כֹ֜ל בְּאֶ֧רֶץ מִצְרַ֣יִם יִפֹּ֗לוּ בַּחֶ֤רֶב בָּֽרָעָב֙ יִתַּ֔מּוּ מִקָּטֹן֙ וְעַד־ גָּד֔וֹל בַּחֶ֥רֶב וּבָרָעָ֖ב יָמֻ֑תוּ וְהָיוּ֙ לְאָלָ֣ה לְשַׁמָּ֔ה וְלִקְלָלָ֖ה וּלְחֶרְפָּֽה׃
13עַ֤ל הַיּֽוֹשְׁבִים֙ בְּאֶ֣רֶץ מִצְרַ֔יִם כַּאֲשֶׁ֥ר פָּקַ֖דְתִּי עַל־ יְרֽוּשָׁלִָ֑ם בַּחֶ֥רֶב בָּרָעָ֖ב וּבַדָּֽבֶר׃
14יִהְיֶ֜ה פָּלִ֤יט וְשָׂרִיד֙ לִשְׁאֵרִ֣ית יְהוּדָ֔ה הַבָּאִ֥ים לָגֽוּר־ שָׁ֖ם בְּאֶ֣רֶץ מִצְרָ֑יִם וְלָשׁ֣וּב ׀ אֶ֣רֶץ יְהוּדָ֗ה אֲשֶׁר־ הֵ֜מָּה מְנַשְּׂאִ֤ים אֶת־ נַפְשָׁם֙ לָשׁוּב֙ לָשֶׁ֣בֶת שָׁ֔ם כִּ֥י לֹֽא־ יָשׁ֖וּבוּ כִּ֥י אִם־ פְּלֵטִֽים׃ ס
15אֶֽת־ יִרְמְיָ֗הוּ כָּל־ הָאֲנָשִׁ֤ים הַיֹּֽדְעִים֙ כִּֽי־ מְקַטְּר֤וֹת נְשֵׁיהֶם֙ לֵאלֹהִ֣ים אֲחֵרִ֔ים וְכָל־ הַנָּשִׁ֥ים הָעֹמְד֖וֹת קָהָ֣ל גָּד֑וֹל וְכָל־ הָעָ֛ם הַיֹּשְׁבִ֥ים בְּאֶֽרֶץ־ מִצְרַ֖יִם בְּפַתְר֥וֹס לֵאמֹֽר׃
16אֲשֶׁר־ דִּבַּ֥רְתָּ אֵלֵ֖ינוּ בְּשֵׁ֣ם יְהוָ֑ה אֵינֶ֥נּוּ שֹׁמְעִ֖ים אֵלֶֽיךָ׃
17עָשֹׂ֨ה נַעֲשֶׂ֜ה אֶֽת־ כָּל־ הַדָּבָ֣ר ׀ אֲשֶׁר־ יָצָ֣א מִפִּ֗ינוּ לְקַטֵּ֞ר לִמְלֶ֣כֶת הַשָּׁמַיִם֮ וְהַסֵּֽיךְ־ לָ֣הּ נְסָכִים֒ כַּאֲשֶׁ֨ר עָשִׂ֜ינוּ אֲנַ֤חְנוּ וַאֲבֹתֵ֙ינוּ֙ מְלָכֵ֣ינוּ וְשָׂרֵ֔ינוּ בְּעָרֵ֣י יְהוּדָ֔ה וּבְחֻצ֖וֹת יְרוּשָׁלִָ֑ם וַנִּֽשְׂבַּֽע־ לֶ֙חֶם֙ וַנִּֽהְיֶ֣ה טוֹבִ֔ים וְרָעָ֖ה לֹ֥א רָאִֽינוּ׃
18אָ֡ז חָדַ֜לְנוּ לְקַטֵּ֨ר לִמְלֶ֧כֶת הַשָּׁמַ֛יִם וְהַסֵּֽךְ־ לָ֥הּ נְסָכִ֖ים חָסַ֣רְנוּ כֹ֑ל וּבַחֶ֥רֶב וּבָרָעָ֖ב תָּֽמְנוּ׃
19אֲנַ֤חְנוּ מְקַטְּרִים֙ לִמְלֶ֣כֶת הַשָּׁמַ֔יִם וּלְהַסֵּ֥ךְ לָ֖הּ נְסָכִ֑ים הֲמִֽבַּלְעֲדֵ֣י אֲנָשֵׁ֗ינוּ עָשִׂ֨ינוּ לָ֤הּ כַּוָּנִים֙ לְהַ֣עֲצִבָ֔ה וְהַסֵּ֥ךְ לָ֖הּ נְסָכִֽים׃ פ
20יִרְמְיָ֖הוּ אֶל־ כָּל־ הָעָ֑ם עַל־ הַגְּבָרִ֤ים וְעַל־ הַנָּשִׁים֙ וְעַל־ כָּל־ הָעָ֔ם הָעֹנִ֥ים אֹת֛וֹ דָּבָ֖ר לֵאמֹֽר׃
21אֶת־ הַקִּטֵּ֗ר אֲשֶׁ֨ר קִטַּרְתֶּ֜ם בְּעָרֵ֤י יְהוּדָה֙ וּבְחֻצ֣וֹת יְרוּשָׁלִַ֔ם אַתֶּ֧ם וַאֲבֽוֹתֵיכֶ֛ם מַלְכֵיכֶ֥ם וְשָׂרֵיכֶ֖ם וְעַ֣ם הָאָ֑רֶץ אֹתָם֙ זָכַ֣ר יְהוָ֔ה וַֽתַּעֲלֶ֖ה עַל־ לִבּֽוֹ׃
22יוּכַל֩ יְהוָ֨ה ע֜וֹד לָשֵׂ֗את מִפְּנֵי֙ רֹ֣עַ מַעַלְלֵיכֶ֔ם מִפְּנֵ֥י הַתּוֹעֵבֹ֖ת אֲשֶׁ֣ר עֲשִׂיתֶ֑ם וַתְּהִ֣י אַ֠רְצְכֶם לְחָרְבָּ֨ה וּלְשַׁמָּ֧ה וְלִקְלָלָ֛ה מֵאֵ֥ין יוֹשֵׁ֖ב כְּהַיּ֥וֹם הַזֶּֽה׃
23אֲשֶׁ֨ר קִטַּרְתֶּ֜ם וַאֲשֶׁ֧ר חֲטָאתֶ֣ם לַיהוָ֗ה וְלֹ֤א שְׁמַעְתֶּם֙ בְּק֣וֹל יְהוָ֔ה וּבְתֹרָת֧וֹ וּבְחֻקֹּתָ֛יו וּבְעֵדְוֹתָ֖יו לֹ֣א הֲלַכְתֶּ֑ם עַל־ כֵּ֞ן קָרָ֥את אֶתְכֶ֛ם הָרָעָ֥ה הַזֹּ֖את כַּיּ֥וֹם הַזֶּֽה׃ ס
24יִרְמְיָ֙הוּ֙ אֶל־ כָּל־ הָעָ֔ם וְאֶ֖ל כָּל־ הַנָּשִׁ֑ים שִׁמְעוּ֙ דְּבַר־ יְהוָ֔ה כָּל־ יְהוּדָ֕ה אֲשֶׁ֖ר בְּאֶ֥רֶץ מִצְרָֽיִם׃ ס
25אָמַ֣ר יְהוָֽה־ צְבָאוֹת֩ אֱלֹהֵ֨י יִשְׂרָאֵ֜ל לֵאמֹ֗ר אַתֶּ֨ם וּנְשֵׁיכֶ֜ם וַתְּדַבֵּ֣רְנָה בְּפִיכֶם֮ וּבִידֵיכֶ֣ם מִלֵּאתֶ֣ם ׀ לֵאמֹר֒ עָשֹׂ֨ה נַעֲשֶׂ֜ה אֶת־ נְדָרֵ֗ינוּ אֲשֶׁ֤ר נָדַ֙רְנוּ֙ לְקַטֵּר֙ לִמְלֶ֣כֶת הַשָּׁמַ֔יִם וּלְהַסֵּ֥ךְ לָ֖הּ נְסָכִ֑ים הָקֵ֤ים תָּקִ֙ימְנָה֙ אֶת־ נִדְרֵיכֶ֔ם וְעָשֹׂ֥ה תַעֲשֶׂ֖ינָה אֶת־ נִדְרֵיכֶֽם׃ פ
26שִׁמְע֣וּ דְבַר־ יְהוָ֔ה כָּל־ יְהוּדָ֕ה הַיֹּשְׁבִ֖ים בְּאֶ֣רֶץ מִצְרָ֑יִם הִנְנִ֨י נִשְׁבַּ֜עְתִּי בִּשְׁמִ֤י הַגָּדוֹל֙ אָמַ֣ר יְהוָ֔ה אִם־ יִהְיֶה֩ ע֨וֹד שְׁמִ֜י נִקְרָ֣א ׀ בְּפִ֣י ׀ כָּל־ אִ֣ישׁ יְהוּדָ֗ה אֹמֵ֛ר חַי־ אֲדֹנָ֥י יְהוִ֖ה בְּכָל־ אֶ֥רֶץ מִצְרָֽיִם׃
27שֹׁקֵ֧ד עֲלֵיהֶ֛ם לְרָעָ֖ה וְלֹ֣א לְטוֹבָ֑ה וְתַמּוּ֩ כָל־ אִ֨ישׁ יְהוּדָ֜ה אֲשֶׁ֧ר בְּאֶֽרֶץ־ מִצְרַ֛יִם בַּחֶ֥רֶב וּבָרָעָ֖ב עַד־ כְּלוֹתָֽם׃
