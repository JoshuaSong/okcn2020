1אָדָ֥ם שֵׁ֖ת אֱנֽוֹשׁ׃
2מַהֲלַלְאֵ֖ל יָֽרֶד׃
3מְתוּשֶׁ֖לַח לָֽמֶךְ׃
4שֵׁ֖ם חָ֥ם וָיָֽפֶת׃ ס
5יֶ֔פֶת גֹּ֣מֶר וּמָג֔וֹג וּמָדַ֖י וְיָוָ֣ן וְתֻבָ֑ל וּמֶ֖שֶׁךְ וְתִירָֽס׃ ס
6גֹּ֑מֶר אַשְׁכֲּנַ֥ז וְדִיפַ֖ת וְתוֹגַרְמָֽה׃
7יָוָ֖ן אֱלִישָׁ֣ה וְתַרְשִׁ֑ישָׁה כִּתִּ֖ים וְרוֹדָנִֽים׃ ס
8חָ֑ם כּ֥וּשׁ וּמִצְרַ֖יִם פּ֥וּט וּכְנָֽעַן׃
9כ֔וּשׁ סְבָא֙ וַחֲוִילָ֔ה וְסַבְתָּ֥א וְרַעְמָ֖א וְסַבְתְּכָ֑א וּבְנֵ֥י רַעְמָ֖א שְׁבָ֥א וּדְדָֽן׃ ס
10יָלַ֣ד אֶת־ נִמְר֑וֹד ה֣וּא הֵחֵ֔ל לִהְי֥וֹת גִּבּ֖וֹר בָּאָֽרֶץ׃ ס
11יָלַ֞ד אֶת־ [לוּדִיִּים כ] (לוּדִ֧ים ק) וְאֶת־ עֲנָמִ֛ים וְאֶת־ לְהָבִ֖ים וְאֶת־ נַפְתֻּחִֽים׃
12פַּתְרֻסִ֞ים וְאֶת־ כַּסְלֻחִ֗ים אֲשֶׁ֨ר יָצְא֥וּ מִשָּׁ֛ם פְּלִשְׁתִּ֖ים וְאֶת־ כַּפְתֹּרִֽים׃ ס
13יָלַ֛ד אֶת־ צִיד֥וֹן בְּכֹר֖וֹ וְאֶת־ חֵֽת׃
14הַיְבוּסִי֙ וְאֶת־ הָ֣אֱמֹרִ֔י וְאֵ֖ת הַגִּרְגָּשִֽׁי׃
15הַחִוִּ֥י וְאֶת־ הַֽעַרְקִ֖י וְאֶת־ הַסִּינִֽי׃
16הָאַרְוָדִ֥י וְאֶת־ הַצְּמָרִ֖י וְאֶת־ הַֽחֲמָתִֽי׃ ס
17שֵׁ֔ם עֵילָ֣ם וְאַשּׁ֔וּר וְאַרְפַּכְשַׁ֖ד וְל֣וּד וַאֲרָ֑ם וְע֥וּץ וְח֖וּל וְגֶ֥תֶר וָמֶֽשֶׁךְ׃ ס
18יָלַ֣ד אֶת־ שָׁ֑לַח וְשֶׁ֖לַח יָלַ֥ד אֶת־ עֵֽבֶר׃
19יֻלַּ֖ד שְׁנֵ֣י בָנִ֑ים שֵׁ֣ם הָאֶחָ֞ד פֶּ֗לֶג כִּ֤י בְיָמָיו֙ נִפְלְגָ֣ה הָאָ֔רֶץ וְשֵׁ֥ם אָחִ֖יו יָקְטָֽן׃
20יָלַ֔ד אֶת־ אַלְמוֹדָ֖ד וְאֶת־ שָׁ֑לֶף וְאֶת־ חֲצַרְמָ֖וֶת וְאֶת־ יָֽרַח׃
21הֲדוֹרָ֥ם וְאֶת־ אוּזָ֖ל וְאֶת־ דִּקְלָֽה׃
22עֵיבָ֥ל וְאֶת־ אֲבִימָאֵ֖ל וְאֶת־ שְׁבָֽא׃
23אוֹפִ֥יר וְאֶת־ חֲוִילָ֖ה וְאֶת־ יוֹבָ֑ב כָּל־ אֵ֖לֶּה בְּנֵ֥י יָקְטָֽן׃ ס
24אַרְפַּכְשַׁ֖ד שָֽׁלַח׃
25פֶּ֖לֶג רְעֽוּ׃
26נָח֖וֹר תָּֽרַח׃
27ה֥וּא אַבְרָהָֽם׃ ס
28אַבְרָהָ֔ם יִצְחָ֖ק וְיִשְׁמָעֵֽאל׃ ס
29תֹּלְדוֹתָ֑ם בְּכ֤וֹר יִשְׁמָעֵאל֙ נְבָי֔וֹת וְקֵדָ֥ר וְאַדְבְּאֵ֖ל וּמִבְשָֽׂם׃
30וְדוּמָ֔ה מַשָּׂ֖א חֲדַ֥ד וְתֵימָֽא׃
31נָפִ֖ישׁ וָקֵ֑דְמָה אֵ֥לֶּה הֵ֖ם בְּנֵ֥י יִשְׁמָעֵֽאל׃ ס
32קְטוּרָ֜ה פִּילֶ֣גֶשׁ אַבְרָהָ֗ם יָלְדָ֞ה אֶת־ זִמְרָ֧ן וְיָקְשָׁ֛ן וּמְדָ֥ן וּמִדְיָ֖ן וְיִשְׁבָּ֣ק וְשׁ֑וּחַ וּבְנֵ֥י יָקְשָׁ֖ן שְׁבָ֥א וּדְדָֽן׃ ס
33מִדְיָ֗ן עֵיפָ֤ה וָעֵ֙פֶר֙ וַחֲנ֔וֹךְ וַאֲבִידָ֖ע וְאֶלְדָּעָ֑ה כָּל־ אֵ֖לֶּה בְּנֵ֥י קְטוּרָֽה׃ ס
34אַבְרָהָ֖ם אֶת־ יִצְחָ֑ק ס בְּנֵ֣י יִצְחָ֔ק עֵשָׂ֖ו וְיִשְׂרָאֵֽל׃ ס
35עֵשָׂ֑ו אֱלִיפַ֛ז רְעוּאֵ֥ל וִיע֖וּשׁ וְיַעְלָ֥ם וְקֹֽרַח׃ ס
36אֱלִיפָ֑ז תֵּימָ֤ן וְאוֹמָר֙ צְפִ֣י וְגַעְתָּ֔ם קְנַ֖ז וְתִמְנָ֥ע וַעֲמָלֵֽק׃ ס
37רְעוּאֵ֑ל נַ֥חַת זֶ֖רַח שַׁמָּ֥ה וּמִזָּֽה׃ ס
38שֵׂעִ֔יר לוֹטָ֥ן וְשׁוֹבָ֖ל וְצִבְע֣וֹן וַֽעֲנָ֑ה וְדִישֹׁ֥ן וְאֵ֖צֶר וְדִישָֽׁן׃
39לוֹטָ֖ן חֹרִ֣י וְהוֹמָ֑ם וַאֲח֥וֹת לוֹטָ֖ן תִּמְנָֽע׃ ס
40שׁוֹבָ֔ל עַלְיָ֧ן וּמָנַ֛חַת וְעֵיבָ֖ל שְׁפִ֣י וְאוֹנָ֑ם ס וּבְנֵ֥י צִבְע֖וֹן אַיָּ֥ה וַעֲנָֽה׃
41עֲנָ֖ה דִּישׁ֑וֹן ס וּבְנֵ֣י דִישׁ֔וֹן חַמְרָ֥ן וְאֶשְׁבָּ֖ן וְיִתְרָ֥ן וּכְרָֽן׃ ס
42אֵ֔צֶר בִּלְהָ֥ן וְזַעֲוָ֖ן יַעֲקָ֑ן בְּנֵ֥י דִישׁ֖וֹן ע֥וּץ וַאֲרָֽן׃ פ
43הַמְּלָכִ֗ים אֲשֶׁ֤ר מָלְכוּ֙ בְּאֶ֣רֶץ אֱד֔וֹם לִפְנֵ֥י מְלָךְ־ מֶ֖לֶךְ לִבְנֵ֣י יִשְׂרָאֵ֑ל בֶּ֚לַע בֶּן־ בְּע֔וֹר וְשֵׁ֥ם עִיר֖וֹ דִּנְהָֽבָה׃
44בָּ֑לַע וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו יוֹבָ֥ב בֶּן־ זֶ֖רַח מִבָּצְרָֽה׃
45יוֹבָ֑ב וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו חוּשָׁ֖ם מֵאֶ֥רֶץ הַתֵּימָנִֽי׃
46חוּשָׁ֑ם וַיִּמְלֹ֨ךְ תַּחְתָּ֜יו הֲדַ֣ד בֶּן־ בְּדַ֗ד הַמַּכֶּ֤ה אֶת־ מִדְיָן֙ בִּשְׂדֵ֣ה מוֹאָ֔ב וְשֵׁ֥ם עִיר֖וֹ [עֲיֹות כ] (עֲוִֽית׃ ק)
47הֲדָ֑ד וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שַׂמְלָ֖ה מִמַּשְׂרֵקָֽה׃
48שַׂמְלָ֑ה וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו שָׁא֖וּל מֵרְחֹב֥וֹת הַנָּהָֽר׃
49שָׁא֑וּל וַיִּמְלֹ֣ךְ תַּחְתָּ֔יו בַּ֥עַל חָנָ֖ן בֶּן־ עַכְבּֽוֹר׃
50בַּ֣עַל חָנָ֔ן וַיִּמְלֹ֤ךְ תַּחְתָּיו֙ הֲדַ֔ד וְשֵׁ֥ם עִיר֖וֹ פָּ֑עִי וְשֵׁ֨ם אִשְׁתּ֤וֹ מְהֵיטַבְאֵל֙ בַּת־ מַטְרֵ֔ד בַּ֖ת מֵ֥י זָהָֽב׃
51הֲדָ֑ד ס וַיִּהְיוּ֙ אַלּוּפֵ֣י אֱד֔וֹם אַלּ֥וּף תִּמְנָ֛ע אַלּ֥וּף [עַלְיָה כ] (עַֽלְוָ֖ה ק) אַלּ֥וּף יְתֵֽת׃
52אָהֳלִיבָמָ֛ה אַלּ֥וּף אֵלָ֖ה אַלּ֥וּף פִּינֹֽן׃
53קְנַ֛ז אַלּ֥וּף תֵּימָ֖ן אַלּ֣וּף מִבְצָֽר׃
