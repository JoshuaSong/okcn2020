1בִּשְׁנַ֣ת שָׁל֔וֹשׁ לְמַלְכ֖וּת יְהוֹיָקִ֣ים מֶֽלֶךְ־ יְהוּדָ֑ה בָּ֣א נְבוּכַדְנֶאצַּ֧ר מֶֽלֶךְ־ בָּבֶ֛ל יְרוּשָׁלִַ֖ם וַיָּ֥צַר עָלֶֽיהָ׃
2אֲדֹנָ֨י בְּיָד֜וֹ אֶת־ יְהוֹיָקִ֣ים מֶֽלֶךְ־ יְהוּדָ֗ה וּמִקְצָת֙ כְּלֵ֣י בֵית־ הָֽאֱלֹהִ֔ים וַיְבִיאֵ֥ם אֶֽרֶץ־ שִׁנְעָ֖ר בֵּ֣ית אֱלֹהָ֑יו וְאֶת־ הַכֵּלִ֣ים הֵבִ֔יא בֵּ֖ית אוֹצַ֥ר אֱלֹהָֽיו׃
3הַמֶּ֔לֶךְ לְאַשְׁפְּנַ֖ז רַ֣ב סָרִיסָ֑יו לְהָבִ֞יא מִבְּנֵ֧י יִשְׂרָאֵ֛ל וּמִזֶּ֥רַע הַמְּלוּכָ֖ה וּמִן־ הַֽפַּרְתְּמִֽים׃
4אֲשֶׁ֣ר אֵֽין־ בָּהֶ֣ם כָּל־ [מְאוּם כ] (מוּם֩ ק) וְטוֹבֵ֨י מַרְאֶ֜ה וּמַשְׂכִּילִ֣ים בְּכָל־ חָכְמָ֗ה וְיֹ֤דְעֵי דַ֙עַת֙ וּמְבִינֵ֣י מַדָּ֔ע וַאֲשֶׁר֙ כֹּ֣חַ בָּהֶ֔ם לַעֲמֹ֖ד בְּהֵיכַ֣ל הַמֶּ֑לֶךְ וּֽלֲלַמְּדָ֥ם סֵ֖פֶר וּלְשׁ֥וֹן כַּשְׂדִּֽים׃
5לָהֶ֨ם הַמֶּ֜לֶךְ דְּבַר־ י֣וֹם בְּיוֹמ֗וֹ מִפַּת־ בַּ֤ג הַמֶּ֙לֶךְ֙ וּמִיֵּ֣ין מִשְׁתָּ֔יו וּֽלְגַדְּלָ֖ם שָׁנִ֣ים שָׁל֑וֹשׁ וּמִ֨קְצָתָ֔ם יַֽעַמְד֖וּ לִפְנֵ֥י הַמֶּֽלֶךְ׃
6בָהֶ֖ם מִבְּנֵ֣י יְהוּדָ֑ה דָּנִיֵּ֣אל חֲנַנְיָ֔ה מִֽישָׁאֵ֖ל וַעֲזַרְיָֽה׃
7לָהֶ֛ם שַׂ֥ר הַסָּרִיסִ֖ים שֵׁמ֑וֹת וַיָּ֨שֶׂם לְדָֽנִיֵּ֜אל בֵּ֣לְטְשַׁאצַּ֗ר וְלַֽחֲנַנְיָה֙ שַׁדְרַ֔ךְ וּלְמִֽישָׁאֵ֣ל מֵישַׁ֔ךְ וְלַעֲזַרְיָ֖ה עֲבֵ֥ד נְגֽוֹ׃
8דָּנִיֵּאל֙ עַל־ לִבּ֔וֹ אֲשֶׁ֧ר לֹֽא־ יִתְגָּאַ֛ל בְּפַתְבַּ֥ג הַמֶּ֖לֶךְ וּבְיֵ֣ין מִשְׁתָּ֑יו וַיְבַקֵּשׁ֙ מִשַּׂ֣ר הַסָּרִיסִ֔ים אֲשֶׁ֖ר לֹ֥א יִתְגָּאָֽל׃
9הָֽאֱלֹהִים֙ אֶת־ דָּ֣נִיֵּ֔אל לְחֶ֖סֶד וּֽלְרַחֲמִ֑ים לִפְנֵ֖י שַׂ֥ר הַסָּרִיסִֽים׃
10שַׂ֤ר הַסָּרִיסִים֙ לְדָ֣נִיֵּ֔אל יָרֵ֤א אֲנִי֙ אֶת־ אֲדֹנִ֣י הַמֶּ֔לֶךְ אֲשֶׁ֣ר מִנָּ֔ה אֶת־ מַאֲכַלְכֶ֖ם וְאֶת־ מִשְׁתֵּיכֶ֑ם אֲשֶׁ֡ר לָמָּה֩ יִרְאֶ֨ה אֶת־ פְּנֵיכֶ֜ם זֹֽעֲפִ֗ים מִן־ הַיְלָדִים֙ אֲשֶׁ֣ר כְּגִֽילְכֶ֔ם וְחִיַּבְתֶּ֥ם אֶת־ רֹאשִׁ֖י לַמֶּֽלֶךְ׃
11דָּנִיֵּ֖אל אֶל־ הַמֶּלְצַ֑ר אֲשֶׁ֤ר מִנָּה֙ שַׂ֣ר הַסָּֽרִיסִ֔ים עַל־ דָּנִיֵּ֣אל חֲנַנְיָ֔ה מִֽישָׁאֵ֖ל וַעֲזַרְיָֽה׃
12נָ֥א אֶת־ עֲבָדֶ֖יךָ יָמִ֣ים עֲשָׂרָ֑ה וְיִתְּנוּ־ לָ֜נוּ מִן־ הַזֵּרֹעִ֛ים וְנֹאכְלָ֖ה וּמַ֥יִם וְנִשְׁתֶּֽה׃
13לְפָנֶ֙יךָ֙ מַרְאֵ֔ינוּ וּמַרְאֵה֙ הַיְלָדִ֔ים הָאֹ֣כְלִ֔ים אֵ֖ת פַּתְבַּ֣ג הַמֶּ֑לֶךְ וְכַאֲשֶׁ֣ר תִּרְאֵ֔ה עֲשֵׂ֖ה עִם־ עֲבָדֶֽיךָ׃
14לָהֶ֖ם לַדָּבָ֣ר הַזֶּ֑ה וַיְנַסֵּ֖ם יָמִ֥ים עֲשָׂרָֽה׃
15יָמִ֣ים עֲשָׂרָ֔ה נִרְאָ֤ה מַרְאֵיהֶם֙ ט֔וֹב וּבְרִיאֵ֖י בָּשָׂ֑ר מִן־ כָּל־ הַיְלָדִ֔ים הָאֹ֣כְלִ֔ים אֵ֖ת פַּתְבַּ֥ג הַמֶּֽלֶךְ׃
16הַמֶּלְצַ֗ר נֹשֵׂא֙ אֶת־ פַּתְבָּגָ֔ם וְיֵ֖ין מִשְׁתֵּיהֶ֑ם וְנֹתֵ֥ן לָהֶ֖ם זֵרְעֹנִֽים׃
17הָאֵ֙לֶּה֙ אַרְבַּעְתָּ֔ם נָתַ֨ן לָהֶ֧ם הָֽאֱלֹהִ֛ים מַדָּ֥ע וְהַשְׂכֵּ֖ל בְּכָל־ סֵ֣פֶר וְחָכְמָ֑ה וְדָנִיֵּ֣אל הֵבִ֔ין בְּכָל־ חָז֖וֹן וַחֲלֹמֽוֹת׃
18הַיָּמִ֔ים אֲשֶׁר־ אָמַ֥ר הַמֶּ֖לֶךְ לַהֲבִיאָ֑ם וַיְבִיאֵם֙ שַׂ֣ר הַסָּרִיסִ֔ים לִפְנֵ֖י נְבֻכַדְנֶצַּֽר׃
19אִתָּם֮ הַמֶּלֶךְ֒ וְלֹ֤א נִמְצָא֙ מִכֻּלָּ֔ם כְּדָנִיֵּ֣אל חֲנַנְיָ֔ה מִֽישָׁאֵ֖ל וַעֲזַרְיָ֑ה וַיַּֽעַמְד֖וּ לִפְנֵ֥י הַמֶּֽלֶךְ׃
20דְּבַר֙ חָכְמַ֣ת בִּינָ֔ה אֲשֶׁר־ בִּקֵּ֥שׁ מֵהֶ֖ם הַמֶּ֑לֶךְ וַֽיִּמְצָאֵ֞ם עֶ֣שֶׂר יָד֗וֹת עַ֤ל כָּל־ הַֽחַרְטֻמִּים֙ הָֽאַשָּׁפִ֔ים אֲשֶׁ֖ר בְּכָל־ מַלְכוּתֽוֹ׃
