1נָֽקְטָ֥ה נַפְשִׁ֗י בְּחַ֫יָּ֥י אֶֽעֶזְבָ֣ה עָלַ֣י שִׂיחִ֑י אֲ֝דַבְּרָה֗ בְּמַ֣ר נַפְשִֽׁי׃
2אֶל־ אֱ֭לוֹהַּ אַל־ תַּרְשִׁיעֵ֑נִי הֽ֝וֹדִיעֵ֗נִי עַ֣ל מַה־ תְּרִיבֵֽנִי׃
3לְךָ֨ ׀ כִּֽי־ תַעֲשֹׁ֗ק כִּֽי־ תִ֭מְאַס יְגִ֣יעַ כַּפֶּ֑יךָ וְעַל־ עֲצַ֖ת רְשָׁעִ֣ים הוֹפָֽעְתָּ׃
4בָשָׂ֣ר לָ֑ךְ אִם־ כִּרְא֖וֹת אֱנ֣וֹשׁ תִּרְאֶֽה׃
5אֱנ֣וֹשׁ יָמֶ֑יךָ אִם־ שְׁ֝נוֹתֶ֗יךָ כִּ֣ימֵי גָֽבֶר׃
6תְבַקֵּ֥שׁ לַעֲוֹנִ֑י וּ֭לְחַטָּאתִ֥י תִדְרֽוֹשׁ׃
7דַּ֭עְתְּךָ כִּי־ לֹ֣א אֶרְשָׁ֑ע וְאֵ֖ין מִיָּדְךָ֣ מַצִּֽיל׃
8עִ֭צְּבוּנִי וַֽיַּעֲשׂ֑וּנִי יַ֥חַד סָ֝בִ֗יב וַֽתְּבַלְּעֵֽנִי׃
9נָ֭א כִּי־ כַחֹ֣מֶר עֲשִׂיתָ֑נִי וְֽאֶל־ עָפָ֥ר תְּשִׁיבֵֽנִי׃
10כֶ֭חָלָב תַּתִּיכֵ֑נִי וְ֝כַגְּבִנָּ֗ה תַּקְפִּיאֵֽנִי׃
11וּ֭בָשָׂר תַּלְבִּישֵׁ֑נִי וּֽבַעֲצָמ֥וֹת וְ֝גִידִ֗ים תְּסֹכְכֵֽנִי׃
12וָ֭חֶסֶד עָשִׂ֣יתָ עִמָּדִ֑י וּ֝פְקֻדָּתְךָ֗ שָֽׁמְרָ֥ה רוּחִֽי׃
13צָפַ֣נְתָּ בִלְבָבֶ֑ךָ יָ֝דַ֗עְתִּי כִּי־ זֹ֥את עִמָּֽךְ׃
14חָטָ֥אתִי וּשְׁמַרְתָּ֑נִי וּ֝מֵעֲוֹנִ֗י לֹ֣א תְנַקֵּֽנִי׃
15רָשַׁ֡עְתִּי אַלְלַ֬י לִ֗י וְ֭צָדַקְתִּי לֹא־ אֶשָּׂ֣א רֹאשִׁ֑י שְׂבַ֥ע קָ֝ל֗וֹן וּרְאֵ֥ה עָנְיִֽי׃
16כַּשַּׁ֣חַל תְּצוּדֵ֑נִי וְ֝תָשֹׁ֗ב תִּתְפַּלָּא־ בִֽי׃
17עֵדֶ֨יךָ ׀ נֶגְדִּ֗י וְתֶ֣רֶב כַּֽ֭עַשְׂךָ עִמָּדִ֑י חֲלִיפ֖וֹת וְצָבָ֣א עִמִּֽי׃
18מֵ֭רֶחֶם הֹצֵאתָ֑נִי אֶ֝גְוַ֗ע וְעַ֣יִן לֹא־ תִרְאֵֽנִי׃
19לֹא־ הָיִ֣יתִי אֶהְיֶ֑ה מִ֝בֶּ֗טֶן לַקֶּ֥בֶר אוּבָֽל׃
20מְעַ֣ט יָמַ֣י [יֶחְדָּל כ] (וַחֲדָ֑ל ק) [יָשִׁית כ] (וְשִׁ֥ית ק) מִ֝מֶּ֗נִּי וְאַבְלִ֥יגָה מְּעָֽט׃
21אֵ֭לֵךְ וְלֹ֣א אָשׁ֑וּב אֶל־ אֶ֖רֶץ חֹ֣שֶׁךְ וְצַלְמָֽוֶת׃
