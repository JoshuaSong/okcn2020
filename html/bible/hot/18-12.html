1וַיַּ֥עַן אִיּ֗וֹב וַיֹּאמַֽר׃
2כִּ֣י אַתֶּם־ עָ֑ם וְ֝עִמָּכֶ֗ם תָּמ֥וּת חָכְמָֽה׃
3לִ֤י לֵבָ֨ב ׀ כְּֽמוֹכֶ֗ם לֹא־ נֹפֵ֣ל אָנֹכִ֣י מִכֶּ֑ם וְאֶת־ מִי־ אֵ֥ין כְּמוֹ־ אֵֽלֶּה׃
4לְרֵעֵ֨הוּ ׀ אֶֽהְיֶ֗ה קֹרֵ֣א לֶ֭אֱלוֹהַּ וַֽיַּעֲנֵ֑הוּ שְׂ֝ח֗וֹק צַדִּ֥יק תָּמִֽים׃
5בּ֭וּז לְעַשְׁתּ֣וּת שַׁאֲנָ֑ן נָ֝כ֗וֹן לְמ֣וֹעֲדֵי רָֽגֶל׃
6אֹֽהָלִ֨ים ׀ לְשֹׁ֥דְדִ֗ים וּֽ֭בַטֻּחוֹת לְמַרְגִּ֣יזֵי אֵ֑ל לַאֲשֶׁ֤ר הֵבִ֖יא אֱל֣וֹהַּ בְּיָדֽוֹ׃
7שְׁאַל־ נָ֣א בְהֵמ֣וֹת וְתֹרֶ֑ךָּ וְע֥וֹף הַ֝שָּׁמַ֗יִם וְיַגֶּד־ לָֽךְ׃
8שִׂ֣יחַ לָאָ֣רֶץ וְתֹרֶ֑ךָּ וִֽיסַפְּר֥וּ לְ֝ךָ֗ דְּגֵ֣י הַיָּֽם׃
9לֹא־ יָדַ֣ע בְּכָל־ אֵ֑לֶּה כִּ֥י יַד־ יְ֝הוָה עָ֣שְׂתָה זֹּֽאת׃
10בְּ֭יָדוֹ נֶ֣פֶשׁ כָּל־ חָ֑י וְ֝ר֗וּחַ כָּל־ בְּשַׂר־ אִֽישׁ׃
11אֹ֭זֶן מִלִּ֣ין תִּבְחָ֑ן וְ֝חֵ֗ךְ אֹ֣כֶל יִטְעַם־ לֽוֹ׃
12חָכְמָ֑ה וְאֹ֖רֶךְ יָמִ֣ים תְּבוּנָֽה׃
13חָכְמָ֣ה וּגְבוּרָ֑ה ל֝֗וֹ עֵצָ֥ה וּתְבוּנָֽה׃
14יַ֭הֲרוֹס וְלֹ֣א יִבָּנֶ֑ה יִסְגֹּ֥ר עַל־ אִ֝֗ישׁ וְלֹ֣א יִפָּתֵֽחַ׃
15יַעְצֹ֣ר בַּמַּ֣יִם וְיִבָ֑שׁוּ וִֽ֝ישַׁלְּחֵ֗ם וְיַ֖הַפְכוּ אָֽרֶץ׃
16עֹ֣ז וְתֽוּשִׁיָּ֑ה ל֝֗וֹ שֹׁגֵ֥ג וּמַשְׁגֶּֽה׃
17יוֹעֲצִ֣ים שׁוֹלָ֑ל וְֽשֹׁפְטִ֥ים יְהוֹלֵֽל׃
18מְלָכִ֣ים פִּתֵּ֑חַ וַיֶּאְסֹ֥ר אֵ֝ז֗וֹר בְּמָתְנֵיהֶֽם׃
19כֹּהֲנִ֣ים שׁוֹלָ֑ל וְאֵֽתָנִ֣ים יְסַלֵּֽף׃
20שָׂ֭פָה לְנֶאֱמָנִ֑ים וְטַ֖עַם זְקֵנִ֣ים יִקָּֽח׃
21בּ֭וּז עַל־ נְדִיבִ֑ים וּמְזִ֖יחַ אֲפִיקִ֣ים רִפָּֽה׃
22עֲ֭מֻקוֹת מִנִּי־ חֹ֑שֶׁךְ וַיֹּצֵ֖א לָא֣וֹר צַלְמָֽוֶת׃
23לַ֭גּוֹיִם וַֽיְאַבְּדֵ֑ם שֹׁטֵ֥חַ לַ֝גּוֹיִ֗ם וַיַּנְחֵֽם׃
24לֵ֭ב רָאשֵׁ֣י עַם־ הָאָ֑רֶץ וַ֝יַּתְעֵ֗ם בְּתֹ֣הוּ לֹא־ דָֽרֶךְ׃
