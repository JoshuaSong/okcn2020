1וַיְדַבֵּ֤ר דָּוִד֙ לַֽיהוָ֔ה אֶת־ דִּבְרֵ֖י הַשִּׁירָ֣ה הַזֹּ֑את בְּיוֹם֩ הִצִּ֨יל יְהוָ֥ה אֹת֛וֹ מִכַּ֥ף כָּל־ אֹיְבָ֖יו וּמִכַּ֥ף שָׁאֽוּל׃
2יְהוָ֛ה סַֽלְעִ֥י וּמְצֻדָתִ֖י וּמְפַלְטִי־ לִֽי׃
3צוּרִ֖י אֶחֱסֶה־ בּ֑וֹ מָגִנִּ֞י וְקֶ֣רֶן יִשְׁעִ֗י מִשְׂגַּבִּי֙ וּמְנוּסִ֔י מֹשִׁעִ֕י מֵחָמָ֖ס תֹּשִׁעֵֽנִי׃
4אֶקְרָ֣א יְהוָ֑ה וּמֵאֹיְבַ֖י אִוָּשֵֽׁעַ׃
5אֲפָפֻ֖נִי מִשְׁבְּרֵי־ מָ֑וֶת נַחֲלֵ֥י בְלִיַּ֖עַל יְבַעֲתֻֽנִי׃
6שְׁא֖וֹל סַבֻּ֑נִי קִדְּמֻ֖נִי מֹֽקְשֵׁי־ מָֽוֶת׃
7לִי֙ אֶקְרָ֣א יְהוָ֔ה וְאֶל־ אֱלֹהַ֖י אֶקְרָ֑א וַיִּשְׁמַ֤ע מֵהֵֽיכָלוֹ֙ קוֹלִ֔י וְשַׁוְעָתִ֖י בְּאָזְנָֽיו׃
8כ] (וַיִּתְגָּעַ֤שׁ ק) וַתִּרְעַשׁ֙ הָאָ֔רֶץ מוֹסְד֥וֹת הַשָּׁמַ֖יִם יִרְגָּ֑זוּ וַיִּֽתְגָּעֲשׁ֖וּ כִּֽי־ חָ֥רָה לֽוֹ׃
9עָשָׁן֙ בְּאַפּ֔וֹ וְאֵ֥שׁ מִפִּ֖יו תֹּאכֵ֑ל גֶּחָלִ֖ים בָּעֲר֥וּ מִמֶּֽנּוּ׃
10שָׁמַ֖יִם וַיֵּרַ֑ד וַעֲרָפֶ֖ל תַּ֥חַת רַגְלָֽיו׃
11עַל־ כְּר֖וּב וַיָּעֹ֑ף וַיֵּרָ֖א עַל־ כַּנְפֵי־ רֽוּחַ׃
12חֹ֛שֶׁךְ סְבִיבֹתָ֖יו סֻכּ֑וֹת חַֽשְׁרַת־ מַ֖יִם עָבֵ֥י שְׁחָקִֽים׃
13נֶגְדּ֑וֹ בָּעֲר֖וּ גַּחֲלֵי־ אֵֽשׁ׃
14מִן־ שָׁמַ֖יִם יְהוָ֑ה וְעֶלְי֖וֹן יִתֵּ֥ן קוֹלֽוֹ׃
15חִצִּ֖ים וַיְפִיצֵ֑ם בָּרָ֖ק [וַיְּהֻמֵּם כ] (וַיָּהֹֽם׃ ק)
16אֲפִ֣קֵי יָ֔ם יִגָּל֖וּ מֹסְד֣וֹת תֵּבֵ֑ל בְּגַעֲרַ֣ת יְהוָ֔ה מִנִּשְׁמַ֖ת ר֥וּחַ אַפּֽוֹ׃
17מִמָּר֖וֹם יִקָּחֵ֑נִי יַֽמְשֵׁ֖נִי מִמַּ֥יִם רַבִּֽים׃
18מֵאֹיְבִ֖י עָ֑ז מִשֹּׂ֣נְאַ֔י כִּ֥י אָמְצ֖וּ מִמֶּֽנִּי׃
19בְּי֣וֹם אֵידִ֑י וַיְהִ֧י יְהוָ֛ה מִשְׁעָ֖ן לִֽי׃
20לַמֶּרְחָ֖ב אֹתִ֑י יְחַלְּצֵ֖נִי כִּי־ חָ֥פֵֽץ בִּֽי׃
21יְהוָ֖ה כְּצִדְקָתִ֑י כְּבֹ֥ר יָדַ֖י יָשִׁ֥יב לִֽי׃
22שָׁמַ֖רְתִּי דַּרְכֵ֣י יְהוָ֑ה וְלֹ֥א רָשַׁ֖עְתִּי מֵאֱלֹהָֽי׃
23כָל־ [מִשְׁפֹּטֹו כ] (מִשְׁפָּטָ֖יו ק) לְנֶגְדִּ֑י וְחֻקֹּתָ֖יו לֹא־ אָס֥וּר מִמֶּֽנָּה׃
24תָמִ֖ים ל֑וֹ וָאֶשְׁתַּמְּרָ֖ה מֵעֲוֹנִֽי׃
25יְהוָ֛ה לִ֖י כְּצִדְקָתִ֑י כְּבֹרִ֖י לְנֶ֥גֶד עֵינָֽיו׃
26חָסִ֖יד תִּתְחַסָּ֑ד עִם־ גִּבּ֥וֹר תָּמִ֖ים תִּתַּמָּֽם׃
27נָבָ֖ר תִּתָּבָ֑ר וְעִם־ עִקֵּ֖שׁ תִּתַּפָּֽל׃
28עַ֥ם עָנִ֖י תּוֹשִׁ֑יעַ וְעֵינֶ֖יךָ עַל־ רָמִ֥ים תַּשְׁפִּֽיל׃
29אַתָּ֥ה נֵירִ֖י יְהוָ֑ה וַיהוָ֖ה יַגִּ֥יהַּ חָשְׁכִּֽי׃
30בְכָ֖ה אָר֣וּץ גְּד֑וּד בֵּאלֹהַ֖י אֲדַלֶּג־ שֽׁוּר׃
31תָּמִ֣ים דַּרְכּ֑וֹ אִמְרַ֤ת יְהוָה֙ צְרוּפָ֔ה מָגֵ֣ן ה֔וּא לְכֹ֖ל הַחֹסִ֥ים בּֽוֹ׃
32מִי־ אֵ֖ל מִבַּלְעֲדֵ֣י יְהוָ֑ה וּמִ֥י צ֖וּר מִֽבַּלְעֲדֵ֥י אֱלֹהֵֽינוּ׃
33מָעוּזִּ֖י חָ֑יִל וַיַּתֵּ֥ר תָּמִ֖ים [דַּרְכֹּו כ] (דַּרְכִּֽי׃ ק)
34[רַגְלָיו כ] (רַגְלַ֖י ק) כָּאַיָּל֑וֹת וְעַ֥ל בָּמוֹתַ֖י יַעֲמִדֵֽנִי׃
35יָדַ֖י לַמִּלְחָמָ֑ה וְנִחַ֥ת קֶֽשֶׁת־ נְחוּשָׁ֖ה זְרֹעֹתָֽי׃
36לִ֖י מָגֵ֣ן יִשְׁעֶ֑ךָ וַעֲנֹתְךָ֖ תַּרְבֵּֽנִי׃
37צַעֲדִ֖י תַּחְתֵּ֑נִי וְלֹ֥א מָעֲד֖וּ קַרְסֻלָּֽי׃
38אֹיְבַ֖י וָאַשְׁמִידֵ֑ם וְלֹ֥א אָשׁ֖וּב עַד־ כַּלּוֹתָֽם׃
39וָאֶמְחָצֵ֖ם וְלֹ֣א יְקוּמ֑וּן וַֽיִּפְּל֖וּ תַּ֥חַת רַגְלָֽי׃
40חַ֖יִל לַמִּלְחָמָ֑ה תַּכְרִ֥יעַ קָמַ֖י תַּחְתֵּֽנִי׃
41תַּ֥תָּה לִּ֖י עֹ֑רֶף מְשַׂנְאַ֖י וָאַצְמִיתֵֽם׃
42וְאֵ֣ין מֹשִׁ֑יעַ אֶל־ יְהוָ֖ה וְלֹ֥א עָנָֽם׃
43כַּעֲפַר־ אָ֑רֶץ כְּטִיט־ חוּצ֥וֹת אֲדִקֵּ֖ם אֶרְקָעֵֽם׃
44מֵרִיבֵ֖י עַמִּ֑י תִּשְׁמְרֵ֙נִי֙ לְרֹ֣אשׁ גּוֹיִ֔ם עַ֥ם לֹא־ יָדַ֖עְתִּי יַעַבְדֻֽנִי׃
45נֵכָ֖ר יִתְכַּֽחֲשׁוּ־ לִ֑י לִשְׁמ֥וֹעַ אֹ֖זֶן יִשָּׁ֥מְעוּ לִֽי׃
46נֵכָ֖ר יִבֹּ֑לוּ וְיַחְגְּר֖וּ מִמִּסְגְּרוֹתָֽם׃
47יְהוָ֖ה וּבָר֣וּךְ צוּרִ֑י וְיָרֻ֕ם אֱלֹהֵ֖י צ֥וּר יִשְׁעִֽי׃
48הַנֹּתֵ֥ן נְקָמֹ֖ת לִ֑י וּמוֹרִ֥יד עַמִּ֖ים תַּחְתֵּֽנִי׃
49מֵאֹֽיְבָ֑י וּמִקָּמַי֙ תְּר֣וֹמְמֵ֔נִי מֵאִ֥ישׁ חֲמָסִ֖ים תַּצִּילֵֽנִי׃
50כֵּ֛ן אוֹדְךָ֥ יְהוָ֖ה בַּגּוֹיִ֑ם וּלְשִׁמְךָ֖ אֲזַמֵּֽר׃
