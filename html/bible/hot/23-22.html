1מַשָּׂ֖א גֵּ֣יא חִזָּי֑וֹן מַה־ לָּ֣ךְ אֵפ֔וֹא כִּֽי־ עָלִ֥ית כֻּלָּ֖ךְ לַגַּגּֽוֹת׃
2מְלֵאָ֗ה עִ֚יר הֽוֹמִיָּ֔ה קִרְיָ֖ה עַלִּיזָ֑ה חֲלָלַ֙יִךְ֙ לֹ֣א חַלְלֵי־ חֶ֔רֶב וְלֹ֖א מֵתֵ֥י מִלְחָמָֽה׃
3קְצִינַ֥יִךְ נָֽדְדוּ־ יַ֖חַד מִקֶּ֣שֶׁת אֻסָּ֑רוּ כָּל־ נִמְצָאַ֙יִךְ֙ אֻסְּר֣וּ יַחְדָּ֔ו מֵרָח֖וֹק בָּרָֽחוּ׃
4כֵּ֥ן אָמַ֛רְתִּי שְׁע֥וּ מִנִּ֖י אֲמָרֵ֣ר בַּבֶּ֑כִי אַל־ תָּאִ֣יצוּ לְנַֽחֲמֵ֔נִי עַל־ שֹׁ֖ד בַּת־ עַמִּֽי׃
5יוֹם֩ מְהוּמָ֨ה וּמְבוּסָ֜ה וּמְבוּכָ֗ה לַֽאדֹנָ֧י יְהוִ֛ה צְבָא֖וֹת בְּגֵ֣יא חִזָּי֑וֹן מְקַרְקַ֥ר קִ֖ר וְשׁ֥וֹעַ אֶל־ הָהָֽר׃
6נָשָׂ֣א אַשְׁפָּ֔ה בְּרֶ֥כֶב אָדָ֖ם פָּֽרָשִׁ֑ים וְקִ֥יר עֵרָ֖ה מָגֵֽן׃
7מִבְחַר־ עֲמָקַ֖יִךְ מָ֣לְאוּ רָ֑כֶב וְהַפָּ֣רָשִׁ֔ים שֹׁ֖ת שָׁ֥תוּ הַשָּֽׁעְרָה׃
8אֵ֖ת מָסַ֣ךְ יְהוּדָ֑ה וַתַּבֵּט֙ בַּיּ֣וֹם הַה֔וּא אֶל־ נֶ֖שֶׁק בֵּ֥ית הַיָּֽעַר׃
9בְּקִיעֵ֧י עִיר־ דָּוִ֛ד רְאִיתֶ֖ם כִּי־ רָ֑בּוּ וַֽתְּקַבְּצ֔וּ אֶת־ מֵ֥י הַבְּרֵכָ֖ה הַתַּחְתּוֹנָֽה׃
10בָּתֵּ֥י יְרוּשָׁלִַ֖ם סְפַרְתֶּ֑ם וַתִּתְֿצוּ֙ הַבָּ֣תִּ֔ים לְבַצֵּ֖ר הַחוֹמָֽה׃
11עֲשִׂיתֶ֗ם בֵּ֚ין הַחֹ֣מֹתַ֔יִם לְמֵ֖י הַבְּרֵכָ֣ה הַיְשָׁנָ֑ה וְלֹ֤א הִבַּטְתֶּם֙ אֶל־ עֹשֶׂ֔יהָ וְיֹצְרָ֥הּ מֵֽרָח֖וֹק לֹ֥א רְאִיתֶֽם׃
12אֲדֹנָ֧י יְהוִ֛ה צְבָא֖וֹת בַּיּ֣וֹם הַה֑וּא לִבְכִי֙ וּלְמִסְפֵּ֔ד וּלְקָרְחָ֖ה וְלַחֲגֹ֥ר שָֽׂק׃
13שָׂשׂ֣וֹן וְשִׂמְחָ֗ה הָרֹ֤ג ׀ בָּקָר֙ וְשָׁחֹ֣ט צֹ֔אן אָכֹ֥ל בָּשָׂ֖ר וְשָׁת֣וֹת יָ֑יִן אָכ֣וֹל וְשָׁת֔וֹ כִּ֥י מָחָ֖ר נָמֽוּת׃
14בְאָזְנָ֖י יְהוָ֣ה צְבָא֑וֹת אִם־ יְ֠כֻפַּר הֶעָוֹ֨ן הַזֶּ֤ה לָכֶם֙ עַד־ תְּמֻת֔וּן אָמַ֛ר אֲדֹנָ֥י יְהוִ֖ה צְבָאֽוֹת׃ פ
15אָמַ֛ר אֲדֹנָ֥י יְהוִ֖ה צְבָא֑וֹת לֶךְ־ בֹּא֙ אֶל־ הַסֹּכֵ֣ן הַזֶּ֔ה עַל־ שֶׁבְנָ֖א אֲשֶׁ֥ר עַל־ הַבָּֽיִת׃
16לְּךָ֥ פֹה֙ וּמִ֣י לְךָ֣ פֹ֔ה כִּֽי־ חָצַ֧בְתָּ לְּךָ֛ פֹּ֖ה קָ֑בֶר חֹצְבִ֤י מָרוֹם֙ קִבְר֔וֹ חֹקְקִ֥י בַסֶּ֖לַע מִשְׁכָּ֥ן לֽוֹ׃
17יְהוָה֙ מְטַלְטֶלְךָ֔ טַלְטֵלָ֖ה גָּ֑בֶר וְעֹטְךָ֖ עָטֹֽה׃
18יִצְנָפְךָ֙ צְנֵפָ֔ה כַּדּ֕וּר אֶל־ אֶ֖רֶץ רַחֲבַ֣ת יָדָ֑יִם שָׁ֣מָּה תָמ֗וּת וְשָׁ֙מָּה֙ מַרְכְּב֣וֹת כְּבוֹדֶ֔ךָ קְל֖וֹן בֵּ֥ית אֲדֹנֶֽיךָ׃
19מִמַּצָּבֶ֑ךָ וּמִמַּעֲמָֽדְךָ֖ יֶהֶרְסֶֽךָ׃
20בַּיּ֣וֹם הַה֑וּא וְקָרָ֣אתִי לְעַבְדִּ֔י לְאֶלְיָקִ֖ים בֶּן־ חִלְקִיָּֽהוּ׃
21כֻּתָּנְתֶּ֗ךָ וְאַבְנֵֽטְךָ֙ אֲחַזְּקֶ֔נּוּ וּמֶֽמְשֶׁלְתְּךָ֖ אֶתֵּ֣ן בְּיָד֑וֹ וְהָיָ֥ה לְאָ֛ב לְיוֹשֵׁ֥ב יְרוּשָׁלִַ֖ם וּלְבֵ֥ית יְהוּדָֽה׃
22מַפְתֵּ֥חַ בֵּית־ דָּוִ֖ד עַל־ שִׁכְמ֑וֹ וּפָתַח֙ וְאֵ֣ין סֹגֵ֔ר וְסָגַ֖ר וְאֵ֥ין פֹּתֵֽחַ׃
23יָתֵ֖ד בְּמָק֣וֹם נֶאֱמָ֑ן וְהָיָ֛ה לְכִסֵּ֥א כָב֖וֹד לְבֵ֥ית אָבִֽיו׃
24עָלָ֜יו כֹּ֣ל ׀ כְּב֣וֹד בֵּית־ אָבִ֗יו הַצֶּֽאֱצָאִים֙ וְהַצְּפִע֔וֹת כֹּ֖ל כְּלֵ֣י הַקָּטָ֑ן מִכְּלֵי֙ הָֽאַגָּנ֔וֹת וְעַ֖ד כָּל־ כְּלֵ֥י הַנְּבָלִֽים׃
