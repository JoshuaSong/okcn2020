1מַשָּׂ֖א צֹ֑ר הֵילִ֣ילוּ ׀ אֳנִיּ֣וֹת תַּרְשִׁ֗ישׁ כִּֽי־ שֻׁדַּ֤ד מִבַּ֙יִת֙ מִבּ֔וֹא מֵאֶ֥רֶץ כִּתִּ֖ים נִגְלָה־ לָֽמוֹ׃
2יֹ֣שְׁבֵי אִ֑י סֹחֵ֥ר צִיד֛וֹן עֹבֵ֥ר יָ֖ם מִלְאֽוּךְ׃
3רַבִּים֙ זֶ֣רַע שִׁחֹ֔ר קְצִ֥יר יְא֖וֹר תְּבֽוּאָתָ֑הּ וַתְּהִ֖י סְחַ֥ר גּוֹיִֽם׃
4צִיד֔וֹן כִּֽי־ אָמַ֣ר יָ֔ם מָע֥וֹז הַיָּ֖ם לֵאמֹ֑ר לֹֽא־ חַ֣לְתִּי וְלֹֽא־ יָלַ֗דְתִּי וְלֹ֥א גִדַּ֛לְתִּי בַּחוּרִ֖ים רוֹמַ֥מְתִּי בְתוּלֽוֹת׃
5שֵׁ֖מַע לְמִצְרָ֑יִם יָחִ֖ילוּ כְּשֵׁ֥מַע צֹֽר׃
6תַּרְשִׁ֑ישָׁה הֵילִ֖ילוּ יֹ֥שְׁבֵי אִֽי׃
7לָכֶ֖ם עַלִּיזָ֑ה מִֽימֵי־ קֶ֤דֶם קַדְמָתָהּ֙ יֹבִל֣וּהָ רַגְלֶ֔יהָ מֵֽרָח֖וֹק לָגֽוּר׃
8יָעַ֣ץ זֹ֔את עַל־ צֹ֖ר הַמַּֽעֲטִירָ֑ה אֲשֶׁ֤ר סֹחֲרֶ֙יה֙ שָׂרִ֔ים כִּנְעָנֶ֖יהָ נִכְבַּדֵּי־ אָֽרֶץ׃
9צְבָא֖וֹת יְעָצָ֑הּ לְחַלֵּל֙ גְּא֣וֹן כָּל־ צְבִ֔י לְהָקֵ֖ל כָּל־ נִכְבַּדֵּי־ אָֽרֶץ׃
10אַרְצֵ֖ךְ כַּיְאֹ֑ר בַּת־ תַּרְשִׁ֕ישׁ אֵ֖ין מֵ֥זַח עֽוֹד׃
11נָטָ֣ה עַל־ הַיָּ֔ם הִרְגִּ֖יז מַמְלָכ֑וֹת יְהוָה֙ צִוָּ֣ה אֶל־ כְּנַ֔עַן לַשְׁמִ֖ד מָעֻזְנֶֽיהָ׃
12לֹֽא־ תוֹסִ֥יפִי ע֖וֹד לַעְל֑וֹז הַֽמְעֻשָּׁקָ֞ה בְּתוּלַ֣ת בַּת־ צִיד֗וֹן [כִּתִּיִּים כ] (כִּתִּים֙ ק) ק֣וּמִי עֲבֹ֔רִי גַּם־ שָׁ֖ם לֹא־ יָנ֥וּחַֽ לָֽךְ׃
13אֶ֣רֶץ כַּשְׂדִּ֗ים זֶ֤ה הָעָם֙ לֹ֣א הָיָ֔ה אַשּׁ֖וּר יְסָדָ֣הּ לְצִיִּ֑ים הֵקִ֣ימוּ [בְחִינָיו כ] (בַחוּנָ֗יו ק) עֹרְרוּ֙ אַרְמְנוֹתֶ֔יהָ שָׂמָ֖הּ לְמַפֵּלָֽה׃
14אֳנִיּ֣וֹת תַּרְשִׁ֑ישׁ כִּ֥י שֻׁדַּ֖ד מָעֻזְּכֶֽן׃ ס
15בַּיּ֣וֹם הַה֔וּא וְנִשְׁכַּ֤חַת צֹר֙ שִׁבְעִ֣ים שָׁנָ֔ה כִּימֵ֖י מֶ֣לֶךְ אֶחָ֑ד מִקֵּ֞ץ שִׁבְעִ֤ים שָׁנָה֙ יִהְיֶ֣ה לְצֹ֔ר כְּשִׁירַ֖ת הַזּוֹנָֽה׃
16כִנּ֛וֹר סֹ֥בִּי עִ֖יר זוֹנָ֣ה נִשְׁכָּחָ֑ה הֵיטִ֤יבִי נַגֵּן֙ הַרְבִּי־ שִׁ֔יר לְמַ֖עַן תִּזָּכֵֽרִי׃
17מִקֵּ֣ץ ׀ שִׁבְעִ֣ים שָׁנָ֗ה יִפְקֹ֤ד יְהוָה֙ אֶת־ צֹ֔ר וְשָׁבָ֖ה לְאֶתְנַנָּ֑ה וְזָֽנְתָ֛ה אֶת־ כָּל־ מַמְלְכ֥וֹת הָאָ֖רֶץ עַל־ פְּנֵ֥י הָאֲדָמָֽה׃
