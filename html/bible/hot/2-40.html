1וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־ מֹשֶׁ֥ה לֵּאמֹֽר׃
2הַחֹ֥דֶשׁ הָרִאשׁ֖וֹן בְּאֶחָ֣ד לַחֹ֑דֶשׁ תָּקִ֕ים אֶת־ מִשְׁכַּ֖ן אֹ֥הֶל מוֹעֵֽד׃
3שָׁ֔ם אֵ֖ת אֲר֣וֹן הָעֵד֑וּת וְסַכֹּתָ֥ עַל־ הָאָרֹ֖ן אֶת־ הַפָּרֹֽכֶת׃
4אֶת־ הַשֻּׁלְחָ֔ן וְעָרַכְתָּ֖ אֶת־ עֶרְכּ֑וֹ וְהֵבֵאתָ֙ אֶת־ הַמְּנֹרָ֔ה וְהַעֲלֵיתָ֖ אֶת־ נֵרֹתֶֽיהָ׃
5אֶת־ מִזְבַּ֤ח הַזָּהָב֙ לִקְטֹ֔רֶת לִפְנֵ֖י אֲר֣וֹן הָעֵדֻ֑ת וְשַׂמְתָּ֛ אֶת־ מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן׃
6אֵ֖ת מִזְבַּ֣ח הָעֹלָ֑ה לִפְנֵ֕י פֶּ֖תַח מִשְׁכַּ֥ן אֹֽהֶל־ מוֹעֵֽד׃
7אֶת־ הַכִּיֹּ֔ר בֵּֽין־ אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וְנָתַתָּ֥ שָׁ֖ם מָֽיִם׃
8אֶת־ הֶחָצֵ֖ר סָבִ֑יב וְנָ֣תַתָּ֔ אֶת־ מָסַ֖ךְ שַׁ֥עַר הֶחָצֵֽר׃
9אֶת־ שֶׁ֣מֶן הַמִּשְׁחָ֔ה וּמָשַׁחְתָּ֥ אֶת־ הַמִּשְׁכָּ֖ן וְאֶת־ כָּל־ אֲשֶׁר־ בּ֑וֹ וְקִדַּשְׁתָּ֥ אֹת֛וֹ וְאֶת־ כָּל־ כֵּלָ֖יו וְהָ֥יָה קֹֽדֶשׁ׃
10אֶת־ מִזְבַּ֥ח הָעֹלָ֖ה וְאֶת־ כָּל־ כֵּלָ֑יו וְקִדַּשְׁתָּ֙ אֶת־ הַמִּזְבֵּ֔חַ וְהָיָ֥ה הַמִּזְבֵּ֖חַ קֹ֥דֶשׁ קָֽדָשִֽׁים׃
11אֶת־ הַכִּיֹּ֖ר וְאֶת־ כַּנּ֑וֹ וְקִדַּשְׁתָּ֖ אֹתֽוֹ׃
12אֶֽת־ אַהֲרֹן֙ וְאֶת־ בָּנָ֔יו אֶל־ פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְרָחַצְתָּ֥ אֹתָ֖ם בַּמָּֽיִם׃
13אֶֽת־ אַהֲרֹ֔ן אֵ֖ת בִּגְדֵ֣י הַקֹּ֑דֶשׁ וּמָשַׁחְתָּ֥ אֹת֛וֹ וְקִדַּשְׁתָּ֥ אֹת֖וֹ וְכִהֵ֥ן לִֽי׃
14בָּנָ֖יו תַּקְרִ֑יב וְהִלְבַּשְׁתָּ֥ אֹתָ֖ם כֻּתֳּנֹֽת׃
15אֹתָ֗ם כַּאֲשֶׁ֤ר מָשַׁ֙חְתָּ֙ אֶת־ אֲבִיהֶ֔ם וְכִהֲנ֖וּ לִ֑י וְ֠הָיְתָה לִהְיֹ֨ת לָהֶ֧ם מָשְׁחָתָ֛ם לִכְהֻנַּ֥ת עוֹלָ֖ם לְדֹרֹתָֽם׃
16מֹשֶׁ֑ה כְּ֠כֹל אֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה אֹת֖וֹ כֵּ֥ן עָשָֽׂה׃ ס
17בַּחֹ֧דֶשׁ הָרִאשׁ֛וֹן בַּשָּׁנָ֥ה הַשֵּׁנִ֖ית בְּאֶחָ֣ד לַחֹ֑דֶשׁ הוּקַ֖ם הַמִּשְׁכָּֽן׃
18מֹשֶׁ֜ה אֶת־ הַמִּשְׁכָּ֗ן וַיִּתֵּן֙ אֶת־ אֲדָנָ֔יו וַיָּ֙שֶׂם֙ אֶת־ קְרָשָׁ֔יו וַיִּתֵּ֖ן אֶת־ בְּרִיחָ֑יו וַיָּ֖קֶם אֶת־ עַמּוּדָֽיו׃
19אֶת־ הָאֹ֙הֶל֙ עַל־ הַמִּשְׁכָּ֔ן וַיָּ֜שֶׂם אֶת־ מִכְסֵ֥ה הָאֹ֛הֶל עָלָ֖יו מִלְמָ֑עְלָה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־ מֹשֶֽׁה׃ ס
20וַיִּתֵּ֤ן אֶת־ הָעֵדֻת֙ אֶל־ הָ֣אָרֹ֔ן וַיָּ֥שֶׂם אֶת־ הַבַּדִּ֖ים עַל־ הָאָרֹ֑ן וַיִּתֵּ֧ן אֶת־ הַכַּפֹּ֛רֶת עַל־ הָאָרֹ֖ן מִלְמָֽעְלָה׃
21אֶת־ הָאָרֹן֮ אֶל־ הַמִּשְׁכָּן֒ וַיָּ֗שֶׂם אֵ֚ת פָּרֹ֣כֶת הַמָּסָ֔ךְ וַיָּ֕סֶךְ עַ֖ל אֲר֣וֹן הָעֵד֑וּת כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־ מֹשֶֽׁה׃ ס
22אֶת־ הַשֻּׁלְחָן֙ בְּאֹ֣הֶל מוֹעֵ֔ד עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן צָפֹ֑נָה מִח֖וּץ לַפָּרֹֽכֶת׃
23עָלָ֛יו עֵ֥רֶךְ לֶ֖חֶם לִפְנֵ֣י יְהוָ֑ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־ מֹשֶֽׁה׃ ס
24אֶת־ הַמְּנֹרָה֙ בְּאֹ֣הֶל מוֹעֵ֔ד נֹ֖כַח הַשֻּׁלְחָ֑ן עַ֛ל יֶ֥רֶךְ הַמִּשְׁכָּ֖ן נֶֽגְבָּה׃
25הַנֵּרֹ֖ת לִפְנֵ֣י יְהוָ֑ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־ מֹשֶֽׁה׃ ס
26אֶת־ מִזְבַּ֥ח הַזָּהָ֖ב בְּאֹ֣הֶל מוֹעֵ֑ד לִפְנֵ֖י הַפָּרֹֽכֶת׃
27עָלָ֖יו קְטֹ֣רֶת סַמִּ֑ים כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־ מֹשֶֽׁה׃ פ
28אֶת־ מָסַ֥ךְ הַפֶּ֖תַח לַמִּשְׁכָּֽן׃
29מִזְבַּ֣ח הָעֹלָ֔ה שָׂ֕ם פֶּ֖תַח מִשְׁכַּ֣ן אֹֽהֶל־ מוֹעֵ֑ד וַיַּ֣עַל עָלָ֗יו אֶת־ הָעֹלָה֙ וְאֶת־ הַמִּנְחָ֔ה כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־ מֹשֶֽׁה׃ ס
30אֶת־ הַכִּיֹּ֔ר בֵּֽין־ אֹ֥הֶל מוֹעֵ֖ד וּבֵ֣ין הַמִּזְבֵּ֑חַ וַיִּתֵּ֥ן שָׁ֛מָּה מַ֖יִם לְרָחְצָֽה׃
31מִמֶּ֔נּוּ מֹשֶׁ֖ה וְאַהֲרֹ֣ן וּבָנָ֑יו אֶת־ יְדֵיהֶ֖ם וְאֶת־ רַגְלֵיהֶֽם׃
32אֶל־ אֹ֣הֶל מוֹעֵ֗ד וּבְקָרְבָתָ֛ם אֶל־ הַמִּזְבֵּ֖חַ יִרְחָ֑צוּ כַּאֲשֶׁ֛ר צִוָּ֥ה יְהוָ֖ה אֶת־ מֹשֶֽׁה׃ ס
33אֶת־ הֶחָצֵ֗ר סָבִיב֙ לַמִּשְׁכָּ֣ן וְלַמִּזְבֵּ֔חַ וַיִּתֵּ֕ן אֶת־ מָסַ֖ךְ שַׁ֣עַר הֶחָצֵ֑ר וַיְכַ֥ל מֹשֶׁ֖ה אֶת־ הַמְּלָאכָֽה׃ פ
34הֶעָנָ֖ן אֶת־ אֹ֣הֶל מוֹעֵ֑ד וּכְב֣וֹד יְהוָ֔ה מָלֵ֖א אֶת־ הַמִּשְׁכָּֽן׃
35יָכֹ֣ל מֹשֶׁ֗ה לָבוֹא֙ אֶל־ אֹ֣הֶל מוֹעֵ֔ד כִּֽי־ שָׁכַ֥ן עָלָ֖יו הֶעָנָ֑ן וּכְב֣וֹד יְהוָ֔ה מָלֵ֖א אֶת־ הַמִּשְׁכָּֽן׃
36הֶֽעָנָן֙ מֵעַ֣ל הַמִּשְׁכָּ֔ן יִסְע֖וּ בְּנֵ֣י יִשְׂרָאֵ֑ל בְּכֹ֖ל מַסְעֵיהֶֽם׃
37לֹ֥א יֵעָלֶ֖ה הֶעָנָ֑ן וְלֹ֣א יִסְע֔וּ עַד־ י֖וֹם הֵעָלֹתֽוֹ׃
