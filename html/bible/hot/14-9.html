1וּמַֽלְכַּת־ שְׁבָ֗א שָֽׁמְעָה֮ אֶת־ שֵׁ֣מַע שְׁלֹמֹה֒ וַתָּב֣וֹא לְנַסּוֹת֩ אֶת־ שְׁלֹמֹ֨ה בְחִיד֜וֹת בִּירֽוּשָׁלִַ֗ם בְּחַ֣יִל כָּבֵ֣ד מְאֹ֡ד וּ֠גְמַלִּים נֹשְׂאִ֨ים בְּשָׂמִ֧ים וְזָהָ֛ב לָרֹ֖ב וְאֶ֣בֶן יְקָרָ֑ה וַתָּבוֹא֙ אֶל־ שְׁלֹמֹ֔ה וַתְּדַבֵּ֣ר עִמּ֔וֹ אֵ֛ת כָּל־ אֲשֶׁ֥ר הָיָ֖ה עִם־ לְבָבָֽהּ׃
2לָ֥הּ שְׁלֹמֹ֖ה אֶת־ כָּל־ דְּבָרֶ֑יהָ וְלֹֽא־ נֶעְלַ֤ם דָּבָר֙ מִשְּׁלֹמֹ֔ה אֲשֶׁ֧ר לֹ֦א הִגִּ֖יד לָֽהּ׃
3מַֽלְכַּת־ שְׁבָ֔א אֵ֖ת חָכְמַ֣ת שְׁלֹמֹ֑ה וְהַבַּ֖יִת אֲשֶׁ֥ר בָּנָֽה׃
4שֻׁלְחָנ֡וֹ וּמוֹשַׁ֣ב עֲבָדָיו֩ וּמַעֲמַ֨ד מְשָׁרְתָ֜יו וּמַלְבּֽוּשֵׁיהֶ֗ם וּמַשְׁקָיו֙ וּמַלְבּ֣וּשֵׁיהֶ֔ם וַעֲלִיָּת֔וֹ אֲשֶׁ֥ר יַעֲלֶ֖ה בֵּ֣ית יְהוָ֑ה וְלֹא־ הָ֥יָה ע֛וֹד בָּ֖הּ רֽוּחַ׃
5אֶל־ הַמֶּ֔לֶךְ אֱמֶת֙ הַדָּבָ֔ר אֲשֶׁ֥ר שָׁמַ֖עְתִּי בְּאַרְצִ֑י עַל־ דְּבָרֶ֖יךָ וְעַל־ חָכְמָתֶֽךָ׃
6הֶאֱמַ֣נְתִּי לְדִבְרֵיהֶ֗ם עַ֤ד אֲשֶׁר־ בָּ֙אתִי֙ וַתִּרְאֶ֣ינָה עֵינַ֔י וְהִנֵּה֙ לֹ֣א הֻגַּד־ לִ֔י חֲצִ֖י מַרְבִּ֣ית חָכְמָתֶ֑ךָ יָסַ֕פְתָּ עַל־ הַשְּׁמוּעָ֖ה אֲשֶׁ֥ר שָׁמָֽעְתִּי׃
7אֲנָשֶׁ֔יךָ וְאַשְׁרֵ֖י עֲבָדֶ֣יךָ אֵ֑לֶּה הָעֹמְדִ֤ים לְפָנֶ֙יךָ֙ תָּמִ֔יד וְשֹׁמְעִ֖ים אֶת־ חָכְמָתֶֽךָ׃
8יְהוָ֤ה אֱלֹהֶ֙יךָ֙ בָּר֔וּךְ אֲשֶׁ֣ר ׀ חָפֵ֣ץ בְּךָ֗ לְתִתְּךָ֤ עַל־ כִּסְאוֹ֙ לְמֶ֔לֶךְ לַיהוָ֖ה אֱלֹהֶ֑יךָ בְּאַהֲבַ֨ת אֱלֹהֶ֤יךָ אֶת־ יִשְׂרָאֵל֙ לְהַעֲמִיד֣וֹ לְעוֹלָ֔ם וַיִּתֶּנְךָ֤ עֲלֵיהֶם֙ לְמֶ֔לֶךְ לַעֲשׂ֖וֹת מִשְׁפָּ֥ט וּצְדָקָֽה׃
9לַמֶּ֜לֶךְ מֵאָ֥ה וְעֶשְׂרִ֣ים ׀ כִּכַּ֣ר זָהָ֗ב וּבְשָׂמִ֛ים לָרֹ֥ב מְאֹ֖ד וְאֶ֣בֶן יְקָרָ֑ה וְלֹ֤א הָיָה֙ כַּבֹּ֣שֶׂם הַה֔וּא אֲשֶׁר־ נָתְנָ֥ה מַֽלְכַּת־ שְׁבָ֖א לַמֶּ֥לֶךְ שְׁלֹמֹֽה׃
10עַבְדֵ֤י [חִירָם כ] (חוּרָם֙ ק) וְעַבְדֵ֣י שְׁלֹמֹ֔ה אֲשֶׁר־ הֵבִ֥יאוּ זָהָ֖ב מֵאוֹפִ֑יר הֵבִ֛יאוּ עֲצֵ֥י אַלְגּוּמִּ֖ים וְאֶ֥בֶן יְקָרָֽה׃
11הַ֠מֶּלֶךְ אֶת־ עֲצֵ֨י הָֽאַלְגּוּמִּ֜ים מְסִלּ֤וֹת לְבֵית־ יְהוָה֙ וּלְבֵ֣ית הַמֶּ֔לֶךְ וְכִנֹּר֥וֹת וּנְבָלִ֖ים לַשָּׁרִ֑ים וְלֹא־ נִרְא֥וּ כָהֵ֛ם לְפָנִ֖ים בְּאֶ֥רֶץ יְהוּדָֽה׃
12שְׁלֹמֹ֜ה נָתַ֣ן לְמַֽלְכַּת־ שְׁבָ֗א אֶת־ כָּל־ חֶפְצָהּ֙ אֲשֶׁ֣ר שָׁאָ֔לָה מִלְּבַ֖ד אֲשֶׁר־ הֵבִ֣יאָה אֶל־ הַמֶּ֑לֶךְ וַֽתַּהֲפֹ֛ךְ וַתֵּ֥לֶךְ לְאַרְצָ֖הּ הִ֥יא וַעֲבָדֶֽיהָ׃ פ
13מִשְׁקַ֣ל הַזָּהָ֔ב אֲשֶׁר־ בָּ֥א לִשְׁלֹמֹ֖ה בְּשָׁנָ֣ה אֶחָ֑ת שֵׁ֥שׁ מֵא֛וֹת וְשִׁשִּׁ֥ים וָשֵׁ֖שׁ כִּכְּרֵ֥י זָהָֽב׃
14מֵאַנְשֵׁ֧י הַתָּרִ֛ים וְהַסֹּחֲרִ֖ים מְבִיאִ֑ים וְכָל־ מַלְכֵ֤י עֲרַב֙ וּפַח֣וֹת הָאָ֔רֶץ מְבִיאִ֛ים זָהָ֥ב וָכֶ֖סֶף לִשְׁלֹמֹֽה׃
15הַמֶּ֧לֶךְ שְׁלֹמֹ֛ה מָאתַ֥יִם צִנָּ֖ה זָהָ֣ב שָׁח֑וּט שֵׁ֤שׁ מֵאוֹת֙ זָהָ֣ב שָׁח֔וּט יַעֲלֶ֖ה עַל־ הַצִּנָּ֥ה הָאֶחָֽת׃
16מֵא֤וֹת מָֽגִנִּים֙ זָהָ֣ב שָׁח֔וּט שְׁלֹ֤שׁ מֵאוֹת֙ זָהָ֔ב יַעֲלֶ֖ה עַל־ הַמָּגֵ֣ן הָאֶחָ֑ת וַיִּתְּנֵ֣ם הַמֶּ֔לֶךְ בְּבֵ֖ית יַ֥עַר הַלְּבָנֽוֹן׃ פ
17הַמֶּ֛לֶךְ כִּסֵּא־ שֵׁ֖ן גָּד֑וֹל וַיְצַפֵּ֖הוּ זָהָ֥ב טָהֽוֹר׃
18מַעֲל֣וֹת לַ֠כִּסֵּא וְכֶ֨בֶשׁ בַּזָּהָ֤ב לַכִּסֵּא֙ מָאֳחָזִ֔ים וְיָד֛וֹת מִזֶּ֥ה וּמִזֶּ֖ה עַל־ מְק֣וֹם הַשָּׁ֑בֶת וּשְׁנַ֣יִם אֲרָי֔וֹת עֹמְדִ֖ים אֵ֥צֶל הַיָּדֽוֹת׃
19עָשָׂ֣ר אֲרָי֗וֹת עֹמְדִ֥ים שָׁ֛ם עַל־ שֵׁ֥שׁ הַֽמַּעֲל֖וֹת מִזֶּ֣ה וּמִזֶּ֑ה לֹא־ נַעֲשָׂ֥ה כֵ֖ן לְכָל־ מַמְלָכָֽה׃
20כְּלֵ֞י מַשְׁקֵ֨ה הַמֶּ֤לֶךְ שְׁלֹמֹה֙ זָהָ֔ב וְכֹ֗ל כְּלֵ֛י בֵּֽית־ יַ֥עַר הַלְּבָנ֖וֹן זָהָ֣ב סָג֑וּר אֵ֣ין כֶּ֗סֶף נֶחְשָׁ֛ב בִּימֵ֥י שְׁלֹמֹ֖ה לִמְאֽוּמָה׃
21אֳנִיּ֤וֹת לַמֶּ֙לֶךְ֙ הֹלְכ֣וֹת תַּרְשִׁ֔ישׁ עִ֖ם עַבְדֵ֣י חוּרָ֑ם אַחַת֩ לְשָׁל֨וֹשׁ שָׁנִ֜ים תָּב֣וֹאנָה ׀ אֳנִיּ֣וֹת תַּרְשִׁ֗ישׁ נֹֽשְׂאוֹת֙ זָהָ֣ב וָכֶ֔סֶף שֶׁנְהַבִּ֥ים וְקוֹפִ֖ים וְתוּכִּיִּֽים׃ פ
22הַמֶּ֣לֶךְ שְׁלֹמֹ֔ה מִכֹּ֖ל מַלְכֵ֣י הָאָ֑רֶץ לְעֹ֖שֶׁר וְחָכְמָֽה׃
23מַלְכֵ֣י הָאָ֔רֶץ מְבַקְשִׁ֖ים אֶת־ פְּנֵ֣י שְׁלֹמֹ֑ה לִשְׁמֹ֙עַ֙ אֶת־ חָכְמָת֔וֹ אֲשֶׁר־ נָתַ֥ן הָאֱלֹהִ֖ים בְּלִבּֽוֹ׃
24מְבִיאִ֣ים אִ֣ישׁ מִנְחָת֡וֹ כְּלֵ֣י כֶסֶף֩ וּכְלֵ֨י זָהָ֤ב וּשְׂלָמוֹת֙ נֵ֣שֶׁק וּבְשָׂמִ֔ים סוּסִ֖ים וּפְרָדִ֑ים דְּבַר־ שָׁנָ֖ה בְּשָׁנָֽה׃ פ
25לִשְׁלֹמֹ֜ה אַרְבַּעַת֩ אֲלָפִ֨ים אֻֽרְי֤וֹת סוּסִים֙ וּמַרְכָּב֔וֹת וּשְׁנֵים־ עָשָׂ֥ר אֶ֖לֶף פָּרָשִׁ֑ים וַיַּנִּיחֵם֙ בְּעָרֵ֣י הָרֶ֔כֶב וְעִם־ הַמֶּ֖לֶךְ בִּירוּשָׁלִָֽם׃
26מוֹשֵׁ֖ל בְּכָל־ הַמְּלָכִ֑ים מִן־ הַנָּהָר֙ וְעַד־ אֶ֣רֶץ פְּלִשְׁתִּ֔ים וְעַ֖ד גְּב֥וּל מִצְרָֽיִם׃
27הַמֶּ֧לֶךְ אֶת־ הַכֶּ֛סֶף בִּירוּשָׁלִַ֖ם כָּאֲבָנִ֑ים וְאֵ֣ת הָאֲרָזִ֗ים נָתַ֛ן כַּשִּׁקְמִ֥ים אֲשֶׁר־ בַּשְּׁפֵלָ֖ה לָרֹֽב׃
28סוּסִ֧ים מִמִּצְרַ֛יִם לִשְׁלֹמֹ֖ה וּמִכָּל־ הָאֲרָצֽוֹת׃
29דִּבְרֵ֣י שְׁלֹמֹ֔ה הָרִאשֹׁנִ֖ים וְהָאֲחֲרוֹנִ֑ים הֲלֹא־ הֵ֣ם כְּתוּבִ֗ים עַל־ דִּבְרֵי֙ נָתָ֣ן הַנָּבִ֔יא וְעַל־ נְבוּאַ֞ת אֲחִיָּ֣ה הַשִּֽׁילוֹנִ֗י וּבַחֲזוֹת֙ [יֶעְדִּי כ] (יֶעְדֹּ֣ו ק) הַחֹזֶ֔ה עַל־ יָרָבְעָ֖ם בֶּן־ נְבָֽט׃
30שְׁלֹמֹ֧ה בִֽירוּשָׁלִַ֛ם עַל־ כָּל־ יִשְׂרָאֵ֖ל אַרְבָּעִ֥ים שָׁנָֽה׃
