1וַיְהִי֙ כַּאֲשֶׁר־ תַּ֣מּוּ כָל־ הַגּ֔וֹי לַעֲב֖וֹר אֶת־ הַיַּרְדֵּ֑ן פ וַיֹּ֣אמֶר יְהוָ֔ה אֶל־ יְהוֹשֻׁ֖עַ לֵאמֹֽר׃
2לָכֶם֙ מִן־ הָעָ֔ם שְׁנֵ֥ים עָשָׂ֖ר אֲנָשִׁ֑ים אִישׁ־ אֶחָ֥ד אִישׁ־ אֶחָ֖ד מִשָּֽׁבֶט׃
3אוֹתָם֮ לֵאמֹר֒ שְׂאֽוּ־ לָכֶ֨ם מִזֶּ֜ה מִתּ֣וֹךְ הַיַּרְדֵּ֗ן מִמַּצַּב֙ רַגְלֵ֣י הַכֹּהֲנִ֔ים הָכִ֖ין שְׁתֵּים־ עֶשְׂרֵ֣ה אֲבָנִ֑ים וְהַעֲבַרְתֶּ֤ם אוֹתָם֙ עִמָּכֶ֔ם וְהִנַּחְתֶּ֣ם אוֹתָ֔ם בַּמָּל֕וֹן אֲשֶׁר־ תָּלִ֥ינוּ ב֖וֹ הַלָּֽיְלָה׃ ס
4יְהוֹשֻׁ֗עַ אֶל־ שְׁנֵ֤ים הֶֽעָשָׂר֙ אִ֔ישׁ אֲשֶׁ֥ר הֵכִ֖ין מִבְּנֵ֣י יִשְׂרָאֵ֑ל אִישׁ־ אֶחָ֥ד אִישׁ־ אֶחָ֖ד מִשָּֽׁבֶט׃
5לָהֶם֙ יְהוֹשֻׁ֔עַ עִ֠בְרוּ לִפְנֵ֨י אֲר֧וֹן יְהוָ֛ה אֱלֹֽהֵיכֶ֖ם אֶל־ תּ֣וֹךְ הַיַּרְדֵּ֑ן וְהָרִ֨ימוּ לָכֶ֜ם אִ֣ישׁ אֶ֤בֶן אַחַת֙ עַל־ שִׁכְמ֔וֹ לְמִסְפַּ֖ר שִׁבְטֵ֥י בְנֵי־ יִשְׂרָאֵֽל׃
6תִּֽהְיֶ֛ה זֹ֥את א֖וֹת בְּקִרְבְּכֶ֑ם כִּֽי־ יִשְׁאָל֨וּן בְּנֵיכֶ֤ם מָחָר֙ לֵאמֹ֔ר מָ֛ה הָאֲבָנִ֥ים הָאֵ֖לֶּה לָכֶֽם׃
7לָהֶ֗ם אֲשֶׁ֨ר נִכְרְת֜וּ מֵימֵ֤י הַיַּרְדֵּן֙ מִפְּנֵי֙ אֲר֣וֹן בְּרִית־ יְהוָ֔ה בְּעָבְרוֹ֙ בַּיַּרְדֵּ֔ן נִכְרְת֖וּ מֵ֣י הַיַּרְדֵּ֑ן וְ֠הָיוּ הָאֲבָנִ֨ים הָאֵ֧לֶּה לְזִכָּר֛וֹן לִבְנֵ֥י יִשְׂרָאֵ֖ל עַד־ עוֹלָֽם׃
8כֵ֣ן בְּנֵי־ יִשְׂרָאֵל֮ כַּאֲשֶׁ֣ר צִוָּ֣ה יְהוֹשֻׁעַ֒ וַיִּשְׂא֡וּ שְׁתֵּֽי־ עֶשְׂרֵ֨ה אֲבָנִ֜ים מִתּ֣וֹךְ הַיַּרְדֵּ֗ן כַּאֲשֶׁ֨ר דִּבֶּ֤ר יְהוָה֙ אֶל־ יְהוֹשֻׁ֔עַ לְמִסְפַּ֖ר שִׁבְטֵ֣י בְנֵֽי־ יִשְׂרָאֵ֑ל וַיַּעֲבִר֤וּם עִמָּם֙ אֶל־ הַמָּל֔וֹן וַיַּנִּח֖וּם שָֽׁם׃
9עֶשְׂרֵ֣ה אֲבָנִ֗ים הֵקִ֣ים יְהוֹשֻׁעַ֮ בְּת֣וֹךְ הַיַּרְדֵּן֒ תַּ֗חַת מַצַּב֙ רַגְלֵ֣י הַכֹּהֲנִ֔ים נֹשְׂאֵ֖י אֲר֣וֹן הַבְּרִ֑ית וַיִּ֣הְיוּ שָׁ֔ם עַ֖ד הַיּ֥וֹם הַזֶּֽה׃
10נֹשְׂאֵ֣י הָאָר֗וֹן עֹמְדִים֮ בְּת֣וֹךְ הַיַּרְדֵּן֒ עַ֣ד תֹּ֣ם כָּֽל־ הַ֠דָּבָר אֲשֶׁר־ צִוָּ֨ה יְהוָ֤ה אֶת־ יְהוֹשֻׁ֙עַ֙ לְדַבֵּ֣ר אֶל־ הָעָ֔ם כְּכֹ֛ל אֲשֶׁר־ צִוָּ֥ה מֹשֶׁ֖ה אֶת־ יְהוֹשֻׁ֑עַ וַיְמַהֲר֥וּ הָעָ֖ם וַֽיַּעֲבֹֽרוּ׃
11כַּֽאֲשֶׁר־ תַּ֥ם כָּל־ הָעָ֖ם לַֽעֲב֑וֹר וַיַּעֲבֹ֧ר אֲרוֹן־ יְהוָ֛ה וְהַכֹּהֲנִ֖ים לִפְנֵ֥י הָעָֽם׃
12בְּנֵי־ רְאוּבֵ֨ן וּבְנֵי־ גָ֜ד וַחֲצִ֨י שֵׁ֤בֶט הַֽמְנַשֶּׁה֙ חֲמֻשִׁ֔ים לִפְנֵ֖י בְּנֵ֣י יִשְׂרָאֵ֑ל כַּאֲשֶׁ֛ר דִּבֶּ֥ר אֲלֵיהֶ֖ם מֹשֶֽׁה׃
13אֶ֖לֶף חֲלוּצֵ֣י הַצָּבָ֑א עָבְר֞וּ לִפְנֵ֤י יְהוָה֙ לַמִּלְחָמָ֔ה אֶ֖ל עַֽרְב֥וֹת יְרִיחֽוֹ׃ ס
14הַה֗וּא גִּדַּ֤ל יְהוָה֙ אֶת־ יְהוֹשֻׁ֔עַ בְּעֵינֵ֖י כָּל־ יִשְׂרָאֵ֑ל וַיִּֽרְא֣וּ אֹת֔וֹ כַּאֲשֶׁ֛ר יָרְא֥וּ אֶת־ מֹשֶׁ֖ה כָּל־ יְמֵ֥י חַיָּֽיו׃ פ
15יְהוָ֔ה אֶל־ יְהוֹשֻׁ֖עַ לֵאמֹֽר׃
16אֶת־ הַכֹּ֣הֲנִ֔ים נֹשְׂאֵ֖י אֲר֣וֹן הָעֵד֑וּת וְיַעֲל֖וּ מִן־ הַיַּרְדֵּֽן׃
17יְהוֹשֻׁ֔עַ אֶת־ הַכֹּהֲנִ֖ים לֵאמֹ֑ר עֲל֖וּ מִן־ הַיַּרְדֵּֽן׃
18[בַּעֲלֹות כ] (כַּעֲלֹ֨ות ק) הַכֹּהֲנִ֜ים נֹשְׂאֵ֨י אֲר֤וֹן בְּרִית־ יְהוָה֙ מִתּ֣וֹךְ הַיַּרְדֵּ֔ן נִתְּק֗וּ כַּפּוֹת֙ רַגְלֵ֣י הַכֹּהֲנִ֔ים אֶ֖ל הֶחָרָבָ֑ה וַיָּשֻׁ֤בוּ מֵֽי־ הַיַּרְדֵּן֙ לִמְקוֹמָ֔ם וַיֵּלְכ֥וּ כִתְמוֹל־ שִׁלְשׁ֖וֹם עַל־ כָּל־ גְּדוֹתָֽיו׃
19עָלוּ֙ מִן־ הַיַּרְדֵּ֔ן בֶּעָשׂ֖וֹר לַחֹ֣דֶשׁ הָרִאשׁ֑וֹן וַֽיַּחֲנוּ֙ בַּגִּלְגָּ֔ל בִּקְצֵ֖ה מִזְרַ֥ח יְרִיחֽוֹ׃
20שְׁתֵּ֨ים עֶשְׂרֵ֤ה הָֽאֲבָנִים֙ הָאֵ֔לֶּה אֲשֶׁ֥ר לָקְח֖וּ מִן־ הַיַּרְדֵּ֑ן הֵקִ֥ים יְהוֹשֻׁ֖עַ בַּגִּלְגָּֽל׃
21אֶל־ בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר אֲשֶׁר֩ יִשְׁאָל֨וּן בְּנֵיכֶ֤ם מָחָר֙ אֶת־ אֲבוֹתָ֣ם לֵאמֹ֔ר מָ֖ה הָאֲבָנִ֥ים הָאֵֽלֶּה׃
22אֶת־ בְּנֵיכֶ֣ם לֵאמֹ֑ר בַּיַּבָּשָׁה֙ עָבַ֣ר יִשְׂרָאֵ֔ל אֶת־ הַיַּרְדֵּ֖ן הַזֶּֽה׃
23הוֹבִישׁ֩ יְהוָ֨ה אֱלֹהֵיכֶ֜ם אֶת־ מֵ֧י הַיַּרְדֵּ֛ן מִפְּנֵיכֶ֖ם עַֽד־ עָבְרְכֶ֑ם כַּאֲשֶׁ֣ר עָשָׂה֩ יְהוָ֨ה אֱלֹהֵיכֶ֧ם לְיַם־ ס֛וּף אֲשֶׁר־ הוֹבִ֥ישׁ מִפָּנֵ֖ינוּ עַד־ עָבְרֵֽנוּ׃
