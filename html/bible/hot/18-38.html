1וַיַּֽעַן־ יְהוָ֣ה אֶת־ אִ֭יּוֹב [מִן כ] [הַסְּעָרָה כ] (מִ֥ן ק) (הַסְּעָרָ֗ה ק) וַיֹּאמַֽר׃
2זֶ֨ה ׀ מַחְשִׁ֖יךְ עֵצָ֥ה בְמִלִּ֗ין בְּֽלִי־ דָֽעַת׃
3נָ֣א כְגֶ֣בֶר חֲלָצֶ֑יךָ וְ֝אֶשְׁאָלְךָ֗ וְהוֹדִיעֵֽנִי׃
4הָ֭יִיתָ בְּיָסְדִי־ אָ֑רֶץ הַ֝גֵּ֗ד אִם־ יָדַ֥עְתָּ בִינָֽה׃
5שָׂ֣ם מְ֭מַדֶּיהָ כִּ֣י תֵדָ֑ע א֤וֹ מִֽי־ נָטָ֖ה עָלֶ֣יהָ קָּֽו׃
6מָ֭ה אֲדָנֶ֣יהָ הָטְבָּ֑עוּ א֥וֹ מִֽי־ יָ֝רָ֗ה אֶ֣בֶן פִּנָּתָֽהּ׃
7יַ֭חַד כּ֣וֹכְבֵי בֹ֑קֶר וַ֝יָּרִ֗יעוּ כָּל־ בְּנֵ֥י אֱלֹהִֽים׃
8בִּדְלָתַ֣יִם יָ֑ם בְּ֝גִיח֗וֹ מֵרֶ֥חֶם יֵצֵֽא׃
9עָנָ֣ן לְבֻשׁ֑וֹ וַ֝עֲרָפֶ֗ל חֲתֻלָּתֽוֹ׃
10עָלָ֣יו חֻקִּ֑י וָֽ֝אָשִׂ֗ים בְּרִ֣יחַ וּדְלָתָֽיִם׃
11עַד־ פֹּ֣ה תָ֭בוֹא וְלֹ֣א תֹסִ֑יף וּפֹ֥א־ יָ֝שִׁ֗ית בִּגְא֥וֹן גַּלֶּֽיךָ׃
12צִוִּ֣יתָ בֹּ֑קֶר [יִדַּעְתָּה כ] [שַׁחַר כ] (יִדַּ֖עְתָּ ק) (הַשַּׁ֣חַר ק) מְקֹמֽוֹ׃
13בְּכַנְפ֣וֹת הָאָ֑רֶץ וְיִנָּעֲר֖וּ רְשָׁ עִ֣ seg>ים מִמֶּֽנָּה׃
14כְּחֹ֣מֶר חוֹתָ֑ם וְ֝יִֽתְיַצְּב֗וּ כְּמ֣וֹ לְבֽוּשׁ׃
15מֵרְשָׁ עִ֣ seg>ים אוֹרָ֑ם וּזְר֥וֹעַ רָ֝מָ֗ה תִּשָּׁבֵֽר׃
16עַד־ נִבְכֵי־ יָ֑ם וּבְחֵ֥קֶר תְּ֝ה֗וֹם הִתְהַלָּֽכְתָּ׃
17לְ֭ךָ שַׁעֲרֵי־ מָ֑וֶת וְשַׁעֲרֵ֖י צַלְמָ֣וֶת תִּרְאֶֽה׃
18עַד־ רַחֲבֵי־ אָ֑רֶץ הַ֝גֵּ֗ד אִם־ יָדַ֥עְתָּ כֻלָּֽהּ׃
19זֶ֣ה הַ֭דֶּרֶךְ יִשְׁכָּן־ א֑וֹר וְ֝חֹ֗שֶׁךְ אֵי־ זֶ֥ה מְקֹמֽוֹ׃
20תִ֭קָּחֶנּוּ אֶל־ גְּבוּל֑וֹ וְכִֽי־ תָ֝בִ֗ין נְתִיב֥וֹת בֵּיתֽוֹ׃
21כִּי־ אָ֣ז תִּוָּלֵ֑ד וּמִסְפַּ֖ר יָמֶ֣יךָ רַבִּֽים׃
22אֶל־ אֹצְר֣וֹת שָׁ֑לֶג וְאֹצְר֖וֹת בָּרָ֣ד תִּרְאֶֽה׃
23חָשַׂ֥כְתִּי לְעֶת־ צָ֑ר לְי֥וֹם קְ֝רָ֗ב וּמִלְחָמָֽה׃
24זֶ֣ה הַ֭דֶּרֶךְ יֵחָ֣לֶק א֑וֹר יָפֵ֖ץ קָדִ֣ים עֲלֵי־ אָֽרֶץ׃
25פִלַּ֣ג לַשֶּׁ֣טֶף תְּעָלָ֑ה וְ֝דֶ֗רֶךְ לַחֲזִ֥יז קֹלֽוֹת׃
26עַל־ אֶ֣רֶץ לֹא־ אִ֑ישׁ מִ֝דְבָּ֗ר לֹא־ אָדָ֥ם בּֽוֹ׃
27שֹׁ֭אָה וּמְשֹׁאָ֑ה וּ֝לְהַצְמִ֗יחַ מֹ֣צָא דֶֽשֶׁא׃
28לַמָּטָ֥ר אָ֑ב א֥וֹ מִי־ ה֝וֹלִ֗יד אֶגְלֵי־ טָֽל׃
29מִ֭י יָצָ֣א הַקָּ֑רַח וּכְפֹ֥ר שָׁ֝מַיִם מִ֣י יְלָדֽוֹ׃
30מַ֣יִם יִתְחַבָּ֑אוּ וּפְנֵ֥י תְ֝ה֗וֹם יִתְלַכָּֽדוּ׃
31מַעֲדַנּ֣וֹת כִּימָ֑ה אֽוֹ־ מֹשְׁכ֖וֹת כְּסִ֣יל תְּפַתֵּֽחַ׃
32מַזָּר֣וֹת בְּעִתּ֑וֹ וְ֝עַ֗יִשׁ עַל־ בָּנֶ֥יהָ תַנְחֵֽם׃
33חֻקּ֣וֹת שָׁמָ֑יִם אִם־ תָּשִׂ֖ים מִשְׁטָר֣וֹ בָאָֽרֶץ׃
34לָעָ֣ב קוֹלֶ֑ךָ וְֽשִׁפְעַת־ מַ֥יִם תְּכַסֶּֽךָּ׃
35בְּרָקִ֣ים וְיֵלֵ֑כוּ וְיֹאמְר֖וּ לְךָ֣ הִנֵּֽנוּ׃
36שָׁ֭ת בַּטֻּח֣וֹת חָכְמָ֑ה א֤וֹ מִֽי־ נָתַ֖ן לַשֶּׂ֣כְוִי בִינָֽה׃
37יְסַפֵּ֣ר שְׁחָקִ֣ים בְּחָכְמָ֑ה וְנִבְלֵ֥י שָׁ֝מַ֗יִם מִ֣י יַשְׁכִּֽיב׃
38עָ֭פָר לַמּוּצָ֑ק וּרְגָבִ֥ים יְדֻבָּֽקוּ׃
39לְלָבִ֣יא טָ֑רֶף וְחַיַּ֖ת כְּפִירִ֣ים תְּמַלֵּֽא׃
40יָשֹׁ֥חוּ בַמְּעוֹנ֑וֹת יֵשְׁב֖וּ בַסֻּכָּ֣ה לְמוֹ־ אָֽרֶב׃
