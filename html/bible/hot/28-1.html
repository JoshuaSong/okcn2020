1דְּבַר־ יְהוָ֣ה ׀ אֲשֶׁ֣ר הָיָ֗ה אֶל־ הוֹשֵׁ֙עַ֙ בֶּן־ בְּאֵרִ֔י בִּימֵ֨י עֻזִּיָּ֥ה יוֹתָ֛ם אָחָ֥ז יְחִזְקִיָּ֖ה מַלְכֵ֣י יְהוּדָ֑ה וּבִימֵ֛י יָרָבְעָ֥ם בֶּן־ יוֹאָ֖שׁ מֶ֥לֶךְ יִשְׂרָאֵֽל׃
2דִּבֶּר־ יְהוָ֖ה בְּהוֹשֵׁ֑עַ פ וַיֹּ֨אמֶר יְהוָ֜ה אֶל־ הוֹשֵׁ֗עַ לֵ֣ךְ קַח־ לְךָ֞ אֵ֤שֶׁת זְנוּנִים֙ וְיַלְדֵ֣י זְנוּנִ֔ים כִּֽי־ זָנֹ֤ה תִזְנֶה֙ הָאָ֔רֶץ מֵֽאַחֲרֵ֖י יְהוָֽה׃
3וַיִּקַּ֔ח אֶת־ גֹּ֖מֶר בַּת־ דִּבְלָ֑יִם וַתַּ֥הַר וַתֵּֽלֶד־ ל֖וֹ בֵּֽן׃
4יְהוָה֙ אֵלָ֔יו קְרָ֥א שְׁמ֖וֹ יִזְרְעֶ֑אל כִּי־ ע֣וֹד מְעַ֗ט וּפָ֨קַדְתִּ֜י אֶת־ דְּמֵ֤י יִזְרְעֶאל֙ עַל־ בֵּ֣ית יֵה֔וּא וְהִ֨שְׁבַּתִּ֔י מַמְלְכ֖וּת בֵּ֥ית יִשְׂרָאֵֽל׃
5בַּיּ֣וֹם הַה֑וּא וְשָֽׁבַרְתִּי֙ אֶת־ קֶ֣שֶׁת יִשְׂרָאֵ֔ל בְּעֵ֖מֶק יִזְרְעֶֽאל׃
6עוֹד֙ וַתֵּ֣לֶד בַּ֔ת וַיֹּ֣אמֶר ל֔וֹ קְרָ֥א שְׁמָ֖הּ לֹ֣א רֻחָ֑מָה כִּי֩ לֹ֨א אוֹסִ֜יף ע֗וֹד אֲרַחֵם֙ אֶת־ בֵּ֣ית יִשְׂרָאֵ֔ל כִּֽי־ נָשֹׂ֥א אֶשָּׂ֖א לָהֶֽם׃
7בֵּ֤ית יְהוּדָה֙ אֲרַחֵ֔ם וְהֽוֹשַׁעְתִּ֖ים בַּיהוָ֣ה אֱלֹֽהֵיהֶ֑ם וְלֹ֣א אֽוֹשִׁיעֵ֗ם בְּקֶ֤שֶׁת וּבְחֶ֙רֶב֙ וּבְמִלְחָמָ֔ה בְּסוּסִ֖ים וּבְפָרָשִֽׁים׃
8אֶת־ לֹ֣א רֻחָ֑מָה וַתַּ֖הַר וַתֵּ֥לֶד בֵּֽן׃
9קְרָ֥א שְׁמ֖וֹ לֹ֣א עַמִּ֑י כִּ֤י אַתֶּם֙ לֹ֣א עַמִּ֔י וְאָנֹכִ֖י לֹֽא־ אֶהְיֶ֥ה לָכֶֽם׃ ס
10מִסְפַּ֤ר בְּנֵֽי־ יִשְׂרָאֵל֙ כְּח֣וֹל הַיָּ֔ם אֲשֶׁ֥ר לֹֽא־ יִמַּ֖ד וְלֹ֣א יִסָּפֵ֑ר וְֽ֠הָיָה בִּמְק֞וֹם אֲשֶׁר־ יֵאָמֵ֤ר לָהֶם֙ לֹֽא־ עַמִּ֣י אַתֶּ֔ם יֵאָמֵ֥ר לָהֶ֖ם בְּנֵ֥י אֵֽל־ חָֽי׃
