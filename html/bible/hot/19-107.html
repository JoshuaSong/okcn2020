1הֹד֣וּ לַיהוָ֣ה כִּי־ ט֑וֹב כִּ֖י לְעוֹלָ֣ם חַסְדּֽוֹ׃
2גְּאוּלֵ֣י יְהוָ֑ה אֲשֶׁ֥ר גְּ֝אָלָ֗ם מִיַּד־ צָֽר׃
3קִ֫בְּצָ֥ם מִמִּזְרָ֥ח וּמִֽמַּעֲרָ֑ב מִצָּפ֥וֹן וּמִיָּֽם׃
4בַ֭מִּדְבָּר בִּישִׁימ֣וֹן דָּ֑רֶךְ עִ֥יר מ֝וֹשָׁ֗ב לֹ֣א מָצָֽאוּ׃
5גַּם־ צְמֵאִ֑ים נַ֝פְשָׁ֗ם בָּהֶ֥ם תִּתְעַטָּֽף׃
6אֶל־ יְ֭הוָה בַּצַּ֣ר לָהֶ֑ם מִ֝מְּצֽוּקוֹתֵיהֶ֗ם יַצִּילֵֽם׃
7בְּדֶ֣רֶךְ יְשָׁרָ֑ה לָ֝לֶ֗כֶת אֶל־ עִ֥יר מוֹשָֽׁב׃
8לַיהוָ֣ה חַסְדּ֑וֹ וְ֝נִפְלְאוֹתָ֗יו לִבְנֵ֥י אָדָֽם׃
9הִ֭שְׂבִּיעַ נֶ֣פֶשׁ שֹׁקֵקָ֑ה וְנֶ֥פֶשׁ רְ֝עֵבָה מִלֵּא־ טֽוֹב׃
10חֹ֣שֶׁךְ וְצַלְמָ֑וֶת אֲסִירֵ֖י עֳנִ֣י וּבַרְזֶֽל׃
11הִמְר֥וּ אִמְרֵי־ אֵ֑ל וַעֲצַ֖ת עֶלְי֣וֹן נָאָֽצוּ׃
12בֶּעָמָ֣ל לִבָּ֑ם כָּ֝שְׁל֗וּ וְאֵ֣ין עֹזֵֽר׃
13אֶל־ יְ֭הוָה בַּצַּ֣ר לָהֶ֑ם מִ֝מְּצֻֽקוֹתֵיהֶ֗ם יוֹשִׁיעֵֽם׃
14מֵחֹ֣שֶׁךְ וְצַלְמָ֑וֶת וּמוֹסְר֖וֹתֵיהֶ֣ם יְנַתֵּֽק׃
15לַיהוָ֣ה חַסְדּ֑וֹ וְ֝נִפְלְאוֹתָ֗יו לִבְנֵ֥י אָדָֽם׃
16שִׁ֭בַּר דַּלְת֣וֹת נְחֹ֑שֶׁת וּבְרִיחֵ֖י בַרְזֶ֣ל גִּדֵּֽעַ׃
17מִדֶּ֣רֶךְ פִּשְׁעָ֑ם וּֽ֝מֵעֲוֹֽנֹתֵיהֶ֗ם יִתְעַנּֽוּ׃
18אֹ֭כֶל תְּתַעֵ֣ב נַפְשָׁ֑ם וַ֝יַּגִּ֗יעוּ עַד־ שַׁ֥עֲרֵי מָֽוֶת׃
19אֶל־ יְ֭הוָה בַּצַּ֣ר לָהֶ֑ם מִ֝מְּצֻֽקוֹתֵיהֶ֗ם יוֹשִׁיעֵֽם׃
20דְּ֭בָרוֹ וְיִרְפָּאֵ֑ם וִֽ֝ימַלֵּ֗ט מִשְּׁחִיתוֹתָֽם׃ ׆
21לַיהוָ֣ה חַסְדּ֑וֹ וְ֝נִפְלְאוֹתָ֗יו לִבְנֵ֥י אָדָֽם׃ ׆
22זִבְחֵ֣י תוֹדָ֑ה וִֽיסַפְּר֖וּ מַעֲשָׂ֣יו בְּרִנָּֽה׃ ׆
23הַ֭יָּם בָּאֳנִיּ֑וֹת עֹשֵׂ֥י מְ֝לָאכָ֗ה בְּמַ֣יִם רַבִּֽים׃ ׆
24רָ֭אוּ מַעֲשֵׂ֣י יְהוָ֑ה וְ֝נִפְלְאוֹתָ֗יו בִּמְצוּלָֽה׃ ׆
25וַֽ֭יַּעֲמֵד ר֣וּחַ סְעָרָ֑ה וַתְּרוֹמֵ֥ם גַּלָּֽיו׃ ׆
26שָׁ֭מַיִם יֵרְד֣וּ תְהוֹמ֑וֹת נַ֝פְשָׁ֗ם בְּרָעָ֥ה תִתְמוֹגָֽג׃
27וְ֭יָנוּעוּ כַּשִּׁכּ֑וֹר וְכָל־ חָ֝כְמָתָ֗ם תִּתְבַּלָּֽע׃
28אֶל־ יְ֭הוָה בַּצַּ֣ר לָהֶ֑ם וּֽ֝מִמְּצֽוּקֹתֵיהֶ֗ם יוֹצִיאֵֽם׃
29סְ֭עָרָה לִדְמָמָ֑ה וַ֝יֶּחֱשׁ֗וּ גַּלֵּיהֶֽם׃
30כִֽי־ יִשְׁתֹּ֑קוּ וַ֝יַּנְחֵ֗ם אֶל־ מְח֥וֹז חֶפְצָֽם׃
31לַיהוָ֣ה חַסְדּ֑וֹ וְ֝נִפְלְאוֹתָ֗יו לִבְנֵ֥י אָדָֽם׃
32בִּקְהַל־ עָ֑ם וּבְמוֹשַׁ֖ב זְקֵנִ֣ים יְהַלְלֽוּהוּ׃
33נְהָר֣וֹת לְמִדְבָּ֑ר וּמֹצָ֥אֵי מַ֝֗יִם לְצִמָּאֽוֹן׃
34פְּ֭רִי לִמְלֵחָ֑ה מֵ֝רָעַ֗ת יֹ֣שְׁבֵי בָֽהּ׃
35מִ֭דְבָּר לַֽאֲגַם־ מַ֑יִם וְאֶ֥רֶץ צִ֝יָּ֗ה לְמֹצָ֥אֵי מָֽיִם׃
36שָׁ֣ם רְעֵבִ֑ים וַ֝יְכוֹנְנ֗וּ עִ֣יר מוֹשָֽׁב׃
37שָׂ֭דוֹת וַיִּטְּע֣וּ כְרָמִ֑ים וַ֝יַּעֲשׂ֗וּ פְּרִ֣י תְבֽוּאָה׃
38וַיִּרְבּ֣וּ מְאֹ֑ד וּ֝בְהֶמְתָּ֗ם לֹ֣א יַמְעִֽיט׃
39וַיָּשֹׁ֑חוּ מֵעֹ֖צֶר רָעָ֣ה וְיָגֽוֹן׃ ׆
40בּ֭וּז עַל־ נְדִיבִ֑ים וַ֝יַּתְעֵ֗ם בְּתֹ֣הוּ לֹא־ דָֽרֶךְ׃
41אֶבְי֣וֹן מֵע֑וֹנִי וַיָּ֥שֶׂם כַּ֝צֹּ֗אן מִשְׁפָּחֽוֹת׃
42יְשָׁרִ֣ים וְיִשְׂמָ֑חוּ וְכָל־ עַ֝וְלָ֗ה קָ֣פְצָה פִּֽיהָ׃
