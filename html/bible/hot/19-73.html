1מִזְמ֗וֹר לְאָ֫סָ֥ף אַ֤ךְ ט֭וֹב לְיִשְׂרָאֵ֥ל אֱלֹהִ֗ים לְבָרֵ֥י לֵבָֽב׃
2כִּ֭מְעַט [נָטוּי כ] (נָטָ֣יוּ ק) רַגְלָ֑י כְּ֝אַ֗יִן [שֻׁפְּכָה כ] (שֻׁפְּכ֥וּ ק) אֲשֻׁרָֽי׃
3קִ֭נֵּאתִי בַּֽהוֹלְלִ֑ים שְׁל֖וֹם רְשָׁעִ֣ים אֶרְאֶֽה׃
4אֵ֖ין חַרְצֻבּ֥וֹת לְמוֹתָ֗ם וּבָרִ֥יא אוּלָֽם׃
5אֱנ֣וֹשׁ אֵינֵ֑מוֹ וְעִם־ אָ֝דָ֗ם לֹ֣א יְנֻגָּֽעוּ׃
6עֲנָקַ֣תְמוֹ גַאֲוָ֑ה יַעֲטָף־ שִׁ֝֗ית חָמָ֥ס לָֽמוֹ׃
7מֵחֵ֣לֶב עֵינֵ֑מוֹ עָ֝בְר֗וּ מַשְׂכִּיּ֥וֹת לֵבָֽב׃
8וִידַבְּר֣וּ בְרָ֣ע עֹ֑שֶׁק מִמָּר֥וֹם יְדַבֵּֽרוּ׃
9בַשָּׁמַ֣יִם פִּיהֶ֑ם וּ֝לְשׁוֹנָ֗ם תִּֽהֲלַ֥ךְ בָּאָֽרֶץ׃
10[יָשִׁיב כ] (יָשׁ֣וּב ק) עַמּ֣וֹ הֲלֹ֑ם וּמֵ֥י מָ֝לֵ֗א יִמָּ֥צוּ לָֽמוֹ׃
11אֵיכָ֥ה יָדַֽע־ אֵ֑ל וְיֵ֖שׁ דֵּעָ֣ה בְעֶלְיֽוֹן׃
12אֵ֥לֶּה רְשָׁעִ֑ים וְשַׁלְוֵ֥י ע֝וֹלָ֗ם הִשְׂגּוּ־ חָֽיִל׃
13רִ֭יק זִכִּ֣יתִי לְבָבִ֑י וָאֶרְחַ֖ץ בְּנִקָּי֣וֹן כַּפָּֽי׃
14נָ֭גוּעַ כָּל־ הַיּ֑וֹם וְ֝תוֹכַחְתִּ֗י לַבְּקָרִֽים׃
15אָ֭מַרְתִּי אֲסַפְּרָ֥ה כְמ֑וֹ הִנֵּ֤ה ד֭וֹר בָּנֶ֣יךָ בָגָֽדְתִּי׃
16לָדַ֣עַת זֹ֑את עָמָ֖ל [הִיא כ] (ה֣וּא ק) בְעֵינָֽי׃
17אָ֭בוֹא אֶל־ מִקְדְּשֵׁי־ אֵ֑ל אָ֝בִ֗ינָה לְאַחֲרִיתָֽם׃
18בַּ֭חֲלָקוֹת תָּשִׁ֣ית לָ֑מוֹ הִ֝פַּלְתָּ֗ם לְמַשּׁוּאֽוֹת׃
19הָי֣וּ לְשַׁמָּ֣ה כְרָ֑גַע סָ֥פוּ תַ֝֗מּוּ מִן־ בַּלָּהֽוֹת׃
20מֵהָקִ֑יץ אֲ֝דֹנָי בָּעִ֤יר ׀ צַלְמָ֬ם תִּבְזֶֽה׃
21יִתְחַמֵּ֣ץ לְבָבִ֑י וְ֝כִלְיוֹתַ֗י אֶשְׁתּוֹנָֽן׃
22בַ֭עַר וְלֹ֣א אֵדָ֑ע בְּ֝הֵמ֗וֹת הָיִ֥יתִי עִמָּֽךְ׃
23תָמִ֣יד עִמָּ֑ךְ אָ֝חַ֗זְתָּ בְּיַד־ יְמִינִֽי׃
24תַנְחֵ֑נִי וְ֝אַחַ֗ר כָּב֥וֹד תִּקָּחֵֽנִי׃
25לִ֥י בַשָּׁמָ֑יִם וְ֝עִמְּךָ֗ לֹא־ חָפַ֥צְתִּי בָאָֽרֶץ׃
26שְׁאֵרִ֗י וּלְבָ֫בִ֥י צוּר־ לְבָבִ֥י וְחֶלְקִ֗י אֱלֹהִ֥ים לְעוֹלָֽם׃
27הִנֵּ֣ה רְחֵקֶ֣יךָ יֹאבֵ֑דוּ הִ֝צְמַ֗תָּה כָּל־ זוֹנֶ֥ה מִמֶּֽךָּ׃
