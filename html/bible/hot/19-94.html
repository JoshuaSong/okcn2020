1אֵל־ נְקָמ֥וֹת יְהוָ֑ה אֵ֖ל נְקָמ֣וֹת הוֹפִֽיַע׃
2שֹׁפֵ֣ט הָאָ֑רֶץ הָשֵׁ֥ב גְּ֝מ֗וּל עַל־ גֵּאִֽים׃
3מָתַ֖י רְשָׁעִ֥ים ׀ יְהוָ֑ה עַד־ מָ֝תַ֗י רְשָׁעִ֥ים יַעֲלֹֽזוּ׃
4יְדַבְּר֣וּ עָתָ֑ק יִֽ֝תְאַמְּר֗וּ כָּל־ פֹּ֥עֲלֵי אָֽוֶן׃
5יְהוָ֣ה יְדַכְּא֑וּ וְֽנַחֲלָתְךָ֥ יְעַנּֽוּ׃
6וְגֵ֣ר יַהֲרֹ֑גוּ וִֽיתוֹמִ֣ים יְרַצֵּֽחוּ׃
7לֹ֣א יִרְאֶה־ יָּ֑הּ וְלֹא־ יָ֝בִ֗ין אֱלֹהֵ֥י יַעֲקֹֽב׃
8בֹּעֲרִ֣ים בָּעָ֑ם וּ֝כְסִילִ֗ים מָתַ֥י תַּשְׂכִּֽילוּ׃
9אֹ֭זֶן הֲלֹ֣א יִשְׁמָ֑ע אִֽם־ יֹ֥צֵֽר עַ֝֗יִן הֲלֹ֣א יַבִּֽיט׃
10גּ֭וֹיִם הֲלֹ֣א יוֹכִ֑יחַ הַֽמְלַמֵּ֖ד אָדָ֣ם דָּֽעַת׃
11יֹ֭דֵעַ מַחְשְׁב֣וֹת אָדָ֑ם כִּי־ הֵ֥מָּה הָֽבֶל׃
12הַגֶּ֣בֶר אֲשֶׁר־ תְּיַסְּרֶ֣נּוּ יָּ֑הּ וּֽמִתּוֹרָתְךָ֥ תְלַמְּדֶֽנּוּ׃
13ל֭וֹ מִ֣ימֵי רָ֑ע עַ֤ד יִכָּרֶ֖ה לָרָשָׁ֣ע שָֽׁחַת׃
14לֹא־ יִטֹּ֣שׁ יְהוָ֣ה עַמּ֑וֹ וְ֝נַחֲלָת֗וֹ לֹ֣א יַעֲזֹֽב׃ כִּֽי־ עַד־
15יָשׁ֣וּב מִשְׁפָּ֑ט וְ֝אַחֲרָ֗יו כָּל־ יִשְׁרֵי־ לֵֽב׃
16יָק֣וּם לִ֭י עִם־ מְרֵעִ֑ים מִֽי־ יִתְיַצֵּ֥ב לִ֝י עִם־ פֹּ֥עֲלֵי אָֽוֶן׃
17יְ֭הוָה עֶזְרָ֣תָה לִּ֑י כִּמְעַ֓ט ׀ שָֽׁכְנָ֖ה דוּמָ֣ה נַפְשִֽׁי׃
18אָ֭מַרְתִּי מָ֣טָה רַגְלִ֑י חַסְדְּךָ֥ יְ֝הוָ֗ה יִסְעָדֵֽנִי׃
19שַׂרְעַפַּ֣י בְּקִרְבִּ֑י תַּ֝נְחוּמֶ֗יךָ יְֽשַׁעַשְׁע֥וּ נַפְשִֽׁי׃
20כִּסֵּ֣א הַוּ֑וֹת יֹצֵ֖ר עָמָ֣ל עֲלֵי־ חֹֽק׃
21עַל־ נֶ֣פֶשׁ צַדִּ֑יק וְדָ֖ם נָקִ֣י יַרְשִֽׁיעוּ׃
22יְהוָ֣ה לִ֣י לְמִשְׂגָּ֑ב וֵ֝אלֹהַ֗י לְצ֣וּר מַחְסִֽי׃
