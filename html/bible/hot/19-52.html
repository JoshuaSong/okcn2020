1לַמְנַצֵּ֗חַ מַשְׂכִּ֥יל לְדָוִֽד׃ בְּב֤וֹא ׀ דּוֹאֵ֣ג הָאֲדֹמִי֮ וַיַּגֵּ֪ד לְשָׁ֫א֥וּל וַיֹּ֥אמֶר ל֑וֹ בָּ֥א דָ֝וִ֗ד אֶל־ בֵּ֥ית אֲחִימֶֽלֶךְ׃ מַה־ תִּתְהַלֵּ֣ל בְּ֭רָעָה הַגִּבּ֑וֹר חֶ֥סֶד אֵ֝֗ל כָּל־ הַיּֽוֹם׃
2תַּחְשֹׁ֣ב לְשׁוֹנֶ֑ךָ כְּתַ֥עַר מְ֝לֻטָּ֗שׁ עֹשֵׂ֥ה רְמִיָּֽה׃
3רָּ֣ע מִטּ֑וֹב שֶׁ֓קֶר ׀ מִדַּבֵּ֖ר צֶ֣דֶק סֶֽלָה׃
4כָֽל־ דִּבְרֵי־ בָ֗לַע לְשׁ֣וֹן מִרְמָֽה׃
5אֵל֮ יִתָּצְךָ֪ לָ֫נֶ֥צַח יַחְתְּךָ֣ וְיִסָּחֲךָ֣ מֵאֹ֑הֶל וְשֵֽׁרֶשְׁךָ֨ מֵאֶ֖רֶץ חַיִּ֣ים סֶֽלָה׃
6צַדִּיקִ֥ים וְיִירָ֗אוּ וְעָלָ֥יו יִשְׂחָֽקוּ׃
7הַגֶּ֗בֶר לֹ֤א יָשִׂ֥ים אֱלֹהִ֗ים מָֽע֫וּזּ֥וֹ וַ֭יִּבְטַח בְּרֹ֣ב עָשְׁר֑וֹ יָ֝עֹ֗ז בְּהַוָּתֽוֹ׃
8כְּזַ֣יִת רַ֭עֲנָן בְּבֵ֣ית אֱלֹהִ֑ים בָּטַ֥חְתִּי בְחֶֽסֶד־ אֱ֝לֹהִ֗ים עוֹלָ֥ם וָעֶֽד׃
