1תְּהִלָּ֗ה לְדָ֫וִ֥ד אֲרוֹמִמְךָ֣ אֱלוֹהַ֣י הַמֶּ֑לֶךְ וַאֲבָרֲכָ֥ה שִׁ֝מְךָ֗ לְעוֹלָ֥ם וָעֶֽד׃
2י֥וֹם אֲבָרֲכֶ֑ךָּ וַאֲהַלְלָ֥ה שִׁ֝מְךָ֗ לְעוֹלָ֥ם וָעֶֽד׃
3יְהוָ֣ה וּמְהֻלָּ֣ל מְאֹ֑ד וְ֝לִגְדֻלָּת֗וֹ אֵ֣ין חֵֽקֶר׃
4לְ֭דוֹר יְשַׁבַּ֣ח מַעֲשֶׂ֑יךָ וּגְב֖וּרֹתֶ֣יךָ יַגִּֽידוּ׃
5כְּב֣וֹד הוֹדֶ֑ךָ וְדִבְרֵ֖י נִפְלְאוֹתֶ֣יךָ אָשִֽׂיחָה׃
6נוֹרְאֹתֶ֣יךָ יֹאמֵ֑רוּ [וּגְדוּלֹּתֶיךָ כ] (וּגְדוּלָּתְךָ֥ ק) אֲסַפְּרֶֽנָּה׃
7רַב־ טוּבְךָ֣ יַבִּ֑יעוּ וְצִדְקָתְךָ֥ יְרַנֵּֽנוּ׃
8וְרַח֣וּם יְהוָ֑ה אֶ֥רֶךְ אַ֝פַּ֗יִם וּגְדָל־ חָֽסֶד׃
9יְהוָ֥ה לַכֹּ֑ל וְ֝רַחֲמָ֗יו עַל־ כָּל־ מַעֲשָֽׂיו׃
10יְ֭הוָה כָּל־ מַעֲשֶׂ֑יךָ וַ֝חֲסִידֶ֗יךָ יְבָרֲכֽוּכָה׃
11מַלְכוּתְךָ֣ יֹאמֵ֑רוּ וּגְבוּרָתְךָ֥ יְדַבֵּֽרוּ׃
12לִבְנֵ֣י הָ֭אָדָם גְּבוּרֹתָ֑יו וּ֝כְב֗וֹד הֲדַ֣ר מַלְכוּתֽוֹ׃
13מַלְכ֥וּת כָּל־ עֹֽלָמִ֑ים וּ֝מֶֽמְשֶׁלְתְּךָ֗ בְּכָל־ דּ֥וֹר וָדֽוֹר׃
14יְ֭הוָה לְכָל־ הַנֹּפְלִ֑ים וְ֝זוֹקֵ֗ף לְכָל־ הַכְּפוּפִֽים׃
15כֹ֭ל אֵלֶ֣יךָ יְשַׂבֵּ֑רוּ וְאַתָּ֤ה נֽוֹתֵן־ לָהֶ֖ם אֶת־ אָכְלָ֣ם בְּעִתּֽוֹ׃
16אֶת־ יָדֶ֑ךָ וּמַשְׂבִּ֖יעַ לְכָל־ חַ֣י רָצֽוֹן׃
17יְ֭הוָה בְּכָל־ דְּרָכָ֑יו וְ֝חָסִ֗יד בְּכָל־ מַעֲשָֽׂיו׃
18יְ֭הוָה לְכָל־ קֹרְאָ֑יו לְכֹ֤ל אֲשֶׁ֖ר יִקְרָאֻ֣הוּ בֶאֱמֶֽת׃
19יְרֵאָ֥יו יַעֲשֶׂ֑ה וְֽאֶת־ שַׁוְעָתָ֥ם יִ֝שְׁמַ֗ע וְיוֹשִׁיעֵֽם׃
20יְ֭הוָה אֶת־ כָּל־ אֹהֲבָ֑יו וְאֵ֖ת כָּל־ הָרְשָׁעִ֣ים יַשְׁמִֽיד׃
