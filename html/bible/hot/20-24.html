1אַל־ תְּ֭קַנֵּא בְּאַנְשֵׁ֣י רָעָ֑ה וְאַל־ [תִּתְאָו כ] (תִּ֝תְאָ֗יו ק) לִהְי֥וֹת אִתָּֽם׃
2שֹׁ֭ד יֶהְגֶּ֣ה לִבָּ֑ם וְ֝עָמָ֗ל שִׂפְתֵיהֶ֥ם תְּדַבֵּֽרְנָה׃
3יִבָּ֣נֶה בָּ֑יִת וּ֝בִתְבוּנָ֗ה יִתְכּוֹנָֽן׃
4חֲדָרִ֣ים יִמָּלְא֑וּ כָּל־ ה֖וֹן יָקָ֣ר וְנָעִֽים׃
5חָכָ֥ם בַּע֑וֹז וְאִֽישׁ־ דַּ֝֗עַת מְאַמֶּץ־ כֹּֽחַ׃
6בְ֭תַחְבֻּלוֹת תַּעֲשֶׂה־ לְּךָ֣ מִלְחָמָ֑ה וּ֝תְשׁוּעָ֗ה בְּרֹ֣ב יוֹעֵֽץ׃
7לֶֽאֱוִ֣יל חָכְמ֑וֹת בַּ֝שַּׁ֗עַר לֹ֣א יִפְתַּח־ פִּֽיהוּ׃
8לְהָרֵ֑עַ ל֝֗וֹ בַּֽעַל־ מְזִמּ֥וֹת יִקְרָֽאוּ׃
9אִוֶּ֣לֶת חַטָּ֑את וְתוֹעֲבַ֖ת לְאָדָ֣ם לֵֽץ׃
10בְּי֥וֹם צָרָ֗ה צַ֣ר כֹּחֶֽכָה׃
11לְקֻחִ֣ים לַמָּ֑וֶת וּמָטִ֥ים לַ֝הֶ֗רֶג אִם־ תַּחְשֽׂוֹךְ׃
12תֹאמַ֗ר הֵן֮ לֹא־ יָדַ֪עְנ֫וּ זֶ֥ה הֲ‍ֽלֹא־ תֹ֘כֵ֤ן לִבּ֨וֹת ׀ הֽוּא־ יָבִ֗ין וְנֹצֵ֣ר נַ֭פְשְׁךָ ה֣וּא יֵדָ֑ע וְהֵשִׁ֖יב לְאָדָ֣ם כְּפָעֳלֽוֹ׃
13בְּנִ֣י דְבַ֣שׁ כִּי־ ט֑וֹב וְנֹ֥פֶת מָ֝ת֗וֹק עַל־ חִכֶּֽךָ׃
14דְּעֶ֥ה חָכְמָ֗ה לְנַ֫פְשֶׁ֥ךָ אִם־ מָ֭צָאתָ וְיֵ֣שׁ אַחֲרִ֑ית וְ֝תִקְוָתְךָ֗ לֹ֣א תִכָּרֵֽת׃ פ
15תֶּאֱרֹ֣ב רָ֭שָׁע לִנְוֵ֣ה צַדִּ֑יק אַֽל־ תְּשַׁדֵּ֥ד רִבְצֽוֹ׃
16שֶׁ֨בַע ׀ יִפּ֣וֹל צַדִּ֣יק וָקָ֑ם וּ֝רְשָׁעִ֗ים יִכָּשְׁל֥וּ בְרָעָֽה׃
17[אֹויְבֶיךָ כ] (אֹֽ֭ויִבְךָ ק) אַל־ תִּשְׂמָ֑ח וּ֝בִכָּשְׁל֗וֹ אַל־ יָגֵ֥ל לִבֶּֽךָ׃
18יִרְאֶ֣ה יְ֭הוָה וְרַ֣ע בְּעֵינָ֑יו וְהֵשִׁ֖יב מֵעָלָ֣יו אַפּֽוֹ׃
19תִּתְחַ֥ר בַּמְּרֵעִ֑ים אַל־ תְּ֝קַנֵּ֗א בָּרְשָׁעִֽים׃
20לֹֽא־ תִהְיֶ֣ה אַחֲרִ֣ית לָרָ֑ע נֵ֖ר רְשָׁעִ֣ים יִדְעָֽךְ׃ יְרָֽא־ אֶת־
21בְּנִ֣י וָמֶ֑לֶךְ עִם־ שׁ֝וֹנִ֗ים אַל־ תִּתְעָרָֽב׃
22פִ֭תְאֹם יָק֣וּם אֵידָ֑ם וּפִ֥יד שְׁ֝נֵיהֶ֗ם מִ֣י יוֹדֵֽעַ׃ ס
23אֵ֥לֶּה לַֽחֲכָמִ֑ים הַֽכֵּר־ פָּנִ֖ים בְּמִשְׁפָּ֣ט בַּל־ טֽוֹב׃
24לְרָשָׁע֮ צַדִּ֪יק אָ֥תָּה יִקְּבֻ֥הוּ עַמִּ֑ים יִזְעָמ֥וּהוּ לְאֻמִּֽים׃
25יִנְעָ֑ם וַֽ֝עֲלֵיהֶ֗ם תָּב֥וֹא בִרְכַּת־ טֽוֹב׃
26יִשָּׁ֑ק מֵ֝שִׁ֗יב דְּבָרִ֥ים נְכֹחִֽים׃
27בַּח֨וּץ ׀ מְלַאכְתֶּ֗ךָ וְעַתְּדָ֣הּ בַּשָּׂדֶ֣ה לָ֑ךְ אַ֝חַ֗ר וּבָנִ֥יתָ בֵיתֶֽךָ׃ פ
28תְּהִ֣י עֵד־ חִנָּ֣ם בְּרֵעֶ֑ךָ וַ֝הֲפִתִּ֗יתָ בִּשְׂפָתֶֽיךָ׃
29תֹּאמַ֗ר כַּאֲשֶׁ֣ר עָֽשָׂה־ לִ֭י כֵּ֤ן אֶֽעֱשֶׂה־ לּ֑וֹ אָשִׁ֖יב לָאִ֣ישׁ כְּפָעֳלֽוֹ׃
30שְׂדֵ֣ה אִישׁ־ עָצֵ֣ל עָבַ֑רְתִּי וְעַל־ כֶּ֝֗רֶם אָדָ֥ם חֲסַר־ לֵֽב׃
31עָ֘לָ֤ה כֻלּ֨וֹ ׀ קִמְּשֹׂנִ֗ים כָּסּ֣וּ פָנָ֣יו חֲרֻלִּ֑ים וְגֶ֖דֶר אֲבָנָ֣יו נֶהֱרָֽסָה׃
32אָ֭נֹכִֽי אָשִׁ֣ית לִבִּ֑י רָ֝אִ֗יתִי לָקַ֥חְתִּי מוּסָֽר׃
33שֵׁ֭נוֹת מְעַ֣ט תְּנוּמ֑וֹת מְעַ֓ט ׀ חִבֻּ֖ק יָדַ֣יִם לִשְׁכָּֽב׃
