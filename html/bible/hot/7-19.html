1וַיְהִי֙ בַּיָּמִ֣ים הָהֵ֔ם וּמֶ֖לֶךְ אֵ֣ין בְּיִשְׂרָאֵ֑ל וַיְהִ֣י ׀ אִ֣ישׁ לֵוִ֗י גָּ֚ר בְּיַרְכְּתֵ֣י הַר־ אֶפְרַ֔יִם וַיִּֽקַּֽח־ לוֹ֙ אִשָּׁ֣ה פִילֶ֔גֶשׁ מִבֵּ֥ית לֶ֖חֶם יְהוּדָֽה׃
2עָלָיו֙ פִּֽילַגְשׁ֔וֹ וַתֵּ֤לֶךְ מֵֽאִתּוֹ֙ אֶל־ בֵּ֣ית אָבִ֔יהָ אֶל־ בֵּ֥ית לֶ֖חֶם יְהוּדָ֑ה וַתְּהִי־ שָׁ֕ם יָמִ֖ים אַרְבָּעָ֥ה חֳדָשִֽׁים׃
3אִישָׁ֜הּ וַיֵּ֣לֶךְ אַחֲרֶ֗יהָ לְדַבֵּ֤ר עַל־ לִבָּהּ֙ [לַהֲשִׁיבֹו כ] (לַהֲשִׁיבָ֔הּ ק) וְנַעֲר֥וֹ עִמּ֖וֹ וְצֶ֣מֶד חֲמֹרִ֑ים וַתְּבִיאֵ֙הוּ֙ בֵּ֣ית אָבִ֔יהָ וַיִּרְאֵ֙הוּ֙ אֲבִ֣י הַֽנַּעֲרָ֔ה וַיִּשְׂמַ֖ח לִקְרָאתֽוֹ׃
4בּ֤וֹ חֹֽתְנוֹ֙ אֲבִ֣י הַֽנַּעֲרָ֔ה וַיֵּ֥שֶׁב אִתּ֖וֹ שְׁלֹ֣שֶׁת יָמִ֑ים וַיֹּאכְלוּ֙ וַיִּשְׁתּ֔וּ וַיָּלִ֖ינוּ שָֽׁם׃
5בַּיּ֣וֹם הָרְבִיעִ֔י וַיַּשְׁכִּ֥ימוּ בַבֹּ֖קֶר וַיָּ֣קָם לָלֶ֑כֶת וַיֹּאמֶר֩ אֲבִ֨י הַֽנַּעֲרָ֜ה אֶל־ חֲתָנ֗וֹ סְעָ֧ד לִבְּךָ֛ פַּת־ לֶ֖חֶם וְאַחַ֥ר תֵּלֵֽכוּ׃
6וַיֹּאכְל֧וּ שְׁנֵיהֶ֛ם יַחְדָּ֖ו וַיִּשְׁתּ֑וּ וַיֹּ֜אמֶר אֲבִ֤י הַֽנַּעֲרָה֙ אֶל־ הָאִ֔ישׁ הֽוֹאֶל־ נָ֥א וְלִ֖ין וְיִטַ֥ב לִבֶּֽךָ׃
7הָאִ֖ישׁ לָלֶ֑כֶת וַיִּפְצַר־ בּוֹ֙ חֹתְנ֔וֹ וַיָּ֖שָׁב וַיָּ֥לֶן שָֽׁם׃
8בַּבֹּ֜קֶר בַּיּ֣וֹם הַחֲמִישִׁי֮ לָלֶכֶת֒ וַיֹּ֣אמֶר ׀ אֲבִ֣י הַֽנַּעֲרָ֗ה סְעָד־ נָא֙ לְבָ֣בְךָ֔ וְהִֽתְמַהְמְה֖וּ עַד־ נְט֣וֹת הַיּ֑וֹם וַיֹּאכְל֖וּ שְׁנֵיהֶֽם׃
9הָאִישׁ֙ לָלֶ֔כֶת ה֥וּא וּפִילַגְשׁ֖וֹ וְנַעֲר֑וֹ וַיֹּ֣אמֶר ל֣וֹ חֹתְנ֣וֹ אֲבִ֣י הַֽנַּעֲרָ֡ה הִנֵּ֣ה נָא֩ רָפָ֨ה הַיּ֜וֹם לַעֲרֹ֗ב לִֽינוּ־ נָ֞א הִנֵּ֨ה חֲנ֤וֹת הַיּוֹם֙ לִ֥ין פֹּה֙ וְיִיטַ֣ב לְבָבֶ֔ךָ וְהִשְׁכַּמְתֶּ֤ם מָחָר֙ לְדַרְכְּכֶ֔ם וְהָלַכְתָּ֖ לְאֹהָלֶֽךָ׃
10אָבָ֤ה הָאִישׁ֙ לָל֔וּן וַיָּ֣קָם וַיֵּ֗לֶךְ וַיָּבֹא֙ עַד־ נֹ֣כַח יְב֔וּס הִ֖יא יְרוּשָׁלִָ֑ם וְעִמּ֗וֹ צֶ֤מֶד חֲמוֹרִים֙ חֲבוּשִׁ֔ים וּפִילַגְשׁ֖וֹ עִמּֽוֹ׃
11עִם־ יְב֔וּס וְהַיּ֖וֹם רַ֣ד מְאֹ֑ד וַיֹּ֨אמֶר הַנַּ֜עַר אֶל־ אֲדֹנָ֗יו לְכָה־ נָּ֛א וְנָס֛וּרָה אֶל־ עִֽיר־ הַיְבוּסִ֥י הַזֹּ֖את וְנָלִ֥ין בָּֽהּ׃
12אֵלָיו֙ אֲדֹנָ֔יו לֹ֤א נָסוּר֙ אֶל־ עִ֣יר נָכְרִ֔י אֲשֶׁ֛ר לֹֽא־ מִבְּנֵ֥י יִשְׂרָאֵ֖ל הֵ֑נָּה וְעָבַ֖רְנוּ עַד־ גִּבְעָֽה׃
13לְנַעֲר֔וֹ לְךָ֥ וְנִקְרְבָ֖ה בְּאַחַ֣ד הַמְּקֹמ֑וֹת וְלַ֥נּוּ בַגִּבְעָ֖ה א֥וֹ בָרָמָֽה׃
14וַיֵּלֵ֑כוּ וַתָּבֹ֤א לָהֶם֙ הַשֶּׁ֔מֶשׁ אֵ֥צֶל הַגִּבְעָ֖ה אֲשֶׁ֥ר לְבִנְיָמִֽן׃
15שָׁ֔ם לָב֖וֹא לָל֣וּן בַּגִּבְעָ֑ה וַיָּבֹ֗א וַיֵּ֙שֶׁב֙ בִּרְח֣וֹב הָעִ֔יר וְאֵ֥ין אִ֛ישׁ מְאַסֵּֽף־ אוֹתָ֥ם הַבַּ֖יְתָה לָלֽוּן׃
16אִ֣ישׁ זָקֵ֗ן בָּ֣א מִֽן־ מַעֲשֵׂ֤הוּ מִן־ הַשָּׂדֶה֙ בָּעֶ֔רֶב וְהָאִישׁ֙ מֵהַ֣ר אֶפְרַ֔יִם וְהוּא־ גָ֖ר בַּגִּבְעָ֑ה וְאַנְשֵׁ֥י הַמָּק֖וֹם בְּנֵ֥י יְמִינִֽי׃
17עֵינָ֗יו וַיַּ֛רְא אֶת־ הָאִ֥ישׁ הָאֹרֵ֖חַ בִּרְחֹ֣ב הָעִ֑יר וַיֹּ֨אמֶר הָאִ֧ישׁ הַזָּקֵ֛ן אָ֥נָה תֵלֵ֖ךְ וּמֵאַ֥יִן תָּבֽוֹא׃
18אֵלָ֗יו עֹבְרִ֨ים אֲנַ֜חְנוּ מִבֵּֽית־ לֶ֣חֶם יְהוּדָה֮ עַד־ יַרְכְּתֵ֣י הַר־ אֶפְרַיִם֒ מִשָּׁ֣ם אָנֹ֔כִי וָאֵלֵ֕ךְ עַד־ בֵּ֥ית לֶ֖חֶם יְהוּדָ֑ה וְאֶת־ בֵּ֤ית יְהוָה֙ אֲנִ֣י הֹלֵ֔ךְ וְאֵ֣ין אִ֔ישׁ מְאַסֵּ֥ף אוֹתִ֖י הַבָּֽיְתָה׃
19תֶּ֤בֶן גַּם־ מִסְפּוֹא֙ יֵ֣שׁ לַחֲמוֹרֵ֔ינוּ וְ֠גַם לֶ֣חֶם וָיַ֤יִן יֶשׁ־ לִי֙ וְלַֽאֲמָתֶ֔ךָ וְלַנַּ֖עַר עִם־ עֲבָדֶ֑יךָ אֵ֥ין מַחְס֖וֹר כָּל־ דָּבָֽר׃
20הָאִ֤ישׁ הַזָּקֵן֙ שָׁל֣וֹם לָ֔ךְ רַ֥ק כָּל־ מַחְסוֹרְךָ֖ עָלָ֑י רַ֥ק בָּרְח֖וֹב אַל־ תָּלַֽן׃
21לְבֵית֔וֹ [וַיִּבֹּול כ] (וַיָּ֖בָל ק) לַחֲמוֹרִ֑ים וַֽיִּרְחֲצוּ֙ רַגְלֵיהֶ֔ם וַיֹּאכְל֖וּ וַיִּשְׁתּֽוּ׃
22מֵיטִיבִ֣ים אֶת־ לִבָּם֒ וְהִנֵּה֩ אַנְשֵׁ֨י הָעִ֜יר אַנְשֵׁ֣י בְנֵֽי־ בְלִיַּ֗עַל נָסַ֙בּוּ֙ אֶת־ הַבַּ֔יִת מִֽתְדַּפְּקִ֖ים עַל־ הַדָּ֑לֶת וַיֹּאמְר֗וּ אֶל־ הָ֠אִישׁ בַּ֣עַל הַבַּ֤יִת הַזָּקֵן֙ לֵאמֹ֔ר הוֹצֵ֗א אֶת־ הָאִ֛ישׁ אֲשֶׁר־ בָּ֥א אֶל־ בֵּיתְךָ֖ וְנֵדָעֶֽנּוּ׃
23אֲלֵיהֶ֗ם הָאִישׁ֙ בַּ֣עַל הַבַּ֔יִת וַיֹּ֣אמֶר אֲלֵהֶ֔ם אַל־ אַחַ֖י אַל־ תָּרֵ֣עוּ נָ֑א אַ֠חֲרֵי אֲשֶׁר־ בָּ֞א הָאִ֤ישׁ הַזֶּה֙ אַל־ בֵּיתִ֔י אַֽל־ תַּעֲשׂ֖וּ אֶת־ הַנְּבָלָ֥ה הַזֹּֽאת׃
24בִתִּ֨י הַבְּתוּלָ֜ה וּפִֽילַגְשֵׁ֗הוּ אוֹצִֽיאָה־ נָּ֤א אוֹתָם֙ וְעַנּ֣וּ אוֹתָ֔ם וַעֲשׂ֣וּ לָהֶ֔ם הַטּ֖וֹב בְּעֵינֵיכֶ֑ם וְלָאִ֤ישׁ הַזֶּה֙ לֹ֣א תַעֲשׂ֔וּ דְּבַ֖ר הַנְּבָלָ֥ה הַזֹּֽאת׃
25אָב֤וּ הָאֲנָשִׁים֙ לִשְׁמֹ֣עַֽ ל֔וֹ וַיַּחֲזֵ֤ק הָאִישׁ֙ בְּפִ֣ילַגְשׁ֔וֹ וַיֹּצֵ֥א אֲלֵיהֶ֖ם הַח֑וּץ וַיֵּדְע֣וּ א֠וֹתָהּ וַיִּֽתְעַלְּלוּ־ בָ֤הּ כָּל־ הַלַּ֙יְלָה֙ עַד־ הַבֹּ֔קֶר וַֽיְשַׁלְּח֖וּהָ [בַּעֲלֹות כ] (כַּעֲלֹ֥ות ק) הַשָּֽׁחַר׃
26הָאִשָּׁ֖ה לִפְנ֣וֹת הַבֹּ֑קֶר וַתִּפֹּ֞ל פֶּ֧תַח בֵּית־ הָאִ֛ישׁ אֲשֶׁר־ אֲדוֹנֶ֥יהָ שָּׁ֖ם עַד־ הָאֽוֹר׃
27אֲדֹנֶ֜יהָ בַּבֹּ֗קֶר וַיִּפְתַּח֙ דַּלְת֣וֹת הַבַּ֔יִת וַיֵּצֵ֖א לָלֶ֣כֶת לְדַרְכּ֑וֹ וְהִנֵּ֧ה הָאִשָּׁ֣ה פִֽילַגְשׁ֗וֹ נֹפֶ֙לֶת֙ פֶּ֣תַח הַבַּ֔יִת וְיָדֶ֖יהָ עַל־ הַסַּֽף׃
28אֵלֶ֛יהָ ק֥וּמִי וְנֵלֵ֖כָה וְאֵ֣ין עֹנֶ֑ה וַיִּקָּחֶ֙הָ֙ עַֽל־ הַחֲמ֔וֹר וַיָּ֣קָם הָאִ֔ישׁ וַיֵּ֖לֶךְ לִמְקֹמֽוֹ׃
29אֶל־ בֵּית֗וֹ וַיִּקַּ֤ח אֶת־ הַֽמַּאֲכֶ֙לֶת֙ וַיַּחֲזֵ֣ק בְּפִֽילַגְשׁ֔וֹ וַֽיְנַתְּחֶ֙הָ֙ לַעֲצָמֶ֔יהָ לִשְׁנֵ֥ים עָשָׂ֖ר נְתָחִ֑ים וַֽיְשַׁלְּחֶ֔הָ בְּכֹ֖ל גְּב֥וּל יִשְׂרָאֵֽל׃
