1לַמְנַצֵּ֥חַ [לִידִיתוּן כ] (לִֽידוּת֗וּן ק) מִזְמ֥וֹר לְדָוִֽד׃ אָמַ֗רְתִּי אֶֽשְׁמְרָ֣ה דְרָכַי֮ מֵחֲט֪וֹא בִלְשׁ֫וֹנִ֥י אֶשְׁמְרָ֥ה לְפִ֥י מַחְס֑וֹם בְּעֹ֖ד רָשָׁ֣ע לְנֶגְדִּֽי׃
2ד֭וּמִיָּה הֶחֱשֵׁ֣יתִי מִטּ֑וֹב וּכְאֵבִ֥י נֶעְכָּֽר׃
3לִבִּ֨י ׀ בְּקִרְבִּ֗י בַּהֲגִיגִ֥י תִבְעַר־ אֵ֑שׁ דִּ֝בַּ֗רְתִּי בִּלְשֽׁוֹנִי׃
4יְהוָ֨ה ׀ קִצִּ֗י וּמִדַּ֣ת יָמַ֣י מַה־ הִ֑יא אֵ֝דְעָ֗ה מֶה־ חָדֵ֥ל אָֽנִי׃
5טְפָח֨וֹת ׀ נָ֘תַ֤תָּה יָמַ֗י וְחֶלְדִּ֣י כְאַ֣יִן נֶגְדֶּ֑ךָ אַ֥ךְ כָּֽל־ הֶ֥בֶל כָּל־ אָ֝דָ֗ם נִצָּ֥ב סֶֽלָה׃
6בְּצֶ֤לֶם ׀ יִֽתְהַלֶּךְ־ אִ֗ישׁ אַךְ־ הֶ֥בֶל יֶהֱמָי֑וּן יִ֝צְבֹּ֗ר וְֽלֹא־ יֵדַ֥ע מִי־ אֹסְפָֽם׃
7מַה־ קִּוִּ֣יתִי אֲדֹנָ֑י תּ֝וֹחַלְתִּ֗י לְךָ֣ הִֽיא׃
8פְּשָׁעַ֥י הַצִּילֵ֑נִי חֶרְפַּ֥ת נָ֝בָ֗ל אַל־ תְּשִׂימֵֽנִי׃
9לֹ֣א אֶפְתַּח־ פִּ֑י כִּ֖י אַתָּ֣ה עָשִֽׂיתָ׃
10מֵעָלַ֣י נִגְעֶ֑ךָ מִתִּגְרַ֥ת יָ֝דְךָ֗ אֲנִ֣י כָלִֽיתִי׃
11עַל־ עָוֹ֨ן ׀ יִסַּ֬רְתָּ אִ֗ישׁ וַתֶּ֣מֶס כָּעָ֣שׁ חֲמוּד֑וֹ אַ֤ךְ הֶ֖בֶל כָּל־ אָדָ֣ם סֶֽלָה׃
12תְפִלָּתִ֨י ׀ יְהוָ֡ה וְשַׁוְעָתִ֨י ׀ הַאֲזִינָה֮ אֶֽל־ דִּמְעָתִ֗י אַֽל־ תֶּ֫חֱרַ֥שׁ כִּ֤י גֵ֣ר אָנֹכִ֣י עִמָּ֑ךְ תּ֝וֹשָׁ֗ב כְּכָל־ אֲבוֹתָֽי׃
