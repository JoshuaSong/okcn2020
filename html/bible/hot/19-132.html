1שִׁ֗יר הַֽמַּ֫עֲל֥וֹת זְכוֹר־ יְהוָ֥ה לְדָוִ֑ד אֵ֝ת כָּל־ עֻנּוֹתֽוֹ׃
2נִ֭שְׁבַּע לַיהוָ֑ה נָ֝דַ֗ר לַאֲבִ֥יר יַעֲקֹֽב׃
3אָ֭בֹא בְּאֹ֣הֶל בֵּיתִ֑י אִם־ אֶ֝עֱלֶ֗ה עַל־ עֶ֥רֶשׂ יְצוּעָֽי׃
4אֶתֵּ֣ן שְׁנַ֣ת לְעֵינָ֑י לְֽעַפְעַפַּ֥י תְּנוּמָֽה׃
5אֶמְצָ֣א מָ֭קוֹם לַיהוָ֑ה מִ֝שְׁכָּנ֗וֹת לַאֲבִ֥יר יַעֲקֹֽב׃
6שְׁמַֽעֲנ֥וּהָ בְאֶפְרָ֑תָה מְ֝צָאנ֗וּהָ בִּשְׂדֵי־ יָֽעַר׃
7לְמִשְׁכְּנוֹתָ֑יו נִ֝שְׁתַּחֲוֶ֗ה לַהֲדֹ֥ם רַגְלָֽיו׃
8יְ֭הוָה לִמְנוּחָתֶ֑ךָ אַ֝תָּ֗ה וַאֲר֥וֹן עֻזֶּֽךָ׃
9יִלְבְּשׁוּ־ צֶ֑דֶק וַחֲסִידֶ֥יךָ יְרַנֵּֽנוּ׃
10דָּוִ֣ד עַבְדֶּ֑ךָ אַל־ תָּ֝שֵׁ֗ב פְּנֵ֣י מְשִׁיחֶֽךָ׃
11יְהוָ֨ה ׀ לְדָוִ֡ד אֱמֶת֮ לֹֽא־ יָשׁ֪וּב מִ֫מֶּ֥נָּה מִפְּרִ֥י בִטְנְךָ֑ אָ֝שִׁ֗ית לְכִסֵּא־ לָֽךְ׃
12יִשְׁמְר֬וּ בָנֶ֨יךָ ׀ בְּרִיתִי֮ וְעֵדֹתִ֥י ז֗וֹ אֲלַ֫מְּדֵ֥ם גַּם־ בְּנֵיהֶ֥ם עֲדֵי־ עַ֑ד יֵ֝שְׁב֗וּ לְכִסֵּא־ לָֽךְ׃
13בָחַ֣ר יְהוָ֣ה בְּצִיּ֑וֹן אִ֝וָּ֗הּ לְמוֹשָׁ֥ב לֽוֹ׃
14מְנוּחָתִ֥י עֲדֵי־ עַ֑ד פֹּֽה־ אֵ֝שֵׁ֗ב כִּ֣י אִוִּתִֽיהָ׃
15בָּרֵ֣ךְ אֲבָרֵ֑ךְ אֶ֝בְיוֹנֶ֗יהָ אַשְׂבִּ֥יעַֽ לָֽחֶם׃
16אַלְבִּ֣ישׁ יֶ֑שַׁע וַ֝חֲסִידֶ֗יהָ רַנֵּ֥ן יְרַנֵּֽנוּ׃
17אַצְמִ֣יחַ קֶ֣רֶן לְדָוִ֑ד עָרַ֥כְתִּי נֵ֝֗ר לִמְשִׁיחִֽי׃
