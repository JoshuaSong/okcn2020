1הֵ֤ן עַבְדִּי֙ אֶתְמָךְ־ בּ֔וֹ בְּחִירִ֖י רָצְתָ֣ה נַפְשִׁ֑י נָתַ֤תִּי רוּחִי֙ עָלָ֔יו מִשְׁפָּ֖ט לַגּוֹיִ֥ם יוֹצִֽיא׃
2יִצְעַ֖ק וְלֹ֣א יִשָּׂ֑א וְלֹֽא־ יַשְׁמִ֥יעַ בַּח֖וּץ קוֹלֽוֹ׃
3רָצוּץ֙ לֹ֣א יִשְׁבּ֔וֹר וּפִשְׁתָּ֥ה כֵהָ֖ה לֹ֣א יְכַבֶּ֑נָּה לֶאֱמֶ֖ת יוֹצִ֥יא מִשְׁפָּֽט׃
4יִכְהֶה֙ וְלֹ֣א יָר֔וּץ עַד־ יָשִׂ֥ים בָּאָ֖רֶץ מִשְׁפָּ֑ט וּלְתוֹרָת֖וֹ אִיִּ֥ים יְיַחֵֽילוּ׃ פ
5אָמַ֞ר הָאֵ֣ל ׀ יְהוָ֗ה בּוֹרֵ֤א הַשָּׁמַ֙יִם֙ וְנ֣וֹטֵיהֶ֔ם רֹקַ֥ע הָאָ֖רֶץ וְצֶאֱצָאֶ֑יהָ נֹתֵ֤ן נְשָׁמָה֙ לָעָ֣ם עָלֶ֔יהָ וְר֖וּחַ לַהֹלְכִ֥ים בָּֽהּ׃
6יְהוָ֛ה קְרָאתִ֥יךָֽ בְצֶ֖דֶק וְאַחְזֵ֣ק בְּיָדֶ֑ךָ וְאֶצָּרְךָ֗ וְאֶתֶּנְךָ֛ לִבְרִ֥ית עָ֖ם לְא֥וֹר גּוֹיִֽם׃
7עֵינַ֣יִם עִוְר֑וֹת לְהוֹצִ֤יא מִמַּסְגֵּר֙ אַסִּ֔יר מִבֵּ֥ית כֶּ֖לֶא יֹ֥שְׁבֵי חֹֽשֶׁךְ׃
8יְהוָ֖ה ה֣וּא שְׁמִ֑י וּכְבוֹדִי֙ לְאַחֵ֣ר לֹֽא־ אֶתֵּ֔ן וּתְהִלָּתִ֖י לַפְּסִילִֽים׃
9הִנֵּה־ בָ֑אוּ וַֽחֲדָשׁוֹת֙ אֲנִ֣י מַגִּ֔יד בְּטֶ֥רֶם תִּצְמַ֖חְנָה אַשְׁמִ֥יע אֶתְכֶֽם׃ פ
10לַֽיהוָה֙ שִׁ֣יר חָדָ֔שׁ תְּהִלָּת֖וֹ מִקְצֵ֣ה הָאָ֑רֶץ יוֹרְדֵ֤י הַיָּם֙ וּמְלֹא֔וֹ אִיִּ֖ים וְיֹשְׁבֵיהֶֽם׃
11מִדְבָּר֙ וְעָרָ֔יו חֲצֵרִ֖ים תֵּשֵׁ֣ב קֵדָ֑ר יָרֹ֙נּוּ֙ יֹ֣שְׁבֵי סֶ֔לַע מֵרֹ֥אשׁ הָרִ֖ים יִצְוָֽחוּ׃
12לַֽיהוָ֖ה כָּב֑וֹד וּתְהִלָּת֖וֹ בָּאִיִּ֥ים יַגִּֽידוּ׃
13כַּגִּבּ֣וֹר יֵצֵ֔א כְּאִ֥ישׁ מִלְחָמ֖וֹת יָעִ֣יר קִנְאָ֑ה יָרִ֙יעַ֙ אַף־ יַצְרִ֔יחַ עַל־ אֹיְבָ֖יו יִתְגַּבָּֽר׃ ס
14מֵֽעוֹלָ֔ם אַחֲרִ֖ישׁ אֶתְאַפָּ֑ק כַּיּוֹלֵדָ֣ה אֶפְעֶ֔ה אֶשֹּׁ֥ם וְאֶשְׁאַ֖ף יָֽחַד׃
15הָרִים֙ וּגְבָע֔וֹת וְכָל־ עֶשְׂבָּ֖ם אוֹבִ֑ישׁ וְשַׂמְתִּ֤י נְהָרוֹת֙ לָֽאִיִּ֔ים וַאֲגַמִּ֖ים אוֹבִֽישׁ׃
16עִוְרִ֗ים בְּדֶ֙רֶךְ֙ לֹ֣א יָדָ֔עוּ בִּנְתִיב֥וֹת לֹֽא־ יָדְע֖וּ אַדְרִיכֵ֑ם אָשִׂים֩ מַחְשָׁ֨ךְ לִפְנֵיהֶ֜ם לָא֗וֹר וּמַֽעֲקַשִּׁים֙ לְמִישׁ֔וֹר אֵ֚לֶּה הַדְּבָרִ֔ים עֲשִׂיתִ֖ם וְלֹ֥א עֲזַבְתִּֽים׃
17אָחוֹר֙ יֵבֹ֣שׁוּ בֹ֔שֶׁת הַבֹּטְחִ֖ים בַּפָּ֑סֶל הָאֹמְרִ֥ים לְמַסֵּכָ֖ה אַתֶּ֥ם אֱלֹהֵֽינוּ׃ ס
18שְׁמָ֑עוּ וְהַעִוְרִ֖ים הַבִּ֥יטוּ לִרְאֽוֹת׃
19עִוֵּר֙ כִּ֣י אִם־ עַבְדִּ֔י וְחֵרֵ֖שׁ כְּמַלְאָכִ֣י אֶשְׁלָ֑ח מִ֤י עִוֵּר֙ כִּמְשֻׁלָּ֔ם וְעִוֵּ֖ר כְּעֶ֥בֶד יְהוָֽה׃
20כ] (רָאֹ֥ות ק) רַבּ֖וֹת וְלֹ֣א תִשְׁמֹ֑ר פָּק֥וֹחַ אָזְנַ֖יִם וְלֹ֥א יִשְׁמָֽע׃
21חָפֵ֖ץ לְמַ֣עַן צִדְק֑וֹ יַגְדִּ֥יל תּוֹרָ֖ה וְיַאְדִּֽיר׃
22עַם־ בָּז֣וּז וְשָׁסוּי֒ הָפֵ֤חַ בַּֽחוּרִים֙ כֻּלָּ֔ם וּבְבָתֵּ֥י כְלָאִ֖ים הָחְבָּ֑אוּ הָי֤וּ לָבַז֙ וְאֵ֣ין מַצִּ֔יל מְשִׁסָּ֖ה וְאֵין־ אֹמֵ֥ר הָשַֽׁב׃
23בָכֶ֖ם יַאֲזִ֣ין זֹ֑את יַקְשִׁ֥ב וְיִשְׁמַ֖ע לְאָחֽוֹר׃
24נָתַ֨ן [לִמְשֹׁוסֶה כ] (לִמְשִׁסָּ֧ה ק) יַעֲקֹ֛ב וְיִשְׂרָאֵ֥ל לְבֹזְזִ֖ים הֲל֣וֹא יְהוָ֑ה ז֚וּ חָטָ֣אנוּ ל֔וֹ וְלֹֽא־ אָב֤וּ בִדְרָכָיו֙ הָל֔וֹךְ וְלֹ֥א שָׁמְע֖וּ בְּתוֹרָתֽוֹ׃
