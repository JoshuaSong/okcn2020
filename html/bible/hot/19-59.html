1לַמְנַצֵּ֣חַ אַל־ תַּשְׁחֵת֮ לְדָוִ֪ד מִ֫כְתָּ֥ם בִּשְׁלֹ֥חַ שָׁא֑וּל וַֽיִּשְׁמְר֥וּ אֶת־ הַ֝בַּ֗יִת לַהֲמִיתֽוֹ׃ הַצִּילֵ֖נִי מֵאֹיְבַ֥י ׀ אֱלֹהָ֑י מִּמִתְקוֹמְמַ֥י תְּשַׂגְּבֵֽנִי׃
2מִפֹּ֣עֲלֵי אָ֑וֶן וּֽמֵאַנְשֵׁ֥י דָ֝מִ֗ים הוֹשִׁיעֵֽנִי׃
3הִנֵּ֪ה אָֽרְב֡וּ לְנַפְשִׁ֗י יָג֣וּרוּ עָלַ֣י עַזִ֑ים לֹא־ פִשְׁעִ֖י וְלֹא־ חַטָּאתִ֣י יְהוָֽה׃
4עָ֭וֹן יְרוּצ֣וּן וְיִכּוֹנָ֑נוּ ע֖וּרָה לִקְרָאתִ֣י וּרְאֵה׃
5יְהוָֽה־ אֱלֹהִ֥ים ׀ צְבָא֡וֹת אֱלֹ֘הֵ֤י יִשְׂרָאֵ֗ל הָקִ֗יצָה לִפְקֹ֥ד כָּֽל־ הַגּוֹיִ֑ם אַל־ תָּחֹ֨ן כָּל־ בֹּ֖גְדֵי אָ֣וֶן סֶֽלָה׃
6לָ֭עֶרֶב יֶהֱמ֥וּ כַכָּ֗לֶב וִיס֥וֹבְבוּ עִֽיר׃
7יַבִּ֘יע֤וּן בְּפִיהֶ֗ם חֲ֭רָבוֹת בְּשִׂפְתוֹתֵיהֶ֑ם כִּי־ מִ֥י שֹׁמֵֽעַ׃
8יְ֭הוָה תִּשְׂחַק־ לָ֑מוֹ תִּ֝לְעַ֗ג לְכָל־ גּוֹיִֽם׃
9אֵלֶ֣יךָ אֶשְׁמֹ֑רָה כִּֽי־ אֱ֝לֹהִ֗ים מִשְׂגַּבִּֽי׃
10[חַסְדֹּו כ] (חַסְדִּ֣י ק) יְקַדְּמֵ֑נִי אֱ֝לֹהִ֗ים יַרְאֵ֥נִי בְשֹׁרְרָֽי׃
11תַּהַרְגֵ֤ם ׀ פֶּֽן־ יִשְׁכְּח֬וּ עַמִּ֗י הֲנִיעֵ֣מוֹ בְ֭חֵילְךָ וְהוֹרִידֵ֑מוֹ מָֽגִנֵּ֣נוּ אֲדֹנָֽי׃
12פִּ֗ימוֹ דְּֽבַר־ שְׂפָ֫תֵ֥ימוֹ וְיִלָּכְד֥וּ בִגְאוֹנָ֑ם וּמֵאָלָ֖ה וּמִכַּ֣חַשׁ יְסַפֵּֽרוּ׃
13בְחֵמָה֮ כַּלֵּ֪ה וְֽאֵ֫ינֵ֥מוֹ וְֽיֵדְע֗וּ כִּֽי־ אֱ֭לֹהִים מֹשֵׁ֣ל בְּיַעֲקֹ֑ב לְאַפְסֵ֖י הָאָ֣רֶץ סֶֽלָה׃
14לָ֭עֶרֶב יֶהֱמ֥וּ כַכָּ֗לֶב וִיס֥וֹבְבוּ עִֽיר׃
15[יְנוּעוּן כ] (יְנִיע֣וּן ק) לֶאֱכֹ֑ל אִם־ לֹ֥א יִ֝שְׂבְּע֗וּ וַיָּלִֽינוּ׃
16אָשִׁ֣יר עֻזֶּךָ֮ וַאֲרַנֵּ֥ן לַבֹּ֗קֶר חַ֫סְדֶּ֥ךָ כִּֽי־ הָיִ֣יתָ מִשְׂגָּ֣ב לִ֑י וּ֝מָנ֗וֹס בְּי֣וֹם צַר־ לִֽי׃
