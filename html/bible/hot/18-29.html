1וַיֹּ֣סֶף אִ֭יּוֹב שְׂאֵ֥ת מְשָׁל֗וֹ וַיֹּאמַֽר׃
2יִתְּנֵ֥נִי כְיַרְחֵי־ קֶ֑דֶם כִּ֝ימֵ֗י אֱל֣וֹהַּ יִשְׁמְרֵֽנִי׃
3נֵ֭רוֹ עֲלֵ֣י רֹאשִׁ֑י לְ֝אוֹרוֹ אֵ֣לֶךְ חֹֽשֶׁךְ׃
4הָ֭יִיתִי בִּימֵ֣י חָרְפִּ֑י בְּס֥וֹד אֱ֝ל֗וֹהַּ עֲלֵ֣י אָהֳלִֽי׃
5שַׁ֭דַּי עִמָּדִ֑י סְבִ֖יבוֹתַ֣י נְעָרָֽי׃
6הֲלִיכַ֣י בְּחֵמָ֑ה וְצ֥וּר יָצ֥וּק עִ֝מָּדִ֗י פַּלְגֵי־ שָֽׁמֶן׃
7שַׁ֣עַר עֲלֵי־ קָ֑רֶת בָּ֝רְח֗וֹב אָכִ֥ין מוֹשָׁבִֽי׃
8נְעָרִ֣ים וְנֶחְבָּ֑אוּ וִֽ֝ישִׁישִׁים קָ֣מוּ עָמָֽדוּ׃
9עָצְר֣וּ בְמִלִּ֑ים וְ֝כַ֗ף יָשִׂ֥ימוּ לְפִיהֶֽם׃
10נְגִידִ֥ים נֶחְבָּ֑אוּ וּ֝לְשׁוֹנָ֗ם לְחִכָּ֥ם דָּבֵֽקָה׃
11אֹ֣זֶן שָׁ֭מְעָה וַֽתְּאַשְּׁרֵ֑נִי וְעַ֥יִן רָ֝אֲתָ֗ה וַתְּעִידֵֽנִי׃
12אֲ֭מַלֵּט עָנִ֣י מְשַׁוֵּ֑עַ וְ֝יָת֗וֹם וְֽלֹא־ עֹזֵ֥ר לֽוֹ׃
13אֹ֭בֵד עָלַ֣י תָּבֹ֑א וְלֵ֖ב אַלְמָנָ֣ה אַרְנִֽן׃
14לָ֭בַשְׁתִּי וַיִּלְבָּשֵׁ֑נִי כִּמְעִ֥יל וְ֝צָנִ֗יף מִשְׁפָּטִֽי׃
15הָ֭יִיתִי לַֽעִוֵּ֑ר וְרַגְלַ֖יִם לַפִּסֵּ֣חַ אָֽנִי׃
16אָ֭נֹכִֽי לָֽאֶבְיוֹנִ֑ים וְרִ֖ב לֹא־ יָדַ֣עְתִּי אֶחְקְרֵֽהוּ׃
17מְתַלְּע֣וֹת עַוָּ֑ל וּ֝מִשִּׁנָּ֗יו אַשְׁלִ֥יךְ טָֽרֶף׃
18עִם־ קִנִּ֣י אֶגְוָ֑ע וְ֝כַח֗וֹל אַרְבֶּ֥ה יָמִֽים׃
19פָת֣וּחַ אֱלֵי־ מָ֑יִם וְ֝טַ֗ל יָלִ֥ין בִּקְצִירִֽי׃
20חָדָ֣שׁ עִמָּדִ֑י וְ֝קַשְׁתִּ֗י בְּיָדִ֥י תַחֲלִֽיף׃
21שָׁמְע֥וּ וְיִחֵ֑לּוּ וְ֝יִדְּמ֗וּ לְמ֣וֹ עֲצָתִֽי׃
22דְ֭בָרִי לֹ֣א יִשְׁנ֑וּ וְ֝עָלֵ֗ימוֹ תִּטֹּ֥ף מִלָּתִֽי׃
23כַמָּטָ֣ר לִ֑י וּ֝פִיהֶ֗ם פָּעֲר֥וּ לְמַלְקֽוֹשׁ׃
24אֲ֭לֵהֶם לֹ֣א יַאֲמִ֑ינוּ וְא֥וֹר פָּ֝נַ֗י לֹ֣א יַפִּילֽוּן׃
