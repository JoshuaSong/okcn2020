1מִזְמ֖וֹר לְדָוִ֣ד לְהַזְכִּֽיר׃ יְֽהוָ֗ה אַל־ בְּקֶצְפְּךָ֥ תוֹכִיחֵ֑נִי וּֽבַחֲמָתְךָ֥ תְיַסְּרֵֽנִי׃
2חִ֭צֶּיךָ נִ֣חֲתוּ בִ֑י וַתִּנְחַ֖ת עָלַ֣י יָדֶֽךָ׃
3מְתֹ֣ם בִּ֭בְשָׂרִי מִפְּנֵ֣י זַעְמֶ֑ךָ אֵין־ שָׁל֥וֹם בַּ֝עֲצָמַ֗י מִפְּנֵ֥י חַטָּאתִֽי׃
4עֲ֭וֹנֹתַי עָבְר֣וּ רֹאשִׁ֑י כְּמַשָּׂ֥א כָ֝בֵ֗ד יִכְבְּד֥וּ מִמֶּֽנִּי׃
5נָ֭מַקּוּ חַבּוּרֹתָ֑י מִ֝פְּנֵ֗י אִוַּלְתִּֽי׃
6שַׁחֹ֣תִי עַד־ מְאֹ֑ד כָּל־ הַ֝יּ֗וֹם קֹדֵ֥ר הִלָּֽכְתִּי׃
7כְ֭סָלַי מָלְא֣וּ נִקְלֶ֑ה וְאֵ֥ין מְ֝תֹ֗ם בִּבְשָׂרִֽי׃
8וְנִדְכֵּ֣יתִי עַד־ מְאֹ֑ד שָׁ֝אַ֗גְתִּי מִֽנַּהֲמַ֥ת לִבִּֽי׃
9נֶגְדְּךָ֥ כָל־ תַּאֲוָתִ֑י וְ֝אַנְחָתִ֗י מִמְּךָ֥ לֹא־ נִסְתָּֽרָה׃
10סְ֭חַרְחַר עֲזָבַ֣נִי כֹחִ֑י וְֽאוֹר־ עֵינַ֥י גַּם־ הֵ֝֗ם אֵ֣ין אִתִּֽי׃
11וְרֵעַ֗י מִנֶּ֣גֶד נִגְעִ֣י יַעֲמֹ֑דוּ וּ֝קְרוֹבַ֗י מֵרָחֹ֥ק עָמָֽדוּ׃
12מְבַקְשֵׁ֬י נַפְשִׁ֗י וְדֹרְשֵׁ֣י רָ֭עָתִי דִּבְּר֣וּ הַוּ֑וֹת וּ֝מִרְמ֗וֹת כָּל־ הַיּ֥וֹם יֶהְגּֽוּ׃
13כְ֭חֵרֵשׁ לֹ֣א אֶשְׁמָ֑ע וּ֝כְאִלֵּ֗ם לֹ֣א יִפְתַּח־ פִּֽיו׃
14כְּ֭אִישׁ אֲשֶׁ֣ר לֹא־ שֹׁמֵ֑עַ וְאֵ֥ין בְּ֝פִ֗יו תּוֹכָחֽוֹת׃
15לְךָ֣ יְהוָ֣ה הוֹחָ֑לְתִּי אַתָּ֥ה תַ֝עֲנֶ֗ה אֲדֹנָ֥י אֱלֹהָֽי׃
16אָ֭מַרְתִּי פֶּן־ יִשְׂמְחוּ־ לִ֑י בְּמ֥וֹט רַ֝גְלִ֗י עָלַ֥י הִגְדִּֽילוּ׃
17אֲ֭נִי לְצֶ֣לַע נָכ֑וֹן וּמַכְאוֹבִ֖י נֶגְדִּ֣י תָמִֽיד׃
18עֲוֹנִ֥י אַגִּ֑יד אֶ֝דְאַ֗ג מֵֽחַטָּאתִֽי׃
19חַיִּ֣ים עָצֵ֑מוּ וְרַבּ֖וּ שֹׂנְאַ֣י שָֽׁקֶר׃
20רָ֭עָה תַּ֣חַת טוֹבָ֑ה יִ֝שְׂטְנ֗וּנִי תַּ֣חַת [רְדֹופִי־ כ] (רָֽדְפִי־ טֽוֹב׃ ק)
21תַּֽעַזְבֵ֥נִי יְהוָ֑ה אֱ֝לֹהַ֗י אַל־ תִּרְחַ֥ק מִמֶּֽנִּי׃
