1וַיְדַבֵּ֣ר יְהוָ֔ה אֶל־ יְהוֹשֻׁ֖עַ לֵאמֹֽר׃
2אֶל־ בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר תְּנ֤וּ לָכֶם֙ אֶת־ עָרֵ֣י הַמִּקְלָ֔ט אֲשֶׁר־ דִּבַּ֥רְתִּי אֲלֵיכֶ֖ם בְּיַד־ מֹשֶֽׁה׃
3שָׁ֙מָּה֙ רוֹצֵ֔חַ מַכֵּה־ נֶ֥פֶשׁ בִּשְׁגָגָ֖ה בִּבְלִי־ דָ֑עַת וְהָי֤וּ לָכֶם֙ לְמִקְלָ֔ט מִגֹּאֵ֖ל הַדָּֽם׃
4אֶל־ אַחַ֣ת ׀ מֵהֶעָרִ֣ים הָאֵ֗לֶּה וְעָמַד֙ פֶּ֚תַח שַׁ֣עַר הָעִ֔יר וְדִבֶּ֛ר בְּאָזְנֵ֛י זִקְנֵ֥י־ הָעִֽיר הַהִ֖יא אֶת־ דְּבָרָ֑יו וְאָסְפ֨וּ אֹת֤וֹ הָעִ֙ירָה֙ אֲלֵיהֶ֔ם וְנָתְנוּ־ ל֥וֹ מָק֖וֹם וְיָשַׁ֥ב עִמָּֽם׃
5יִרְדֹּ֜ף גֹּאֵ֤ל הַדָּם֙ אַֽחֲרָ֔יו וְלֹֽא־ יַסְגִּ֥רוּ אֶת־ הָרֹצֵ֖חַ בְּיָד֑וֹ כִּ֤י בִבְלִי־ דַ֙עַת֙ הִכָּ֣ה אֶת־ רֵעֵ֔הוּ וְלֹֽא־ שֹׂנֵ֥א ה֛וּא ל֖וֹ מִתְּמ֥וֹל שִׁלְשֽׁוֹם׃
6בָּעִ֣יר הַהִ֗יא עַד־ עָמְד֞וֹ לִפְנֵ֤י הָֽעֵדָה֙ לַמִּשְׁפָּ֔ט עַד־ מוֹת֙ הַכֹּהֵ֣ן הַגָּד֔וֹל אֲשֶׁ֥ר יִהְיֶ֖ה בַּיָּמִ֣ים הָהֵ֑ם אָ֣ז ׀ יָשׁ֣וּב הָרוֹצֵ֗חַ וּבָ֤א אֶל־ עִירוֹ֙ וְאֶל־ בֵּית֔וֹ אֶל־ הָעִ֖יר אֲשֶׁר־ נָ֥ס מִשָּֽׁם׃
7אֶת־ קֶ֤דֶשׁ בַּגָּלִיל֙ בְּהַ֣ר נַפְתָּלִ֔י וְאֶת־ שְׁכֶ֖ם בְּהַ֣ר אֶפְרָ֑יִם וְאֶת־ קִרְיַ֥ת אַרְבַּ֛ע הִ֥יא חֶבְר֖וֹן בְּהַ֥ר יְהוּדָֽה׃
8לְיַרְדֵּ֤ן יְרִיחוֹ֙ מִזְרָ֔חָה נָתְנ֞וּ אֶת־ בֶּ֧צֶר בַּמִּדְבָּ֛ר בַּמִּישֹׁ֖ר מִמַּטֵּ֣ה רְאוּבֵ֑ן וְאֶת־ רָאמֹ֤ת בַּגִּלְעָד֙ מִמַּטֵּה־ גָ֔ד וְאֶת־ [גָּלֹון כ] (גֹּולָ֥ן ק) בַּבָּשָׁ֖ן מִמַּטֵּ֥ה מְנַשֶּֽׁה׃
