1שִׁגָּי֗וֹן לְדָ֫וִ֥ד אֲשֶׁר־ שָׁ֥ר לַיהוָ֑ה עַל־ דִּבְרֵי־ כ֝֗וּשׁ בֶּן־ יְמִינִֽי׃ יְהוָ֣ה אֱ֭לֹהַי בְּךָ֣ חָסִ֑יתִי הוֹשִׁיעֵ֥נִי מִכָּל־ רֹ֝דְפַ֗י וְהַצִּילֵֽנִי׃
2יִטְרֹ֣ף כְּאַרְיֵ֣ה נַפְשִׁ֑י פֹּ֝רֵ֗ק וְאֵ֣ין מַצִּֽיל׃
3אֱ֭לֹהַי אִם־ עָשִׂ֣יתִי זֹ֑את אִֽם־ יֶשׁ־ עָ֥וֶל בְּכַפָּֽי׃
4גָּ֭מַלְתִּי שֽׁוֹלְמִ֥י רָ֑ע וָאֲחַלְּצָ֖ה צוֹרְרִ֣י רֵיקָֽם׃
5אוֹיֵ֨ב ׀ נַפְשִׁ֡י וְיַשֵּׂ֗ג וְיִרְמֹ֣ס לָאָ֣רֶץ חַיָּ֑י וּכְבוֹדִ֓י ׀ לֶעָפָ֖ר יַשְׁכֵּ֣ן סֶֽלָה׃
6יְהוָ֨ה ׀ בְּאַפֶּ֗ךָ הִ֭נָּשֵׂא בְּעַבְר֣וֹת צוֹרְרָ֑י וְע֥וּרָה אֵ֝לַ֗י מִשְׁפָּ֥ט צִוִּֽיתָ׃
7לְ֭אֻמִּים תְּסוֹבְבֶ֑ךָּ וְ֝עָלֶ֗יהָ לַמָּר֥וֹם שֽׁוּבָה׃
8יָדִ֪ין עַ֫מִּ֥ים שָׁפְטֵ֥נִי יְהוָ֑ה כְּצִדְקִ֖י וּכְתֻמִּ֣י עָלָֽי׃
9נָ֬א רַ֨ע ׀ רְשָׁעִים֮ וּתְכוֹנֵ֪ן צַ֫דִּ֥יק וּבֹחֵ֣ן לִ֭בּ֗וֹת וּכְלָי֗וֹת אֱלֹהִ֥ים צַדִּֽיק׃
10עַל־ אֱלֹהִ֑ים מ֝וֹשִׁ֗יעַ יִשְׁרֵי־ לֵֽב׃
11שׁוֹפֵ֣ט צַדִּ֑יק וְ֝אֵ֗ל זֹעֵ֥ם בְּכָל־ יֽוֹם׃
12לֹ֣א יָ֭שׁוּב חַרְבּ֣וֹ יִלְט֑וֹשׁ קַשְׁתּ֥וֹ דָ֝רַ֗ךְ וַֽיְכוֹנְנֶֽהָ׃
13הֵכִ֣ין כְּלֵי־ מָ֑וֶת חִ֝צָּ֗יו לְֽדֹלְקִ֥ים יִפְעָֽל׃
14יְחַבֶּל־ אָ֑וֶן וְהָרָ֥ה עָ֝מָ֗ל וְיָ֣לַד שָֽׁקֶר׃
15כָּ֭רָֽה וַֽיַּחְפְּרֵ֑הוּ וַ֝יִּפֹּ֗ל בְּשַׁ֣חַת יִפְעָֽל׃
16עֲמָל֣וֹ בְרֹאשׁ֑וֹ וְעַ֥ל קָ֝דְקֳד֗וֹ חֲמָס֥וֹ יֵרֵֽד׃
