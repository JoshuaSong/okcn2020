1מִזְמ֡וֹר שִׁיר־ חֲנֻכַּ֖ת הַבַּ֣יִת לְדָוִֽד׃ אֲרוֹמִמְךָ֣ יְ֭הוָה כִּ֣י דִלִּיתָ֑נִי וְלֹא־ שִׂמַּ֖חְתָּ אֹיְבַ֣י לִֽי׃
2אֱלֹהָ֑י שִׁוַּ֥עְתִּי אֵ֝לֶ֗יךָ וַתִּרְפָּאֵֽנִי׃
3הֶֽעֱלִ֣יתָ מִן־ שְׁא֣וֹל נַפְשִׁ֑י חִ֝יִּיתַ֗נִי [מִיֹּורְדֵי־ כ] (מִיָּֽרְדִי־ בֽוֹר׃ ק)
4לַיהוָ֣ה חֲסִידָ֑יו וְ֝הוֹד֗וּ לְזֵ֣כֶר קָדְשֽׁוֹ׃
5רֶ֨גַע ׀ בְּאַפּוֹ֮ חַיִּ֪ים בִּרְצ֫וֹנ֥וֹ בָּ֭עֶרֶב יָלִ֥ין בֶּ֗כִי וְלַבֹּ֥קֶר רִנָּֽה׃
6אָמַ֣רְתִּי בְשַׁלְוִ֑י בַּל־ אֶמּ֥וֹט לְעוֹלָֽם׃
7בִּרְצוֹנְךָ֮ הֶעֱמַ֪דְתָּה לְֽהַרְרִ֫י עֹ֥ז הִסְתַּ֥רְתָּ פָנֶ֗יךָ הָיִ֥יתִי נִבְהָֽל׃
8יְהוָ֣ה אֶקְרָ֑א וְאֶל־ אֲ֝דֹנָ֗י אֶתְחַנָּֽן׃
9בֶּ֥צַע בְּדָמִי֮ בְּרִדְתִּ֪י אֶ֫ל־ שָׁ֥חַת הֲיוֹדְךָ֥ עָפָ֑ר הֲיַגִּ֥יד אֲמִתֶּֽךָ׃
10יְהוָ֥ה וְחָנֵּ֑נִי יְ֝הוָה הֱ‍ֽיֵה־ עֹזֵ֥ר לִֽי׃
11מִסְפְּדִי֮ לְמָח֪וֹל לִ֥י פִּתַּ֥חְתָּ שַׂקִּ֑י וַֽתְּאַזְּרֵ֥נִי שִׂמְחָֽה׃
