1וְאֵ֗לֶּה שְׁמוֹת֙ בְּנֵ֣י יִשְׂרָאֵ֔ל הַבָּאִ֖ים מִצְרָ֑יְמָה אֵ֣ת יַעֲקֹ֔ב אִ֥ישׁ וּבֵית֖וֹ בָּֽאוּ׃
2שִׁמְע֔וֹן לֵוִ֖י וִיהוּדָֽה׃
3זְבוּלֻ֖ן וּבְנְיָמִֽן׃
4וְנַפְתָּלִ֖י גָּ֥ד וְאָשֵֽׁר׃
5כָּל־ נֶ֛פֶשׁ יֹצְאֵ֥י יֶֽרֶךְ־ יַעֲקֹ֖ב שִׁבְעִ֣ים נָ֑פֶשׁ וְיוֹסֵ֖ף הָיָ֥ה בְמִצְרָֽיִם׃
6יוֹסֵף֙ וְכָל־ אֶחָ֔יו וְכֹ֖ל הַדּ֥וֹר הַהֽוּא׃
7יִשְׂרָאֵ֗ל פָּר֧וּ וַֽיִּשְׁרְצ֛וּ וַיִּרְבּ֥וּ וַיַּֽעַצְמ֖וּ בִּמְאֹ֣ד מְאֹ֑ד וַתִּמָּלֵ֥א הָאָ֖רֶץ אֹתָֽם׃ פ
8מֶֽלֶךְ־ חָדָ֖שׁ עַל־ מִצְרָ֑יִם אֲשֶׁ֥ר לֹֽא־ יָדַ֖ע אֶת־ יוֹסֵֽף׃
9אֶל־ עַמּ֑וֹ הִנֵּ֗ה עַ֚ם בְּנֵ֣י יִשְׂרָאֵ֔ל רַ֥ב וְעָצ֖וּם מִמֶּֽנּוּ׃
10נִֽתְחַכְּמָ֖ה ל֑וֹ פֶּן־ יִרְבֶּ֗ה וְהָיָ֞ה כִּֽי־ תִקְרֶ֤אנָה מִלְחָמָה֙ וְנוֹסַ֤ף גַּם־ הוּא֙ עַל־ שֹׂ֣נְאֵ֔ינוּ וְנִלְחַם־ בָּ֖נוּ וְעָלָ֥ה מִן־ הָאָֽרֶץ׃
11עָלָיו֙ שָׂרֵ֣י מִסִּ֔ים לְמַ֥עַן עַנֹּת֖וֹ בְּסִבְלֹתָ֑ם וַיִּ֜בֶן עָרֵ֤י מִסְכְּנוֹת֙ לְפַרְעֹ֔ה אֶת־ פִּתֹ֖ם וְאֶת־ רַעַמְסֵֽס׃
12יְעַנּ֣וּ אֹת֔וֹ כֵּ֥ן יִרְבֶּ֖ה וְכֵ֣ן יִפְרֹ֑ץ וַיָּקֻ֕צוּ מִפְּנֵ֖י בְּנֵ֥י יִשְׂרָאֵֽל׃
13מִצְרַ֛יִם אֶת־ בְּנֵ֥י יִשְׂרָאֵ֖ל בְּפָֽרֶךְ׃
14אֶת־ חַיֵּיהֶ֜ם בַּעֲבֹדָ֣ה קָשָׁ֗ה בְּחֹ֙מֶר֙ וּבִלְבֵנִ֔ים וּבְכָל־ עֲבֹדָ֖ה בַּשָּׂדֶ֑ה אֵ֚ת כָּל־ עֲבֹ֣דָתָ֔ם אֲשֶׁר־ עָבְד֥וּ בָהֶ֖ם בְּפָֽרֶךְ׃
15מֶ֣לֶךְ מִצְרַ֔יִם לַֽמְיַלְּדֹ֖ת הָֽעִבְרִיֹּ֑ת אֲשֶׁ֨ר שֵׁ֤ם הָֽאַחַת֙ שִׁפְרָ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖ית פּוּעָֽה׃
16בְּיַלֶּדְכֶן֙ אֶת־ הָֽעִבְרִיּ֔וֹת וּרְאִיתֶ֖ן עַל־ הָאָבְנָ֑יִם אִם־ בֵּ֥ן הוּא֙ וַהֲמִתֶּ֣ן אֹת֔וֹ וְאִם־ בַּ֥ת הִ֖יא וָחָֽיָה׃
17הַֽמְיַלְּדֹת֙ אֶת־ הָ֣אֱלֹהִ֔ים וְלֹ֣א עָשׂ֔וּ כַּאֲשֶׁ֛ר דִּבֶּ֥ר אֲלֵיהֶ֖ן מֶ֣לֶךְ מִצְרָ֑יִם וַתְּחַיֶּ֖יןָ אֶת־ הַיְלָדִֽים׃
18מֶֽלֶךְ־ מִצְרַ֙יִם֙ לַֽמְיַלְּדֹ֔ת וַיֹּ֣אמֶר לָהֶ֔ן מַדּ֥וּעַ עֲשִׂיתֶ֖ן הַדָּבָ֣ר הַזֶּ֑ה וַתְּחַיֶּ֖יןָ אֶת־ הַיְלָדִֽים׃
19הַֽמְיַלְּדֹת֙ אֶל־ פַּרְעֹ֔ה כִּ֣י לֹ֧א כַנָּשִׁ֛ים הַמִּצְרִיֹּ֖ת הָֽעִבְרִיֹּ֑ת כִּֽי־ חָי֣וֹת הֵ֔נָּה בְּטֶ֨רֶם תָּב֧וֹא אֲלֵהֶ֛ן הַמְיַלֶּ֖דֶת וְיָלָֽדוּ׃
20אֱלֹהִ֖ים לַֽמְיַלְּדֹ֑ת וַיִּ֧רֶב הָעָ֛ם וַיַּֽעַצְמ֖וּ מְאֹֽד׃
21כִּֽי־ יָֽרְא֥וּ הַֽמְיַלְּדֹ֖ת אֶת־ הָאֱלֹהִ֑ים וַיַּ֥עַשׂ לָהֶ֖ם בָּתִּֽים׃
