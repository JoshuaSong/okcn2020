1חַכְמ֣וֹת נָ֭שִׁים בָּנְתָ֣ה בֵיתָ֑הּ וְ֝אִוֶּ֗לֶת בְּיָדֶ֥יהָ תֶהֶרְסֶֽנּוּ׃
2בְּ֭יָשְׁרוֹ יְרֵ֣א יְהוָ֑ה וּנְל֖וֹז דְּרָכָ֣יו בּוֹזֵֽהוּ׃
3אֱ֭וִיל חֹ֣טֶר גַּאֲוָ֑ה וְשִׂפְתֵ֥י חֲ֝כָמִ֗ים תִּשְׁמוּרֵֽם׃
4אֲ֭לָפִים אֵב֣וּס בָּ֑ר וְרָב־ תְּ֝בוּא֗וֹת בְּכֹ֣חַ שֽׁוֹר׃
5אֱ֭מוּנִים לֹ֣א יְכַזֵּ֑ב וְיָפִ֥יחַ כְּ֝זָבִ֗ים עֵ֣ד שָֽׁקֶר׃
6לֵ֣ץ חָכְמָ֣ה וָאָ֑יִן וְדַ֖עַת לְנָב֣וֹן נָקָֽל׃
7מִ֭נֶּגֶד לְאִ֣ישׁ כְּסִ֑יל וּבַל־ יָ֝דַ֗עְתָּ שִׂפְתֵי־ דָֽעַת׃
8עָ֭רוּם הָבִ֣ין דַּרְכּ֑וֹ וְאִוֶּ֖לֶת כְּסִילִ֣ים מִרְמָֽה׃
9יָלִ֣יץ אָשָׁ֑ם וּבֵ֖ין יְשָׁרִ֣ים רָצֽוֹן׃
10י֭וֹדֵעַ מָרַּ֣ת נַפְשׁ֑וֹ וּ֝בְשִׂמְחָת֗וֹ לֹא־ יִתְעָ֥רַב זָֽר׃
11רְ֭שָׁעִים יִשָּׁמֵ֑ד וְאֹ֖הֶל יְשָׁרִ֣ים יַפְרִֽיחַ׃
12דֶּ֣רֶךְ יָ֭שָׁר לִפְנֵי־ אִ֑ישׁ וְ֝אַחֲרִיתָ֗הּ דַּרְכֵי־ מָֽוֶת׃
13בִּשְׂח֥וֹק יִכְאַב־ לֵ֑ב וְאַחֲרִיתָ֖הּ שִׂמְחָ֣ה תוּגָֽה׃
14יִ֭שְׂבַּע ס֣וּג לֵ֑ב וּ֝מֵעָלָ֗יו אִ֣ישׁ טֽוֹב׃
15יַאֲמִ֣ין לְכָל־ דָּבָ֑ר וְ֝עָר֗וּם יָבִ֥ין לַאֲשֻׁרֽוֹ׃
16יָ֭רֵא וְסָ֣ר מֵרָ֑ע וּ֝כְסִ֗יל מִתְעַבֵּ֥ר וּבוֹטֵֽחַ׃
17אַ֭פַּיִם יַעֲשֶׂ֣ה אִוֶּ֑לֶת וְאִ֥ישׁ מְ֝זִמּ֗וֹת יִשָּׂנֵֽא׃
18פְתָאיִ֣ם אִוֶּ֑לֶת וַֽ֝עֲרוּמִ֗ים יַכְתִּ֥רוּ דָֽעַת׃
19רָ֭עִים לִפְנֵ֣י טוֹבִ֑ים וּ֝רְשָׁעִ֗ים עַֽל־ שַׁעֲרֵ֥י צַדִּֽיק׃
20לְ֭רֵעֵהוּ יִשָּׂ֣נֵא רָ֑שׁ וְאֹהֲבֵ֖י עָשִׁ֣יר רַבִּֽים׃
21לְרֵעֵ֥הוּ חוֹטֵ֑א וּמְחוֹנֵ֖ן [עֲנָיִים כ] (עֲנָוִ֣ים ק) אַשְׁרָֽיו׃
22יִ֭תְעוּ חֹ֣רְשֵׁי רָ֑ע וְחֶ֥סֶד וֶ֝אֱמֶ֗ת חֹ֣רְשֵׁי טֽוֹב׃
23עֶ֭צֶב יִהְיֶ֣ה מוֹתָ֑ר וּדְבַר־ שְׂ֝פָתַ֗יִם אַךְ־ לְמַחְסֽוֹר׃
24חֲכָמִ֣ים עָשְׁרָ֑ם אִוֶּ֖לֶת כְּסִילִ֣ים אִוֶּֽלֶת׃
25נְ֭פָשׁוֹת עֵ֣ד אֱמֶ֑ת וְיָפִ֖חַ כְּזָבִ֣ים מִרְמָֽה׃
26יְ֭הוָה מִבְטַח־ עֹ֑ז וּ֝לְבָנָ֗יו יִהְיֶ֥ה מַחְסֶֽה׃
27יְ֭הוָה מְק֣וֹר חַיִּ֑ים לָ֝ס֗וּר מִמֹּ֥קְשֵׁי מָֽוֶת׃
28עָ֥ם הַדְרַת־ מֶ֑לֶךְ וּבְאֶ֥פֶס לְ֝אֹ֗ם מְחִתַּ֥ת רָזֽוֹן׃
29אַ֭פַּיִם רַב־ תְּבוּנָ֑ה וּקְצַר־ ר֝֗וּחַ מֵרִ֥ים אִוֶּֽלֶת׃
30בְ֭שָׂרִים לֵ֣ב מַרְפֵּ֑א וּרְקַ֖ב עֲצָמ֣וֹת קִנְאָֽה׃
31דָּ֭ל חֵרֵ֣ף עֹשֵׂ֑הוּ וּ֝מְכַבְּד֗וֹ חֹנֵ֥ן אֶבְיֽוֹן׃
32יִדָּחֶ֣ה רָשָׁ֑ע וְחֹסֶ֖ה בְמוֹת֣וֹ צַדִּֽיק׃
33נָ֭בוֹן תָּנ֣וּחַ חָכְמָ֑ה וּבְקֶ֥רֶב כְּ֝סִילִ֗ים תִּוָּדֵֽעַ׃
34תְרֽוֹמֵֽם־ גּ֑וֹי וְחֶ֖סֶד לְאֻמִּ֣ים חַטָּֽאת׃
