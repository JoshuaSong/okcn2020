1בִּשְׁנַ֨ת עֶשְׂרִ֤ים וָשֶׁ֙בַע֙ שָׁנָ֔ה לְיָרָבְעָ֖ם מֶ֣לֶךְ יִשְׂרָאֵ֑ל מָלַ֛ךְ עֲזַרְיָ֥ה בֶן־ אֲמַצְיָ֖ה מֶ֥לֶךְ יְהוּדָֽה׃
2שֵׁ֨שׁ עֶשְׂרֵ֤ה שָׁנָה֙ הָיָ֣ה בְמָלְכ֔וֹ וַחֲמִשִּׁ֤ים וּשְׁתַּ֙יִם֙ שָׁנָ֔ה מָלַ֖ךְ בִּירוּשָׁלִָ֑ם וְשֵׁ֣ם אִמּ֔וֹ יְכָלְיָ֖הוּ מִירוּשָׁלִָֽם׃
3הַיָּשָׁ֖ר בְּעֵינֵ֣י יְהוָ֑ה כְּכֹ֥ל אֲשֶׁר־ עָשָׂ֖ה אֲמַצְיָ֥הוּ אָבִֽיו׃
4הַבָּמ֖וֹת לֹא־ סָ֑רוּ ע֥וֹד הָעָ֛ם מְזַבְּחִ֥ים וּֽמְקַטְּרִ֖ים בַּבָּמֽוֹת׃
5יְהוָ֜ה אֶת־ הַמֶּ֗לֶךְ וַיְהִ֤י מְצֹרָע֙ עַד־ י֣וֹם מֹת֔וֹ וַיֵּ֖שֶׁב בְּבֵ֣ית הַחָפְשִׁ֑ית וְיוֹתָ֤ם בֶּן־ הַמֶּ֙לֶךְ֙ עַל־ הַבַּ֔יִת שֹׁפֵ֖ט אֶת־ עַ֥ם הָאָֽרֶץ׃
6דִּבְרֵ֥י עֲזַרְיָ֖הוּ וְכָל־ אֲשֶׁ֣ר עָשָׂ֑ה הֲלֹא־ הֵ֣ם כְּתוּבִ֗ים עַל־ סֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לְמַלְכֵ֥י יְהוּדָֽה׃
7עֲזַרְיָה֙ עִם־ אֲבֹתָ֔יו וַיִּקְבְּר֥וּ אֹת֛וֹ עִם־ אֲבֹתָ֖יו בְּעִ֣יר דָּוִ֑ד וַיִּמְלֹ֛ךְ יוֹתָ֥ם בְּנ֖וֹ תַּחְתָּֽיו׃ פ
8שְׁלֹשִׁ֤ים וּשְׁמֹנֶה֙ שָׁנָ֔ה לַעֲזַרְיָ֖הוּ מֶ֣לֶךְ יְהוּדָ֑ה מָ֠לַךְ זְכַרְיָ֨הוּ בֶן־ יָרָבְעָ֧ם עַל־ יִשְׂרָאֵ֛ל בְּשֹׁמְר֖וֹן שִׁשָּׁ֥ה חֳדָשִֽׁים׃
9הָרַע֙ בְּעֵינֵ֣י יְהוָ֔ה כַּאֲשֶׁ֥ר עָשׂ֖וּ אֲבֹתָ֑יו לֹ֣א סָ֗ר מֵֽחַטֹּאות֙ יָרָבְעָ֣ם בֶּן־ נְבָ֔ט אֲשֶׁ֥ר הֶחֱטִ֖יא אֶת־ יִשְׂרָאֵֽל׃
10עָלָיו֙ שַׁלֻּ֣ם בֶּן־ יָבֵ֔שׁ וַיַּכֵּ֥הוּ קָֽבָלְ־ עָ֖ם וַיְמִיתֵ֑הוּ וַיִּמְלֹ֖ךְ תַּחְתָּֽיו׃
11דִּבְרֵ֣י זְכַרְיָ֑ה הִנָּ֣ם כְּתוּבִ֗ים עַל־ סֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לְמַלְכֵ֥י יִשְׂרָאֵֽל׃
12דְבַר־ יְהוָ֗ה אֲשֶׁ֨ר דִּבֶּ֤ר אֶל־ יֵהוּא֙ לֵאמֹ֔ר בְּנֵ֣י רְבִיעִ֔ים יֵשְׁב֥וּ לְךָ֖ עַל־ כִּסֵּ֣א יִשְׂרָאֵ֑ל וַֽיְהִי־ כֵֽן׃ פ
13בֶּן־ יָבֵישׁ֙ מָלַ֔ךְ בִּשְׁנַ֨ת שְׁלֹשִׁ֤ים וָתֵ֙שַׁע֙ שָׁנָ֔ה לְעֻזִיָּ֖ה מֶ֣לֶךְ יְהוּדָ֑ה וַיִּמְלֹ֥ךְ יֶֽרַח־ יָמִ֖ים בְּשֹׁמְרֽוֹן׃
14מְנַחֵ֨ם בֶּן־ גָּדִ֜י מִתִּרְצָ֗ה וַיָּבֹא֙ שֹׁמְר֔וֹן וַיַּ֛ךְ אֶת־ שַׁלּ֥וּם בֶּן־ יָבֵ֖ישׁ בְּשֹׁמְר֑וֹן וַיְמִיתֵ֖הוּ וַיִּמְלֹ֥ךְ תַּחְתָּֽיו׃
15דִּבְרֵ֣י שַׁלּ֔וּם וְקִשְׁר֖וֹ אֲשֶׁ֣ר קָשָׁ֑ר הִנָּ֣ם כְּתֻבִ֗ים עַל־ סֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לְמַלְכֵ֥י יִשְׂרָאֵֽל׃ ס
16יַכֶּֽה־ מְ֠נַחֵם אֶת־ תִּפְסַ֨ח וְאֶת־ כָּל־ אֲשֶׁר־ בָּ֤הּ וְאֶת־ גְּבוּלֶ֙יהָ֙ מִתִּרְצָ֔ה כִּ֛י לֹ֥א פָתַ֖ח וַיַּ֑ךְ אֵ֛ת כָּל־ הֶהָ֥רוֹתֶ֖יהָ בִּקֵּֽעַ׃ פ
17שְׁלֹשִׁ֤ים וָתֵ֙שַׁע֙ שָׁנָ֔ה לַעֲזַרְיָ֖ה מֶ֣לֶךְ יְהוּדָ֑ה מָ֠לַךְ מְנַחֵ֨ם בֶּן־ גָּדִ֧י עַל־ יִשְׂרָאֵ֛ל עֶ֥שֶׂר שָׁנִ֖ים בְּשֹׁמְרֽוֹן׃
18הָרַ֖ע בְּעֵינֵ֣י יְהוָ֑ה לֹ֣א סָ֠ר מֵעַ֨ל חַטֹּ֜אות יָרָבְעָ֧ם בֶּן־ נְבָ֛ט אֲשֶׁר־ הֶחֱטִ֥יא אֶת־ יִשְׂרָאֵ֖ל כָּל־ יָמָֽיו׃
19פ֤וּל מֶֽלֶךְ־ אַשּׁוּר֙ עַל־ הָאָ֔רֶץ וַיִּתֵּ֤ן מְנַחֵם֙ לְפ֔וּל אֶ֖לֶף כִּכַּר־ כָּ֑סֶף לִהְי֤וֹת יָדָיו֙ אִתּ֔וֹ לְהַחֲזִ֥יק הַמַּמְלָכָ֖ה בְּיָדֽוֹ׃
20מְנַחֵ֨ם אֶת־ הַכֶּ֜סֶף עַל־ יִשְׂרָאֵ֗ל עַ֚ל כָּל־ גִּבּוֹרֵ֣י הַחַ֔יִל לָתֵת֙ לְמֶ֣לֶךְ אַשּׁ֔וּר חֲמִשִּׁ֧ים שְׁקָלִ֛ים כֶּ֖סֶף לְאִ֣ישׁ אֶחָ֑ד וַיָּ֙שָׁב֙ מֶ֣לֶךְ אַשּׁ֔וּר וְלֹא־ עָ֥מַד שָׁ֖ם בָּאָֽרֶץ׃
21דִּבְרֵ֥י מְנַחֵ֖ם וְכָל־ אֲשֶׁ֣ר עָשָׂ֑ה הֲלוֹא־ הֵ֣ם כְּתוּבִ֗ים עַל־ סֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לְמַלְכֵ֥י יִשְׂרָאֵֽל׃
22מְנַחֵ֖ם עִם־ אֲבֹתָ֑יו וַיִּמְלֹ֛ךְ פְּקַחְיָ֥ה בְנ֖וֹ תַּחְתָּֽיו׃ פ
23חֲמִשִּׁ֣ים שָׁנָ֔ה לַעֲזַרְיָ֖ה מֶ֣לֶךְ יְהוּדָ֑ה מָ֠לַךְ פְּקַֽחְיָ֨ה בֶן־ מְנַחֵ֧ם עַל־ יִשְׂרָאֵ֛ל בְּשֹׁמְר֖וֹן שְׁנָתָֽיִם׃
24הָרַ֖ע בְּעֵינֵ֣י יְהוָ֑ה לֹ֣א סָ֗ר מֵֽחַטֹּאות֙ יָרָבְעָ֣ם בֶּן־ נְבָ֔ט אֲשֶׁ֥ר הֶחֱטִ֖יא אֶת־ יִשְׂרָאֵֽל׃
25עָלָיו֩ פֶּ֨קַח בֶּן־ רְמַלְיָ֜הוּ שָׁלִישׁ֗וֹ וַיַּכֵּ֨הוּ בְשֹׁמְר֜וֹן בְּאַרְמ֤וֹן בֵּית־ [מֶלֶךְ כ] (הַמֶּ֙לֶךְ֙ ק) אֶת־ אַרְגֹּ֣ב וְאֶת־ הָאַרְיֵ֔ה וְעִמּ֛וֹ חֲמִשִּׁ֥ים אִ֖ישׁ מִבְּנֵ֣י גִלְעָדִ֑ים וַיְמִיתֵ֖הוּ וַיִּמְלֹ֥ךְ תַּחְתָּֽיו׃
26דִּבְרֵ֥י פְקַחְיָ֖ה וְכָל־ אֲשֶׁ֣ר עָשָׂ֑ה הִנָּ֣ם כְּתוּבִ֗ים עַל־ סֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לְמַלְכֵ֥י יִשְׂרָאֵֽל׃ פ
27חֲמִשִּׁ֤ים וּשְׁתַּ֙יִם֙ שָׁנָ֔ה לַעֲזַרְיָ֖ה מֶ֣לֶךְ יְהוּדָ֑ה מָ֠לַךְ פֶּ֣קַח בֶּן־ רְמַלְיָ֧הוּ עַל־ יִשְׂרָאֵ֛ל בְּשֹׁמְר֖וֹן עֶשְׂרִ֥ים שָׁנָֽה׃
28הָרַ֖ע בְּעֵינֵ֣י יְהוָ֑ה לֹ֣א סָ֗ר מִן־ חַטֹּאות֙ יָרָבְעָ֣ם בֶּן־ נְבָ֔ט אֲשֶׁ֥ר הֶחֱטִ֖יא אֶת־ יִשְׂרָאֵֽל׃
29פֶּ֣קַח מֶֽלֶךְ־ יִשְׂרָאֵ֗ל בָּא֮ תִּגְלַ֣ת פִּלְאֶסֶר֮ מֶ֣לֶךְ אַשּׁוּר֒ וַיִּקַּ֣ח אֶת־ עִיּ֡וֹן וְאֶת־ אָבֵ֣ל בֵּֽית־ מַעֲכָ֡ה וְאֶת־ יָ֠נוֹחַ וְאֶת־ קֶ֨דֶשׁ וְאֶת־ חָצ֤וֹר וְאֶת־ הַגִּלְעָד֙ וְאֶת־ הַגָּלִ֔ילָה כֹּ֖ל אֶ֣רֶץ נַפְתָּלִ֑י וַיַּגְלֵ֖ם אַשּֽׁוּרָה׃
30קֶ֜שֶׁר הוֹשֵׁ֣עַ בֶּן־ אֵלָ֗ה עַל־ פֶּ֙קַח֙ בֶּן־ רְמַלְיָ֔הוּ וַיַּכֵּ֙הוּ֙ וַיְמִיתֵ֔הוּ וַיִּמְלֹ֖ךְ תַּחְתָּ֑יו בִּשְׁנַ֣ת עֶשְׂרִ֔ים לְיוֹתָ֖ם בֶּן־ עֻזִיָּֽה׃
31דִּבְרֵי־ פֶ֖קַח וְכָל־ אֲשֶׁ֣ר עָשָׂ֑ה הִנָּ֣ם כְּתוּבִ֗ים עַל־ סֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לְמַלְכֵ֥י יִשְׂרָאֵֽל׃ פ
32שְׁתַּ֔יִם לְפֶ֥קַח בֶּן־ רְמַלְיָ֖הוּ מֶ֣לֶךְ יִשְׂרָאֵ֑ל מָלַ֛ךְ יוֹתָ֥ם בֶּן־ עֻזִיָּ֖הוּ מֶ֥לֶךְ יְהוּדָֽה׃
33עֶשְׂרִ֨ים וְחָמֵ֤שׁ שָׁנָה֙ הָיָ֣ה בְמָלְכ֔וֹ וְשֵׁשׁ־ עֶשְׂרֵ֣ה שָׁנָ֔ה מָלַ֖ךְ בִּירוּשָׁלִָ֑ם וְשֵׁ֣ם אִמּ֔וֹ יְרוּשָׁ֖א בַּת־ צָדֽוֹק׃
34הַיָּשָׁ֖ר בְּעֵינֵ֣י יְהוָ֑ה כְּכֹ֧ל אֲשֶׁר־ עָשָׂ֛ה עֻזִיָּ֥הוּ אָבִ֖יו עָשָֽׂה׃
35הַבָּמוֹת֙ לֹ֣א סָ֔רוּ ע֗וֹד הָעָ֛ם מְזַבְּחִ֥ים וּֽמְקַטְּרִ֖ים בַּבָּמ֑וֹת ה֗וּא בָּנָ֛ה אֶת־ שַׁ֥עַר בֵּית־ יְהוָ֖ה הָעֶלְיֽוֹן׃
36דִּבְרֵ֥י יוֹתָ֖ם אֲשֶׁ֣ר עָשָׂ֑ה הֲלֹא־ הֵ֣ם כְּתוּבִ֗ים עַל־ סֵ֛פֶר דִּבְרֵ֥י הַיָּמִ֖ים לְמַלְכֵ֥י יְהוּדָֽה׃
37הָהֵ֔ם הֵחֵ֣ל יְהוָ֗ה לְהַשְׁלִ֙יחַ֙ בִּֽיהוּדָ֔ה רְצִ֖ין מֶ֣לֶךְ אֲרָ֑ם וְאֵ֖ת פֶּ֥קַח בֶּן־ רְמַלְיָֽהוּ׃
