1כַּשֶּׁ֤לֶג ׀ בַּקַּ֗יִץ וְכַמָּטָ֥ר בַּקָּצִ֑יר כֵּ֤ן לֹא־ נָאוֶ֖ה לִכְסִ֣יל כָּבֽוֹד׃
2לָ֭נוּד כַּדְּר֣וֹר לָע֑וּף כֵּ֥ן קִֽלְלַ֥ת חִ֝נָּ֗ם [לֹא כ] (לֹ֣ו ק) תָבֹֽא׃
3לַ֭סּוּס מֶ֣תֶג לַחֲמ֑וֹר וְ֝שֵׁ֗בֶט לְגֵ֣ו כְּסִילִֽים׃
4תַּ֣עַן כְּ֭סִיל כְּאִוַּלְתּ֑וֹ פֶּֽן־ תִּשְׁוֶה־ לּ֥וֹ גַם־ אָֽתָּה׃
5כְ֭סִיל כְּאִוַּלְתּ֑וֹ פֶּן־ יִהְיֶ֖ה חָכָ֣ם בְּעֵינָֽיו׃
6רַ֭גְלַיִם חָמָ֣ס שֹׁתֶ֑ה שֹׁלֵ֖חַ דְּבָרִ֣ים בְּיַד־ כְּסִֽיל׃
7שֹׁ֭קַיִם מִפִּסֵּ֑חַ וּ֝מָשָׁ֗ל בְּפִ֣י כְסִילִֽים׃
8אֶ֭בֶן בְּמַרְגֵּמָ֑ה כֵּן־ נוֹתֵ֖ן לִכְסִ֣יל כָּבֽוֹד׃
9עָלָ֣ה בְיַד־ שִׁכּ֑וֹר וּ֝מָשָׁ֗ל בְּפִ֣י כְסִילִֽים׃
10מְחֽוֹלֵֽל־ כֹּ֑ל וְשֹׂכֵ֥ר כְּ֝סִ֗יל וְשֹׂכֵ֥ר עֹבְרִֽים׃
11שָׁ֣ב עַל־ קֵא֑וֹ כְּ֝סִ֗יל שׁוֹנֶ֥ה בְאִוַּלְתּֽוֹ׃
12אִ֭ישׁ חָכָ֣ם בְּעֵינָ֑יו תִּקְוָ֖ה לִכְסִ֣יל מִמֶּֽנּוּ׃
13עָ֭צֵל שַׁ֣חַל בַּדָּ֑רֶךְ אֲ֝רִ֗י בֵּ֣ין הָרְחֹבֽוֹת׃
14תִּסּ֣וֹב עַל־ צִירָ֑הּ וְ֝עָצֵ֗ל עַל־ מִטָּתֽוֹ׃
15עָצֵ֣ל יָ֭דוֹ בַּצַּלָּ֑חַת נִ֝לְאָ֗ה לַֽהֲשִׁיבָ֥הּ אֶל־ פִּֽיו׃
16עָצֵ֣ל בְּעֵינָ֑יו מִ֝שִּׁבְעָ֗ה מְשִׁ֣יבֵי טָֽעַם׃
17בְּאָזְנֵי־ כָ֑לֶב עֹבֵ֥ר מִ֝תְעַבֵּ֗ר עַל־ רִ֥יב לֹּֽא־ לֽוֹ׃
18הַיֹּרֶ֥ה זִקִּ֗ים חִצִּ֥ים וָמָֽוֶת׃
19אִ֭ישׁ רִמָּ֣ה אֶת־ רֵעֵ֑הוּ וְ֝אָמַ֗ר הֲ‍ֽלֹא־ מְשַׂחֵ֥ק אָֽנִי׃
20עֵ֭צִים תִּכְבֶּה־ אֵ֑שׁ וּבְאֵ֥ין נִ֝רְגָּ֗ן יִשְׁתֹּ֥ק מָדֽוֹן׃
21לְ֭גֶחָלִים וְעֵצִ֣ים לְאֵ֑שׁ וְאִ֥ישׁ [מִדֹונִים כ] (מִ֝דְיָנִ֗ים ק) לְחַרְחַר־ רִֽיב׃ פ
22נִ֭רְגָּן כְּמִֽתְלַהֲמִ֑ים וְ֝הֵ֗ם יָרְד֥וּ חַדְרֵי־ בָֽטֶן׃
23סִ֭יגִים מְצֻפֶּ֣ה עַל־ חָ֑רֶשׂ שְׂפָתַ֖יִם דֹּלְקִ֣ים וְלֶב־ רָֽע׃
24כ] (בִּ֭שְׂפָתָיו ק) יִנָּכֵ֣ר שׂוֹנֵ֑א וּ֝בְקִרְבּ֗וֹ יָשִׁ֥ית מִרְמָֽה׃
25יְחַנֵּ֣ן ק֭וֹלוֹ אַל־ תַּֽאֲמֶן־ בּ֑וֹ כִּ֤י שֶׁ֖בַע תּוֹעֵב֣וֹת בְּלִבּֽוֹ׃
26שִׂ֭נְאָה בְּמַשָּׁא֑וֹן תִּגָּלֶ֖ה רָעָת֣וֹ בְקָהָֽל׃
27שַּׁ֭חַת בָּ֣הּ יִפֹּ֑ל וְגֹ֥לֵ֥ל אֶ֝בֶן אֵלָ֥יו תָּשֽׁוּב׃
