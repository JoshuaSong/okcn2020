1וַיְדַבֵּ֥ר יְהוָ֖ה אֶל־ מֹשֶׁ֥ה לֵּאמֹֽר׃
2קָרָ֣אתִֽי בְשֵׁ֑ם בְּצַלְאֵ֛ל בֶּן־ אוּרִ֥י בֶן־ ח֖וּר לְמַטֵּ֥ה יְהוּדָֽה׃
3אֹת֖וֹ ר֣וּחַ אֱלֹהִ֑ים בְּחָכְמָ֛ה וּבִתְבוּנָ֥ה וּבְדַ֖עַת וּבְכָל־ מְלָאכָֽה׃
4מַחֲשָׁבֹ֑ת לַעֲשׂ֛וֹת בַּזָּהָ֥ב וּבַכֶּ֖סֶף וּבַנְּחֹֽשֶׁת׃
5אֶ֛בֶן לְמַלֹּ֖את וּבַחֲרֹ֣שֶׁת עֵ֑ץ לַעֲשׂ֖וֹת בְּכָל־ מְלָאכָֽה׃
6הִנֵּ֧ה נָתַ֣תִּי אִתּ֗וֹ אֵ֣ת אָהֳלִיאָ֞ב בֶּן־ אֲחִֽיסָמָךְ֙ לְמַטֵּה־ דָ֔ן וּבְלֵ֥ב כָּל־ חֲכַם־ לֵ֖ב נָתַ֣תִּי חָכְמָ֑ה וְעָשׂ֕וּ אֵ֖ת כָּל־ אֲשֶׁ֥ר צִוִּיתִֽךָ׃
7אֹ֣הֶל מוֹעֵ֗ד וְאֶת־ הָֽאָרֹן֙ לָֽעֵדֻ֔ת וְאֶת־ הַכַּפֹּ֖רֶת אֲשֶׁ֣ר עָלָ֑יו וְאֵ֖ת כָּל־ כְּלֵ֥י הָאֹֽהֶל׃
8הַשֻּׁלְחָן֙ וְאֶת־ כֵּלָ֔יו וְאֶת־ הַמְּנֹרָ֥ה הַטְּהֹרָ֖ה וְאֶת־ כָּל־ כֵּלֶ֑יהָ וְאֵ֖ת מִזְבַּ֥ח הַקְּטֹֽרֶת׃
9מִזְבַּ֥ח הָעֹלָ֖ה וְאֶת־ כָּל־ כֵּלָ֑יו וְאֶת־ הַכִּיּ֖וֹר וְאֶת־ כַּנּֽוֹ׃
10בִּגְדֵ֣י הַשְּׂרָ֑ד וְאֶת־ בִּגְדֵ֤י הַקֹּ֙דֶשׁ֙ לְאַהֲרֹ֣ן הַכֹּהֵ֔ן וְאֶת־ בִּגְדֵ֥י בָנָ֖יו לְכַהֵֽן׃
11שֶׁ֧מֶן הַמִּשְׁחָ֛ה וְאֶת־ קְטֹ֥רֶת הַסַּמִּ֖ים לַקֹּ֑דֶשׁ כְּכֹ֥ל אֲשֶׁר־ צִוִּיתִ֖ךָ יַעֲשֽׂוּ׃ פ
12יְהוָ֖ה אֶל־ מֹשֶׁ֥ה לֵּאמֹֽר׃
13דַּבֵּ֨ר אֶל־ בְּנֵ֤י יִשְׂרָאֵל֙ לֵאמֹ֔ר אַ֥ךְ אֶת־ שַׁבְּתֹתַ֖י תִּשְׁמֹ֑רוּ כִּי֩ א֨וֹת הִ֜וא בֵּינִ֤י וּבֵֽינֵיכֶם֙ לְדֹרֹ֣תֵיכֶ֔ם לָדַ֕עַת כִּ֛י אֲנִ֥י יְהוָ֖ה מְקַדִּשְׁכֶֽם׃
14אֶת־ הַשַּׁבָּ֔ת כִּ֛י קֹ֥דֶשׁ הִ֖וא לָכֶ֑ם מְחַֽלְלֶ֙יהָ֙ מ֣וֹת יוּמָ֔ת כִּ֗י כָּל־ הָעֹשֶׂ֥ה בָהּ֙ מְלָאכָ֔ה וְנִכְרְתָ֛ה הַנֶּ֥פֶשׁ הַהִ֖וא מִקֶּ֥רֶב עַמֶּֽיהָ׃
15יָמִים֮ יֵעָשֶׂ֣ה מְלָאכָה֒ וּבַיּ֣וֹם הַשְּׁבִיעִ֗י שַׁבַּ֧ת שַׁבָּת֛וֹן קֹ֖דֶשׁ לַיהוָ֑ה כָּל־ הָעֹשֶׂ֧ה מְלָאכָ֛ה בְּי֥וֹם הַשַּׁבָּ֖ת מ֥וֹת יוּמָֽת׃
16בְנֵֽי־ יִשְׂרָאֵ֖ל אֶת־ הַשַּׁבָּ֑ת לַעֲשׂ֧וֹת אֶת־ הַשַּׁבָּ֛ת לְדֹרֹתָ֖ם בְּרִ֥ית עוֹלָֽם׃
17וּבֵין֙ בְּנֵ֣י יִשְׂרָאֵ֔ל א֥וֹת הִ֖וא לְעֹלָ֑ם כִּי־ שֵׁ֣שֶׁת יָמִ֗ים עָשָׂ֤ה יְהוָה֙ אֶת־ הַשָּׁמַ֣יִם וְאֶת־ הָאָ֔רֶץ וּבַיּוֹם֙ הַשְּׁבִיעִ֔י שָׁבַ֖ת וַיִּנָּפַֽשׁ׃ ס
