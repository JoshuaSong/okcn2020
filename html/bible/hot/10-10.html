1וַֽיְהִי֙ אַֽחֲרֵי־ כֵ֔ן וַיָּ֕מָת מֶ֖לֶךְ בְּנֵ֣י עַמּ֑וֹן וַיִּמְלֹ֛ךְ חָנ֥וּן בְּנ֖וֹ תַּחְתָּֽיו׃
2דָּוִ֜ד אֶעֱשֶׂה־ חֶ֣סֶד ׀ עִם־ חָנ֣וּן בֶּן־ נָחָ֗שׁ כַּאֲשֶׁר֩ עָשָׂ֨ה אָבִ֤יו עִמָּדִי֙ חֶ֔סֶד וַיִּשְׁלַ֨ח דָּוִ֧ד לְנַחֲמ֛וֹ בְּיַד־ עֲבָדָ֖יו אֶל־ אָבִ֑יו וַיָּבֹ֙אוּ֙ עַבְדֵ֣י דָוִ֔ד אֶ֖רֶץ בְּנֵ֥י עַמּֽוֹן׃
3שָׂרֵ֨י בְנֵֽי־ עַמּ֜וֹן אֶל־ חָנ֣וּן אֲדֹֽנֵיהֶ֗ם הַֽמְכַבֵּ֨ד דָּוִ֤ד אֶת־ אָבִ֙יךָ֙ בְּעֵינֶ֔יךָ כִּֽי־ שָׁלַ֥ח לְךָ֖ מְנַֽחֲמִ֑ים הֲ֠לוֹא בַּעֲב֞וּר חֲק֤וֹר אֶת־ הָעִיר֙ וּלְרַגְּלָ֣הּ וּלְהָפְכָ֔הּ שָׁלַ֥ח דָּוִ֛ד אֶת־ עֲבָדָ֖יו אֵלֶֽיךָ׃
4חָנ֜וּן אֶת־ עַבְדֵ֣י דָוִ֗ד וַיְגַלַּח֙ אֶת־ חֲצִ֣י זְקָנָ֔ם וַיִּכְרֹ֧ת אֶת־ מַדְוֵיהֶ֛ם בַּחֵ֖צִי עַ֣ד שְׁתֽוֹתֵיהֶ֑ם וַֽיְשַׁלְּחֵֽם׃
5לְדָוִד֙ וַיִּשְׁלַ֣ח לִקְרָאתָ֔ם כִּֽי־ הָי֥וּ הָאֲנָשִׁ֖ים נִכְלָמִ֣ים מְאֹ֑ד וַיֹּ֤אמֶר הַמֶּ֙לֶךְ֙ שְׁב֣וּ בִֽירֵח֔וֹ עַד־ יְצַמַּ֥ח זְקַנְכֶ֖ם וְשַׁבְתֶּֽם׃
6בְּנֵ֣י עַמּ֔וֹן כִּ֥י נִבְאֲשׁ֖וּ בְּדָוִ֑ד וַיִּשְׁלְח֣וּ בְנֵֽי־ עַמּ֡וֹן וַיִּשְׂכְּרוּ֩ אֶת־ אֲרַ֨ם בֵּית־ רְח֜וֹב וְאֶת־ אֲרַ֣ם צוֹבָ֗א עֶשְׂרִ֥ים אֶ֙לֶף֙ רַגְלִ֔י וְאֶת־ מֶ֤לֶךְ מַֽעֲכָה֙ אֶ֣לֶף אִ֔ישׁ וְאִ֣ישׁ ט֔וֹב שְׁנֵים־ עָשָׂ֥ר אֶ֖לֶף אִֽישׁ׃
7דָּוִ֑ד וַיִּשְׁלַח֙ אֶת־ יוֹאָ֔ב וְאֵ֥ת כָּל־ הַצָּבָ֖א הַגִּבֹּרִֽים׃
8בְּנֵ֣י עַמּ֔וֹן וַיַּעַרְכ֥וּ מִלְחָמָ֖ה פֶּ֣תַח הַשָּׁ֑עַר וַאֲרַ֨ם צוֹבָ֤א וּרְחוֹב֙ וְאִֽישׁ־ ט֣וֹב וּמַֽעֲכָ֔ה לְבַדָּ֖ם בַּשָּׂדֶֽה׃
9יוֹאָ֗ב כִּֽי־ הָיְתָ֤ה אֵלָיו֙ פְּנֵ֣י הַמִּלְחָמָ֔ה מִפָּנִ֖ים וּמֵֽאָח֑וֹר וַיִּבְחַ֗ר מִכֹּל֙ בְּחוּרֵ֣י [בְּיִשְׂרָאֵל כ] (יִשְׂרָאֵ֔ל ק) וַֽיַּעֲרֹ֖ךְ לִקְרַ֥את אֲרָֽם׃
10יֶ֣תֶר הָעָ֔ם נָתַ֕ן בְּיַ֖ד אַבְשַׁ֣י אָחִ֑יו וַֽיַּעֲרֹ֕ךְ לִקְרַ֖את בְּנֵ֥י עַמּֽוֹן׃
11אִם־ תֶּחֱזַ֤ק אֲרָם֙ מִמֶּ֔נִּי וְהָיִ֥תָה לִּ֖י לִֽישׁוּעָ֑ה וְאִם־ בְּנֵ֤י עַמּוֹן֙ יֶחֱזְק֣וּ מִמְּךָ֔ וְהָלַכְתִּ֖י לְהוֹשִׁ֥יעַֽ לָֽךְ׃
12וְנִתְחַזַּק֙ בְּעַד־ עַמֵּ֔נוּ וּבְעַ֖ד עָרֵ֣י אֱלֹהֵ֑ינוּ וַֽיהוָ֔ה יַעֲשֶׂ֥ה הַטּ֖וֹב בְּעֵינָֽיו׃
13יוֹאָ֗ב וְהָעָם֙ אֲשֶׁ֣ר עִמּ֔וֹ לַמִּלְחָמָ֖ה בַּֽאֲרָ֑ם וַיָּנֻ֖סוּ מִפָּנָֽיו׃
14עַמּ֤וֹן רָאוּ֙ כִּי־ נָ֣ס אֲרָ֔ם וַיָּנֻ֙סוּ֙ מִפְּנֵ֣י אֲבִישַׁ֔י וַיָּבֹ֖אוּ הָעִ֑יר וַיָּ֣שָׁב יוֹאָ֗ב מֵעַל֙ בְּנֵ֣י עַמּ֔וֹן וַיָּבֹ֖א יְרוּשָׁלִָֽם׃
15אֲרָ֔ם כִּ֥י נִגַּ֖ף לִפְנֵ֣י יִשְׂרָאֵ֑ל וַיֵּאָסְפ֖וּ יָֽחַד׃
16הֲדַדְעֶ֗זֶר וַיֹּצֵ֤א אֶת־ אֲרָם֙ אֲשֶׁר֙ מֵעֵ֣בֶר הַנָּהָ֔ר וַיָּבֹ֖אוּ חֵילָ֑ם וְשׁוֹבַ֛ךְ שַׂר־ צְבָ֥א הֲדַדְעֶ֖זֶר לִפְנֵיהֶֽם׃
17לְדָוִ֗ד ס וַיֶּאֱסֹ֤ף אֶת־ כָּל־ יִשְׂרָאֵל֙ וַיַּעֲבֹ֣ר אֶת־ הַיַּרְדֵּ֔ן וַיָּבֹ֖א חֵלָ֑אמָה וַיַּעַרְכ֤וּ אֲרָם֙ לִקְרַ֣את דָּוִ֔ד וַיִּלָּחֲמ֖וּ עִמּֽוֹ׃
18אֲרָם֮ מִפְּנֵ֣י יִשְׂרָאֵל֒ וַיַּהֲרֹ֨ג דָּוִ֜ד מֵאֲרָ֗ם שְׁבַ֤ע מֵאוֹת֙ רֶ֔כֶב וְאַרְבָּעִ֥ים אֶ֖לֶף פָּרָשִׁ֑ים וְאֵ֨ת שׁוֹבַ֧ךְ שַׂר־ צְבָא֛וֹ הִכָּ֖ה וַיָּ֥מָת שָֽׁם׃
