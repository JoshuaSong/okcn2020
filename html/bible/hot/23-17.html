1מַשָּׂ֖א דַּמָּ֑שֶׂק הִנֵּ֤ה דַמֶּ֙שֶׂק֙ מוּסָ֣ר מֵעִ֔יר וְהָיְתָ֖ה מְעִ֥י מַפָּלָֽה׃
2עָרֵ֣י עֲרֹעֵ֑ר לַעֲדָרִ֣ים תִּֽהְיֶ֔ינָה וְרָבְצ֖וּ וְאֵ֥ין מַחֲרִֽיד׃
3מִבְצָר֙ מֵֽאֶפְרַ֔יִם וּמַמְלָכָ֥ה מִדַּמֶּ֖שֶׂק וּשְׁאָ֣ר אֲרָ֑ם כִּכְב֤וֹד בְּנֵֽי־ יִשְׂרָאֵל֙ יִֽהְי֔וּ נְאֻ֖ם יְהוָ֥ה צְבָאֽוֹת׃ ס
4בַּיּ֣וֹם הַה֔וּא יִדַּ֖ל כְּב֣וֹד יַעֲקֹ֑ב וּמִשְׁמַ֥ן בְּשָׂר֖וֹ יֵרָזֶֽה׃
5כֶּֽאֱסֹף֙ קָצִ֣יר קָמָ֔ה וּזְרֹע֖וֹ שִׁבֳּלִ֣ים יִקְצ֑וֹר וְהָיָ֛ה כִּמְלַקֵּ֥ט שִׁבֳּלִ֖ים בְּעֵ֥מֶק רְפָאִֽים׃
6בּ֤וֹ עֽוֹלֵלֹת֙ כְּנֹ֣קֶף זַ֔יִת שְׁנַ֧יִם שְׁלֹשָׁ֛ה גַּרְגְּרִ֖ים בְּרֹ֣אשׁ אָמִ֑יר אַרְבָּעָ֣ה חֲמִשָּׁ֗ה בִּסְעִפֶ֙יהָ֙ פֹּֽרִיָּ֔ה נְאֻם־ יְהוָ֖ה אֱלֹהֵ֥י יִשְׂרָאֵֽל׃ ס
7הַה֔וּא יִשְׁעֶ֥ה הָאָדָ֖ם עַל־ עֹשֵׂ֑הוּ וְעֵינָ֕יו אֶל־ קְד֥וֹשׁ יִשְׂרָאֵ֖ל תִּרְאֶֽינָה׃
8יִשְׁעֶ֔ה אֶל־ הַֽמִּזְבְּח֖וֹת מַעֲשֵׂ֣ה יָדָ֑יו וַאֲשֶׁ֨ר עָשׂ֤וּ אֶצְבְּעֹתָיו֙ לֹ֣א יִרְאֶ֔ה וְהָאֲשֵׁרִ֖ים וְהָחַמָּנִֽים׃
9הַה֜וּא יִהְי֣וּ ׀ עָרֵ֣י מָעֻזּ֗וֹ כַּעֲזוּבַ֤ת הַחֹ֙רֶשׁ֙ וְהָ֣אָמִ֔יר אֲשֶׁ֣ר עָזְב֔וּ מִפְּנֵ֖י בְּנֵ֣י יִשְׂרָאֵ֑ל וְהָיְתָ֖ה שְׁמָמָֽה׃
10שָׁכַ֙חַתְּ֙ אֱלֹהֵ֣י יִשְׁעֵ֔ךְ וְצ֥וּר מָעֻזֵּ֖ךְ לֹ֣א זָכָ֑רְתְּ עַל־ כֵּ֗ן תִּטְּעִי֙ נִטְעֵ֣י נַעֲמָנִ֔ים וּזְמֹ֥רַת זָ֖ר תִּזְרָעֶֽנּוּ׃
11נִטְעֵךְ֙ תְּשַׂגְשֵׂ֔גִי וּבַבֹּ֖קֶר זַרְעֵ֣ךְ תַּפְרִ֑יחִי נֵ֥ד קָצִ֛יר בְּי֥וֹם נַחֲלָ֖ה וּכְאֵ֥ב אָנֽוּשׁ׃ ס
12הֲמוֹן֙ עַמִּ֣ים רַבִּ֔ים כַּהֲמ֥וֹת יַמִּ֖ים יֶהֱמָי֑וּן וּשְׁא֣וֹן לְאֻמִּ֔ים כִּשְׁא֛וֹן מַ֥יִם כַּבִּירִ֖ים יִשָּׁאֽוּן׃
13כִּשְׁא֞וֹן מַ֤יִם רַבִּים֙ יִשָּׁא֔וּן וְגָ֥עַר בּ֖וֹ וְנָ֣ס מִמֶּרְחָ֑ק וְרֻדַּ֗ף כְּמֹ֤ץ הָרִים֙ לִפְנֵי־ ר֔וּחַ וּכְגַלְגַּ֖ל לִפְנֵ֥י סוּפָֽה׃
