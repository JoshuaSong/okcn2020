1וַיָּ֣מָת שְׁמוּאֵ֔ל וַיִּקָּבְצ֤וּ כָל־ יִשְׂרָאֵל֙ וַיִּסְפְּדוּ־ ל֔וֹ וַיִּקְבְּרֻ֥הוּ בְּבֵית֖וֹ בָּרָמָ֑ה וַיָּ֣קָם דָּוִ֔ד וַיֵּ֖רֶד אֶל־ מִדְבַּ֥ר פָּארָֽן׃ ס
2בְּמָע֜וֹן וּמַעֲשֵׂ֣הוּ בַכַּרְמֶ֗ל וְהָאִישׁ֙ גָּד֣וֹל מְאֹ֔ד וְל֛וֹ צֹ֥אן שְׁלֹֽשֶׁת־ אֲלָפִ֖ים וְאֶ֣לֶף עִזִּ֑ים וַיְהִ֛י בִּגְזֹ֥ז אֶת־ צֹאנ֖וֹ בַּכַּרְמֶֽל׃
3הָאִישׁ֙ נָבָ֔ל וְשֵׁ֥ם אִשְׁתּ֖וֹ אֲבִגָ֑יִל וְהָאִשָּׁ֤ה טֽוֹבַת־ שֶׂ֙כֶל֙ וִ֣יפַת תֹּ֔אַר וְהָאִ֥ישׁ קָשֶׁ֛ה וְרַ֥ע מַעֲלָלִ֖ים וְה֥וּא [כָלֶבֹו כ] (כָלִבִּֽי׃ ק)
4דָּוִ֖ד בַּמִּדְבָּ֑ר כִּֽי־ גֹזֵ֥ז נָבָ֖ל אֶת־ צֹאנֽוֹ׃
5דָּוִ֖ד עֲשָׂרָ֣ה נְעָרִ֑ים וַיֹּ֨אמֶר דָּוִ֜ד לַנְּעָרִ֗ים עֲל֤וּ כַרְמֶ֙לָה֙ וּבָאתֶ֣ם אֶל־ נָבָ֔ל וּשְׁאֶלְתֶּם־ ל֥וֹ בִשְׁמִ֖י לְשָׁלֽוֹם׃
6כֹּ֖ה לֶחָ֑י וְאַתָּ֤ה שָׁלוֹם֙ וּבֵיתְךָ֣ שָׁל֔וֹם וְכֹ֥ל אֲשֶׁר־ לְךָ֖ שָׁלֽוֹם׃
7שָׁמַ֔עְתִּי כִּ֥י גֹזְזִ֖ים לָ֑ךְ עַתָּ֗ה הָרֹעִ֤ים אֲשֶׁר־ לְךָ֙ הָי֣וּ עִמָּ֔נוּ לֹ֣א הֶכְלַמְנ֗וּם וְלֹֽא־ נִפְקַ֤ד לָהֶם֙ מְא֔וּמָה כָּל־ יְמֵ֖י הֱיוֹתָ֥ם בַּכַּרְמֶֽל׃
8אֶת־ נְעָרֶ֜יךָ וְיַגִּ֣ידוּ לָ֗ךְ וְיִמְצְא֨וּ הַנְּעָרִ֥ים חֵן֙ בְּעֵינֶ֔יךָ כִּֽי־ עַל־ י֥וֹם ט֖וֹב בָּ֑נוּ תְּנָה־ נָּ֗א אֵת֩ אֲשֶׁ֨ר תִּמְצָ֤א יָֽדְךָ֙ לַעֲבָדֶ֔יךָ וּלְבִנְךָ֖ לְדָוִֽד׃
9נַעֲרֵ֣י דָוִ֔ד וַיְדַבְּר֧וּ אֶל־ נָבָ֛ל כְּכָל־ הַדְּבָרִ֥ים הָאֵ֖לֶּה בְּשֵׁ֣ם דָּוִ֑ד וַיָּנֽוּחוּ׃
10נָבָ֜ל אֶת־ עַבְדֵ֤י דָוִד֙ וַיֹּ֔אמֶר מִ֥י דָוִ֖ד וּמִ֣י בֶן־ יִשָׁ֑י הַיּוֹם֙ רַבּ֣וּ עֲבָדִ֔ים הַמִּתְפָּ֣רְצִ֔ים אִ֖ישׁ מִפְּנֵ֥י אֲדֹנָֽיו׃
11אֶת־ לַחְמִי֙ וְאֶת־ מֵימַ֔י וְאֵת֙ טִבְחָתִ֔י אֲשֶׁ֥ר טָבַ֖חְתִּי לְגֹֽזְזָ֑י וְנָֽתַתִּי֙ לַֽאֲנָשִׁ֔ים אֲשֶׁר֙ לֹ֣א יָדַ֔עְתִּי אֵ֥י מִזֶּ֖ה הֵֽמָּה׃
12נַעֲרֵֽי־ דָוִ֖ד לְדַרְכָּ֑ם וַיָּשֻׁ֙בוּ֙ וַיָּבֹ֔אוּ וַיַּגִּ֣דוּ ל֔וֹ כְּכֹ֖ל הַדְּבָרִ֥ים הָאֵֽלֶּה׃
13דָּוִ֨ד לַאֲנָשָׁ֜יו חִגְר֣וּ ׀ אִ֣ישׁ אֶת־ חַרְבּ֗וֹ וַֽיַּחְגְּרוּ֙ אִ֣ישׁ אֶת־ חַרְבּ֔וֹ וַיַּחְגֹּ֥ר גַּם־ דָּוִ֖ד אֶת־ חַרְבּ֑וֹ וַֽיַּעֲל֣וּ ׀ אַחֲרֵ֣י דָוִ֗ד כְּאַרְבַּ֤ע מֵאוֹת֙ אִ֔ישׁ וּמָאתַ֖יִם יָשְׁב֥וּ עַל־ הַכֵּלִֽים׃
14אֵ֣שֶׁת נָבָ֔ל הִגִּ֧יד נַֽעַר־ אֶחָ֛ד מֵהַנְּעָרִ֖ים לֵאמֹ֑ר הִנֵּ֣ה שָׁלַח֩ דָּוִ֨ד מַלְאָכִ֧ים ׀ מֵֽהַמִּדְבָּ֛ר לְבָרֵ֥ךְ אֶת־ אֲדֹנֵ֖ינוּ וַיָּ֥עַט בָּהֶֽם׃
15טֹבִ֥ים לָ֖נוּ מְאֹ֑ד וְלֹ֤א הָכְלַ֙מְנוּ֙ וְלֹֽא־ פָקַ֣דְנוּ מְא֔וּמָה כָּל־ יְמֵי֙ הִתְהַלַּ֣כְנוּ אִתָּ֔ם בִּֽהְיוֹתֵ֖נוּ בַּשָּׂדֶֽה׃
16הָי֣וּ עָלֵ֔ינוּ גַּם־ לַ֖יְלָה גַּם־ יוֹמָ֑ם כָּל־ יְמֵ֛י הֱיוֹתֵ֥נוּ עִמָּ֖ם רֹעִ֥ים הַצֹּֽאן׃
17דְּעִ֤י וּרְאִי֙ מַֽה־ תַּעֲשִׂ֔י כִּֽי־ כָלְתָ֧ה הָרָעָ֛ה אֶל־ אֲדֹנֵ֖ינוּ וְעַ֣ל כָּל־ בֵּית֑וֹ וְהוּא֙ בֶּן־ בְּלִיַּ֔עַל מִדַּבֵּ֖ר אֵלָֽיו׃
18[אֲבֹוגַיִל כ] (אֲבִיגַ֡יִל ק) וַתִּקַּח֩ מָאתַ֨יִם לֶ֜חֶם וּשְׁנַ֣יִם נִבְלֵי־ יַ֗יִן וְחָמֵ֨שׁ צֹ֤אן [עֲשָׂוֹות כ] (עֲשׂוּיֹת֙ ק) וְחָמֵ֤שׁ סְאִים֙ קָלִ֔י וּמֵאָ֥ה צִמֻּקִ֖ים וּמָאתַ֣יִם דְּבֵלִ֑ים וַתָּ֖שֶׂם עַל־ הַחֲמֹרִֽים׃
19לִנְעָרֶ֙יהָ֙ עִבְר֣וּ לְפָנַ֔י הִנְנִ֖י אַחֲרֵיכֶ֣ם בָּאָ֑ה וּלְאִישָׁ֥הּ נָבָ֖ל לֹ֥א הִגִּֽידָה׃
20הִ֣יא ׀ רֹכֶ֣בֶת עַֽל־ הַחֲמ֗וֹר וְיֹרֶ֙דֶת֙ בְּסֵ֣תֶר הָהָ֔ר וְהִנֵּ֤ה דָוִד֙ וַאֲנָשָׁ֔יו יֹרְדִ֖ים לִקְרָאתָ֑הּ וַתִּפְגֹּ֖שׁ אֹתָֽם׃
21אָמַ֗ר אַךְ֩ לַשֶּׁ֨קֶר שָׁמַ֜רְתִּי אֶֽת־ כָּל־ אֲשֶׁ֤ר לָזֶה֙ בַּמִּדְבָּ֔ר וְלֹא־ נִפְקַ֥ד מִכָּל־ אֲשֶׁר־ ל֖וֹ מְא֑וּמָה וַיָּֽשֶׁב־ לִ֥י רָעָ֖ה תַּ֥חַת טוֹבָֽה׃
22יַעֲשֶׂ֧ה אֱלֹהִ֛ים לְאֹיְבֵ֥י דָוִ֖ד וְכֹ֣ה יֹסִ֑יף אִם־ אַשְׁאִ֧יר מִכָּל־ אֲשֶׁר־ ל֛וֹ עַד־ הַבֹּ֖קֶר מַשְׁתִּ֥ין בְּקִֽיר׃
23אֲבִיגַ֙יִל֙ אֶת־ דָּוִ֔ד וַתְּמַהֵ֕ר וַתֵּ֖רֶד מֵעַ֣ל הַחֲמ֑וֹר וַתִּפֹּ֞ל לְאַפֵּ֤י דָוִד֙ עַל־ פָּנֶ֔יהָ וַתִּשְׁתַּ֖חוּ אָֽרֶץ׃
24עַל־ רַגְלָ֔יו וַתֹּ֕אמֶר בִּי־ אֲנִ֥י אֲדֹנִ֖י הֶֽעָוֹ֑ן וּֽתְדַבֶּר־ נָ֤א אֲמָֽתְךָ֙ בְּאָזְנֶ֔יךָ וּשְׁמַ֕ע אֵ֖ת דִּבְרֵ֥י אֲמָתֶֽךָ׃
25נָ֣א יָשִׂ֣ים אֲדֹנִ֣י ׀ אֶת־ לִבּ֡וֹ אֶל־ אִישׁ֩ הַבְּלִיַּ֨עַל הַזֶּ֜ה עַל־ נָבָ֗ל כִּ֤י כִשְׁמוֹ֙ כֶּן־ ה֔וּא נָבָ֣ל שְׁמ֔וֹ וּנְבָלָ֖ה עִמּ֑וֹ וַֽאֲנִי֙ אֲמָ֣תְךָ֔ לֹ֥א רָאִ֛יתִי אֶת־ נַעֲרֵ֥י אֲדֹנִ֖י אֲשֶׁ֥ר שָׁלָֽחְתָּ׃
26אֲדֹנִ֗י חַי־ יְהוָ֤ה וְחֵֽי־ נַפְשְׁךָ֙ אֲשֶׁ֨ר מְנָעֲךָ֤ יְהוָה֙ מִבּ֣וֹא בְדָמִ֔ים וְהוֹשֵׁ֥עַ יָדְךָ֖ לָ֑ךְ וְעַתָּ֗ה יִֽהְי֤וּ כְנָבָל֙ אֹיְבֶ֔יךָ וְהַֽמְבַקְשִׁ֥ים אֶל־ אֲדֹנִ֖י רָעָֽה׃
27הַבְּרָכָ֣ה הַזֹּ֔את אֲשֶׁר־ הֵבִ֥יא שִׁפְחָתְךָ֖ לַֽאדֹנִ֑י וְנִתְּנָה֙ לַנְּעָרִ֔ים הַמִּֽתְהַלְּכִ֖ים בְּרַגְלֵ֥י אֲדֹנִֽי׃
28נָ֖א לְפֶ֣שַׁע אֲמָתֶ֑ךָ כִּ֣י עָשֹֽׂה־ יַעֲשֶׂה֩ יְהוָ֨ה לַֽאדֹנִ֜י בַּ֣יִת נֶאֱמָ֗ן כִּי־ מִלְחֲמ֤וֹת יְהוָה֙ אֲדֹנִ֣י נִלְחָ֔ם וְרָעָ֛ה לֹא־ תִמָּצֵ֥א בְךָ֖ מִיָּמֶֽיךָ׃
29אָדָם֙ לִרְדָפְךָ֔ וּלְבַקֵּ֖שׁ אֶת־ נַפְשֶׁ֑ךָ וְֽהָיְתָה֩ נֶ֨פֶשׁ אֲדֹנִ֜י צְרוּרָ֣ה ׀ בִּצְר֣וֹר הַחַיִּ֗ים אֵ֚ת יְהוָ֣ה אֱלֹהֶ֔יךָ וְאֵ֨ת נֶ֤פֶשׁ אֹיְבֶ֙יךָ֙ יְקַלְּעֶ֔נָּה בְּת֖וֹךְ כַּ֥ף הַקָּֽלַע׃
30כִּֽי־ יַעֲשֶׂ֤ה יְהוָה֙ לַֽאדֹנִ֔י כְּכֹ֛ל אֲשֶׁר־ דִּבֶּ֥ר אֶת־ הַטּוֹבָ֖ה עָלֶ֑יךָ וְצִוְּךָ֥ לְנָגִ֖יד עַל־ יִשְׂרָאֵֽל׃
31תִהְיֶ֣ה זֹ֣את ׀ לְךָ֡ לְפוּקָה֩ וּלְמִכְשׁ֨וֹל לֵ֜ב לַאדֹנִ֗י וְלִשְׁפָּךְ־ דָּם֙ חִנָּ֔ם וּלְהוֹשִׁ֥יעַ אֲדֹנִ֖י ל֑וֹ וְהֵיטִ֤ב יְהוָה֙ לַֽאדֹנִ֔י וְזָכַרְתָּ֖ אֶת־ אֲמָתֶֽךָ׃ ס
32דָּוִ֖ד לַאֲבִיגַ֑ל בָּר֤וּךְ יְהוָה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֧ר שְׁלָחֵ֛ךְ הַיּ֥וֹם הַזֶּ֖ה לִקְרָאתִֽי׃
33טַעְמֵ֖ךְ וּבְרוּכָ֣ה אָ֑תְּ אֲשֶׁ֨ר כְּלִתִ֜נִי הַיּ֤וֹם הַזֶּה֙ מִבּ֣וֹא בְדָמִ֔ים וְהֹשֵׁ֥עַ יָדִ֖י לִֽי׃
34חַי־ יְהוָה֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל אֲשֶׁ֣ר מְנָעַ֔נִי מֵהָרַ֖ע אֹתָ֑ךְ כִּ֣י ׀ לוּלֵ֣י מִהַ֗רְתְּ [וַתָּבֹאתִי כ] (וַתָּבֹאת֙ ק) לִקְרָאתִ֔י כִּ֣י אִם־ נוֹתַ֧ר לְנָבָ֛ל עַד־ א֥וֹר הַבֹּ֖קֶר מַשְׁתִּ֥ין בְּקִֽיר׃
35דָּוִד֙ מִיָּדָ֔הּ אֵ֥ת אֲשֶׁר־ הֵבִ֖יאָה ל֑וֹ וְלָ֣הּ אָמַ֗ר עֲלִ֤י לְשָׁלוֹם֙ לְבֵיתֵ֔ךְ רְאִי֙ שָׁמַ֣עְתִּי בְקוֹלֵ֔ךְ וָאֶשָּׂ֖א פָּנָֽיִךְ׃
36אֲבִיגַ֣יִל ׀ אֶל־ נָבָ֡ל וְהִנֵּה־ לוֹ֩ מִשְׁתֶּ֨ה בְּבֵית֜וֹ כְּמִשְׁתֵּ֣ה הַמֶּ֗לֶךְ וְלֵ֤ב נָבָל֙ ט֣וֹב עָלָ֔יו וְה֥וּא שִׁכֹּ֖ר עַד־ מְאֹ֑ד וְלֹֽא־ הִגִּ֣ידָה לּ֗וֹ דָּבָ֥ר קָטֹ֛ן וְגָד֖וֹל עַד־ א֥וֹר הַבֹּֽקֶר׃
37בַבֹּ֗קֶר בְּצֵ֤את הַיַּ֙יִן֙ מִנָּבָ֔ל וַתַּגֶּד־ ל֣וֹ אִשְׁתּ֔וֹ אֶת־ הַדְּבָרִ֖ים הָאֵ֑לֶּה וַיָּ֤מָת לִבּוֹ֙ בְּקִרְבּ֔וֹ וְה֖וּא הָיָ֥ה לְאָֽבֶן׃
38כַּעֲשֶׂ֣רֶת הַיָּמִ֑ים וַיִּגֹּ֧ף יְהוָ֛ה אֶת־ נָבָ֖ל וַיָּמֹֽת׃
39דָּוִד֮ כִּ֣י מֵ֣ת נָבָל֒ וַיֹּ֡אמֶר בָּר֣וּךְ יְהוָ֡ה אֲשֶׁ֣ר רָב֩ אֶת־ רִ֨יב חֶרְפָּתִ֜י מִיַּ֣ד נָבָ֗ל וְאֶת־ עַבְדּוֹ֙ חָשַׂ֣ךְ מֵֽרָעָ֔ה וְאֵת֙ רָעַ֣ת נָבָ֔ל הֵשִׁ֥יב יְהוָ֖ה בְּרֹאשׁ֑וֹ וַיִּשְׁלַ֤ח דָּוִד֙ וַיְדַבֵּ֣ר בַּאֲבִיגַ֔יִל לְקַחְתָּ֥הּ ל֖וֹ לְאִשָּֽׁה׃
40עַבְדֵ֥י דָוִ֛ד אֶל־ אֲבִיגַ֖יִל הַכַּרְמֶ֑לָה וַיְדַבְּר֤וּ אֵלֶ֙יהָ֙ לֵאמֹ֔ר דָּוִד֙ שְׁלָחָ֣נוּ אֵלַ֔יִךְ לְקַחְתֵּ֥ךְ ל֖וֹ לְאִשָּֽׁה׃
41וַתִּשְׁתַּ֥חוּ אַפַּ֖יִם אָ֑רְצָה וַתֹּ֗אמֶר הִנֵּ֤ה אֲמָֽתְךָ֙ לְשִׁפְחָ֔ה לִרְחֹ֕ץ רַגְלֵ֖י עַבְדֵ֥י אֲדֹנִֽי׃
42וַתָּ֣קָם אֲבִיגַ֗יִל וַתִּרְכַּב֙ עַֽל־ הַחֲמ֔וֹר וְחָמֵשׁ֙ נַעֲרֹתֶ֔יהָ הַהֹלְכ֖וֹת לְרַגְלָ֑הּ וַתֵּ֗לֶךְ אַֽחֲרֵי֙ מַלְאֲכֵ֣י דָוִ֔ד וַתְּהִי־ ל֖וֹ לְאִשָּֽׁה׃
