1מִזְמ֗וֹר לְאָ֫סָ֥ף אֵ֤ל ׀ אֱ‍ֽלֹהִ֡ים יְֽהוָ֗ה דִּבֶּ֥ר וַיִּקְרָא־ אָ֑רֶץ מִמִּזְרַח־ שֶׁ֝֗מֶשׁ עַד־ מְבֹאֽוֹ׃
2מִכְלַל־ יֹ֗פִי אֱלֹהִ֥ים הוֹפִֽיעַ׃
3אֱלֹהֵ֗ינוּ וְֽאַל־ יֶ֫חֱרַ֥שׁ אֵשׁ־ לְפָנָ֥יו תֹּאכֵ֑ל וּ֝סְבִיבָ֗יו נִשְׂעֲרָ֥ה מְאֹֽד׃
4אֶל־ הַשָּׁמַ֣יִם מֵעָ֑ל וְאֶל־ הָ֝אָ֗רֶץ לָדִ֥ין עַמּֽוֹ׃
5לִ֥י חֲסִידָ֑י כֹּרְתֵ֖י בְרִיתִ֣י עֲלֵי־ זָֽבַח׃
6שָׁמַ֣יִם צִדְק֑וֹ כִּֽי־ אֱלֹהִ֓ים ׀ שֹׁפֵ֖ט ה֣וּא סֶֽלָה׃
7עַמִּ֨י ׀ וַאֲדַבֵּ֗רָה יִ֭שְׂרָאֵל וְאָעִ֣ידָה בָּ֑ךְ אֱלֹהִ֖ים אֱלֹהֶ֣יךָ אָנֹֽכִי׃
8עַל־ זְ֭בָחֶיךָ אוֹכִיחֶ֑ךָ וְעוֹלֹתֶ֖יךָ לְנֶגְדִּ֣י תָמִֽיד׃
9אֶקַּ֣ח מִבֵּיתְךָ֣ פָ֑ר מִ֝מִּכְלְאֹתֶ֗יךָ עַתּוּדִֽים׃
10לִ֥י כָל־ חַיְתוֹ־ יָ֑עַר בְּ֝הֵמ֗וֹת בְּהַרְרֵי־ אָֽלֶף׃
11כָּל־ ע֣וֹף הָרִ֑ים וְזִ֥יז שָׂ֝דַ֗י עִמָּדִֽי׃
12אֶ֭רְעַב לֹא־ אֹ֣מַר לָ֑ךְ כִּי־ לִ֥י תֵ֝בֵ֗ל וּמְלֹאָֽהּ׃
13בְּשַׂ֣ר אַבִּירִ֑ים וְדַ֖ם עַתּוּדִ֣ים אֶשְׁתֶּֽה׃
14לֵאלֹהִ֣ים תּוֹדָ֑ה וְשַׁלֵּ֖ם לְעֶלְי֣וֹן נְדָרֶֽיךָ׃
15בְּי֣וֹם צָרָ֑ה אֲ֝חַלֶּצְךָ֗ וּֽתְכַבְּדֵֽנִי׃
16אָ֘מַ֤ר אֱלֹהִ֗ים מַה־ לְּ֭ךָ לְסַפֵּ֣ר חֻקָּ֑י וַתִּשָּׂ֖א בְרִיתִ֣י עֲלֵי־ פִֽיךָ׃
17שָׂנֵ֣אתָ מוּסָ֑ר וַתַּשְׁלֵ֖ךְ דְּבָרַ֣י אַחֲרֶֽיךָ׃
18רָאִ֣יתָ גַ֭נָּב וַתִּ֣רֶץ עִמּ֑וֹ וְעִ֖ם מְנָאֲפִ֣ים חֶלְקֶֽךָ׃
19שָׁלַ֣חְתָּ בְרָעָ֑ה וּ֝לְשׁוֹנְךָ֗ תַּצְמִ֥יד מִרְמָֽה׃
20בְּאָחִ֣יךָ תְדַבֵּ֑ר בְּבֶֽן־ אִ֝מְּךָ֗ תִּתֶּן־ דֹּֽפִי׃
21עָשִׂ֨יתָ ׀ וְֽהֶחֱרַ֗שְׁתִּי דִּמִּ֗יתָ הֱֽיוֹת־ אֶֽהְיֶ֥ה כָמ֑וֹךָ אוֹכִיחֲךָ֖ וְאֶֽעֶרְכָ֣ה לְעֵינֶֽיךָ׃
22נָ֣א זֹ֭את שֹׁכְחֵ֣י אֱל֑וֹהַּ פֶּן־ אֶ֝טְרֹ֗ף וְאֵ֣ין מַצִּֽיל׃
