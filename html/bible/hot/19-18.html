1לַמְנַצֵּ֤חַ ׀ לְעֶ֥בֶד יְהוָ֗ה לְדָ֫וִ֥ד אֲשֶׁ֤ר דִּבֶּ֨ר ׀ לַיהוָ֗ה אֶת־ דִּ֭בְרֵי הַשִּׁירָ֣ה הַזֹּ֑את בְּי֤וֹם הִֽצִּיל־ יְהוָ֘ה אוֹת֥וֹ מִכַּ֥ף כָּל־ אֹ֝יְבָ֗יו וּמִיַּ֥ד שָׁאֽוּל׃ וַיֹּאמַ֡ר אֶרְחָמְךָ֖ יְהוָ֣ה חִזְקִֽי׃
2סַֽלְעִ֥י וּמְצוּדָתִ֗י וּמְפַ֫לְטִ֥י אֵלִ֣י צ֭וּרִי אֶֽחֱסֶה־ בּ֑וֹ מָֽגִנִּ֥י וְקֶֽרֶן־ יִ֝שְׁעִ֗י מִשְׂגַּבִּֽי׃
3אֶקְרָ֣א יְהוָ֑ה וּמִן־ אֹ֝יְבַ֗י אִוָּשֵֽׁעַ׃
4חֶבְלֵי־ מָ֑וֶת וְֽנַחֲלֵ֖י בְלִיַּ֣עַל יְבַֽעֲתֽוּנִי׃
5שְׁא֣וֹל סְבָב֑וּנִי קִ֝דְּמ֗וּנִי מ֣וֹקְשֵׁי מָֽוֶת׃
6לִ֤י ׀ אֶֽקְרָ֣א יְהוָה֮ וְאֶל־ אֱלֹהַ֪י אֲשַׁ֫וֵּ֥עַ יִשְׁמַ֣ע מֵהֵיכָל֣וֹ קוֹלִ֑י וְ֝שַׁוְעָתִ֗י לְפָנָ֤יו ׀ תָּב֬וֹא בְאָזְנָֽיו׃
7וַתִּרְעַ֨שׁ ׀ הָאָ֗רֶץ וּמוֹסְדֵ֣י הָרִ֣ים יִרְגָּ֑זוּ וַ֝יִּתְגָּֽעֲשׁ֗וּ כִּי־ חָ֥רָה לֽוֹ׃
8עָשָׁ֨ן ׀ בְּאַפּ֗וֹ וְאֵשׁ־ מִפִּ֥יו תֹּאכֵ֑ל גֶּ֝חָלִ֗ים בָּעֲר֥וּ מִמֶּֽנּוּ׃
9שָׁ֭מַיִם וַיֵּרַ֑ד וַ֝עֲרָפֶ֗ל תַּ֣חַת רַגְלָֽיו׃
10עַל־ כְּ֭רוּב וַיָּעֹ֑ף וַ֝יֵּ֗דֶא עַל־ כַּנְפֵי־ רֽוּחַ׃
11חֹ֨שֶׁךְ ׀ סִתְר֗וֹ סְבִֽיבוֹתָ֥יו סֻכָּת֑וֹ חֶשְׁכַת־ מַ֝֗יִם עָבֵ֥י שְׁחָקִֽים׃
12נֶ֫גְדּ֥וֹ עָבָ֥יו עָבְר֑וּ בָּ֝רָ֗ד וְגַֽחֲלֵי־ אֵֽשׁ׃
13בַּשָּׁמַ֨יִם ׀ יְֽהוָ֗ה וְ֭עֶלְיוֹן יִתֵּ֣ן קֹל֑וֹ בָּ֝רָ֗ד וְגַֽחֲלֵי־ אֵֽשׁ׃
14חִ֭צָּיו וַיְפִיצֵ֑ם וּבְרָקִ֥ים רָ֝ב וַיְהֻמֵּֽם׃
15אֲפִ֥יקֵי מַ֗יִם וַֽיִּגָּלוּ֮ מוֹסְד֪וֹת תֵּ֫בֵ֥ל מִגַּעֲרָ֣תְךָ֣ יְהוָ֑ה מִ֝נִּשְׁמַ֗ת ר֣וּחַ אַפֶּֽךָ׃
16מִ֭מָּרוֹם יִקָּחֵ֑נִי יַֽ֝מְשֵׁ֗נִי מִמַּ֥יִם רַבִּֽים׃
17מֵאֹיְבִ֥י עָ֑ז וּ֝מִשֹּׂנְאַ֗י כִּֽי־ אָמְצ֥וּ מִמֶּֽנִּי׃
18בְיוֹם־ אֵידִ֑י וַֽיְהִי־ יְהוָ֖ה לְמִשְׁעָ֣ן לִֽי׃
19לַמֶּרְחָ֑ב יְ֝חַלְּצֵ֗נִי כִּ֘י חָ֥פֵֽץ בִּֽי׃
20יְהוָ֣ה כְּצִדְקִ֑י כְּבֹ֥ר יָ֝דַ֗י יָשִׁ֥יב לִֽי׃
21שָׁ֭מַרְתִּי דַּרְכֵ֣י יְהוָ֑ה וְלֹֽא־ רָ֝שַׁ֗עְתִּי מֵאֱלֹהָֽי׃
22כָל־ מִשְׁפָּטָ֣יו לְנֶגְדִּ֑י וְ֝חֻקֹּתָ֗יו לֹא־ אָסִ֥יר מֶֽנִּי׃
23תָמִ֣ים עִמּ֑וֹ וָ֝אֶשְׁתַּמֵּ֗ר מֵעֲוֹנִֽי׃
24יְהוָ֣ה לִ֣י כְצִדְקִ֑י כְּבֹ֥ר יָ֝דַ֗י לְנֶ֣גֶד עֵינָֽיו׃
25חָסִ֥יד תִּתְחַסָּ֑ד עִם־ גְּבַ֥ר תָּ֝מִ֗ים תִּתַּמָּֽם׃
26נָבָ֥ר תִּתְבָּרָ֑ר וְעִם־ עִ֝קֵּ֗שׁ תִּתְפַּתָּֽל׃
27אַ֭תָּה עַם־ עָנִ֣י תוֹשִׁ֑יעַ וְעֵינַ֖יִם רָמ֣וֹת תַּשְׁפִּֽיל׃
28אַ֭תָּה תָּאִ֣יר נֵרִ֑י יְהוָ֥ה אֱ֝לֹהַ֗י יַגִּ֥יהַּ חָשְׁכִּֽי׃
29בְ֭ךָ אָרֻ֣ץ גְּד֑וּד וּ֝בֵֽאלֹהַ֗י אֲדַלֶּג־ שֽׁוּר׃
30תָּמִ֪ים דַּ֫רְכּ֥וֹ אִמְרַֽת־ יְהוָ֥ה צְרוּפָ֑ה מָגֵ֥ן ה֝֗וּא לְכֹ֤ל ׀ הַחֹסִ֬ים בּֽוֹ׃
31מִ֣י אֱ֭לוֹהַּ מִבַּלְעֲדֵ֣י יְהוָ֑ה וּמִ֥י צ֝֗וּר זוּלָתִ֥י אֱלֹהֵֽינוּ׃
32הַמְאַזְּרֵ֣נִי חָ֑יִל וַיִּתֵּ֖ן תָּמִ֣ים דַּרְכִּֽי׃
33רַ֭גְלַי כָּאַיָּל֑וֹת וְעַ֥ל בָּ֝מֹתַ֗י יַעֲמִידֵֽנִי׃
34יָ֭דַי לַמִּלְחָמָ֑ה וְֽנִחֲתָ֥ה קֶֽשֶׁת־ נְ֝חוּשָׁ֗ה זְרוֹעֹתָֽי׃
35לִי֮ מָגֵ֪ן יִ֫שְׁעֶ֥ךָ וִֽימִינְךָ֥ תִסְעָדֵ֑נִי וְֽעַנְוַתְךָ֥ תַרְבֵּֽנִי׃
36צַעֲדִ֣י תַחְתָּ֑י וְלֹ֥א מָ֝עֲד֗וּ קַרְסֻלָּֽי׃
37א֭וֹיְבַי וְאַשִּׂיגֵ֑ם וְלֹֽא־ אָ֝שׁוּב עַד־ כַּלּוֹתָֽם׃
38וְלֹא־ יֻ֣כְלוּ ק֑וּם יִ֝פְּל֗וּ תַּ֣חַת רַגְלָֽי׃
39חַ֭יִל לַמִּלְחָמָ֑ה תַּכְרִ֖יעַ קָמַ֣י תַּחְתָּֽי׃
40נָתַ֣תָּה לִּ֣י עֹ֑רֶף וּ֝מְשַׂנְאַ֗י אַצְמִיתֵֽם׃
41וְאֵין־ מוֹשִׁ֑יעַ עַל־ יְ֝הוָ֗ה וְלֹ֣א עָנָֽם׃
42כְּעָפָ֥ר עַל־ פְּנֵי־ ר֑וּחַ כְּטִ֖יט חוּצ֣וֹת אֲרִיקֵֽם׃
43מֵרִ֪יבֵ֫י עָ֥ם תְּ֭שִׂימֵנִי לְרֹ֣אשׁ גּוֹיִ֑ם עַ֖ם לֹא־ יָדַ֣עְתִּי יַֽעַבְדֽוּנִי׃
44אֹ֭זֶן יִשָּׁ֣מְעוּ לִ֑י בְּנֵֽי־ נֵ֝כָ֗ר יְכַחֲשׁוּ־ לִֽי׃
45נֵכָ֥ר יִבֹּ֑לוּ וְ֝יַחְרְג֗וּ מִֽמִּסְגְּרֽוֹתֵיהֶֽם׃
46יְ֭הוָה וּבָר֣וּךְ צוּרִ֑י וְ֝יָר֗וּם אֱלוֹהֵ֥י יִשְׁעִֽי׃
47הַנּוֹתֵ֣ן נְקָמ֣וֹת לִ֑י וַיַּדְבֵּ֖ר עַמִּ֣ים תַּחְתָּֽי׃
48מֵאֹ֫יְבָ֥י אַ֣ף מִן־ קָ֭מַי תְּרוֹמְמֵ֑נִי מֵאִ֥ישׁ חָ֝מָ֗ס תַּצִּילֵֽנִי׃
49כֵּ֤ן ׀ אוֹדְךָ֖ בַגּוֹיִ֥ם ׀ יְהוָ֑ה וּלְשִׁמְךָ֥ אֲזַמֵּֽרָה׃
