1וַיְדַבֵּ֤ר מֹשֶׁה֙ אֶל־ רָאשֵׁ֣י הַמַּטּ֔וֹת לִבְנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר זֶ֣ה הַדָּבָ֔ר אֲשֶׁ֖ר צִוָּ֥ה יְהוָֽה׃
2כִּֽי־ יִדֹּ֨ר נֶ֜דֶר לַֽיהוָ֗ה אֽוֹ־ הִשָּׁ֤בַע שְׁבֻעָה֙ לֶאְסֹ֤ר אִסָּר֙ עַל־ נַפְשׁ֔וֹ לֹ֥א יַחֵ֖ל דְּבָר֑וֹ כְּכָל־ הַיֹּצֵ֥א מִפִּ֖יו יַעֲשֶֽׂה׃
3כִּֽי־ תִדֹּ֥ר נֶ֖דֶר לַיהוָ֑ה וְאָסְרָ֥ה אִסָּ֛ר בְּבֵ֥ית אָבִ֖יהָ בִּנְעֻרֶֽיהָ׃
4אָבִ֜יהָ אֶת־ נִדְרָ֗הּ וֶֽאֱסָרָהּ֙ אֲשֶׁ֣ר אָֽסְרָ֣ה עַל־ נַפְשָׁ֔הּ וְהֶחֱרִ֥ישׁ לָ֖הּ אָבִ֑יהָ וְקָ֙מוּ֙ כָּל־ נְדָרֶ֔יהָ וְכָל־ אִסָּ֛ר אֲשֶׁר־ אָסְרָ֥ה עַל־ נַפְשָׁ֖הּ יָקֽוּם׃
5הֵנִ֨יא אָבִ֣יהָ אֹתָהּ֮ בְּי֣וֹם שָׁמְעוֹ֒ כָּל־ נְדָרֶ֗יהָ וֶֽאֱסָרֶ֛יהָ אֲשֶׁר־ אָסְרָ֥ה עַל־ נַפְשָׁ֖הּ לֹ֣א יָק֑וּם וַֽיהוָה֙ יִֽסְלַח־ לָ֔הּ כִּי־ הֵנִ֥יא אָבִ֖יהָ אֹתָֽהּ׃
6הָי֤וֹ תִֽהְיֶה֙ לְאִ֔ישׁ וּנְדָרֶ֖יהָ עָלֶ֑יהָ א֚וֹ מִבְטָ֣א שְׂפָתֶ֔יהָ אֲשֶׁ֥ר אָסְרָ֖ה עַל־ נַפְשָֽׁהּ׃
7אִישָׁ֛הּ בְּי֥וֹם שָׁמְע֖וֹ וְהֶחֱרִ֣ישׁ לָ֑הּ וְקָ֣מוּ נְדָרֶ֗יהָ וֶֽאֱסָרֶ֛הָ אֲשֶׁר־ אָסְרָ֥ה עַל־ נַפְשָׁ֖הּ יָקֻֽמוּ׃
8בְּי֨וֹם שְׁמֹ֣עַ אִישָׁהּ֮ יָנִ֣יא אוֹתָהּ֒ וְהֵפֵ֗ר אֶת־ נִדְרָהּ֙ אֲשֶׁ֣ר עָלֶ֔יהָ וְאֵת֙ מִבְטָ֣א שְׂפָתֶ֔יהָ אֲשֶׁ֥ר אָסְרָ֖ה עַל־ נַפְשָׁ֑הּ וַיהוָ֖ה יִֽסְלַֽח־ לָֽהּ׃
9אַלְמָנָ֖ה וּגְרוּשָׁ֑ה כֹּ֛ל אֲשֶׁר־ אָסְרָ֥ה עַל־ נַפְשָׁ֖הּ יָק֥וּם עָלֶֽיהָ׃
10בֵּ֥ית אִישָׁ֖הּ נָדָ֑רָה אֽוֹ־ אָסְרָ֥ה אִסָּ֛ר עַל־ נַפְשָׁ֖הּ בִּשְׁבֻעָֽה׃
11אִישָׁהּ֙ וְהֶחֱרִ֣שׁ לָ֔הּ לֹ֥א הֵנִ֖יא אֹתָ֑הּ וְקָ֙מוּ֙ כָּל־ נְדָרֶ֔יהָ וְכָל־ אִסָּ֛ר אֲשֶׁר־ אָסְרָ֥ה עַל־ נַפְשָׁ֖הּ יָקֽוּם׃
12הָפֵר֩ יָפֵ֨ר אֹתָ֥ם ׀ אִישָׁהּ֮ בְּי֣וֹם שָׁמְעוֹ֒ כָּל־ מוֹצָ֨א שְׂפָתֶ֧יהָ לִנְדָרֶ֛יהָ וּלְאִסַּ֥ר נַפְשָׁ֖הּ לֹ֣א יָק֑וּם אִישָׁ֣הּ הֲפֵרָ֔ם וַיהוָ֖ה יִֽסְלַֽח־ לָֽהּ׃
13נֵ֛דֶר וְכָל־ שְׁבֻעַ֥ת אִסָּ֖ר לְעַנֹּ֣ת נָ֑פֶשׁ אִישָׁ֥הּ יְקִימֶ֖נּוּ וְאִישָׁ֥הּ יְפֵרֶֽנּוּ׃
14הַחֲרֵשׁ֩ יַחֲרִ֨ישׁ לָ֥הּ אִישָׁהּ֮ מִיּ֣וֹם אֶל־ יוֹם֒ וְהֵקִים֙ אֶת־ כָּל־ נְדָרֶ֔יהָ א֥וֹ אֶת־ כָּל־ אֱסָרֶ֖יהָ אֲשֶׁ֣ר עָלֶ֑יהָ הֵקִ֣ים אֹתָ֔ם כִּי־ הֶחֱרִ֥שׁ לָ֖הּ בְּי֥וֹם שָׁמְעֽוֹ׃
15הָפֵ֥ר יָפֵ֛ר אֹתָ֖ם אַחֲרֵ֣י שָׁמְע֑וֹ וְנָשָׂ֖א אֶת־ עֲוֹנָֽהּ׃
