1וְאִשָּׁ֣ה אַחַ֣ת מִנְּשֵׁ֣י בְנֵֽי־ הַ֠נְּבִיאִים צָעֲקָ֨ה אֶל־ אֱלִישָׁ֜ע לֵאמֹ֗ר עַבְדְּךָ֤ אִישִׁי֙ מֵ֔ת וְאַתָּ֣ה יָדַ֔עְתָּ כִּ֣י עַבְדְּךָ֔ הָיָ֥ה יָרֵ֖א אֶת־ יְהוָ֑ה וְהַ֨נֹּשֶׁ֔ה בָּ֗א לָקַ֜חַת אֶת־ שְׁנֵ֧י יְלָדַ֛י ל֖וֹ לַעֲבָדִֽים׃
2אֵלֶ֤יהָ אֱלִישָׁע֙ מָ֣ה אֶֽעֱשֶׂה־ לָּ֔ךְ הַגִּ֣ידִי לִ֔י מַה־ יֶּשׁ־ [לְכִי כ] (לָ֖ךְ ק) בַּבָּ֑יִת וַתֹּ֗אמֶר אֵ֣ין לְשִׁפְחָתְךָ֥ כֹל֙ בַּבַּ֔יִת כִּ֖י אִם־ אָס֥וּךְ שָֽׁמֶן׃
3לְכִ֨י שַׁאֲלִי־ לָ֤ךְ כֵּלִים֙ מִן־ הַח֔וּץ מֵאֵ֖ת כָּל־ [שִׁכְנָכֵי כ] (שְׁכֵנָ֑יִךְ ק) כֵּלִ֥ים רֵקִ֖ים אַל־ תַּמְעִֽיטִי׃
4וְסָגַ֤רְתְּ הַדֶּ֙לֶת֙ בַּעֲדֵ֣ךְ וּבְעַד־ בָּנַ֔יִךְ וְיָצַ֕קְתְּ עַ֥ל כָּל־ הַכֵּלִ֖ים הָאֵ֑לֶּה וְהַמָּלֵ֖א תַּסִּֽיעִי׃
5מֵֽאִתּ֔וֹ וַתִּסְגֹּ֣ר הַדֶּ֔לֶת בַּעֲדָ֖הּ וּבְעַ֣ד בָּנֶ֑יהָ הֵ֛ם מַגִּשִׁ֥ים אֵלֶ֖יהָ וְהִ֥יא [מֵיצָקֶת כ] (מֹוצָֽקֶת׃ ק)
6כִּמְלֹ֣את הַכֵּלִ֗ים וַתֹּ֤אמֶר אֶל־ בְּנָהּ֙ הַגִּ֨ישָׁה אֵלַ֥י עוֹד֙ כֶּ֔לִי וַיֹּ֣אמֶר אֵלֶ֔יהָ אֵ֥ין ע֖וֹד כֶּ֑לִי וַֽיַּעֲמֹ֖ד הַשָּֽׁמֶן׃
7וַתַּגֵּד֙ לְאִ֣ישׁ הָאֱלֹהִ֔ים וַיֹּ֗אמֶר לְכִי֙ מִכְרִ֣י אֶת־ הַשֶּׁ֔מֶן וְשַׁלְּמִ֖י אֶת־ [נִשְׁיֵכִי כ] (נִשְׁיֵ֑ךְ ק) וְאַ֣תְּ [בְּנֵיכִי כ] (וּבָנַ֔יִךְ ק) תִֽחְיִ֖י בַּנּוֹתָֽר׃ פ
8הַיּ֜וֹם וַיַּעֲבֹ֧ר אֱלִישָׁ֣ע אֶל־ שׁוּנֵ֗ם וְשָׁם֙ אִשָּׁ֣ה גְדוֹלָ֔ה וַתַּחֲזֶק־ בּ֖וֹ לֶאֱכָל־ לָ֑חֶם וַֽיְהִי֙ מִדֵּ֣י עָבְר֔וֹ יָסֻ֥ר שָׁ֖מָּה לֶאֱכָל־ לָֽחֶם׃
9אֶל־ אִישָׁ֔הּ הִנֵּה־ נָ֣א יָדַ֔עְתִּי כִּ֛י אִ֥ישׁ אֱלֹהִ֖ים קָד֣וֹשׁ ה֑וּא עֹבֵ֥ר עָלֵ֖ינוּ תָּמִֽיד׃
10נָּ֤א עֲלִיַּת־ קִיר֙ קְטַנָּ֔ה וְנָשִׂ֨ים ל֥וֹ שָׁ֛ם מִטָּ֥ה וְשֻׁלְחָ֖ן וְכִסֵּ֣א וּמְנוֹרָ֑ה וְהָיָ֛ה בְּבֹא֥וֹ אֵלֵ֖ינוּ יָס֥וּר שָֽׁמָּה׃
11הַיּ֖וֹם וַיָּ֣בֹא שָׁ֑מָּה וַיָּ֥סַר אֶל־ הָעֲלִיָּ֖ה וַיִּשְׁכַּב־ שָֽׁמָּה׃
12אֶל־ גֵּחֲזִ֣י נַעֲר֔וֹ קְרָ֖א לַשּׁוּנַמִּ֣ית הַזֹּ֑את וַיִּקְרָא־ לָ֔הּ וַֽתַּעֲמֹ֖ד לְפָנָֽיו׃
13ל֗וֹ אֱמָר־ נָ֣א אֵלֶיהָ֮ הִנֵּ֣ה חָרַ֣דְתְּ ׀ אֵלֵינוּ֮ אֶת־ כָּל־ הַחֲרָדָ֣ה הַזֹּאת֒ מֶ֚ה לַעֲשׂ֣וֹת לָ֔ךְ הֲיֵ֤שׁ לְדַבֶּר־ לָךְ֙ אֶל־ הַמֶּ֔לֶךְ א֖וֹ אֶל־ שַׂ֣ר הַצָּבָ֑א וַתֹּ֕אמֶר בְּת֥וֹךְ עַמִּ֖י אָנֹכִ֥י יֹשָֽׁבֶת׃
14וּמֶ֖ה לַעֲשׂ֣וֹת לָ֑הּ וַיֹּ֣אמֶר גֵּיחֲזִ֗י אֲבָ֛ל בֵּ֥ן אֵֽין־ לָ֖הּ וְאִישָׁ֥הּ זָקֵֽן׃
15קְרָא־ לָ֑הּ וַיִּקְרָא־ לָ֔הּ וַֽתַּעֲמֹ֖ד בַּפָּֽתַח׃
16לַמּוֹעֵ֤ד הַזֶּה֙ כָּעֵ֣ת חַיָּ֔ה [אַתִּי כ] (אַ֖תְּ ק) חֹבֶ֣קֶת בֵּ֑ן וַתֹּ֗אמֶר אַל־ אֲדֹנִי֙ אִ֣ישׁ הָאֱלֹהִ֔ים אַל־ תְּכַזֵּ֖ב בְּשִׁפְחָתֶֽךָ׃
17הָאִשָּׁ֖ה וַתֵּ֣לֶד בֵּ֑ן לַמּוֹעֵ֤ד הַזֶּה֙ כָּעֵ֣ת חַיָּ֔ה אֲשֶׁר־ דִּבֶּ֥ר אֵלֶ֖יהָ אֱלִישָֽׁע׃
18הַיָּ֑לֶד וַיְהִ֣י הַיּ֔וֹם וַיֵּצֵ֥א אֶל־ אָבִ֖יו אֶל־ הַקֹּצְרִֽים׃
19אֶל־ אָבִ֖יו רֹאשִׁ֣י ׀ רֹאשִׁ֑י וַיֹּ֙אמֶר֙ אֶל־ הַנַּ֔עַר שָׂאֵ֖הוּ אֶל־ אִמּֽוֹ׃
20וַיְבִיאֵ֖הוּ אֶל־ אִמּ֑וֹ וַיֵּ֧שֶׁב עַל־ בִּרְכֶּ֛יהָ עַד־ הַֽצָּהֳרַ֖יִם וַיָּמֹֽת׃
21וַתַּשְׁכִּבֵ֔הוּ עַל־ מִטַּ֖ת אִ֣ישׁ הָאֱלֹהִ֑ים וַתִּסְגֹּ֥ר בַּעֲד֖וֹ וַתֵּצֵֽא׃
22אֶל־ אִישָׁהּ֒ וַתֹּ֗אמֶר שִׁלְחָ֨ה נָ֥א לִי֙ אֶחָ֣ד מִן־ הַנְּעָרִ֔ים וְאַחַ֖ת הָאֲתֹנ֑וֹת וְאָר֛וּצָה עַד־ אִ֥ישׁ הָאֱלֹהִ֖ים וְאָשֽׁוּבָה׃
23מַ֠דּוּעַ [אַתִּי כ] (אַ֣תְּ ק) [הֹלַכְתִּי כ] (הֹלֶ֤כֶת ק) אֵלָיו֙ הַיּ֔וֹם לֹֽא־ חֹ֖דֶשׁ וְלֹ֣א שַׁבָּ֑ת וַתֹּ֖אמֶר שָׁלֽוֹם׃
24הָֽאָת֔וֹן וַתֹּ֥אמֶר אֶֽל־ נַעֲרָ֖הּ נְהַ֣ג וָלֵ֑ךְ אַל־ תַּעֲצָר־ לִ֣י לִרְכֹּ֔ב כִּ֖י אִם־ אָמַ֥רְתִּי לָֽךְ׃
25וַתָּב֛וֹא אֶל־ אִ֥ישׁ הָאֱלֹהִ֖ים אֶל־ הַ֣ר הַכַּרְמֶ֑ל וַ֠יְהִי כִּרְא֨וֹת אִישׁ־ הָאֱלֹהִ֤ים אֹתָהּ֙ מִנֶּ֔גֶד וַיֹּ֙אמֶר֙ אֶל־ גֵּיחֲזִ֣י נַעֲר֔וֹ הִנֵּ֖ה הַשּׁוּנַמִּ֥ית הַלָּֽז׃
26רֽוּץ־ נָ֣א לִקְרָאתָהּ֒ וֶאֱמָר־ לָ֗הּ הֲשָׁל֥וֹם לָ֛ךְ הֲשָׁל֥וֹם לְאִישֵׁ֖ךְ הֲשָׁל֣וֹם לַיָּ֑לֶד וַתֹּ֖אמֶר שָׁלֽוֹם׃
27אֶל־ אִ֤ישׁ הָֽאֱלֹהִים֙ אֶל־ הָהָ֔ר וַֽתַּחֲזֵ֖ק בְּרַגְלָ֑יו וַיִּגַּ֨שׁ גֵּֽיחֲזִ֜י לְהָדְפָ֗הּ וַיֹּאמֶר֩ אִ֨ישׁ הָאֱלֹהִ֤ים הַרְפֵּֽה־ לָהּ֙ כִּֽי־ נַפְשָׁ֣הּ מָֽרָה־ לָ֔הּ וַֽיהוָה֙ הֶעְלִ֣ים מִמֶּ֔נִּי וְלֹ֥א הִגִּ֖יד לִֽי׃
28הֲשָׁאַ֥לְתִּי בֵ֖ן מֵאֵ֣ת אֲדֹנִ֑י הֲלֹ֣א אָמַ֔רְתִּי לֹ֥א תַשְׁלֶ֖ה אֹתִֽי׃
29לְגֵיחֲזִ֜י חֲגֹ֣ר מָתְנֶ֗יךָ וְקַ֨ח מִשְׁעַנְתִּ֣י בְיָדְךָ֮ וָלֵךְ֒ כִּֽי־ תִמְצָ֥א אִישׁ֙ לֹ֣א תְבָרְכֶ֔נּוּ וְכִֽי־ יְבָרֶכְךָ֥ אִ֖ישׁ לֹ֣א תַעֲנֶנּ֑וּ וְשַׂמְתָּ֥ מִשְׁעַנְתִּ֖י עַל־ פְּנֵ֥י הַנָּֽעַר׃
30אֵ֣ם הַנַּ֔עַר חַי־ יְהוָ֥ה וְחֵֽי־ נַפְשְׁךָ֖ אִם־ אֶעֶזְבֶ֑ךָּ וַיָּ֖קָם וַיֵּ֥לֶךְ אַחֲרֶֽיהָ׃
31עָבַ֣ר לִפְנֵיהֶ֗ם וַיָּ֤שֶׂם אֶת־ הַמִּשְׁעֶ֙נֶת֙ עַל־ פְּנֵ֣י הַנַּ֔עַר וְאֵ֥ין ק֖וֹל וְאֵ֣ין קָ֑שֶׁב וַיָּ֤שָׁב לִקְרָאתוֹ֙ וַיַּגֶּד־ ל֣וֹ לֵאמֹ֔ר לֹ֥א הֵקִ֖יץ הַנָּֽעַר׃
32אֱלִישָׁ֖ע הַבָּ֑יְתָה וְהִנֵּ֤ה הַנַּ֙עַר֙ מֵ֔ת מֻשְׁכָּ֖ב עַל־ מִטָּתֽוֹ׃
33וַיִּסְגֹּ֥ר הַדֶּ֖לֶת בְּעַ֣ד שְׁנֵיהֶ֑ם וַיִּתְפַּלֵּ֖ל אֶל־ יְהוָֽה׃
34וַיִּשְׁכַּ֣ב עַל־ הַיֶּ֗לֶד וַיָּשֶׂם֩ פִּ֨יו עַל־ פִּ֜יו וְעֵינָ֤יו עַל־ עֵינָיו֙ וְכַפָּ֣יו עַל־ [כַּפֹּו כ] (כַּפָּ֔יו ק) וַיִּגְהַ֖ר עָלָ֑יו וַיָּ֖חָם בְּשַׂ֥ר הַיָּֽלֶד׃
35וַיֵּ֣לֶךְ בַּבַּ֗יִת אַחַ֥ת הֵ֙נָּה֙ וְאַחַ֣ת הֵ֔נָּה וַיַּ֖עַל וַיִּגְהַ֣ר עָלָ֑יו וַיְזוֹרֵ֤ר הַנַּ֙עַר֙ עַד־ שֶׁ֣בַע פְּעָמִ֔ים וַיִּפְקַ֥ח הַנַּ֖עַר אֶת־ עֵינָֽיו׃
36אֶל־ גֵּיחֲזִ֗י וַיֹּ֙אמֶר֙ קְרָא֙ אֶל־ הַשֻּׁנַמִּ֣ית הַזֹּ֔את וַיִּקְרָאֶ֖הָ וַתָּב֣וֹא אֵלָ֑יו וַיֹּ֖אמֶר שְׂאִ֥י בְנֵֽךְ׃
37וַתִּפֹּ֣ל עַל־ רַגְלָ֔יו וַתִּשְׁתַּ֖חוּ אָ֑רְצָה וַתִּשָּׂ֥א אֶת־ בְּנָ֖הּ וַתֵּצֵֽא׃ פ
38שָׁ֤ב הַגִּלְגָּ֙לָה֙ וְהָרָעָ֣ב בָּאָ֔רֶץ וּבְנֵי֙ הַנְּבִיאִ֔ים יֹשְׁבִ֖ים לְפָנָ֑יו וַיֹּ֣אמֶר לְנַעֲר֗וֹ שְׁפֹת֙ הַסִּ֣יר הַגְּדוֹלָ֔ה וּבַשֵּׁ֥ל נָזִ֖יד לִבְנֵ֥י הַנְּבִיאִֽים׃
39אֶחָ֣ד אֶל־ הַשָּׂדֶה֮ לְלַקֵּ֣ט אֹרֹת֒ וַיִּמְצָא֙ גֶּ֣פֶן שָׂדֶ֔ה וַיְלַקֵּ֥ט מִמֶּ֛נּוּ פַּקֻּעֹ֥ת שָׂדֶ֖ה מְלֹ֣א בִגְד֑וֹ וַיָּבֹ֗א וַיְפַלַּ֛ח אֶל־ סִ֥יר הַנָּזִ֖יד כִּֽי־ לֹ֥א יָדָֽעוּ׃
40לַאֲנָשִׁ֖ים לֶאֱכ֑וֹל וַ֠יְהִי כְּאָכְלָ֨ם מֵהַנָּזִ֜יד וְהֵ֣מָּה צָעָ֗קוּ וַיֹּֽאמְרוּ֙ מָ֤וֶת בַּסִּיר֙ אִ֣ישׁ הָאֱלֹהִ֔ים וְלֹ֥א יָכְל֖וּ לֶאֱכֹֽל׃
41וּקְחוּ־ קֶ֔מַח וַיַּשְׁלֵ֖ךְ אֶל־ הַסִּ֑יר וַיֹּ֗אמֶר צַ֤ק לָעָם֙ וְיֹאכֵ֔לוּ וְלֹ֥א הָיָ֛ה דָּבָ֥ר רָ֖ע בַּסִּֽיר׃ ס
42בָּ֜א מִבַּ֣עַל שָׁלִ֗שָׁה וַיָּבֵא֩ לְאִ֨ישׁ הָאֱלֹהִ֜ים לֶ֤חֶם בִּכּוּרִים֙ עֶשְׂרִֽים־ לֶ֣חֶם שְׂעֹרִ֔ים וְכַרְמֶ֖ל בְּצִקְלֹנ֑וֹ וַיֹּ֕אמֶר תֵּ֥ן לָעָ֖ם וְיֹאכֵֽלוּ׃
