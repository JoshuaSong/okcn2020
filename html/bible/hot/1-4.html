1וְהָ֣אָדָ֔ם יָדַ֖ע אֶת־ חַוָּ֣ה אִשְׁתּ֑וֹ וַתַּ֙הַר֙ וַתֵּ֣לֶד אֶת־ קַ֔יִן וַתֹּ֕אמֶר קָנִ֥יתִי אִ֖ישׁ אֶת־ יְהוָֽה׃
2לָלֶ֔דֶת אֶת־ אָחִ֖יו אֶת־ הָ֑בֶל וַֽיְהִי־ הֶ֙בֶל֙ רֹ֣עֵה צֹ֔אן וְקַ֕יִן הָיָ֖ה עֹבֵ֥ד אֲדָמָֽה׃
3מִקֵּ֣ץ יָמִ֑ים וַיָּבֵ֨א קַ֜יִן מִפְּרִ֧י הָֽאֲדָמָ֛ה מִנְחָ֖ה לַֽיהוָֽה׃
4הֵבִ֥יא גַם־ ה֛וּא מִבְּכֹר֥וֹת צֹאנ֖וֹ וּמֵֽחֶלְבֵהֶ֑ן וַיִּ֣שַׁע יְהוָ֔ה אֶל־ הֶ֖בֶל וְאֶל־ מִנְחָתֽוֹ׃
5קַ֥יִן וְאֶל־ מִנְחָת֖וֹ לֹ֣א שָׁעָ֑ה וַיִּ֤חַר לְקַ֙יִן֙ מְאֹ֔ד וַֽיִּפְּל֖וּ פָּנָֽיו׃
6יְהוָ֖ה אֶל־ קָ֑יִן לָ֚מָּה חָ֣רָה לָ֔ךְ וְלָ֖מָּה נָפְל֥וּ פָנֶֽיךָ׃
7אִם־ תֵּיטִיב֙ שְׂאֵ֔ת וְאִם֙ לֹ֣א תֵיטִ֔יב לַפֶּ֖תַח חַטָּ֣את רֹבֵ֑ץ וְאֵלֶ֙יךָ֙ תְּשׁ֣וּקָת֔וֹ וְאַתָּ֖ה תִּמְשָׁל־ בּֽוֹ׃
8קַ֖יִן אֶל־ הֶ֣בֶל אָחִ֑יו וַֽיְהִי֙ בִּהְיוֹתָ֣ם בַּשָּׂדֶ֔ה וַיָּ֥קָם קַ֛יִן אֶל־ הֶ֥בֶל אָחִ֖יו וַיַּהַרְגֵֽהוּ׃
9יְהוָה֙ אֶל־ קַ֔יִן אֵ֖י הֶ֣בֶל אָחִ֑יךָ וַיֹּ֙אמֶר֙ לֹ֣א יָדַ֔עְתִּי הֲשֹׁמֵ֥ר אָחִ֖י אָנֹֽכִי׃
10מֶ֣ה עָשִׂ֑יתָ ק֚וֹל דְּמֵ֣י אָחִ֔יךָ צֹעֲקִ֥ים אֵלַ֖י מִן־ הָֽאֲדָמָֽה׃
11אָר֣וּר אָ֑תָּה מִן־ הָֽאֲדָמָה֙ אֲשֶׁ֣ר פָּצְתָ֣ה אֶת־ פִּ֔יהָ לָקַ֛חַת אֶת־ דְּמֵ֥י אָחִ֖יךָ מִיָּדֶֽךָ׃
12תַֽעֲבֹד֙ אֶת־ הָ֣אֲדָמָ֔ה לֹֽא־ תֹסֵ֥ף תֵּת־ כֹּחָ֖הּ לָ֑ךְ נָ֥ע וָנָ֖ד תִּֽהְיֶ֥ה בָאָֽרֶץ׃
13קַ֖יִן אֶל־ יְהוָ֑ה גָּד֥וֹל עֲוֹנִ֖י מִנְּשֹֽׂא׃
14גֵּרַ֨שְׁתָּ אֹתִ֜י הַיּ֗וֹם מֵעַל֙ פְּנֵ֣י הָֽאֲדָמָ֔ה וּמִפָּנֶ֖יךָ אֶסָּתֵ֑ר וְהָיִ֜יתִי נָ֤ע וָנָד֙ בָּאָ֔רֶץ וְהָיָ֥ה כָל־ מֹצְאִ֖י יַֽהַרְגֵֽנִי׃
15ל֣וֹ יְהוָ֗ה לָכֵן֙ כָּל־ הֹרֵ֣ג קַ֔יִן שִׁבְעָתַ֖יִם יֻקָּ֑ם וַיָּ֨שֶׂם יְהוָ֤ה לְקַ֙יִן֙ א֔וֹת לְבִלְתִּ֥י הַכּוֹת־ אֹת֖וֹ כָּל־ מֹצְאֽוֹ׃
16קַ֖יִן מִלִּפְנֵ֣י יְהוָ֑ה וַיֵּ֥שֶׁב בְּאֶֽרֶץ־ נ֖וֹד קִדְמַת־ עֵֽדֶן׃
17קַ֙יִן֙ אֶת־ אִשְׁתּ֔וֹ וַתַּ֖הַר וַתֵּ֣לֶד אֶת־ חֲנ֑וֹךְ וַֽיְהִי֙ בֹּ֣נֶה עִ֔יר וַיִּקְרָא֙ שֵׁ֣ם הָעִ֔יר כְּשֵׁ֖ם בְּנ֥וֹ חֲנֽוֹךְ׃
18לַֽחֲנוֹךְ֙ אֶת־ עִירָ֔ד וְעִירָ֕ד יָלַ֖ד אֶת־ מְחֽוּיָאֵ֑ל וּמְחִיּיָאֵ֗ל יָלַד֙ אֶת־ מְת֣וּשָׁאֵ֔ל וּמְתוּשָׁאֵ֖ל יָלַ֥ד אֶת־ לָֽמֶךְ׃
19ל֥וֹ לֶ֖מֶךְ שְׁתֵּ֣י נָשִׁ֑ים שֵׁ֤ם הָֽאַחַת֙ עָדָ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖ית צִלָּֽה׃
20עָדָ֖ה אֶת־ יָבָ֑ל ה֣וּא הָיָ֔ה אֲבִ֕י יֹשֵׁ֥ב אֹ֖הֶל וּמִקְנֶֽה׃
21אָחִ֖יו יוּבָ֑ל ה֣וּא הָיָ֔ה אֲבִ֕י כָּל־ תֹּפֵ֥שׂ כִּנּ֖וֹר וְעוּגָֽב׃
22גַם־ הִ֗וא יָֽלְדָה֙ אֶת־ תּ֣וּבַל קַ֔יִן לֹטֵ֕שׁ כָּל־ חֹרֵ֥שׁ נְחֹ֖שֶׁת וּבַרְזֶ֑ל וַֽאֲח֥וֹת תּֽוּבַל־ קַ֖יִן נַֽעֲמָֽה׃
23לֶ֜מֶךְ לְנָשָׁ֗יו עָדָ֤ה וְצִלָּה֙ שְׁמַ֣עַן קוֹלִ֔י נְשֵׁ֣י לֶ֔מֶךְ הַאְזֵ֖נָּה אִמְרָתִ֑י כִּ֣י אִ֤ישׁ הָרַ֙גְתִּי֙ לְפִצְעִ֔י וְיֶ֖לֶד לְחַבֻּרָתִֽי׃
24שִׁבְעָתַ֖יִם יֻקַּם־ קָ֑יִן וְלֶ֖מֶךְ שִׁבְעִ֥ים וְשִׁבְעָֽה׃
25אָדָ֥ם עוֹד֙ אֶת־ אִשְׁתּ֔וֹ וַתֵּ֣לֶד בֵּ֔ן וַתִּקְרָ֥א אֶת־ שְׁמ֖וֹ שֵׁ֑ת כִּ֣י שָֽׁת־ לִ֤י אֱלֹהִים֙ זֶ֣רַע אַחֵ֔ר תַּ֣חַת הֶ֔בֶל כִּ֥י הֲרָג֖וֹ קָֽיִן׃
