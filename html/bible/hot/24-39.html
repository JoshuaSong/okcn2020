1בַּשָּׁנָ֣ה הַ֠תְּשִׁעִית לְצִדְקִיָּ֨הוּ מֶלֶךְ־ יְהוּדָ֜ה בַּחֹ֣דֶשׁ הָעֲשִׂרִ֗י בָּ֠א נְבוּכַדְרֶאצַּ֨ר מֶֽלֶךְ־ בָּבֶ֤ל וְכָל־ חֵילוֹ֙ אֶל־ יְר֣וּשָׁלִַ֔ם וַיָּצֻ֖רוּ עָלֶֽיהָ׃ ס
2עֶשְׂרֵ֤ה שָׁנָה֙ לְצִדְקִיָּ֔הוּ בַּחֹ֥דֶשׁ הָרְבִיעִ֖י בְּתִשְׁעָ֣ה לַחֹ֑דֶשׁ הָבְקְעָ֖ה הָעִֽיר׃
3כֹּ֚ל שָׂרֵ֣י מֶֽלֶךְ־ בָּבֶ֔ל וַיֵּשְׁב֖וּ בְּשַׁ֣עַר הַתָּ֑וֶךְ נֵרְגַ֣ל שַׂר־ אֶ֠צֶר סַֽמְגַּר־ נְב֞וּ שַׂר־ סְכִ֣ים רַב־ סָרִ֗יס נֵרְגַ֤ל שַׂר־ אֶ֙צֶר֙ רַב־ מָ֔ג וְכָל־ שְׁאֵרִ֔ית שָׂרֵ֖י מֶ֥לֶךְ בָּבֶֽל׃
4כַּאֲשֶׁ֣ר רָ֠אָם צִדְקִיָּ֨הוּ מֶֽלֶךְ־ יְהוּדָ֜ה וְכֹ֣ל ׀ אַנְשֵׁ֣י הַמִּלְחָמָ֗ה וַֽ֠יִּבְרְחוּ וַיֵּצְא֨וּ לַ֤יְלָה מִן־ הָעִיר֙ דֶּ֚רֶךְ גַּ֣ן הַמֶּ֔לֶךְ בְּשַׁ֖עַר בֵּ֣ין הַחֹמֹתָ֑יִם וַיֵּצֵ֖א דֶּ֥רֶךְ הָעֲרָבָֽה׃
5חֵיל־ כַּשְׂדִּ֜ים אַחֲרֵיהֶ֗ם וַיַּשִּׂ֣גוּ אֶת־ צִדְקִיָּהוּ֮ בְּעַֽרְב֣וֹת יְרֵחוֹ֒ וַיִּקְח֣וּ אֹת֗וֹ וַֽ֠יַּעֲלֻהוּ אֶל־ נְבוּכַדְרֶאצַּ֧ר מֶֽלֶךְ־ בָּבֶ֛ל רִבְלָ֖תָה בְּאֶ֣רֶץ חֲמָ֑ת וַיְדַבֵּ֥ר אִתּ֖וֹ מִשְׁפָּטִֽים׃
6מֶ֨לֶךְ בָּבֶ֜ל אֶת־ בְּנֵ֧י צִדְקִיָּ֛הוּ בְּרִבְלָ֖ה לְעֵינָ֑יו וְאֵת֙ כָּל־ חֹרֵ֣י יְהוּדָ֔ה שָׁחַ֖ט מֶ֥לֶךְ בָּבֶֽל׃
7עֵינֵ֥י צִדְקִיָּ֖הוּ עִוֵּ֑ר וַיַּאַסְרֵ֙הוּ֙ בַּֽנְחֻשְׁתַּ֔יִם לָבִ֥יא אֹת֖וֹ בָּבֶֽלָה׃
8בֵּ֤ית הַמֶּ֙לֶךְ֙ וְאֶת־ בֵּ֣ית הָעָ֔ם שָׂרְפ֥וּ הַכַּשְׂדִּ֖ים בָּאֵ֑שׁ וְאֶת־ חֹמ֥וֹת יְרוּשָׁלִַ֖ם נָתָֽצוּ׃
9יֶ֨תֶר הָעָ֜ם הַנִּשְׁאָרִ֣ים בָּעִ֗יר וְאֶת־ הַנֹּֽפְלִים֙ אֲשֶׁ֣ר נָפְל֣וּ עָלָ֔יו וְאֵ֛ת יֶ֥תֶר הָעָ֖ם הַנִּשְׁאָרִ֑ים הֶגְלָ֛ה נְבֽוּזַר־ אֲדָ֥ן רַב־ טַבָּחִ֖ים בָּבֶֽל׃
10הָעָ֣ם הַדַּלִּ֗ים אֲשֶׁ֤ר אֵין־ לָהֶם֙ מְא֔וּמָה הִשְׁאִ֛יר נְבוּזַרְאֲדָ֥ן רַב־ טַבָּחִ֖ים בְּאֶ֣רֶץ יְהוּדָ֑ה וַיִּתֵּ֥ן לָהֶ֛ם כְּרָמִ֥ים וִֽיגֵבִ֖ים בַּיּ֥וֹם הַהֽוּא׃
11נְבוּכַדְרֶאצַּ֥ר מֶֽלֶךְ־ בָּבֶ֖ל עַֽל־ יִרְמְיָ֑הוּ בְּיַ֛ד נְבוּזַרְאֲדָ֥ן רַב־ טַבָּחִ֖ים לֵאמֹֽר׃
12וְעֵינֶ֙יךָ֙ שִׂ֣ים עָלָ֔יו וְאַל־ תַּ֥עַשׂ ל֖וֹ מְא֣וּמָה רָּ֑ע כִּ֗י [אִם כ] כַּֽאֲשֶׁר֙ יְדַבֵּ֣ר אֵלֶ֔יךָ כֵּ֖ן עֲשֵׂ֥ה עִמּֽוֹ׃

14וַיִּקְח֨וּ אֶֽת־ יִרְמְיָ֜הוּ מֵחֲצַ֣ר הַמַּטָּרָ֗ה וַיִּתְּנ֤וּ אֹתוֹ֙ אֶל־ גְּדַלְיָ֙הוּ֙ בֶּן־ אֲחִיקָ֣ם בֶּן־ שָׁפָ֔ן לְהוֹצִאֵ֖הוּ אֶל־ הַבָּ֑יִת וַיֵּ֖שֶׁב בְּת֥וֹךְ הָעָֽם׃ ס
15יִרְמְיָ֖הוּ הָיָ֣ה דְבַר־ יְהוָ֑ה בִּֽהְיֹת֣וֹ עָצ֔וּר בַּחֲצַ֥ר הַמַּטָּרָ֖ה לֵאמֹֽר׃
16וְאָמַרְתָּ֡ לְעֶבֶד־ מֶ֨לֶךְ הַכּוּשִׁ֜י לֵאמֹ֗ר כֹּֽה־ אָמַ֞ר יְהוָ֤ה צְבָאוֹת֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל הִנְנִי֩ [מֵבִי כ] (מֵבִ֨יא ק) אֶת־ דְּבָרַ֜י אֶל־ הָעִ֥יר הַזֹּ֛את לְרָעָ֖ה וְלֹ֣א לְטוֹבָ֑ה וְהָי֥וּ לְפָנֶ֖יךָ בַּיּ֥וֹם הַהֽוּא׃
