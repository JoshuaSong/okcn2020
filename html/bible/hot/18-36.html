1וַיֹּ֥סֶף אֱלִיה֗וּא וַיֹּאמַֽר׃
2לִ֣י זְ֭עֵיר וַאֲחַוֶּ֑ךָּ כִּ֤י ע֖וֹד לֶאֱל֣וֹהַּ מִלִּֽים׃
3דֵ֭עִי לְמֵרָח֑וֹק וּ֝לְפֹעֲלִ֗י אֶֽתֵּֽן־ צֶֽדֶק׃
4אָ֭מְנָם לֹא־ שֶׁ֣קֶר מִלָּ֑י תְּמִ֖ים דֵּע֣וֹת עִמָּֽךְ׃
5אֵ֣ל כַּ֭בִּיר וְלֹ֣א יִמְאָ֑ס כַּ֝בִּ֗יר כֹּ֣חַֽ לֵֽב׃
6יְחַיֶּ֥ה רָשָׁ֑ע וּמִשְׁפַּ֖ט עֲנִיִּ֣ים יִתֵּֽן׃
7יִגְרַ֥ע מִצַּדִּ֗יק עֵ֫ינָ֥יו וְאֶת־ מְלָכִ֥ים לַכִּסֵּ֑א וַיֹּשִׁיבֵ֥ם לָ֝נֶ֗צַח וַיִּגְבָּֽהוּ׃
8אֲסוּרִ֥ים בַּזִּקִּ֑ים יִ֝לָּכְד֗וּן בְּחַבְלֵי־ עֹֽנִי׃
9לָהֶ֣ם פָּעֳלָ֑ם וּ֝פִשְׁעֵיהֶ֗ם כִּ֣י יִתְגַּבָּֽרוּ׃
10אָ֭זְנָם לַמּוּסָ֑ר וַ֝יֹּ֗אמֶר כִּֽי־ יְשֻׁב֥וּן מֵאָֽוֶן׃
11יִשְׁמְע֗וּ וְֽיַ֫עֲבֹ֥דוּ יְכַלּ֣וּ יְמֵיהֶ֣ם בַּטּ֑וֹב וּ֝שְׁנֵיהֶ֗ם בַּנְּעִימִֽים׃
12לֹ֣א יִ֭שְׁמְעוּ בְּשֶׁ֣לַח יַעֲבֹ֑רוּ וְ֝יִגְוְע֗וּ כִּבְלִי־ דָֽעַת׃
13לֵ֭ב יָשִׂ֣ימוּ אָ֑ף לֹ֥א יְ֝שַׁוְּע֗וּ כִּ֣י אֲסָרָֽם׃
14בַּנֹּ֣עַר נַפְשָׁ֑ם וְ֝חַיָּתָ֗ם בַּקְּדֵשִֽׁים׃
15עָנִ֣י בְעָנְי֑וֹ וְיִ֖גֶל בַּלַּ֣חַץ אָזְנָֽם׃
16הֲסִיתְךָ֨ ׀ מִפִּי־ צָ֗ר רַ֭חַב לֹא־ מוּצָ֣ק תַּחְתֶּ֑יהָ וְנַ֥חַת שֻׁ֝לְחָנְךָ֗ מָ֣לֵא דָֽשֶׁן׃
17רָשָׁ֥ע מָלֵ֑אתָ דִּ֖ין וּמִשְׁפָּ֣ט יִתְמֹֽכוּ׃
18חֵ֭מָה פֶּן־ יְסִֽיתְךָ֣ בְסָ֑פֶק וְרָב־ כֹּ֝֗פֶר אַל־ יַטֶּֽךָּ׃
19שׁ֭וּעֲךָ לֹ֣א בְצָ֑ר וְ֝כֹ֗ל מַאֲמַצֵּי־ כֹֽחַ׃
20תִּשְׁאַ֥ף הַלָּ֑יְלָה לַעֲל֖וֹת עַמִּ֣ים תַּחְתָּֽם׃
21אַל־ תֵּ֣פֶן אֶל־ אָ֑וֶן כִּֽי־ עַל־ זֶ֝֗ה בָּחַ֥רְתָּ מֵעֹֽנִי׃
22אֵ֭ל יַשְׂגִּ֣יב בְּכֹח֑וֹ מִ֖י כָמֹ֣הוּ מוֹרֶֽה׃
23פָקַ֣ד עָלָ֣יו דַּרְכּ֑וֹ וּמִֽי־ אָ֝מַ֗ר פָּעַ֥לְתָּ עַוְלָֽה׃
24כִּֽי־ תַשְׂגִּ֣יא פָעֳל֑וֹ אֲשֶׁ֖ר שֹׁרְר֣וּ אֲנָשִֽׁים׃
25אָדָ֥ם חָֽזוּ־ ב֑וֹ אֱ֝נ֗וֹשׁ יַבִּ֥יט מֵרָחֽוֹק׃
26אֵ֣ל שַׂ֭גִּיא וְלֹ֣א נֵדָ֑ע מִסְפַּ֖ר שָׁנָ֣יו וְלֹא־ חֵֽקֶר׃
27יְגָרַ֣ע נִטְפֵי־ מָ֑יִם יָזֹ֖קּוּ מָטָ֣ר לְאֵדֽוֹ׃
28יִזְּל֥וּ שְׁחָקִ֑ים יִ֝רְעֲפ֗וּ עֲלֵ֤י ׀ אָדָ֬ם רָֽב׃
29אִם־ יָ֭בִין מִפְרְשֵׂי־ עָ֑ב תְּ֝שֻׁא֗וֹת סֻכָּתֽוֹ׃
30פָּרַ֣שׂ עָלָ֣יו אוֹר֑וֹ וְשָׁרְשֵׁ֖י הַיָּ֣ם כִּסָּֽה׃
31בָ֭ם יָדִ֣ין עַמִּ֑ים יִֽתֶּן־ אֹ֥כֶל לְמַכְבִּֽיר׃
32כַּפַּ֥יִם כִּסָּה־ א֑וֹר וַיְצַ֖ו עָלֶ֣יהָ בְמַפְגִּֽיעַ׃
