1וַיְדַבֵּ֣ר אֱלֹהִ֔ים אֵ֛ת כָּל־ הַדְּבָרִ֥ים הָאֵ֖לֶּה לֵאמֹֽר׃ ס
2יְהוָ֣ה אֱלֹהֶ֑֔יךָ אֲשֶׁ֧ר הוֹצֵאתִ֛יךָ מֵאֶ֥רֶץ מִצְרַ֖יִם מִבֵּ֣֥ית עֲבָדִֽ֑ים׃
3יִהְיֶֽה־ לְךָ֛֩ אֱלֹהִ֥֨ים אֲחֵרִ֖֜ים עַל־ פָּנָֽ֗יַ
4תַֽעֲשֶׂ֨ה־ לְךָ֥֣ פֶ֣֙סֶל֙ ׀ וְכָל־ תְּמוּנָ֡֔ה אֲשֶׁ֤֣ר בַּשָּׁמַ֣֙יִם֙ ׀ מִמַּ֡֔עַל וַֽאֲשֶׁ֥ר֩ בָּאָ֖֨רֶץ מִתַָּ֑֜חַת וַאֲשֶׁ֥֣ר בַּמַּ֖֣יִם ׀ מִתַּ֥֣חַת לָאָֽ֗רֶץ
5תִשְׁתַּחְוֶ֥֣ה לָהֶ֖ם֮ וְלֹ֣א תָעָבְדֵ֑ם֒ כִּ֣י אָֽנֹכִ֞י יְהוָ֤ה אֱלֹהֶ֙יךָ֙ אֵ֣ל קַנָּ֔א פֹּ֠קֵד עֲוֹ֨ן אָבֹ֧ת עַל־ בָּנִ֛ים עַל־ שִׁלֵּשִׁ֥ים וְעַל־ רִבֵּעִ֖ים לְשֹׂנְאָֽ֑י׃
6חֶ֖֙סֶד֙ לַאֲלָפִ֑֔ים לְאֹהֲבַ֖י וּלְשֹׁמְרֵ֥י מִצְוֹתָֽי׃ ס
7תִשָּׂ֛א אֶת־ שֵֽׁם־ יְהוָ֥ה אֱלֹהֶ֖יךָ לַשָּׁ֑וְא כִּ֣י לֹ֤א יְנַקֶּה֙ יְהוָ֔ה אֵ֛ת אֲשֶׁר־ יִשָּׂ֥א אֶת־ שְׁמ֖וֹ לַשָּֽׁוְא׃ פ
8אֶת־ י֥֨וֹם הַשַּׁבָּ֖֜ת לְקַדְּשֽׁ֗וֹ
9יָמִ֣ים֙ תַּֽעֲבֹ֔ד֮ וְעָשִׂ֖֣יתָ כָּל־ מְלַאכְתֶּֽךָ֒
10הַשְּׁבִיעִ֔֜י שַׁבָּ֖֣ת ׀ לַיהוָ֣ה אֱלֹהֶ֑֗יךָ לֹֽ֣א־ תַעֲשֶׂ֣֨ה כָל־ מְלָאכָ֡֜ה אַתָּ֣ה ׀ וּבִנְךָֽ֣־ וּ֠בִתֶּ֗ךָ עַבְדְּךָ֤֨ וַאֲמָֽתְךָ֜֙ וּבְהֶמְתֶּ֔֗ךָ וְגֵרְךָ֖֙ אֲשֶׁ֥֣ר בִּשְׁעָרֶֽ֔יךָ
11שֵֽׁשֶׁת־ יָמִים֩ עָשָׂ֨ה יְהוָ֜ה אֶת־ הַשָּׁמַ֣יִם וְאֶת־ הָאָ֗רֶץ אֶת־ הַיָּם֙ וְאֶת־ כָּל־ אֲשֶׁר־ בָּ֔ם וַיָּ֖נַח בַּיּ֣וֹם הַשְּׁבִיעִ֑י עַל־ כֵּ֗ן בֵּרַ֧ךְ יְהוָ֛ה אֶת־ י֥וֹם הַשַּׁבָּ֖ת וַֽיְקַדְּשֵֽׁהוּ׃ ס
12אֶת־ אָבִ֖יךָ וְאֶת־ אִמֶּ֑ךָ לְמַ֙עַן֙ יַאֲרִכ֣וּן יָמֶ֔יךָ עַ֚ל הָאֲדָמָ֔ה אֲשֶׁר־ יְהוָ֥ה אֱלֹהֶ֖יךָ נֹתֵ֥ן לָֽךְ׃ ס
13תִּֿרְצָֽ֖ח׃ ס
14תִּֿנְאָֽ֑ף׃ ס
15תִּֿגְנֹֽ֔ב׃ ס
16תַעֲנֶ֥ה בְרֵעֲךָ֖ עֵ֥ד שָֽׁקֶר׃ ס
17תַחְמֹ֖ד בֵּ֣ית רֵעֶ֑ךָ לֹֽא־ תַחְמֹ֞ד אֵ֣שֶׁת רֵעֶ֗ךָ וְעַבְדּ֤וֹ וַאֲמָתוֹ֙ וְשׁוֹר֣וֹ וַחֲמֹר֔וֹ וְכֹ֖ל אֲשֶׁ֥ר לְרֵעֶֽךָ׃ פ
18הָעָם֩ רֹאִ֨ים אֶת־ הַקּוֹלֹ֜ת וְאֶת־ הַלַּפִּידִ֗ם וְאֵת֙ ק֣וֹל הַשֹּׁפָ֔ר וְאֶת־ הָהָ֖ר עָשֵׁ֑ן וַיַּ֤רְא הָעָם֙ וַיָּנֻ֔עוּ וַיַּֽעַמְד֖וּ מֵֽרָחֹֽק׃
19אֶל־ מֹשֶׁ֔ה דַּבֵּר־ אַתָּ֥ה עִמָּ֖נוּ וְנִשְׁמָ֑עָה וְאַל־ יְדַבֵּ֥ר עִמָּ֛נוּ אֱלֹהִ֖ים פֶּן־ נָמֽוּת׃
20מֹשֶׁ֣ה אֶל־ הָעָם֮ אַל־ תִּירָאוּ֒ כִּ֗י לְבַֽעֲבוּר֙ נַסּ֣וֹת אֶתְכֶ֔ם בָּ֖א הָאֱלֹהִ֑ים וּבַעֲב֗וּר תִּהְיֶ֧ה יִרְאָת֛וֹ עַל־ פְּנֵיכֶ֖ם לְבִלְתִּ֥י תֶחֱטָֽאוּ׃
21הָעָ֖ם מֵרָחֹ֑ק וּמֹשֶׁה֙ נִגַּ֣שׁ אֶל־ הָֽעֲרָפֶ֔ל אֲשֶׁר־ שָׁ֖ם הָאֱלֹהִֽים׃ פ
22יְהוָה֙ אֶל־ מֹשֶׁ֔ה כֹּ֥ה תֹאמַ֖ר אֶל־ בְּנֵ֣י יִשְׂרָאֵ֑ל אַתֶּ֣ם רְאִיתֶ֔ם כִּ֚י מִן־ הַשָּׁמַ֔יִם דִּבַּ֖רְתִּי עִמָּכֶֽם׃
23תַעֲשׂ֖וּן אִתִּ֑י אֱלֹ֤הֵי כֶ֙סֶף֙ וֵאלֹהֵ֣י זָהָ֔ב לֹ֥א תַעֲשׂ֖וּ לָכֶֽם׃
24אֲדָמָה֮ תַּעֲשֶׂה־ לִּי֒ וְזָבַחְתָּ֣ עָלָ֗יו אֶת־ עֹלֹתֶ֙יךָ֙ וְאֶת־ שְׁלָמֶ֔יךָ אֶת־ צֹֽאנְךָ֖ וְאֶת־ בְּקָרֶ֑ךָ בְּכָל־ הַמָּקוֹם֙ אֲשֶׁ֣ר אַזְכִּ֣יר אֶת־ שְׁמִ֔י אָב֥וֹא אֵלֶ֖יךָ וּבֵרַכְתִּֽיךָ׃
25מִזְבַּ֤ח אֲבָנִים֙ תַּֽעֲשֶׂה־ לִּ֔י לֹֽא־ תִבְנֶ֥ה אֶתְהֶ֖ן גָּזִ֑ית כִּ֧י חַרְבְּךָ֛ הֵנַ֥פְתָּ עָלֶ֖יהָ וַתְּחַֽלְלֶֽהָ׃
