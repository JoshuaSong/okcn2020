1וַיִּקְרָ֖א אֶל־ מֹשֶׁ֑ה וַיְדַבֵּ֤ר יְהוָה֙ אֵלָ֔יו מֵאֹ֥הֶל מוֹעֵ֖ד לֵאמֹֽר׃
2אֶל־ בְּנֵ֤י יִשְׂרָאֵל֙ וְאָמַרְתָּ֣ אֲלֵהֶ֔ם אָדָ֗ם כִּֽי־ יַקְרִ֥יב מִכֶּ֛ם קָרְבָּ֖ן לַֽיהוָ֑ה מִן־ הַבְּהֵמָ֗ה מִן־ הַבָּקָר֙ וּמִן־ הַצֹּ֔אן תַּקְרִ֖יבוּ אֶת־ קָרְבַּנְכֶֽם׃
3עֹלָ֤ה קָרְבָּנוֹ֙ מִן־ הַבָּקָ֔ר זָכָ֥ר תָּמִ֖ים יַקְרִיבֶ֑נּוּ אֶל־ פֶּ֝תַח אֹ֤הֶל מוֹעֵד֙ יַקְרִ֣יב אֹת֔וֹ לִרְצֹנ֖וֹ לִפְנֵ֥י יְהוָֽה׃
4יָד֔וֹ עַ֖ל רֹ֣אשׁ הָעֹלָ֑ה וְנִרְצָ֥ה ל֖וֹ לְכַפֵּ֥ר עָלָֽיו׃
5אֶת־ בֶּ֥ן הַבָּקָ֖ר לִפְנֵ֣י יְהוָ֑ה וְ֠הִקְרִיבוּ בְּנֵ֨י אַהֲרֹ֤ן הַֽכֹּֽהֲנִים֙ אֶת־ הַדָּ֔ם וְזָרְק֨וּ אֶת־ הַדָּ֤ם עַל־ הַמִּזְבֵּ֙חַ֙ סָבִ֔יב אֲשֶׁר־ פֶּ֖תַח אֹ֥הֶל מוֹעֵֽד׃
6אֶת־ הָעֹלָ֑ה וְנִתַּ֥ח אֹתָ֖הּ לִנְתָחֶֽיהָ׃
7בְּנֵ֨י אַהֲרֹ֧ן הַכֹּהֵ֛ן אֵ֖שׁ עַל־ הַמִּזְבֵּ֑חַ וְעָרְכ֥וּ עֵצִ֖ים עַל־ הָאֵֽשׁ׃
8בְּנֵ֤י אַהֲרֹן֙ הַכֹּ֣הֲנִ֔ים אֵ֚ת הַנְּתָחִ֔ים אֶת־ הָרֹ֖אשׁ וְאֶת־ הַפָּ֑דֶר עַל־ הָעֵצִים֙ אֲשֶׁ֣ר עַל־ הָאֵ֔שׁ אֲשֶׁ֖ר עַל־ הַמִּזְבֵּֽחַ׃
9וּכְרָעָ֖יו יִרְחַ֣ץ בַּמָּ֑יִם וְהִקְטִ֨יר הַכֹּהֵ֤ן אֶת־ הַכֹּל֙ הַמִּזְבֵּ֔חָה עֹלָ֛ה אִשֵּׁ֥ה רֵֽיחַ־ נִיח֖וֹחַ לַֽיהוָֽה׃ ס
10מִן־ הַצֹּ֨אן קָרְבָּנ֧וֹ מִן־ הַכְּשָׂבִ֛ים א֥וֹ מִן־ הָעִזִּ֖ים לְעֹלָ֑ה זָכָ֥ר תָּמִ֖ים יַקְרִיבֶֽנּוּ׃
11אֹת֜וֹ עַ֣ל יֶ֧רֶךְ הַמִּזְבֵּ֛חַ צָפֹ֖נָה לִפְנֵ֣י יְהוָ֑ה וְזָרְק֡וּ בְּנֵי֩ אַהֲרֹ֨ן הַכֹּהֲנִ֧ים אֶת־ דָּמ֛וֹ עַל־ הַמִּזְבֵּ֖חַ סָבִֽיב׃
12אֹתוֹ֙ לִנְתָחָ֔יו וְאֶת־ רֹאשׁ֖וֹ וְאֶת־ פִּדְר֑וֹ וְעָרַ֤ךְ הַכֹּהֵן֙ אֹתָ֔ם עַל־ הָֽעֵצִים֙ אֲשֶׁ֣ר עַל־ הָאֵ֔שׁ אֲשֶׁ֖ר עַל־ הַמִּזְבֵּֽחַ׃
13וְהַכְּרָעַ֖יִם יִרְחַ֣ץ בַּמָּ֑יִם וְהִקְרִ֨יב הַכֹּהֵ֤ן אֶת־ הַכֹּל֙ וְהִקְטִ֣יר הַמִּזְבֵּ֔חָה עֹלָ֣ה ה֗וּא אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַיהוָֽה׃ פ
14מִן־ הָע֛וֹף עֹלָ֥ה קָרְבָּנ֖וֹ לַֽיהוָ֑ה וְהִקְרִ֣יב מִן־ הַתֹּרִ֗ים א֛וֹ מִן־ בְּנֵ֥י הַיּוֹנָ֖ה אֶת־ קָרְבָּנֽוֹ׃
15הַכֹּהֵן֙ אֶל־ הַמִּזְבֵּ֔חַ וּמָלַק֙ אֶת־ רֹאשׁ֔וֹ וְהִקְטִ֖יר הַמִּזְבֵּ֑חָה וְנִמְצָ֣ה דָמ֔וֹ עַ֖ל קִ֥יר הַמִּזְבֵּֽחַ׃
16אֶת־ מֻרְאָת֖וֹ בְּנֹצָתָ֑הּ וְהִשְׁלִ֨יךְ אֹתָ֜הּ אֵ֤צֶל הַמִּזְבֵּ֙חַ֙ קֵ֔דְמָה אֶל־ מְק֖וֹם הַדָּֽשֶׁן׃
