1בְּךָֽ־ יְהוָ֥ה חָסִ֑יתִי אַל־ אֵב֥וֹשָׁה לְעוֹלָֽם׃
2תַּצִּילֵ֥נִי וּֽתְפַלְּטֵ֑נִי הַטֵּֽה־ אֵלַ֥י אָ֝זְנְךָ֗ וְהוֹשִׁיעֵֽנִי׃
3לִ֨י ׀ לְצ֥וּר מָע֡וֹן לָב֗וֹא תָּמִ֗יד צִוִּ֥יתָ לְהוֹשִׁיעֵ֑נִי כִּֽי־ סַלְעִ֖י וּמְצוּדָתִ֣י אָֽתָּה׃
4פַּ֭לְּטֵנִי מִיַּ֣ד רָשָׁ֑ע מִכַּ֖ף מְעַוֵּ֣ל וְחוֹמֵץ׃
5אַתָּ֥ה תִקְוָתִ֑י אֲדֹנָ֥י יְ֝הוִ֗ה מִבְטַחִ֥י מִנְּעוּרָֽי׃
6נִסְמַ֬כְתִּי מִבֶּ֗טֶן מִמְּעֵ֣י אִ֭מִּי אַתָּ֣ה גוֹזִ֑י בְּךָ֖ תְהִלָּתִ֣י תָמִֽיד׃
7הָיִ֣יתִי לְרַבִּ֑ים וְ֝אַתָּ֗ה מַֽחֲסִי־ עֹֽז׃
8פִ֭י תְּהִלָּתֶ֑ךָ כָּל־ הַ֝יּ֗וֹם תִּפְאַרְתֶּֽךָ׃
9תַּ֭שְׁלִיכֵנִי לְעֵ֣ת זִקְנָ֑ה כִּכְל֥וֹת כֹּ֝חִ֗י אַֽל־ תַּעַזְבֵֽנִי׃
10אָמְר֣וּ אוֹיְבַ֣י לִ֑י וְשֹׁמְרֵ֥י נַ֝פְשִׁ֗י נוֹעֲצ֥וּ יַחְדָּֽו׃
11אֱלֹהִ֣ים עֲזָב֑וֹ רִֽדְפ֥וּ וְ֝תִפְשׂ֗וּהוּ כִּי־ אֵ֥ין מַצִּֽיל׃
12אַל־ תִּרְחַ֣ק מִמֶּ֑נִּי אֱ֝לֹהַ֗י לְעֶזְרָ֥תִי [חִישָׁה כ] (חֽוּשָׁה׃ ק)
13יִכְלוּ֮ שֹׂטְנֵ֪י נַ֫פְשִׁ֥י יַֽעֲט֣וּ חֶ֭רְפָּה וּכְלִמָּ֑ה מְ֝בַקְשֵׁ֗י רָעָתִֽי׃
14תָּמִ֣יד אֲיַחֵ֑ל וְ֝הוֹסַפְתִּ֗י עַל־ כָּל־ תְּהִלָּתֶֽךָ׃
15יְסַפֵּ֬ר צִדְקָתֶ֗ךָ כָּל־ הַיּ֥וֹם תְּשׁוּעָתֶ֑ךָ כִּ֤י לֹ֖א יָדַ֣עְתִּי סְפֹרֽוֹת׃
16בִּ֭גְבֻרוֹת אֲדֹנָ֣י יְהוִ֑ה אַזְכִּ֖יר צִדְקָתְךָ֣ לְבַדֶּֽךָ׃
17לִמַּדְתַּ֥נִי מִנְּעוּרָ֑י וְעַד־ הֵ֝֗נָּה אַגִּ֥יד נִפְלְאוֹתֶֽיךָ׃
18עַד־ זִקְנָ֨ה ׀ וְשֵׂיבָה֮ אֱלֹהִ֪ים אַֽל־ תַּעַ֫זְבֵ֥נִי עַד־ אַגִּ֣יד זְרוֹעֲךָ֣ לְד֑וֹר לְכָל־ יָ֝ב֗וֹא גְּבוּרָתֶֽךָ׃
19אֱלֹהִ֗ים עַד־ מָ֫ר֥וֹם אֲשֶׁר־ עָשִׂ֥יתָ גְדֹל֑וֹת אֱ֝לֹהִ֗ים מִ֣י כָמֽוֹךָ׃
20[הִרְאִיתַנוּ כ] (הִרְאִיתַ֨נִי ק) צָר֥וֹת רַבּ֗וֹת וְרָ֫ע֥וֹת תָּשׁ֥וּב [תְּחַיֵּינוּ כ] (תְּחַיֵּ֑ינִי ק) וּֽמִתְּהֹמ֥וֹת הָ֝אָ֗רֶץ תָּשׁ֥וּב תַּעֲלֵֽנִי׃
21גְּֽדֻלָּתִ֗י וְתִסֹּ֥ב תְּֽנַחֲמֵֽנִי׃
22אֲנִ֤י ׀ אוֹדְךָ֣ בִכְלִי־ נֶבֶל֮ אֲמִתְּךָ֪ אֱלֹ֫הָ֥י אֲזַמְּרָ֣ה לְךָ֣ בְכִנּ֑וֹר קְ֝ד֗וֹשׁ יִשְׂרָאֵֽל׃
23שְׂ֭פָתַי כִּ֣י אֲזַמְּרָה־ לָּ֑ךְ וְ֝נַפְשִׁ֗י אֲשֶׁ֣ר פָּדִֽיתָ׃
