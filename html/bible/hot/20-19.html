1טֽוֹב־ רָ֭שׁ הוֹלֵ֣ךְ בְּתֻמּ֑וֹ מֵעִקֵּ֥שׁ שְׂ֝פָתָ֗יו וְה֣וּא כְסִֽיל׃
2בְּלֹא־ דַ֣עַת נֶ֣פֶשׁ לֹא־ ט֑וֹב וְאָ֖ץ בְּרַגְלַ֣יִם חוֹטֵֽא׃
3אָ֭דָם תְּסַלֵּ֣ף דַּרְכּ֑וֹ וְעַל־ יְ֝הוָ֗ה יִזְעַ֥ף לִבּֽוֹ׃
4יֹ֭סִיף רֵעִ֣ים רַבִּ֑ים וְ֝דָ֗ל מֵרֵ֥עהוּ יִפָּרֵֽד׃
5שְׁ֭קָרִים לֹ֣א יִנָּקֶ֑ה וְיָפִ֥יחַ כְּ֝זָבִ֗ים לֹ֣א יִמָּלֵֽט׃
6יְחַלּ֣וּ פְנֵֽי־ נָדִ֑יב וְכָל־ הָ֝רֵ֗עַ לְאִ֣ישׁ מַתָּֽן׃
7אֲחֵי־ רָ֨שׁ ׀ שְֽׂנֵאֻ֗הוּ אַ֤ף כִּ֣י מְ֭רֵעֵהוּ רָחֲק֣וּ מִמֶּ֑נּוּ מְרַדֵּ֖ף אֲמָרִ֣ים [לֹא־ כ] (לֹו־ הֵֽמָּה׃ ק)
8לֵּ֭ב אֹהֵ֣ב נַפְשׁ֑וֹ שֹׁמֵ֥ר תְּ֝בוּנָ֗ה לִמְצֹא־ טֽוֹב׃
9שְׁ֭קָרִים לֹ֣א יִנָּקֶ֑ה וְיָפִ֖יחַ כְּזָבִ֣ים יֹאבֵֽד׃ פ
10נָאוֶ֣ה לִכְסִ֣יל תַּעֲנ֑וּג אַ֝֗ף כִּֽי־ לְעֶ֤בֶד ׀ מְשֹׁ֬ל בְּשָׂרִֽים׃
11אָ֭דָם הֶאֱרִ֣יךְ אַפּ֑וֹ וְ֝תִפאַרְתּ֗וֹ עֲבֹ֣ר עַל־ פָּֽשַׁע׃
12כַּ֭כְּפִיר זַ֣עַף מֶ֑לֶךְ וּכְטַ֖ל עַל־ עֵ֣שֶׂב רְצוֹנֽוֹ׃
13לְ֭אָבִיו בֵּ֣ן כְּסִ֑יל וְדֶ֥לֶף טֹ֝רֵ֗ד מִדְיְנֵ֥י אִשָּֽׁה׃
14וָ֭הוֹן נַחֲלַ֣ת אָב֑וֹת וּ֝מֵיְהוָ֗ה אִשָּׁ֥ה מַשְׂכָּֽלֶת׃
15תַּפִּ֣יל תַּרְדֵּמָ֑ה וְנֶ֖פֶשׁ רְמִיָּ֣ה תִרְעָֽב׃
16מִ֭צְוָה שֹׁמֵ֣ר נַפְשׁ֑וֹ בּוֹזֵ֖ה דְרָכָ֣יו [יוּמָת כ] (יָמֽוּת׃ ק)
17יְ֭הוָה ח֣וֹנֵֽן דָּ֑ל וּ֝גְמֻל֗וֹ יְשַׁלֶּם־ לֽוֹ׃
18בִּ֭נְךָ כִּי־ יֵ֣שׁ תִּקְוָ֑ה וְאֶל־ הֲ֝מִית֗וֹ אַל־ תִּשָּׂ֥א נַפְשֶֽׁךָ׃
19כ] (גְּֽדָל־ חֵ֭מָה ק) נֹ֣שֵׂא עֹ֑נֶשׁ כִּ֥י אִם־ תַּ֝צִּ֗יל וְע֣וֹד תּוֹסִֽף׃
20עֵ֭צָה וְקַבֵּ֣ל מוּסָ֑ר לְ֝מַ֗עַן תֶּחְכַּ֥ם בְּאַחֲרִיתֶֽךָ׃
21מַחֲשָׁב֣וֹת בְּלֶב־ אִ֑ישׁ וַעֲצַ֥ת יְ֝הוָ֗ה הִ֣יא תָקֽוּם׃
22אָדָ֣ם חַסְדּ֑וֹ וְטֽוֹב־ רָ֝שׁ מֵאִ֥ישׁ כָּזָֽב׃
23יְהוָ֣ה לְחַיִּ֑ים וְשָׂבֵ֥עַ יָ֝לִ֗ין בַּל־ יִפָּ֥קֶד רָֽע׃
24עָצֵ֣ל יָ֭דוֹ בַּצַּלָּ֑חַת גַּם־ אֶל־ פִּ֝֗יהוּ לֹ֣א יְשִׁיבֶֽנָּה׃
25תַּ֭כֶּה וּפֶ֣תִי יַעְרִ֑ם וְהוֹכִ֥יחַ לְ֝נָב֗וֹן יָבִ֥ין דָּֽעַת׃
26אָ֭ב יַבְרִ֣יחַ אֵ֑ם בֵּ֝֗ן מֵבִ֥ישׁ וּמַחְפִּֽיר׃
27בְּ֭נִי לִשְׁמֹ֣עַ מוּסָ֑ר לִ֝שְׁג֗וֹת מֵֽאִמְרֵי־ דָֽעַת׃
28בְּ֭לִיַּעַל יָלִ֣יץ מִשְׁפָּ֑ט וּפִ֥י רְ֝שָׁעִ֗ים יְבַלַּע־ אָֽוֶן׃
