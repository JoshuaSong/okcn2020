1בְּ֭נִי לְחָכְמָתִ֣י הַקְשִׁ֑יבָה לִ֝תְבוּנָתִ֗י הַט־ אָזְנֶֽךָ׃
2מְזִמּ֑וֹת וְ֝דַ֗עַת שְׂפָתֶ֥יךָ יִנְצֹֽרוּ׃
3נֹ֣פֶת תִּ֭טֹּפְנָה שִׂפְתֵ֣י זָרָ֑ה וְחָלָ֖ק מִשֶּׁ֣מֶן חִכָּֽהּ׃
4מָרָ֣ה כַֽלַּעֲנָ֑ה חַ֝דָּ֗ה כְּחֶ֣רֶב פִּיּֽוֹת׃
5יֹרְד֣וֹת מָ֑וֶת שְׁ֝א֗וֹל צְעָדֶ֥יהָ יִתְמֹֽכוּ׃
6חַ֭יִּים פֶּן־ תְּפַלֵּ֑ס נָע֥וּ מַ֝עְגְּלֹתֶ֗יהָ לֹ֣א תֵדָֽע׃ פ
7בָ֭נִים שִׁמְעוּ־ לִ֑י וְאַל־ תָּ֝ס֗וּרוּ מֵאִמְרֵי־ פִֽי׃
8מֵעָלֶ֣יהָ דַרְכֶּ֑ךָ וְאַל־ תִּ֝קְרַ֗ב אֶל־ פֶּ֥תַח בֵּיתָֽהּ׃
9תִּתֵּ֣ן לַאֲחֵרִ֣ים הוֹדֶ֑ךָ וּ֝שְׁנֹתֶ֗יךָ לְאַכְזָרִֽי׃
10יִשְׂבְּע֣וּ זָרִ֣ים כֹּחֶ֑ךָ וַ֝עֲצָבֶ֗יךָ בְּבֵ֣ית נָכְרִֽי׃
11בְאַחֲרִיתֶ֑ךָ בִּכְל֥וֹת בְּ֝שָׂרְךָ֗ וּשְׁאֵרֶֽךָ׃
12אֵ֭יךְ שָׂנֵ֣אתִי מוּסָ֑ר וְ֝תוֹכַ֗חַת נָאַ֥ץ לִבִּֽי׃
13שָׁ֭מַעְתִּי בְּק֣וֹל מוֹרָ֑י וְ֝לִֽמְלַמְּדַ֗י לֹא־ הִטִּ֥יתִי אָזְנִֽי׃
14הָיִ֣יתִי בְכָל־ רָ֑ע בְּת֖וֹךְ קָהָ֣ל וְעֵדָֽה׃
15מַ֥יִם מִבּוֹרֶ֑ךָ וְ֝נֹזְלִ֗ים מִתּ֥וֹךְ בְּאֵרֶֽךָ׃
16מַעְיְנֹתֶ֣יךָ ח֑וּצָה בָּ֝רְחֹב֗וֹת פַּלְגֵי־ מָֽיִם׃
17לְךָ֥ לְבַדֶּ֑ךָ וְאֵ֖ין לְזָרִ֣ים אִתָּֽךְ׃
18מְקוֹרְךָ֥ בָר֑וּךְ וּ֝שְׂמַ֗ח מֵאֵ֥שֶׁת נְעוּרֶֽךָ׃
19אֲהָבִ֗ים וְֽיַעֲלַ֫ת־ חֵ֥ן דַּ֭דֶּיהָ יְרַוֻּ֣ךָ בְכָל־ עֵ֑ת בְּ֝אַהֲבָתָ֗הּ תִּשְׁגֶּ֥ה תָמִֽיד׃
20תִשְׁגֶּ֣ה בְנִ֣י בְזָרָ֑ה וּ֝תְחַבֵּ֗ק חֵ֣ק נָכְרִיָּֽה׃
21נֹ֨כַח ׀ עֵינֵ֣י יְ֭הוָה דַּרְכֵי־ אִ֑ישׁ וְֽכָל־ מַעְגְּלֹתָ֥יו מְפַלֵּֽס׃
22יִלְכְּדֻנ֥וֹ אֶת־ הָרָשָׁ֑ע וּבְחַבְלֵ֥י חַ֝טָּאת֗וֹ יִתָּמֵֽךְ׃
