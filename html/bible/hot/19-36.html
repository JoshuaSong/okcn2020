1לַמְנַצֵּ֬חַ ׀ לְעֶֽבֶד־ יְהוָ֬ה לְדָוִֽד׃ נְאֻֽם־ פֶּ֣שַׁע לָ֭רָשָׁע בְּקֶ֣רֶב לִבִּ֑י אֵֽין־ פַּ֥חַד אֱ֝לֹהִ֗ים לְנֶ֣גֶד עֵינָֽיו׃
2הֶחֱלִ֣יק אֵלָ֣יו בְּעֵינָ֑יו לִמְצֹ֖א עֲוֹנ֣וֹ לִשְׂנֹֽא׃
3פִ֭יו אָ֣וֶן וּמִרְמָ֑ה חָדַ֖ל לְהַשְׂכִּ֣יל לְהֵיטִֽיב׃
4יַחְשֹׁ֗ב עַֽל־ מִשְׁכָּ֫ב֥וֹ יִ֭תְיַצֵּב עַל־ דֶּ֣רֶךְ לֹא־ ט֑וֹב רָ֝֗ע לֹ֣א יִמְאָֽס׃
5בְּהַשָּׁמַ֣יִם חַסְדֶּ֑ךָ אֱ֝מֽוּנָתְךָ֗ עַד־ שְׁחָקִֽים׃
6כְּֽהַרְרֵי־ אֵ֗ל מִ֭שְׁפָּטֶךָ תְּה֣וֹם רַבָּ֑ה אָ֤דָֽם־ וּבְהֵמָ֖ה תוֹשִׁ֣יעַ יְהוָֽה׃
7יָּקָ֥ר חַסְדְּךָ֗ אֱלֹ֫הִ֥ים וּבְנֵ֥י אָדָ֑ם בְּצֵ֥ל כְּ֝נָפֶ֗יךָ יֶחֱסָיֽוּן׃
8מִדֶּ֣שֶׁן בֵּיתֶ֑ךָ וְנַ֖חַל עֲדָנֶ֣יךָ תַשְׁקֵֽם׃
9עִ֭מְּךָ מְק֣וֹר חַיִּ֑ים בְּ֝אוֹרְךָ֗ נִרְאֶה־ אֽוֹר׃
10חַ֭סְדְּךָ לְיֹדְעֶ֑יךָ וְ֝צִדְקָֽתְךָ֗ לְיִשְׁרֵי־ לֵֽב׃
11תְּ֭בוֹאֵנִי רֶ֣גֶל גַּאֲוָ֑ה וְיַד־ רְ֝שָׁעִ֗ים אַל־ תְּנִדֵֽנִי׃
