1וַ֭יַּעַן אֱלִיפַ֥ז הַֽתֵּמָנִ֗י וַיֹּאמַֽר׃
2יִסְכָּן־ גָּ֑בֶר כִּֽי־ יִסְכֹּ֖ן עָלֵ֣ימוֹ מַשְׂכִּֽיל׃
3לְ֭שַׁדַּי כִּ֣י תִצְדָּ֑ק וְאִם־ בֶּ֝֗צַע כִּֽי־ תַתֵּ֥ם דְּרָכֶֽיךָ׃
4יֹכִיחֶ֑ךָ יָב֥וֹא עִ֝מְּךָ֗ בַּמִּשְׁפָּֽט׃
5רָעָֽתְךָ֣ רַבָּ֑ה וְאֵֽין־ קֵ֝֗ץ לַעֲוֹנֹתֶֽיךָ׃
6תַחְבֹּ֣ל אַחֶ֣יךָ חִנָּ֑ם וּבִגְדֵ֖י עֲרוּמִּ֣ים תַּפְשִֽׁיט׃
7מַ֭יִם עָיֵ֣ף תַּשְׁקֶ֑ה וּ֝מֵרָעֵ֗ב תִּֽמְנַֽע־ לָֽחֶם׃
8זְ֭רוֹעַ ל֣וֹ הָאָ֑רֶץ וּנְשׂ֥וּא פָ֝נִ֗ים יֵ֣שֶׁב בָּֽהּ׃
9שִׁלַּ֣חְתָּ רֵיקָ֑ם וּזְרֹע֖וֹת יְתֹמִ֣ים יְדֻכָּֽא׃
10כֵּ֭ן סְבִיבוֹתֶ֣יךָ פַחִ֑ים וִֽ֝יבַהֶלְךָ פַּ֣חַד פִּתְאֹֽם׃
11חֹ֥שֶׁךְ לֹֽא־ תִרְאֶ֑ה וְֽשִׁפְעַת־ מַ֥יִם תְּכַסֶּֽךָּ׃
12אֱ֭לוֹהַּ גֹּ֣בַהּ שָׁמָ֑יִם וּרְאֵ֤ה רֹ֖אשׁ כּוֹכָבִ֣ים כִּי־ רָֽמּוּ׃
13מַה־ יָּ֣דַֽע אֵ֑ל הַבְעַ֖ד עֲרָפֶ֣ל יִשְׁפּֽוֹט׃
14סֵֽתֶר־ ל֭וֹ וְלֹ֣א יִרְאֶ֑ה וְח֥וּג שָׁ֝מַ֗יִם יִתְהַלָּֽךְ׃
15עוֹלָ֣ם תִּשְׁמֹ֑ר אֲשֶׁ֖ר דָּרְכ֣וּ מְתֵי־ אָֽוֶן׃
16קֻמְּט֥וּ וְלֹא־ עֵ֑ת נָ֝הָ֗ר יוּצַ֥ק יְסוֹדָֽם׃
17לָ֭אֵל ס֣וּר מִמֶּ֑נּוּ וּמַה־ יִּפְעַ֖ל שַׁדַּ֣י לָֽמוֹ׃
18מִלֵּ֣א בָתֵּיהֶ֣ם ט֑וֹב וַעֲצַ֥ת רְ֝שָׁעִ֗ים רָ֣חֲקָה מֶֽנִּי׃
19צַדִּיקִ֣ים וְיִשְׂמָ֑חוּ וְ֝נָקִ֗י יִלְעַג־ לָֽמוֹ׃
20לֹ֣א נִכְחַ֣ד קִימָ֑נוּ וְ֝יִתְרָ֗ם אָ֣כְלָה אֵֽשׁ׃
21נָ֣א עִמּ֑וֹ וּשְׁלם בָּ֝הֶ֗ם תְּֽבוֹאַתְךָ֥ טוֹבָֽה׃
22נָ֣א מִפִּ֣יו תּוֹרָ֑ה וְשִׂ֥ים אֲ֝מָרָ֗יו בִּלְבָבֶֽךָ׃
23תָּשׁ֣וּב עַד־ שַׁ֭דַּי תִּבָּנֶ֑ה תַּרְחִ֥יק עַ֝וְלָ֗ה מֵאָהֳלֶֽךָ׃ וְשִׁית־ עַל־
24בָּ֑צֶר וּבְצ֖וּר נְחָלִ֣ים אוֹפִֽיר׃
25שַׁדַּ֣י בְּצָרֶ֑יךָ וְכֶ֖סֶף תּוֹעָפ֣וֹת לָֽךְ׃
26אָ֭ז עַל־ שַׁדַּ֣י תִּתְעַנָּ֑ג וְתִשָּׂ֖א אֶל־ אֱל֣וֹהַּ פָּנֶֽיךָ׃
27אֵ֭לָיו וְיִשְׁמָעֶ֑ךָּ וּנְדָרֶ֥יךָ תְשַׁלֵּֽם׃
28א֭וֹמֶר וְיָ֣קָם לָ֑ךְ וְעַל־ דְּ֝רָכֶ֗יךָ נָ֣גַֽהּ אֽוֹר׃
29הִ֭שְׁפִּילוּ וַתֹּ֣אמֶר גֵּוָ֑ה וְשַׁ֖ח עֵינַ֣יִם יוֹשִֽׁעַ׃
