1לְדָוִ֨ד ׀ בָּרֲכִ֣י נַ֭פְשִׁי אֶת־ יְהוָ֑ה וְכָל־ קְ֝רָבַ֗י אֶת־ שֵׁ֥ם קָדְשֽׁוֹ׃
2נַ֭פְשִׁי אֶת־ יְהוָ֑ה וְאַל־ תִּ֝שְׁכְּחִ֗י כָּל־ גְּמוּלָֽיו׃
3לְכָל־ עֲוֹנֵ֑כִי הָ֝רֹפֵ֗א לְכָל־ תַּחֲלֻאָֽיְכִי׃
4מִשַּׁ֣חַת חַיָּ֑יְכִי הַֽ֝מְעַטְּרֵ֗כִי חֶ֣סֶד וְרַחֲמִֽים׃
5בַּטּ֣וֹב עֶדְיֵ֑ךְ תִּתְחַדֵּ֖שׁ כַּנֶּ֣שֶׁר נְעוּרָֽיְכִי׃
6צְדָק֣וֹת יְהוָ֑ה וּ֝מִשְׁפָּטִ֗ים לְכָל־ עֲשׁוּקִֽים׃
7דְּרָכָ֣יו לְמֹשֶׁ֑ה לִבְנֵ֥י יִ֝שְׂרָאֵ֗ל עֲלִילֽוֹתָיו׃
8וְחַנּ֣וּן יְהוָ֑ה אֶ֖רֶךְ אַפַּ֣יִם וְרַב־ חָֽסֶד׃
9לָנֶ֥צַח יָרִ֑יב וְלֹ֖א לְעוֹלָ֣ם יִטּֽוֹר׃
10כַ֭חֲטָאֵינוּ עָ֣שָׂה לָ֑נוּ וְלֹ֥א כַ֝עֲוֹנֹתֵ֗ינוּ גָּמַ֥ל עָלֵֽינוּ׃
11כִגְבֹ֣הַּ שָׁ֭מַיִם עַל־ הָאָ֑רֶץ גָּבַ֥ר חַ֝סְדּ֗וֹ עַל־ יְרֵאָֽיו׃
12מִ֭זְרָח מִֽמַּֽעֲרָ֑ב הִֽרְחִ֥יק מִ֝מֶּ֗נּוּ אֶת־ פְּשָׁעֵֽינוּ׃
13אָ֭ב עַל־ בָּנִ֑ים רִחַ֥ם יְ֝הוָ֗ה עַל־ יְרֵאָֽיו׃
14ה֖וּא יָדַ֣ע יִצְרֵ֑נוּ זָ֝כ֗וּר כִּי־ עָפָ֥ר אֲנָֽחְנוּ׃
15כֶּחָצִ֣יר יָמָ֑יו כְּצִ֥יץ הַ֝שָּׂדֶ֗ה כֵּ֣ן יָצִֽיץ׃
16ר֣וּחַ עָֽבְרָה־ בּ֣וֹ וְאֵינֶ֑נּוּ וְלֹא־ יַכִּירֶ֖נּוּ ע֣וֹד מְקוֹמֽוֹ׃
17יְהוָ֨ה ׀ מֵעוֹלָ֣ם וְעַד־ ע֭וֹלָם עַל־ יְרֵאָ֑יו וְ֝צִדְקָת֗וֹ לִבְנֵ֥י בָנִֽים׃
18בְרִית֑וֹ וּלְזֹכְרֵ֥י פִ֝קֻּדָ֗יו לַעֲשׂוֹתָֽם׃
19בַּ֭שָּׁמַיִם הֵכִ֣ין כִּסְא֑וֹ וּ֝מַלְכוּת֗וֹ בַּכֹּ֥ל מָשָֽׁלָה׃
20יְהוָ֗ה מַלְאָ֫כָ֥יו גִּבֹּ֣רֵי כֹ֭חַ עֹשֵׂ֣י דְבָר֑וֹ לִ֝שְׁמֹ֗עַ בְּק֣וֹל דְּבָרֽוֹ׃
21יְ֭הוָה כָּל־ צְבָאָ֑יו מְ֝שָׁרְתָ֗יו עֹשֵׂ֥י רְצוֹנֽוֹ׃
