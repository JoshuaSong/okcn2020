1וְכָל־ יִשְׂרָאֵל֙ הִתְיַחְשׂ֔וּ וְהִנָּ֣ם כְּתוּבִ֔ים עַל־ סֵ֖פֶר מַלְכֵ֣י יִשְׂרָאֵ֑ל וִיהוּדָ֛ה הָגְל֥וּ לְבָבֶ֖ל בְּמַעֲלָֽם׃ ס
2הָרִ֣אשֹׁנִ֔ים אֲשֶׁ֥ר בַּאֲחֻזָּתָ֖ם בְּעָרֵיהֶ֑ם יִשְׂרָאֵל֙ הַכֹּ֣הֲנִ֔ים הַלְוִיִּ֖ם וְהַנְּתִינִֽים׃
3יָשְׁב֔וּ מִן־ בְּנֵ֥י יְהוּדָ֖ה וּמִן־ בְּנֵ֣י בִנְיָמִ֑ן וּמִן־ בְּנֵ֥י אֶפְרַ֖יִם וּמְנַשֶּֽׁה׃
4בֶּן־ עַמִּיה֤וּד בֶּן־ עָמְרִי֙ בֶּן־ אִמְרִ֣י בֶן־ [בָּנִימִן־ כ] (בָּנִ֔י ק) (מִן־ בְּנֵי־ פֶ֖רֶץ ק) בֶּן־ יְהוּדָֽה׃
5הַשִּׁ֣ילוֹנִ֔י עֲשָׂיָ֥ה הַבְּכ֖וֹר וּבָנָֽיו׃
6בְּנֵי־ זֶ֖רַח יְעוּאֵ֑ל וַאֲחֵיהֶ֖ם שֵׁשׁ־ מֵא֥וֹת וְתִשְׁעִֽים׃
7בְּנֵ֖י בִּנְיָמִ֑ן סַלּוּא֙ בֶּן־ מְשֻׁלָּ֔ם בֶּן־ הוֹדַוְיָ֖ה בֶּן־ הַסְּנֻאָֽה׃
8בֶּן־ יְרֹחָ֔ם וְאֵלָ֥ה בֶן־ עֻזִּ֖י בֶּן־ מִכְרִ֑י וּמְשֻׁלָּם֙ בֶּן־ שְׁפַטְיָ֔ה בֶּן־ רְעוּאֵ֖ל בֶּן־ יִבְנִיָּֽה׃
9לְתֹ֣לְדוֹתָ֔ם תְּשַׁ֥ע מֵא֖וֹת וַחֲמִשִּׁ֣ים וְשִׁשָּׁ֑ה כָּל־ אֵ֣לֶּה אֲנָשִׁ֔ים רָאשֵׁ֥י אָב֖וֹת לְבֵ֥ית אֲבֹתֵיהֶֽם׃ ס
10הַֽכֹּהֲנִ֑ים יְדַֽעְיָ֥ה וִיהוֹיָרִ֖יב וְיָכִֽין׃
11בֶן־ חִלְקִיָּ֜ה בֶּן־ מְשֻׁלָּ֣ם בֶּן־ צָד֗וֹק בֶּן־ מְרָיוֹת֙ בֶּן־ אֲחִיט֔וּב נְגִ֖יד בֵּ֥ית הָאֱלֹהִֽים׃ ס
12בֶּן־ יְרֹחָ֔ם בֶּן־ פַּשְׁח֖וּר בֶּן־ מַלְכִּיָּ֑ה וּמַעְשַׂ֨י בֶּן־ עֲדִיאֵ֧ל בֶּן־ יַחְזֵ֛רָה בֶּן־ מְשֻׁלָּ֥ם בֶּן־ מְשִׁלֵּמִ֖ית בֶּן־ אִמֵּֽר׃
13רָאשִׁים֙ לְבֵ֣ית אֲבוֹתָ֔ם אֶ֕לֶף וּשְׁבַ֥ע מֵא֖וֹת וְשִׁשִּׁ֑ים גִּבּ֣וֹרֵי חֵ֔יל מְלֶ֖אכֶת עֲבוֹדַ֥ת בֵּית־ הָאֱלֹהִֽים׃
14הַלְוִיִּ֑ם שְׁמַֽעְיָ֧ה בֶן־ חַשּׁ֛וּב בֶּן־ עַזְרִיקָ֥ם בֶּן־ חֲשַׁבְיָ֖ה מִן־ בְּנֵ֥י מְרָרִֽי׃
15חֶ֖רֶשׁ וְגָלָ֑ל וּמַתַּנְיָה֙ בֶּן־ מִיכָ֔א בֶּן־ זִכְרִ֖י בֶּן־ אָסָֽף׃
16בֶּֽן־ שְׁמַֽעְיָ֔ה בֶּן־ גָּלָ֖ל בֶּן־ יְדוּת֑וּן וּבֶרֶכְיָ֤ה בֶן־ אָסָא֙ בֶּן־ אֶלְקָנָ֔ה הַיּוֹשֵׁ֖ב בְּחַצְרֵ֥י נְטוֹפָתִֽי׃
17שַׁלּ֣וּם וְעַקּ֔וּב וְטַלְמֹ֖ן וַאֲחִימָ֑ן וַאֲחִיהֶ֥ם שַׁלּ֖וּם הָרֹֽאשׁ׃
18הֵ֔נָּה בְּשַׁ֥עַר הַמֶּ֖לֶךְ מִזְרָ֑חָה הֵ֚מָּה הַשֹּׁ֣עֲרִ֔ים לְמַחֲנ֖וֹת בְּנֵ֥י לֵוִֽי׃
19בֶּן־ ק֠וֹרֵא בֶּן־ אֶבְיָסָ֨ף בֶּן־ קֹ֜רַח וְֽאֶחָ֧יו לְבֵית־ אָבִ֣יו הַקָּרְחִ֗ים עַ֚ל מְלֶ֣אכֶת הָעֲבוֹדָ֔ה שֹׁמְרֵ֥י הַסִּפִּ֖ים לָאֹ֑הֶל וַאֲבֹֽתֵיהֶם֙ עַל־ מַחֲנֵ֣ה יְהוָ֔ה שֹׁמְרֵ֖י הַמָּבֽוֹא׃
20בֶּן־ אֶלְעָזָ֗ר נָגִ֨יד הָיָ֧ה עֲלֵיהֶ֛ם לְפָנִ֖ים יְהוָ֥ה ׀ עִמּֽוֹ׃
21בֶּ֣ן מְשֶֽׁלֶמְיָ֔ה שֹׁעֵ֥ר פֶּ֖תַח לְאֹ֥הֶל מוֹעֵֽד׃
22הַבְּרוּרִים֙ לְשֹׁעֲרִ֣ים בַּסִּפִּ֔ים מָאתַ֖יִם וּשְׁנֵ֣ים עָשָׂ֑ר הֵ֤מָּה בְחַצְרֵיהֶם֙ הִתְיַחְשָׂ֔ם הֵ֣מָּה יִסַּ֥ד דָּוִ֛יד וּשְׁמוּאֵ֥ל הָרֹאֶ֖ה בֶּאֱמוּנָתָֽם׃
23וּבְנֵיהֶ֜ם עַל־ הַשְּׁעָרִ֧ים לְבֵית־ יְהוָ֛ה לְבֵ֥ית־ הָאֹ֖הֶל לְמִשְׁמָרֽוֹת׃
24רוּח֔וֹת יִהְי֖וּ הַשֹּׁעֲרִ֑ים מִזְרָ֥ח יָ֖מָּה צָפ֥וֹנָה וָנֶֽגְבָּה׃
25בְּחַצְרֵיהֶ֜ם לָב֨וֹא לְשִׁבְעַ֧ת הַיָּמִ֛ים מֵעֵ֥ת אֶל־ עֵ֖ת עִם־ אֵֽלֶּה׃
26בֶאֱמוּנָ֞ה הֵ֗מָּה אַרְבַּ֙עַת֙ גִּבֹּרֵ֣י הַשֹּׁעֲרִ֔ים הֵ֖ם הַלְוִיִּ֑ם וְהָיוּ֙ עַל־ הַלְּשָׁכ֔וֹת וְעַ֥ל הָאֹצְר֖וֹת בֵּ֥ית הָאֱלֹהִֽים׃
27בֵּית־ הָאֱלֹהִ֖ים יָלִ֑ינוּ כִּֽי־ עֲלֵיהֶ֣ם מִשְׁמֶ֔רֶת וְהֵ֥ם עַל־ הַמַּפְתֵּ֖חַ וְלַבֹּ֥קֶר לַבֹּֽקֶר׃
28עַל־ כְּלֵ֣י הָעֲבוֹדָ֑ה כִּֽי־ בְמִסְפָּ֣ר יְבִיא֔וּם וּבְמִסְפָּ֖ר יוֹצִיאֽוּם׃
29מְמֻנִּים֙ עַל־ הַכֵּלִ֔ים וְעַ֖ל כָּל־ כְּלֵ֣י הַקֹּ֑דֶשׁ וְעַל־ הַסֹּ֙לֶת֙ וְהַיַּ֣יִן וְהַשֶּׁ֔מֶן וְהַלְּבוֹנָ֖ה וְהַבְּשָׂמִֽים׃
30בְּנֵי֙ הַכֹּ֣הֲנִ֔ים רֹקְחֵ֥י הַמִּרְקַ֖חַת לַבְּשָׂמִֽים׃
31מִן־ הַלְוִיִּ֔ם ה֥וּא הַבְּכ֖וֹר לְשַׁלֻּ֣ם הַקָּרְחִ֑י בֶּאֱמוּנָ֕ה עַ֖ל מַעֲשֵׂ֥ה הַחֲבִתִּֽים׃
32בְּנֵ֧י הַקְּהָתִ֛י מִן־ אֲחֵיהֶ֖ם עַל־ לֶ֣חֶם הַֽמַּעֲרָ֑כֶת לְהָכִ֖ין שַׁבַּ֥ת שַׁבָּֽת׃ ס
33הַ֠מְשֹׁרְרִים רָאשֵׁ֨י אָב֧וֹת לַלְוִיִּ֛ם בַּלְּשָׁכֹ֖ת [פְּטִירִים כ] (פְּטוּרִ֑ים ק) כִּֽי־ יוֹמָ֥ם וָלַ֛יְלָה עֲלֵיהֶ֖ם בַּמְּלָאכָֽה׃
34רָאשֵׁ֨י הָאָב֧וֹת לַלְוִיִּ֛ם לְתֹלְדוֹתָ֖ם רָאשִׁ֑ים אֵ֖לֶּה יָשְׁב֥וּ בִירוּשָׁלִָֽם׃ פ
35יָשְׁב֥וּ אֲבִֽי־ גִבְע֖וֹן [יְעוּאֵל כ] (יְעִיאֵ֑ל ק) וְשֵׁ֥ם אִשְׁתּ֖וֹ מַעֲכָֽה׃
36הַבְּכ֖וֹר עַבְדּ֑וֹן וְצ֣וּר וְקִ֔ישׁ וּבַ֥עַל וְנֵ֖ר וְנָדָֽב׃
37וְאַחְי֔וֹ וּזְכַרְיָ֖ה וּמִקְלֽוֹת׃
38הוֹלִ֣יד אֶת־ שִׁמְאָ֑ם וְאַף־ הֵ֗ם נֶ֧גֶד אֲחֵיהֶ֛ם יָשְׁב֥וּ בִירֽוּשָׁלִַ֖ם עִם־ אֲחֵיהֶֽם׃ ס
39הוֹלִ֣יד אֶת־ קִ֔ישׁ וְקִ֖ישׁ הוֹלִ֣יד אֶת־ שָׁא֑וּל וְשָׁא֗וּל הוֹלִ֤יד אֶת־ יְהֽוֹנָתָן֙ וְאֶת־ מַלְכִּי־ שׁ֔וּעַ וְאֶת־ אֲבִינָדָ֖ב וְאֶת־ אֶשְׁבָּֽעַל׃
40יְהוֹנָתָ֖ן מְרִ֣יב בָּ֑עַל וּמְרִי־ בַ֖עַל הוֹלִ֥יד אֶת־ מִיכָֽה׃
41מִיכָ֑ה פִּית֥וֹן וָמֶ֖לֶךְ וְתַחְרֵֽעַ׃
42הוֹלִ֣יד אֶת־ יַעְרָ֔ה וְיַעְרָ֗ה הוֹלִ֛יד אֶת־ עָלֶ֥מֶת וְאֶת־ עַזְמָ֖וֶת וְאֶת־ זִמְרִ֑י וְזִמְרִ֖י הוֹלִ֥יד אֶת־ מוֹצָֽא׃
43הוֹלִ֣יד אֶת־ בִּנְעָ֑א וּרְפָיָ֥ה בְנ֛וֹ אֶלְעָשָׂ֥ה בְנ֖וֹ אָצֵ֥ל בְּנֽוֹ׃
