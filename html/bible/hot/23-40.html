1נַחֲמ֥וּ נַחֲמ֖וּ עַמִּ֑י יֹאמַ֖ר אֱלֹהֵיכֶֽם׃
2עַל־ לֵ֤ב יְרֽוּשָׁלִַ֙ם֙ וְקִרְא֣וּ אֵלֶ֔יהָ כִּ֤י מָֽלְאָה֙ צְבָאָ֔הּ כִּ֥י נִרְצָ֖ה עֲוֹנָ֑הּ כִּ֤י לָקְחָה֙ מִיַּ֣ד יְהוָ֔ה כִּפְלַ֖יִם בְּכָל־ חַטֹּאתֶֽיהָ׃ ס
3קוֹרֵ֔א בַּמִּדְבָּ֕ר פַּנּ֖וּ דֶּ֣רֶךְ יְהוָ֑ה יַשְּׁרוּ֙ בָּעֲרָבָ֔ה מְסִלָּ֖ה לֵאלֹהֵֽינוּ׃
4גֶּיא֙ יִנָּשֵׂ֔א וְכָל־ הַ֥ר וְגִבְעָ֖ה יִשְׁפָּ֑לוּ וְהָיָ֤ה הֶֽעָקֹב֙ לְמִישׁ֔וֹר וְהָרְכָסִ֖ים לְבִקְעָֽה׃
5כְּב֣וֹד יְהוָ֑ה וְרָא֤וּ כָל־ בָּשָׂר֙ יַחְדָּ֔ו כִּ֛י פִּ֥י יְהוָ֖ה דִּבֵּֽר׃ ס
6אֹמֵ֣ר קְרָ֔א וְאָמַ֖ר מָ֣ה אֶקְרָ֑א כָּל־ הַבָּשָׂ֣ר חָצִ֔יר וְכָל־ חַסְדּ֖וֹ כְּצִ֥יץ הַשָּׂדֶֽה׃
7חָצִיר֙ נָ֣בֵֽל צִ֔יץ כִּ֛י ר֥וּחַ יְהוָ֖ה נָ֣שְׁבָה בּ֑וֹ אָכֵ֥ן חָצִ֖יר הָעָֽם׃
8חָצִ֖יר נָ֣בֵֽל צִ֑יץ וּדְבַר־ אֱלֹהֵ֖ינוּ יָק֥וּם לְעוֹלָֽם׃ ס
9הַר־ גָּבֹ֤הַ עֲלִי־ לָךְ֙ מְבַשֶּׂ֣רֶת צִיּ֔וֹן הָרִ֤ימִי בַכֹּ֙חַ֙ קוֹלֵ֔ךְ מְבַשֶּׂ֖רֶת יְרוּשָׁלִָ֑ם הָרִ֙ימִי֙ אַל־ תִּירָ֔אִי אִמְרִי֙ לְעָרֵ֣י יְהוּדָ֔ה הִנֵּ֖ה אֱלֹהֵיכֶֽם׃
10אֲדֹנָ֤י יְהוִה֙ בְּחָזָ֣ק יָב֔וֹא וּזְרֹע֖וֹ מֹ֣שְׁלָה ל֑וֹ הִנֵּ֤ה שְׂכָרוֹ֙ אִתּ֔וֹ וּפְעֻלָּת֖וֹ לְפָנָֽיו׃
11עֶדְר֣וֹ יִרְעֶ֔ה בִּזְרֹעוֹ֙ יְקַבֵּ֣ץ טְלָאִ֔ים וּבְחֵיק֖וֹ יִשָּׂ֑א עָל֖וֹת יְנַהֵֽל׃ ס
12מָדַ֨ד בְּשָׁעֳל֜וֹ מַ֗יִם וְשָׁמַ֙יִם֙ בַּזֶּ֣רֶת תִּכֵּ֔ן וְכָ֥ל בַּשָּׁלִ֖שׁ עֲפַ֣ר הָאָ֑רֶץ וְשָׁקַ֤ל בַּפֶּ֙לֶס֙ הָרִ֔ים וּגְבָע֖וֹת בְּמֹאזְנָֽיִם׃
13תִכֵּ֥ן אֶת־ ר֖וּחַ יְהוָ֑ה וְאִ֥ישׁ עֲצָת֖וֹ יוֹדִיעֶֽנּוּ׃
14מִ֤י נוֹעָץ֙ וַיְבִינֵ֔הוּ וַֽיְלַמְּדֵ֖הוּ בְּאֹ֣רַח מִשְׁפָּ֑ט וַיְלַמְּדֵ֣הוּ דַ֔עַת וְדֶ֥רֶךְ תְּבוּנ֖וֹת יוֹדִיעֶֽנּוּ׃
15גּוֹיִם֙ כְּמַ֣ר מִדְּלִ֔י וּכְשַׁ֥חַק מֹאזְנַ֖יִם נֶחְשָׁ֑בוּ הֵ֥ן אִיִּ֖ים כַּדַּ֥ק יִטּֽוֹל׃
16אֵ֥ין דֵּ֖י בָּעֵ֑ר וְחַיָּת֔וֹ אֵ֥ין דֵּ֖י עוֹלָֽה׃ ס
17הַגּוֹיִ֖ם כְּאַ֣יִן נֶגְדּ֑וֹ מֵאֶ֥פֶס וָתֹ֖הוּ נֶחְשְׁבוּ־ לֽוֹ׃
18מִ֖י תְּדַמְּי֣וּן אֵ֑ל וּמַה־ דְּמ֖וּת תַּ֥עַרְכוּ לֽוֹ׃
19נָסַ֣ךְ חָרָ֔שׁ וְצֹרֵ֖ף בַּזָּהָ֣ב יְרַקְּעֶ֑נּוּ וּרְתֻק֥וֹת כֶּ֖סֶף צוֹרֵֽף׃
20תְּרוּמָ֔ה עֵ֥ץ לֹֽא־ יִרְקַ֖ב יִבְחָ֑ר חָרָ֤שׁ חָכָם֙ יְבַקֶּשׁ־ ל֔וֹ לְהָכִ֥ין פֶּ֖סֶל לֹ֥א יִמּֽוֹט׃
21תֵֽדְעוּ֙ הֲל֣וֹא תִשְׁמָ֔עוּ הֲל֛וֹא הֻגַּ֥ד מֵרֹ֖אשׁ לָכֶ֑ם הֲלוֹא֙ הֲבִ֣ינֹתֶ֔ם מוֹסְד֖וֹת הָאָֽרֶץ׃
22עַל־ ח֣וּג הָאָ֔רֶץ וְיֹשְׁבֶ֖יהָ כַּחֲגָבִ֑ים הַנּוֹטֶ֤ה כַדֹּק֙ שָׁמַ֔יִם וַיִּמְתָּחֵ֥ם כָּאֹ֖הֶל לָשָֽׁבֶת׃
23רוֹזְנִ֖ים לְאָ֑יִן שֹׁ֥פְטֵי אֶ֖רֶץ כַּתֹּ֥הוּ עָשָֽׂה׃
24בַּל־ נִטָּ֗עוּ אַ֚ף בַּל־ זֹרָ֔עוּ אַ֛ף בַּל־ שֹׁרֵ֥שׁ בָּאָ֖רֶץ גִּזְעָ֑ם וְגַם־ נָשַׁ֤ף בָּהֶם֙ וַיִּבָ֔שׁוּ וּסְעָרָ֖ה כַּקַּ֥שׁ תִּשָּׂאֵֽם׃ ס
25מִ֥י תְדַמְּי֖וּנִי וְאֶשְׁוֶ֑ה יֹאמַ֖ר קָדֽוֹשׁ׃
26מָר֨וֹם עֵינֵיכֶ֤ם וּרְאוּ֙ מִי־ בָרָ֣א אֵ֔לֶּה הַמּוֹצִ֥יא בְמִסְפָּ֖ר צְבָאָ֑ם לְכֻלָּם֙ בְּשֵׁ֣ם יִקְרָ֔א מֵרֹ֤ב אוֹנִים֙ וְאַמִּ֣יץ כֹּ֔חַ אִ֖ישׁ לֹ֥א נֶעְדָּֽר׃ ס
27תֹאמַר֙ יַֽעֲקֹ֔ב וּתְדַבֵּ֖ר יִשְׂרָאֵ֑ל נִסְתְּרָ֤ה דַרְכִּי֙ מֵיְהוָ֔ה וּמֵאֱלֹהַ֖י מִשְׁפָּטִ֥י יַעֲבֽוֹר׃
28יָדַ֜עְתָּ אִם־ לֹ֣א שָׁמַ֗עְתָּ אֱלֹהֵ֨י עוֹלָ֤ם ׀ יְהוָה֙ בּוֹרֵא֙ קְצ֣וֹת הָאָ֔רֶץ לֹ֥א יִיעַ֖ף וְלֹ֣א יִיגָ֑ע אֵ֥ין חֵ֖קֶר לִתְבוּנָתֽוֹ׃
29לַיָּעֵ֖ף כֹּ֑חַ וּלְאֵ֥ין אוֹנִ֖ים עָצְמָ֥ה יַרְבֶּֽה׃
30נְעָרִ֖ים וְיִגָ֑עוּ וּבַחוּרִ֖ים כָּשׁ֥וֹל יִכָּשֵֽׁלוּ׃
