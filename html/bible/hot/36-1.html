1דְּבַר־ יְהוָ֣ה ׀ אֲשֶׁ֣ר הָיָ֗ה אֶל־ צְפַנְיָה֙ בֶּן־ כּוּשִׁ֣י בֶן־ גְּדַלְיָ֔ה בֶּן־ אֲמַרְיָ֖ה בֶּן־ חִזְקִיָּ֑ה בִּימֵ֛י יֹאשִׁיָּ֥הוּ בֶן־ אָמ֖וֹן מֶ֥לֶךְ יְהוּדָֽה׃
2אָסֵ֜ף כֹּ֗ל מֵעַ֛ל פְּנֵ֥י הָאֲדָמָ֖ה נְאֻם־ יְהוָֽה׃
3אָדָ֜ם וּבְהֵמָ֗ה אָסֵ֤ף עוֹף־ הַשָּׁמַ֙יִם֙ וּדְגֵ֣י הַיָּ֔ם וְהַמַּכְשֵׁל֖וֹת אֶת־ הָרְשָׁעִ֑ים וְהִכְרַתִּ֣י אֶת־ הָאָדָ֗ם מֵעַ֛ל פְּנֵ֥י הָאֲדָמָ֖ה נְאֻם־ יְהוָֽה׃
4יָדִי֙ עַל־ יְהוּדָ֔ה וְעַ֖ל כָּל־ יוֹשְׁבֵ֣י יְרוּשָׁלִָ֑ם וְהִכְרַתִּ֞י מִן־ הַמָּק֤וֹם הַזֶּה֙ אֶת־ שְׁאָ֣ר הַבַּ֔עַל אֶת־ שֵׁ֥ם הַכְּמָרִ֖ים עִם־ הַכֹּהֲנִֽים׃
5הַמִּשְׁתַּחֲוִ֥ים עַל־ הַגַּגּ֖וֹת לִצְבָ֣א הַשָּׁמָ֑יִם וְאֶת־ הַמִּֽשְׁתַּחֲוִים֙ הַנִּשְׁבָּעִ֣ים לַֽיהוָ֔ה וְהַנִּשְׁבָּעִ֖ים בְּמַלְכָּֽם׃
6הַנְּסוֹגִ֖ים מֵאַחֲרֵ֣י יְהוָ֑ה וַאֲשֶׁ֛ר לֹֽא־ בִקְשׁ֥וּ אֶת־ יְהוָ֖ה וְלֹ֥א דְרָשֻֽׁהוּ׃
7מִפְּנֵ֖י אֲדֹנָ֣י יְהוִ֑ה כִּ֤י קָרוֹב֙ י֣וֹם יְהוָ֔ה כִּֽי־ הֵכִ֧ין יְהוָ֛ה זֶ֖בַח הִקְדִּ֥ישׁ קְרֻאָֽיו׃
8בְּיוֹם֙ זֶ֣בַח יְהוָ֔ה וּפָקַדְתִּ֥י עַל־ הַשָּׂרִ֖ים וְעַל־ בְּנֵ֣י הַמֶּ֑לֶךְ וְעַ֥ל כָּל־ הַלֹּבְשִׁ֖ים מַלְבּ֥וּשׁ נָכְרִֽי׃
9עַ֧ל כָּל־ הַדּוֹלֵ֛ג עַל־ הַמִּפְתָּ֖ן בַּיּ֣וֹם הַה֑וּא הַֽמְמַלְאִ֛ים בֵּ֥ית אֲדֹנֵיהֶ֖ם חָמָ֥ס וּמִרְמָֽה׃ ס
10בַיּ֨וֹם הַה֜וּא נְאֻם־ יְהוָ֗ה ק֤וֹל צְעָקָה֙ מִשַּׁ֣עַר הַדָּגִ֔ים וִֽילָלָ֖ה מִן־ הַמִּשְׁנֶ֑ה וְשֶׁ֥בֶר גָּד֖וֹל מֵהַגְּבָעֽוֹת׃
11יֹשְׁבֵ֣י הַמַּכְתֵּ֑שׁ כִּ֤י נִדְמָה֙ כָּל־ עַ֣ם כְּנַ֔עַן נִכְרְת֖וּ כָּל־ נְטִ֥ילֵי כָֽסֶף׃
12בָּעֵ֣ת הַהִ֔יא אֲחַפֵּ֥שׂ אֶת־ יְרוּשָׁלִַ֖ם בַּנֵּר֑וֹת וּפָקַדְתִּ֣י עַל־ הָאֲנָשִׁ֗ים הַקֹּֽפְאִים֙ עַל־ שִׁמְרֵיהֶ֔ם הָאֹֽמְרִים֙ בִּלְבָבָ֔ם לֹֽא־ יֵיטִ֥יב יְהוָ֖ה וְלֹ֥א יָרֵֽעַ׃
13חֵילָם֙ לִמְשִׁסָּ֔ה וּבָתֵּיהֶ֖ם לִשְׁמָמָ֑ה וּבָנ֤וּ בָתִּים֙ וְלֹ֣א יֵשֵׁ֔בוּ וְנָטְע֣וּ כְרָמִ֔ים וְלֹ֥א יִשְׁתּ֖וּ אֶת־ יֵינָֽם׃
14יוֹם־ יְהוָה֙ הַגָּד֔וֹל קָר֖וֹב וּמַהֵ֣ר מְאֹ֑ד ק֚וֹל י֣וֹם יְהוָ֔ה מַ֥ר צֹרֵ֖חַ שָׁ֥ם גִּבּֽוֹר׃
15עֶבְרָ֖ה הַיּ֣וֹם הַה֑וּא י֧וֹם צָרָ֣ה וּמְצוּקָ֗ה י֤וֹם שֹׁאָה֙ וּמְשׁוֹאָ֔ה י֥וֹם חֹ֙שֶׁךְ֙ וַאֲפֵלָ֔ה י֥וֹם עָנָ֖ן וַעֲרָפֶֽל׃
16שׁוֹפָ֖ר וּתְרוּעָ֑ה עַ֚ל הֶעָרִ֣ים הַבְּצֻר֔וֹת וְעַ֖ל הַפִּנּ֥וֹת הַגְּבֹהֽוֹת׃
17לָאָדָ֗ם וְהָֽלְכוּ֙ כַּֽעִוְרִ֔ים כִּ֥י לַֽיהוָ֖ה חָטָ֑אוּ וְשֻׁפַּ֤ךְ דָּמָם֙ כֶּֽעָפָ֔ר וּלְחֻמָ֖ם כַּגְּלָלִֽים׃
