1לַמְנַצֵּ֥חַ עַֽל־ יְדוּת֗וּן מִזְמ֥וֹר לְדָוִֽד׃ אַ֣ךְ אֶל־ אֱ֭לֹהִים דּֽוּמִיָּ֣ה נַפְשִׁ֑י מִ֝מֶּ֗נּוּ יְשׁוּעָתִֽי׃
2ה֣וּא צ֭וּרִי וִֽישׁוּעָתִ֑י מִ֝שְׂגַּבִּ֗י לֹא־ אֶמּ֥וֹט רַבָּֽה׃
3אָ֤נָה ׀ תְּהֽוֹתְת֣וּ עַל אִישׁ֮ תְּרָצְּח֪וּ כֻ֫לְּכֶ֥ם כְּקִ֥יר נָט֑וּי גָּ֝דֵ֗ר הַדְּחוּיָֽה׃
4מִשְּׂאֵת֨וֹ ׀ יָעֲצ֣וּ לְהַדִּיחַ֮ יִרְצ֪וּ כָ֫זָ֥ב בְּפִ֥יו יְבָרֵ֑כוּ וּ֝בְקִרְבָּ֗ם יְקַלְלוּ־ סֶֽלָה׃
5לֵ֭אלֹהִים דּ֣וֹמִּי נַפְשִׁ֑י כִּי־ מִ֝מֶּ֗נּוּ תִּקְוָתִֽי׃
6ה֣וּא צ֭וּרִי וִֽישׁוּעָתִ֑י מִ֝שְׂגַּבִּ֗י לֹ֣א אֶמּֽוֹט׃
7אֱ֭לֹהִים יִשְׁעִ֣י וּכְבוֹדִ֑י צוּר־ עֻזִּ֥י מַ֝חְסִ֗י בֵּֽאלֹהִֽים׃
8ב֤וֹ בְכָל־ עֵ֨ת ׀ עָ֗ם שִׁפְכֽוּ־ לְפָנָ֥יו לְבַבְכֶ֑ם אֱלֹהִ֖ים מַחֲסֶה־ לָּ֣נוּ סֶֽלָה׃
9הֶ֥בֶל בְּנֵֽי־ אָדָם֮ כָּזָ֪ב בְּנֵ֫י אִ֥ישׁ בְּמֹאזְנַ֥יִם לַעֲל֑וֹת הֵ֝֗מָּה מֵהֶ֥בֶל יָֽחַד׃
10תִּבְטְח֣וּ בְעֹשֶׁק֮ וּבְגָזֵ֪ל אַל־ תֶּ֫הְבָּ֥לוּ חַ֤יִל ׀ כִּֽי־ יָנ֑וּב אַל־ תָּשִׁ֥יתוּ לֵֽב׃
11דִּבֶּ֬ר אֱלֹהִ֗ים שְׁתַּֽיִם־ ז֥וּ שָׁמָ֑עְתִּי כִּ֥י עֹ֝֗ז לֵאלֹהִֽים׃
