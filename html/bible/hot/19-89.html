1מַ֝שְׂכִּ֗יל לְאֵיתָ֥ן הָֽאֶזְרָחִֽי׃ חַֽסְדֵ֣י יְ֭הוָה עוֹלָ֣ם אָשִׁ֑ירָה לְדֹ֥ר וָדֹ֓ר ׀ אוֹדִ֖יעַ אֱמוּנָתְךָ֣ בְּפִֽי׃
2אָמַ֗רְתִּי ע֭וֹלָם חֶ֣סֶד יִבָּנֶ֑ה שָׁמַ֓יִם ׀ תָּכִ֖ן אֱמוּנָתְךָ֣ בָהֶֽם׃
3בְ֭רִית לִבְחִירִ֑י נִ֝שְׁבַּ֗עְתִּי לְדָוִ֥ד עַבְדִּֽי׃
4ע֭וֹלָם אָכִ֣ין זַרְעֶ֑ךָ וּבָנִ֨יתִי לְדֹר־ וָד֖וֹר כִּסְאֲךָ֣ סֶֽלָה׃
5שָׁמַ֣יִם פִּלְאֲךָ֣ יְהוָ֑ה אַף־ אֱ֝מֽוּנָתְךָ֗ בִּקְהַ֥ל קְדֹשִֽׁים׃
6מִ֣י בַ֭שַּׁחַק יַעֲרֹ֣ךְ לַיהוָ֑ה יִדְמֶ֥ה לַ֝יהוָ֗ה בִּבְנֵ֥י אֵלִים׃
7נַ֭עֲרָץ בְּסוֹד־ קְדֹשִׁ֣ים רַבָּ֑ה וְ֝נוֹרָ֗א עַל־ כָּל־ סְבִיבָֽיו׃
8אֱלֹ֘הֵ֤י צְבָא֗וֹת מִֽי־ כָֽמ֖וֹךָ חֲסִ֥ין ׀ יָ֑הּ וֶ֝אֱמֽוּנָתְךָ֗ סְבִיבוֹתֶֽיךָ׃
9מ֭וֹשֵׁל בְּגֵא֣וּת הַיָּ֑ם בְּשׂ֥וֹא גַ֝לָּ֗יו אַתָּ֥ה תְשַׁבְּחֵֽם׃
10דִכִּ֣אתָ כֶחָלָ֣ל רָ֑הַב בִּזְר֥וֹעַ עֻ֝זְּךָ֗ פִּזַּ֥רְתָּ אוֹיְבֶֽיךָ׃
11שָׁ֭מַיִם אַף־ לְךָ֥ אָ֑רֶץ תֵּבֵ֥ל וּ֝מְלֹאָ֗הּ אַתָּ֥ה יְסַדְתָּֽם׃
12וְ֭יָמִין אַתָּ֣ה בְרָאתָ֑ם תָּב֥וֹר וְ֝חֶרְמ֗וֹן בְּשִׁמְךָ֥ יְרַנֵּֽנוּ׃
13זְ֭רוֹעַ עִם־ גְּבוּרָ֑ה תָּעֹ֥ז יָ֝דְךָ֗ תָּר֥וּם יְמִינֶֽךָ׃
14וּ֭מִשְׁפָּט מְכ֣וֹן כִּסְאֶ֑ךָ חֶ֥סֶד וֶ֝אֱמֶ֗ת יְֽקַדְּמ֥וּ פָנֶֽיךָ׃
15הָ֭עָם יוֹדְעֵ֣י תְרוּעָ֑ה יְ֝הוָ֗ה בְּֽאוֹר־ פָּנֶ֥יךָ יְהַלֵּכֽוּן׃
16יְגִיל֣וּן כָּל־ הַיּ֑וֹם וּבְצִדְקָתְךָ֥ יָרֽוּמוּ׃
17תִפְאֶ֣רֶת עֻזָּ֣מוֹ אָ֑תָּה וּ֝בִרְצֹנְךָ֗ [תָּרִים כ] (תָּר֥וּם ק) קַרְנֵֽנוּ׃
18לַֽ֭יהוָה מָֽגִנֵּ֑נוּ וְלִקְד֖וֹשׁ יִשְׂרָאֵ֣ל מַלְכֵּֽנוּ׃
19דִּבַּ֥רְתָּֽ־ בְחָ֡זוֹן לַֽחֲסִידֶ֗יךָ וַתֹּ֗אמֶר שִׁוִּ֣יתִי עֵ֭זֶר עַל־ גִּבּ֑וֹר הֲרִימ֖וֹתִי בָח֣וּר מֵעָֽם׃
20דָּוִ֣ד עַבְדִּ֑י בְּשֶׁ֖מֶן קָדְשִׁ֣י מְשַׁחְתִּֽיו׃
21יָ֭דִי תִּכּ֣וֹן עִמּ֑וֹ אַף־ זְרוֹעִ֥י תְאַמְּצֶֽנּוּ׃
22יַשִּׁ֣א אוֹיֵ֣ב בּ֑וֹ וּבֶן־ עַ֝וְלָ֗ה לֹ֣א יְעַנֶּֽנּוּ׃
23מִפָּנָ֣יו צָרָ֑יו וּמְשַׂנְאָ֥יו אֶגּֽוֹף׃
24וְחַסְדִּ֣י עִמּ֑וֹ וּ֝בִשְׁמִ֗י תָּר֥וּם קַרְנֽוֹ׃
25בַיָּ֣ם יָד֑וֹ וּֽבַנְּהָר֥וֹת יְמִינֽוֹ׃
26יִ֭קְרָאֵנִי אָ֣בִי אָ֑תָּה אֵ֝לִ֗י וְצ֣וּר יְשׁוּעָתִֽי׃
27אָ֭נִי בְּכ֣וֹר אֶתְּנֵ֑הוּ עֶ֝לְי֗וֹן לְמַלְכֵי־ אָֽרֶץ׃
28[אֶשְׁמֹור־ כ] (אֶשְׁמָר־ ל֣וֹ ק) חַסְדִּ֑י וּ֝בְרִיתִ֗י נֶאֱמֶ֥נֶת לֽוֹ׃
29לָעַ֣ד זַרְע֑וֹ וְ֝כִסְא֗וֹ כִּימֵ֥י שָׁמָֽיִם׃
30יַֽעַזְב֣וּ בָ֭נָיו תּוֹרָתִ֑י וּ֝בְמִשְׁפָּטַ֗י לֹ֣א יֵלֵכֽוּן׃
31חֻקֹּתַ֥י יְחַלֵּ֑לוּ וּ֝מִצְוֹתַ֗י לֹ֣א יִשְׁמֹֽרוּ׃
32בְשֵׁ֣בֶט פִּשְׁעָ֑ם וּבִנְגָעִ֥ים עֲוֹנָֽם׃
33לֹֽא־ אָפִ֣יר מֵֽעִמּ֑וֹ וְלֹֽא־ אֲ֝שַׁקֵּ֗ר בֶּאֱמוּנָתִֽי׃
34אֲחַלֵּ֥ל בְּרִיתִ֑י וּמוֹצָ֥א שְׂ֝פָתַ֗י לֹ֣א אֲשַׁנֶּֽה׃
35נִשְׁבַּ֣עְתִּי בְקָדְשִׁ֑י אִֽם־ לְדָוִ֥ד אֲכַזֵּֽב׃
36לְעוֹלָ֣ם יִהְיֶ֑ה וְכִסְא֖וֹ כַשֶּׁ֣מֶשׁ נֶגְדִּֽי׃
37יִכּ֣וֹן עוֹלָ֑ם וְעֵ֥ד בַּ֝שַּׁ֗חַק נֶאֱמָ֥ן סֶֽלָה׃
38זָ֭נַחְתָּ וַתִּמְאָ֑ס הִ֝תְעַבַּ֗רְתָּ עִם־ מְשִׁיחֶֽךָ׃
39בְּרִ֣ית עַבְדֶּ֑ךָ חִלַּ֖לְתָּ לָאָ֣רֶץ נִזְרֽוֹ׃
40כָל־ גְּדֵרֹתָ֑יו שַׂ֖מְתָּ מִבְצָרָ֣יו מְחִתָּה׃
41כָּל־ עֹ֣בְרֵי דָ֑רֶךְ הָיָ֥ה חֶ֝רְפָּ֗ה לִשְׁכֵנָֽיו׃
42יְמִ֣ין צָרָ֑יו הִ֝שְׂמַ֗חְתָּ כָּל־ אוֹיְבָֽיו׃
43תָּ֭שִׁיב צ֣וּר חַרְבּ֑וֹ וְלֹ֥א הֲ֝קֵימֹת֗וֹ בַּמִּלְחָמָֽה׃
44מִטְּהָר֑וֹ וְ֝כִסְא֗וֹ לָאָ֥רֶץ מִגַּֽרְתָּה׃
45יְמֵ֣י עֲלוּמָ֑יו הֶֽעֱטִ֨יתָ עָלָ֖יו בּוּשָׁ֣ה סֶֽלָה׃
46מָ֣ה יְ֭הוָה תִּסָּתֵ֣ר לָנֶ֑צַח תִּבְעַ֖ר כְּמוֹ־ אֵ֣שׁ חֲמָתֶֽךָ׃
47אֲנִ֥י מֶה־ חָ֑לֶד עַל־ מַה־ שָּׁ֝֗וְא בָּרָ֥אתָ כָל־ בְּנֵי־ אָדָֽם׃
48גֶ֣בֶר יִֽ֭חְיֶה וְלֹ֣א יִרְאֶה־ מָּ֑וֶת יְמַלֵּ֨ט נַפְשׁ֖וֹ מִיַּד־ שְׁא֣וֹל סֶֽלָה׃
49חֲסָדֶ֖יךָ הָרִאשֹׁנִ֥ים ׀ אֲדֹנָ֑י נִשְׁבַּ֥עְתָּ לְ֝דָוִ֗ד בֶּאֱמוּנָתֶֽךָ׃
50אֲ֭דֹנָי חֶרְפַּ֣ת עֲבָדֶ֑יךָ שְׂאֵתִ֥י בְ֝חֵיקִ֗י כָּל־ רַבִּ֥ים עַמִּֽים׃
51חֵרְפ֖וּ אוֹיְבֶ֥יךָ ׀ יְהוָ֑ה אֲשֶׁ֥ר חֵ֝רְפ֗וּ עִקְּב֥וֹת מְשִׁיחֶֽךָ׃
