1מִזְמ֗וֹר לְאָ֫סָ֥ף אֱ‍ֽלֹהִ֡ים בָּ֤אוּ גוֹיִ֨ם ׀ בְּֽנַחֲלָתֶ֗ךָ טִ֭מְּאוּ אֶת־ הֵיכַ֣ל קָדְשֶׁ֑ךָ שָׂ֖מוּ אֶת־ יְרוּשָׁלִַ֣ם לְעִיִּֽים׃
2אֶת־ נִבְלַ֬ת עֲבָדֶ֗יךָ מַ֭אֲכָל לְע֣וֹף הַשָּׁמָ֑יִם בְּשַׂ֥ר חֲ֝סִידֶ֗יךָ לְחַיְתוֹ־ אָֽרֶץ׃
3דָמָ֨ם ׀ כַּמַּ֗יִם סְֽבִ֘יב֤וֹת יְֽרוּשָׁלִָ֗ם וְאֵ֣ין קוֹבֵֽר׃
4חֶ֭רְפָּה לִשְׁכֵנֵ֑ינוּ לַ֥עַג וָ֝קֶ֗לֶס לִסְבִיבוֹתֵֽינוּ׃
5מָ֣ה יְ֭הוָה תֶּאֱנַ֣ף לָנֶ֑צַח תִּבְעַ֥ר כְּמוֹ־ אֵ֝֗שׁ קִנְאָתֶֽךָ׃
6חֲמָתְךָ֨ אֶֽל־ הַגּוֹיִם֮ אֲשֶׁ֪ר לֹא־ יְדָ֫ע֥וּךָ וְעַ֥ל מַמְלָכ֑וֹת אֲשֶׁ֥ר בְּ֝שִׁמְךָ֗ לֹ֣א קָרָֽאוּ׃
7אָכַ֣ל אֶֽת־ יַעֲקֹ֑ב וְֽאֶת־ נָוֵ֥הוּ הֵשַֽׁמּוּ׃ אַֽל־ תִּזְכָּר־
8עֲוֹנֹ֪ת רִאשֹׁ֫נִ֥ים מַ֭הֵר יְקַדְּמ֣וּנוּ רַחֲמֶ֑יךָ כִּ֖י דַלּ֣וֹנוּ מְאֹֽד׃
9אֱלֹ֘הֵ֤י יִשְׁעֵ֗נוּ עַל־ דְּבַ֥ר כְּבֽוֹד־ שְׁמֶ֑ךָ וְהַצִּילֵ֥נוּ וְכַפֵּ֥ר עַל־ חַ֝טֹּאתֵ֗ינוּ לְמַ֣עַן שְׁמֶֽךָ׃
10יֹאמְר֣וּ הַגּוֹיִם֮ אַיֵּ֪ה אֱ‍ֽלֹהֵ֫יהֶ֥ם יִוָּדַ֣ע [בַּגִּיִּים כ] (בַּגֹּויִ֣ם ק) לְעֵינֵ֑ינוּ נִ֝קְמַ֗ת דַּֽם־ עֲבָדֶ֥יךָ הַשָּׁפֽוּךְ׃
11לְפָנֶיךָ֮ אֶנְקַ֪ת אָ֫סִ֥יר כְּגֹ֥דֶל זְרוֹעֲךָ֑ ה֝וֹתֵ֗ר בְּנֵ֣י תְמוּתָֽה׃
12לִשְׁכֵנֵ֣ינוּ שִׁ֭בְעָתַיִם אֶל־ חֵיקָ֑ם חֶרְפָּ֘תָ֤ם אֲשֶׁ֖ר חֵרְפ֣וּךָ אֲדֹנָֽי׃
