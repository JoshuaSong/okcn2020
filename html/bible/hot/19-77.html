1לַמְנַצֵּ֥חַ עַֽל־ [יְדִיתוּן כ] (יְדוּת֗וּן ק) לְאָסָ֥ף מִזְמֽוֹר׃ קוֹלִ֣י אֶל־ אֱלֹהִ֣ים וְאֶצְעָ֑קָה קוֹלִ֥י אֶל־ אֱ֝לֹהִ֗ים וְהַאֲזִ֥ין אֵלָֽי׃
2צָרָתִי֮ אֲדֹנָ֪י דָּ֫רָ֥שְׁתִּי יָדִ֤י ׀ לַ֣יְלָה נִ֭גְּרָה וְלֹ֣א תָפ֑וּג מֵאֲנָ֖ה הִנָּחֵ֣ם נַפְשִֽׁי׃
3אֱלֹהִ֣ים וְאֶֽהֱמָיָ֑ה אָשִׂ֓יחָה ׀ וְתִתְעַטֵּ֖ף רוּחִ֣י סֶֽלָה׃
4שְׁמֻר֣וֹת עֵינָ֑י נִ֝פְעַ֗מְתִּי וְלֹ֣א אֲדַבֵּֽר׃
5יָמִ֣ים מִקֶּ֑דֶם שְׁ֝נ֗וֹת עוֹלָמִֽים׃
6נְגִינָתִ֗י בַּ֫לָּ֥יְלָה עִם־ לְבָבִ֥י אָשִׂ֑יחָה וַיְחַפֵּ֥שׂ רוּחִֽי׃
7יִזְנַ֥ח ׀ אֲדֹנָ֑י וְלֹֽא־ יֹסִ֖יף לִרְצ֣וֹת עֽוֹד׃
8לָנֶ֣צַח חַסְדּ֑וֹ גָּ֥מַר אֹ֝֗מֶר לְדֹ֣ר וָדֹֽר׃
9חַנּ֣וֹת אֵ֑ל אִם־ קָפַ֥ץ בְּ֝אַ֗ף רַחֲמָ֥יו סֶֽלָה׃
10חַלּ֣וֹתִי הִ֑יא שְׁ֝נ֗וֹת יְמִ֣ין עֶלְיֽוֹן׃
11כ] (אֶזְכֹּ֥ור ק) מַֽעַלְלֵי־ יָ֑הּ כִּֽי־ אֶזְכְּרָ֖ה מִקֶּ֣דֶם פִּלְאֶֽךָ׃
12בְכָל־ פָּעֳלֶ֑ךָ וּֽבַעֲלִ֖ילוֹתֶ֣יךָ אָשִֽׂיחָה׃
13בַּקֹּ֣דֶשׁ דַּרְכֶּ֑ךָ מִי־ אֵ֥ל גָּ֝ד֗וֹל כֵּֽאלֹהִֽים׃
14הָ֭אֵל עֹ֣שֵׂה פֶ֑לֶא הוֹדַ֖עְתָּ בָעַמִּ֣ים עֻזֶּֽךָ׃
15בִּזְר֣וֹעַ עַמֶּ֑ךָ בְּנֵי־ יַעֲקֹ֖ב וְיוֹסֵ֣ף סֶֽלָה׃
16מַּ֨יִם ׀ אֱ‍ֽלֹהִ֗ים רָא֣וּךָ מַּ֣יִם יָחִ֑ילוּ אַ֝֗ף יִרְגְּז֥וּ תְהֹמֽוֹת׃
17מַ֨יִם ׀ עָב֗וֹת ק֭וֹל נָתְנ֣וּ שְׁחָקִ֑ים אַף־ חֲ֝צָצֶ֗יךָ יִתְהַלָּֽכוּ׃
18רַעַמְךָ֨ ׀ בַּגַּלְגַּ֗ל הֵאִ֣ירוּ בְרָקִ֣ים תֵּבֵ֑ל רָגְזָ֖ה וַתִּרְעַ֣שׁ הָאָֽרֶץ׃
19דַּרְכֶּ֗ךָ [וּשְׁבִילֶיךָ כ] (וּֽ֭שְׁבִֽילְךָ ק) בְּמַ֣יִם רַבִּ֑ים וְ֝עִקְּבוֹתֶ֗יךָ לֹ֣א נֹדָֽעוּ׃
