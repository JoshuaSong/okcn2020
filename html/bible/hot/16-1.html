1דִּבְרֵ֥י נְחֶמְיָ֖ה בֶּן־ חֲכַלְיָ֑ה וַיְהִ֤י בְחֹֽדֶשׁ־ [כִּסְלֵו כ] (כִּסְלֵיו֙ ק) שְׁנַ֣ת עֶשְׂרִ֔ים וַאֲנִ֥י הָיִ֖יתִי בְּשׁוּשַׁ֥ן הַבִּירָֽה׃
2חֲנָ֜נִי אֶחָ֧ד מֵאַחַ֛י ה֥וּא וַאֲנָשִׁ֖ים מִֽיהוּדָ֑ה וָאֶשְׁאָלֵ֞ם עַל־ הַיְּהוּדִ֧ים הַפְּלֵיטָ֛ה אֲשֶֽׁר־ נִשְׁאֲר֥וּ מִן־ הַשֶּׁ֖בִי וְעַל־ יְרוּשָׁלִָֽם׃
3לִי֒ הַֽנִּשְׁאָרִ֞ים אֲשֶֽׁר־ נִשְׁאֲר֤וּ מִן־ הַשְּׁבִי֙ שָׁ֣ם בַּמְּדִינָ֔ה בְּרָעָ֥ה גְדֹלָ֖ה וּבְחֶרְפָּ֑ה וְחוֹמַ֤ת יְרוּשָׁלִַ֙ם֙ מְפֹרָ֔צֶת וּשְׁעָרֶ֖יהָ נִצְּת֥וּ בָאֵֽשׁ׃
4כְּשָׁמְעִ֣י ׀ אֶת־ הַדְּבָרִ֣ים הָאֵ֗לֶּה יָשַׁ֙בְתִּי֙ וָֽאֶבְכֶּ֔ה וָאֶתְאַבְּלָ֖ה יָמִ֑ים וָֽאֱהִ֥י צָם֙ וּמִתְפַּלֵּ֔ל לִפְנֵ֖י אֱלֹהֵ֥י הַשָּׁמָֽיִם׃
5אָֽנָּ֤א יְהוָה֙ אֱלֹהֵ֣י הַשָּׁמַ֔יִם הָאֵ֥ל הַגָּד֖וֹל וְהַנּוֹרָ֑א שֹׁמֵ֤ר הַבְּרִית֙ וָחֶ֔סֶד לְאֹהֲבָ֖יו וּלְשֹׁמְרֵ֥י מִצְוֹתָֽיו׃
6נָ֣א אָזְנְךָֽ־ קַשֶּׁ֣בֶת וְֽעֵינֶ֪יךָ פְתֻוּח֟וֹת לִשְׁמֹ֣עַ אֶל־ תְּפִלַּ֣ת עַבְדְּךָ֡ אֲשֶׁ֣ר אָנֹכִי֩ מִתְפַּלֵּ֨ל לְפָנֶ֤יךָ הַיּוֹם֙ יוֹמָ֣ם וָלַ֔יְלָה עַל־ בְּנֵ֥י יִשְׂרָאֵ֖ל עֲבָדֶ֑יךָ וּמִתְוַדֶּ֗ה עַל־ חַטֹּ֤אות בְּנֵֽי־ יִשְׂרָאֵל֙ אֲשֶׁ֣ר חָטָ֣אנוּ לָ֔ךְ וַאֲנִ֥י וּבֵית־ אָבִ֖י חָטָֽאנוּ׃
7חָבַ֣לְנוּ לָ֑ךְ וְלֹא־ שָׁמַ֣רְנוּ אֶת־ הַמִּצְוֹ֗ת וְאֶת־ הַֽחֻקִּים֙ וְאֶת־ הַמִּשְׁפָּטִ֔ים אֲשֶׁ֥ר צִוִּ֖יתָ אֶת־ מֹשֶׁ֥ה עַבְדֶּֽךָ׃
8נָא֙ אֶת־ הַדָּבָ֔ר אֲשֶׁ֥ר צִוִּ֛יתָ אֶת־ מֹשֶׁ֥ה עַבְדְּךָ֖ לֵאמֹ֑ר אַתֶּ֣ם תִּמְעָ֔לוּ אֲנִ֕י אָפִ֥יץ אֶתְכֶ֖ם בָּעַמִּֽים׃
9אֵלַ֔י וּשְׁמַרְתֶּם֙ מִצְוֹתַ֔י וַעֲשִׂיתֶ֖ם אֹתָ֑ם אִם־ יִהְיֶ֨ה נִֽדַּחֲכֶ֜ם בִּקְצֵ֤ה הַשָּׁמַ֙יִם֙ מִשָּׁ֣ם אֲקַבְּצֵ֔ם [וַהֲבֹואֹתִים כ] (וַהֲבִֽיאֹותִים֙ ק) אֶל־ הַמָּק֔וֹם אֲשֶׁ֣ר בָּחַ֔רְתִּי לְשַׁכֵּ֥ן אֶת־ שְׁמִ֖י שָֽׁם׃
10עֲבָדֶ֖יךָ וְעַמֶּ֑ךָ אֲשֶׁ֤ר פָּדִ֙יתָ֙ בְּכֹחֲךָ֣ הַגָּד֔וֹל וּבְיָדְךָ֖ הַחֲזָקָֽה׃
