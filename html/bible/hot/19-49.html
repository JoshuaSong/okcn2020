1לַמְנַצֵּ֬חַ ׀ לִבְנֵי־ קֹ֬רַח מִזְמֽוֹר׃ שִׁמְעוּ־ זֹ֭את כָּל־ הָֽעַמִּ֑ים הַ֝אֲזִ֗ינוּ כָּל־ יֹ֥שְׁבֵי חָֽלֶד׃
2בְּנֵ֣י אָ֭דָם גַּם־ בְּנֵי־ אִ֑ישׁ יַ֝֗חַד עָשִׁ֥יר וְאֶבְיֽוֹן׃
3יְדַבֵּ֣ר חָכְמ֑וֹת וְהָג֖וּת לִבִּ֣י תְבוּנֽוֹת׃
4לְמָשָׁ֣ל אָזְנִ֑י אֶפְתַּ֥ח בְּ֝כִנּ֗וֹר חִידָתִֽי׃
5אִ֭ירָא בִּ֣ימֵי רָ֑ע עֲוֹ֖ן עֲקֵבַ֣י יְסוּבֵּֽנִי׃
6עַל־ חֵילָ֑ם וּבְרֹ֥ב עָ֝שְׁרָ֗ם יִתְהַלָּֽלוּ׃
7לֹא־ פָדֹ֣ה יִפְדֶּ֣ה אִ֑ישׁ לֹא־ יִתֵּ֖ן לֵאלֹהִ֣ים כָּפְרֽוֹ׃
8פִּדְי֥וֹן נַפְשָׁ֗ם וְחָדַ֥ל לְעוֹלָֽם׃
9ע֥וֹד לָנֶ֑צַח לֹ֖א יִרְאֶ֣ה הַשָּֽׁחַת׃
10יִרְאֶ֨ה ׀ חֲכָ֘מִ֤ים יָמ֗וּתוּ יַ֤חַד כְּסִ֣יל וָבַ֣עַר יֹאבֵ֑דוּ וְעָזְב֖וּ לַאֲחֵרִ֣ים חֵילָֽם׃
11בָּתֵּ֨ימוֹ ׀ לְֽעוֹלָ֗ם מִ֭שְׁכְּנֹתָם לְדֹ֣ר וָדֹ֑ר קָֽרְא֥וּ בִ֝שְׁמוֹתָ֗ם עֲלֵ֣י אֲדָמֽוֹת׃
12בִּ֭יקָר בַּל־ יָלִ֑ין נִמְשַׁ֖ל כַּבְּהֵמ֣וֹת נִדְמֽוּ׃
13דַ֭רְכָּם כֵּ֣סֶל לָ֑מוֹ וְאַחֲרֵיהֶ֓ם ׀ בְּפִיהֶ֖ם יִרְצ֣וּ סֶֽלָה׃
14לִֽשְׁא֣וֹל שַׁתּוּ֮ מָ֤וֶת יִ֫רְעֵ֥ם וַיִּרְדּ֘וּ בָ֤ם יְשָׁרִ֨ים ׀ לַבֹּ֗קֶר [וְצִירָם כ] (וְ֭צוּרָם ק) לְבַלּ֥וֹת שְׁא֗וֹל מִזְּבֻ֥ל לֽוֹ׃
15אֱלֹהִ֗ים יִפְדֶּ֣ה נַ֭פְשִׁי מִֽיַּד־ שְׁא֑וֹל כִּ֖י יִקָּחֵ֣נִי סֶֽלָה׃
16תִּ֭ירָא כִּֽי־ יַעֲשִׁ֣ר אִ֑ישׁ כִּֽי־ יִ֝רְבֶּה כְּב֣וֹד בֵּיתֽוֹ׃
17לֹ֣א בְ֭מוֹתוֹ יִקַּ֣ח הַכֹּ֑ל לֹא־ יֵרֵ֖ד אַחֲרָ֣יו כְּבוֹדֽוֹ׃
18נַ֭פְשׁוֹ בְּחַיָּ֣יו יְבָרֵ֑ךְ וְ֝יוֹדֻ֗ךָ כִּי־ תֵיטִ֥יב לָֽךְ׃
19עַד־ דּ֣וֹר אֲבוֹתָ֑יו עַד־ נֵ֝֗צַח לֹ֣א יִרְאוּ־ אֽוֹר׃
