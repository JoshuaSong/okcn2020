1וַיְהִ֣י אַֽחֲרֵי־ כֵ֗ן וּלְאַבְשָׁל֧וֹם בֶּן־ דָּוִ֛ד אָח֥וֹת יָפָ֖ה וּשְׁמָ֣הּ תָּמָ֑ר וַיֶּאֱהָבֶ֖הָ אַמְנ֥וֹן בֶּן־ דָּוִֽד׃
2לְאַמְנ֜וֹן לְהִתְחַלּ֗וֹת בַּֽעֲבוּר֙ תָּמָ֣ר אֲחֹת֔וֹ כִּ֥י בְתוּלָ֖ה הִ֑יא וַיִּפָּלֵא֙ בְּעֵינֵ֣י אַמְנ֔וֹן לַעֲשׂ֥וֹת לָ֖הּ מְאֽוּמָה׃
3רֵ֗עַ וּשְׁמוֹ֙ יֽוֹנָדָ֔ב בֶּן־ שִׁמְעָ֖ה אֲחִ֣י דָוִ֑ד וְי֣וֹנָדָ֔ב אִ֥ישׁ חָכָ֖ם מְאֹֽד׃
4ל֗וֹ מַדּ֣וּעַ אַ֠תָּה כָּ֣כָה דַּ֤ל בֶּן־ הַמֶּ֙לֶךְ֙ בַּבֹּ֣קֶר בַּבֹּ֔קֶר הֲל֖וֹא תַּגִּ֣יד לִ֑י וַיֹּ֤אמֶר לוֹ֙ אַמְנ֔וֹן אֶת־ תָּמָ֗ר אֲח֛וֹת אַבְשָׁלֹ֥ם אָחִ֖י אֲנִ֥י אֹהֵֽב׃
5לוֹ֙ יְה֣וֹנָדָ֔ב שְׁכַ֥ב עַל־ מִשְׁכָּבְךָ֖ וְהִתְחָ֑ל וּבָ֧א אָבִ֣יךָ לִרְאוֹתֶ֗ךָ וְאָמַרְתָּ֣ אֵלָ֡יו תָּ֣בֹא נָא֩ תָמָ֨ר אֲחוֹתִ֜י וְתַבְרֵ֣נִי לֶ֗חֶם וְעָשְׂתָ֤ה לְעֵינַי֙ אֶת־ הַבִּרְיָ֔ה לְמַ֙עַן֙ אֲשֶׁ֣ר אֶרְאֶ֔ה וְאָכַלְתִּ֖י מִיָּדָֽהּ׃
6אַמְנ֖וֹן וַיִּתְחָ֑ל וַיָּבֹ֨א הַמֶּ֜לֶךְ לִרְאֹת֗וֹ וַיֹּ֨אמֶר אַמְנ֤וֹן אֶל־ הַמֶּ֙לֶךְ֙ תָּֽבוֹא־ נָ֞א תָּמָ֣ר אֲחֹתִ֗י וּתְלַבֵּ֤ב לְעֵינַי֙ שְׁתֵּ֣י לְבִב֔וֹת וְאֶבְרֶ֖ה מִיָּדָֽהּ׃
7דָּוִ֛ד אֶל־ תָּמָ֖ר הַבַּ֣יְתָה לֵאמֹ֑ר לְכִ֣י נָ֗א בֵּ֚ית אַמְנ֣וֹן אָחִ֔יךְ וַעֲשִׂי־ ל֖וֹ הַבִּרְיָֽה׃
8תָּמָ֗ר בֵּ֛ית אַמְנ֥וֹן אָחִ֖יהָ וְה֣וּא שֹׁכֵ֑ב וַתִּקַּ֨ח אֶת־ הַבָּצֵ֤ק [וַתָּלֹושׁ כ] (וַתָּ֙לָשׁ֙ ק) וַתְּלַבֵּ֣ב לְעֵינָ֔יו וַתְּבַשֵּׁ֖ל אֶת־ הַלְּבִבֽוֹת׃
9אֶת־ הַמַּשְׂרֵת֙ וַתִּצֹ֣ק לְפָנָ֔יו וַיְמָאֵ֖ן לֶאֱכ֑וֹל וַיֹּ֣אמֶר אַמְנ֗וֹן הוֹצִ֤יאוּ כָל־ אִישׁ֙ מֵֽעָלַ֔י וַיֵּצְא֥וּ כָל־ אִ֖ישׁ מֵעָלָֽיו׃
10אַמְנ֜וֹן אֶל־ תָּמָ֗ר הָבִ֤יאִי הַבִּרְיָה֙ הַחֶ֔דֶר וְאֶבְרֶ֖ה מִיָּדֵ֑ךְ וַתִּקַּ֣ח תָּמָ֗ר אֶת־ הַלְּבִבוֹת֙ אֲשֶׁ֣ר עָשָׂ֔תָה וַתָּבֵ֛א לְאַמְנ֥וֹן אָחִ֖יהָ הֶחָֽדְרָה׃
11אֵלָ֖יו לֶֽאֱכֹ֑ל וַיַּֽחֲזֶק־ בָּהּ֙ וַיֹּ֣אמֶר לָ֔הּ בּ֛וֹאִי שִׁכְבִ֥י עִמִּ֖י אֲחוֹתִֽי׃
12ל֗וֹ אַל־ אָחִי֙ אַל־ תְּעַנֵּ֔נִי כִּ֛י לֹא־ יֵֽעָשֶׂ֥ה כֵ֖ן בְּיִשְׂרָאֵ֑ל אַֽל־ תַּעֲשֵׂ֖ה אֶת־ הַנְּבָלָ֥ה הַזֹּֽאת׃
13אָ֤נָה אוֹלִיךְ֙ אֶת־ חֶרְפָּתִ֔י וְאַתָּ֗ה תִּהְיֶ֛ה כְּאַחַ֥ד הַנְּבָלִ֖ים בְּיִשְׂרָאֵ֑ל וְעַתָּה֙ דַּבֶּר־ נָ֣א אֶל־ הַמֶּ֔לֶךְ כִּ֛י לֹ֥א יִמְנָעֵ֖נִי מִמֶּֽךָּ׃
14אָבָ֖ה לִשְׁמֹ֣עַ בְּקוֹלָ֑הּ וַיֶּחֱזַ֤ק מִמֶּ֙נָּה֙ וַיְעַנֶּ֔הָ וַיִּשְׁכַּ֖ב אֹתָֽהּ׃
15אַמְנ֗וֹן שִׂנְאָה֙ גְּדוֹלָ֣ה מְאֹ֔ד כִּ֣י גְדוֹלָ֗ה הַשִּׂנְאָה֙ אֲשֶׁ֣ר שְׂנֵאָ֔הּ מֵאַהֲבָ֖ה אֲשֶׁ֣ר אֲהֵבָ֑הּ וַֽיֹּאמֶר־ לָ֥הּ אַמְנ֖וֹן ק֥וּמִי לֵֽכִי׃
16ל֗וֹ אַל־ אוֹדֹ֞ת הָרָעָ֤ה הַגְּדוֹלָה֙ הַזֹּ֔את מֵאַחֶ֛רֶת אֲשֶׁר־ עָשִׂ֥יתָ עִמִּ֖י לְשַׁלְּחֵ֑נִי וְלֹ֥א אָבָ֖ה לִשְׁמֹ֥עַֽ לָֽהּ׃
17אֶֽת־ נַעֲרוֹ֙ מְשָׁ֣רְת֔וֹ וַיֹּ֕אמֶר שִׁלְחוּ־ נָ֥א אֶת־ זֹ֛את מֵעָלַ֖י הַח֑וּצָה וּנְעֹ֥ל הַדֶּ֖לֶת אַחֲרֶֽיהָ׃
18כְּתֹ֣נֶת פַּסִּ֔ים כִּי֩ כֵ֨ן תִּלְבַּ֧שְׁןָ בְנוֹת־ הַמֶּ֛לֶךְ הַבְּתוּלֹ֖ת מְעִילִ֑ים וַיֹּצֵ֨א אוֹתָ֤הּ מְשָֽׁרְתוֹ֙ הַח֔וּץ וְנָעַ֥ל הַדֶּ֖לֶת אַחֲרֶֽיהָ׃
19תָּמָ֥ר אֵ֙פֶר֙ עַל־ רֹאשָׁ֔הּ וּכְתֹ֧נֶת הַפַּסִּ֛ים אֲשֶׁ֥ר עָלֶ֖יהָ קָרָ֑עָה וַתָּ֤שֶׂם יָדָהּ֙ עַל־ רֹאשָׁ֔הּ וַתֵּ֥לֶךְ הָל֖וֹךְ וְזָעָֽקָה׃
20אֵלֶ֜יהָ אַבְשָׁל֣וֹם אָחִ֗יהָ הַאֲמִינ֣וֹן אָחִיךְ֮ הָיָ֣ה עִמָּךְ֒ וְעַתָּ֞ה אֲחוֹתִ֤י הַחֲרִ֙ישִׁי֙ אָחִ֣יךְ ה֔וּא אַל־ תָּשִׁ֥יתִי אֶת־ לִבֵּ֖ךְ לַדָּבָ֣ר הַזֶּ֑ה וַתֵּ֤שֶׁב תָּמָר֙ וְשֹׁ֣מֵמָ֔ה בֵּ֖ית אַבְשָׁל֥וֹם אָחִֽיהָ׃
21דָּוִ֔ד שָׁמַ֕ע אֵ֥ת כָּל־ הַדְּבָרִ֖ים הָאֵ֑לֶּה וַיִּ֥חַר ל֖וֹ מְאֹֽד׃
22דִבֶּ֧ר אַבְשָׁל֛וֹם עִם־ אַמְנ֖וֹן לְמֵרָ֣ע וְעַד־ ט֑וֹב כִּֽי־ שָׂנֵ֤א אַבְשָׁלוֹם֙ אֶת־ אַמְנ֔וֹן עַל־ דְּבַר֙ אֲשֶׁ֣ר עִנָּ֔ה אֵ֖ת תָּמָ֥ר אֲחֹתֽוֹ׃ פ
23לִשְׁנָתַ֣יִם יָמִ֔ים וַיִּהְי֤וּ גֹֽזְזִים֙ לְאַבְשָׁל֔וֹם בְּבַ֥עַל חָצ֖וֹר אֲשֶׁ֣ר עִם־ אֶפְרָ֑יִם וַיִּקְרָ֥א אַבְשָׁל֖וֹם לְכָל־ בְּנֵ֥י הַמֶּֽלֶךְ׃
24אַבְשָׁלוֹם֙ אֶל־ הַמֶּ֔לֶךְ וַיֹּ֕אמֶר הִנֵּה־ נָ֥א גֹזְזִ֖ים לְעַבְדֶּ֑ךָ יֵֽלֶךְ־ נָ֥א הַמֶּ֛לֶךְ וַעֲבָדָ֖יו עִם־ עַבְדֶּֽךָ׃
25הַמֶּ֜לֶךְ אֶל־ אַבְשָׁל֗וֹם אַל־ בְּנִי֙ אַל־ נָ֤א נֵלֵךְ֙ כֻּלָּ֔נוּ וְלֹ֥א נִכְבַּ֖ד עָלֶ֑יךָ וַיִּפְרָץ־ בּ֛וֹ וְלֹֽא־ אָבָ֥ה לָלֶ֖כֶת וַֽיְבָרֲכֵֽהוּ׃
26אַבְשָׁל֔וֹם וָלֹ֕א יֵֽלֶךְ־ נָ֥א אִתָּ֖נוּ אַמְנ֣וֹן אָחִ֑י וַיֹּ֤אמֶר לוֹ֙ הַמֶּ֔לֶךְ לָ֥מָּה יֵלֵ֖ךְ עִמָּֽךְ׃
27בּ֖וֹ אַבְשָׁל֑וֹם וַיִּשְׁלַ֤ח אִתּוֹ֙ אֶת־ אַמְנ֔וֹן וְאֵ֖ת כָּל־ בְּנֵ֥י הַמֶּֽלֶךְ׃ ס
28אַבְשָׁל֨וֹם אֶת־ נְעָרָ֜יו לֵאמֹ֗ר רְא֣וּ נָ֠א כְּט֨וֹב לֵב־ אַמְנ֤וֹן בַּיַּ֙יִן֙ וְאָמַרְתִּ֣י אֲלֵיכֶ֔ם הַכּ֧וּ אֶת־ אַמְנ֛וֹן וַהֲמִתֶּ֥ם אֹת֖וֹ אַל־ תִּירָ֑אוּ הֲל֗וֹא כִּ֤י אָֽנֹכִי֙ צִוִּ֣יתִי אֶתְכֶ֔ם חִזְק֖וּ וִהְי֥וּ לִבְנֵי־ חָֽיִל׃
29נַעֲרֵ֤י אַבְשָׁלוֹם֙ לְאַמְנ֔וֹן כַּאֲשֶׁ֥ר צִוָּ֖ה אַבְשָׁל֑וֹם וַיָּקֻ֣מוּ ׀ כָּל־ בְּנֵ֣י הַמֶּ֗לֶךְ וַֽיִּרְכְּב֛וּ אִ֥ישׁ עַל־ פִּרְדּ֖וֹ וַיָּנֻֽסוּ׃
30הֵ֣מָּה בַדֶּ֔רֶךְ וְהַשְּׁמֻעָ֣ה בָ֔אָה אֶל־ דָּוִ֖ד לֵאמֹ֑ר הִכָּ֤ה אַבְשָׁלוֹם֙ אֶת־ כָּל־ בְּנֵ֣י הַמֶּ֔לֶךְ וְלֹֽא־ נוֹתַ֥ר מֵהֶ֖ם אֶחָֽד׃ ס
31הַמֶּ֛לֶךְ וַיִּקְרַ֥ע אֶת־ בְּגָדָ֖יו וַיִּשְׁכַּ֣ב אָ֑רְצָה וְכָל־ עֲבָדָ֥יו נִצָּבִ֖ים קְרֻעֵ֥י בְגָדִֽים׃ ס
32יוֹנָדָ֣ב ׀ בֶּן־ שִׁמְעָ֨ה אֲחִֽי־ דָוִ֜ד וַיֹּ֗אמֶר אַל־ יֹאמַ֤ר אֲדֹנִי֙ אֵ֣ת כָּל־ הַנְּעָרִ֤ים בְּנֵֽי־ הַמֶּ֙לֶךְ֙ הֵמִ֔יתוּ כִּֽי־ אַמְנ֥וֹן לְבַדּ֖וֹ מֵ֑ת כִּֽי־ עַל־ פִּ֤י אַבְשָׁלוֹם֙ הָיְתָ֣ה שׂוּמָ֔ה מִיּוֹם֙ עַנֹּת֔וֹ אֵ֖ת תָּמָ֥ר אֲחֹתֽוֹ׃
33אַל־ יָשֵׂם֩ אֲדֹנִ֨י הַמֶּ֤לֶךְ אֶל־ לִבּוֹ֙ דָּבָ֣ר לֵאמֹ֔ר כָּל־ בְּנֵ֥י הַמֶּ֖לֶךְ מֵ֑תוּ כִּֽי־ אִם־ אַמְנ֥וֹן לְבַדּ֖וֹ מֵֽת׃ פ
34אַבְשָׁל֑וֹם וַיִּשָּׂ֞א הַנַּ֤עַר הַצֹּפֶה֙ אֶת־ [עֵינֹו כ] (עֵינָ֔יו ק) וַיַּ֗רְא וְהִנֵּ֨ה עַם־ רַ֜ב הֹלְכִ֥ים מִדֶּ֛רֶךְ אַחֲרָ֖יו מִצַּ֥ד הָהָֽר׃
35יֽוֹנָדָב֙ אֶל־ הַמֶּ֔לֶךְ הִנֵּ֥ה בְנֵֽי־ הַמֶּ֖לֶךְ בָּ֑אוּ כִּדְבַ֥ר עַבְדְּךָ֖ כֵּ֥ן הָיָֽה׃
36כְּכַלֹּת֣וֹ לְדַבֵּ֗ר וְהִנֵּ֤ה בְנֵֽי־ הַמֶּ֙לֶךְ֙ בָּ֔אוּ וַיִּשְׂא֥וּ קוֹלָ֖ם וַיִּבְכּ֑וּ וְגַם־ הַמֶּ֙לֶךְ֙ וְכָל־ עֲבָדָ֔יו בָּכ֕וּ בְּכִ֖י גָּד֥וֹל מְאֹֽד׃
37בָּרַ֔ח וַיֵּ֛לֶךְ אֶל־ תַּלְמַ֥י בֶּן־ [עַמִּיחוּר כ] (עַמִּיה֖וּד ק) מֶ֣לֶךְ גְּשׁ֑וּר וַיִּתְאַבֵּ֥ל עַל־ בְּנ֖וֹ כָּל־ הַיָּמִֽים׃
38בָּרַ֖ח וַיֵּ֣לֶךְ גְּשׁ֑וּר וַיְהִי־ שָׁ֖ם שָׁלֹ֥שׁ שָׁנִֽים׃
