1מַשְׂכִּ֗יל לְאָ֫סָ֥ף הַאֲזִ֣ינָה עַ֭מִּי תּוֹרָתִ֑י הַטּ֥וּ אָ֝זְנְכֶ֗ם לְאִמְרֵי־ פִֽי׃
2בְמָשָׁ֣ל פִּ֑י אַבִּ֥יעָה חִ֝יד֗וֹת מִנִּי־ קֶֽדֶם׃
3שָׁ֭מַעְנוּ וַנֵּדָעֵ֑ם וַ֝אֲבוֹתֵ֗ינוּ סִפְּרוּ־ לָֽנוּ׃
4נְכַחֵ֨ד ׀ מִבְּנֵיהֶ֗ם לְד֥וֹר אַחֲר֗וֹן מְֽ֭סַפְּרִים תְּהִלּ֣וֹת יְהוָ֑ה וֶעֱזוּז֥וֹ וְ֝נִפְלְאוֹתָ֗יו אֲשֶׁ֣ר עָשָֽׂה׃
5עֵד֨וּת ׀ בְּֽיַעֲקֹ֗ב וְתוֹרָה֮ שָׂ֤ם בְּיִשְׂרָ֫אֵ֥ל אֲשֶׁ֣ר צִ֭וָּה אֶת־ אֲבוֹתֵ֑ינוּ לְ֝הוֹדִיעָ֗ם לִבְנֵיהֶֽם׃
6יֵדְע֨וּ ׀ דּ֣וֹר אַ֭חֲרוֹן בָּנִ֣ים יִוָּלֵ֑דוּ יָ֝קֻ֗מוּ וִֽיסַפְּר֥וּ לִבְנֵיהֶֽם׃
7בֵֽאלֹהִ֗ים כִּ֫סְלָ֥ם וְלֹ֣א יִ֭שְׁכְּחוּ מַֽעַלְלֵי־ אֵ֑ל וּמִצְוֹתָ֥יו יִנְצֹֽרוּ׃
8יִהְי֨וּ ׀ כַּאֲבוֹתָ֗ם דּוֹר֮ סוֹרֵ֪ר וּמֹ֫רֶ֥ה דּ֭וֹר לֹא־ הֵכִ֣ין לִבּ֑וֹ וְלֹא־ נֶאֶמְנָ֖ה אֶת־ אֵ֣ל רוּחֽוֹ׃
9אֶפְרַ֗יִם נוֹשְׁקֵ֥י רוֹמֵי־ קָ֑שֶׁת הָ֝פְכ֗וּ בְּי֣וֹם קְרָֽב׃
10שָׁ֭מְרוּ בְּרִ֣ית אֱלֹהִ֑ים וּ֝בְתוֹרָת֗וֹ מֵאֲנ֥וּ לָלֶֽכֶת׃
11עֲלִילוֹתָ֑יו וְ֝נִפְלְאוֹתָ֗יו אֲשֶׁ֣ר הֶרְאָֽם׃
12אֲ֭בוֹתָם עָ֣שָׂה פֶ֑לֶא בְּאֶ֖רֶץ מִצְרַ֣יִם שְׂדֵה־ צֹֽעַן׃
13יָ֭ם וַיַּֽעֲבִירֵ֑ם וַֽיַּצֶּב־ מַ֥יִם כְּמוֹ־ נֵֽד׃
14בֶּעָנָ֣ן יוֹמָ֑ם וְכָל־ הַ֝לַּ֗יְלָה בְּא֣וֹר אֵֽשׁ׃
15צֻ֭רִים בַּמִּדְבָּ֑ר וַ֝יַּ֗שְׁקְ כִּתְהֹמ֥וֹת רַבָּֽה׃
16נוֹזְלִ֣ים מִסָּ֑לַע וַיּ֖וֹרֶד כַּנְּהָר֣וֹת מָֽיִם׃
17ע֭וֹד לַחֲטֹא־ ל֑וֹ לַֽמְר֥וֹת עֶ֝לְי֗וֹן בַּצִּיָּֽה׃
18אֵ֥ל בִּלְבָבָ֑ם לִֽשְׁאָל־ אֹ֥כֶל לְנַפְשָֽׁם׃
19בֵּֽאלֹ֫הִ֥ים אָ֭מְרוּ הֲי֣וּכַל אֵ֑ל לַעֲרֹ֥ךְ שֻׁ֝לְחָ֗ן בַּמִּדְבָּֽר׃
20הִכָּה־ צ֨וּר ׀ וַיָּז֣וּבוּ מַיִם֮ וּנְחָלִ֪ים יִ֫שְׁטֹ֥פוּ הֲגַם־ לֶ֭חֶם י֣וּכַל תֵּ֑ת אִם־ יָכִ֖ין שְׁאֵ֣ר לְעַמּֽוֹ׃
21שָׁמַ֥ע יְהוָ֗ה וַֽיִּתְעַבָּ֥ר וְ֭אֵשׁ נִשְּׂקָ֣ה בְיַעֲקֹ֑ב וְגַם־ אַ֝֗ף עָלָ֥ה בְיִשְׂרָאֵֽל׃
22לֹ֣א הֶ֭אֱמִינוּ בֵּאלֹהִ֑ים וְלֹ֥א בָ֝טְח֗וּ בִּֽישׁוּעָתֽוֹ׃
23שְׁחָקִ֣ים מִמָּ֑עַל וְדַלְתֵ֖י שָׁמַ֣יִם פָּתָֽח׃
24עֲלֵיהֶ֣ם מָ֣ן לֶאֱכֹ֑ל וּדְגַן־ שָׁ֝מַ֗יִם נָ֣תַן לָֽמוֹ׃
25אַ֭בִּירִים אָ֣כַל אִ֑ישׁ צֵידָ֬ה שָׁלַ֖ח לָהֶ֣ם לָשֹֽׂבַע׃
26קָ֭דִים בַּשָּׁמָ֑יִם וַיְנַהֵ֖ג בְּעֻזּ֣וֹ תֵימָֽן׃
27עֲלֵיהֶ֣ם כֶּעָפָ֣ר שְׁאֵ֑ר וּֽכְח֥וֹל יַ֝מִּ֗ים ע֣וֹף כָּנָֽף׃
28בְּקֶ֣רֶב מַחֲנֵ֑הוּ סָ֝בִ֗יב לְמִשְׁכְּנֹתָֽיו׃
29וַיִּשְׂבְּע֣וּ מְאֹ֑ד וְ֝תַֽאֲוָתָ֗ם יָבִ֥א לָהֶֽם׃
30זָר֥וּ מִתַּאֲוָתָ֑ם ע֝֗וֹד אָכְלָ֥ם בְּפִיהֶֽם׃
31אֱלֹהִ֨ים ׀ עָ֘לָ֤ה בָהֶ֗ם וַֽ֭יַּהֲרֹג בְּמִשְׁמַנֵּיהֶ֑ם וּבַחוּרֵ֖י יִשְׂרָאֵ֣ל הִכְרִֽיעַ׃
32זֹ֭את חָֽטְאוּ־ ע֑וֹד וְלֹֽא־ הֶ֝אֱמִ֗ינוּ בְּנִפְלְאוֹתָֽיו׃
33בַּהֶ֥בֶל יְמֵיהֶ֑ם וּ֝שְׁנוֹתָ֗ם בַּבֶּהָלָֽה׃
34הֲרָגָ֥ם וּדְרָשׁ֑וּהוּ וְ֝שָׁ֗בוּ וְשִֽׁחֲרוּ־ אֵֽל׃
35כִּֽי־ אֱלֹהִ֣ים צוּרָ֑ם וְאֵ֥ל עֶ֝לְיוֹן גֹּאֲלָֽם׃
36בְּפִיהֶ֑ם וּ֝בִלְשׁוֹנָ֗ם יְכַזְּבוּ־ לֽוֹ׃
37לֹא־ נָכ֣וֹן עִמּ֑וֹ וְלֹ֥א נֶ֝אֶמְנ֗וּ בִּבְרִיתֽוֹ׃
38רַח֨וּם ׀ יְכַפֵּ֥ר עָוֹן֮ וְֽלֹא־ יַ֫שְׁחִ֥ית וְ֭הִרְבָּה לְהָשִׁ֣יב אַפּ֑וֹ וְלֹֽא־ יָ֝עִיר כָּל־ חֲמָתֽוֹ׃
39כִּי־ בָשָׂ֣ר הֵ֑מָּה ר֥וּחַ ה֝וֹלֵ֗ךְ וְלֹ֣א יָשֽׁוּב׃
40יַמְר֣וּהוּ בַמִּדְבָּ֑ר יַ֝עֲצִיב֗וּהוּ בִּֽישִׁימֽוֹן׃
41וַיְנַסּ֣וּ אֵ֑ל וּקְד֖וֹשׁ יִשְׂרָאֵ֣ל הִתְווּ׃
42זָכְר֥וּ אֶת־ יָד֑וֹ י֝֗וֹם אֲ‍ֽשֶׁר־ פָּדָ֥ם מִנִּי־ צָֽר׃
43שָׂ֣ם בְּ֭מִצְרַיִם אֹֽתוֹתָ֑יו וּ֝מוֹפְתָ֗יו בִּשְׂדֵה־ צֹֽעַן׃
44לְ֭דָם יְאֹרֵיהֶ֑ם וְ֝נֹזְלֵיהֶ֗ם בַּל־ יִשְׁתָּיֽוּן׃
45בָּהֶ֣ם עָ֭רֹב וַיֹּאכְלֵ֑ם וּ֝צְפַרְדֵּ֗עַ וַתַּשְׁחִיתֵֽם׃
46לֶחָסִ֣יל יְבוּלָ֑ם וִֽ֝יגִיעָ֗ם לָאַרְבֶּֽה׃
47בַּבָּרָ֣ד גַּפְנָ֑ם וְ֝שִׁקְמוֹתָ֗ם בַּֽחֲנָמַֽל׃
48לַבָּרָ֣ד בְּעִירָ֑ם וּ֝מִקְנֵיהֶ֗ם לָרְשָׁפִֽים׃
49בָּ֨ם ׀ חֲר֬וֹן אַפּ֗וֹ עֶבְרָ֣ה וָזַ֣עַם וְצָרָ֑ה מִ֝שְׁלַ֗חַת מַלְאֲכֵ֥י רָעִֽים׃
50נָתִ֗יב לְאַ֫פּ֥וֹ לֹא־ חָשַׂ֣ךְ מִמָּ֣וֶת נַפְשָׁ֑ם וְ֝חַיָּתָ֗ם לַדֶּ֥בֶר הִסְגִּֽיר׃
51כָּל־ בְּכ֣וֹר בְּמִצְרָ֑יִם רֵאשִׁ֥ית א֝וֹנִ֗ים בְּאָהֳלֵי־ חָֽם׃
52כַּצֹּ֣אן עַמּ֑וֹ וַֽיְנַהֲגֵ֥ם כַּ֝עֵ֗דֶר בַּמִּדְבָּֽר׃
53לָ֭בֶטַח וְלֹ֣א פָחָ֑דוּ וְאֶת־ א֝וֹיְבֵיהֶ֗ם כִּסָּ֥ה הַיָּֽם׃
54אֶל־ גְּב֣וּל קָדְשׁ֑וֹ הַר־ זֶ֝֗ה קָנְתָ֥ה יְמִינֽוֹ׃
55מִפְּנֵיהֶ֨ם ׀ גּוֹיִ֗ם וַֽ֭יַּפִּילֵם בְּחֶ֣בֶל נַחֲלָ֑ה וַיַּשְׁכֵּ֥ן בְּ֝אָהֳלֵיהֶ֗ם שִׁבְטֵ֥י יִשְׂרָאֵֽל׃
56וַ֭יַּמְרוּ אֶת־ אֱלֹהִ֣ים עֶלְי֑וֹן וְ֝עֵדוֹתָ֗יו לֹ֣א שָׁמָֽרוּ׃
57וַֽ֭יִּבְגְּדוּ כַּאֲבוֹתָ֑ם נֶ֝הְפְּכ֗וּ כְּקֶ֣שֶׁת רְמִיָּֽה׃
58בְּבָמוֹתָ֑ם וּ֝בִפְסִילֵיהֶ֗ם יַקְנִיאֽוּהוּ׃
59אֱ֭לֹהִים וַֽיִּתְעַבָּ֑ר וַיִּמְאַ֥ס מְ֝אֹ֗ד בְּיִשְׂרָאֵֽל׃
60מִשְׁכַּ֣ן שִׁל֑וֹ אֹ֝֗הֶל שִׁכֵּ֥ן בָּאָדָֽם׃
61לַשְּׁבִ֣י עֻזּ֑וֹ וְֽתִפְאַרְתּ֥וֹ בְיַד־ צָֽר׃
62לַחֶ֣רֶב עַמּ֑וֹ וּ֝בְנַחֲלָת֗וֹ הִתְעַבָּֽר׃
63אָֽכְלָה־ אֵ֑שׁ וּ֝בְתוּלֹתָ֗יו לֹ֣א הוּלָּֽלוּ׃
64בַּחֶ֣רֶב נָפָ֑לוּ וְ֝אַלְמְנֹתָ֗יו לֹ֣א תִבְכֶּֽינָה׃
65כְּיָשֵׁ֥ן ׀ אֲדֹנָ֑י כְּ֝גִבּ֗וֹר מִתְרוֹנֵ֥ן מִיָּֽיִן׃
66צָרָ֥יו אָח֑וֹר חֶרְפַּ֥ת ע֝וֹלָ֗ם נָ֣תַן לָֽמוֹ׃
67בְּאֹ֣הֶל יוֹסֵ֑ף וּֽבְשֵׁ֥בֶט אֶ֝פְרַ֗יִם לֹ֣א בָחָֽר׃
68אֶת־ שֵׁ֣בֶט יְהוּדָ֑ה אֶֽת־ הַ֥ר צִ֝יּ֗וֹן אֲשֶׁ֣ר אָהֵֽב׃
69כְּמוֹ־ רָ֭מִים מִקְדָּשׁ֑וֹ כְּ֝אֶ֗רֶץ יְסָדָ֥הּ לְעוֹלָֽם׃
70בְּדָוִ֣ד עַבְדּ֑וֹ וַ֝יִּקָּחֵ֗הוּ מִֽמִּכְלְאֹ֥ת צֹֽאן׃
71עָל֗וֹת הֱ֫בִיא֥וֹ לִ֭רְעוֹת בְּיַעֲקֹ֣ב עַמּ֑וֹ וּ֝בְיִשְׂרָאֵ֗ל נַחֲלָתֽוֹ׃
