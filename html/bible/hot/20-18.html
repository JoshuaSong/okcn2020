1לְֽ֭תַאֲוָה יְבַקֵּ֣שׁ נִפְרָ֑ד בְּכָל־ תּ֝וּשִׁיָּ֗ה יִתְגַּלָּֽע׃
2יַחְפֹּ֣ץ כְּ֭סִיל בִּתְבוּנָ֑ה כִּ֝֗י אִֽם־ בְּהִתְגַּלּ֥וֹת לִבּֽוֹ׃
3רָ֭שָׁע בָּ֣א גַם־ בּ֑וּז וְֽעִם־ קָל֥וֹן חֶרְפָּֽה׃
4עֲ֭מֻקִּים דִּבְרֵ֣י פִי־ אִ֑ישׁ נַ֥חַל נֹ֝בֵ֗עַ מְק֣וֹר חָכְמָֽה׃
5פְּנֵי־ רָשָׁ֣ע לֹא־ ט֑וֹב לְהַטּ֥וֹת צַ֝דִּ֗יק בַּמִּשְׁפָּֽט׃
6כְ֭סִיל יָבֹ֣אֽוּ בְרִ֑יב וּ֝פִ֗יו לְֽמַהֲלֻמ֥וֹת יִקְרָֽא׃
7כְ֭סִיל מְחִתָּה־ ל֑וֹ וּ֝שְׂפָתָ֗יו מוֹקֵ֥שׁ נַפְשֽׁוֹ׃
8נִ֭רְגָּן כְּמִֽתְלַהֲמִ֑ים וְ֝הֵ֗ם יָרְד֥וּ חַדְרֵי־ בָֽטֶן׃
9מִתְרַפֶּ֣ה בִמְלַאכְתּ֑וֹ אָ֥ח ה֝֗וּא לְבַ֣עַל מַשְׁחִֽית׃
10עֹ֭ז שֵׁ֣ם יְהוָ֑ה בּֽוֹ־ יָר֖וּץ צַדִּ֣יק וְנִשְׂגָּֽב׃
11עָ֭שִׁיר קִרְיַ֣ת עֻזּ֑וֹ וּכְחוֹמָ֥ה נִ֝שְׂגָּבָ֗ה בְּמַשְׂכִּיתֽוֹ׃
12שֶׁ֭בֶר יִגְבַּ֣הּ לֵב־ אִ֑ישׁ וְלִפְנֵ֖י כָב֣וֹד עֲנָוָֽה׃
13דָּ֭בָר בְּטֶ֣רֶם יִשְׁמָ֑ע אִוֶּ֥לֶת הִיא־ ל֝֗וֹ וּכְלִמָּֽה׃
14אִ֭ישׁ יְכַלְכֵּ֣ל מַחֲלֵ֑הוּ וְר֥וּחַ נְ֝כֵאָ֗ה מִ֣י יִשָּׂאֶֽנָּה׃
15נָ֭בוֹן יִקְנֶה־ דָּ֑עַת וְאֹ֥זֶן חֲ֝כָמִ֗ים תְּבַקֶּשׁ־ דָּֽעַת׃
16אָ֭דָם יַרְחִ֣יב ל֑וֹ וְלִפְנֵ֖י גְדֹלִ֣ים יַנְחֶֽנּוּ׃
17הָרִאשׁ֣וֹן בְּרִיב֑וֹ [יָבֹא־ כ] (וּבָֽא־ ק) וַחֲקָרֽוֹ׃
18יַשְׁבִּ֣ית הַגּוֹרָ֑ל וּבֵ֖ין עֲצוּמִ֣ים יַפְרִֽיד׃
19נִפְשָׁ֥ע מִקִּרְיַת־ עֹ֑ז [וּמְדֹונִים כ] (וּ֝מִדְיָנִ֗ים ק) כִּבְרִ֥יחַ אַרְמֽוֹן׃
20פִי־ אִ֭ישׁ תִּשְׂבַּ֣ע בִּטְנ֑וֹ תְּבוּאַ֖ת שְׂפָתָ֣יו יִשְׂבָּֽע׃
21וְ֭חַיִּים בְּיַד־ לָשׁ֑וֹן וְ֝אֹהֲבֶ֗יהָ יֹאכַ֥ל פִּרְיָֽהּ׃
22אִ֭שָּׁה מָ֣צָא ט֑וֹב וַיָּ֥פֶק רָ֝צ֗וֹן מֵיְהוָֽה׃
23יְדַבֶּר־ רָ֑שׁ וְ֝עָשִׁ֗יר יַעֲנֶ֥ה עַזּֽוֹת׃
