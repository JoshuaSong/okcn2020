1וַיַּ֥עַן אִיּ֗וֹב וַיֹּאמַֽר׃
2כְאֵ֣לֶּה רַבּ֑וֹת מְנַחֲמֵ֖י עָמָ֣ל כֻּלְּכֶֽם׃
3לְדִבְרֵי־ ר֑וּחַ א֥וֹ מַה־ יַּ֝מְרִֽיצְךָ֗ כִּ֣י תַעֲנֶֽה׃
4אָנֹכִי֮ כָּכֶ֪ם אֲדַ֫בֵּ֥רָה ל֤וּ־ יֵ֪שׁ נַפְשְׁכֶ֡ם תַּ֤חַת נַפְשִׁ֗י אַחְבִּ֣ירָה עֲלֵיכֶ֣ם בְּמִלִּ֑ים וְאָנִ֥יעָה עֲ֝לֵיכֶ֗ם בְּמ֣וֹ רֹאשִֽׁי׃
5בְּמוֹ־ פִ֑י וְנִ֖יד שְׂפָתַ֣י יַחְשֹֽׂךְ׃
6אֲ֭דַבְּרָה לֹא־ יֵחָשֵׂ֣ךְ כְּאֵבִ֑י וְ֝אַחְדְּלָ֗ה מַה־ מִנִּ֥י יַהֲלֹֽךְ׃
7עַתָּ֥ה הֶלְאָ֑נִי הֲ֝שִׁמּ֗וֹתָ כָּל־ עֲדָתִֽי׃
8לְעֵ֣ד הָיָ֑ה וַיָּ֥קָם בִּ֥י כַ֝חֲשִׁ֗י בְּפָנַ֥י יַעֲנֶֽה׃
9טָרַ֨ף ׀ וַֽיִּשְׂטְמֵ֗נִי חָרַ֣ק עָלַ֣י בְּשִׁנָּ֑יו צָרִ֓י ׀ יִלְט֖וֹשׁ עֵינָ֣יו לִֽי׃
10עָלַ֨י ׀ בְּפִיהֶ֗ם בְּ֭חֶרְפָּה הִכּ֣וּ לְחָיָ֑י יַ֝֗חַד עָלַ֥י יִתְמַלָּאֽוּן׃
11אֵ֭ל אֶ֣ל עֲוִ֑יל וְעַל־ יְדֵ֖י רְשָׁעִ֣ים יִרְטֵֽנִי׃
12הָיִ֨יתִי ׀ וַֽיְפַרְפְּרֵ֗נִי וְאָחַ֣ז בְּ֭עָרְפִּי וַֽיְפַצְפְּצֵ֑נִי וַיְקִימֵ֥נִי ל֝֗וֹ לְמַטָּרָֽה׃
13עָלַ֨י ׀ רַבָּ֗יו יְפַלַּ֣ח כִּ֭לְיוֹתַי וְלֹ֣א יַחְמ֑וֹל יִשְׁפֹּ֥ךְ לָ֝אָ֗רֶץ מְרֵרָֽתִי׃
14פֶ֭רֶץ עַל־ פְּנֵי־ פָ֑רֶץ יָרֻ֖ץ עָלַ֣י כְּגִבּֽוֹר׃
15תָּ֭פַרְתִּי עֲלֵ֣י גִלְדִּ֑י וְעֹלַ֖לְתִּי בֶעָפָ֣ר קַרְנִֽי׃
16[חֳמַרְמְרָה כ] (חֳ֭מַרְמְרוּ ק) מִנִּי־ בֶ֑כִי וְעַ֖ל עַפְעַפַּ֣י צַלְמָֽוֶת׃
17לֹא־ חָמָ֣ס בְּכַפָּ֑י וּֽתְפִלָּתִ֥י זַכָּֽה׃
18אַל־ תְּכַסִּ֣י דָמִ֑י וְֽאַל־ יְהִ֥י מָ֝ק֗וֹם לְזַעֲקָתִֽי׃
19עַ֭תָּה הִנֵּה־ בַשָּׁמַ֣יִם עֵדִ֑י וְ֝שָׂהֲדִ֗י בַּמְּרוֹמִֽים׃
20רֵעָ֑י אֶל־ אֱ֝ל֗וֹהַ דָּלְפָ֥ה עֵינִֽי׃
21לְגֶ֣בֶר עִם־ אֱל֑וֹהַּ וּֽבֶן־ אָדָ֥ם לְרֵעֵֽהוּ׃
