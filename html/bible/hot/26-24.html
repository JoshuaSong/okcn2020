1וַיְהִי֩ דְבַר־ יְהוָ֨ה אֵלַ֜י בַּשָּׁנָ֤ה הַתְּשִׁיעִית֙ בַּחֹ֣דֶשׁ הָעֲשִׂירִ֔י בֶּעָשׂ֥וֹר לַחֹ֖דֶשׁ לֵאמֹֽר׃
2אָדָ֗ם [כְּתֹוב־ כ] (כְּתָב־ לְךָ֙ ק) אֶת־ שֵׁ֣ם הַיּ֔וֹם אֶת־ עֶ֖צֶם הַיּ֣וֹם הַזֶּ֑ה סָמַ֤ךְ מֶֽלֶךְ־ בָּבֶל֙ אֶל־ יְר֣וּשָׁלִַ֔ם בְּעֶ֖צֶם הַיּ֥וֹם הַזֶּֽה׃
3אֶל־ בֵּית־ הַמֶּ֙רִי֙ מָשָׁ֔ל וְאָמַרְתָּ֣ אֲלֵיהֶ֔ם כֹּ֥ה אָמַ֖ר אֲדֹנָ֣י יְהוִ֑ה שְׁפֹ֤ת הַסִּיר֙ שְׁפֹ֔ת וְגַם־ יְצֹ֥ק בּ֖וֹ מָֽיִם׃
4נְתָחֶ֙יהָ֙ אֵלֶ֔יהָ כָּל־ נֵ֥תַח ט֖וֹב יָרֵ֣ךְ וְכָתֵ֑ף מִבְחַ֥ר עֲצָמִ֖ים מַלֵּֽא׃
5הַצֹּאן֙ לָק֔וֹחַ וְגַ֛ם דּ֥וּר הָעֲצָמִ֖ים תַּחְתֶּ֑יהָ רַתַּ֣ח רְתָחֶ֔יהָ גַּם־ בָּשְׁל֥וּ עֲצָמֶ֖יהָ בְּתוֹכָֽהּ׃ ס
6כֹּה־ אָמַ֣ר ׀ אֲדֹנָ֣י יְהֹוִ֗ה אוֹי֮ עִ֣יר הַדָּמִים֒ סִ֚יר אֲשֶׁ֣ר חֶלְאָתָ֣ה בָ֔הּ וְחֶ֨לְאָתָ֔הּ לֹ֥א יָצְאָ֖ה מִמֶּ֑נָּה לִנְתָחֶ֤יהָ לִנְתָחֶ֙יהָ֙ הוֹצִיאָ֔הּ לֹא־ נָפַ֥ל עָלֶ֖יהָ גּוֹרָֽל׃
7דָמָהּ֙ בְּתוֹכָ֣הּ הָיָ֔ה עַל־ צְחִ֥יחַ סֶ֖לַע שָׂמָ֑תְהוּ לֹ֤א שְׁפָכַ֙תְהוּ֙ עַל־ הָאָ֔רֶץ לְכַסּ֥וֹת עָלָ֖יו עָפָֽר׃
8חֵמָה֙ לִנְקֹ֣ם נָקָ֔ם נָתַ֥תִּי אֶת־ דָּמָ֖הּ עַל־ צְחִ֣יחַ סָ֑לַע לְבִלְתִּ֖י הִכָּסֽוֹת׃ פ
9כֹּ֤ה אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה א֖וֹי עִ֣יר הַדָּמִ֑ים גַּם־ אֲנִ֖י אַגְדִּ֥יל הַמְּדוּרָֽה׃
10הָעֵצִים֙ הַדְלֵ֣ק הָאֵ֔שׁ הָתֵ֖ם הַבָּשָׂ֑ר וְהַרְקַח֙ הַמֶּרְקָחָ֔ה וְהָעֲצָמ֖וֹת יֵחָֽרוּ׃
11עַל־ גֶּחָלֶ֖יהָ רֵקָ֑ה לְמַ֨עַן תֵּחַ֜ם וְחָ֣רָה נְחֻשְׁתָּ֗הּ וְנִתְּכָ֤ה בְתוֹכָהּ֙ טֻמְאָתָ֔הּ תִּתֻּ֖ם חֶלְאָתָֽהּ׃
12הֶלְאָ֑ת וְלֹֽא־ תֵצֵ֤א מִמֶּ֙נָּה֙ רַבַּ֣ת חֶלְאָתָ֔הּ בְּאֵ֖שׁ חֶלְאָתָֽהּ׃
13זִמָּ֑ה יַ֤עַן טִֽהַרְתִּיךְ֙ וְלֹ֣א טָהַ֔רְתְּ מִטֻּמְאָתֵךְ֙ לֹ֣א תִטְהֲרִי־ ע֔וֹד עַד־ הֲנִיחִ֥י אֶת־ חֲמָתִ֖י בָּֽךְ׃
14יְהוָ֤ה דִּבַּ֙רְתִּי֙ בָּאָ֣ה וְעָשִׂ֔יתִי לֹֽא־ אֶפְרַ֥ע וְלֹא־ אָח֖וּס וְלֹ֣א אֶנָּחֵ֑ם כִּדְרָכַ֤יִךְ וְכַעֲלִילוֹתַ֙יִךְ֙ שְׁפָט֔וּךְ נְאֻ֖ם אֲדֹנָ֥י יְהֹוִֽה׃ פ
15דְבַר־ יְהוָ֖ה אֵלַ֥י לֵאמֹֽר׃
16אָדָ֕ם הִנְנִ֨י לֹקֵ֧חַ מִמְּךָ֛ אֶת־ מַחְמַ֥ד עֵינֶ֖יךָ בְּמַגֵּפָ֑ה וְלֹ֤א תִסְפֹּד֙ וְלֹ֣א תִבְכֶּ֔ה וְל֥וֹא תָב֖וֹא דִּמְעָתֶֽךָ׃
17דֹּ֗ם מֵתִים֙ אֵ֣בֶל לֹֽא־ תַֽעֲשֶׂ֔ה פְאֵֽרְךָ֙ חֲב֣וֹשׁ עָלֶ֔יךָ וּנְעָלֶ֖יךָ תָּשִׂ֣ים בְּרַגְלֶ֑יךָ וְלֹ֤א תַעְטֶה֙ עַל־ שָׂפָ֔ם וְלֶ֥חֶם אֲנָשִׁ֖ים לֹ֥א תֹאכֵֽל׃
18אֶל־ הָעָם֙ בַּבֹּ֔קֶר וַתָּ֥מָת אִשְׁתִּ֖י בָּעָ֑רֶב וָאַ֥עַשׂ בַּבֹּ֖קֶר כַּאֲשֶׁ֥ר צֻוֵּֽיתִי׃
19אֵלַ֖י הָעָ֑ם הֲלֹֽא־ תַגִּ֥יד לָ֙נוּ֙ מָה־ אֵ֣לֶּה לָּ֔נוּ כִּ֥י אַתָּ֖ה עֹשֶֽׂה׃
20אֲלֵיהֶ֑ם דְּבַ֨ר־ יְהוָ֔ה הָיָ֥ה אֵלַ֖י לֵאמֹֽר׃
21לְבֵ֣ית יִשְׂרָאֵ֗ל כֹּֽה־ אָמַר֮ אֲדֹנָ֣י יְהוִה֒ הִנְנִ֨י מְחַלֵּ֤ל אֶת־ מִקְדָּשִׁי֙ גְּא֣וֹן עֻזְּכֶ֔ם מַחְמַ֥ד עֵֽינֵיכֶ֖ם וּמַחְמַ֣ל נַפְשְׁכֶ֑ם וּבְנֵיכֶ֧ם וּבְנֽוֹתֵיכֶ֛ם אֲשֶׁ֥ר עֲזַבְתֶּ֖ם בַּחֶ֥רֶב יִפֹּֽלוּ׃
22כַּאֲשֶׁ֣ר עָשִׂ֑יתִי עַל־ שָׂפָם֙ לֹ֣א תַעְט֔וּ וְלֶ֥חֶם אֲנָשִׁ֖ים לֹ֥א תֹאכֵֽלוּ׃
23עַל־ רָאשֵׁיכֶ֗ם וְנַֽעֲלֵיכֶם֙ בְּרַגְלֵיכֶ֔ם לֹ֥א תִסְפְּד֖וּ וְלֹ֣א תִבְכּ֑וּ וּנְמַקֹּתֶם֙ בַּעֲוֹנֹ֣תֵיכֶ֔ם וּנְהַמְתֶּ֖ם אִ֥ישׁ אֶל־ אָחִֽיו׃
24יְחֶזְקֵ֤אל לָכֶם֙ לְמוֹפֵ֔ת כְּכֹ֥ל אֲשֶׁר־ עָשָׂ֖ה תַּעֲשׂ֑וּ בְּבֹאָ֕הּ וִֽידַעְתֶּ֕ם כִּ֥י אֲנִ֖י אֲדֹנָ֥י יְהוִֽה׃ ס
25בֶן־ אָדָ֔ם הֲל֗וֹא בְּי֨וֹם קַחְתִּ֤י מֵהֶם֙ אֶת־ מָ֣עוּזָּ֔ם מְשׂ֖וֹשׂ תִּפְאַרְתָּ֑ם אֶת־ מַחְמַ֤ד עֵֽינֵיהֶם֙ וְאֶת־ מַשָּׂ֣א נַפְשָׁ֔ם בְּנֵיהֶ֖ם וּבְנוֹתֵיהֶֽם׃
26הַה֔וּא יָב֥וֹא הַפָּלִ֖יט אֵלֶ֑יךָ לְהַשְׁמָע֖וּת אָזְנָֽיִם׃
