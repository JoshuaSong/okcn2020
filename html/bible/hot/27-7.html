1בִּשְׁנַ֣ת חֲדָ֗ה לְבֵלְאשַׁצַּר֙ מֶ֣לֶךְ בָּבֶ֔ל דָּנִיֵּאל֙ חֵ֣לֶם חֲזָ֔ה וְחֶזְוֵ֥י רֵאשֵׁ֖הּ עַֽל־ מִשְׁכְּבֵ֑הּ בֵּאדַ֙יִן֙ חֶלְמָ֣א כְתַ֔ב רֵ֥אשׁ מִלִּ֖ין אֲמַֽר׃
2דָנִיֵּאל֙ וְאָמַ֔ר חָזֵ֥ה הֲוֵ֛ית בְּחֶזְוִ֖י עִם־ לֵֽילְיָ֑א וַאֲר֗וּ אַרְבַּע֙ רוּחֵ֣י שְׁמַיָּ֔א מְגִיחָ֖ן לְיַמָּ֥א רַבָּֽא׃
3חֵיוָן֙ רַבְרְבָ֔ן סָלְקָ֖ן מִן־ יַמָּ֑א שָׁנְיָ֖ן דָּ֥א מִן־ דָּֽא׃
4כְאַרְיֵ֔ה וְגַפִּ֥ין דִּֽי־ נְשַׁ֖ר לַ֑הּ חָזֵ֣ה הֲוֵ֡ית עַד֩ דִּי־ מְּרִ֨יטוּ גַפַּ֜יהּ וּנְטִ֣ילַת מִן־ אַרְעָ֗א וְעַל־ רַגְלַ֙יִן֙ כֶּאֱנָ֣שׁ הֳקִימַ֔ת וּלְבַ֥ב אֱנָ֖שׁ יְהִ֥יב לַֽהּ׃
5חֵיוָה֩ אָחֳרִ֨י תִנְיָנָ֜ה דָּמְיָ֣ה לְדֹ֗ב וְלִשְׂטַר־ חַד֙ הֳקִמַ֔ת וּתְלָ֥ת עִלְעִ֛ין בְּפֻמַּ֖הּ בֵּ֣ין [שִׁנַּיַּהּ כ] (שִׁנַּ֑הּ ק) וְכֵן֙ אָמְרִ֣ין לַ֔הּ ק֥וּמִֽי אֲכֻ֖לִי בְּשַׂ֥ר שַׂגִּֽיא׃
6דְּנָ֜ה חָזֵ֣ה הֲוֵ֗ית וַאֲר֤וּ אָֽחֳרִי֙ כִּנְמַ֔ר וְלַ֨הּ גַּפִּ֥ין אַרְבַּ֛ע דִּי־ ע֖וֹף עַל־ [גַּבַּיַּהּ כ] (גַּבַּ֑הּ ק) וְאַרְבְּעָ֤ה רֵאשִׁין֙ לְחֵ֣יוְתָ֔א וְשָׁלְטָ֖ן יְהִ֥יב לַֽהּ׃
7דְּנָה֩ חָזֵ֨ה הֲוֵ֜ית בְּחֶזְוֵ֣י לֵֽילְיָ֗א וַאֲר֣וּ חֵיוָ֣ה [רְבִיעָיָה כ] (רְֽבִיעָאָ֡ה ק) דְּחִילָה֩ וְאֵֽימְתָנִ֨י וְתַקִּיפָ֜א יַתִּ֗ירָא וְשִׁנַּ֨יִן דִּֽי־ פַרְזֶ֥ל לַהּ֙ רַבְרְבָ֔ן אָֽכְלָ֣ה וּמַדֱּקָ֔ה וּשְׁאָרָ֖א [בְּרַגְלַיַּהּ כ] (בְּרַגְלַ֣הּ ק) רָפְסָ֑ה וְהִ֣יא מְשַׁנְּיָ֗ה מִן־ כָּל־ חֵֽיוָתָא֙ דִּ֣י קָֽדָמַ֔יהּ וְקַרְנַ֥יִן עֲשַׂ֖ר לַֽהּ׃
8הֲוֵ֜ית בְּקַרְנַיָּ֗א וַ֠אֲלוּ קֶ֣רֶן אָחֳרִ֤י זְעֵירָה֙ סִלְקָ֣ת [בֵּינֵיהֹון כ] (בֵּֽינֵיהֵ֔ן ק) וּתְלָ֗ת מִן־ קַרְנַיָּא֙ קַדְמָ֣יָתָ֔א [אֶתְעֲקַרוּ כ] (אֶתְעֲקַ֖רָה ק) מִן־ [קֳדָמַיַּהּ כ] (קֳדָמַ֑הּ ק) וַאֲל֨וּ עַיְנִ֜ין כְּעַיְנֵ֤י אֲנָשָׁא֙ בְּקַרְנָא־ דָ֔א וּפֻ֖ם מְמַלִּ֥ל רַבְרְבָֽן׃
9הֲוֵ֗ית עַ֣ד דִּ֤י כָרְסָוָן֙ רְמִ֔יו וְעַתִּ֥יק יוֹמִ֖ין יְתִ֑ב לְבוּשֵׁ֣הּ ׀ כִּתְלַ֣ג חִוָּ֗ר וּשְׂעַ֤ר רֵאשֵׁהּ֙ כַּעֲמַ֣ר נְקֵ֔א כָּרְסְיֵהּ֙ שְׁבִיבִ֣ין דִּי־ נ֔וּר גַּלְגִּלּ֖וֹהִי נ֥וּר דָּלִֽק׃
10דִּי־ נ֗וּר נָגֵ֤ד וְנָפֵק֙ מִן־ קֳדָמ֔וֹהִי אֶ֤לֶף [אַלְפַיִם כ] (אַלְפִין֙ ק) יְשַׁמְּשׁוּנֵּ֔הּ וְרִבּ֥וֹ [רַבְּוָן כ] (רִבְבָ֖ן ק) קָֽדָמ֣וֹהִי יְקוּמ֑וּן דִּינָ֥א יְתִ֖ב וְסִפְרִ֥ין פְּתִֽיחוּ׃
11הֲוֵ֔ית בֵּאדַ֗יִן מִן־ קָל֙ מִלַּיָּ֣א רַבְרְבָתָ֔א דִּ֥י קַרְנָ֖א מְמַלֱּלָ֑ה חָזֵ֣ה הֲוֵ֡ית עַד֩ דִּ֨י קְטִילַ֤ת חֵֽיוְתָא֙ וְהוּבַ֣ד גִּשְׁמַ֔הּ וִיהִיבַ֖ת לִיקֵדַ֥ת אֶשָּֽׁא׃
12חֵֽיוָתָ֔א הֶעְדִּ֖יו שָׁלְטָנְה֑וֹן וְאַרְכָ֧ה בְחַיִּ֛ין יְהִ֥יבַת לְה֖וֹן עַד־ זְמַ֥ן וְעִדָּֽן׃
13הֲוֵית֙ בְּחֶזְוֵ֣י לֵֽילְיָ֔א וַאֲרוּ֙ עִם־ עֲנָנֵ֣י שְׁמַיָּ֔א כְּבַ֥ר אֱנָ֖שׁ אָתֵ֣ה הֲוָ֑ה וְעַד־ עַתִּ֤יק יֽוֹמַיָּא֙ מְטָ֔ה וּקְדָמ֖וֹהִי הַקְרְבֽוּהִי׃
14יְהִ֤יב שָׁלְטָן֙ וִיקָ֣ר וּמַלְכ֔וּ וְכֹ֣ל עַֽמְמַיָּ֗א אֻמַיָּ֛א וְלִשָּׁנַיָּ֖א לֵ֣הּ יִפְלְח֑וּן שָׁלְטָנֵ֞הּ שָׁלְטָ֤ן עָלַם֙ דִּֽי־ לָ֣א יֶעְדֵּ֔ה וּמַלְכוּתֵ֖הּ דִּי־ לָ֥א תִתְחַבַּֽל׃ פ
15רוּחִ֛י אֲנָ֥ה דָנִיֵּ֖אל בְּג֣וֹא נִדְנֶ֑ה וְחֶזְוֵ֥י רֵאשִׁ֖י יְבַהֲלֻנַּֽנִי׃
16עַל־ חַד֙ מִן־ קָ֣אֲמַיָּ֔א וְיַצִּיבָ֥א אֶבְעֵֽא־ מִנֵּ֖הּ עַֽל־ כָּל־ דְּנָ֑ה וַאֲמַר־ לִ֕י וּפְשַׁ֥ר מִלַּיָּ֖א יְהוֹדְעִנַּֽנִי׃
17חֵיוָתָ֣א רַבְרְבָתָ֔א דִּ֥י אִנִּ֖ין אַרְבַּ֑ע אַרְבְּעָ֥ה מַלְכִ֖ין יְקוּמ֥וּן מִן־ אַרְעָֽא׃
18מַלְכוּתָ֔א קַדִּישֵׁ֖י עֶלְיוֹנִ֑ין וְיַחְסְנ֤וּן מַלְכוּתָא֙ עַֽד־ עָ֣לְמָ֔א וְעַ֖ד עָלַ֥ם עָלְמַיָּֽא׃
19צְבִית֙ לְיַצָּבָ֔א עַל־ חֵֽיוְתָא֙ רְבִיעָ֣יְתָ֔א דִּֽי־ הֲוָ֥ת שָֽׁנְיָ֖ה מִן־ [כָּלְּהֹון כ] (כָּלְּהֵ֑ין ק) דְּחִילָ֣ה יַתִּ֗ירָה [שִׁנַּיַּהּ כ] (שִׁנַּ֤הּ ק) דִּֽי־ פַרְזֶל֙ וְטִפְרַ֣יהּ דִּֽי־ נְחָ֔שׁ אָֽכְלָ֣ה מַדֲּקָ֔ה וּשְׁאָרָ֖א בְּרַגְלַ֥יהּ רָֽפְסָֽה׃
20קַרְנַיָּ֤א עֲשַׂר֙ דִּ֣י בְרֵאשַׁ֔הּ וְאָחֳרִי֙ דִּ֣י סִלְקַ֔ת [וּנְפַלוּ כ] (וּנְפַ֥לָה ק) מִן־ [קֳדָמַיַּהּ כ] (קֳדָמַ֖הּ ק) תְּלָ֑ת וְקַרְנָ֨א דִכֵּ֜ן וְעַיְנִ֣ין לַ֗הּ וְפֻם֙ מְמַלִּ֣ל רַבְרְבָ֔ן וְחֶזְוַ֖הּ רַ֥ב מִן־ חַבְרָתַֽהּ׃
21הֲוֵ֔ית וְקַרְנָ֣א דִכֵּ֔ן עָבְדָ֥ה קְרָ֖ב עִם־ קַדִּישִׁ֑ין וְיָכְלָ֖ה לְהֽוֹן׃
22דִּֽי־ אֲתָ֗ה עַתִּיק֙ יֽוֹמַיָּ֔א וְדִינָ֣א יְהִ֔ב לְקַדִּישֵׁ֖י עֶלְיוֹנִ֑ין וְזִמְנָ֣א מְטָ֔ה וּמַלְכוּתָ֖א הֶחֱסִ֥נוּ קַדִּישִֽׁין׃
23אֲמַר֒ חֵֽיוְתָא֙ רְבִיעָ֣יְתָ֔א מַלְכ֤וּ [רְבִיעָיָא כ] (רְבִיעָאָה֙ ק) תֶּהֱוֵ֣א בְאַרְעָ֔א דִּ֥י תִשְׁנֵ֖א מִן־ כָּל־ מַלְכְוָתָ֑א וְתֵאכֻל֙ כָּל־ אַרְעָ֔א וּתְדוּשִׁנַּ֖הּ וְתַדְּקִנַּֽהּ׃
24עֲשַׂ֔ר מִנַּהּ֙ מַלְכוּתָ֔ה עַשְׂרָ֥ה מַלְכִ֖ין יְקֻמ֑וּן וְאָחֳרָ֞ן יְק֣וּם אַחֲרֵיה֗וֹן וְה֤וּא יִשְׁנֵא֙ מִן־ קַדְמָיֵ֔א וּתְלָתָ֥ה מַלְכִ֖ין יְהַשְׁפִּֽל׃
25לְצַ֤ד [עִלָּיָא כ] (עִלָּאָה֙ ק) יְמַלִּ֔ל וּלְקַדִּישֵׁ֥י עֶלְיוֹנִ֖ין יְבַלֵּ֑א וְיִסְבַּ֗ר לְהַשְׁנָיָה֙ זִמְנִ֣ין וְדָ֔ת וְיִתְיַהֲב֣וּן בִּידֵ֔הּ עַד־ עִדָּ֥ן וְעִדָּנִ֖ין וּפְלַ֥ג עִדָּֽן׃
26יִתִּ֑ב וְשָׁלְטָנֵ֣הּ יְהַעְדּ֔וֹן לְהַשְׁמָדָ֥ה וּלְהוֹבָדָ֖ה עַד־ סוֹפָֽא׃
27וְשָׁלְטָנָ֜א וּרְבוּתָ֗א דִּ֚י מַלְכְוָת֙ תְּח֣וֹת כָּל־ שְׁמַיָּ֔א יְהִיבַ֕ת לְעַ֖ם קַדִּישֵׁ֣י עֶלְיוֹנִ֑ין מַלְכוּתֵהּ֙ מַלְכ֣וּת עָלַ֔ם וְכֹל֙ שָׁלְטָ֣נַיָּ֔א לֵ֥הּ יִפְלְח֖וּן וְיִֽשְׁתַּמְּעֽוּן׃
