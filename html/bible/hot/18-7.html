1הֲלֹא־ צָבָ֣א לֶאֱנ֣וֹשׁ [עַל־ כ] (עֲלֵי־ אָ֑רֶץ ק) וְכִימֵ֖י שָׂכִ֣יר יָמָֽיו׃
2יִשְׁאַף־ צֵ֑ל וּ֝כְשָׂכִ֗יר יְקַוֶּ֥ה פָעֳלֽוֹ׃
3הָנְחַ֣לְתִּי לִ֭י יַרְחֵי־ שָׁ֑וְא וְלֵיל֥וֹת עָ֝מָ֗ל מִנּוּ־ לִֽי׃
4שָׁכַ֗בְתִּי וְאָמַ֗רְתִּי מָתַ֣י אָ֭קוּם וּמִדַּד־ עָ֑רֶב וְשָׂבַ֖עְתִּי נְדֻדִ֣ים עֲדֵי־ נָֽשֶׁף׃
5בְּשָׂרִ֣י רִ֭מָּה [וְגִישׁ כ] (וְג֣וּשׁ ק) עָפָ֑ר עוֹרִ֥י רָ֝גַ֗ע וַיִּמָּאֵֽס׃
6קַ֭לּוּ מִנִּי־ אָ֑רֶג וַ֝יִּכְל֗וּ בְּאֶ֣פֶס תִּקְוָֽה׃
7כִּי־ ר֣וּחַ חַיָּ֑י לֹא־ תָשׁ֥וּב עֵ֝ינִ֗י לִרְא֥וֹת טֽוֹב׃
8תְ֭שׁוּרֵנִי עֵ֣ין רֹ֑אִי עֵינֶ֖יךָ בִּ֣י וְאֵינֶֽנִּי׃
9עָ֭נָן וַיֵּלַ֑ךְ כֵּ֥ן יוֹרֵ֥ד שְׁ֝א֗וֹל לֹ֣א יַעֲלֶֽה׃
10יָשׁ֣וּב ע֣וֹד לְבֵית֑וֹ וְלֹא־ יַכִּירֶ֖נּוּ ע֣וֹד מְקֹמֽוֹ׃
11אֲנִי֮ לֹ֤א אֶחֱשָׂ֫ךְ פִּ֥י אֲ‍ֽ֭דַבְּרָה בְּצַ֣ר רוּחִ֑י אָ֝שִׂ֗יחָה בְּמַ֣ר נַפְשִֽׁי׃
12אָ֭נִי אִם־ תַּנִּ֑ין כִּֽי־ תָשִׂ֖ים עָלַ֣י מִשְׁמָֽר׃
13אָ֭מַרְתִּי תְּנַחֲמֵ֣נִי עַרְשִׂ֑י יִשָּׂ֥א בְ֝שִׂיחִ֗י מִשְׁכָּבִֽי׃
14בַחֲלֹמ֑וֹת וּֽמֵחֶזְיֹנ֥וֹת תְּבַעֲתַֽנִּי׃
15מַחֲנָ֣ק נַפְשִׁ֑י מָ֝֗וֶת מֵֽעַצְמוֹתָֽי׃
16לֹא־ לְעֹלָ֣ם אֶֽחְיֶ֑ה חֲדַ֥ל מִ֝מֶּ֗נִּי כִּי־ הֶ֥בֶל יָמָֽי׃
17אֱ֭נוֹשׁ כִּ֣י תְגַדְּלֶ֑נּוּ וְכִי־ תָשִׁ֖ית אֵלָ֣יו לִבֶּֽךָ׃
18לִבְקָרִ֑ים לִ֝רְגָעִ֗ים תִּבְחָנֶֽנּוּ׃
19לֹא־ תִשְׁעֶ֣ה מִמֶּ֑נִּי לֹֽא־ תַ֝רְפֵּ֗נִי עַד־ בִּלְעִ֥י רֻקִּֽי׃
20מָ֤ה אֶפְעַ֨ל ׀ לָךְ֮ נֹצֵ֪ר הָאָ֫דָ֥ם לָ֤מָה שַׂמְתַּ֣נִי לְמִפְגָּ֣ע לָ֑ךְ וָאֶהְיֶ֖ה עָלַ֣י לְמַשָּֽׂא׃
