1וְעַתָּ֤ה ׀ שָֽׂחֲק֣וּ עָלַי֮ צְעִירִ֥ים מִמֶּ֗נִּי לְיָ֫מִ֥ים אֲשֶׁר־ מָאַ֥סְתִּי אֲבוֹתָ֑ם לָ֝שִׁ֗ית עִם־ כַּלְבֵ֥י צֹאנִֽי׃
2כֹּ֣חַ יְ֭דֵיהֶם לָ֣מָּה לִּ֑י עָ֝לֵ֗ימוֹ אָ֣בַד כָּֽלַח׃
3וּבְכָפָ֗ן גַּ֫לְמ֥וּד הַֽעֹרְקִ֥ים צִיָּ֑ה אֶ֝֗מֶשׁ שׁוֹאָ֥ה וּמְשֹׁאָֽה׃
4מַלּ֣וּחַ עֲלֵי־ שִׂ֑יחַ וְשֹׁ֖רֶשׁ רְתָמִ֣ים לַחְמָֽם׃
5גֵּ֥ו יְגֹרָ֑שׁוּ יָרִ֥יעוּ עָ֝לֵ֗ימוֹ כַּגַּנָּֽב׃
6נְחָלִ֣ים לִשְׁכֹּ֑ן חֹרֵ֖י עָפָ֣ר וְכֵפִֽים׃
7שִׂיחִ֥ים יִנְהָ֑קוּ תַּ֖חַת חָר֣וּל יְסֻפָּֽחוּ׃
8נָ֭בָל גַּם־ בְּנֵ֣י בְלִי־ שֵׁ֑ם נִ֝כְּא֗וּ מִן־ הָאָֽרֶץ׃
9נְגִינָתָ֣ם הָיִ֑יתִי וָאֱהִ֖י לָהֶ֣ם לְמִלָּֽה׃
10רָ֣חֲקוּ מֶ֑נִּי וּ֝מִפָּנַ֗י לֹא־ חָ֥שְׂכוּ רֹֽק׃
11[יִתְרֹו כ] (יִתְרִ֣י ק) פִ֭תַּח וַיְעַנֵּ֑נִי וְ֝רֶ֗סֶן מִפָּנַ֥י שִׁלֵּֽחוּ׃
12יָמִין֮ פִּרְחַ֪ח יָ֫ק֥וּמוּ רַגְלַ֥י שִׁלֵּ֑חוּ וַיָּסֹ֥לּוּ עָ֝לַ֗י אָרְח֥וֹת אֵידָֽם׃
13נְֽתִיבָ֫תִ֥י לְהַוָּתִֽי יֹעִ֑ילוּ לֹ֖א עֹזֵ֣ר לָֽמוֹ׃
14רָחָ֣ב יֶאֱתָ֑יוּ תַּ֥חַת שֹׁ֝אָ֗ה הִתְגַּלְגָּֽלוּ׃
15עָלַ֗י בַּלָּ֫ה֥וֹת תִּרְדֹּ֣ף כָּ֭רוּחַ נְדִבָתִ֑י וּ֝כְעָ֗ב עָבְרָ֥ה יְשֻׁעָתִֽי׃
16עָ֭לַי תִּשְׁתַּפֵּ֣ךְ נַפְשִׁ֑י יֹ֖אחֲז֣וּנִי יְמֵי־ עֹֽנִי׃
17עֲ֭צָמַי נִקַּ֣ר מֵעָלָ֑י וְ֝עֹרְקַ֗י לֹ֣א יִשְׁכָּבֽוּן׃
18כֹּ֭חַ יִתְחַפֵּ֣שׂ לְבוּשִׁ֑י כְּפִ֖י כֻתָּנְתִּ֣י יַֽאַזְרֵֽנִי׃
19לַחֹ֑מֶר וָ֝אֶתְמַשֵּׁ֗ל כֶּעָפָ֥ר וָאֵֽפֶר׃
20אֵ֭לֶיךָ וְלֹ֣א תַעֲנֵ֑נִי עָ֝מַ֗דְתִּי וַתִּתְבֹּ֥נֶן בִּֽי׃
21לְאַכְזָ֣ר לִ֑י בְּעֹ֖צֶם יָדְךָ֣ תִשְׂטְמֵֽנִי׃
22אֶל־ ר֭וּחַ תַּרְכִּיבֵ֑נִי וּ֝תְמֹגְגֵ֗נִי [תְּשֻׁוֶּה כ] (תּוּשִׁיָּֽה׃ ק)
23יָ֭דַעְתִּי מָ֣וֶת תְּשִׁיבֵ֑נִי וּבֵ֖ית מוֹעֵ֣ד לְכָל־ חָֽי׃
24לֹא־ בְ֭עִי יִשְׁלַח־ יָ֑ד אִם־ בְּ֝פִיד֗וֹ לָהֶ֥ן שֽׁוּעַ׃
25לֹ֣א בָ֭כִיתִי לִקְשֵׁה־ י֑וֹם עָֽגְמָ֥ה נַ֝פְשִׁ֗י לָאֶבְיֽוֹן׃
26ט֣וֹב קִ֭וִּיתִי וַיָּ֣בֹא רָ֑ע וַֽאֲיַחֲלָ֥ה לְ֝א֗וֹר וַיָּ֥בֹא אֹֽפֶל׃
27רֻתְּח֥וּ וְלֹא־ דָ֗מּוּ קִדְּמֻ֥נִי יְמֵי־ עֹֽנִי׃
28הִ֭לַּכְתִּי בְּלֹ֣א חַמָּ֑ה קַ֖מְתִּי בַקָּהָ֣ל אֲשַׁוֵּֽעַ׃
29הָיִ֣יתִי לְתַנִּ֑ים וְ֝רֵ֗עַ לִבְנ֥וֹת יַעֲנָֽה׃
30שָׁחַ֣ר מֵעָלָ֑י וְעַצְמִי־ חָ֝֗רָה מִנִּי־ חֹֽרֶב׃
