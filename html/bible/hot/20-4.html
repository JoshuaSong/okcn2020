1שִׁמְע֣וּ בָ֭נִים מ֣וּסַר אָ֑ב וְ֝הַקְשִׁ֗יבוּ לָדַ֥עַת בִּינָֽה׃
2לֶ֣קַח ט֭וֹב נָתַ֣תִּי לָכֶ֑ם תּֽ֝וֹרָתִ֗י אַֽל־ תַּעֲזֹֽבוּ׃
3בֵ֭ן הָיִ֣יתִי לְאָבִ֑י רַ֥ךְ וְ֝יָחִ֗יד לִפְנֵ֥י אִמִּֽי׃
4וַיֹּ֥אמֶר לִ֗י יִֽתְמָךְ־ דְּבָרַ֥י לִבֶּ֑ךָ שְׁמֹ֖ר מִצְוֹתַ֣י וֶֽחְיֵֽה׃
5חָ֭כְמָה קְנֵ֣ה בִינָ֑ה אַל־ תִּשְׁכַּ֥ח וְאַל־ תֵּ֝֗ט מֵֽאִמְרֵי־ פִֽי׃
6תַּעַזְבֶ֥הָ וְתִשְׁמְרֶ֑ךָּ אֱהָבֶ֥הָ וְתִצְּרֶֽךָּ׃
7חָ֭כְמָה קְנֵ֣ה חָכְמָ֑ה וּבְכָל־ קִ֝נְיָנְךָ֗ קְנֵ֣ה בִינָֽה׃
8וּֽתְרוֹמְמֶ֑ךָּ תְּ֝כַבֵּ֗דְךָ כִּ֣י תְחַבְּקֶֽנָּה׃
9לְ֭רֹאשְׁךָ לִוְיַת־ חֵ֑ן עֲטֶ֖רֶת תִּפְאֶ֣רֶת תְּמַגְּנֶֽךָּ׃
10בְּ֭נִי וְקַ֣ח אֲמָרָ֑י וְיִרְבּ֥וּ לְ֝ךָ֗ שְׁנ֣וֹת חַיִּֽים׃
11חָ֭כְמָה הֹרֵתִ֑יךָ הִ֝דְרַכְתִּ֗יךָ בְּמַעְגְּלֵי־ יֹֽשֶׁר׃
12לֹא־ יֵצַ֣ר צַעֲדֶ֑ךָ וְאִם־ תָּ֝ר֗וּץ לֹ֣א תִכָּשֵֽׁל׃
13בַּמּוּסָ֣ר אַל־ תֶּ֑רֶף נִ֝צְּרֶ֗הָ כִּי־ הִ֥יא חַיֶּֽיךָ׃
14רְ֭שָׁעִים אַל־ תָּבֹ֑א וְאַל־ תְּ֝אַשֵּׁ֗ר בְּדֶ֣רֶךְ רָעִֽים׃
15אַל־ תַּעֲבָר־ בּ֑וֹ שְׂטֵ֖ה מֵעָלָ֣יו וַעֲבֽוֹר׃
16לֹ֣א יִֽ֭שְׁנוּ אִם־ לֹ֣א יָרֵ֑עוּ וְֽנִגְזְלָ֥ה שְׁ֝נָתָ֗ם אִם־ לֹ֥א [יִכְשֹׁולוּ כ] (יַכְשִֽׁילוּ׃ ק)
17לָ֭חֲמוּ לֶ֣חֶם רֶ֑שַׁע וְיֵ֖ין חֲמָסִ֣ים יִשְׁתּֽוּ׃
18צַ֭דִּיקִים כְּא֣וֹר נֹ֑גַהּ הוֹלֵ֥ךְ וָ֝א֗וֹר עַד־ נְכ֥וֹן הַיּֽוֹם׃
19רְ֭שָׁעִים כָּֽאֲפֵלָ֑ה לֹ֥א יָ֝דְע֗וּ בַּמֶּ֥ה יִכָּשֵֽׁלוּ׃ פ
20לִדְבָרַ֣י הַקְשִׁ֑יבָה לַ֝אֲמָרַ֗י הַט־ אָזְנֶֽךָ׃
21יַלִּ֥יזוּ מֵעֵינֶ֑יךָ שָׁ֝מְרֵ֗ם בְּת֣וֹךְ לְבָבֶֽךָ׃
22חַיִּ֣ים הֵ֭ם לְמֹצְאֵיהֶ֑ם וּֽלְכָל־ בְּשָׂר֥וֹ מַרְפֵּֽא׃
23מִ֭שְׁמָר נְצֹ֣ר לִבֶּ֑ךָ כִּֽי־ מִ֝מֶּ֗נּוּ תּוֹצְא֥וֹת חַיִּֽים׃
24מִ֭מְּךָ עִקְּשׁ֣וּת פֶּ֑ה וּלְז֥וּת שְׂ֝פָתַ֗יִם הַרְחֵ֥ק מִמֶּֽךָּ׃
25לְנֹ֣כַח יַבִּ֑יטוּ וְ֝עַפְעַפֶּ֗יךָ יַיְשִׁ֥רוּ נֶגְדֶּֽךָ׃
26מַעְגַּ֣ל רַגְלֶ֑ךָ וְֽכָל־ דְּרָכֶ֥יךָ יִכֹּֽנוּ׃ אַֽל־ תֵּט־
