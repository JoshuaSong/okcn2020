1הֶן־ כֹּ֭ל רָאֲתָ֣ה עֵינִ֑י שָֽׁמְעָ֥ה אָ֝זְנִ֗י וַתָּ֥בֶן לָֽהּ׃
2יָדַ֣עְתִּי גַם־ אָ֑נִי לֹא־ נֹפֵ֖ל אָנֹכִ֣י מִכֶּֽם׃
3אֲ֭נִי אֶל־ שַׁדַּ֣י אֲדַבֵּ֑ר וְהוֹכֵ֖חַ אֶל־ אֵ֣ל אֶחְפָּֽץ׃
4אַתֶּ֥ם טֹֽפְלֵי־ שָׁ֑קֶר רֹפְאֵ֖י אֱלִ֣ל כֻּלְּכֶֽם׃
5יִ֭תֵּן הַחֲרֵ֣שׁ תַּחֲרִישׁ֑וּן וּתְהִ֖י לָכֶ֣ם לְחָכְמָֽה׃
6נָ֥א תוֹכַחְתִּ֑י וְרִב֖וֹת שְׂפָתַ֣י הַקְשִֽׁיבוּ׃
7תְּדַבְּר֣וּ עַוְלָ֑ה וְ֝ל֗וֹ תְּֽדַבְּר֥וּ רְמִיָּֽה׃
8תִּשָּׂא֑וּן אִם־ לָאֵ֥ל תְּרִיבֽוּן׃
9כִּֽי־ יַחְקֹ֣ר אֶתְכֶ֑ם אִם־ כְּהָתֵ֥ל בֶּ֝אֱנ֗וֹשׁ תְּהָתֵ֥לּוּ בֽוֹ׃
10יוֹכִ֣יחַ אֶתְכֶ֑ם אִם־ בַּ֝סֵּ֗תֶר פָּנִ֥ים תִּשָּׂאֽוּן׃
11שְׂ֭אֵתוֹ תְּבַעֵ֣ת אֶתְכֶ֑ם וּ֝פַחְדּ֗וֹ יִפֹּ֥ל עֲלֵיכֶֽם׃
12מִשְׁלֵי־ אֵ֑פֶר לְגַבֵּי־ חֹ֝֗מֶר גַּבֵּיכֶֽם׃
13מִ֭מֶּנִּי וַאֲדַבְּרָה־ אָ֑נִי וְיַעֲבֹ֖ר עָלַ֣י מָֽה׃
14מָ֤ה ׀ אֶשָּׂ֣א בְשָׂרִ֣י בְשִׁנָּ֑י וְ֝נַפְשִׁ֗י אָשִׂ֥ים בְּכַפִּֽי׃
15יִ֭קְטְלֵנִי [לֹא כ] (לֹ֣ו ק) אֲיַחֵ֑ל אַךְ־ דְּ֝רָכַ֗י אֶל־ פָּנָ֥יו אוֹכִֽיחַ׃ גַּם־ הוּא־
16לִֽישׁוּעָ֑ה כִּי־ לֹ֥א לְ֝פָנָ֗יו חָנֵ֥ף יָבֽוֹא׃
17שָׁ֭מוֹעַ מִלָּתִ֑י וְ֝אַֽחֲוָתִ֗י בְּאָזְנֵיכֶֽם׃
18נָ֭א עָרַ֣כְתִּי מִשְׁפָּ֑ט יָ֝דַ֗עְתִּי כִּֽי־ אֲנִ֥י אֶצְדָּֽק׃
19ה֭וּא יָרִ֣יב עִמָּדִ֑י כִּֽי־ עַתָּ֖ה אַחֲרִ֣ישׁ וְאֶגְוָֽע׃
20שְׁ֭תַּיִם אַל־ תַּ֣עַשׂ עִמָּדִ֑י אָ֥ז מִ֝פָּנֶ֗יךָ לֹ֣א אֶסָּתֵֽר׃
21מֵעָלַ֣י הַרְחַ֑ק וְ֝אֵ֥מָתְךָ֗ אַֽל־ תְּבַעֲתַֽנִּי׃
22וְאָנֹכִ֣י אֶֽעֱנֶ֑ה אֽוֹ־ אֲ֝דַבֵּ֗ר וַהֲשִׁיבֵֽנִי׃
23לִ֭י עֲוֹנ֣וֹת וְחַטָּא֑וֹת פִּֽשְׁעִ֥י וְ֝חַטָּאתִ֗י הֹדִיעֵֽנִי׃
24פָנֶ֥יךָ תַסְתִּ֑יר וְתַחְשְׁבֵ֖נִי לְאוֹיֵ֣ב לָֽךְ׃
25נִדָּ֣ף תַּעֲר֑וֹץ וְאֶת־ קַ֖שׁ יָבֵ֣שׁ תִּרְדֹּֽף׃
26תִכְתֹּ֣ב עָלַ֣י מְרֹר֑וֹת וְ֝תוֹרִישֵׁ֗נִי עֲוֹנ֥וֹת נְעוּרָֽי׃
27בַּסַּ֨ד ׀ רַגְלַ֗י וְתִשְׁמ֥וֹר כָּל־ אָרְחוֹתָ֑י עַל־ שָׁרְשֵׁ֥י רַ֝גְלַ֗י תִּתְחַקֶּֽה׃
