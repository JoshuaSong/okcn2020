1אֹהֵ֣ב מ֭וּסָר אֹ֣הֵֽב דָּ֑עַת וְשֹׂנֵ֖א תוֹכַ֣חַת בָּֽעַר׃
2יָפִ֣יק רָ֭צוֹן מֵיְהוָ֑ה וְאִ֖ישׁ מְזִמּ֣וֹת יַרְשִֽׁיעַ׃
3יִכּ֣וֹן אָדָ֣ם בְּרֶ֑שַׁע וְשֹׁ֥רֶשׁ צַ֝דִּיקִ֗ים בַּל־ יִמּֽוֹט׃
4חַ֭יִל עֲטֶ֣רֶת בַּעְלָ֑הּ וּכְרָקָ֖ב בְּעַצְמוֹתָ֣יו מְבִישָֽׁה׃
5צַדִּיקִ֣ים מִשְׁפָּ֑ט תַּחְבֻּל֖וֹת רְשָׁעִ֣ים מִרְמָֽה׃
6רְשָׁעִ֣ים אֱרָב־ דָּ֑ם וּפִ֥י יְ֝שָׁרִ֗ים יַצִּילֵֽם׃
7רְשָׁעִ֣ים וְאֵינָ֑ם וּבֵ֖ית צַדִּיקִ֣ים יַעֲמֹֽד׃
8שִׂ֭כְלוֹ יְהֻלַּל־ אִ֑ישׁ וְנַעֲוֵה־ לֵ֝֗ב יִהְיֶ֥ה לָבֽוּז׃
9נִ֭קְלֶה וְעֶ֣בֶד ל֑וֹ מִ֝מְּתַכַּבֵּ֗ד וַחֲסַר־ לָֽחֶם׃
10צַ֭דִּיק נֶ֣פֶשׁ בְּהֶמְתּ֑וֹ וְֽרַחֲמֵ֥י רְ֝שָׁעִ֗ים אַכְזָרִֽי׃
11אַ֭דְמָתוֹ יִֽשְׂבַּֽע־ לָ֑חֶם וּמְרַדֵּ֖ף רֵיקִ֣ים חֲסַר־ לֵֽב׃
12רָ֭שָׁע מְצ֣וֹד רָעִ֑ים וְשֹׁ֖רֶשׁ צַדִּיקִ֣ים יִתֵּֽן׃
13שְׂ֭פָתַיִם מוֹקֵ֣שׁ רָ֑ע וַיֵּצֵ֖א מִצָּרָ֣ה צַדִּֽיק׃
14פִי־ אִ֭ישׁ יִשְׂבַּע־ ט֑וֹב וּגְמ֥וּל יְדֵי־ אָ֝דָ֗ם [יָשׁוּב כ] (יָשִׁ֥יב ק) לֽוֹ׃
15אֱ֭וִיל יָשָׁ֣ר בְּעֵינָ֑יו וְשֹׁמֵ֖עַ לְעֵצָ֣ה חָכָֽם׃
16בַּ֭יּוֹם יִוָּדַ֣ע כַּעְס֑וֹ וְכֹסֶ֖ה קָל֣וֹן עָרֽוּם׃
17אֱ֭מוּנָה יַגִּ֣יד צֶ֑דֶק וְעֵ֖ד שְׁקָרִ֣ים מִרְמָֽה׃
18בּ֭וֹטֶה כְּמַדְקְר֣וֹת חָ֑רֶב וּלְשׁ֖וֹן חֲכָמִ֣ים מַרְפֵּֽא׃
19אֱ֭מֶת תִּכּ֣וֹן לָעַ֑ד וְעַד־ אַ֝רְגִּ֗יעָה לְשׁ֣וֹן שָֽׁקֶר׃
20בְּלֶב־ חֹ֣רְשֵׁי רָ֑ע וּֽלְיֹעֲצֵ֖י שָׁל֣וֹם שִׂמְחָֽה׃
21יְאֻנֶּ֣ה לַצַּדִּ֣יק כָּל־ אָ֑וֶן וּ֝רְשָׁעִ֗ים מָ֣לְאוּ רָֽע׃
22יְ֭הוָה שִׂפְתֵי־ שָׁ֑קֶר וְעֹשֵׂ֖י אֱמוּנָ֣ה רְצוֹנֽוֹ׃
23עָ֭רוּם כֹּ֣סֶה דָּ֑עַת וְלֵ֥ב כְּ֝סִילִ֗ים יִקְרָ֥א אִוֶּֽלֶת׃
24חָרוּצִ֥ים תִּמְשׁ֑וֹל וּ֝רְמִיָּ֗ה תִּהְיֶ֥ה לָמַֽס׃
25בְלֶב־ אִ֣ישׁ יַשְׁחֶ֑נָּה וְדָבָ֖ר ט֣וֹב יְשַׂמְּחֶֽנָּה׃
26מֵרֵעֵ֣הוּ צַדִּ֑יק וְדֶ֖רֶךְ רְשָׁעִ֣ים תַּתְעֵֽם׃
27יַחֲרֹ֣ךְ רְמִיָּ֣ה צֵיד֑וֹ וְהוֹן־ אָדָ֖ם יָקָ֣ר חָרֽוּץ׃
