1וּבִ֨נְיָמִ֔ן הוֹלִ֖יד אֶת־ בֶּ֣לַע בְּכֹר֑וֹ אַשְׁבֵּל֙ הַשֵּׁנִ֔י וְאַחְרַ֖ח הַשְּׁלִישִֽׁי׃
2הָֽרְבִיעִ֔י וְרָפָ֖א הַחֲמִישִֽׁי׃ ס
3בָנִ֖ים לְבָ֑לַע אַדָּ֥ר וְגֵרָ֖א וַאֲבִיהֽוּד׃
4וְנַעֲמָ֖ן וַאֲחֽוֹחַ׃
5וּשְׁפוּפָ֖ן וְחוּרָֽם׃
6בְּנֵ֣י אֵח֑וּד אֵ֣לֶּה הֵ֞ם רָאשֵׁ֤י אָבוֹת֙ לְי֣וֹשְׁבֵי גֶ֔בַע וַיַּגְל֖וּם אֶל־ מָנָֽחַת׃
7וַאֲחִיָּ֛ה וְגֵרָ֖א ה֣וּא הֶגְלָ֑ם וְהוֹלִ֥יד אֶת־ עֻזָּ֖א וְאֶת־ אֲחִיחֻֽד׃
8הוֹלִיד֙ בִּשְׂדֵ֣ה מוֹאָ֔ב מִן־ שִׁלְח֖וֹ אֹתָ֑ם חוּשִׁ֥ים וְאֶֽת־ בַּעֲרָ֖א נָשָֽׁיו׃
9מִן־ חֹ֣דֶשׁ אִשְׁתּ֑וֹ אֶת־ יוֹבָב֙ וְאֶת־ צִבְיָ֔א וְאֶת־ מֵישָׁ֖א וְאֶת־ מַלְכָּֽם׃
10יְע֥וּץ וְאֶת־ שָֽׂכְיָ֖ה וְאֶת־ מִרְמָ֑ה אֵ֥לֶּה בָנָ֖יו רָאשֵׁ֥י אָבֽוֹת׃
11הוֹלִ֥יד אֶת־ אֲבִיט֖וּב וְאֶת־ אֶלְפָּֽעַל׃
12אֶלְפַּ֔עַל עֵ֥בֶר וּמִשְׁעָ֖ם וָשָׁ֑מֶד ה֚וּא בָּנָ֣ה אֶת־ אוֹנ֔וֹ וְאֶת־ לֹ֖ד וּבְנֹתֶֽיהָ׃
13וָשֶׁ֔מַע הֵ֚מָּה רָאשֵׁ֣י הָאָב֔וֹת לְיוֹשְׁבֵ֖י אַיָּל֑וֹן הֵ֥מָּה הִבְרִ֖יחוּ אֶת־ י֥וֹשְׁבֵי גַֽת׃
14שָׁשָׁ֖ק וִירֵמֽוֹת׃
15וַעֲרָ֖ד וָעָֽדֶר׃
16וְיִשְׁפָּ֛ה וְיוֹחָ֖א בְּנֵ֥י בְרִיעָֽה׃
17וּמְשֻׁלָּ֖ם וְחִזְקִ֥י וָחָֽבֶר׃
18וְיִזְלִיאָ֛ה וְיוֹבָ֖ב בְּנֵ֥י אֶלְפָּֽעַל׃
19וְזִכְרִ֖י וְזַבְדִּֽי׃
20וְצִלְּתַ֖י וֶאֱלִיאֵֽל׃
21וּבְרָאיָ֛ה וְשִׁמְרָ֖ת בְּנֵ֥י שִׁמְעִֽי׃
22וָעֵ֖בֶר וֶאֱלִיאֵֽל׃
23וְזִכְרִ֖י וְחָנָֽן׃
24וְעֵילָ֖ם וְעַנְתֹתִיָּֽה׃
25[וּפְנִיאֵל כ] (וּפְנוּאֵ֖ל ק) בְּנֵ֥י שָׁשָֽׁק׃
26וּשְׁחַרְיָ֖ה וַעֲתַלְיָֽה׃
27וְאֵלִיָּ֛ה וְזִכְרִ֖י בְּנֵ֥י יְרֹחָֽם׃
28רָאשֵׁ֥י אָב֛וֹת לְתֹלְדוֹתָ֖ם רָאשִׁ֑ים אֵ֖לֶּה יָשְׁב֥וּ בִירוּשָׁלִָֽם׃ ס
29יָשְׁב֖וּ אֲבִ֣י גִבְע֑וֹן וְשֵׁ֥ם אִשְׁתּ֖וֹ מַעֲכָֽה׃
30הַבְּכ֖וֹר עַבְדּ֑וֹן וְצ֥וּר וְקִ֖ישׁ וּבַ֥עַל וְנָדָֽב׃
31וְאַחְי֖וֹ וָזָֽכֶר׃
32הוֹלִ֣יד אֶת־ שִׁמְאָ֑ה וְאַף־ הֵ֗מָּה נֶ֧גֶד אֲחֵיהֶ֛ם יָשְׁב֥וּ בִירוּשָׁלִַ֖ם עִם־ אֲחֵיהֶֽם׃ ס
33הוֹלִ֣יד אֶת־ קִ֔ישׁ וְקִ֖ישׁ הוֹלִ֣יד אֶת־ שָׁא֑וּל וְשָׁא֗וּל הוֹלִ֤יד אֶת־ יְהֽוֹנָתָן֙ וְאֶת־ מַלְכִּי־ שׁ֔וּעַ וְאֶת־ אֲבִֽינָדָ֖ב וְאֶת־ אֶשְׁבָּֽעַל׃
34יְהוֹנָתָ֖ן מְרִ֣יב בָּ֑עַל וּמְרִ֥יב בַּ֖עַל הוֹלִ֥יד אֶת־ מִיכָֽה׃ ס
35מִיכָ֑ה פִּית֥וֹן וָמֶ֖לֶךְ וְתַאְרֵ֥עַ וְאָחָֽז׃
36הוֹלִ֣יד אֶת־ יְהוֹעַדָּ֔ה וִיהֽוֹעַדָּ֗ה הוֹלִ֛יד אֶת־ עָלֶ֥מֶת וְאֶת־ עַזְמָ֖וֶת וְאֶת־ זִמְרִ֑י וְזִמְרִ֖י הוֹלִ֥יד אֶת־ מוֹצָֽא׃
37הוֹלִ֣יד אֶת־ בִּנְעָ֑א רָפָ֥ה בְנ֛וֹ אֶלְעָשָׂ֥ה בְנ֖וֹ אָצֵ֥ל בְּנֽוֹ׃
38שִׁשָּׁ֣ה בָנִים֒ וְאֵ֣לֶּה שְׁמוֹתָ֗ם עַזְרִיקָ֥ם ׀ בֹּ֙כְרוּ֙ וְיִשְׁמָעֵ֣אל וּשְׁעַרְיָ֔ה וְעֹבַדְיָ֖ה וְחָנָ֑ן כָּל־ אֵ֖לֶּה בְּנֵ֥י אָצַֽל׃
39עֵ֣שֶׁק אָחִ֑יו אוּלָ֣ם בְּכֹר֔וֹ יְעוּשׁ֙ הַשֵּׁנִ֔י וֶֽאֱלִיפֶ֖לֶט הַשְּׁלִשִֽׁי׃
