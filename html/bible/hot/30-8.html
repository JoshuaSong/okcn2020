1כֹּ֥ה הִרְאַ֖נִי אֲדֹנָ֣י יְהוִ֑ה וְהִנֵּ֖ה כְּל֥וּב קָֽיִץ׃
2מָֽה־ אַתָּ֤ה רֹאֶה֙ עָמ֔וֹס וָאֹמַ֖ר כְּל֣וּב קָ֑יִץ וַיֹּ֨אמֶר יְהוָ֜ה אֵלַ֗י בָּ֤א הַקֵּץ֙ אֶל־ עַמִּ֣י יִשְׂרָאֵ֔ל לֹא־ אוֹסִ֥יף ע֖וֹד עֲב֥וֹר לֽוֹ׃
3שִׁיר֤וֹת הֵיכָל֙ בַּיּ֣וֹם הַה֔וּא נְאֻ֖ם אֲדֹנָ֣י יְהוִ֑ה רַ֣ב הַפֶּ֔גֶר בְּכָל־ מָק֖וֹם הִשְׁלִ֥יךְ הָֽס׃ פ
4זֹ֕את הַשֹּׁאֲפִ֖ים אֶבְי֑וֹן וְלַשְׁבִּ֖ית [עַנְוֵי־ כ] (עֲנִיֵּי־ אָֽרֶץ׃ ק)
5מָתַ֞י יַעֲבֹ֤ר הַחֹ֙דֶשׁ֙ וְנַשְׁבִּ֣ירָה שֶּׁ֔בֶר וְהַשַּׁבָּ֖ת וְנִפְתְּחָה־ בָּ֑ר לְהַקְטִ֤ין אֵיפָה֙ וּלְהַגְדִּ֣יל שֶׁ֔קֶל וּלְעַוֵּ֖ת מֹאזְנֵ֥י מִרְמָֽה׃
6בַּכֶּ֙סֶף֙ דַּלִּ֔ים וְאֶבְי֖וֹן בַּעֲב֣וּר נַעֲלָ֑יִם וּמַפַּ֥ל בַּ֖ר נַשְׁבִּֽיר׃
7יְהוָ֖ה בִּגְא֣וֹן יַעֲקֹ֑ב אִם־ אֶשְׁכַּ֥ח לָנֶ֖צַח כָּל־ מַעֲשֵׂיהֶֽם׃
8זֹאת֙ לֹֽא־ תִרְגַּ֣ז הָאָ֔רֶץ וְאָבַ֖ל כָּל־ יוֹשֵׁ֣ב בָּ֑הּ וְעָלְתָ֤ה כָאֹר֙ כֻּלָּ֔הּ וְנִגְרְשָׁ֥ה [וְנִשְׁקָה כ] (וְנִשְׁקְעָ֖ה ק) כִּיא֥וֹר מִצְרָֽיִם׃ ס
9בַּיּ֣וֹם הַה֗וּא נְאֻם֙ אֲדֹנָ֣י יְהוִ֔ה וְהֵבֵאתִ֥י הַשֶּׁ֖מֶשׁ בַּֽצָּהֳרָ֑יִם וְהַחֲשַׁכְתִּ֥י לָאָ֖רֶץ בְּי֥וֹם אֽוֹר׃
10חַגֵּיכֶ֜ם לְאֵ֗בֶל וְכָל־ שִֽׁירֵיכֶם֙ לְקִינָ֔ה וְהַעֲלֵיתִ֤י עַל־ כָּל־ מָתְנַ֙יִם֙ שָׂ֔ק וְעַל־ כָּל־ רֹ֖אשׁ קָרְחָ֑ה וְשַׂמְתִּ֙יהָ֙ כְּאֵ֣בֶל יָחִ֔יד וְאַחֲרִיתָ֖הּ כְּי֥וֹם מָֽר׃
11יָמִ֣ים בָּאִ֗ים נְאֻם֙ אֲדֹנָ֣י יְהוִ֔ה וְהִשְׁלַחְתִּ֥י רָעָ֖ב בָּאָ֑רֶץ לֹֽא־ רָעָ֤ב לַלֶּ֙חֶם֙ וְלֹֽא־ צָמָ֣א לַמַּ֔יִם כִּ֣י אִם־ לִשְׁמֹ֔עַ אֵ֖ת דִּבְרֵ֥י יְהוָֽה׃
12מִיָּ֣ם עַד־ יָ֔ם וּמִצָּפ֖וֹן וְעַד־ מִזְרָ֑ח יְשֽׁוֹטְט֛וּ לְבַקֵּ֥שׁ אֶת־ דְּבַר־ יְהוָ֖ה וְלֹ֥א יִמְצָֽאוּ׃
13הַה֜וּא תִּ֠תְעַלַּפְנָה הַבְּתוּלֹ֧ת הַיָּפ֛וֹת וְהַבַּחוּרִ֖ים בַּצָּמָֽא׃
