1בָּעֵ֣ת הַהִ֡וא שָׁלַ֡ח מְרֹדַ֣ךְ בַּ֠לְאֲדָן בֶּֽן־ בַּלְאֲדָ֧ן מֶֽלֶךְ־ בָּבֶ֛ל סְפָרִ֥ים וּמִנְחָ֖ה אֶל־ חִזְקִיָּ֑הוּ וַיִּשְׁמַ֕ע כִּ֥י חָלָ֖ה וַֽיֶּחֱזָֽק׃
2עֲלֵיהֶם֮ חִזְקִיָּהוּ֒ וַיַּרְאֵ֣ם אֶת־ בֵּ֣ית [נְכֹתָה כ] (נְכֹתֹ֡ו ק) אֶת־ הַכֶּסֶף֩ וְאֶת־ הַזָּהָ֨ב וְאֶת־ הַבְּשָׂמִ֜ים וְאֵ֣ת ׀ הַשֶּׁ֣מֶן הַטּ֗וֹב וְאֵת֙ כָּל־ בֵּ֣ית כֵּלָ֔יו וְאֵ֛ת כָּל־ אֲשֶׁ֥ר נִמְצָ֖א בְּאֹֽצְרֹתָ֑יו לֹֽא־ הָיָ֣ה דָבָ֗ר אֲ֠שֶׁר לֹֽא־ הֶרְאָ֧ם חִזְקִיָּ֛הוּ בְּבֵית֖וֹ וּבְכָל־ מֶמְשַׁלְתּֽוֹ׃
3יְשַׁעְיָ֣הוּ הַנָּבִ֔יא אֶל־ הַמֶּ֖לֶךְ חִזְקִיָּ֑הוּ וַיֹּ֨אמֶר אֵלָ֜יו מָ֥ה אָמְר֣וּ ׀ הָאֲנָשִׁ֣ים הָאֵ֗לֶּה וּמֵאַ֙יִן֙ יָבֹ֣אוּ אֵלֶ֔יךָ וַיֹּ֙אמֶר֙ חִזְקִיָּ֔הוּ מֵאֶ֧רֶץ רְחוֹקָ֛ה בָּ֥אוּ אֵלַ֖י מִבָּבֶֽל׃
4מָ֥ה רָא֖וּ בְּבֵיתֶ֑ךָ וַיֹּ֣אמֶר חִזְקִיָּ֗הוּ אֵ֣ת כָּל־ אֲשֶׁ֤ר בְּבֵיתִי֙ רָא֔וּ לֹֽא־ הָיָ֥ה דָבָ֛ר אֲשֶׁ֥ר לֹֽא־ הִרְאִיתִ֖ים בְּאוֹצְרֹתָֽי׃
5יְשַׁעְיָ֖הוּ אֶל־ חִזְקִיָּ֑הוּ שְׁמַ֖ע דְּבַר־ יְהוָ֥ה צְבָאֽוֹת׃
6יָמִ֣ים בָּאִים֒ וְנִשָּׂ֣א ׀ כָּל־ אֲשֶׁ֣ר בְּבֵיתֶ֗ךָ וַאֲשֶׁ֨ר אָצְר֧וּ אֲבֹתֶ֛יךָ עַד־ הַיּ֥וֹם הַזֶּ֖ה בָּבֶ֑ל לֹֽא־ יִוָּתֵ֥ר דָּבָ֖ר אָמַ֥ר יְהוָֽה׃
7אֲשֶׁ֨ר יֵצְא֧וּ מִמְּךָ֛ אֲשֶׁ֥ר תּוֹלִ֖יד יִקָּ֑חוּ וְהָיוּ֙ סָרִיסִ֔ים בְּהֵיכַ֖ל מֶ֥לֶךְ בָּבֶֽל׃
