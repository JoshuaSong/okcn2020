1לֹ֤א לָ֥נוּ יְהוָ֗ה לֹ֫א לָ֥נוּ כִּֽי־ לְ֭שִׁמְךָ תֵּ֣ן כָּב֑וֹד עַל־ חַ֝סְדְּךָ֗ עַל־ אֲמִתֶּֽךָ׃
2יֹאמְר֣וּ הַגּוֹיִ֑ם אַיֵּה־ נָ֝֗א אֱלֹהֵיהֶֽם׃
3בַשָּׁמָ֑יִם כֹּ֭ל אֲשֶׁר־ חָפֵ֣ץ עָשָֽׂה׃
4כֶּ֣סֶף וְזָהָ֑ב מַ֝עֲשֵׂ֗ה יְדֵ֣י אָדָֽם׃
5לָ֭הֶם וְלֹ֣א יְדַבֵּ֑רוּ עֵינַ֥יִם לָ֝הֶ֗ם וְלֹ֣א יִרְאֽוּ׃
6לָ֭הֶם וְלֹ֣א יִשְׁמָ֑עוּ אַ֥ף לָ֝הֶ֗ם וְלֹ֣א יְרִיחֽוּן׃
7וְלֹ֬א יְמִישׁ֗וּן רַ֭גְלֵיהֶם וְלֹ֣א יְהַלֵּ֑כוּ לֹֽא־ יֶ֝הְגּ֗וּ בִּגְרוֹנָֽם׃
8יִהְי֣וּ עֹשֵׂיהֶ֑ם כֹּ֭ל אֲשֶׁר־ בֹּטֵ֣חַ בָּהֶֽם׃
9בְּטַ֣ח בַּיהוָ֑ה עֶזְרָ֖ם וּמָגִנָּ֣ם הֽוּא׃
10אַ֭הֲרֹן בִּטְח֣וּ בַיהוָ֑ה עֶזְרָ֖ם וּמָגִנָּ֣ם הֽוּא׃
11יְ֭הוָה בִּטְח֣וּ בַיהוָ֑ה עֶזְרָ֖ם וּמָגִנָּ֣ם הֽוּא׃
12זְכָרָ֪נוּ יְבָ֫רֵ֥ךְ יְ֭בָרֵךְ אֶת־ בֵּ֣ית יִשְׂרָאֵ֑ל יְ֝בָרֵ֗ךְ אֶת־ בֵּ֥ית אַהֲרֹֽן׃
13יִרְאֵ֣י יְהוָ֑ה הַ֝קְּטַנִּ֗ים עִם־ הַגְּדֹלִֽים׃
14יְהוָ֣ה עֲלֵיכֶ֑ם עֲ֝לֵיכֶ֗ם וְעַל־ בְּנֵיכֶֽם׃
15אַ֭תֶּם לַיהוָ֑ה עֹ֝שֵׂ֗ה שָׁמַ֥יִם וָאָֽרֶץ׃
16שָׁ֭מַיִם לַיהוָ֑ה וְ֝הָאָ֗רֶץ נָתַ֥ן לִבְנֵי־ אָדָֽם׃
17הַ֭מֵּתִים יְהַֽלְלוּ־ יָ֑הּ וְ֝לֹ֗א כָּל־ יֹרְדֵ֥י דוּמָֽה׃
