1מַֽעֲנֶה־ רַּ֭ךְ יָשִׁ֣יב חֵמָ֑ה וּדְבַר־ עֶ֝֗צֶב יַעֲלֶה־ אָֽף׃
2חֲ֭כָמִים תֵּיטִ֣יב דָּ֑עַת וּפִ֥י כְ֝סִילִ֗ים יַבִּ֥יעַ אִוֶּֽלֶת׃
3מָ֭קוֹם עֵינֵ֣י יְהוָ֑ה צֹ֝פ֗וֹת רָעִ֥ים וטוֹבִֽים׃
4לָ֭שׁוֹן עֵ֣ץ חַיִּ֑ים וְסֶ֥לֶף בָּ֝֗הּ שֶׁ֣בֶר בְּרֽוּחַ׃
5יִ֭נְאַץ מוּסַ֣ר אָבִ֑יו וְשֹׁמֵ֖ר תּוֹכַ֣חַת יַעְרִֽם׃
6צַ֭דִּיק חֹ֣סֶן רָ֑ב וּבִתְבוּאַ֖ת רָשָׁ֣ע נֶעְכָּֽרֶת׃
7חֲ֭כָמִים יְזָ֣רוּ דָ֑עַת וְלֵ֖ב כְּסִילִ֣ים לֹא־ כֵֽן׃
8רְ֭שָׁעִים תּוֹעֲבַ֣ת יְהוָ֑ה וּתְפִלַּ֖ת יְשָׁרִ֣ים רְצוֹנֽוֹ׃
9יְ֭הוָה דֶּ֣רֶךְ רָשָׁ֑ע וּמְרַדֵּ֖ף צְדָקָ֣ה יֶאֱהָֽב׃
10רָ֭ע לְעֹזֵ֣ב אֹ֑רַח שׂוֹנֵ֖א תוֹכַ֣חַת יָמֽוּת׃
11וַ֭אֲבַדּוֹן נֶ֣גֶד יְהוָ֑ה אַ֝֗ף כִּֽי־ לִבּ֥וֹת בְּֽנֵי־ אָדָֽם׃
12יֶאֱהַב־ לֵ֭ץ הוֹכֵ֣חַֽ ל֑וֹ אֶל־ חֲ֝כָמִ֗ים לֹ֣א יֵלֵֽךְ׃
13שָׂ֭מֵחַ יֵיטִ֣ב פָּנִ֑ים וּבְעַצְּבַת־ לֵ֝ב ר֣וּחַ נְכֵאָֽה׃
14נָ֭בוֹן יְבַקֶּשׁ־ דָּ֑עַת [וּפְנֵי כ] (וּפִ֥י ק) כְ֝סִילִ֗ים יִרְעֶ֥ה אִוֶּֽלֶת׃
15יְמֵ֣י עָנִ֣י רָעִ֑ים וְטֽוֹב־ לֵ֝֗ב מִשְׁתֶּ֥ה תָמִֽיד׃
16מְ֭עַט בְּיִרְאַ֣ת יְהוָ֑ה מֵאוֹצָ֥ר רָ֝֗ב וּמְה֥וּמָה בֽוֹ׃
17אֲרֻחַ֣ת יָ֭רָק וְאַהֲבָה־ שָׁ֑ם מִשּׁ֥וֹר אָ֝ב֗וּס וְשִׂנְאָה־ בֽוֹ׃
18חֵ֭מָה יְגָרֶ֣ה מָד֑וֹן וְאֶ֥רֶך אַ֝פַּ֗יִם יַשְׁקִ֥יט רִֽיב׃
19עָ֭צֵל כִּמְשֻׂ֣כַת חָ֑דֶק וְאֹ֖רַח יְשָׁרִ֣ים סְלֻלָֽה׃
20חָ֭כָם יְשַׂמַּח־ אָ֑ב וּכְסִ֥יל אָ֝דָ֗ם בּוֹזֶ֥ה אִמּֽוֹ׃
21שִׂמְחָ֣ה לַחֲסַר־ לֵ֑ב וְאִ֥ישׁ תְּ֝בוּנָ֗ה יְיַשֶּׁר־ לָֽכֶת׃
22מַ֭חֲשָׁבוֹת בְּאֵ֣ין ס֑וֹד וּבְרֹ֖ב יוֹעֲצִ֣ים תָּקֽוּם׃
23לָ֭אִישׁ בְּמַעֲנֵה־ פִ֑יו וְדָבָ֖ר בְּעִתּ֣וֹ מַה־ טּֽוֹב׃
24חַ֭יִּים לְמַ֣עְלָה לְמַשְׂכִּ֑יל לְמַ֥עַן ס֝֗וּר מִשְּׁא֥וֹל מָֽטָּה׃
25גֵּ֭אִים יִסַּ֥ח ׀ יְהוָ֑ה וְ֝יַצֵּ֗ב גְּב֣וּל אַלְמָנָֽה׃
26יְ֭הוָה מַחְשְׁב֣וֹת רָ֑ע וּ֝טְהֹרִ֗ים אִמְרֵי־ נֹֽעַם׃
27בֵּ֭יתוֹ בּוֹצֵ֣עַ בָּ֑צַע וְשׂוֹנֵ֖א מַתָּנֹ֣ת יִחְיֶֽה׃
28צַ֭דִּיק יֶהְגֶּ֣ה לַעֲנ֑וֹת וּפִ֥י רְ֝שָׁעִ֗ים יַבִּ֥יעַ רָעֽוֹת׃
29יְ֭הוָה מֵרְשָׁעִ֑ים וּתְפִלַּ֖ת צַדִּיקִ֣ים יִשְׁמָֽע׃
30עֵ֭ינַיִם יְשַׂמַּֽח־ לֵ֑ב שְׁמוּעָ֥ה ט֝וֹבָ֗ה תְּדַשֶּׁן־ עָֽצֶם׃
31שֹׁ֭מַעַת תּוֹכַ֣חַת חַיִּ֑ים בְּקֶ֖רֶב חֲכָמִ֣ים תָּלִֽין׃
32מ֭וּסָר מוֹאֵ֣ס נַפְשׁ֑וֹ וְשׁוֹמֵ֥עַ תּ֝וֹכַ֗חַת ק֣וֹנֶה לֵּֽב׃
