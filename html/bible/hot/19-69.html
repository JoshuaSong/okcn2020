1לַמְנַצֵּ֬חַ עַֽל־ שׁוֹשַׁנִּ֬ים לְדָוִֽד׃ הוֹשִׁיעֵ֥נִי אֱלֹהִ֑ים כִּ֤י בָ֖אוּ מַ֣יִם עַד־ נָֽפֶשׁ׃
2בִּיוֵ֣ן מְ֭צוּלָה וְאֵ֣ין מָעֳמָ֑ד בָּ֥אתִי בְמַעֲמַקֵּי־ מַ֝֗יִם וְשִׁבֹּ֥לֶת שְׁטָפָֽתְנִי׃
3בְקָרְאִי֮ נִחַ֪ר גְּר֫וֹנִ֥י כָּל֥וּ עֵינַ֑י מְ֝יַחֵ֗ל לֵאלֹהָֽי׃
4מִשַּׂעֲר֣וֹת רֹאשִׁי֮ שֹׂנְאַ֪י חִ֫נָּ֥ם עָצְמ֣וּ מַ֭צְמִיתַי אֹיְבַ֣י שֶׁ֑קֶר אֲשֶׁ֥ר לֹא־ גָ֝זַ֗לְתִּי אָ֣ז אָשִֽׁיב׃
5אַתָּ֣ה יָ֭דַעְתָּ לְאִוַּלְתִּ֑י וְ֝אַשְׁמוֹתַ֗י מִמְּךָ֥ לֹא־ נִכְחָֽדוּ׃
6יֵ֘בֹ֤שׁוּ בִ֨י ׀ קֹוֶיךָ֮ אֲדֹנָ֥י יְהוִ֗ה צְבָ֫א֥וֹת אַל־ יִכָּ֣לְמוּ בִ֣י מְבַקְשֶׁ֑יךָ אֱ֝לֹהֵ֗י יִשְׂרָאֵֽל׃
7עָ֭לֶיךָ נָשָׂ֣אתִי חֶרְפָּ֑ה כִּסְּתָ֖ה כְלִמָּ֣ה פָנָֽי׃
8הָיִ֣יתִי לְאֶחָ֑י וְ֝נָכְרִ֗י לִבְנֵ֥י אִמִּֽי׃
9קִנְאַ֣ת בֵּיתְךָ֣ אֲכָלָ֑תְנִי וְחֶרְפּ֥וֹת ח֝וֹרְפֶ֗יךָ נָפְל֥וּ עָלָֽי׃
10בַצּ֣וֹם נַפְשִׁ֑י וַתְּהִ֖י לַחֲרָפ֣וֹת לִֽי׃
11לְבוּשִׁ֣י שָׂ֑ק וָאֱהִ֖י לָהֶ֣ם לְמָשָֽׁל׃
12בִ֭י יֹ֣שְׁבֵי שָׁ֑עַר וּ֝נְגִינ֗וֹת שׁוֹתֵ֥י שֵׁכָֽר׃
13תְפִלָּתִֽי־ לְךָ֨ ׀ יְהוָ֡ה עֵ֤ת רָצ֗וֹן אֱלֹהִ֥ים בְּרָב־ חַסְדֶּ֑ךָ עֲ֝נֵ֗נִי בֶּאֱמֶ֥ת יִשְׁעֶֽךָ׃
14מִ֭טִּיט וְאַל־ אֶטְבָּ֑עָה אִנָּצְלָ֥ה מִ֝שֹּֽׂנְאַ֗י וּמִמַּֽעֲמַקֵּי־ מָֽיִם׃
15תִּשְׁטְפֵ֤נִי ׀ שִׁבֹּ֣לֶת מַ֭יִם וְאַל־ תִּבְלָעֵ֣נִי מְצוּלָ֑ה וְאַל־ תֶּאְטַר־ עָלַ֖י בְּאֵ֣ר פִּֽיהָ׃
16יְ֭הוָה כִּי־ ט֣וֹב חַסְדֶּ֑ךָ כְּרֹ֥ב רַ֝חֲמֶ֗יךָ פְּנֵ֣ה אֵלָֽי׃
17תַּסְתֵּ֣ר פָּ֭נֶיךָ מֵֽעַבְדֶּ֑ךָ כִּֽי־ צַר־ לִ֝֗י מַהֵ֥ר עֲנֵֽנִי׃
18אֶל־ נַפְשִׁ֣י גְאָלָ֑הּ לְמַ֖עַן אֹיְבַ֣י פְּדֵֽנִי׃
19יָדַ֗עְתָּ חֶרְפָּתִ֣י וּ֭בָשְׁתִּי וּכְלִמָּתִ֑י נֶ֝גְדְּךָ֗ כָּל־ צוֹרְרָֽי׃
20שָֽׁבְרָ֥ה לִבִּ֗י וָֽאָ֫נ֥וּשָׁה וָאֲקַוֶּ֣ה לָנ֣וּד וָאַ֑יִן וְ֝לַמְנַחֲמִ֗ים וְלֹ֣א מָצָֽאתִי׃
21בְּבָרוּתִ֣י רֹ֑אשׁ וְ֝לִצְמָאִ֗י יַשְׁק֥וּנִי חֹֽמֶץ׃
22שֻׁלְחָנָ֣ם לִפְנֵיהֶ֣ם לְפָ֑ח וְלִשְׁלוֹמִ֥ים לְמוֹקֵֽשׁ׃
23עֵ֭ינֵיהֶם מֵרְא֑וֹת וּ֝מָתְנֵ֗יהֶם תָּמִ֥יד הַמְעַֽד׃
24עֲלֵיהֶ֥ם זַעְמֶ֑ךָ וַחֲר֥וֹן אַ֝פְּךָ֗ יַשִּׂיגֵֽם׃
25טִֽירָתָ֥ם נְשַׁמָּ֑ה בְּ֝אָהֳלֵיהֶ֗ם אַל־ יְהִ֥י יֹשֵֽׁב׃
26אַתָּ֣ה אֲשֶׁר־ הִכִּ֣יתָ רָדָ֑פוּ וְאֶל־ מַכְא֖וֹב חֲלָלֶ֣יךָ יְסַפֵּֽרוּ׃
27עָ֭וֹן עַל־ עֲוֹנָ֑ם וְאַל־ יָ֝בֹ֗אוּ בְּצִדְקָתֶֽךָ׃
28מִסֵּ֣פֶר חַיִּ֑ים וְעִ֥ם צַ֝דִּיקִ֗ים אַל־ יִכָּתֵֽבוּ׃
29עָנִ֣י וְכוֹאֵ֑ב יְשׁוּעָתְךָ֖ אֱלֹהִ֣ים תְּשַׂגְּבֵֽנִי׃
30שֵׁם־ אֱלֹהִ֣ים בְּשִׁ֑יר וַאֲגַדְּלֶ֥נּוּ בְתוֹדָֽה׃
31לַֽ֭יהוָה מִשּׁ֥וֹר פָּ֗ר מַקְרִ֥ן מַפְרִֽיס׃
32עֲנָוִ֣ים יִשְׂמָ֑חוּ דֹּרְשֵׁ֥י אֱ֝לֹהִ֗ים וִיחִ֥י לְבַבְכֶֽם׃
33שֹׁמֵ֣עַ אֶל־ אֶבְיוֹנִ֣ים יְהוָ֑ה וְאֶת־ אֲ֝סִירָ֗יו לֹ֣א בָזָֽה׃
34שָׁמַ֣יִם וָאָ֑רֶץ יַ֝מִּ֗ים וְֽכָל־ רֹמֵ֥שׂ בָּֽם׃
35אֱלֹהִ֨ים ׀ י֘וֹשִׁ֤יעַ צִיּ֗וֹן וְ֭יִבְנֶה עָרֵ֣י יְהוּדָ֑ה וְיָ֥שְׁבוּ שָׁ֝֗ם וִירֵשֽׁוּהָ׃
