1וְאַתָּ֣ה בֶן־ אָדָ֔ם הִנָּבֵ֖א אֶל־ הָרֵ֣י יִשְׂרָאֵ֑ל וְאָ֣מַרְתָּ֔ הָרֵי֙ יִשְׂרָאֵ֔ל שִׁמְע֖וּ דְּבַר־ יְהוָֽה׃
2אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה יַ֣עַן אָמַ֧ר הָאוֹיֵ֛ב עֲלֵיכֶ֖ם הֶאָ֑ח וּבָמ֣וֹת עוֹלָ֔ם לְמֽוֹרָשָׁ֖ה הָ֥יְתָה לָּֽנוּ׃
3הִנָּבֵ֣א וְאָמַרְתָּ֔ כֹּ֥ה אָמַ֖ר אֲדֹנָ֣י יְהוִ֑ה יַ֣עַן בְּיַ֡עַן שַׁמּוֹת֩ וְשָׁאֹ֨ף אֶתְכֶ֜ם מִסָּבִ֗יב לִֽהְיוֹתְכֶ֤ם מֽוֹרָשָׁה֙ לִשְׁאֵרִ֣ית הַגּוֹיִ֔ם וַתֵּֽעֲל֛וּ עַל־ שְׂפַ֥ת לָשׁ֖וֹן וְדִבַּת־ עָֽם׃
4הָרֵ֣י יִשְׂרָאֵ֔ל שִׁמְע֖וּ דְּבַר־ אֲדֹנָ֣י יְהוִ֑ה כֹּֽה־ אָמַ֣ר אֲדֹנָ֣י יְ֠הוִה לֶהָרִ֨ים וְלַגְּבָע֜וֹת לָאֲפִיקִ֣ים וְלַגֵּאָי֗וֹת וְלֶחֳרָב֤וֹת הַשֹּֽׁמְמוֹת֙ וְלֶעָרִ֣ים הַנֶּעֱזָב֔וֹת אֲשֶׁ֨ר הָי֤וּ לְבַז֙ וּלְלַ֔עַג לִשְׁאֵרִ֥ית הַגּוֹיִ֖ם אֲשֶׁ֥ר מִסָּבִֽיב׃ ס
5כֹּֽה־ אָמַר֮ אֲדֹנָ֣י יְהוִה֒ אִם־ לֹ֠א בְּאֵ֨שׁ קִנְאָתִ֥י דִבַּ֛רְתִּי עַל־ שְׁאֵרִ֥ית הַגּוֹיִ֖ם וְעַל־ אֱד֣וֹם כֻּלָּ֑א אֲשֶׁ֣ר נָתְנֽוּ־ אֶת־ אַרְצִ֣י ׀ לָ֠הֶם לְמ֨וֹרָשָׁ֜ה בְּשִׂמְחַ֤ת כָּל־ לֵבָב֙ בִּשְׁאָ֣ט נֶ֔פֶשׁ לְמַ֥עַן מִגְרָשָׁ֖הּ לָבַֽז׃
6הִנָּבֵ֖א עַל־ אַדְמַ֣ת יִשְׂרָאֵ֑ל וְאָמַרְתָּ֡ לֶהָרִ֣ים וְ֠לַגְּבָעוֹת לָאֲפִיקִ֨ים וְלַגֵּאָי֜וֹת כֹּֽה־ אָמַ֣ר ׀ אֲדֹנָ֣י יְהוִ֗ה הִנְנִ֨י בְקִנְאָתִ֤י וּבַחֲמָתִי֙ דִּבַּ֔רְתִּי יַ֛עַן כְּלִמַּ֥ת גּוֹיִ֖ם נְשָׂאתֶֽם׃
7כֹּ֤ה אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה אֲנִ֖י נָשָׂ֣אתִי אֶת־ יָדִ֑י אִם־ לֹ֤א הַגּוֹיִם֙ אֲשֶׁ֣ר לָכֶ֣ם מִסָּבִ֔יב הֵ֖מָּה כְּלִמָּתָ֥ם יִשָּֽׂאוּ׃
8הָרֵ֤י יִשְׂרָאֵל֙ עַנְפְּכֶ֣ם תִּתֵּ֔נוּ וּפֶרְיְכֶ֥ם תִּשְׂא֖וּ לְעַמִּ֣י יִשְׂרָאֵ֑ל כִּ֥י קֵרְב֖וּ לָבֽוֹא׃
9הִנְנִ֣י אֲלֵיכֶ֑ם וּפָנִ֣יתִי אֲלֵיכֶ֔ם וְנֶעֱבַדְתֶּ֖ם וְנִזְרַעְתֶּֽם׃
10עֲלֵיכֶם֙ אָדָ֔ם כָּל־ בֵּ֥ית יִשְׂרָאֵ֖ל כֻּלֹּ֑ה וְנֹֽשְׁבוּ֙ הֶֽעָרִ֔ים וְהֶחֳרָב֖וֹת תִּבָּנֶֽינָה׃
11עֲלֵיכֶ֛ם אָדָ֥ם וּבְהֵמָ֖ה וְרָב֣וּ וּפָר֑וּ וְהוֹשַׁבְתִּ֨י אֶתְכֶ֜ם כְּקַדְמֽוֹתֵיכֶ֗ם וְהֵטִֽבֹתִי֙ מֵרִאשֹׁ֣תֵיכֶ֔ם וִֽידַעְתֶּ֖ם כִּֽי־ אֲנִ֥י יְהוָֽה׃
12עֲלֵיכֶ֨ם אָדָ֜ם אֶת־ עַמִּ֤י יִשְׂרָאֵל֙ וִֽירֵשׁ֔וּךָ וְהָיִ֥יתָ לָהֶ֖ם לְנַחֲלָ֑ה וְלֹא־ תוֹסִ֥ף ע֖וֹד לְשַׁכְּלָֽם׃ ס
13אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה יַ֚עַן אֹמְרִ֣ים לָכֶ֔ם אֹכֶ֥לֶת אָדָ֖ם [אָתִּי כ] (אָ֑תְּ ק) וּמְשַׁכֶּ֥לֶת [גֹּויֵךְ כ] (גֹּויַ֖יִךְ ק) הָיִֽית׃
14אָדָם֙ לֹא־ תֹ֣אכְלִי ע֔וֹד [וְגֹויֵךְ כ] (וְגֹויַ֖יִךְ ק) לֹ֣א [תְכַשְּׁלִי־ כ] (תְשַׁכְּלִי־ ע֑וֹד ק) נְאֻ֖ם אֲדֹנָ֥י יְהוִֽה׃
15אַשְׁמִ֨יעַ אֵלַ֤יִךְ עוֹד֙ כְּלִמַּ֣ת הַגּוֹיִ֔ם וְחֶרְפַּ֥ת עַמִּ֖ים לֹ֣א תִשְׂאִי־ ע֑וֹד [וְגֹויֵךְ כ] (וְגֹויַ֙יִךְ֙ ק) לֹא־ תַכְשִׁ֣לִי ע֔וֹד נְאֻ֖ם אֲדֹנָ֥י יְהוִֽה׃ ס
16דְבַר־ יְהוָ֖ה אֵלַ֥י לֵאמֹֽר׃
17אָדָ֗ם בֵּ֤ית יִשְׂרָאֵל֙ יֹשְׁבִ֣ים עַל־ אַדְמָתָ֔ם וַיְטַמְּא֣וּ אוֹתָ֔הּ בְּדַרְכָּ֖ם וּבַעֲלִֽילוֹתָ֑ם כְּטֻמְאַת֙ הַנִּדָּ֔ה הָיְתָ֥ה דַרְכָּ֖ם לְפָנָֽי׃
18חֲמָתִי֙ עֲלֵיהֶ֔ם עַל־ הַדָּ֖ם אֲשֶׁר־ שָׁפְכ֣וּ עַל־ הָאָ֑רֶץ וּבְגִלּוּלֵיהֶ֖ם טִמְּאֽוּהָ׃
19אֹתָם֙ בַּגּוֹיִ֔ם וַיִּזָּר֖וּ בָּאֲרָצ֑וֹת כְּדַרְכָּ֥ם וְכַעֲלִילוֹתָ֖ם שְׁפַטְתִּֽים׃
20אֶל־ הַגּוֹיִם֙ אֲשֶׁר־ בָּ֣אוּ שָׁ֔ם וַֽיְחַלְּל֖וּ אֶת־ שֵׁ֣ם קָדְשִׁ֑י בֶּאֱמֹ֤ר לָהֶם֙ עַם־ יְהוָ֣ה אֵ֔לֶּה וּמֵאַרְצ֖וֹ יָצָֽאוּ׃
21עַל־ שֵׁ֣ם קָדְשִׁ֑י אֲשֶׁ֤ר חִלְּל֙וּהוּ֙ בֵּ֣ית יִשְׂרָאֵ֔ל בַּגּוֹיִ֖ם אֲשֶׁר־ בָּ֥אוּ שָֽׁמָּה׃ ס
22אֱמֹ֣ר לְבֵֽית־ יִשְׂרָאֵ֗ל כֹּ֤ה אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה לֹ֧א לְמַעַנְכֶ֛ם אֲנִ֥י עֹשֶׂ֖ה בֵּ֣ית יִשְׂרָאֵ֑ל כִּ֤י אִם־ לְשֵׁם־ קָדְשִׁי֙ אֲשֶׁ֣ר חִלַּלְתֶּ֔ם בַּגּוֹיִ֖ם אֲשֶׁר־ בָּ֥אתֶם שָֽׁם׃
23אֶת־ שְׁמִ֣י הַגָּד֗וֹל הַֽמְחֻלָּל֙ בַּגּוֹיִ֔ם אֲשֶׁ֥ר חִלַּלְתֶּ֖ם בְּתוֹכָ֑ם וְיָדְע֨וּ הַגּוֹיִ֜ם כִּי־ אֲנִ֣י יְהוָ֗ה נְאֻם֙ אֲדֹנָ֣י יְהוִ֔ה בְּהִקָּדְשִׁ֥י בָכֶ֖ם לְעֵינֵיהֶֽם׃
24אֶתְכֶם֙ מִן־ הַגּוֹיִ֔ם וְקִבַּצְתִּ֥י אֶתְכֶ֖ם מִכָּל־ הָאֲרָצ֑וֹת וְהֵבֵאתִ֥י אֶתְכֶ֖ם אֶל־ אַדְמַתְכֶֽם׃
25עֲלֵיכֶ֛ם מַ֥יִם טְהוֹרִ֖ים וּטְהַרְתֶּ֑ם מִכֹּ֧ל טֻמְאוֹתֵיכֶ֛ם וּמִכָּל־ גִּלּ֥וּלֵיכֶ֖ם אֲטַהֵ֥ר אֶתְכֶֽם׃
26לָכֶם֙ לֵ֣ב חָדָ֔שׁ וְר֥וּחַ חֲדָשָׁ֖ה אֶתֵּ֣ן בְּקִרְבְּכֶ֑ם וַהֲסִ֨רֹתִ֜י אֶת־ לֵ֤ב הָאֶ֙בֶן֙ מִבְּשַׂרְכֶ֔ם וְנָתַתִּ֥י לָכֶ֖ם לֵ֥ב בָּשָֽׂר׃
27רוּחִ֖י אֶתֵּ֣ן בְּקִרְבְּכֶ֑ם וְעָשִׂ֗יתִי אֵ֤ת אֲשֶׁר־ בְּחֻקַּי֙ תֵּלֵ֔כוּ וּמִשְׁפָּטַ֥י תִּשְׁמְר֖וּ וַעֲשִׂיתֶֽם׃
28בָּאָ֔רֶץ אֲשֶׁ֥ר נָתַ֖תִּי לַאֲבֹֽתֵיכֶ֑ם וִהְיִ֤יתֶם לִי֙ לְעָ֔ם וְאָ֣נֹכִ֔י אֶהְיֶ֥ה לָכֶ֖ם לֵאלֹהִֽים׃
29אֶתְכֶ֔ם מִכֹּ֖ל טֻמְאֽוֹתֵיכֶ֑ם וְקָרָ֤אתִי אֶל־ הַדָּגָן֙ וְהִרְבֵּיתִ֣י אֹת֔וֹ וְלֹא־ אֶתֵּ֥ן עֲלֵיכֶ֖ם רָעָֽב׃
30אֶת־ פְּרִ֣י הָעֵ֔ץ וּתְנוּבַ֖ת הַשָּׂדֶ֑ה לְמַ֗עַן אֲ֠שֶׁר לֹ֣א תִקְח֥וּ ע֛וֹד חֶרְפַּ֥ת רָעָ֖ב בַּגּוֹיִֽם׃
31אֶת־ דַּרְכֵיכֶ֣ם הָרָעִ֔ים וּמַעַלְלֵיכֶ֖ם אֲשֶׁ֣ר לֹֽא־ טוֹבִ֑ים וּנְקֹֽטֹתֶם֙ בִּפְנֵיכֶ֔ם עַ֚ל עֲוֹנֹ֣תֵיכֶ֔ם וְעַ֖ל תּוֹעֲבֽוֹתֵיכֶֽם׃
32לְמַעַנְכֶ֣ם אֲנִֽי־ עֹשֶׂ֗ה נְאֻם֙ אֲדֹנָ֣י יְהוִ֔ה יִוָּדַ֖ע לָכֶ֑ם בּ֧וֹשׁוּ וְהִכָּלְמ֛וּ מִדַּרְכֵיכֶ֖ם בֵּ֥ית יִשְׂרָאֵֽל׃ ס
33אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה בְּיוֹם֙ טַהֲרִ֣י אֶתְכֶ֔ם מִכֹּ֖ל עֲוֹנֽוֹתֵיכֶ֑ם וְהֽוֹשַׁבְתִּי֙ אֶת־ הֶ֣עָרִ֔ים וְנִבְנ֖וּ הֶחֳרָבֽוֹת׃
34הַנְּשַׁמָּ֖ה תֵּֽעָבֵ֑ד תַּ֚חַת אֲשֶׁ֣ר הָיְתָ֣ה שְׁמָמָ֔ה לְעֵינֵ֖י כָּל־ עוֹבֵֽר׃
35הָאָ֤רֶץ הַלֵּ֙זוּ֙ הַנְּשַׁמָּ֔ה הָיְתָ֖ה כְּגַן־ עֵ֑דֶן וְהֶעָרִ֧ים הֶחֳרֵב֛וֹת וְהַֽנְשַׁמּ֥וֹת וְהַנֶּהֱרָס֖וֹת בְּצוּר֥וֹת יָשָֽׁבוּ׃
36הַגּוֹיִ֗ם אֲשֶׁ֣ר יִֽשָּׁאֲרוּ֮ סְבִיבוֹתֵיכֶם֒ כִּ֣י ׀ אֲנִ֣י יְהוָ֗ה בָּנִ֙יתִי֙ הַנֶּ֣הֱרָס֔וֹת נָטַ֖עְתִּי הַנְּשַׁמָּ֑ה אֲנִ֥י יְהוָ֖ה דִּבַּ֥רְתִּי וְעָשִֽׂיתִי׃ ס
37אָמַר֙ אֲדֹנָ֣י יְהוִ֔ה ע֗וֹד זֹ֛את אִדָּרֵ֥שׁ לְבֵֽית־ יִשְׂרָאֵ֖ל לַעֲשׂ֣וֹת לָהֶ֑ם אַרְבֶּ֥ה אֹתָ֛ם כַּצֹּ֖אן אָדָֽם׃
