1וְיוֹסֵ֖ף הוּרַ֣ד מִצְרָ֑יְמָה וַיִּקְנֵ֡הוּ פּוֹטִיפַר֩ סְרִ֨יס פַּרְעֹ֜ה שַׂ֤ר הַטַּבָּחִים֙ אִ֣ישׁ מִצְרִ֔י מִיַּד֙ הַיִּשְׁמְעֵאלִ֔ים אֲשֶׁ֥ר הוֹרִדֻ֖הוּ שָֽׁמָּה׃
2יְהוָה֙ אֶת־ יוֹסֵ֔ף וַיְהִ֖י אִ֣ישׁ מַצְלִ֑יחַ וַיְהִ֕י בְּבֵ֥ית אֲדֹנָ֖יו הַמִּצְרִֽי׃
3אֲדֹנָ֔יו כִּ֥י יְהוָ֖ה אִתּ֑וֹ וְכֹל֙ אֲשֶׁר־ ה֣וּא עֹשֶׂ֔ה יְהוָ֖ה מַצְלִ֥יחַ בְּיָדֽוֹ׃
4יוֹסֵ֥ף חֵ֛ן בְּעֵינָ֖יו וַיְשָׁ֣רֶת אֹת֑וֹ וַיַּפְקִדֵ֙הוּ֙ עַל־ בֵּית֔וֹ וְכָל־ יֶשׁ־ ל֖וֹ נָתַ֥ן בְּיָדֽוֹ׃
5מֵאָז֩ הִפְקִ֨יד אֹת֜וֹ בְּבֵית֗וֹ וְעַל֙ כָּל־ אֲשֶׁ֣ר יֶשׁ־ ל֔וֹ וַיְבָ֧רֶךְ יְהוָ֛ה אֶת־ בֵּ֥ית הַמִּצְרִ֖י בִּגְלַ֣ל יוֹסֵ֑ף וַיְהִ֞י בִּרְכַּ֤ת יְהוָה֙ בְּכָל־ אֲשֶׁ֣ר יֶשׁ־ ל֔וֹ בַּבַּ֖יִת וּבַשָּׂדֶֽה׃
6כָּל־ אֲשֶׁר־ לוֹ֮ בְּיַד־ יוֹסֵף֒ וְלֹא־ יָדַ֤ע אִתּוֹ֙ מְא֔וּמָה כִּ֥י אִם־ הַלֶּ֖חֶם אֲשֶׁר־ ה֣וּא אוֹכֵ֑ל וַיְהִ֣י יוֹסֵ֔ף יְפֵה־ תֹ֖אַר וִיפֵ֥ה מַרְאֶֽה׃
7אַחַר֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וַתִּשָּׂ֧א אֵֽשֶׁת־ אֲדֹנָ֛יו אֶת־ עֵינֶ֖יהָ אֶל־ יוֹסֵ֑ף וַתֹּ֖אמֶר שִׁכְבָ֥ה עִמִּֽי׃
8וַיֹּ֙אמֶר֙ אֶל־ אֵ֣שֶׁת אֲדֹנָ֔יו הֵ֣ן אֲדֹנִ֔י לֹא־ יָדַ֥ע אִתִּ֖י מַה־ בַּבָּ֑יִת וְכֹ֥ל אֲשֶׁר־ יֶשׁ־ ל֖וֹ נָתַ֥ן בְּיָדִֽי׃
9גָד֜וֹל בַּבַּ֣יִת הַזֶּה֮ מִמֶּנִּי֒ וְלֹֽא־ חָשַׂ֤ךְ מִמֶּ֙נִּי֙ מְא֔וּמָה כִּ֥י אִם־ אוֹתָ֖ךְ בַּאֲשֶׁ֣ר אַתְּ־ אִשְׁתּ֑וֹ וְאֵ֨יךְ אֶֽעֱשֶׂ֜ה הָרָעָ֤ה הַגְּדֹלָה֙ הַזֹּ֔את וְחָטָ֖אתִי לֵֽאלֹהִֽים׃
10כְּדַבְּרָ֥הּ אֶל־ יוֹסֵ֖ף י֣וֹם ׀ י֑וֹם וְלֹא־ שָׁמַ֥ע אֵלֶ֛יהָ לִשְׁכַּ֥ב אֶצְלָ֖הּ לִהְי֥וֹת עִמָּֽהּ׃
11כְּהַיּ֣וֹם הַזֶּ֔ה וַיָּבֹ֥א הַבַּ֖יְתָה לַעֲשׂ֣וֹת מְלַאכְתּ֑וֹ וְאֵ֨ין אִ֜ישׁ מֵאַנְשֵׁ֥י הַבַּ֛יִת שָׁ֖ם בַּבָּֽיִת׃
12בְּבִגְד֛וֹ לֵאמֹ֖ר שִׁכְבָ֣ה עִמִּ֑י וַיַּעֲזֹ֤ב בִּגְדוֹ֙ בְּיָדָ֔הּ וַיָּ֖נָס וַיֵּצֵ֥א הַחֽוּצָה׃
13כִּרְאוֹתָ֔הּ כִּֽי־ עָזַ֥ב בִּגְד֖וֹ בְּיָדָ֑הּ וַיָּ֖נָס הַחֽוּצָה׃
14לְאַנְשֵׁ֣י בֵיתָ֗הּ וַתֹּ֤אמֶר לָהֶם֙ לֵאמֹ֔ר רְא֗וּ הֵ֥בִיא לָ֛נוּ אִ֥ישׁ עִבְרִ֖י לְצַ֣חֶק בָּ֑נוּ בָּ֤א אֵלַי֙ לִשְׁכַּ֣ב עִמִּ֔י וָאֶקְרָ֖א בְּק֥וֹל גָּדֽוֹל׃
15כְשָׁמְע֔וֹ כִּֽי־ הֲרִימֹ֥תִי קוֹלִ֖י וָאֶקְרָ֑א וַיַּעֲזֹ֤ב בִּגְדוֹ֙ אֶצְלִ֔י וַיָּ֖נָס וַיֵּצֵ֥א הַחֽוּצָה׃
16בִּגְד֖וֹ אֶצְלָ֑הּ עַד־ בּ֥וֹא אֲדֹנָ֖יו אֶל־ בֵּיתֽוֹ׃
17אֵלָ֔יו כַּדְּבָרִ֥ים הָאֵ֖לֶּה לֵאמֹ֑ר בָּֽא־ אֵלַ֞י הָעֶ֧בֶד הָֽעִבְרִ֛י אֲשֶׁר־ הֵבֵ֥אתָ לָּ֖נוּ לְצַ֥חֶק בִּֽי׃
18כַּהֲרִימִ֥י קוֹלִ֖י וָאֶקְרָ֑א וַיַּעֲזֹ֥ב בִּגְד֛וֹ אֶצְלִ֖י וַיָּ֥נָס הַחֽוּצָה׃
19כִשְׁמֹ֨עַ אֲדֹנָ֜יו אֶת־ דִּבְרֵ֣י אִשְׁתּ֗וֹ אֲשֶׁ֨ר דִּבְּרָ֤ה אֵלָיו֙ לֵאמֹ֔ר כַּדְּבָרִ֣ים הָאֵ֔לֶּה עָ֥שָׂהּ לִ֖י עַבְדֶּ֑ךָ וַיִּ֖חַר אַפּֽוֹ׃
20אֲדֹנֵ֨י יוֹסֵ֜ף אֹת֗וֹ וַֽיִּתְּנֵ֙הוּ֙ אֶל־ בֵּ֣ית הַסֹּ֔הַר מְק֕וֹם אֲשֶׁר־ [אֲסוּרֵי כ] (אֲסִירֵ֥י ק) הַמֶּ֖לֶךְ אֲסוּרִ֑ים וַֽיְהִי־ שָׁ֖ם בְּבֵ֥ית הַסֹּֽהַר׃
21יְהוָה֙ אֶת־ יוֹסֵ֔ף וַיֵּ֥ט אֵלָ֖יו חָ֑סֶד וַיִּתֵּ֣ן חִנּ֔וֹ בְּעֵינֵ֖י שַׂ֥ר בֵּית־ הַסֹּֽהַר׃
22שַׂ֤ר בֵּית־ הַסֹּ֙הַר֙ בְּיַד־ יוֹסֵ֔ף אֵ֚ת כָּל־ הָ֣אֲסִירִ֔ם אֲשֶׁ֖ר בְּבֵ֣ית הַסֹּ֑הַר וְאֵ֨ת כָּל־ אֲשֶׁ֤ר עֹשִׂים֙ שָׁ֔ם ה֖וּא הָיָ֥ה עֹשֶֽׂה׃
