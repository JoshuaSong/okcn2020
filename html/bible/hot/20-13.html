1בֵּ֣ן חָ֭כָם מ֣וּסַר אָ֑ב וְ֝לֵ֗ץ לֹא־ שָׁמַ֥ע גְּעָרָֽה׃
2פִי־ אִ֭ישׁ יֹ֣אכַל ט֑וֹב וְנֶ֖פֶשׁ בֹּגְדִ֣ים חָמָֽס׃
3פִּ֭יו שֹׁמֵ֣ר נַפְשׁ֑וֹ פֹּשֵׂ֥ק שְׂ֝פָתָ֗יו מְחִתָּה־ לֽוֹ׃
4וָ֭אַיִן נַפְשׁ֣וֹ עָצֵ֑ל וְנֶ֖פֶשׁ חָרֻצִ֣ים תְּדֻשָּֽׁן׃
5שֶׁ֭קֶר יִשְׂנָ֣א צַדִּ֑יק וְ֝רָשָׁ֗ע יַבְאִ֥ישׁ וְיַחְפִּֽיר׃
6תִּצֹּ֣ר תָּם־ דָּ֑רֶךְ וְ֝רִשְׁעָ֗ה תְּסַלֵּ֥ף חַטָּֽאת׃
7מִ֭תְעַשֵּׁר וְאֵ֣ין כֹּ֑ל מִ֝תְרוֹשֵׁ֗שׁ וְה֣וֹן רָֽב׃
8נֶֽפֶשׁ־ אִ֣ישׁ עָשְׁר֑וֹ וְ֝רָ֗שׁ לֹא־ שָׁמַ֥ע גְּעָרָֽה׃
9צַדִּיקִ֥ים יִשְׂמָ֑ח וְנֵ֖ר רְשָׁעִ֣ים יִדְעָֽךְ׃
10בְּ֭זָדוֹן יִתֵּ֣ן מַצָּ֑ה וְאֶת־ נ֖וֹעָצִ֣ים חָכְמָֽה׃
11מֵהֶ֣בֶל יִמְעָ֑ט וְקֹבֵ֖ץ עַל־ יָ֣ד יַרְבֶּֽה׃
12מְ֭מֻשָּׁכָה מַחֲלָה־ לֵ֑ב וְעֵ֥ץ חַ֝יִּ֗ים תַּאֲוָ֥ה בָאָֽה׃
13לְ֭דָבָר יֵחָ֣בֶל ל֑וֹ וִירֵ֥א מִ֝צְוָ֗ה ה֣וּא יְשֻׁלָּֽם׃
14חָ֭כָם מְק֣וֹר חַיִּ֑ים לָ֝ס֗וּר מִמֹּ֥קְשֵׁי מָֽוֶת׃
15ט֭וֹב יִתֶּן־ חֵ֑ן וְדֶ֖רֶךְ בֹּגְדִ֣ים אֵיתָֽן׃
16עָ֭רוּם יַעֲשֶׂ֣ה בְדָ֑עַת וּ֝כְסִ֗יל יִפְרֹ֥שׂ אִוֶּֽלֶת׃
17רָ֭שָׁע יִפֹּ֣ל בְּרָ֑ע וְצִ֖יר אֱמוּנִ֣ים מַרְפֵּֽא׃
18וְ֭קָלוֹן פּוֹרֵ֣עַ מוּסָ֑ר וְשׁוֹמֵ֖ר תּוֹכַ֣חַת יְכֻבָּֽד׃
19נִ֭הְיָה תֶּעֱרַ֣ב לְנָ֑פֶשׁ וְתוֹעֲבַ֥ת כְּ֝סִילִ֗ים ס֣וּר מֵרָֽע׃
20כ] (הֹולֵ֣ךְ ק) אֶת־ חֲכָמִ֣ים [וַחֲכָם כ] (יֶחְכָּ֑ם ק) וְרֹעֶ֖ה כְסִילִ֣ים יֵרֽוֹעַ׃
21תְּרַדֵּ֣ף רָעָ֑ה וְאֶת־ צַ֝דִּיקִ֗ים יְשַׁלֶּם־ טֽוֹב׃
22יַנְחִ֥יל בְּנֵֽי־ בָנִ֑ים וְצָפ֥וּן לַ֝צַּדִּ֗יק חֵ֣יל חוֹטֵֽא׃
23אֹ֭כֶל נִ֣יר רָאשִׁ֑ים וְיֵ֥שׁ נִ֝סְפֶּ֗ה בְּלֹ֣א מִשְׁפָּֽט׃
24שִׁ֭בְטוֹ שׂוֹנֵ֣א בְנ֑וֹ וְ֝אֹהֲב֗וֹ שִֽׁחֲר֥וֹ מוּסָֽר׃
