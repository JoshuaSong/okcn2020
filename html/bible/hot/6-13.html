1וִיהוֹשֻׁ֣עַ זָקֵ֔ן בָּ֖א בַּיָּמִ֑ים וַיֹּ֨אמֶר יְהוָ֜ה אֵלָ֗יו אַתָּ֤ה זָקַ֙נְתָּה֙ בָּ֣אתָ בַיָּמִ֔ים וְהָאָ֛רֶץ נִשְׁאֲרָ֥ה הַרְבֵּֽה־ מְאֹ֖ד לְרִשְׁתָּֽהּ׃
2הָאָ֖רֶץ הַנִּשְׁאָ֑רֶת כָּל־ גְּלִיל֥וֹת הַפְּלִשְׁתִּ֖ים וְכָל־ הַגְּשׁוּרִֽי׃
3הַשִּׁיח֞וֹר אֲשֶׁ֣ר ׀ עַל־ פְּנֵ֣י מִצְרַ֗יִם וְעַ֨ד גְּב֤וּל עֶקְרוֹן֙ צָפ֔וֹנָה לַֽכְּנַעֲנִ֖י תֵּחָשֵׁ֑ב חֲמֵ֣שֶׁת ׀ סַרְנֵ֣י פְלִשְׁתִּ֗ים הָעַזָּתִ֤י וְהָאַשְׁדּוֹדִי֙ הָאֶשְׁקְלוֹנִ֣י הַגִּתִּ֔י וְהָעֶקְרוֹנִ֖י וְהָעַוִּֽים׃
4כָּל־ אֶ֣רֶץ הַֽכְּנַעֲנִ֗י וּמְעָרָ֛ה אֲשֶׁ֥ר לַצִּידֹנִ֖ים עַד־ אֲפֵ֑קָה עַ֖ד גְּב֥וּל הָאֱמֹרִֽי׃
5הַגִּבְלִ֗י וְכָל־ הַלְּבָנוֹן֙ מִזְרַ֣ח הַשֶּׁ֔מֶשׁ מִבַּ֣עַל גָּ֔ד תַּ֖חַת הַר־ חֶרְמ֑וֹן עַ֖ד לְב֥וֹא חֲמָֽת׃
6יֹשְׁבֵ֣י הָ֠הָר מִֽן־ הַלְּבָנ֞וֹן עַד־ מִשְׂרְפֹ֥ת מַ֙יִם֙ כָּל־ צִ֣ידֹנִ֔ים אָֽנֹכִי֙ אוֹרִישֵׁ֔ם מִפְּנֵ֖י בְּנֵ֣י יִשְׂרָאֵ֑ל רַ֠ק הַפִּלֶ֤הָ לְיִשְׂרָאֵל֙ בְּֽנַחֲלָ֔ה כַּאֲשֶׁ֖ר צִוִּיתִֽיךָ׃
7חַלֵּ֞ק אֶת־ הָאָ֧רֶץ הַזֹּ֛את בְּנַחֲלָ֖ה לְתִשְׁעַ֣ת הַשְּׁבָטִ֑ים וַחֲצִ֖י הַשֵּׁ֥בֶט הַֽמְנַשֶּֽׁה׃
8הָרֽאוּבֵנִי֙ וְהַגָּדִ֔י לָקְח֖וּ נַחֲלָתָ֑ם אֲשֶׁר֩ נָתַ֨ן לָהֶ֜ם מֹשֶׁ֗ה בְּעֵ֤בֶר הַיַּרְדֵּן֙ מִזְרָ֔חָה כַּאֲשֶׁר֙ נָתַ֣ן לָהֶ֔ם מֹשֶׁ֖ה עֶ֥בֶד יְהוָֽה׃
9אֲשֶׁר֩ עַל־ שְׂפַת־ נַ֨חַל אַרְנ֜וֹן וְהָעִ֨יר אֲשֶׁ֧ר בְּתוֹךְ־ הַנַּ֛חַל וְכָל־ הַמִּישֹׁ֥ר מֵידְבָ֖א עַד־ דִּיבֽוֹן׃
10עָרֵי֙ סִיחוֹן֙ מֶ֣לֶךְ הָאֱמֹרִ֔י אֲשֶׁ֥ר מָלַ֖ךְ בְּחֶשְׁבּ֑וֹן עַד־ גְּב֖וּל בְּנֵ֥י עַמּֽוֹן׃
11וּגְב֧וּל הַגְּשׁוּרִ֣י וְהַמַּעֲכָתִ֗י וְכֹ֨ל הַ֥ר חֶרְמ֛וֹן וְכָל־ הַבָּשָׁ֖ן עַד־ סַלְכָֽה׃
12מַמְלְכ֥וּת עוֹג֙ בַּבָּשָׁ֔ן אֲשֶׁר־ מָלַ֥ךְ בְּעַשְׁתָּר֖וֹת וּבְאֶדְרֶ֑עִי ה֤וּא נִשְׁאַר֙ מִיֶּ֣תֶר הָרְפָאִ֔ים וַיַּכֵּ֥ם מֹשֶׁ֖ה וַיֹּרִשֵֽׁם׃
13הוֹרִ֙ישׁוּ֙ בְּנֵ֣י יִשְׂרָאֵ֔ל אֶת־ הַגְּשׁוּרִ֖י וְאֶת־ הַמַּעֲכָתִ֑י וַיֵּ֨שֶׁב גְּשׁ֤וּר וּמַֽעֲכָת֙ בְּקֶ֣רֶב יִשְׂרָאֵ֔ל עַ֖ד הַיּ֥וֹם הַזֶּֽה׃
14לְשֵׁ֣בֶט הַלֵּוִ֔י לֹ֥א נָתַ֖ן נַחֲלָ֑ה אִשֵּׁ֨י יְהוָ֜ה אֱלֹהֵ֤י יִשְׂרָאֵל֙ ה֣וּא נַחֲלָת֔וֹ כַּאֲשֶׁ֖ר דִּבֶּר־ לֽוֹ׃ ס
15מֹשֶׁ֔ה לְמַטֵּ֥ה בְנֵֽי־ רְאוּבֵ֖ן לְמִשְׁפְּחֹתָֽם׃
16לָהֶ֜ם הַגְּב֗וּל מֵעֲרוֹעֵ֡ר אֲשֶׁר֩ עַל־ שְׂפַת־ נַ֨חַל אַרְנ֜וֹן וְהָעִ֨יר אֲשֶׁ֧ר בְּתוֹךְ־ הַנַּ֛חַל וְכָל־ הַמִּישֹׁ֖ר עַל־ מֵידְבָֽא׃
17וְכָל־ עָרֶ֖יהָ אֲשֶׁ֣ר בַּמִּישֹׁ֑ר דִּיבוֹן֙ וּבָמ֣וֹת בַּ֔עַל וּבֵ֖ית בַּ֥עַל מְעֽוֹן׃
18וּקְדֵמֹ֖ת וּמֵפָֽעַת׃
19וְשִׂבְמָ֔ה וְצֶ֥רֶת הַשַּׁ֖חַר בְּהַ֥ר הָעֵֽמֶק׃
20פְּע֛וֹר וְאַשְׁדּ֥וֹת הַפִּסְגָּ֖ה וּבֵ֥ית הַיְשִׁמֽוֹת׃
21עָרֵ֣י הַמִּישֹׁ֔ר וְכָֽל־ מַמְלְכ֗וּת סִיחוֹן֙ מֶ֣לֶךְ הָאֱמֹרִ֔י אֲשֶׁ֥ר מָלַ֖ךְ בְּחֶשְׁבּ֑וֹן אֲשֶׁר֩ הִכָּ֨ה מֹשֶׁ֜ה אֹת֣וֹ ׀ וְאֶת־ נְשִׂיאֵ֣י מִדְיָ֗ן אֶת־ אֱוִ֤י וְאֶת־ רֶ֙קֶם֙ וְאֶת־ צ֤וּר וְאֶת־ חוּר֙ וְאֶת־ רֶ֔בַע נְסִיכֵ֣י סִיח֔וֹן יֹשְׁבֵ֖י הָאָֽרֶץ׃
22בִּלְעָ֥ם בֶּן־ בְּע֖וֹר הַקּוֹסֵ֑ם הָרְג֧וּ בְנֵֽי־ יִשְׂרָאֵ֛ל בַּחֶ֖רֶב אֶל־ חַלְלֵיהֶֽם׃
23גְּבוּל֙ בְּנֵ֣י רְאוּבֵ֔ן הַיַּרְדֵּ֖ן וּגְב֑וּל זֹ֣את נַחֲלַ֤ת בְּנֵֽי־ רְאוּבֵן֙ לְמִשְׁפְּחֹתָ֔ם הֶעָרִ֖ים וְחַצְרֵיהֶֽן׃ פ
24מֹשֶׁה֙ לְמַטֵּה־ גָ֔ד לִבְנֵי־ גָ֖ד לְמִשְׁפְּחֹתָֽם׃
25לָהֶם֙ הַגְּב֔וּל יַעְזֵר֙ וְכָל־ עָרֵ֣י הַגִּלְעָ֔ד וַחֲצִ֕י אֶ֖רֶץ בְּנֵ֣י עַמּ֑וֹן עַד־ עֲרוֹעֵ֕ר אֲשֶׁ֖ר עַל־ פְּנֵ֥י רַבָּֽה׃
26עַד־ רָמַ֥ת הַמִּצְפֶּ֖ה וּבְטֹנִ֑ים וּמִֽמַּחֲנַ֖יִם עַד־ גְּב֥וּל לִדְבִֽר׃
27בֵּ֣ית הָרָם֩ וּבֵ֨ית נִמְרָ֜ה וְסֻכּ֣וֹת וְצָפ֗וֹן יֶ֚תֶר מַמְלְכ֗וּת סִיחוֹן֙ מֶ֣לֶךְ חֶשְׁבּ֔וֹן הַיַּרְדֵּ֖ן וּגְבֻ֑ל עַד־ קְצֵה֙ יָם־ כִּנֶּ֔רֶת עֵ֥בֶר הַיַּרְדֵּ֖ן מִזְרָֽחָה׃
28נַחֲלַ֥ת בְּנֵי־ גָ֖ד לְמִשְׁפְּחֹתָ֑ם הֶעָרִ֖ים וְחַצְרֵיהֶֽם׃
29מֹשֶׁ֔ה לַחֲצִ֖י שֵׁ֣בֶט מְנַשֶּׁ֑ה וַיְהִ֗י לַחֲצִ֛י מַטֵּ֥ה בְנֵֽי־ מְנַשֶּׁ֖ה לְמִשְׁפְּחוֹתָֽם׃
30גְבוּלָ֗ם מִמַּחֲנַ֨יִם כָּֽל־ הַבָּשָׁ֜ן כָּֽל־ מַמְלְכ֣וּת ׀ ע֣וֹג מֶֽלֶךְ־ הַבָּשָׁ֗ן וְכָל־ חַוֹּ֥ת יָאִ֛יר אֲשֶׁ֥ר בַּבָּשָׁ֖ן שִׁשִּׁ֥ים עִֽיר׃
31הַגִּלְעָד֙ וְעַשְׁתָּר֣וֹת וְאֶדְרֶ֔עִי עָרֵ֛י מַמְלְכ֥וּת ע֖וֹג בַּבָּשָׁ֑ן לִבְנֵ֤י מָכִיר֙ בֶּן־ מְנַשֶּׁ֔ה לַחֲצִ֥י בְנֵֽי־ מָכִ֖יר לְמִשְׁפְּחוֹתָֽם׃
32אֲשֶׁר־ נִחַ֥ל מֹשֶׁ֖ה בְּעַֽרְב֣וֹת מוֹאָ֑ב מֵעֵ֛בֶר לְיַרְדֵּ֥ן יְרִיח֖וֹ מִזְרָֽחָה׃ ס
