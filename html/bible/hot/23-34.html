1קִרְב֤וּ גוֹיִם֙ לִשְׁמֹ֔עַ וּלְאֻמִּ֖ים הַקְשִׁ֑יבוּ תִּשְׁמַ֤ע הָאָ֙רֶץ֙ וּמְלֹאָ֔הּ תֵּבֵ֖ל וְכָל־ צֶאֱצָאֶֽיהָ׃
2קֶ֤צֶף לַֽיהוָה֙ עַל־ כָּל־ הַגּוֹיִ֔ם וְחֵמָ֖ה עַל־ כָּל־ צְבָאָ֑ם הֶחֱרִימָ֖ם נְתָנָ֥ם לַטָּֽבַח׃
3יֻשְׁלָ֔כוּ וּפִגְרֵיהֶ֖ם יַעֲלֶ֣ה בָאְשָׁ֑ם וְנָמַ֥סּוּ הָרִ֖ים מִדָּמָֽם׃
4כָּל־ צְבָ֣א הַשָּׁמַ֔יִם וְנָגֹ֥לּוּ כַסֵּ֖פֶר הַשָּׁמָ֑יִם וְכָל־ צְבָאָ֣ם יִבּ֔וֹל כִּנְבֹ֤ל עָלֶה֙ מִגֶּ֔פֶן וּכְנֹבֶ֖לֶת מִתְּאֵנָֽה׃
5רִוְּתָ֥ה בַשָּׁמַ֖יִם חַרְבִּ֑י הִנֵּה֙ עַל־ אֱד֣וֹם תֵּרֵ֔ד וְעַל־ עַ֥ם חֶרְמִ֖י לְמִשְׁפָּֽט׃
6לַיהוָ֞ה מָלְאָ֥ה דָם֙ הֻדַּ֣שְׁנָה מֵחֵ֔לֶב מִדַּ֤ם כָּרִים֙ וְעַתּוּדִ֔ים מֵחֵ֖לֶב כִּלְי֣וֹת אֵילִ֑ים כִּ֣י זֶ֤בַח לַֽיהוָה֙ בְּבָצְרָ֔ה וְטֶ֥בַח גָּד֖וֹל בְּאֶ֥רֶץ אֱדֽוֹם׃
7רְאֵמִים֙ עִמָּ֔ם וּפָרִ֖ים עִם־ אַבִּירִ֑ים וְרִוְּתָ֤ה אַרְצָם֙ מִדָּ֔ם וַעֲפָרָ֖ם מֵחֵ֥לֶב יְדֻשָּֽׁן׃
8י֥וֹם נָקָ֖ם לַֽיהוָ֑ה שְׁנַ֥ת שִׁלּוּמִ֖ים לְרִ֥יב צִיּֽוֹן׃
9נְחָלֶ֙יהָ֙ לְזֶ֔פֶת וַעֲפָרָ֖הּ לְגָפְרִ֑ית וְהָיְתָ֣ה אַרְצָ֔הּ לְזֶ֖פֶת בֹּעֵרָֽה׃
10וְיוֹמָם֙ לֹ֣א תִכְבֶּ֔ה לְעוֹלָ֖ם יַעֲלֶ֣ה עֲשָׁנָ֑הּ מִדּ֤וֹר לָדוֹר֙ תֶּחֱרָ֔ב לְנֵ֣צַח נְצָחִ֔ים אֵ֥ין עֹבֵ֖ר בָּֽהּ׃
11קָאַ֣ת וְקִפּ֔וֹד וְיַנְשׁ֥וֹף וְעֹרֵ֖ב יִשְׁכְּנוּ־ בָ֑הּ וְנָטָ֥ה עָלֶ֛יהָ קַֽו־ תֹ֖הוּ וְאַבְנֵי־ בֹֽהוּ׃
12וְאֵֽין־ שָׁ֖ם מְלוּכָ֣ה יִקְרָ֑אוּ וְכָל־ שָׂרֶ֖יהָ יִ֥הְיוּ אָֽפֶס׃
13אַרְמְנֹתֶ֙יהָ֙ סִירִ֔ים קִמּ֥וֹשׂ וָח֖וֹחַ בְּמִבְצָרֶ֑יהָ וְהָיְתָה֙ נְוֵ֣ה תַנִּ֔ים חָצִ֖יר לִבְנ֥וֹת יַעֲנָֽה׃
14צִיִּים֙ אֶת־ אִיִּ֔ים וְשָׂעִ֖יר עַל־ רֵעֵ֣הוּ יִקְרָ֑א אַךְ־ שָׁם֙ הִרְגִּ֣יעָה לִּילִ֔ית וּמָצְאָ֥ה לָ֖הּ מָנֽוֹחַ׃
15קִנְּנָ֤ה קִפּוֹז֙ וַתְּמַלֵּ֔ט וּבָקְעָ֖ה וְדָגְרָ֣ה בְצִלָּ֑הּ אַךְ־ שָׁ֛ם נִקְבְּצ֥וּ דַיּ֖וֹת אִשָּׁ֥ה רְעוּתָֽהּ׃
16מֵֽעַל־ סֵ֤פֶר יְהוָה֙ וּֽקְרָ֔אוּ אַחַ֤ת מֵהֵ֙נָּה֙ לֹ֣א נֶעְדָּ֔רָה אִשָּׁ֥ה רְעוּתָ֖הּ לֹ֣א פָקָ֑דוּ כִּֽי־ פִי֙ ה֣וּא צִוָּ֔ה וְרוּח֖וֹ ה֥וּא קִבְּצָֽן׃
