1וְאִם־ זֶ֥בַח שְׁלָמִ֖ים קָרְבָּנ֑וֹ אִ֤ם מִן־ הַבָּקָר֙ ה֣וּא מַקְרִ֔יב אִם־ זָכָר֙ אִם־ נְקֵבָ֔ה תָּמִ֥ים יַקְרִיבֶ֖נּוּ לִפְנֵ֥י יְהוָֽה׃
2יָדוֹ֙ עַל־ רֹ֣אשׁ קָרְבָּנ֔וֹ וּשְׁחָט֕וֹ פֶּ֖תַח אֹ֣הֶל מוֹעֵ֑ד וְזָרְק֡וּ בְּנֵי֩ אַהֲרֹ֨ן הַכֹּהֲנִ֧ים אֶת־ הַדָּ֛ם עַל־ הַמִּזְבֵּ֖חַ סָבִֽיב׃
3מִזֶּ֣בַח הַשְּׁלָמִ֔ים אִשֶּׁ֖ה לַיהוָ֑ה אֶת־ הַחֵ֙לֶב֙ הַֽמְכַסֶּ֣ה אֶת־ הַקֶּ֔רֶב וְאֵת֙ כָּל־ הַחֵ֔לֶב אֲשֶׁ֖ר עַל־ הַקֶּֽרֶב׃
4שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־ הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־ הַכְּסָלִ֑ים וְאֶת־ הַיֹּתֶ֙רֶת֙ עַל־ הַכָּבֵ֔ד עַל־ הַכְּלָי֖וֹת יְסִירֶֽנָּה׃
5אֹת֤וֹ בְנֵֽי־ אַהֲרֹן֙ הַמִּזְבֵּ֔חָה עַל־ הָ֣עֹלָ֔ה אֲשֶׁ֥ר עַל־ הָעֵצִ֖ים אֲשֶׁ֣ר עַל־ הָאֵ֑שׁ אִשֵּׁ֛ה רֵ֥יחַ נִיחֹ֖חַ לַֽיהוָֽה׃ פ
6מִן־ הַצֹּ֧אן קָרְבָּנ֛וֹ לְזֶ֥בַח שְׁלָמִ֖ים לַיהוָ֑ה זָכָר֙ א֣וֹ נְקֵבָ֔ה תָּמִ֖ים יַקְרִיבֶֽנּוּ׃
7כֶּ֥שֶׂב הֽוּא־ מַקְרִ֖יב אֶת־ קָרְבָּנ֑וֹ וְהִקְרִ֥יב אֹת֖וֹ לִפְנֵ֥י יְהוָֽה׃
8אֶת־ יָדוֹ֙ עַל־ רֹ֣אשׁ קָרְבָּנ֔וֹ וְשָׁחַ֣ט אֹת֔וֹ לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְ֠זָרְקוּ בְּנֵ֨י אַהֲרֹ֧ן אֶת־ דָּמ֛וֹ עַל־ הַמִּזְבֵּ֖חַ סָבִֽיב׃
9מִזֶּ֣בַח הַשְּׁלָמִים֮ אִשֶּׁ֣ה לַיהוָה֒ חֶלְבּוֹ֙ הָאַלְיָ֣ה תְמִימָ֔ה לְעֻמַּ֥ת הֶעָצֶ֖ה יְסִירֶ֑נָּה וְאֶת־ הַחֵ֙לֶב֙ הַֽמְכַסֶּ֣ה אֶת־ הַקֶּ֔רֶב וְאֵת֙ כָּל־ הַחֵ֔לֶב אֲשֶׁ֖ר עַל־ הַקֶּֽרֶב׃
10שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־ הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־ הַכְּסָלִ֑ים וְאֶת־ הַיֹּתֶ֙רֶת֙ עַל־ הַכָּבֵ֔ד עַל־ הַכְּלָיֹ֖ת יְסִירֶֽנָּה׃
11הַכֹּהֵ֖ן הַמִּזְבֵּ֑חָה לֶ֥חֶם אִשֶּׁ֖ה לַיהוָֽה׃ פ
12עֵ֖ז קָרְבָּנ֑וֹ וְהִקְרִיב֖וֹ לִפְנֵ֥י יְהוָֽה׃
13אֶת־ יָדוֹ֙ עַל־ רֹאשׁ֔וֹ וְשָׁחַ֣ט אֹת֔וֹ לִפְנֵ֖י אֹ֣הֶל מוֹעֵ֑ד וְ֠זָרְקוּ בְּנֵ֨י אַהֲרֹ֧ן אֶת־ דָּמ֛וֹ עַל־ הַמִּזְבֵּ֖חַ סָבִֽיב׃
14מִמֶּ֙נּוּ֙ קָרְבָּנ֔וֹ אִשֶּׁ֖ה לַֽיהוָ֑ה אֶת־ הַחֵ֙לֶב֙ הַֽמְכַסֶּ֣ה אֶת־ הַקֶּ֔רֶב וְאֵת֙ כָּל־ הַחֵ֔לֶב אֲשֶׁ֖ר עַל־ הַקֶּֽרֶב׃
15שְׁתֵּ֣י הַכְּלָיֹ֔ת וְאֶת־ הַחֵ֙לֶב֙ אֲשֶׁ֣ר עֲלֵהֶ֔ן אֲשֶׁ֖ר עַל־ הַכְּסָלִ֑ים וְאֶת־ הַיֹּתֶ֙רֶת֙ עַל־ הַכָּבֵ֔ד עַל־ הַכְּלָיֹ֖ת יְסִירֶֽנָּה׃
16הַכֹּהֵ֖ן הַמִּזְבֵּ֑חָה לֶ֤חֶם אִשֶּׁה֙ לְרֵ֣יחַ נִיחֹ֔חַ כָּל־ חֵ֖לֶב לַיהוָֽה׃
