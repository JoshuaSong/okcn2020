1מַשְׂכִּ֥יל לְדָוִ֑ד בִּהְיוֹת֖וֹ בַמְּעָרָ֣ה תְפִלָּֽה׃ ק֭וֹלִי אֶל־ יְהוָ֣ה אֶזְעָ֑ק ק֝וֹלִ֗י אֶל־ יְהוָ֥ה אֶתְחַנָּֽן׃
2לְפָנָ֣יו שִׂיחִ֑י צָ֝רָתִ֗י לְפָנָ֥יו אַגִּֽיד׃
3עָלַ֨י ׀ רוּחִ֗י וְאַתָּה֮ יָדַ֪עְתָּ נְֽתִיבָ֫תִ֥י בְּאֹֽרַח־ ז֥וּ אֲהַלֵּ֑ךְ טָמְנ֖וּ פַ֣ח לִֽי׃
4יָמִ֨ין ׀ וּרְאֵה֮ וְאֵֽין־ לִ֪י מַ֫כִּ֥יר אָבַ֣ד מָנ֣וֹס מִמֶּ֑נִּי אֵ֖ין דּוֹרֵ֣שׁ לְנַפְשִֽׁי׃
5אֵלֶ֗יךָ יְה֫וָ֥ה אָ֭מַרְתִּי אַתָּ֣ה מַחְסִ֑י חֶ֝לְקִ֗י בְּאֶ֣רֶץ הַֽחַיִּים׃
6אֶֽל־ רִנָּתִי֮ כִּֽי־ דַלּ֪וֹתִ֫י מְאֹ֥ד הַצִּילֵ֥נִי מֵרֹדְפַ֑י כִּ֖י אָמְצ֣וּ מִמֶּֽנִּי׃
