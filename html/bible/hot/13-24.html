1וְלִבְנֵ֥י אַהֲרֹ֖ן מַחְלְקוֹתָ֑ם בְּנֵ֣י אַהֲרֹ֔ן נָדָב֙ וַאֲבִיה֔וּא אֶלְעָזָ֖ר וְאִיתָמָֽר׃
2נָדָ֤ב וַאֲבִיהוּא֙ לִפְנֵ֣י אֲבִיהֶ֔ם וּבָנִ֖ים לֹא־ הָי֣וּ לָהֶ֑ם וַֽיְכַהֲנ֔וּ אֶלְעָזָ֖ר וְאִיתָמָֽר׃
3דָּוִ֔יד וְצָדוֹק֙ מִן־ בְּנֵ֣י אֶלְעָזָ֔ר וַאֲחִימֶ֖לֶךְ מִן־ בְּנֵ֣י אִיתָמָ֑ר לִפְקֻדָּתָ֖ם בַּעֲבֹדָתָֽם׃
4בְֽנֵי־ אֶלְעָזָ֡ר רַבִּ֞ים לְרָאשֵׁ֧י הַגְּבָרִ֛ים מִן־ בְּנֵ֥י אִיתָמָ֖ר וַֽיַּחְלְק֑וּם לִבְנֵ֨י אֶלְעָזָ֜ר רָאשִׁ֤ים לְבֵית־ אָבוֹת֙ שִׁשָּׁ֣ה עָשָׂ֔ר וְלִבְנֵ֧י אִיתָמָ֛ר לְבֵ֥ית אֲבוֹתָ֖ם שְׁמוֹנָֽה׃
5בְּגוֹרָל֖וֹת אֵ֣לֶּה עִם־ אֵ֑לֶּה כִּי־ הָי֤וּ שָֽׂרֵי־ קֹ֙דֶשׁ֙ וְשָׂרֵ֣י הָאֱלֹהִ֔ים מִבְּנֵ֥י אֶלְעָזָ֖ר וּבִבְנֵ֥י אִיתָמָֽר׃ ס
6שְֽׁמַֽעְיָה֩ בֶן־ נְתַנְאֵ֨ל הַסּוֹפֵ֜ר מִן־ הַלֵּוִ֗י לִפְנֵ֨י הַמֶּ֤לֶךְ וְהַשָּׂרִים֙ וְצָד֣וֹק הַכֹּהֵ֗ן וַאֲחִימֶ֙לֶךְ֙ בֶּן־ אֶבְיָתָ֔ר וְרָאשֵׁי֙ הָֽאָב֔וֹת לַכֹּהֲנִ֖ים וְלַלְוִיִּ֑ם בֵּֽית־ אָ֣ב אֶחָ֗ד אָחֻז֙ לְאֶלְעָזָ֔ר וְאָחֻ֥ז ׀ אָחֻ֖ז לְאִיתָמָֽר׃ פ
7הַגּוֹרָ֤ל הָרִאשׁוֹן֙ לִיה֣וֹיָרִ֔יב לִֽידַעְיָ֖ה הַשֵּׁנִֽי׃
8הַשְּׁלִישִׁ֔י לִשְׂעֹרִ֖ים הָרְבִעִֽי׃
9הַחֲמִישִׁ֔י לְמִיָּמִ֖ן הַשִּׁשִּֽׁי׃
10הַשְּׁבִעִ֔י לַאֲבִיָּ֖ה הַשְּׁמִינִֽי׃
11הַתְּשִׁעִ֔י לִשְׁכַנְיָ֖הוּ הָעֲשִׂרִֽי׃
12עַשְׁתֵּ֣י עָשָׂ֔ר לְיָקִ֖ים שְׁנֵ֥ים עָשָֽׂר׃
13שְׁלֹשָׁ֣ה עָשָׂ֔ר לְיֶֽשֶׁבְאָ֖ב אַרְבָּעָ֥ה עָשָֽׂר׃
14חֲמִשָּׁ֣ה עָשָׂ֔ר לְאִמֵּ֖ר שִׁשָּׁ֥ה עָשָֽׂר׃
15שִׁבְעָ֣ה עָשָׂ֔ר לְהַפִּצֵּ֖ץ שְׁמוֹנָ֥ה עָשָֽׂר׃
16תִּשְׁעָ֣ה עָשָׂ֔ר לִֽיחֶזְקֵ֖אל הָעֶשְׂרִֽים׃
17אֶחָ֣ד וְעֶשְׂרִ֔ים לְגָמ֖וּל שְׁנַ֥יִם וְעֶשְׂרִֽים׃
18שְׁלֹשָׁ֣ה וְעֶשְׂרִ֔ים לְמַֽעַזְיָ֖הוּ אַרְבָּעָ֥ה וְעֶשְׂרִֽים׃ פ
19פְקֻדָּתָ֞ם לַעֲבֹדָתָ֗ם לָב֤וֹא לְבֵית־ יְהוָה֙ כְּמִשְׁפָּטָ֔ם בְּיַ֖ד אַהֲרֹ֣ן אֲבִיהֶ֑ם כַּאֲשֶׁ֣ר צִוָּ֔הוּ יְהוָ֖ה אֱלֹהֵ֥י יִשְׂרָאֵֽל׃ פ
20לֵוִ֖י הַנּוֹתָרִ֑ים לִבְנֵ֤י עַמְרָם֙ שֽׁוּבָאֵ֔ל לִבְנֵ֥י שׁוּבָאֵ֖ל יֶחְדְּיָֽהוּ׃
21לִבְנֵ֣י רְחַבְיָ֔הוּ הָרֹ֖אשׁ יִשִּׁיָּֽה׃
22שְׁלֹמ֔וֹת לִבְנֵ֥י שְׁלֹמ֖וֹת יָֽחַת׃
23יְרִיָּ֑הוּ אֲמַרְיָ֙הוּ֙ הַשֵּׁנִ֔י יַחֲזִיאֵל֙ הַשְּׁלִישִׁ֔י יְקַמְעָ֖ם הָרְבִיעִֽי׃
24עֻזִּיאֵל֙ מִיכָ֔ה לִבְנֵ֥י מִיכָ֖ה [שָׁמוּר כ] (שָׁמִֽיר׃ ק)
25מִיכָה֙ יִשִּׁיָּ֔ה לִבְנֵ֥י יִשִּׁיָּ֖ה זְכַרְיָֽהוּ׃
26מְרָרִי֙ מַחְלִ֣י וּמוּשִׁ֔י בְּנֵ֖י יַעֲזִיָּ֥הֽוּ בְנֽוֹ׃
27מְרָרִ֑י לְיַֽעֲזִיָּ֣הֽוּ בְנ֔וֹ וְשֹׁ֥הַם וְזַכּ֖וּר וְעִבְרִֽי׃
28אֶלְעָזָ֔ר וְלֹא־ הָ֥יָה ל֖וֹ בָּנִֽים׃
29בְּנֵי־ קִ֥ישׁ יְרַחְמְאֵֽל׃
30מוּשִׁ֔י מַחְלִ֥י וְעֵ֖דֶר וִירִימ֑וֹת אֵ֛לֶּה בְּנֵ֥י הַלְוִיִּ֖ם לְבֵ֥ית אֲבֹתֵיהֶֽם׃
