1לַמְנַצֵּ֥חַ אֶל־ שֹׁשַׁנִּ֑ים עֵד֖וּת לְאָסָ֣ף מִזְמֽוֹר׃ רֹ֘עֵ֤ה יִשְׂרָאֵ֨ל ׀ הַאֲזִ֗ינָה נֹהֵ֣ג כַּצֹּ֣אן יוֹסֵ֑ף יֹשֵׁ֖ב הַכְּרוּבִ֣ים הוֹפִֽיעָה׃
2אֶפְרַ֨יִם ׀ וּבִנְיָ֘מִ֤ן וּמְנַשֶּׁ֗ה עוֹרְרָ֥ה אֶת־ גְּבֽוּרָתֶ֑ךָ וּלְכָ֖ה לִישֻׁעָ֣תָה לָּֽנוּ׃
3הֲשִׁיבֵ֑נוּ וְהָאֵ֥ר פָּ֝נֶ֗יךָ וְנִוָּשֵֽׁעָה׃
4אֱלֹהִ֣ים צְבָא֑וֹת עַד־ מָתַ֥י עָ֝שַׁ֗נְתָּ בִּתְפִלַּ֥ת עַמֶּֽךָ׃
5לֶ֣חֶם דִּמְעָ֑ה וַ֝תַּשְׁקֵ֗מוֹ בִּדְמָע֥וֹת שָׁלִֽישׁ׃
6מָ֭דוֹן לִשְׁכֵנֵ֑ינוּ וְ֝אֹיְבֵ֗ינוּ יִלְעֲגוּ־ לָֽמוֹ׃
7צְבָא֣וֹת הֲשִׁיבֵ֑נוּ וְהָאֵ֥ר פָּ֝נֶ֗יךָ וְנִוָּשֵֽׁעָה׃
8מִמִּצְרַ֣יִם תַּסִּ֑יעַ תְּגָרֵ֥שׁ גּ֝וֹיִ֗ם וַתִּטָּעֶֽהָ׃
9לְפָנֶ֑יהָ וַתַּשְׁרֵ֥שׁ שָׁ֝רָשֶׁ֗יהָ וַתְּמַלֵּא־ אָֽרֶץ׃
10הָרִ֣ים צִלָּ֑הּ וַ֝עֲנָפֶ֗יהָ אַֽרְזֵי־ אֵֽל׃
11קְצִירֶ֣הָ עַד־ יָ֑ם וְאֶל־ נָ֝הָ֗ר יֽוֹנְקוֹתֶֽיהָ׃
12פָּרַ֣צְתָּ גְדֵרֶ֑יהָ וְ֝אָר֗וּהָ כָּל־ עֹ֥בְרֵי דָֽרֶךְ׃
13חֲזִ֣יר מִיָּ֑ עַ seg>ר וְזִ֖יז שָׂדַ֣י יִרְעֶֽנָּה׃
14צְבָאוֹת֮ שֽׁ֫וּב־ נָ֥א הַבֵּ֣ט מִשָּׁמַ֣יִם וּרְאֵ֑ה וּ֝פְקֹ֗ד גֶּ֣פֶן זֹֽאת׃
15אֲשֶׁר־ נָטְעָ֣ה יְמִינֶ֑ךָ וְעַל־ בֵּ֝֗ן אִמַּ֥צְתָּה לָּֽךְ׃
16בָאֵ֣שׁ כְּסוּחָ֑ה מִגַּעֲרַ֖ת פָּנֶ֣יךָ יֹאבֵֽדוּ׃
17יָ֭דְךָ עַל־ אִ֣ישׁ יְמִינֶ֑ךָ עַל־ בֶּן־ אָ֝דָ֗ם אִמַּ֥צְתָּ לָּֽךְ׃
18נָס֥וֹג מִמֶּ֑ךָּ תְּ֝חַיֵּ֗נוּ וּבְשִׁמְךָ֥ נִקְרָֽא׃
