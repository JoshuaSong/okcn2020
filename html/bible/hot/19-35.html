1לְדָוִ֨ד ׀ רִיבָ֣ה יְ֭הוָה אֶת־ יְרִיבַ֑י לְ֝חַ֗ם אֶת־ לֹֽחֲמָֽי׃
2מָגֵ֣ן וְצִנָּ֑ה וְ֝ק֗וּמָה בְּעֶזְרָתִֽי׃
3חֲנִ֣ית וּ֭סְגֹר לִקְרַ֣את רֹדְפָ֑י אֱמֹ֥ר לְ֝נַפְשִׁ֗י יְֽשֻׁעָתֵ֥ךְ אָֽנִי׃
4וְיִכָּלְמוּ֮ מְבַקְשֵׁ֪י נַ֫פְשִׁ֥י יִסֹּ֣גוּ אָח֣וֹר וְיַחְפְּר֑וּ חֹ֝שְׁבֵ֗י רָעָתִֽי׃
5כְּמֹ֥ץ לִפְנֵי־ ר֑וּחַ וּמַלְאַ֖ךְ יְהוָ֣ה דּוֹחֶֽה׃
6דַרְכָּ֗ם חֹ֥שֶׁךְ וַחֲלַקְלַקּ֑וֹת וּמַלְאַ֥ךְ יְ֝הוָ֗ה רֹדְפָֽם׃
7חִנָּ֣ם טָֽמְנוּ־ לִ֭י שַׁ֣חַת רִשְׁתָּ֑ם חִ֝נָּ֗ם חָפְר֥וּ לְנַפְשִֽׁי׃
8שׁוֹאָה֮ לֹֽא־ יֵ֫דָ֥ע וְרִשְׁתּ֣וֹ אֲשֶׁר־ טָמַ֣ן תִּלְכְּד֑וֹ בְּ֝שׁוֹאָ֗ה יִפָּל־ בָּֽהּ׃
9תָּגִ֣יל בַּיהוָ֑ה תָּ֝שִׂישׂ בִּישׁוּעָתֽוֹ׃
10עַצְמוֹתַ֨י ׀ תֹּאמַרְנָה֮ יְהוָ֗ה מִ֥י כָ֫מ֥וֹךָ מַצִּ֣יל עָ֭נִי מֵחָזָ֣ק מִמֶּ֑נּוּ וְעָנִ֥י וְ֝אֶבְי֗וֹן מִגֹּזְלֽוֹ׃
11עֵדֵ֣י חָמָ֑ס אֲשֶׁ֥ר לֹא־ יָ֝דַ֗עְתִּי יִשְׁאָלֽוּנִי׃
12רָ֭עָה תַּ֥חַת טוֹבָ֗ה שְׁכ֣וֹל לְנַפְשִֽׁי׃
13בַּחֲלוֹתָ֡ם לְב֬וּשִׁי שָׂ֗ק עִנֵּ֣יתִי בַצּ֣וֹם נַפְשִׁ֑י וּ֝תְפִלָּתִ֗י עַל־ חֵיקִ֥י תָשֽׁוּב׃
14כְּאָ֣ח לִ֭י הִתְהַלָּ֑כְתִּי כַּאֲבֶל־ אֵ֝֗ם קֹדֵ֥ר שַׁחֽוֹתִי׃
15שָׂמְח֪וּ וְֽנֶאֱ֫סָ֥פוּ נֶאֶסְפ֬וּ עָלַ֣י נֵ֭כִים וְלֹ֣א יָדַ֑עְתִּי קָֽרְע֥וּ וְלֹא־ דָֽמּוּ׃
16לַעֲגֵ֣י מָע֑וֹג חָרֹ֖ק עָלַ֣י שִׁנֵּֽימוֹ׃
17כַּמָּ֪ה תִּ֫רְאֶ֥ה הָשִׁ֣יבָה נַ֭פְשִׁי מִשֹּׁאֵיהֶ֑ם מִ֝כְּפִירִ֗ים יְחִידָתִֽי׃
18בְּקָהָ֣ל רָ֑ב בְּעַ֖ם עָצ֣וּם אֲהַֽלְלֶֽךָּ׃ אַֽל־ יִשְׂמְחוּ־
19אֹיְבַ֣י שֶׁ֑קֶר שֹׂנְאַ֥י חִ֝נָּ֗ם יִקְרְצוּ־ עָֽיִן׃
20לֹ֥א שָׁל֗וֹם יְדַ֫בֵּ֥רוּ וְעַ֥ל רִגְעֵי־ אֶ֑רֶץ דִּבְרֵ֥י מִ֝רְמוֹת יַחֲשֹׁבֽוּן׃
21עָלַ֗י פִּ֫יהֶ֥ם אָ֭מְרוּ הֶאָ֣ח ׀ הֶאָ֑ח רָאֲתָ֥ה עֵינֵֽינוּ׃
22יְ֭הוָה אַֽל־ תֶּחֱרַ֑שׁ אֲ֝דֹנָ֗י אֲל־ תִּרְחַ֥ק מִמֶּֽנִּי׃
23וְ֭הָקִיצָה לְמִשְׁפָּטִ֑י אֱלֹהַ֖י וַֽאדֹנָ֣י לְרִיבִֽי׃
24כְ֭צִדְקְךָ יְהוָ֥ה אֱלֹהָ֗י וְאַל־ יִשְׂמְחוּ־ לִֽי׃
25יֹאמְר֣וּ בְ֭לִבָּם הֶאָ֣ח נַפְשֵׁ֑נוּ אַל־ יֹ֝אמְר֗וּ בִּֽלַּעֲנֽוּהוּ׃
26וְיַחְפְּר֨וּ ׀ יַחְדָּו֮ שְׂמֵחֵ֪י רָעָ֫תִ֥י יִֽלְבְּשׁוּ־ בֹ֥שֶׁת וּכְלִמָּ֑ה הַֽמַּגְדִּילִ֥ים עָלָֽי׃
27וְיִשְׂמְחוּ֮ חֲפֵצֵ֪י צִ֫דְקִ֥י וְיֹאמְר֣וּ תָ֭מִיד יִגְדַּ֣ל יְהוָ֑ה הֶ֝חָפֵ֗ץ שְׁל֣וֹם עַבְדּֽוֹ׃
