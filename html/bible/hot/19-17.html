1תְּפִלָּ֗ה לְדָ֫וִ֥ד שִׁמְעָ֤ה יְהוָ֨ה ׀ צֶ֗דֶק הַקְשִׁ֥יבָה רִנָּתִ֗י הַאֲזִ֥ינָה תְפִלָּתִ֑י בְּ֝לֹ֗א שִׂפְתֵ֥י מִרְמָֽה׃
2מִשְׁפָּטִ֣י יֵצֵ֑א עֵ֝ינֶ֗יךָ תֶּחֱזֶ֥ינָה מֵישָׁרִֽים׃
3לִבִּ֨י ׀ פָּ֘קַ֤דְתָּ לַּ֗יְלָה צְרַפְתַּ֥נִי בַל־ תִּמְצָ֑א זַ֝מֹּתִ֗י בַּל־ יַעֲבָר־ פִּֽי׃
4אָ֭דָם בִּדְבַ֣ר שְׂפָתֶ֑יךָ אֲנִ֥י שָׁ֝מַ֗רְתִּי אָרְח֥וֹת פָּרִֽיץ׃
5אֲ֭שֻׁרַי בְּמַעְגְּלוֹתֶ֑יךָ בַּל־ נָמ֥וֹטּוּ פְעָמָֽי׃
6קְרָאתִ֣יךָ כִֽי־ תַעֲנֵ֣נִי אֵ֑ל הַֽט־ אָזְנְךָ֥ לִ֝֗י שְׁמַ֣ע אִמְרָתִֽי׃
7חֲ֭סָדֶיךָ מוֹשִׁ֣יעַ חוֹסִ֑ים מִ֝מִּתְקוֹמְמִ֗ים בִּֽימִינֶֽךָ׃
8כְּאִישׁ֣וֹן בַּת־ עָ֑יִן בְּצֵ֥ל כְּ֝נָפֶ֗יךָ תַּסְתִּירֵֽנִי׃
9רְ֭שָׁעִים ז֣וּ שַׁדּ֑וּנִי אֹיְבַ֥י בְּ֝נֶ֗פֶשׁ יַקִּ֥יפוּ עָלָֽי׃
10סָּגְר֑וּ פִּ֝֗ימוֹ דִּבְּר֥וּ בְגֵאֽוּת׃
11עַתָּ֣ה [סְבָבוּנִי כ] (סְבָב֑וּנוּ ק) עֵינֵיהֶ֥ם יָ֝שִׁ֗יתוּ לִנְט֥וֹת בָּאָֽרֶץ׃
12כְּ֭אַרְיֵה יִכְס֣וֹף לִטְר֑וֹף וְ֝כִכְפִ֗יר יֹשֵׁ֥ב בְּמִסְתָּרִֽים׃
13יְהוָ֗ה קַדְּמָ֣ה פָ֭נָיו הַכְרִיעֵ֑הוּ פַּלְּטָ֥ה נַ֝פְשִׁ֗י מֵרָשָׁ֥ע חַרְבֶּֽךָ׃
14יָדְךָ֨ ׀ יְהוָ֡ה מִֽמְתִ֬ים מֵחֶ֗לֶד חֶלְקָ֥ם בַּֽחַיִּים֮ [וּצְפִינְךָ כ] (וּֽצְפוּנְךָ֮ ק) תְּמַלֵּ֪א בִ֫טְנָ֥ם יִשְׂבְּע֥וּ בָנִ֑ים וְהִנִּ֥יחוּ יִ֝תְרָ֗ם לְעוֹלְלֵיהֶֽם׃
