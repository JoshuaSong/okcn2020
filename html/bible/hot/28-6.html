1לְכוּ֙ וְנָשׁ֣וּבָה אֶל־ יְהוָ֔ה כִּ֛י ה֥וּא טָרָ֖ף וְיִרְפָּאֵ֑נוּ יַ֖ךְ וְיַחְבְּשֵֽׁנוּ׃
2מִיֹּמָ֑יִם בַּיּוֹם֙ הַשְּׁלִישִׁ֔י יְקִמֵ֖נוּ וְנִחְיֶ֥ה לְפָנָֽיו׃
3נִרְדְּפָ֗ה לָדַ֙עַת֙ אֶת־ יְהוָ֔ה כְּשַׁ֖חַר נָכ֣וֹן מֽוֹצָא֑וֹ וְיָב֤וֹא כַגֶּ֙שֶׁם֙ לָ֔נוּ כְּמַלְק֖וֹשׁ י֥וֹרֶה אָֽרֶץ׃
4אֶֽעֱשֶׂה־ לְּךָ֙ אֶפְרַ֔יִם מָ֥ה אֶעֱשֶׂה־ לְּךָ֖ יְהוּדָ֑ה וְחַסְדְּכֶם֙ כַּֽעֲנַן־ בֹּ֔קֶר וְכַטַּ֖ל מַשְׁכִּ֥ים הֹלֵֽךְ׃
5כֵּ֗ן חָצַ֙בְתִּי֙ בַּנְּבִיאִ֔ים הֲרַגְתִּ֖ים בְּאִמְרֵי־ פִ֑י וּמִשְׁפָּטֶ֖יךָ א֥וֹר יֵצֵֽא׃
6חֶ֥סֶד חָפַ֖צְתִּי וְלֹא־ זָ֑בַח וְדַ֥עַת אֱלֹהִ֖ים מֵעֹלֽוֹת׃
7כְּאָדָ֖ם עָבְר֣וּ בְרִ֑ית שָׁ֖ם בָּ֥גְדוּ בִֽי׃
8קִרְיַ֖ת פֹּ֣עֲלֵי אָ֑וֶן עֲקֻבָּ֖ה מִדָּֽם׃
9אִ֜ישׁ גְּדוּדִ֗ים חֶ֚בֶר כֹּֽהֲנִ֔ים דֶּ֖רֶךְ יְרַצְּחוּ־ שֶׁ֑כְמָה כִּ֥י זִמָּ֖ה עָשֽׂוּ׃
10יִשְׂרָאֵ֔ל רָאִ֖יתִי [שַׁעֲרִירִיָּה כ] (שַׁעֲרֽוּרִיָּ֑ה ק) שָׁ֚ם זְנ֣וּת לְאֶפְרַ֔יִם נִטְמָ֖א יִשְׂרָאֵֽל׃
