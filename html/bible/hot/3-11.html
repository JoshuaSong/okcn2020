2אֶל־ בְּנֵ֥י יִשְׂרָאֵ֖ל לֵאמֹ֑ר זֹ֤את הַֽחַיָּה֙ אֲשֶׁ֣ר תֹּאכְל֔וּ מִכָּל־ הַבְּהֵמָ֖ה אֲשֶׁ֥ר עַל־ הָאָֽרֶץ׃
3מַפְרֶ֣סֶת פַּרְסָ֗ה וְשֹׁסַ֤עַת שֶׁ֙סַע֙ פְּרָסֹ֔ת מַעֲלַ֥ת גֵּרָ֖ה בַּבְּהֵמָ֑ה אֹתָ֖הּ תֹּאכֵֽלוּ׃
4אֶת־ זֶה֙ לֹ֣א תֹֽאכְל֔וּ מִֽמַּעֲלֵי֙ הַגֵּרָ֔ה וּמִמַּפְרִיסֵ֖י הַפַּרְסָ֑ה אֶֽת־ הַ֠גָּמָל כִּֽי־ מַעֲלֵ֨ה גֵרָ֜ה ה֗וּא וּפַרְסָה֙ אֵינֶ֣נּוּ מַפְרִ֔יס טָמֵ֥א ה֖וּא לָכֶֽם׃
5הַשָּׁפָ֗ן כִּֽי־ מַעֲלֵ֤ה גֵרָה֙ ה֔וּא וּפַרְסָ֖ה לֹ֣א יַפְרִ֑יס טָמֵ֥א ה֖וּא לָכֶֽם׃
6הָאַרְנֶ֗בֶת כִּֽי־ מַעֲלַ֤ת גֵּרָה֙ הִ֔וא וּפַרְסָ֖ה לֹ֣א הִפְרִ֑יסָה טְמֵאָ֥ה הִ֖וא לָכֶֽם׃
7הַ֠חֲזִיר כִּֽי־ מַפְרִ֨יס פַּרְסָ֜ה ה֗וּא וְשֹׁסַ֥ע שֶׁ֙סַע֙ פַּרְסָ֔ה וְה֖וּא גֵּרָ֣ה לֹֽא־ יִגָּ֑ר טָמֵ֥א ה֖וּא לָכֶֽם׃
8לֹ֣א תֹאכֵ֔לוּ וּבְנִבְלָתָ֖ם לֹ֣א תִגָּ֑עוּ טְמֵאִ֥ים הֵ֖ם לָכֶֽם׃
9זֶה֙ תֹּֽאכְל֔וּ מִכֹּ֖ל אֲשֶׁ֣ר בַּמָּ֑יִם כֹּ֣ל אֲשֶׁר־ לוֹ֩ סְנַפִּ֨יר וְקַשְׂקֶ֜שֶׂת בַּמַּ֗יִם בַּיַּמִּ֛ים וּבַנְּחָלִ֖ים אֹתָ֥ם תֹּאכֵֽלוּ׃
10אֲשֶׁ֨ר אֵֽין־ ל֜וֹ סְנַפִּ֣יר וְקַשְׂקֶ֗שֶׂת בַּיַּמִּים֙ וּבַנְּחָלִ֔ים מִכֹּל֙ שֶׁ֣רֶץ הַמַּ֔יִם וּמִכֹּ֛ל נֶ֥פֶשׁ הַחַיָּ֖ה אֲשֶׁ֣ר בַּמָּ֑יִם שֶׁ֥קֶץ הֵ֖ם לָכֶֽם׃
11יִהְי֣וּ לָכֶ֑ם מִבְּשָׂרָם֙ לֹ֣א תֹאכֵ֔לוּ וְאֶת־ נִבְלָתָ֖ם תְּשַׁקֵּֽצוּ׃
12אֲשֶׁ֥ר אֵֽין־ ל֛וֹ סְנַפִּ֥יר וְקַשְׂקֶ֖שֶׂת בַּמָּ֑יִם שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃
13אֵ֙לֶּה֙ תְּשַׁקְּצ֣וּ מִן־ הָע֔וֹף לֹ֥א יֵאָכְל֖וּ שֶׁ֣קֶץ הֵ֑ם אֶת־ הַנֶּ֙שֶׁר֙ וְאֶת־ הַפֶּ֔רֶס וְאֵ֖ת הָעָזְנִיָּֽה׃
14הַ֨דָּאָ֔ה וְאֶת־ הָאַיָּ֖ה לְמִינָֽהּ׃

16בַּ֣ת הַֽיַּעֲנָ֔ה וְאֶת־ הַתַּחְמָ֖ס וְאֶת־ הַשָּׁ֑חַף וְאֶת־ הַנֵּ֖ץ לְמִינֵֽהוּ׃
17הַכּ֥וֹס וְאֶת־ הַשָּׁלָ֖ךְ וְאֶת־ הַיַּנְשֽׁוּף׃
18הַתִּנְשֶׁ֥מֶת וְאֶת־ הַקָּאָ֖ת וְאֶת־ הָרָחָֽם׃
19הַחֲסִידָ֔ה הָאֲנָפָ֖ה לְמִינָ֑הּ וְאֶת־ הַדּוּכִיפַ֖ת וְאֶת־ הָעֲטַלֵּֽף׃
20שֶׁ֣רֶץ הָע֔וֹף הַהֹלֵ֖ךְ עַל־ אַרְבַּ֑ע שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃ ס
21אֶת־ זֶה֙ תֹּֽאכְל֔וּ מִכֹּל֙ שֶׁ֣רֶץ הָע֔וֹף הַהֹלֵ֖ךְ עַל־ אַרְבַּ֑ע אֲשֶׁר־ [לֹא כ] (לֹ֤ו ק) כְרָעַ֙יִם֙ מִמַּ֣עַל לְרַגְלָ֔יו לְנַתֵּ֥ר בָּהֵ֖ן עַל־ הָאָֽרֶץ׃
22אֵ֤לֶּה מֵהֶם֙ תֹּאכֵ֔לוּ אֶת־ הָֽאַרְבֶּ֣ה לְמִינ֔וֹ וְאֶת־ הַסָּלְעָ֖ם לְמִינֵ֑הוּ וְאֶת־ הַחַרְגֹּ֣ל לְמִינֵ֔הוּ וְאֶת־ הֶחָגָ֖ב לְמִינֵֽהוּ׃
23שֶׁ֣רֶץ הָע֔וֹף אֲשֶׁר־ ל֖וֹ אַרְבַּ֣ע רַגְלָ֑יִם שֶׁ֥קֶץ ה֖וּא לָכֶֽם׃
24תִּטַּמָּ֑אוּ כָּל־ הַנֹּגֵ֥עַ בְּנִבְלָתָ֖ם יִטְמָ֥א עַד־ הָעָֽרֶב׃
25הַנֹּשֵׂ֖א מִנִּבְלָתָ֑ם יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֥א עַד־ הָעָֽרֶב׃
26הַבְּהֵמָ֡ה אֲשֶׁ֣ר הִוא֩ מַפְרֶ֨סֶת פַּרְסָ֜ה וְשֶׁ֣סַע ׀ אֵינֶ֣נָּה שֹׁסַ֗עַת וְגֵרָה֙ אֵינֶ֣נָּה מַעֲלָ֔ה טְמֵאִ֥ים הֵ֖ם לָכֶ֑ם כָּל־ הַנֹּגֵ֥עַ בָּהֶ֖ם יִטְמָֽא׃
27הוֹלֵ֣ךְ עַל־ כַּפָּ֗יו בְּכָל־ הַֽחַיָּה֙ הַהֹלֶ֣כֶת עַל־ אַרְבַּ֔ע טְמֵאִ֥ים הֵ֖ם לָכֶ֑ם כָּל־ הַנֹּגֵ֥עַ בְּנִבְלָתָ֖ם יִטְמָ֥א עַד־ הָעָֽרֶב׃
28אֶת־ נִבְלָתָ֔ם יְכַבֵּ֥ס בְּגָדָ֖יו וְטָמֵ֣א עַד־ הָעָ֑רֶב טְמֵאִ֥ים הֵ֖מָּה לָכֶֽם׃ ס
29לָכֶם֙ הַטָּמֵ֔א בַּשֶּׁ֖רֶץ הַשֹּׁרֵ֣ץ עַל־ הָאָ֑רֶץ הַחֹ֥לֶד וְהָעַכְבָּ֖ר וְהַצָּ֥ב לְמִינֵֽהוּ׃
30וְהַכֹּ֖חַ וְהַלְּטָאָ֑ה וְהַחֹ֖מֶט וְהַתִּנְשָֽׁמֶת׃
31הַטְּמֵאִ֥ים לָכֶ֖ם בְּכָל־ הַשָּׁ֑רֶץ כָּל־ הַנֹּגֵ֧עַ בָּהֶ֛ם בְּמֹתָ֖ם יִטְמָ֥א עַד־ הָעָֽרֶב׃
32אֲשֶׁר־ יִפֹּל־ עָלָיו֩ מֵהֶ֨ם ׀ בְּמֹתָ֜ם יִטְמָ֗א מִכָּל־ כְּלִי־ עֵץ֙ א֣וֹ בֶ֤גֶד אוֹ־ עוֹר֙ א֣וֹ שָׂ֔ק כָּל־ כְּלִ֕י אֲשֶׁר־ יֵעָשֶׂ֥ה מְלָאכָ֖ה בָּהֶ֑ם בַּמַּ֧יִם יוּבָ֛א וְטָמֵ֥א עַד־ הָעֶ֖רֶב וְטָהֵֽר׃

35אֲשֶׁר־ יִפֹּ֨ל מִנִּבְלָתָ֥ם ׀ עָלָיו֮ יִטְמָא֒ תַּנּ֧וּר וְכִירַ֛יִם יֻתָּ֖ץ טְמֵאִ֣ים הֵ֑ם וּטְמֵאִ֖ים יִהְי֥וּ לָכֶֽם׃
