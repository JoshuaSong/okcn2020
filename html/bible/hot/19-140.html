1לַמְנַצֵּ֗חַ מִזְמ֥וֹר לְדָוִֽד׃ חַלְּצֵ֣נִי יְ֭הוָה מֵאָדָ֣ם רָ֑ע מֵאִ֖ישׁ חֲמָסִ֣ים תִּנְצְרֵֽנִי׃
2חָשְׁב֣וּ רָע֣וֹת בְּלֵ֑ב כָּל־ י֝֗וֹם יָג֥וּרוּ מִלְחָמֽוֹת׃
3לְשׁוֹנָם֮ כְּֽמוֹ־ נָ֫חָ֥שׁ חֲמַ֥ת עַכְשׁ֑וּב תַּ֖חַת שְׂפָתֵ֣ימוֹ סֶֽלָה׃
4יְהוָ֨ה ׀ מִ֘ידֵ֤י רָשָׁ֗ע מֵאִ֣ישׁ חֲמָסִ֣ים תִּנְצְרֵ֑נִי אֲשֶׁ֥ר חָ֝שְׁב֗וּ לִדְח֥וֹת פְּעָמָֽי׃
5גֵאִ֨ים ׀ פַּ֡ח לִ֗י וַחֲבָלִ֗ים פָּ֣רְשׂוּ רֶ֭שֶׁת לְיַד־ מַעְגָּ֑ל מֹקְשִׁ֖ים שָֽׁתוּ־ לִ֣י סֶֽלָה׃
6לַ֭יהוָה אֵ֣לִי אָ֑תָּה הַאֲזִ֥ינָה יְ֝הוָ֗ה ק֣וֹל תַּחֲנוּנָֽי׃
7אֲ֭דֹנָי עֹ֣ז יְשׁוּעָתִ֑י סַכֹּ֥תָה לְ֝רֹאשִׁ֗י בְּי֣וֹם נָֽשֶׁק׃
8תִּתֵּ֣ן יְ֭הוָה מַאֲוַיֵּ֣י רָשָׁ֑ע זְמָמ֥וֹ אַל־ תָּ֝פֵ֗ק יָר֥וּמוּ סֶֽלָה׃
9מְסִבָּ֑י עֲמַ֖ל שְׂפָתֵ֣ימוֹ [יְכַסּוּמֹו כ] (יְכַסֵּֽמֹו ק)
10כ] (יִמֹּ֥וטוּ ק) עֲלֵיהֶ֗ם גֶּֽחָ֫לִ֥ים בָּאֵ֥שׁ יַפִּלֵ֑ם בְּ֝מַהֲמֹר֗וֹת בַּֽל־ יָקֽוּמוּ׃
11לָשׁוֹן֮ בַּל־ יִכּ֪וֹן בָּ֫אָ֥רֶץ אִישׁ־ חָמָ֥ס רָ֑ע יְ֝צוּדֶ֗נּוּ לְמַדְחֵפֹֽת׃
12כ] (יָדַ֗עְתִּי ק) כִּֽי־ יַעֲשֶׂ֣ה יְ֭הוָה דִּ֣ין עָנִ֑י מִ֝שְׁפַּ֗ט אֶבְיֹנִֽים׃
