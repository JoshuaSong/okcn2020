1לַמְנַצֵּ֣חַ אַל־ תַּ֭שְׁחֵת לְדָוִ֣ד מִכְתָּ֑ם בְּבָרְח֥וֹ מִפְּנֵי־ שָׁ֝א֗וּל בַּמְּעָרָֽה׃ חָנֵּ֤נִי אֱלֹהִ֨ים ׀ חָנֵּ֗נִי כִּ֥י בְךָ֮ חָסָ֪יָה נַ֫פְשִׁ֥י וּבְצֵֽל־ כְּנָפֶ֥יךָ אֶחְסֶ֑ה עַ֝֗ד יַעֲבֹ֥ר הַוּֽוֹת׃
2לֵֽאלֹהִ֣ים עֶלְי֑וֹן לָ֝אֵ֗ל גֹּמֵ֥ר עָלָֽי׃
3מִשָּׁמַ֨יִם ׀ וְֽיוֹשִׁיעֵ֗נִי חֵרֵ֣ף שֹׁאֲפִ֣י סֶ֑לָה יִשְׁלַ֥ח אֱ֝לֹהִ֗ים חַסְדּ֥וֹ וַאֲמִתּֽוֹ׃
4בְּת֥וֹךְ לְבָאִם֮ אֶשְׁכְּבָ֪ה לֹ֫הֲטִ֥ים בְּֽנֵי־ אָדָ֗ם שִׁ֭נֵּיהֶם חֲנִ֣ית וְחִצִּ֑ים וּ֝לְשׁוֹנָ֗ם חֶ֣רֶב חַדָּֽה׃
5עַל־ הַשָּׁמַ֣יִם אֱלֹהִ֑ים עַ֖ל כָּל־ הָאָ֣רֶץ כְּבוֹדֶֽךָ׃
6הֵכִ֣ינוּ לִפְעָמַי֮ כָּפַ֪ף נַ֫פְשִׁ֥י כָּר֣וּ לְפָנַ֣י שִׁיחָ֑ה נָפְל֖וּ בְתוֹכָ֣הּ סֶֽלָה׃
7לִבִּ֣י אֱ֭לֹהִים נָכ֣וֹן לִבִּ֑י אָ֝שִׁ֗ירָה וַאֲזַמֵּֽרָה׃
8כְבוֹדִ֗י ע֭וּרָֽה הַנֵּ֥בֶל וְכִנּ֗וֹר אָעִ֥ירָה שָּֽׁחַר׃
9בָעַמִּ֥ים ׀ אֲדֹנָ֑י אֲ֝זַמֶּרְךָ֗ בַּל־ אֻמִּֽים׃
10גָדֹ֣ל עַד־ שָׁמַ֣יִם חַסְדֶּ֑ךָ וְֽעַד־ שְׁחָקִ֥ים אֲמִתֶּֽךָ׃
