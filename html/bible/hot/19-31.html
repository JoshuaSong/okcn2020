1לַמְנַצֵּ֗חַ מִזְמ֥וֹר לְדָוִֽד׃ בְּךָ֖ יְהוָ֣ה חָ֭סִיתִי אַל־ אֵב֣וֹשָׁה לְעוֹלָ֑ם בְּצִדְקָתְךָ֥ פַלְּטֵֽנִי׃
2אֵלַ֨י ׀ אָזְנְךָ֮ מְהֵרָ֪ה הַצִּ֫ילֵ֥נִי הֱיֵ֤ה לִ֨י ׀ לְֽצוּר־ מָ֭עוֹז לְבֵ֥ית מְצוּד֗וֹת לְהוֹשִׁיעֵֽנִי׃
3סַלְעִ֣י וּמְצוּדָתִ֣י אָ֑תָּה וּלְמַ֥עַן שִׁ֝מְךָ֗ תַּֽנְחֵ֥נִי וּֽתְנַהֲלֵֽנִי׃
4מֵרֶ֣שֶׁת ז֭וּ טָ֣מְנוּ לִ֑י כִּֽי־ אַ֝תָּה מָֽעוּזִּֽי׃
5אַפְקִ֪יד ר֫וּחִ֥י פָּדִ֖יתָה אוֹתִ֥י יְהוָ֗ה אֵ֣ל אֱמֶֽת׃
6הַשֹּׁמְרִ֥ים הַבְלֵי־ שָׁ֑וְא וַ֝אֲנִ֗י אֶל־ יְהוָ֥ה בָּטָֽחְתִּי׃
7וְאֶשְׂמְחָ֗ה בְּחַ֫סְדֶּ֥ךָ אֲשֶׁ֣ר רָ֭אִיתָ אֶת־ עָנְיִ֑י יָ֝דַ֗עְתָּ בְּצָר֥וֹת נַפְשִֽׁי׃
8הִ֭סְגַּרְתַּנִי בְּיַד־ אוֹיֵ֑ב הֶֽעֱמַ֖דְתָּ בַמֶּרְחָ֣ב רַגְלָֽי׃
9יְהוָה֮ כִּ֤י צַ֫ר־ לִ֥י עָשְׁשָׁ֖ה בְכַ֥עַס עֵינִ֗י נַפְשִׁ֥י וּבִטְנִֽי׃
10כָל֪וּ בְיָג֡וֹן חַיַּי֮ וּשְׁנוֹתַ֪י בַּאֲנָ֫חָ֥ה כָּשַׁ֣ל בַּעֲוֹנִ֣י כֹחִ֑י וַעֲצָמַ֥י עָשֵֽׁשׁוּ׃
11צֹרְרַ֨י הָיִ֪יתִי חֶרְפָּ֡ה וְלִשֲׁכֵנַ֨י ׀ מְאֹד֮ וּפַ֪חַד לִֽמְיֻדָּ֫עָ֥י רֹאַ֥י בַּח֑וּץ נָדְד֥וּ מִמֶּֽנִּי׃
12כְּמֵ֣ת מִלֵּ֑ב הָ֝יִ֗יתִי כִּכְלִ֥י אֹבֵֽד׃
13שָׁמַ֨עְתִּי ׀ דִּבַּ֥ת רַבִּים֮ מָג֪וֹר מִסָּ֫בִ֥יב בְּהִוָּסְדָ֣ם יַ֣חַד עָלַ֑י לָקַ֖חַת נַפְשִׁ֣י זָמָֽמוּ׃
14עָלֶ֣יךָ בָטַ֣חְתִּי יְהוָ֑ה אָ֝מַ֗רְתִּי אֱלֹהַ֥י אָֽתָּה׃
15עִתֹּתָ֑י הַצִּ֘ילֵ֤נִי מִיַּד־ א֝וֹיְבַ֗י וּמֵרֹדְפָֽי׃
16פָ֭נֶיךָ עַל־ עַבְדֶּ֑ךָ ה֖וֹשִׁיעֵ֣נִי בְחַסְדֶּֽךָ׃
17אַל־ אֵ֭בוֹשָׁה כִּ֣י קְרָאתִ֑יךָ יֵבֹ֥שׁוּ רְ֝שָׁעִ֗ים יִדְּמ֥וּ לִשְׁאֽוֹל׃
18שִׂפְתֵ֫י שָׁ֥קֶר הַדֹּבְר֖וֹת עַל־ צַדִּ֥יק עָתָ֗ק בְּגַאֲוָ֥ה וָבֽוּז׃
19רַֽב־ טוּבְךָ֮ אֲשֶׁר־ צָפַ֪נְתָּ לִּֽירֵ֫אֶ֥יךָ פָּ֭עַלְתָּ לַחֹסִ֣ים בָּ֑ךְ נֶ֝֗גֶד בְּנֵ֣י אָדָם׃
20בְּסֵ֥תֶר פָּנֶיךָ֮ מֵֽרֻכְסֵ֫י אִ֥ישׁ תִּצְפְּנֵ֥ם בְּסֻכָּ֗ה מֵרִ֥יב לְשֹׁנֽוֹת׃
21יְהוָ֑ה כִּ֥י הִפְלִ֘יא חַסְדּ֥וֹ לִ֝֗י בְּעִ֣יר מָצֽוֹר׃
22אָ֘מַ֤רְתִּי בְחָפְזִ֗י נִגְרַזְתִּי֮ מִנֶּ֪גֶד עֵ֫ינֶ֥יךָ אָכֵ֗ן שָׁ֭מַעְתָּ ק֥וֹל תַּחֲנוּנַ֗י בְּשַׁוְּעִ֥י אֵלֶֽיךָ׃
23אֶת־ יְהוָ֗ה כָּֽל־ חֲסִ֫ידָ֥יו אֱ֭מוּנִים נֹצֵ֣ר יְהוָ֑ה וּמְשַׁלֵּ֥ם עַל־ יֶ֝֗תֶר עֹשֵׂ֥ה גַאֲוָֽה׃
