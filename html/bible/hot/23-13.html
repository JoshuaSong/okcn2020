1מַשָּׂ֖א בָּבֶ֑ל אֲשֶׁ֣ר חָזָ֔ה יְשַׁעְיָ֖הוּ בֶּן־ אָמֽוֹץ׃
2הַר־ נִשְׁפֶּה֙ שְֽׂאוּ־ נֵ֔ס הָרִ֥ימוּ ק֖וֹל לָהֶ֑ם הָנִ֣יפוּ יָ֔ד וְיָבֹ֖אוּ פִּתְחֵ֥י נְדִיבִֽים׃
3צִוֵּ֖יתִי לִמְקֻדָּשָׁ֑י גַּ֣ם קָרָ֤אתִי גִבּוֹרַי֙ לְאַפִּ֔י עַלִּיזֵ֖י גַּאֲוָתִֽי׃
4הָמ֛וֹן בֶּֽהָרִ֖ים דְּמ֣וּת עַם־ רָ֑ב ק֠וֹל שְׁא֞וֹן מַמְלְכ֤וֹת גּוֹיִם֙ נֶֽאֱסָפִ֔ים יְהוָ֣ה צְבָא֔וֹת מְפַקֵּ֖ד צְבָ֥א מִלְחָמָֽה׃
5מֵאֶ֥רֶץ מֶרְחָ֖ק מִקְצֵ֣ה הַשָּׁמָ֑יִם יְהוָה֙ וּכְלֵ֣י זַעְמ֔וֹ לְחַבֵּ֖ל כָּל־ הָאָֽרֶץ׃
6כִּ֥י קָר֖וֹב י֣וֹם יְהוָ֑ה כְּשֹׁ֖ד מִשַּׁדַּ֥י יָבֽוֹא׃
7כֵּ֖ן כָּל־ יָדַ֣יִם תִּרְפֶּ֑ינָה וְכָל־ לְבַ֥ב אֱנ֖וֹשׁ יִמָּֽס׃
8צִירִ֤ים וַֽחֲבָלִים֙ יֹֽאחֵז֔וּן כַּיּוֹלֵדָ֖ה יְחִיל֑וּן אִ֤ישׁ אֶל־ רֵעֵ֙הוּ֙ יִתְמָ֔הוּ פְּנֵ֥י לְהָבִ֖ים פְּנֵיהֶֽם׃
9יוֹם־ יְהוָה֙ בָּ֔א אַכְזָרִ֥י וְעֶבְרָ֖ה וַחֲר֣וֹן אָ֑ף לָשׂ֤וּם הָאָ֙רֶץ֙ לְשַׁמָּ֔ה וְחַטָּאֶ֖יהָ יַשְׁמִ֥יד מִמֶּֽנָּה׃
10כוֹכְבֵ֤י הַשָּׁמַ֙יִם֙ וּכְסִ֣ילֵיהֶ֔ם לֹ֥א יָהֵ֖לּוּ אוֹרָ֑ם חָשַׁ֤ךְ הַשֶּׁ֙מֶשׁ֙ בְּצֵאת֔וֹ וְיָרֵ֖חַ לֹֽא־ יַגִּ֥יהַ אוֹרֽוֹ׃
11עַל־ תֵּבֵל֙ רָעָ֔ה וְעַל־ רְשָׁעִ֖ים עֲוֹנָ֑ם וְהִשְׁבַּתִּי֙ גְּא֣וֹן זֵדִ֔ים וְגַאֲוַ֥ת עָרִיצִ֖ים אַשְׁפִּֽיל׃
12אֱנ֖וֹשׁ מִפָּ֑ז וְאָדָ֖ם מִכֶּ֥תֶם אוֹפִֽיר׃
13כֵּן֙ שָׁמַ֣יִם אַרְגִּ֔יז וְתִרְעַ֥שׁ הָאָ֖רֶץ מִמְּקוֹמָ֑הּ בְּעֶבְרַת֙ יְהוָ֣ה צְבָא֔וֹת וּבְי֖וֹם חֲר֥וֹן אַפּֽוֹ׃
14כִּצְבִ֣י מֻדָּ֔ח וּכְצֹ֖אן וְאֵ֣ין מְקַבֵּ֑ץ אִ֤ישׁ אֶל־ עַמּוֹ֙ יִפְנ֔וּ וְאִ֥ישׁ אֶל־ אַרְצ֖וֹ יָנֽוּסוּ׃
15הַנִּמְצָ֖א יִדָּקֵ֑ר וְכָל־ הַנִּסְפֶּ֖ה יִפּ֥וֹל בֶּחָֽרֶב׃
16יְרֻטְּשׁ֖וּ לְעֵֽינֵיהֶ֑ם יִשַּׁ֙סּוּ֙ בָּֽתֵּיהֶ֔ם וּנְשֵׁיהֶ֖ם [תִּשָּׁגַלְנָה כ] (תִּשָּׁכַֽבְנָה׃ ק)
17מֵעִ֥יר עֲלֵיהֶ֖ם אֶת־ מָדָ֑י אֲשֶׁר־ כֶּ֙סֶף֙ לֹ֣א יַחְשֹׁ֔בוּ וְזָהָ֖ב לֹ֥א יַחְפְּצוּ־ בֽוֹ׃
18נְעָרִ֣ים תְּרַטַּ֑שְׁנָה וּפְרִי־ בֶ֙טֶן֙ לֹ֣א יְרַחֵ֔מוּ עַל־ בָּנִ֖ים לֹֽא־ תָח֥וּס עֵינָֽם׃
19בָבֶל֙ צְבִ֣י מַמְלָכ֔וֹת תִּפְאֶ֖רֶת גְּא֣וֹן כַּשְׂדִּ֑ים כְּמַהְפֵּכַ֣ת אֱלֹהִ֔ים אֶת־ סְדֹ֖ם וְאֶת־ עֲמֹרָֽה׃
20תֵשֵׁ֣ב לָנֶ֔צַח וְלֹ֥א תִשְׁכֹּ֖ן עַד־ דּ֣וֹר וָד֑וֹר וְלֹֽא־ יַהֵ֥ל שָׁם֙ עֲרָבִ֔י וְרֹעִ֖ים לֹא־ יַרְבִּ֥צוּ שָֽׁם׃
21שָׁ֣ם צִיִּ֔ים וּמָלְא֥וּ בָתֵּיהֶ֖ם אֹחִ֑ים וְשָׁ֤כְנוּ שָׁם֙ בְּנ֣וֹת יַֽעֲנָ֔ה וּשְׂעִירִ֖ים יְרַקְּדוּ־ שָֽׁם׃
