1וַֽיִּגְּשׁ֗וּ רָאשֵׁי֙ אֲב֣וֹת הַלְוִיִּ֔ם אֶל־ אֶלְעָזָר֙ הַכֹּהֵ֔ן וְאֶל־ יְהוֹשֻׁ֖עַ בִּן־ נ֑וּן וְאֶל־ רָאשֵׁ֛י אֲב֥וֹת הַמַּטּ֖וֹת לִבְנֵ֥י יִשְׂרָאֵֽל׃
2אֲלֵיהֶ֜ם בְּשִׁלֹ֗ה בְּאֶ֤רֶץ כְּנַ֙עַן֙ לֵאמֹ֔ר יְהוָה֙ צִוָּ֣ה בְיַד־ מֹשֶׁ֔ה לָֽתֶת־ לָ֥נוּ עָרִ֖ים לָשָׁ֑בֶת וּמִגְרְשֵׁיהֶ֖ן לִבְהֶמְתֵּֽנוּ׃
3בְנֵי־ יִשְׂרָאֵ֧ל לַלְוִיִּ֛ם מִנַּחֲלָתָ֖ם אֶל־ פִּ֣י יְהוָ֑ה אֶת־ הֶעָרִ֥ים הָאֵ֖לֶּה וְאֶת־ מִגְרְשֵׁיהֶֽן׃
4הַגּוֹרָ֖ל לְמִשְׁפְּחֹ֣ת הַקְּהָתִ֑י וַיְהִ֡י לִבְנֵי֩ אַהֲרֹ֨ן הַכֹּהֵ֜ן מִן־ הַלְוִיִּ֗ם מִמַּטֵּ֣ה יְ֠הוּדָה וּמִמַּטֵּ֨ה הַשִּׁמְעֹנִ֜י וּמִמַּטֵּ֤ה בִנְיָמִן֙ בַּגּוֹרָ֔ל עָרִ֖ים שְׁלֹ֥שׁ עֶשְׂרֵֽה׃ ס
5קְהָ֜ת הַנּוֹתָרִ֗ים מִמִּשְׁפְּחֹ֣ת מַטֵּֽה־ אֶ֠פְרַיִם וּֽמִמַּטֵּה־ דָ֞ן וּמֵחֲצִ֨י מַטֵּ֧ה מְנַשֶּׁ֛ה בַּגּוֹרָ֖ל עָרִ֥ים עָֽשֶׂר׃ ס
6גֵרְשׁ֗וֹן מִמִּשְׁפְּח֣וֹת מַטֵּֽה־ יִשָּׂשכָ֣ר וּמִמַּטֵּֽה־ אָ֠שֵׁר וּמִמַּטֵּ֨ה נַפְתָּלִ֜י וּ֠מֵחֲצִי מַטֵּ֨ה מְנַשֶּׁ֤ה בַבָּשָׁן֙ בַּגּוֹרָ֔ל עָרִ֖ים שְׁלֹ֥שׁ עֶשְׂרֵֽה׃ ס
7מְרָרִ֜י לְמִשְׁפְּחֹתָ֗ם מִמַּטֵּ֨ה רְאוּבֵ֤ן וּמִמַּטֵּה־ גָד֙ וּמִמַּטֵּ֣ה זְבוּלֻ֔ן עָרִ֖ים שְׁתֵּ֥ים עֶשְׂרֵֽה׃
8בְנֵֽי־ יִשְׂרָאֵל֙ לַלְוִיִּ֔ם אֶת־ הֶעָרִ֥ים הָאֵ֖לֶּה וְאֶת־ מִגְרְשֵׁיהֶ֑ן כַּאֲשֶׁ֨ר צִוָּ֧ה יְהוָ֛ה בְּיַד־ מֹשֶׁ֖ה בַּגּוֹרָֽל׃ פ
9מִמַּטֵּה֙ בְּנֵ֣י יְהוּדָ֔ה וּמִמַּטֵּ֖ה בְּנֵ֣י שִׁמְע֑וֹן אֵ֚ת הֶֽעָרִ֣ים הָאֵ֔לֶּה אֲשֶׁר־ יִקְרָ֥א אֶתְהֶ֖ן בְּשֵֽׁם׃
10לִבְנֵ֣י אַהֲרֹ֔ן מִמִּשְׁפְּח֥וֹת הַקְּהָתִ֖י מִבְּנֵ֣י לֵוִ֑י כִּ֥י לָהֶ֛ם הָיָ֥ה הַגּוֹרָ֖ל רִיאשֹׁנָֽה׃
11לָהֶ֜ם אֶת־ קִרְיַת֩ אַרְבַּ֨ע אֲבִ֧י הָֽעֲנ֛וֹק הִ֥יא חֶבְר֖וֹן בְּהַ֣ר יְהוּדָ֑ה וְאֶת־ מִגְרָשֶׁ֖הָ סְבִיבֹתֶֽיהָ׃
12שְׂדֵ֥ה הָעִ֖יר וְאֶת־ חֲצֵרֶ֑יהָ נָֽתְנ֛וּ לְכָלֵ֥ב בֶּן־ יְפֻנֶּ֖ה בַּאֲחֻזָּתֽוֹ׃ ס
13אַהֲרֹ֣ן הַכֹּהֵ֗ן נָֽתְנוּ֙ אֶת־ עִיר֙ מִקְלַ֣ט הָרֹצֵ֔חַ אֶת־ חֶבְר֖וֹן וְאֶת־ מִגְרָשֶׁ֑הָ וְאֶת־ לִבְנָ֖ה וְאֶת־ מִגְרָשֶֽׁהָ׃
14יַתִּר֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶת־ אֶשְׁתְּמֹ֖עַ וְאֶת־ מִגְרָשֶֽׁהָ׃
15חֹלֹן֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶת־ דְּבִ֖ר וְאֶת־ מִגְרָשֶֽׁהָ׃
16עַ֣יִן וְאֶת־ מִגְרָשֶׁ֗הָ וְאֶת־ יֻטָּה֙ וְאֶת־ מִגְרָשֶׁ֔הָ אֶת־ בֵּ֥ית שֶׁ֖מֶשׁ וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֣ים תֵּ֔שַׁע מֵאֵ֕ת שְׁנֵ֥י הַשְּׁבָטִ֖ים הָאֵֽלֶּה׃ פ
17בִנְיָמִ֔ן אֶת־ גִּבְע֖וֹן וְאֶת־ מִגְרָשֶׁ֑הָ אֶת־ גֶּ֖בַע וְאֶת־ מִגְרָשֶֽׁהָ׃
18עֲנָתוֹת֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶת־ עַלְמ֖וֹן וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים אַרְבַּֽע׃
19עָרֵ֥י בְנֵֽי־ אַהֲרֹ֖ן הַכֹּֽהֲנִ֑ים שְׁלֹשׁ־ עֶשְׂרֵ֥ה עָרִ֖ים וּמִגְרְשֵׁיהֶֽן׃ ס
20בְּנֵֽי־ קְהָת֙ הַלְוִיִּ֔ם הַנּוֹתָרִ֖ים מִבְּנֵ֣י קְהָ֑ת וַֽיְהִי֙ עָרֵ֣י גֽוֹרָלָ֔ם מִמַּטֵּ֖ה אֶפְרָֽיִם׃
21לָהֶ֜ם אֶת־ עִ֨יר מִקְלַ֧ט הָרֹצֵ֛חַ אֶת־ שְׁכֶ֥ם וְאֶת־ מִגְרָשֶׁ֖הָ בְּהַ֣ר אֶפְרָ֑יִם וְאֶת־ גֶּ֖זֶר וְאֶת־ מִגְרָשֶֽׁהָ׃
22קִבְצַ֙יִם֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶת־ בֵּ֥ית חוֹרֹ֖ן וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים אַרְבַּֽע׃ ס
23דָ֔ן אֶֽת־ אֶלְתְּקֵ֖א וְאֶת־ מִגְרָשֶׁ֑הָ אֶֽת־ גִּבְּת֖וֹן וְאֶת־ מִגְרָשֶֽׁהָ׃
24אַיָּלוֹן֙ וְאֶת־ מִגְרָשֶׁ֔הָ אֶת־ גַּת־ רִמּ֖וֹן וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים אַרְבַּֽע׃ ס
25מַטֵּ֣ה מְנַשֶּׁ֔ה אֶת־ תַּעְנַךְ֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶת־ גַּת־ רִמּ֖וֹן וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים שְׁתָּֽיִם׃
26עָרִ֥ים עֶ֖שֶׂר וּמִגְרְשֵׁיהֶ֑ן לְמִשְׁפְּח֥וֹת בְּנֵֽי־ קְהָ֖ת הַנּוֹתָרִֽים׃ ס
27גֵרְשׁוֹן֮ מִמִּשְׁפְּחֹ֣ת הַלְוִיִּם֒ מֵחֲצִ֞י מַטֵּ֣ה מְנַשֶּׁ֗ה אֶת־ עִיר֙ מִקְלַ֣ט הָרֹצֵ֔חַ אֶת־ [גָּלֹון כ] (גֹּולָ֤ן ק) בַּבָּשָׁן֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶֽת־ בְּעֶשְׁתְּרָ֖ה וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים שְׁתָּֽיִם׃ ס
28יִשָּׂשכָ֔ר אֶת־ קִשְׁי֖וֹן וְאֶת־ מִגְרָשֶׁ֑הָ אֶת־ דָּֽבְרַ֖ת וְאֶת־ מִגְרָשֶֽׁהָ׃
29יַרְמוּת֙ וְאֶת־ מִגְרָשֶׁ֔הָ אֶת־ עֵ֥ין גַּנִּ֖ים וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים אַרְבַּֽע׃ ס
30אָשֵׁ֔ר אֶת־ מִשְׁאָ֖ל וְאֶת־ מִגְרָשֶׁ֑הָ אֶת־ עַבְדּ֖וֹן וְאֶת־ מִגְרָשֶֽׁהָ׃
31חֶלְקָת֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶת־ רְחֹ֖ב וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים אַרְבַּֽע׃ ס
32נַפְתָּלִ֜י אֶת־ עִ֣יר ׀ מִקְלַ֣ט הָֽרֹצֵ֗חַ אֶת־ קֶ֨דֶשׁ בַּגָּלִ֤יל וְאֶת־ מִגְרָשֶׁ֙הָ֙ וְאֶת־ חַמֹּ֥ת דֹּאר֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶת־ קַרְתָּ֖ן וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים שָׁלֹֽשׁ׃
33עָרֵ֥י הַגֵּרְשֻׁנִּ֖י לְמִשְׁפְּחֹתָ֑ם שְׁלֹשׁ־ עֶשְׂרֵ֥ה עִ֖יר וּמִגְרְשֵׁיהֶֽן׃ ס
34בְּנֵֽי־ מְרָרִי֮ הַלְוִיִּ֣ם הַנּוֹתָרִים֒ מֵאֵת֙ מַטֵּ֣ה זְבוּלֻ֔ן אֶֽת־ יָקְנְעָ֖ם וְאֶת־ מִגְרָשֶׁ֑הָ אֶת־ קַרְתָּ֖ה וְאֶת־ מִגְרָשֶֽׁהָ׃
35דִּמְנָה֙ וְאֶת־ מִגְרָשֶׁ֔הָ אֶֽת־ נַהֲלָ֖ל וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים אַרְבַּֽע׃
36רְאוּבֵ֔ן אֶת־ בֶּ֖צֶר וְאֶת־ מִגְרָשֶׁ֑הָ וְאֶת־ יַ֖הְצָה וְאֶת־ מִגְרָשֶֽׁהָ׃
37קְדֵמוֹת֙ וְאֶת־ מִגְרָשֶׁ֔הָ וְאֶת־ מֵיפָ֖עַת וְאֶת־ מִגְרָשֶׁ֑הָ עָרִ֖ים אַרְבַּֽע׃ ס
38גָ֗ד אֶת־ עִיר֙ מִקְלַ֣ט הָרֹצֵ֔חַ אֶת־ רָמֹ֥ת בַּגִּלְעָ֖ד וְאֶת־ מִגְרָשֶׁ֑הָ וְאֶֽת־ מַחֲנַ֖יִם וְאֶת־ מִגְרָשֶֽׁהָ׃
39חֶשְׁבּוֹן֙ וְאֶת־ מִגְרָשֶׁ֔הָ אֶת־ יַעְזֵ֖ר וְאֶת־ מִגְרָשֶׁ֑הָ כָּל־ עָרִ֖ים אַרְבַּֽע׃
40הֶ֨עָרִ֜ים לִבְנֵ֤י מְרָרִי֙ לְמִשְׁפְּחֹתָ֔ם הַנּוֹתָרִ֖ים מִמִּשְׁפְּח֣וֹת הַלְוִיִּ֑ם וַיְהִי֙ גּוֹרָלָ֔ם עָרִ֖ים שְׁתֵּ֥ים עֶשְׂרֵֽה׃
41עָרֵ֣י הַלְוִיִּ֔ם בְּת֖וֹךְ אֲחֻזַּ֣ת בְּנֵֽי־ יִשְׂרָאֵ֑ל עָרִ֛ים אַרְבָּעִ֥ים וּשְׁמֹנֶ֖ה וּמִגְרְשֵׁיהֶֽן׃
42הֶעָרִ֣ים הָאֵ֔לֶּה עִ֣יר עִ֔יר וּמִגְרָשֶׁ֖יהָ סְבִיבֹתֶ֑יהָ כֵּ֖ן לְכָל־ הֶעָרִ֥ים הָאֵֽלֶּה׃ ס
43יְהוָה֙ לְיִשְׂרָאֵ֔ל אֶת־ כָּל־ הָאָ֔רֶץ אֲשֶׁ֥ר נִשְׁבַּ֖ע לָתֵ֣ת לַאֲבוֹתָ֑ם וַיִּרָשׁ֖וּהָ וַיֵּ֥שְׁבוּ בָֽהּ׃
