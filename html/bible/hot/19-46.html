1לַמְנַצֵּ֥חַ לִבְנֵי־ קֹ֑רַח עַֽל־ עֲלָמ֥וֹת שִֽׁיר׃ אֱלֹהִ֣ים לָ֭נוּ מַחֲסֶ֣ה וָעֹ֑ז עֶזְרָ֥ה בְ֝צָר֗וֹת נִמְצָ֥א מְאֹֽד׃
2כֵּ֣ן לֹא־ נִ֭ירָא בְּהָמִ֣יר אָ֑רֶץ וּבְמ֥וֹט הָ֝רִ֗ים בְּלֵ֣ב יַמִּֽים׃
3יֶחְמְר֣וּ מֵימָ֑יו יִֽרְעֲשֽׁוּ־ הָרִ֖ים בְּגַאֲוָת֣וֹ סֶֽלָה׃
4פְּלָגָ֗יו יְשַׂמְּח֥וּ עִיר־ אֱלֹהִ֑ים קְ֝דֹ֗שׁ מִשְׁכְּנֵ֥י עֶלְיֽוֹן׃
5בְּ֭קִרְבָּהּ בַּל־ תִּמּ֑וֹט יַעְזְרֶ֥הָ אֱ֝לֹהִ֗ים לִפְנ֥וֹת בֹּֽקֶר׃
6ג֭וֹיִם מָ֣טוּ מַמְלָכ֑וֹת נָתַ֥ן בְּ֝קוֹל֗וֹ תָּמ֥וּג אָֽרֶץ׃
7צְבָא֣וֹת עִמָּ֑נוּ מִשְׂגָּֽב־ לָ֝נוּ אֱלֹהֵ֖י יַעֲקֹ֣ב סֶֽלָה׃
8חֲ֭זוּ מִפְעֲל֣וֹת יְהוָ֑ה אֲשֶׁר־ שָׂ֖ם שַׁמּ֣וֹת בָּאָֽרֶץ׃
9מִלְחָמוֹת֮ עַד־ קְצֵ֪ה הָ֫אָ֥רֶץ קֶ֣שֶׁת יְ֭שַׁבֵּר וְקִצֵּ֣ץ חֲנִ֑ית עֲ֝גָל֗וֹת יִשְׂרֹ֥ף בָּאֵֽשׁ׃
10וּ֭דְעוּ כִּי־ אָנֹכִ֣י אֱלֹהִ֑ים אָר֥וּם בַּ֝גּוֹיִ֗ם אָר֥וּם בָּאָֽרֶץ׃
