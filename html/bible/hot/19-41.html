1לַמְנַצֵּ֗חַ מִזְמ֥וֹר לְדָוִֽד׃ אַ֭שְׁרֵי מַשְׂכִּ֣יל אֶל־ דָּ֑ל בְּי֥וֹם רָ֝עָ֗ה יְֽמַלְּטֵ֥הוּ יְהוָֽה׃
2יִשְׁמְרֵ֣הוּ וִֽ֭יחַיֵּהוּ [יֶאְשֹּׁר כ] (וְאֻשַּׁ֣ר ק) בָּאָ֑רֶץ וְאַֽל־ תִּ֝תְּנֵ֗הוּ בְּנֶ֣פֶשׁ אֹיְבָֽיו׃
3יִ֭סְעָדֶנּוּ עַל־ עֶ֣רֶשׂ דְּוָ֑י כָּל־ מִ֝שְׁכָּב֗וֹ הָפַ֥כְתָּ בְחָלְיֽוֹ׃
4אָ֭מַרְתִּי יְהוָ֣ה חָנֵּ֑נִי רְפָאָ֥ה נַ֝פְשִׁ֗י כִּי־ חָטָ֥אתִי לָֽךְ׃
5יֹאמְר֣וּ רַ֣ע לִ֑י מָתַ֥י יָ֝מ֗וּת וְאָבַ֥ד שְׁמֽוֹ׃
6בָּ֤א לִרְא֨וֹת ׀ שָׁ֤וְא יְדַבֵּ֗ר לִבּ֗וֹ יִקְבָּץ־ אָ֥וֶן ל֑וֹ יֵצֵ֖א לַח֣וּץ יְדַבֵּֽר׃
7עָלַ֣י יִ֭תְלַחֲשׁוּ כָּל־ שֹׂנְאָ֑י עָלַ֓י ׀ יַחְשְׁב֖וּ רָעָ֣ה לִֽי׃
8בְּ֭לִיַּעַל יָצ֣וּק בּ֑וֹ וַאֲשֶׁ֥ר שָׁ֝כַ֗ב לֹא־ יוֹסִ֥יף לָקֽוּם׃
9אִ֤ישׁ שְׁלוֹמִ֨י ׀ אֲשֶׁר־ בָּטַ֣חְתִּי ב֭וֹ אוֹכֵ֣ל לַחְמִ֑י הִגְדִּ֖יל עָלַ֣י עָקֵֽב׃
10יְהוָ֗ה חָנֵּ֥נִי וַהֲקִימֵ֑נִי וַֽאֲשַׁלְּמָ֥ה לָהֶֽם׃
11יָ֭דַעְתִּי כִּֽי־ חָפַ֣צְתָּ בִּ֑י כִּ֤י לֹֽא־ יָרִ֖יעַ אֹיְבִ֣י עָלָֽי׃
12בְּ֭תֻמִּי תָּמַ֣כְתָּ בִּ֑י וַתַּצִּיבֵ֖נִי לְפָנֶ֣יךָ לְעוֹלָֽם׃
