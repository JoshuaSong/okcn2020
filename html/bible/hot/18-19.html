1וַיַּ֥עַן אִיּ֗וֹב וַיֹּאמַֽר׃
2אָ֭נָה תּוֹגְי֣וּן נַפְשִׁ֑י וּֽתְדַכְּאוּנַ֥נִי בְמִלִּֽים׃
3עֶ֣שֶׂר פְּ֭עָמִים תַּכְלִימ֑וּנִי לֹֽא־ תֵ֝בֹ֗שׁוּ תַּהְכְּרוּ־ לִֽי׃
4אָמְנָ֥ם שָׁגִ֑יתִי אִ֝תִּ֗י תָּלִ֥ין מְשׁוּגָתִֽי׃
5אָ֭מְנָם עָלַ֣י תַּגְדִּ֑ילוּ וְתוֹכִ֥יחוּ עָ֝לַ֗י חֶרְפָּתִּֽי׃
6אֵ֭פוֹ כִּי־ אֱל֣וֹהַּ עִוְּתָ֑נִי וּ֝מְצוּד֗וֹ עָלַ֥י הִקִּֽיף׃
7אֶצְעַ֣ק חָ֭מָס וְלֹ֣א אֵעָנֶ֑ה אֲ֝שַׁוַּ֗ע וְאֵ֣ין מִשְׁפָּֽט׃
8גָ֭דַר וְלֹ֣א אֶעֱב֑וֹר וְעַ֥ל נְ֝תִיבוֹתַ֗י חֹ֣שֶׁךְ יָשִֽׂים׃
9מֵעָלַ֣י הִפְשִׁ֑יט וַ֝יָּ֗סַר עֲטֶ֣רֶת רֹאשִֽׁי׃
10סָ֭בִיב וָאֵלַ֑ךְ וַיַּסַּ֥ע כָּ֝עֵ֗ץ תִּקְוָתִֽי׃
11עָלַ֣י אַפּ֑וֹ וַיַּחְשְׁבֵ֖נִי ל֣וֹ כְצָרָֽיו׃
12יָ֘בֹ֤אוּ גְדוּדָ֗יו וַיָּסֹ֣לּוּ עָלַ֣י דַּרְכָּ֑ם וַיַּחֲנ֖וּ סָבִ֣יב לְאָהֳלִֽי׃
13מֵעָלַ֣י הִרְחִ֑יק וְ֝יֹדְעַ֗י אַךְ־ זָ֥רוּ מִמֶּֽנִּי׃
14קְרוֹבָ֑י וּֽמְיֻדָּעַ֥י שְׁכֵחֽוּנִי׃
15בֵיתִ֣י וְ֭אַמְהֹתַי לְזָ֣ר תַּחְשְׁבֻ֑נִי נָ֝כְרִ֗י הָיִ֥יתִי בְעֵינֵיהֶֽם׃
16קָ֭רָאתִי וְלֹ֣א יַעֲנֶ֑ה בְּמוֹ־ פִ֝֗י אֶתְחַנֶּן־ לֽוֹ׃
17זָ֣רָה לְאִשְׁתִּ֑י וְ֝חַנֹּתִ֗י לִבְנֵ֥י בִטְנִֽי׃
18עֲ֭וִילִים מָ֣אֲסוּ בִ֑י אָ֝ק֗וּמָה וַיְדַבְּרוּ־ בִֽי׃
19כָּל־ מְתֵ֣י סוֹדִ֑י וְזֶֽה־ אָ֝הַ֗בְתִּי נֶהְפְּכוּ־ בִֽי׃
20וּ֭בִבְשָׂרִי דָּבְקָ֣ה עַצְמִ֑י וָ֝אֶתְמַלְּטָ֗ה בְּע֣וֹר שִׁנָּֽי׃
21חָנֻּ֣נִי אַתֶּ֣ם רֵעָ֑י כִּ֥י יַד־ אֱ֝ל֗וֹהַּ נָ֣גְעָה בִּֽי׃
22תִּרְדְּפֻ֣נִי כְמוֹ־ אֵ֑ל וּ֝מִבְּשָׂרִ֗י לֹ֣א תִשְׂבָּֽעוּ׃
23יִתֵּ֣ן אֵ֭פוֹ וְיִכָּתְב֣וּן מִלָּ֑י מִֽי־ יִתֵּ֖ן בַּסֵּ֣פֶר וְיֻחָֽקוּ׃
24בַּרְזֶ֥ל וְעֹפָ֑רֶת לָ֝עַ֗ד בַּצּ֥וּר יֵחָצְבֽוּן׃
25יָ֭דַעְתִּי גֹּ֣אֲלִי חָ֑י וְ֝אַחֲר֗וֹן עַל־ עָפָ֥ר יָקֽוּם׃
26ע֭וֹרִֽי נִקְּפוּ־ זֹ֑את וּ֝מִבְּשָׂרִ֗י אֶֽחֱזֶ֥ה אֱלֽוֹהַּ׃
27אֲנִ֨י ׀ אֶֽחֱזֶה־ לִּ֗י וְעֵינַ֣י רָא֣וּ וְלֹא־ זָ֑ר כָּל֖וּ כִלְיֹתַ֣י בְּחֵקִֽי׃
28תֹ֭אמְרוּ מַה־ נִּרְדָּף־ ל֑וֹ וְשֹׁ֥רֶשׁ דָּ֝בָ֗ר נִמְצָא־ בִֽי׃
