1וְעַ֖ל הַחֲתוּמִ֑ים נְחֶמְיָ֧ה הַתִּרְשָׁ֛תָא בֶּן־ חֲכַלְיָ֖ה וְצִדְקִיָּֽה׃
2עֲזַרְיָ֖ה יִרְמְיָֽה׃
3אֲמַרְיָ֖ה מַלְכִּיָּֽה׃
4שְׁבַנְיָ֖ה מַלּֽוּךְ׃
5מְרֵמ֖וֹת עֹֽבַדְיָֽה׃
6גִּנְּת֖וֹן בָּרֽוּךְ׃
7אֲבִיָּ֖ה מִיָּמִֽן׃
8בִלְגַּ֖י שְׁמַֽעְיָ֑ה אֵ֖לֶּה הַכֹּהֲנִֽים׃ ס
9וְיֵשׁ֙וּעַ֙ בֶּן־ אֲזַנְיָ֔ה בִּנּ֕וּי מִבְּנֵ֥י חֵנָדָ֖ד קַדְמִיאֵֽל׃
10שְׁבַנְיָ֧ה הֽוֹדִיָּ֛ה קְלִיטָ֖א פְּלָאיָ֥ה חָנָֽן׃
11רְח֖וֹב חֲשַׁבְיָֽה׃
12שֵׁרֵֽבְיָ֖ה שְׁבַנְיָֽה׃
13בָנִ֖י בְּנִֽינוּ׃ ס
14הָעָ֑ם פַּרְעֹשׁ֙ פַּחַ֣ת מוֹאָ֔ב עֵילָ֥ם זַתּ֖וּא בָּנִֽי׃
15עַזְגָּ֖ד בֵּבָֽי׃
16בִגְוַ֖י עָדִֽין׃
17חִזְקִיָּ֖ה עַזּֽוּר׃
18חָשֻׁ֖ם בֵּצָֽי׃
19עֲנָת֖וֹת [נֹובָי כ] (נֵיבָֽי׃ ק)
20מְשֻׁלָּ֖ם חֵזִֽיר׃
21צָד֖וֹק יַדּֽוּעַ׃
22חָנָ֖ן עֲנָיָֽה׃
23חֲנַנְיָ֖ה חַשּֽׁוּב׃
24פִּלְחָ֖א שׁוֹבֵֽק׃
25חֲשַׁבְנָ֖ה מַעֲשֵׂיָֽה׃
26חָנָ֖ן עָנָֽן׃
27חָרִ֖ם בַּעֲנָֽה׃
28הָעָ֡ם הַכֹּהֲנִ֣ים הַ֠לְוִיִּם הַשּׁוֹעֲרִ֨ים הַמְשֹׁרְרִ֜ים הַנְּתִינִ֗ים וְֽכָל־ הַנִּבְדָּ֞ל מֵעַמֵּ֤י הָאֲרָצוֹת֙ אֶל־ תּוֹרַ֣ת הָאֱלֹהִ֔ים נְשֵׁיהֶ֖ם בְּנֵיהֶ֣ם וּבְנֹתֵיהֶ֑ם כֹּ֖ל יוֹדֵ֥עַ מֵבִֽין׃
29עַל־ אֲחֵיהֶם֮ אַדִּירֵיהֶם֒ וּבָאִ֞ים בְּאָלָ֣ה וּבִשְׁבוּעָ֗ה לָלֶ֙כֶת֙ בְּתוֹרַ֣ת הָאֱלֹהִ֔ים אֲשֶׁ֣ר נִתְּנָ֔ה בְּיַ֖ד מֹשֶׁ֣ה עֶֽבֶד־ הָֽאֱלֹהִ֑ים וְלִשְׁמ֣וֹר וְלַעֲשׂ֗וֹת אֶת־ כָּל־ מִצְוֹת֙ יְהוָ֣ה אֲדֹנֵ֔ינוּ וּמִשְׁפָּטָ֖יו וְחֻקָּֽיו׃
30לֹא־ נִתֵּ֥ן בְּנֹתֵ֖ינוּ לְעַמֵּ֣י הָאָ֑רֶץ וְאֶת־ בְּנֹ֣תֵיהֶ֔ם לֹ֥א נִקַּ֖ח לְבָנֵֽינוּ׃
31הָאָ֡רֶץ הַֽמְבִיאִים֩ אֶת־ הַמַּקָּח֨וֹת וְכָל־ שֶׁ֜בֶר בְּי֤וֹם הַשַּׁבָּת֙ לִמְכּ֔וֹר לֹא־ נִקַּ֥ח מֵהֶ֛ם בַּשַּׁבָּ֖ת וּבְי֣וֹם קֹ֑דֶשׁ וְנִטֹּ֛שׁ אֶת־ הַשָּׁנָ֥ה הַשְּׁבִיעִ֖ית וּמַשָּׁ֥א כָל־ יָֽד׃
32עָלֵ֙ינוּ֙ מִצְוֹ֔ת לָתֵ֥ת עָלֵ֛ינוּ שְׁלִשִׁ֥ית הַשֶּׁ֖קֶל בַּשָּׁנָ֑ה לַעֲבֹדַ֖ת בֵּ֥ית אֱלֹהֵֽינוּ׃
33הַֽמַּעֲרֶ֡כֶת וּמִנְחַ֣ת הַתָּמִ֣יד וּלְעוֹלַ֣ת הַ֠תָּמִיד הַשַּׁבָּת֨וֹת הֶחֳדָשִׁ֜ים לַמּוֹעֲדִ֗ים וְלַקֳּדָשִׁים֙ וְלַ֣חַטָּא֔וֹת לְכַפֵּ֖ר עַל־ יִשְׂרָאֵ֑ל וְכֹ֖ל מְלֶ֥אכֶת בֵּית־ אֱלֹהֵֽינוּ׃ ס
34הִפַּ֜לְנוּ עַל־ קֻרְבַּ֣ן הָעֵצִ֗ים הַכֹּהֲנִ֣ים הַלְוִיִּם֮ וְהָעָם֒ לְ֠הָבִיא לְבֵ֨ית אֱלֹהֵ֧ינוּ לְבֵית־ אֲבֹתֵ֛ינוּ לְעִתִּ֥ים מְזֻמָּנִ֖ים שָׁנָ֣ה בְשָׁנָ֑ה לְבַעֵ֗ר עַל־ מִזְבַּח֙ יְהוָ֣ה אֱלֹהֵ֔ינוּ כַּכָּת֖וּב בַּתּוֹרָֽה׃
35אֶת־ בִּכּוּרֵ֣י אַדְמָתֵ֗נוּ וּבִכּוּרֵ֛י כָּל־ פְּרִ֥י כָל־ עֵ֖ץ שָׁנָ֣ה בְשָׁנָ֑ה לְבֵ֖ית יְהוָֽה׃
36בְּכֹר֤וֹת בָּנֵ֙ינוּ֙ וּבְהֶמְתֵּ֔ינוּ כַּכָּת֖וּב בַּתּוֹרָ֑ה וְאֶת־ בְּכוֹרֵ֨י בְקָרֵ֜ינוּ וְצֹאנֵ֗ינוּ לְהָבִיא֙ לְבֵ֣ית אֱלֹהֵ֔ינוּ לַכֹּ֣הֲנִ֔ים הַמְשָׁרְתִ֖ים בְּבֵ֥ית אֱלֹהֵֽינוּ׃
37רֵאשִׁ֣ית עֲרִיסֹתֵ֣ינוּ וּ֠תְרוּמֹתֵינוּ וּפְרִ֨י כָל־ עֵ֜ץ תִּיר֣וֹשׁ וְיִצְהָ֗ר נָבִ֤יא לַכֹּהֲנִים֙ אֶל־ לִשְׁכ֣וֹת בֵּית־ אֱלֹהֵ֔ינוּ וּמַעְשַׂ֥ר אַדְמָתֵ֖נוּ לַלְוִיִּ֑ם וְהֵם֙ הַלְוִיִּ֔ם הַֽמְעַשְּׂרִ֔ים בְּכֹ֖ל עָרֵ֥י עֲבֹדָתֵֽנוּ׃
38הַכֹּהֵ֧ן בֶּֽן־ אַהֲרֹ֛ן עִם־ הַלְוִיִּ֖ם בַּעְשֵׂ֣ר הַלְוִיִּ֑ם וְהַלְוִיִּ֞ם יַעֲל֨וּ אֶת־ מַעֲשַׂ֤ר הַֽמַּעֲשֵׂר֙ לְבֵ֣ית אֱלֹהֵ֔ינוּ אֶל־ הַלְּשָׁכ֖וֹת לְבֵ֥ית הָאוֹצָֽר׃
