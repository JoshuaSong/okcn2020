1וַיַּ֥עַן אֱלִיה֗וּ וַיֹּאמַֽר׃
2חָשַׁ֣בְתָּ לְמִשְׁפָּ֑ט אָ֝מַ֗רְתָּ צִדְקִ֥י מֵאֵֽל׃
3תֹ֭אמַר מַה־ יִּסְכָּן־ לָ֑ךְ מָֽה־ אֹ֝עִ֗יל מֵֽחַטָּאתִֽי׃
4אֲשִֽׁיבְךָ֣ מִלִּ֑ין וְֽאֶת־ רֵעֶ֥יךָ עִמָּֽךְ׃
5שָׁמַ֣יִם וּרְאֵ֑ה וְשׁ֥וּר שְׁ֝חָקִ֗ים גָּבְה֥וּ מִמֶּֽךָּ׃
6חָ֭טָאתָ מַה־ תִּפְעָל־ בּ֑וֹ וְרַבּ֥וּ פְ֝שָׁעֶ֗יךָ מַה־ תַּעֲשֶׂה־ לּֽוֹ׃
7צָ֭דַקְתָּ מַה־ תִּתֶּן־ ל֑וֹ א֥וֹ מַה־ מִיָּדְךָ֥ יִקָּֽח׃
8כָּמ֥וֹךָ רִשְׁעֶ֑ךָ וּלְבֶן־ אָ֝דָ֗ם צִדְקָתֶֽךָ׃
9עֲשׁוּקִ֣ים יַזְעִ֑יקוּ יְשַׁוְּע֖וּ מִזְּר֣וֹעַ רַבִּֽים׃
10אָמַ֗ר אַ֭יֵּה אֱל֣וֹהַּ עֹשָׂ֑י נֹתֵ֖ן זְמִר֣וֹת בַּלָּֽיְלָה׃
11מִבַּהֲמ֣וֹת אָ֑רֶץ וּמֵע֖וֹף הַשָּׁמַ֣יִם יְחַכְּמֵֽנוּ׃
12יִ֭צְעֲקוּ וְלֹ֣א יַעֲנֶ֑ה מִ֝פְּנֵ֗י גְּא֣וֹן רָעִֽים׃
13שָׁ֭וְא לֹא־ יִשְׁמַ֥ע ׀ אֵ֑ל וְ֝שַׁדַּ֗י לֹ֣א יְשׁוּרֶֽנָּה׃
14כִּֽי־ תֹ֭אמַר לֹ֣א תְשׁוּרֶ֑נּוּ דִּ֥ין לְ֝פָנָ֗יו וּתְח֥וֹלֵֽל לֽוֹ׃
15כִּי־ אַ֭יִן פָּקַ֣ד אַפּ֑וֹ וְלֹֽא־ יָדַ֖ע בַּפַּ֣שׁ מְאֹֽד׃
