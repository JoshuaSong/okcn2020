1פַּלְגֵי־ מַ֣יִם לֶב־ מֶ֭לֶךְ בְּיַד־ יְהוָ֑ה עַֽל־ כָּל־ אֲשֶׁ֖ר יַחְפֹּ֣ץ יַטֶּֽנּוּ׃ כָּֽל־ דֶּרֶךְ־
2יָשָׁ֣ר בְּעֵינָ֑יו וְתֹכֵ֖ן לִבּ֣וֹת יְהוָֽה׃
3צְדָקָ֣ה וּמִשְׁפָּ֑ט נִבְחָ֖ר לַיהוָ֣ה מִזָּֽבַח׃
4עֵ֭ינַיִם וּרְחַב־ לֵ֑ב נִ֖ר רְשָׁעִ֣ים חַטָּֽאת׃
5חָ֭רוּץ אַךְ־ לְמוֹתָ֑ר וְכָל־ אָ֝֗ץ אַךְ־ לְמַחְסֽוֹר׃
6א֭וֹצָרוֹת בִּלְשׁ֣וֹן שָׁ֑קֶר הֶ֥בֶל נִ֝דָּ֗ף מְבַקְשֵׁי־ מָֽוֶת׃
7רְשָׁעִ֥ים יְגוֹרֵ֑ם כִּ֥י מֵ֝אֲנ֗וּ לַעֲשׂ֥וֹת מִשְׁפָּֽט׃
8דֶּ֣רֶךְ אִ֣ישׁ וָזָ֑ר וְ֝זַ֗ךְ יָשָׁ֥ר פָּעֳלֽוֹ׃
9לָשֶׁ֥בֶת עַל־ פִּנַּת־ גָּ֑ג מֵאֵ֥שֶׁת מִ֝דְיָנִ֗ים וּבֵ֥ית חָֽבֶר׃
10רָ֭שָׁע אִוְּתָה־ רָ֑ע לֹא־ יֻחַ֖ן בְּעֵינָ֣יו רֵעֵֽהוּ׃
11לֵ֭ץ יֶחְכַּם־ פֶּ֑תִי וּבְהַשְׂכִּ֥יל לְ֝חָכָ֗ם יִקַּח־ דָּֽעַת׃
12צַ֭דִּיק לְבֵ֣ית רָשָׁ֑ע מְסַלֵּ֖ף רְשָׁעִ֣ים לָרָֽע׃
13אָ֭זְנוֹ מִזַּעֲקַת־ דָּ֑ל גַּֽם־ ה֥וּא יִ֝קְרָ֗א וְלֹ֣א יֵעָנֶֽה׃
14בַּ֭סֵּתֶר יִכְפֶּה־ אָ֑ף וְשֹׁ֥חַד בַּ֝חֵ֗ק חֵמָ֥ה עַזָּֽה׃
15לַ֭צַּדִּיק עֲשׂ֣וֹת מִשְׁפָּ֑ט וּ֝מְחִתָּ֗ה לְפֹ֣עֲלֵי אָֽוֶן׃
16תּ֭וֹעֶה מִדֶּ֣רֶךְ הַשְׂכֵּ֑ל בִּקְהַ֖ל רְפָאִ֣ים יָנֽוּחַ׃
17מַ֭חְסוֹר אֹהֵ֣ב שִׂמְחָ֑ה אֹהֵ֥ב יַֽיִן־ וָ֝שֶׁ֗מֶן לֹ֣א יַעֲשִֽׁיר׃
18לַצַּדִּ֣יק רָשָׁ֑ע וְתַ֖חַת יְשָׁרִ֣ים בּוֹגֵֽד׃
19שֶׁ֥בֶת בְּאֶֽרֶץ־ מִדְבָּ֑ר מֵאֵ֖שֶׁת [מְדֹונִים כ] (מִדְיָנִ֣ים ק) וָכָֽעַס׃
20נֶחְמָ֣ד וָ֭שֶׁמֶן בִּנְוֵ֣ה חָכָ֑ם וּכְסִ֖יל אָדָ֣ם יְבַלְּעֶֽנּוּ׃
21צְדָקָ֣ה וָחָ֑סֶד יִמְצָ֥א חַ֝יִּ֗ים צְדָקָ֥ה וְכָבֽוֹד׃
22גִּ֭בֹּרִים עָלָ֣ה חָכָ֑ם וַ֝יֹּ֗רֶד עֹ֣ז מִבְטֶחָֽה׃
23פִּ֭יו וּלְשׁוֹנ֑וֹ שֹׁמֵ֖ר מִצָּר֣וֹת נַפְשֽׁוֹ׃
24יָ֭הִיר לֵ֣ץ שְׁמ֑וֹ ע֝וֹשֶׂ֗ה בְּעֶבְרַ֥ת זָדֽוֹן׃
25עָצֵ֣ל תְּמִיתֶ֑נּוּ כִּֽי־ מֵאֲנ֖וּ יָדָ֣יו לַעֲשֽׂוֹת׃
26הַ֭יּוֹם הִתְאַוָּ֣ה תַאֲוָ֑ה וְצַדִּ֥יק יִ֝תֵּ֗ן וְלֹ֣א יַחְשֹֽׂךְ׃
27רְ֭שָׁעִים תּוֹעֵבָ֑ה אַ֝֗ף כִּֽי־ בְזִמָּ֥ה יְבִיאֶֽנּוּ׃
28כְּזָבִ֥ים יֹאבֵ֑ד וְאִ֥ישׁ שׁ֝וֹמֵ֗עַ לָנֶ֥צַח יְדַבֵּֽר׃
29אִ֣ישׁ רָשָׁ֣ע בְּפָנָ֑יו וְ֝יָשָׁ֗ר ה֤וּא ׀ [יָכִין כ] (יָבִ֬ין ק) [דְּרָכָיו כ] (דַּרְכֹּֽו׃ ק)
30חָ֭כְמָה וְאֵ֣ין תְּבוּנָ֑ה וְאֵ֥ין עֵ֝צָ֗ה לְנֶ֣גֶד יְהוָֽה׃ פ
