1וַיְהִ֗י אַחַר֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וְהָ֣אֱלֹהִ֔ים נִסָּ֖ה אֶת־ אַבְרָהָ֑ם וַיֹּ֣אמֶר אֵלָ֔יו אַבְרָהָ֖ם וַיֹּ֥אמֶר הִנֵּֽנִי׃
2קַח־ נָ֠א אֶת־ בִּנְךָ֨ אֶת־ יְחִֽידְךָ֤ אֲשֶׁר־ אָהַ֙בְתָּ֙ אֶת־ יִצְחָ֔ק וְלֶךְ־ לְךָ֔ אֶל־ אֶ֖רֶץ הַמֹּרִיָּ֑ה וְהַעֲלֵ֤הוּ שָׁם֙ לְעֹלָ֔ה עַ֚ל אַחַ֣ד הֶֽהָרִ֔ים אֲשֶׁ֖ר אֹמַ֥ר אֵלֶֽיךָ׃
3אַבְרָהָ֜ם בַּבֹּ֗קֶר וַֽיַּחֲבֹשׁ֙ אֶת־ חֲמֹר֔וֹ וַיִּקַּ֞ח אֶת־ שְׁנֵ֤י נְעָרָיו֙ אִתּ֔וֹ וְאֵ֖ת יִצְחָ֣ק בְּנ֑וֹ וַיְבַקַּע֙ עֲצֵ֣י עֹלָ֔ה וַיָּ֣קָם וַיֵּ֔לֶךְ אֶל־ הַמָּק֖וֹם אֲשֶׁר־ אָֽמַר־ ל֥וֹ הָאֱלֹהִֽים׃
4הַשְּׁלִישִׁ֗י וַיִּשָּׂ֨א אַבְרָהָ֧ם אֶת־ עֵינָ֛יו וַיַּ֥רְא אֶת־ הַמָּק֖וֹם מֵרָחֹֽק׃
5אַבְרָהָ֜ם אֶל־ נְעָרָ֗יו שְׁבוּ־ לָכֶ֥ם פֹּה֙ עִֽם־ הַחֲמ֔וֹר וַאֲנִ֣י וְהַנַּ֔עַר נֵלְכָ֖ה עַד־ כֹּ֑ה וְנִֽשְׁתַּחֲוֶ֖ה וְנָשׁ֥וּבָה אֲלֵיכֶֽם׃
6אַבְרָהָ֜ם אֶת־ עֲצֵ֣י הָעֹלָ֗ה וַיָּ֙שֶׂם֙ עַל־ יִצְחָ֣ק בְּנ֔וֹ וַיִּקַּ֣ח בְּיָד֔וֹ אֶת־ הָאֵ֖שׁ וְאֶת־ הַֽמַּאֲכֶ֑לֶת וַיֵּלְכ֥וּ שְׁנֵיהֶ֖ם יַחְדָּֽו׃
7יִצְחָ֜ק אֶל־ אַבְרָהָ֤ם אָבִיו֙ וַיֹּ֣אמֶר אָבִ֔י וַיֹּ֖אמֶר הִנֶּ֣נִּֽי בְנִ֑י וַיֹּ֗אמֶר הִנֵּ֤ה הָאֵשׁ֙ וְהָ֣עֵצִ֔ים וְאַיֵּ֥ה הַשֶּׂ֖ה לְעֹלָֽה׃
8אַבְרָהָ֔ם אֱלֹהִ֞ים יִרְאֶה־ לּ֥וֹ הַשֶּׂ֛ה לְעֹלָ֖ה בְּנִ֑י וַיֵּלְכ֥וּ שְׁנֵיהֶ֖ם יַחְדָּֽו׃
9אֶֽל־ הַמָּקוֹם֮ אֲשֶׁ֣ר אָֽמַר־ ל֣וֹ הָאֱלֹהִים֒ וַיִּ֨בֶן שָׁ֤ם אַבְרָהָם֙ אֶת־ הַמִּזְבֵּ֔חַ וַֽיַּעֲרֹ֖ךְ אֶת־ הָעֵצִ֑ים וַֽיַּעֲקֹד֙ אֶת־ יִצְחָ֣ק בְּנ֔וֹ וַיָּ֤שֶׂם אֹתוֹ֙ עַל־ הַמִּזְבֵּ֔חַ מִמַּ֖עַל לָעֵצִֽים׃
10אַבְרָהָם֙ אֶת־ יָד֔וֹ וַיִּקַּ֖ח אֶת־ הַֽמַּאֲכֶ֑לֶת לִשְׁחֹ֖ט אֶת־ בְּנֽוֹ׃
11אֵלָ֜יו מַלְאַ֤ךְ יְהוָה֙ מִן־ הַשָּׁמַ֔יִם וַיֹּ֖אמֶר אַבְרָהָ֣ם ׀ אַבְרָהָ֑ם וַיֹּ֖אמֶר הִנֵּֽנִי׃
12אַל־ תִּשְׁלַ֤ח יָֽדְךָ֙ אֶל־ הַנַּ֔עַר וְאַל־ תַּ֥עַשׂ ל֖וֹ מְא֑וּמָּה כִּ֣י ׀ עַתָּ֣ה יָדַ֗עְתִּי כִּֽי־ יְרֵ֤א אֱלֹהִים֙ אַ֔תָּה וְלֹ֥א חָשַׂ֛כְתָּ אֶת־ בִּנְךָ֥ אֶת־ יְחִידְךָ֖ מִמֶּֽנִּי׃
13אַבְרָהָ֜ם אֶת־ עֵינָ֗יו וַיַּרְא֙ וְהִנֵּה־ אַ֔יִל אַחַ֕ר נֶאֱחַ֥ז בַּסְּבַ֖ךְ בְּקַרְנָ֑יו וַיֵּ֤לֶךְ אַבְרָהָם֙ וַיִּקַּ֣ח אֶת־ הָאַ֔יִל וַיַּעֲלֵ֥הוּ לְעֹלָ֖ה תַּ֥חַת בְּנֽוֹ׃
14אַבְרָהָ֛ם שֵֽׁם־ הַמָּק֥וֹם הַה֖וּא יְהוָ֣ה ׀ יִרְאֶ֑ה אֲשֶׁר֙ יֵאָמֵ֣ר הַיּ֔וֹם בְּהַ֥ר יְהוָ֖ה יֵרָאֶֽה׃
15מַלְאַ֥ךְ יְהוָ֖ה אֶל־ אַבְרָהָ֑ם שֵׁנִ֖ית מִן־ הַשָּׁמָֽיִם׃
16בִּ֥י נִשְׁבַּ֖עְתִּי נְאֻם־ יְהוָ֑ה כִּ֗י יַ֚עַן אֲשֶׁ֤ר עָשִׂ֙יתָ֙ אֶת־ הַדָּבָ֣ר הַזֶּ֔ה וְלֹ֥א חָשַׂ֖כְתָּ אֶת־ בִּנְךָ֥ אֶת־ יְחִידֶֽךָ׃
17בָרֵ֣ךְ אֲבָרֶכְךָ֗ וְהַרְבָּ֨ה אַרְבֶּ֤ה אֶֽת־ זַרְעֲךָ֙ כְּכוֹכְבֵ֣י הַשָּׁמַ֔יִם וְכַח֕וֹל אֲשֶׁ֖ר עַל־ שְׂפַ֣ת הַיָּ֑ם וְיִרַ֣שׁ זַרְעֲךָ֔ אֵ֖ת שַׁ֥עַר אֹיְבָֽיו׃
18בְזַרְעֲךָ֔ כֹּ֖ל גּוֹיֵ֣י הָאָ֑רֶץ עֵ֕קֶב אֲשֶׁ֥ר שָׁמַ֖עְתָּ בְּקֹלִֽי׃
19אַבְרָהָם֙ אֶל־ נְעָרָ֔יו וַיָּקֻ֛מוּ וַיֵּלְכ֥וּ יַחְדָּ֖ו אֶל־ בְּאֵ֣ר שָׁ֑בַע וַיֵּ֥שֶׁב אַבְרָהָ֖ם בִּבְאֵ֥ר שָֽׁבַע׃ פ
20אַחֲרֵי֙ הַדְּבָרִ֣ים הָאֵ֔לֶּה וַיֻּגַּ֥ד לְאַבְרָהָ֖ם לֵאמֹ֑ר הִ֠נֵּה יָלְדָ֨ה מִלְכָּ֥ה גַם־ הִ֛וא בָּנִ֖ים לְנָח֥וֹר אָחִֽיךָ׃
21ע֥וּץ בְּכֹר֖וֹ וְאֶת־ בּ֣וּז אָחִ֑יו וְאֶת־ קְמוּאֵ֖ל אֲבִ֥י אֲרָֽם׃
22כֶּ֣שֶׂד וְאֶת־ חֲז֔וֹ וְאֶת־ פִּלְדָּ֖שׁ וְאֶת־ יִדְלָ֑ף וְאֵ֖ת בְּתוּאֵֽל׃
23יָלַ֣ד אֶת־ רִבְקָ֑ה שְׁמֹנָ֥ה אֵ֙לֶּה֙ יָלְדָ֣ה מִלְכָּ֔ה לְנָח֖וֹר אֲחִ֥י אַבְרָהָֽם׃
