1לַ֭מְנַצֵּחַ לְדָוִ֣ד מִזְמ֑וֹר אֱלֹהֵ֥י תְ֝הִלָּתִ֗י אַֽל־ תֶּחֱרַֽשׁ׃
2פִ֪י רָשָׁ֡ע וּֽפִי־ מִ֭רְמָה עָלַ֣י פָּתָ֑חוּ דִּבְּר֥וּ אִ֝תִּ֗י לְשׁ֣וֹן שָֽׁקֶר׃
3שִׂנְאָ֣ה סְבָב֑וּנִי וַיִּֽלָּחֲמ֥וּנִי חִנָּֽם׃
4אַהֲבָתִ֥י יִשְׂטְנ֗וּנִי וַאֲנִ֥י תְפִלָּֽה׃
5עָלַ֣י רָ֭עָה תַּ֣חַת טוֹבָ֑ה וְ֝שִׂנְאָ֗ה תַּ֣חַת אַהֲבָתִֽי׃
6עָלָ֣יו רָשָׁ֑ע וְ֝שָׂטָ֗ן יַעֲמֹ֥ד עַל־ יְמִינֽוֹ׃
7יֵצֵ֣א רָשָׁ֑ע וּ֝תְפִלָּת֗וֹ תִּהְיֶ֥ה לַֽחֲטָאָֽה׃
8יָמָ֥יו מְעַטִּ֑ים פְּ֝קֻדָּת֗וֹ יִקַּ֥ח אַחֵֽר׃
9בָנָ֥יו יְתוֹמִ֑ים וְ֝אִשְׁתּוֹ אַלְמָנָֽה׃
10יָנ֣וּעוּ בָנָ֣יו וְשִׁאֵ֑לוּ וְ֝דָרְשׁ֗וּ מֵחָרְבוֹתֵיהֶֽם׃
11נ֭וֹשֶׁה לְכָל־ אֲשֶׁר־ ל֑וֹ וְיָבֹ֖זּוּ זָרִ֣ים יְגִיעֽוֹ׃ אַל־ יְהִי־
12מֹשֵׁ֣ךְ חָ֑סֶד וְֽאַל־ יְהִ֥י ח֝וֹנֵ֗ן לִיתוֹמָֽיו׃
13אַחֲרִית֥וֹ לְהַכְרִ֑ית בְּד֥וֹר אַ֝חֵ֗ר יִמַּ֥ח שְׁמָֽם׃
14עֲוֹ֣ן אֲ֭בֹתָיו אֶל־ יְהוָ֑ה וְחַטַּ֥את אִ֝מּ֗וֹ אַל־ תִּמָּֽח׃
15נֶֽגֶד־ יְהוָ֣ה תָּמִ֑יד וְיַכְרֵ֖ת מֵאֶ֣רֶץ זִכְרָֽם׃
16אֲשֶׁ֤ר ׀ לֹ֥א זָכַר֮ עֲשׂ֪וֹת חָ֥סֶד וַיִּרְדֹּ֡ף אִישׁ־ עָנִ֣י וְ֭אֶבְיוֹן וְנִכְאֵ֨ה לֵבָ֬ב לְמוֹתֵֽת׃
17קְ֭לָלָה וַתְּבוֹאֵ֑הוּ וְֽלֹא־ חָפֵ֥ץ בִּ֝בְרָכָ֗ה וַתִּרְחַ֥ק מִמֶּֽנּוּ׃
18קְלָלָ֗ה כְּמַ֫דּ֥וֹ וַתָּבֹ֣א כַמַּ֣יִם בְּקִרְבּ֑וֹ וְ֝כַשֶּׁ֗מֶן בְּעַצְמוֹתָֽיו׃
19ל֭וֹ כְּבֶ֣גֶד יַעְטֶ֑ה וּ֝לְמֵ֗זַח תָּמִ֥יד יַחְגְּרֶֽהָ׃
20פְּעֻלַּ֣ת שֹׂ֭טְנַי מֵאֵ֣ת יְהוָ֑ה וְהַדֹּבְרִ֥ים רָ֝֗ע עַל־ נַפְשִֽׁי׃
21יְה֘וִ֤ה אֲדֹנָ֗י עֲ‍ֽשֵׂה־ אִ֭תִּי לְמַ֣עַן שְׁמֶ֑ךָ כִּי־ ט֥וֹב חַ֝סְדְּךָ֗ הַצִּילֵֽנִי׃
22עָנִ֣י וְאֶבְי֣וֹן אָנֹ֑כִי וְ֝לִבִּ֗י חָלַ֥ל בְּקִרְבִּֽי׃
23כִּנְטוֹת֥וֹ נֶהֱלָ֑כְתִּי נִ֝נְעַ֗רְתִּי כָּֽאַרְבֶּֽה׃
24כָּשְׁל֣וּ מִצּ֑וֹם וּ֝בְשָׂרִ֗י כָּחַ֥שׁ מִשָּֽׁמֶן׃
25הָיִ֣יתִי חֶרְפָּ֣ה לָהֶ֑ם יִ֝רְא֗וּנִי יְנִיע֥וּן רֹאשָֽׁם׃
26יְהוָ֣ה אֱלֹהָ֑י ה֭וֹשִׁיעֵ֣נִי כְחַסְדֶּֽךָ׃
27כִּי־ יָ֣דְךָ זֹּ֑את אַתָּ֖ה יְהוָ֣ה עֲשִׂיתָֽהּ׃
28הֵמָּה֮ וְאַתָּ֪ה תְבָ֫רֵ֥ךְ קָ֤מוּ ׀ וַיֵּבֹ֗שׁוּ וְֽעַבְדְּךָ֥ יִשְׂמָֽח׃
29שׂוֹטְנַ֣י כְּלִמָּ֑ה וְיַעֲט֖וּ כַמְעִ֣יל בָּשְׁתָּֽם׃
30יְהוָ֣ה מְאֹ֣ד בְּפִ֑י וּבְת֖וֹךְ רַבִּ֣ים אֲהַֽלְלֶֽנּוּ׃
