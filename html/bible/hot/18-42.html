1וַיַּ֖עַן אִיּ֥וֹב אֶת־ יְהוָ֗ה וַיֹּאמַֽר׃
2כ] (יָ֭דַעְתִּי ק) כִּי־ כֹ֣ל תּוּכָ֑ל וְלֹא־ יִבָּצֵ֖ר מִמְּךָ֣ מְזִמָּֽה׃
3זֶ֨ה ׀ מַעְלִ֥ים עֵצָ֗ה בְּֽלִ֫י דָ֥עַת לָכֵ֣ן הִ֭גַּדְתִּי וְלֹ֣א אָבִ֑ין נִפְלָא֥וֹת מִ֝מֶּ֗נִּי וְלֹ֣א אֵדָֽע׃
4נָ֭א וְאָנֹכִ֣י אֲדַבֵּ֑ר אֶ֝שְׁאָלְךָ֗ וְהוֹדִיעֵֽנִי׃
5אֹ֥זֶן שְׁמַעְתִּ֑יךָ וְ֝עַתָּ֗ה עֵינִ֥י רָאָֽתְךָ׃
6כֵּ֭ן אֶמְאַ֣ס וְנִחַ֑מְתִּי עַל־ עָפָ֥ר וָאֵֽפֶר׃ פ
7אַחַ֨ר דִּבֶּ֧ר יְהוָ֛ה אֶת־ הַדְּבָרִ֥ים הָאֵ֖לֶּה אֶל־ אִיּ֑וֹב וַיֹּ֨אמֶר יְהוָ֜ה אֶל־ אֱלִיפַ֣ז הַתֵּֽימָנִ֗י חָרָ֨ה אַפִּ֤י בְךָ֙ וּבִשְׁנֵ֣י רֵעֶ֔יךָ כִּ֠י לֹ֣א דִבַּרְתֶּ֥ם אֵלַ֛י נְכוֹנָ֖ה כְּעַבְדִּ֥י אִיּֽוֹב׃
8קְחֽוּ־ לָכֶ֣ם שִׁבְעָֽה־ פָרִים֩ וְשִׁבְעָ֨ה אֵילִ֜ים וּלְכ֣וּ ׀ אֶל־ עַבְדִּ֣י אִיּ֗וֹב וְהַעֲלִיתֶ֤ם עוֹלָה֙ בַּֽעַדְכֶ֔ם וְאִיּ֣וֹב עַבְדִּ֔י יִתְפַּלֵּ֖ל עֲלֵיכֶ֑ם כִּ֧י אִם־ פָּנָ֣יו אֶשָּׂ֗א לְבִלְתִּ֞י עֲשׂ֤וֹת עִמָּכֶם֙ נְבָלָ֔ה כִּ֠י לֹ֣א דִבַּרְתֶּ֥ם אֵלַ֛י נְכוֹנָ֖ה כְּעַבְדִּ֥י אִיּֽוֹב׃
9אֱלִיפַ֨ז הַתֵּֽימָנִ֜י וּבִלְדַּ֣ד הַשּׁוּחִ֗י צֹפַר֙ הַנַּ֣עֲמָתִ֔י וַֽיַּעֲשׂ֔וּ כַּאֲשֶׁ֛ר דִּבֶּ֥ר אֲלֵיהֶ֖ם יְהוָ֑ה וַיִּשָּׂ֥א יְהוָ֖ה אֶת־ פְּנֵ֥י אִיּֽוֹב׃
10שָׁ֚ב אֶת־ [שְׁבִית כ] (שְׁב֣וּת ק) אִיּ֔וֹב בְּהִֽתְפַּֽלְל֖וֹ בְּעַ֣ד רֵעֵ֑הוּ וַ֧יֹּסֶף יְהוָ֛ה אֶת־ כָּל־ אֲשֶׁ֥ר לְאִיּ֖וֹב לְמִשְׁנֶֽה׃
11אֵ֠לָיו כָּל־ אֶחָ֨יו וְכָל־ [אַחְיֹתָיו כ] (אַחְיֹותָ֜יו ק) וְכָל־ יֹדְעָ֣יו לְפָנִ֗ים וַיֹּאכְל֨וּ עִמּ֣וֹ לֶחֶם֮ בְּבֵיתוֹ֒ וַיָּנֻ֤דוּ לוֹ֙ וַיְנַחֲמ֣וּ אֹת֔וֹ עַ֚ל כָּל־ הָ֣רָעָ֔ה אֲשֶׁר־ הֵבִ֥יא יְהוָ֖ה עָלָ֑יו וַיִּתְּנוּ־ ל֗וֹ אִ֚ישׁ קְשִׂיטָ֣ה אֶחָ֔ת וְאִ֕ישׁ נֶ֥זֶם זָהָ֖ב אֶחָֽד׃ ס
12בֵּרַ֛ךְ אֶת־ אַחֲרִ֥ית אִיּ֖וֹב מֵרֵאשִׁת֑וֹ וַֽיְהִי־ ל֡וֹ אַרְבָּעָה֩ עָשָׂ֨ר אֶ֜לֶף צֹ֗אן וְשֵׁ֤שֶׁת אֲלָפִים֙ גְּמַלִּ֔ים וְאֶֽלֶף־ צֶ֥מֶד בָּקָ֖ר וְאֶ֥לֶף אֲתוֹנֽוֹת׃
13ל֛וֹ שִׁבְעָ֥נָה בָנִ֖ים וְשָׁל֥וֹשׁ בָּנֽוֹת׃
14שֵׁם־ הָֽאַחַת֙ יְמִימָ֔ה וְשֵׁ֥ם הַשֵּׁנִ֖ית קְצִיעָ֑ה וְשֵׁ֥ם הַשְּׁלִישִׁ֖ית קֶ֥רֶן הַפּֽוּךְ׃
15נִמְצָ֜א נָשִׁ֥ים יָפ֛וֹת כִּבְנ֥וֹת אִיּ֖וֹב בְּכָל־ הָאָ֑רֶץ וַיִּתֵּ֨ן לָהֶ֧ם אֲבִיהֶ֛ם נַחֲלָ֖ה בְּת֥וֹךְ אֲחֵיהֶֽם׃ ס
16אִיּוֹב֙ אַֽחֲרֵי־ זֹ֔את מֵאָ֥ה וְאַרְבָּעִ֖ים שָׁנָ֑ה [וַיַּרְא כ] (וַיִּרְאֶ֗ה ק) אֶת־ בָּנָיו֙ וְאֶת־ בְּנֵ֣י בָנָ֔יו אַרְבָּעָ֖ה דֹּרֽוֹת׃
