1לִשְׁלֹמֹ֨ה ׀ אֱ‍ֽלֹהִ֗ים מִ֭שְׁפָּטֶיךָ לְמֶ֣לֶךְ תֵּ֑ן וְצִדְקָתְךָ֥ לְבֶן־ מֶֽלֶךְ׃
2עַמְּךָ֣ בְצֶ֑דֶק וַעֲנִיֶּ֥יךָ בְמִשְׁפָּֽט׃
3הָרִ֓ים שָׁ֘ל֥וֹם לָעָ֑ם וּ֝גְבָע֗וֹת בִּצְדָקָֽה׃
4עֲ‍ֽנִיֵּי־ עָ֗ם י֭וֹשִׁיעַ לִבְנֵ֣י אֶבְי֑וֹן וִֽידַכֵּ֣א עוֹשֵֽׁק׃
5עִם־ שָׁ֑מֶשׁ וְלִפְנֵ֥י יָ֝רֵ֗חַ דּ֣וֹר דּוֹרִֽים׃
6כְּמָטָ֣ר עַל־ גֵּ֑ז כִּ֝רְבִיבִ֗ים זַרְזִ֥יף אָֽרֶץ׃
7בְּיָמָ֥יו צַדִּ֑יק וְרֹ֥ב שָׁ֝ל֗וֹם עַד־ בְּלִ֥י יָרֵֽחַ׃
8מִיָּ֣ם עַד־ יָ֑ם וּ֝מִנָּהָ֗ר עַד־ אַפְסֵי־ אָֽרֶץ׃
9יִכְרְע֣וּ צִיִּ֑ים וְ֝אֹיְבָ֗יו עָפָ֥ר יְלַחֵֽכוּ׃
10תַרְשִׁ֣ישׁ וְ֭אִיִּים מִנְחָ֣ה יָשִׁ֑יבוּ מַלְכֵ֥י שְׁבָ֥א וּ֝סְבָ֗א אֶשְׁכָּ֥ר יַקְרִֽיבוּ׃
11ל֥וֹ כָל־ מְלָכִ֑ים כָּל־ גּוֹיִ֥ם יַֽעַבְדֽוּהוּ׃
12יַ֭צִּיל אֶבְי֣וֹן מְשַׁוֵּ֑עַ וְ֝עָנִ֗י וְֽאֵין־ עֹזֵ֥ר לֽוֹ׃
13עַל־ דַּ֣ל וְאֶבְי֑וֹן וְנַפְשׁ֖וֹת אֶבְיוֹנִ֣ים יוֹשִֽׁיעַ׃
14וּ֭מֵחָמָס יִגְאַ֣ל נַפְשָׁ֑ם וְיֵיקַ֖ר דָּמָ֣ם בְּעֵינָֽיו׃
15וְיִתֶּן־ לוֹ֮ מִזְּהַ֪ב שְׁ֫בָ֥א וְיִתְפַּלֵּ֣ל בַּעֲד֣וֹ תָמִ֑יד כָּל־ הַ֝יּ֗וֹם יְבָרֲכֶֽנְהֽוּ׃
16פִסַּת־ בַּ֨ר ׀ בָּאָרֶץ֮ בְּרֹ֪אשׁ הָ֫רִ֥ים יִרְעַ֣שׁ כַּלְּבָנ֣וֹן פִּרְי֑וֹ וְיָצִ֥יצוּ מֵ֝עִ֗יר כְּעֵ֣שֶׂב הָאָֽרֶץ׃
17שְׁמ֨וֹ לְֽעוֹלָ֗ם לִפְנֵי־ שֶׁמֶשׁ֮ [יָנִין כ] (יִנֹּ֪ון ק) שְׁ֫מ֥וֹ וְיִתְבָּ֥רְכוּ ב֑וֹ כָּל־ גּוֹיִ֥ם יְאַשְּׁרֽוּהוּ׃
18יְהוָ֣ה אֱ֭לֹהִים אֱלֹהֵ֣י יִשְׂרָאֵ֑ל עֹשֵׂ֖ה נִפְלָא֣וֹת לְבַדּֽוֹ׃
19שֵׁ֥ם כְּבוֹד֗וֹ לְע֫וֹלָ֥ם וְיִמָּלֵ֣א כְ֭בוֹדוֹ אֶת־ כֹּ֥ל הָאָ֗רֶץ אָ֘מֵ֥ן ׀ וְאָמֵֽן׃
