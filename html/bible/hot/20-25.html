1גַּם־ אֵ֭לֶּה מִשְׁלֵ֣י שְׁלֹמֹ֑ה אֲשֶׁ֥ר הֶ֝עְתִּ֗יקוּ אַנְשֵׁ֤י ׀ חִזְקִיָּ֬ה מֶֽלֶךְ־ יְהוּדָֽה׃
2אֱ֭לֹהִים הַסְתֵּ֣ר דָּבָ֑ר וּכְבֹ֥ד מְ֝לָכִ֗ים חֲקֹ֣ר דָּבָֽר׃
3לָ֭רוּם וָאָ֣רֶץ לָעֹ֑מֶק וְלֵ֥ב מְ֝לָכִ֗ים אֵ֣ין חֵֽקֶר׃
4סִיגִ֣ים מִכָּ֑סֶף וַיֵּצֵ֖א לַצֹּרֵ֣ף כֶּֽלִי׃
5רָ֭שָׁע לִפְנֵי־ מֶ֑לֶךְ וְיִכּ֖וֹן בַּצֶּ֣דֶק כִּסְאֽוֹ׃
6תִּתְהַדַּ֥ר לִפְנֵי־ מֶ֑לֶךְ וּבִמְק֥וֹם גְּ֝דֹלִ֗ים אַֽל־ תַּעֲמֹֽד׃
7ט֥וֹב אֲמָר־ לְךָ֗ עֲ‍ֽלֵ֫ה הֵ֥נָּה מֵֽ֭הַשְׁפִּ֣ילְךָ לִפְנֵ֣י נָדִ֑יב אֲשֶׁ֖ר רָא֣וּ עֵינֶֽיךָ׃
8תֵּצֵ֥א לָרִ֗ב מַ֫הֵ֥ר פֶּ֣ן מַה־ תַּ֭עֲשֶׂה בְּאַחֲרִיתָ֑הּ בְּהַכְלִ֖ים אֹתְךָ֣ רֵעֶֽךָ׃
9רִ֣יב אֶת־ רֵעֶ֑ךָ וְס֖וֹד אַחֵ֣ר אַל־ תְּגָֽל׃
10יְחַסֶּדְךָ֥ שֹׁמֵ֑עַ וְ֝דִבָּתְךָ֗ לֹ֣א תָשֽׁוּב׃
11זָ֭הָב בְּמַשְׂכִּיּ֥וֹת כָּ֑סֶף דָּ֝בָ֗ר דָּבֻ֥ר עַל־ אָפְנָֽיו׃
12זָ֭הָב וַחֲלִי־ כָ֑תֶם מוֹכִ֥יחַ חָ֝כָ֗ם עַל־ אֹ֥זֶן שֹׁמָֽעַת׃
13שֶׁ֨לֶג ׀ בְּי֬וֹם קָצִ֗יר צִ֣יר נֶ֭אֱמָן לְשֹׁלְחָ֑יו וְנֶ֖פֶשׁ אֲדֹנָ֣יו יָשִֽׁיב׃ פ
14וְ֭רוּחַ וְגֶ֣שֶׁם אָ֑יִן אִ֥ישׁ מִ֝תְהַלֵּ֗ל בְּמַתַּת־ שָֽׁקֶר׃
15אַ֭פַּיִם יְפֻתֶּ֣ה קָצִ֑ין וְלָשׁ֥וֹן רַ֝כָּ֗ה תִּשְׁבָּר־ גָּֽרֶם׃
16מָ֭צָאתָ אֱכֹ֣ל דַּיֶּ֑ךָּ פֶּן־ תִּ֝שְׂבָּעֶ֗נּוּ וַהֲקֵֽאתֽוֹ׃
17רַ֭גְלְךָ מִבֵּ֣ית רֵעֶ֑ךָ פֶּן־ יִ֝שְׂבָּעֲךָ֗ וּשְׂנֵאֶֽךָ׃
18וְ֭חֶרֶב וְחֵ֣ץ שָׁנ֑וּן אִ֥ישׁ עֹנֶ֥ה בְ֝רֵעֵ֗הוּ עֵ֣ד שָֽׁקֶר׃
19רֹ֭עָה וְרֶ֣גֶל מוּעָ֑דֶת מִבְטָ֥ח בּ֝וֹגֵ֗ד בְּי֣וֹם צָרָֽה׃
20בֶּ֨גֶד ׀ בְּי֣וֹם קָ֭רָה חֹ֣מֶץ עַל־ נָ֑תֶר וְשָׁ֥ר בַּ֝שִּׁרִ֗ים עַ֣ל לֶב־ רָֽע׃ פ
21רָעֵ֣ב שֹׂ֭נַאֲךָ הַאֲכִלֵ֣הוּ לָ֑חֶם וְאִם־ צָ֝מֵ֗א הַשְׁקֵ֥הוּ מָֽיִם׃
22גֶֽחָלִ֗ים אַ֭תָּה חֹתֶ֣ה עַל־ רֹאשׁ֑וֹ וַֽ֝יהוָ֗ה יְשַׁלֶּם־ לָֽךְ׃
23צָ֭פוֹן תְּח֣וֹלֵֽל גָּ֑שֶׁם וּפָנִ֥ים נִ֝זְעָמִ֗ים לְשׁ֣וֹן סָֽתֶר׃
24שֶׁ֥בֶת עַל־ פִּנַּת־ גָּ֑ג מֵאֵ֥שֶׁת [מִדֹונִים כ] (מִ֝דְיָנִ֗ים ק) וּבֵ֥ית חָֽבֶר׃
25קָ֭רִים עַל־ נֶ֣פֶשׁ עֲיֵפָ֑ה וּשְׁמוּעָ֥ה ט֝וֹבָ֗ה מֵאֶ֥רֶץ מֶרְחָֽק׃
26נִ֭רְפָּשׂ וּמָק֣וֹר מָשְׁחָ֑ת צַ֝דִּ֗יק מָ֣ט לִפְנֵֽי־ רָשָֽׁע׃
27דְּבַ֣שׁ הַרְבּ֣וֹת לֹא־ ט֑וֹב וְחֵ֖קֶר כְּבֹדָ֣ם כָּבֽוֹד׃
