1וַיִּקְרָ֣א בְאָזְנַ֗י ק֤וֹל גָּדוֹל֙ לֵאמֹ֔ר קָרְב֖וּ פְּקֻדּ֣וֹת הָעִ֑יר וְאִ֛ישׁ כְּלִ֥י מַשְׁחֵת֖וֹ בְּיָדֽוֹ׃
2שִׁשָּׁ֣ה אֲנָשִׁ֡ים בָּאִ֣ים ׀ מִדֶּרֶךְ־ שַׁ֨עַר הָעֶלְי֜וֹן אֲשֶׁ֣ר ׀ מָפְנֶ֣ה צָפ֗וֹנָה וְאִ֨ישׁ כְּלִ֤י מַפָּצוֹ֙ בְּיָד֔וֹ וְאִישׁ־ אֶחָ֤ד בְּתוֹכָם֙ לָבֻ֣שׁ בַּדִּ֔ים וְקֶ֥סֶת הַסֹּפֵ֖ר בְּמָתְנָ֑יו וַיָּבֹ֙אוּ֙ וַיַּ֣עַמְד֔וּ אֵ֖צֶל מִזְבַּ֥ח הַנְּחֹֽשֶׁת׃
3אֱלֹהֵ֣י יִשְׂרָאֵ֗ל נַעֲלָה֙ מֵעַ֤ל הַכְּרוּב֙ אֲשֶׁ֣ר הָיָ֣ה עָלָ֔יו אֶ֖ל מִפְתַּ֣ן הַבָּ֑יִת וַיִּקְרָ֗א אֶל־ הָאִישׁ֙ הַלָּבֻ֣שׁ הַבַּדִּ֔ים אֲשֶׁ֛ר קֶ֥סֶת הַסֹּפֵ֖ר בְּמָתְנָֽיו׃ ס
4יְהוָה֙ [אֵלֹו כ] (אֵלָ֔יו ק) עֲבֹר֙ בְּת֣וֹךְ הָעִ֔יר בְּת֖וֹךְ יְרֽוּשָׁלִָ֑ם וְהִתְוִ֨יתָ תָּ֜ו עַל־ מִצְח֣וֹת הָאֲנָשִׁ֗ים הַנֶּֽאֱנָחִים֙ וְהַנֶּ֣אֱנָקִ֔ים עַ֚ל כָּל־ הַתּ֣וֹעֵב֔וֹת הַֽנַּעֲשׂ֖וֹת בְּתוֹכָֽהּ׃
5אָמַ֣ר בְּאָזְנַ֔י עִבְר֥וּ בָעִ֛יר אַחֲרָ֖יו וְהַכּ֑וּ [עַל־ כ] (אַל־ תָּחֹ֥ס ק) [עֵינֵיכֶם כ] (עֵינְכֶ֖ם ק) וְאַל־ תַּחְמֹֽלוּ׃
6בָּח֣וּר וּבְתוּלָה֩ וְטַ֨ף וְנָשִׁ֜ים תַּהַרְג֣וּ לְמַשְׁחִ֗ית וְעַל־ כָּל־ אִ֨ישׁ אֲשֶׁר־ עָלָ֤יו הַתָּו֙ אַל־ תִּגַּ֔שׁוּ וּמִמִּקְדָּשִׁ֖י תָּחֵ֑לּוּ וַיָּחֵ֙לּוּ֙ בָּאֲנָשִׁ֣ים הַזְּקֵנִ֔ים אֲשֶׁ֖ר לִפְנֵ֥י הַבָּֽיִת׃
7אֲלֵיהֶ֜ם טַמְּא֣וּ אֶת־ הַבַּ֗יִת וּמַלְא֧וּ אֶת־ הַחֲצֵר֛וֹת חֲלָלִ֖ים צֵ֑אוּ וְיָצְא֖וּ וְהִכּ֥וּ בָעִֽיר׃
8כְּהַכּוֹתָ֔ם וְנֵֽאשֲׁאַ֖ר אָ֑נִי וָאֶפְּלָ֨ה עַל־ פָּנַ֜י וָאֶזְעַ֗ק וָֽאֹמַר֙ אֲהָהּ֙ אֲדֹנָ֣י יְהוִ֔ה הֲמַשְׁחִ֣ית אַתָּ֗ה אֵ֚ת כָּל־ שְׁאֵרִ֣ית יִשְׂרָאֵ֔ל בְּשָׁפְכְּךָ֥ אֶת־ חֲמָתְךָ֖ עַל־ יְרוּשָׁלִָֽם׃
9אֵלַ֗י עֲוֹ֨ן בֵּֽית־ יִשְׂרָאֵ֤ל וִֽיהוּדָה֙ גָּדוֹל֙ בִּמְאֹ֣ד מְאֹ֔ד וַתִּמָּלֵ֤א הָאָ֙רֶץ֙ דָּמִ֔ים וְהָעִ֖יר מָלְאָ֣ה מֻטֶּ֑ה כִּ֣י אָמְר֗וּ עָזַ֤ב יְהוָה֙ אֶת־ הָאָ֔רֶץ וְאֵ֥ין יְהוָ֖ה רֹאֶֽה׃
10אֲנִ֔י לֹא־ תָח֥וֹס עֵינִ֖י וְלֹ֣א אֶחְמֹ֑ל דַּרְכָּ֖ם בְּרֹאשָׁ֥ם נָתָֽתִּי׃
