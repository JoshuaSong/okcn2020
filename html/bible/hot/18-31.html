1בְּ֭רִית כָּרַ֣תִּי לְעֵינָ֑י וּמָ֥ה אֶ֝תְבּוֹנֵ֗ן עַל־ בְּתוּלָֽה׃
2חֵ֣לֶק אֱל֣וֹהַּ מִמָּ֑עַל וְֽנַחֲלַ֥ת שַׁ֝דַּ֗י מִמְּרֹמִֽים׃
3אֵ֥יד לְעַוָּ֑ל וְ֝נֵ֗כֶר לְפֹ֣עֲלֵי אָֽוֶן׃
4ה֭וּא יִרְאֶ֣ה דְרָכָ֑י וְֽכָל־ צְעָדַ֥י יִסְפּֽוֹר׃
5הָלַ֥כְתִּי עִם־ שָׁ֑וְא וַתַּ֖חַשׁ עַל־ מִרְמָ֣ה רַגְלִֽי׃
6בְמֹאזְנֵי־ צֶ֑דֶק וְיֵדַ֥ע אֱ֝ל֗וֹהַּ תֻּמָּתִֽי׃
7תִּטֶּ֣ה אַשֻּׁרִי֮ מִנִּ֪י הַ֫דָּ֥רֶךְ וְאַחַ֣ר עֵ֭ינַי הָלַ֣ךְ לִבִּ֑י וּ֝בְכַפַּ֗י דָּ֣בַק מֻאֽוּם׃ פ
8וְאַחֵ֣ר יֹאכֵ֑ל וְֽצֶאֱצָאַ֥י יְשֹׁרָֽשׁוּ׃
9נִפְתָּ֣ה לִ֭בִּי עַל־ אִשָּׁ֑ה וְעַל־ פֶּ֖תַח רֵעִ֣י אָרָֽבְתִּי׃
10לְאַחֵ֣ר אִשְׁתִּ֑י וְ֝עָלֶ֗יהָ יִכְרְע֥וּן אֲחֵרִֽין׃
11[הוּא כ] (הִ֥יא ק) זִמָּ֑ה [וְהִיא כ] (וְ֝ה֗וּא ק) עָוֹ֥ן פְּלִילִֽים׃
12אֵ֣שׁ הִ֭יא עַד־ אֲבַדּ֣וֹן תֹּאכֵ֑ל וּֽבְכָל־ תְּב֖וּאָתִ֣י תְשָׁרֵֽשׁ׃
13אֶמְאַ֗ס מִשְׁפַּ֣ט עַ֭בְדִּי וַאֲמָתִ֑י בְּ֝רִבָ֗ם עִמָּדִֽי׃
14אֶֽ֭עֱשֶׂה כִּֽי־ יָק֣וּם אֵ֑ל וְכִֽי־ יִ֝פְקֹ֗ד מָ֣ה אֲשִׁיבֶֽנּוּ׃
15בַ֭בֶּטֶן עֹשֵׂ֣נִי עָשָׂ֑הוּ וַ֝יְכֻנֶ֗נּוּ בָּרֶ֥חֶם אֶחָֽד׃
16אֶ֭מְנַע מֵחֵ֣פֶץ דַּלִּ֑ים וְעֵינֵ֖י אַלְמָנָ֣ה אֲכַלֶּֽה׃
17פִּתִּ֣י לְבַדִּ֑י וְלֹא־ אָכַ֖ל יָת֣וֹם מִמֶּֽנָּה׃
18מִ֭נְּעוּרַי גְּדֵלַ֣נִי כְאָ֑ב וּמִבֶּ֖טֶן אִמִּ֣י אַנְחֶֽנָּה׃
19אֶרְאֶ֣ה א֭וֹבֵד מִבְּלִ֣י לְב֑וּשׁ וְאֵ֥ין כְּ֝ס֗וּת לָאֶבְיֽוֹן׃
20לֹ֣א בֵרֲכ֣וּנִי [חֲלָצֹו כ] (חֲלָצָ֑יו ק) וּמִגֵּ֥ז כְּ֝בָשַׂי יִתְחַמָּֽם׃
21הֲנִיפ֣וֹתִי עַל־ יָת֣וֹם יָדִ֑י כִּֽי־ אֶרְאֶ֥ה בַ֝שַּׁ֗עַר עֶזְרָתִֽי׃
22מִשִּׁכְמָ֣ה תִפּ֑וֹל וְ֝אֶזְרֹעִ֗י מִקָּנָ֥ה תִשָּׁבֵֽר׃
23פַ֣חַד אֵ֭לַי אֵ֣יד אֵ֑ל וּ֝מִשְּׂאֵת֗וֹ לֹ֣א אוּכָֽל׃
24שַׂ֣מְתִּי זָהָ֣ב כִּסְלִ֑י וְ֝לַכֶּ֗תֶם אָמַ֥רְתִּי מִבְטַחִֽי׃
25אֶ֭שְׂמַח כִּי־ רַ֣ב חֵילִ֑י וְכִֽי־ כַ֝בִּ֗יר מָצְאָ֥ה יָדִֽי׃
26אֶרְאֶ֣ה א֖וֹר כִּ֣י יָהֵ֑ל וְ֝יָרֵ֗חַ יָקָ֥ר הֹלֵֽךְ׃
27בַּסֵּ֣תֶר לִבִּ֑י וַתִּשַּׁ֖ק יָדִ֣י לְפִֽי׃
28ה֭וּא עָוֹ֣ן פְּלִילִ֑י כִּֽי־ כִחַ֖שְׁתִּי לָאֵ֣ל מִמָּֽעַל׃
29אֶ֭שְׂמַח בְּפִ֣יד מְשַׂנְאִ֑י וְ֝הִתְעֹרַ֗רְתִּי כִּֽי־ מְצָ֥אוֹ רָֽע׃
30נָתַ֣תִּי לַחֲטֹ֣א חִכִּ֑י לִשְׁאֹ֖ל בְּאָלָ֣ה נַפְשֽׁוֹ׃
31לֹ֣א אָ֭מְרוּ מְתֵ֣י אָהֳלִ֑י מִֽי־ יִתֵּ֥ן מִ֝בְּשָׂר֗וֹ לֹ֣א נִשְׂבָּֽע׃
32לֹא־ יָלִ֣ין גֵּ֑ר דְּ֝לָתַ֗י לָאֹ֥רַח אֶפְתָּֽח׃
33כִּסִּ֣יתִי כְאָדָ֣ם פְּשָׁעָ֑י לִטְמ֖וֹן בְּחֻבִּ֣י עֲוֹֽנִי׃
34אֶֽעֱר֨וֹץ ׀ הָ֘מ֤וֹן רַבָּ֗ה וּבוּז־ מִשְׁפָּח֥וֹת יְחִתֵּ֑נִי וָ֝אֶדֹּ֗ם לֹא־ אֵ֥צֵא פָֽתַח׃
35יִתֶּן־ לִ֨י ׀ שֹׁ֘מֵ֤עַֽ לִ֗י הֶן־ תָּ֭וִי שַׁדַּ֣י יַעֲנֵ֑נִי וְסֵ֥פֶר כָּ֝תַ֗ב אִ֣ישׁ רִיבִֽי׃
36לֹ֣א עַל־ שִׁ֭כְמִי אֶשָּׂאֶ֑נּוּ אֶֽעֶנְדֶ֖נּוּ עֲטָר֣וֹת לִֽי׃
37צְ֭עָדַי אַגִּידֶ֑נּוּ כְּמוֹ־ נָ֝גִ֗יד אֲקָרֲבֶֽנּוּ׃
38עָ֭לַי אַדְמָתִ֣י תִזְעָ֑ק וְ֝יַ֗חַד תְּלָמֶ֥יהָ יִבְכָּיֽוּן׃
39כֹּ֭חָהּ אָכַ֣לְתִּי בְלִי־ כָ֑סֶף וְנֶ֖פֶשׁ בְּעָלֶ֣יהָ הִפָּֽחְתִּי׃
