1אִ֣ישׁ תּ֭וֹכָחוֹת מַקְשֶׁה־ עֹ֑רֶף פֶּ֥תַע יִ֝שָּׁבֵ֗ר וְאֵ֣ין מַרְפֵּֽא׃
2צַ֭דִּיקִים יִשְׂמַ֣ח הָעָ֑ם וּבִמְשֹׁ֥ל רָ֝שָׁ֗ע יֵאָ֥נַֽח עָֽם׃
3אֹהֵ֣ב חָ֭כְמָה יְשַׂמַּ֣ח אָבִ֑יו וְרֹעֶ֥ה ז֝וֹנ֗וֹת יְאַבֶּד־ הֽוֹן׃
4בְּ֭מִשְׁפָּט יַעֲמִ֣יד אָ֑רֶץ וְאִ֖ישׁ תְּרוּמ֣וֹת יֶֽהֶרְסֶֽנָּה׃
5מַחֲלִ֣יק עַל־ רֵעֵ֑הוּ רֶ֝֗שֶׁת פּוֹרֵ֥שׂ עַל־ פְּעָמָֽיו׃
6אִ֣ישׁ רָ֣ע מוֹקֵ֑שׁ וְ֝צַדִּ֗יק יָר֥וּן וְשָׂמֵֽחַ׃
7צַ֭דִּיק דִּ֣ין דַּלִּ֑ים רָ֝שָׁ֗ע לֹא־ יָבִ֥ין דָּֽעַת׃
8לָ֭צוֹן יָפִ֣יחוּ קִרְיָ֑ה וַ֝חֲכָמִ֗ים יָשִׁ֥יבוּ אָֽף׃
9חָכָ֗ם נִ֭שְׁפָּט אֶת־ אִ֣ישׁ אֱוִ֑יל וְרָגַ֥ז וְ֝שָׂחַ֗ק וְאֵ֣ין נָֽחַת׃
10דָ֭מִים יִשְׂנְאוּ־ תָ֑ם וִֽ֝ישָׁרִ֗ים יְבַקְשׁ֥וּ נַפְשֽׁוֹ׃
11ר֭וּחוֹ יוֹצִ֣יא כְסִ֑יל וְ֝חָכָ֗ם בְּאָח֥וֹר יְשַׁבְּחֶֽנָּה׃
12מַקְשִׁ֣יב עַל־ דְּבַר־ שָׁ֑קֶר כָּֽל־ מְשָׁרְתָ֥יו רְשָׁעִֽים׃
13וְאִ֣ישׁ תְּכָכִ֣ים נִפְגָּ֑שׁוּ מֵ֤אִיר־ עֵינֵ֖י שְׁנֵיהֶ֣ם יְהוָֽה׃
14שׁוֹפֵ֣ט בֶּֽאֱמֶ֣ת דַּלִּ֑ים כִּ֝סְא֗וֹ לָעַ֥ד יִכּֽוֹן׃
15וְ֭תוֹכַחַת יִתֵּ֣ן חָכְמָ֑ה וְנַ֥עַר מְ֝שֻׁלָּ֗ח מֵבִ֥ישׁ אִמּֽוֹ׃
16רְ֭שָׁעִים יִרְבֶּה־ פָּ֑שַׁע וְ֝צַדִּיקִ֗ים בְּֽמַפַּלְתָּ֥ם יִרְאֽוּ׃
17בִּ֭נְךָ וִֽינִיחֶ֑ךָ וְיִתֵּ֖ן מַעֲדַנִּ֣ים לְנַפְשֶֽׁךָ׃ פ
18חָ֭זוֹן יִפָּ֣רַֽע עָ֑ם וְשֹׁמֵ֖ר תּוֹרָ֣ה אַשְׁרֵֽהוּ׃
19לֹא־ יִוָּ֣סֶר עָ֑בֶד כִּֽי־ יָ֝בִ֗ין וְאֵ֣ין מַעֲנֶֽה׃
20אִ֭ישׁ אָ֣ץ בִּדְבָרָ֑יו תִּקְוָ֖ה לִכְסִ֣יל מִמֶּֽנּוּ׃
21מִנֹּ֣עַר עַבְדּ֑וֹ וְ֝אַחֲרִית֗וֹ יִהְיֶ֥ה מָנֽוֹן׃
22אַ֭ף יְגָרֶ֣ה מָד֑וֹן וּבַ֖עַל חֵמָ֣ה רַב־ פָּֽשַׁע׃
23אָ֭דָם תַּשְׁפִּילֶ֑נּוּ וּשְׁפַל־ ר֝֗וּחַ יִתְמֹ֥ךְ כָּבֽוֹד׃
24עִם־ גַּ֭נָּב שׂוֹנֵ֣א נַפְשׁ֑וֹ אָלָ֥ה יִ֝שְׁמַ֗ע וְלֹ֣א יַגִּֽיד׃
25אָ֭דָם יִתֵּ֣ן מוֹקֵ֑שׁ וּבוֹטֵ֖חַ בַּיהוָ֣ה יְשֻׂגָּֽב׃
26מְבַקְשִׁ֣ים פְּנֵי־ מוֹשֵׁ֑ל וּ֝מֵיְהוָ֗ה מִשְׁפַּט־ אִֽישׁ׃
