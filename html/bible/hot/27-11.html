1וַאֲנִי֙ בִּשְׁנַ֣ת אַחַ֔ת לְדָרְיָ֖וֶשׁ הַמָּדִ֑י עָמְדִ֛י לְמַחֲזִ֥יק וּלְמָע֖וֹז לֽוֹ׃
2אֱמֶ֖ת אַגִּ֣יד לָ֑ךְ הִנֵּה־ עוֹד֩ שְׁלֹשָׁ֨ה מְלָכִ֜ים עֹמְדִ֣ים לְפָרַ֗ס וְהָֽרְבִיעִי֙ יַעֲשִׁ֤יר עֹֽשֶׁר־ גָּדוֹל֙ מִכֹּ֔ל וּכְחֶזְקָת֣וֹ בְעָשְׁר֔וֹ יָעִ֣יר הַכֹּ֔ל אֵ֖ת מַלְכ֥וּת יָוָֽן׃
3מֶ֣לֶךְ גִּבּ֑וֹר וּמָשַׁל֙ מִמְשָׁ֣ל רַ֔ב וְעָשָׂ֖ה כִּרְצוֹנֽוֹ׃
4תִּשָּׁבֵ֣ר מַלְכוּת֔וֹ וְתֵחָ֕ץ לְאַרְבַּ֖ע רוּח֣וֹת הַשָּׁמָ֑יִם וְלֹ֣א לְאַחֲרִית֗וֹ וְלֹ֤א כְמָשְׁלוֹ֙ אֲשֶׁ֣ר מָשָׁ֔ל כִּ֤י תִנָּתֵשׁ֙ מַלְכוּת֔וֹ וְלַאֲחֵרִ֖ים מִלְּבַד־ אֵֽלֶּה׃
5מֶֽלֶךְ־ הַנֶּ֖גֶב וּמִן־ שָׂרָ֑יו וְיֶחֱזַ֤ק עָלָיו֙ וּמָשָׁ֔ל מִמְשָׁ֥ל רַ֖ב מֶמְשַׁלְתּֽוֹ׃
6שָׁנִים֙ יִתְחַבָּ֔רוּ וּבַ֣ת מֶֽלֶךְ־ הַנֶּ֗גֶב תָּבוֹא֙ אֶל־ מֶ֣לֶךְ הַצָּפ֔וֹן לַעֲשׂ֖וֹת מֵישָׁרִ֑ים וְלֹֽא־ תַעְצֹ֞ר כּ֣וֹחַ הַזְּר֗וֹעַ וְלֹ֤א יַעֲמֹד֙ וּזְרֹע֔וֹ וְתִנָּתֵ֨ן הִ֤יא וּמְבִיאֶ֙יהָ֙ וְהַיֹּ֣לְדָ֔הּ וּמַחֲזִקָ֖הּ בָּעִתִּֽים׃
7מִנֵּ֥צֶר שָׁרָשֶׁ֖יהָ כַּנּ֑וֹ וְיָבֹ֣א אֶל־ הַחַ֗יִל וְיָבֹא֙ בְּמָעוֹז֙ מֶ֣לֶךְ הַצָּפ֔וֹן וְעָשָׂ֥ה בָהֶ֖ם וְהֶחֱזִֽיק׃
8אֱ‍ֽלֹהֵיהֶ֡ם עִם־ נְסִֽכֵיהֶם֩ עִם־ כְּלֵ֨י חֶמְדָּתָ֜ם כֶּ֧סֶף וְזָהָ֛ב בַּשְּׁבִ֖י יָבִ֣א מִצְרָ֑יִם וְהוּא֙ שָׁנִ֣ים יַעֲמֹ֔ד מִמֶּ֖לֶךְ הַצָּפֽוֹן׃
9בְּמַלְכוּת֙ מֶ֣לֶךְ הַנֶּ֔גֶב וְשָׁ֖ב אֶל־ אַדְמָתֽוֹ׃
10כ] (וּבָנָ֣יו ק) יִתְגָּר֗וּ וְאָסְפוּ֙ הֲמוֹן֙ חֲיָלִ֣ים רַבִּ֔ים וּבָ֥א ב֖וֹא וְשָׁטַ֣ף וְעָבָ֑ר וְיָשֹׁ֥ב [וְיִתְגָּרוּ כ] (וְיִתְגָּרֶ֖ה ק) עַד־ [מָעֻזָּה כ] (מָעֻזֹּֽו׃ ק)
11מֶ֣לֶךְ הַנֶּ֔גֶב וְיָצָ֕א וְנִלְחַ֥ם עִמּ֖וֹ עִם־ מֶ֣לֶךְ הַצָּפ֑וֹן וְהֶעֱמִיד֙ הָמ֣וֹן רָ֔ב וְנִתַּ֥ן הֶהָמ֖וֹן בְּיָדֽוֹ׃
12הֶהָמ֖וֹן [יָרוּם כ] (וְרָ֣ם ק) לְבָב֑וֹ וְהִפִּ֛יל רִבֹּא֖וֹת וְלֹ֥א יָעֽוֹז׃
13מֶ֣לֶךְ הַצָּפ֔וֹן וְהֶעֱמִ֣יד הָמ֔וֹן רַ֖ב מִן־ הָרִאשׁ֑וֹן וּלְקֵ֨ץ הָֽעִתִּ֤ים שָׁנִים֙ יָ֣בוֹא ב֔וֹא בְּחַ֥יִל גָּד֖וֹל וּבִרְכ֥וּשׁ רָֽב׃
14הָהֵ֔ם רַבִּ֥ים יַֽעַמְד֖וּ עַל־ מֶ֣לֶךְ הַנֶּ֑גֶב וּבְנֵ֣י ׀ פָּרִיצֵ֣י עַמְּךָ֗ יִֽנַּשְּׂא֛וּ לְהַעֲמִ֥יד חָז֖וֹן וְנִכְשָֽׁלוּ׃
15מֶ֣לֶךְ הַצָּפ֔וֹן וְיִשְׁפֹּךְ֙ סֽוֹלֲלָ֔ה וְלָכַ֖ד עִ֣יר מִבְצָר֑וֹת וּזְרֹע֤וֹת הַנֶּ֙גֶב֙ לֹ֣א יַעֲמֹ֔דוּ וְעַם֙ מִבְחָרָ֔יו וְאֵ֥ין כֹּ֖חַ לַעֲמֹֽד׃
16הַבָּ֤א אֵלָיו֙ כִּרְצוֹנ֔וֹ וְאֵ֥ין עוֹמֵ֖ד לְפָנָ֑יו וְיַעֲמֹ֥ד בְּאֶֽרֶץ־ הַצְּבִ֖י וְכָלָ֥ה בְיָדֽוֹ׃
17פָּ֠נָיו לָב֞וֹא בְּתֹ֧קֶף כָּל־ מַלְכוּת֛וֹ וִישָׁרִ֥ים עִמּ֖וֹ וְעָשָׂ֑ה וּבַ֤ת הַנָּשִׁים֙ יִתֶּן־ ל֣וֹ לְהַשְׁחִיתָ֔הּ וְלֹ֥א תַעֲמֹ֖ד וְלֹא־ ל֥וֹ תִהְיֶֽה׃
18כ] (וְיָשֵׂ֧ם ק) פָּנָ֛יו לְאִיִּ֖ים וְלָכַ֣ד רַבִּ֑ים וְהִשְׁבִּ֨ית קָצִ֤ין חֶרְפָּתוֹ֙ ל֔וֹ בִּלְתִּ֥י חֶרְפָּת֖וֹ יָשִׁ֥יב לֽוֹ׃
19פָּנָ֔יו לְמָעוּזֵּ֖י אַרְצ֑וֹ וְנִכְשַׁ֥ל וְנָפַ֖ל וְלֹ֥א יִמָּצֵֽא׃
20עַל־ כַּנּ֛וֹ מַעֲבִ֥יר נוֹגֵ֖שׂ הֶ֣דֶר מַלְכ֑וּת וּבְיָמִ֤ים אֲחָדִים֙ יִשָּׁבֵ֔ר וְלֹ֥א בְאַפַּ֖יִם וְלֹ֥א בְמִלְחָמָֽה׃
21עַל־ כַּנּוֹ֙ נִבְזֶ֔ה וְלֹא־ נָתְנ֥וּ עָלָ֖יו ה֣וֹד מַלְכ֑וּת וּבָ֣א בְשַׁלְוָ֔ה וְהֶחֱזִ֥יק מַלְכ֖וּת בַּחֲלַקְלַקּֽוֹת׃
22הַשֶּׁ֛טֶף יִשָּׁטְפ֥וּ מִלְּפָנָ֖יו וְיִשָּׁבֵ֑רוּ וְגַ֖ם נְגִ֥יד בְּרִֽית׃
23הִֽתְחַבְּר֥וּת אֵלָ֖יו יַעֲשֶׂ֣ה מִרְמָ֑ה וְעָלָ֥ה וְעָצַ֖ם בִּמְעַט־ גּֽוֹי׃
24וּבְמִשְׁמַנֵּ֣י מְדִינָה֮ יָבוֹא֒ וְעָשָׂ֗ה אֲשֶׁ֨ר לֹא־ עָשׂ֤וּ אֲבֹתָיו֙ וַאֲב֣וֹת אֲבֹתָ֔יו בִּזָּ֧ה וְשָׁלָ֛ל וּרְכ֖וּשׁ לָהֶ֣ם יִבְז֑וֹר וְעַ֧ל מִבְצָרִ֛ים יְחַשֵּׁ֥ב מַחְשְׁבֹתָ֖יו וְעַד־ עֵֽת׃
25כֹּח֨וֹ וּלְבָב֜וֹ עַל־ מֶ֣לֶךְ הַנֶּגֶב֮ בְּחַ֣יִל גָּדוֹל֒ וּמֶ֣לֶךְ הַנֶּ֗גֶב יִתְגָּרֶה֙ לַמִּלְחָמָ֔ה בְּחַֽיִל־ גָּד֥וֹל וְעָצ֖וּם עַד־ מְאֹ֑ד וְלֹ֣א יַעֲמֹ֔ד כִּֽי־ יַחְשְׁב֥וּ עָלָ֖יו מַחֲשָׁבֽוֹת׃
26פַת־ בָּג֛וֹ יִשְׁבְּר֖וּהוּ וְחֵיל֣וֹ יִשְׁט֑וֹף וְנָפְל֖וּ חֲלָלִ֥ים רַבִּֽים׃
27הַמְּלָכִים֙ לְבָבָ֣ם לְמֵרָ֔ע וְעַל־ שֻׁלְחָ֥ן אֶחָ֖ד כָּזָ֣ב יְדַבֵּ֑רוּ וְלֹ֣א תִצְלָ֔ח כִּי־ ע֥וֹד קֵ֖ץ לַמּוֹעֵֽד׃
28אַרְצוֹ֙ בִּרְכ֣וּשׁ גָּד֔וֹל וּלְבָב֖וֹ עַל־ בְּרִ֣ית קֹ֑דֶשׁ וְעָשָׂ֖ה וְשָׁ֥ב לְאַרְצֽוֹ׃
29יָשׁ֖וּב וּבָ֣א בַנֶּ֑גֶב וְלֹֽא־ תִהְיֶ֥ה כָרִאשֹׁנָ֖ה וְכָאַחֲרֹנָֽה׃
30ב֜וֹ צִיִּ֤ים כִּתִּים֙ וְנִכְאָ֔ה וְשָׁ֛ב וְזָעַ֥ם עַל־ בְּרִֽית־ ק֖וֹדֶשׁ וְעָשָׂ֑ה וְשָׁ֣ב וְיָבֵ֔ן עַל־ עֹזְבֵ֖י בְּרִ֥ית קֹֽדֶשׁ׃
31מִמֶּ֣נּוּ יַעֲמֹ֑דוּ וְחִלְּל֞וּ הַמִּקְדָּ֤שׁ הַמָּעוֹז֙ וְהֵסִ֣ירוּ הַתָּמִ֔יד וְנָתְנ֖וּ הַשִּׁקּ֥וּץ מְשׁוֹמֵֽם׃
32בְרִ֔ית יַחֲנִ֖יף בַּחֲלַקּ֑וֹת וְעַ֛ם יֹדְעֵ֥י אֱלֹהָ֖יו יַחֲזִ֥קוּ וְעָשֽׂוּ׃
33עָ֔ם יָבִ֖ינוּ לָֽרַבִּ֑ים וְנִכְשְׁל֞וּ בְּחֶ֧רֶב וּבְלֶהָבָ֛ה בִּשְׁבִ֥י וּבְבִזָּ֖ה יָמִֽים׃
34יֵעָזְר֖וּ עֵ֣זֶר מְעָ֑ט וְנִלְו֧וּ עֲלֵיהֶ֛ם רַבִּ֖ים בַּחֲלַקְלַקּֽוֹת׃
35הַמַּשְׂכִּילִ֣ים יִכָּֽשְׁל֗וּ לִצְר֥וֹף בָּהֶ֛ם וּלְבָרֵ֥ר וְלַלְבֵּ֖ן עַד־ עֵ֣ת קֵ֑ץ כִּי־ ע֖וֹד לַמּוֹעֵֽד׃
36כִרְצוֹנ֜וֹ הַמֶּ֗לֶךְ וְיִתְרוֹמֵ֤ם וְיִתְגַּדֵּל֙ עַל־ כָּל־ אֵ֔ל וְעַל֙ אֵ֣ל אֵלִ֔ים יְדַבֵּ֖ר נִפְלָא֑וֹת וְהִצְלִ֙יחַ֙ עַד־ כָּ֣לָה זַ֔עַם כִּ֥י נֶחֱרָצָ֖ה נֶעֱשָֽׂתָה׃
37אֱלֹהֵ֤י אֲבֹתָיו֙ לֹ֣א יָבִ֔ין וְעַל־ חֶמְדַּ֥ת נָשִׁ֛ים וְעַֽל־ כָּל־ אֱל֖וֹהַּ לֹ֣א יָבִ֑ין כִּ֥י עַל־ כֹּ֖ל יִתְגַּדָּֽל׃
38מָֽעֻזִּ֔ים עַל־ כַּנּ֖וֹ יְכַבֵּ֑ד וְלֶאֱל֜וֹהַּ אֲשֶׁ֧ר לֹא־ יְדָעֻ֣הוּ אֲבֹתָ֗יו יְכַבֵּ֛ד בְּזָהָ֥ב וּבְכֶ֛סֶף וּבְאֶ֥בֶן יְקָרָ֖ה וּבַחֲמֻדֽוֹת׃
39לְמִבְצְרֵ֤י מָֽעֻזִּים֙ עִם־ אֱל֣וֹהַּ נֵכָ֔ר אֲשֶׁ֥ר [הִכִּיר כ] (יַכִּ֖יר ק) יַרְבֶּ֣ה כָב֑וֹד וְהִמְשִׁילָם֙ בָּֽרַבִּ֔ים וַאֲדָמָ֖ה יְחַלֵּ֥ק בִּמְחִֽיר׃
40קֵ֗ץ יִתְנַגַּ֤ח עִמּוֹ֙ מֶ֣לֶךְ הַנֶּ֔גֶב וְיִשְׂתָּעֵ֨ר עָלָ֜יו מֶ֣לֶךְ הַצָּפ֗וֹן בְּרֶ֙כֶב֙ וּבְפָ֣רָשִׁ֔ים וּבָאֳנִיּ֖וֹת רַבּ֑וֹת וּבָ֥א בַאֲרָצ֖וֹת וְשָׁטַ֥ף וְעָבָֽר׃
41בְּאֶ֣רֶץ הַצְּבִ֔י וְרַבּ֖וֹת יִכָּשֵׁ֑לוּ וְאֵ֙לֶּה֙ יִמָּלְט֣וּ מִיָּד֔וֹ אֱד֣וֹם וּמוֹאָ֔ב וְרֵאשִׁ֖ית בְּנֵ֥י עַמּֽוֹן׃
42יָד֖וֹ בַּאֲרָצ֑וֹת וְאֶ֣רֶץ מִצְרַ֔יִם לֹ֥א תִהְיֶ֖ה לִפְלֵיטָֽה׃
43בְּמִכְמַנֵּי֙ הַזָּהָ֣ב וְהַכֶּ֔סֶף וּבְכֹ֖ל חֲמֻד֣וֹת מִצְרָ֑יִם וְלֻבִ֥ים וְכֻשִׁ֖ים בְּמִצְעָדָֽיו׃
44יְבַהֲלֻ֔הוּ מִמִּזְרָ֖ח וּמִצָּפ֑וֹן וְיָצָא֙ בְּחֵמָ֣א גְדֹלָ֔ה לְהַשְׁמִ֥יד וּֽלְהַחֲרִ֖ים רַבִּֽים׃
