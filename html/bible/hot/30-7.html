1כֹּ֤ה הִרְאַ֙נִי֙ אֲדֹנָ֣י יְהוִ֔ה וְהִנֵּה֙ יוֹצֵ֣ר גֹּבַ֔י בִּתְחִלַּ֖ת עֲל֣וֹת הַלָּ֑קֶשׁ וְהִ֨נֵּה־ לֶ֔קֶשׁ אַחַ֖ר גִּזֵּ֥י הַמֶּֽלֶךְ׃
2אִם־ כִּלָּה֙ לֶֽאֱכוֹל֙ אֶת־ עֵ֣שֶׂב הָאָ֔רֶץ וָאֹמַ֗ר אֲדֹנָ֤י יְהוִה֙ סְֽלַֽח־ נָ֔א מִ֥י יָק֖וּם יַֽעֲקֹ֑ב כִּ֥י קָטֹ֖ן הֽוּא׃
3יְהוָ֖ה עַל־ זֹ֑את לֹ֥א תִהְיֶ֖ה אָמַ֥ר יְהוָֽה׃
4הִרְאַ֙נִי֙ אֲדֹנָ֣י יְהוִ֔ה וְהִנֵּ֥ה קֹרֵ֛א לָרִ֥ב בָּאֵ֖שׁ אֲדֹנָ֣י יְהוִ֑ה וַתֹּ֙אכַל֙ אֶת־ תְּה֣וֹם רַבָּ֔ה וְאָכְלָ֖ה אֶת־ הַחֵֽלֶק׃
5אֲדֹנָ֤י יְהוִה֙ חֲדַל־ נָ֔א מִ֥י יָק֖וּם יַעֲקֹ֑ב כִּ֥י קָטֹ֖ן הֽוּא׃
6יְהוָ֖ה עַל־ זֹ֑את גַּם־ הִיא֙ לֹ֣א תִֽהְיֶ֔ה אָמַ֖ר אֲדֹנָ֥י יְהוִֽה׃ ס
7הִרְאַ֔נִי וְהִנֵּ֧ה אֲדֹנָ֛י נִצָּ֖ב עַל־ חוֹמַ֣ת אֲנָ֑ךְ וּבְיָד֖וֹ אֲנָֽךְ׃
8יְהוָ֜ה אֵלַ֗י מָֽה־ אַתָּ֤ה רֹאֶה֙ עָמ֔וֹס וָאֹמַ֖ר אֲנָ֑ךְ וַיֹּ֣אמֶר אֲדֹנָ֗י הִנְנִ֨י שָׂ֤ם אֲנָךְ֙ בְּקֶ֙רֶב֙ עַמִּ֣י יִשְׂרָאֵ֔ל לֹֽא־ אוֹסִ֥יף ע֖וֹד עֲב֥וֹר לֽוֹ׃
9בָּמ֣וֹת יִשְׂחָ֔ק וּמִקְדְּשֵׁ֥י יִשְׂרָאֵ֖ל יֶחֱרָ֑בוּ וְקַמְתִּ֛י עַל־ בֵּ֥ית יָרָבְעָ֖ם בֶּחָֽרֶב׃ פ
10אֲמַצְיָה֙ כֹּהֵ֣ן בֵּֽית־ אֵ֔ל אֶל־ יָרָבְעָ֥ם מֶֽלֶךְ־ יִשְׂרָאֵ֖ל לֵאמֹ֑ר קָשַׁ֨ר עָלֶ֜יךָ עָמ֗וֹס בְּקֶ֙רֶב֙ בֵּ֣ית יִשְׂרָאֵ֔ל לֹא־ תוּכַ֣ל הָאָ֔רֶץ לְהָכִ֖יל אֶת־ כָּל־ דְּבָרָֽיו׃
11כֹה֙ אָמַ֣ר עָמ֔וֹס בַּחֶ֖רֶב יָמ֣וּת יָרָבְעָ֑ם וְיִ֨שְׂרָאֵ֔ל גָּלֹ֥ה יִגְלֶ֖ה מֵעַ֥ל אַדְמָתֽוֹ׃ ס
12אֲמַצְיָה֙ אֶל־ עָמ֔וֹס חֹזֶ֕ה לֵ֥ךְ בְּרַח־ לְךָ֖ אֶל־ אֶ֣רֶץ יְהוּדָ֑ה וֶאֱכָל־ שָׁ֣ם לֶ֔חֶם וְשָׁ֖ם תִּנָּבֵֽא׃
13אֵ֔ל לֹֽא־ תוֹסִ֥יף ע֖וֹד לְהִנָּבֵ֑א כִּ֤י מִקְדַּשׁ־ מֶ֙לֶךְ֙ ה֔וּא וּבֵ֥ית מַמְלָכָ֖ה הֽוּא׃ ס
14עָמוֹס֙ וַיֹּ֣אמֶר אֶל־ אֲמַצְיָ֔ה לֹא־ נָבִ֣יא אָנֹ֔כִי וְלֹ֥א בֶן־ נָבִ֖יא אָנֹ֑כִי כִּֽי־ בוֹקֵ֥ר אָנֹ֖כִי וּבוֹלֵ֥ס שִׁקְמִֽים׃
15יְהוָ֔ה מֵאַחֲרֵ֖י הַצֹּ֑אן וַיֹּ֤אמֶר אֵלַי֙ יְהוָ֔ה לֵ֥ךְ הִנָּבֵ֖א אֶל־ עַמִּ֥י יִשְׂרָאֵֽל׃
16שְׁמַ֣ע דְּבַר־ יְהוָ֑ה אַתָּ֣ה אֹמֵ֗ר לֹ֤א תִנָּבֵא֙ עַל־ יִשְׂרָאֵ֔ל וְלֹ֥א תַטִּ֖יף עַל־ בֵּ֥ית יִשְׂחָֽק׃
