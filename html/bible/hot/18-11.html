1וַ֭יַּעַן צֹפַ֥ר הַֽנַּעֲמָתִ֗י וַיֹּאמַֽר׃
2דְּ֭בָרִים לֹ֣א יֵעָנֶ֑ה וְאִם־ אִ֖ישׁ שְׂפָתַ֣יִם יִצְדָּֽק׃
3מְתִ֣ים יַחֲרִ֑ישׁו וַ֝תִּלְעַ֗ג וְאֵ֣ין מַכְלִֽם׃
4זַ֣ךְ לִקְחִ֑י וּ֝בַ֗ר הָיִ֥יתִי בְעֵינֶֽיךָ׃
5מִֽי־ יִתֵּ֣ן אֱל֣וֹהַּ דַּבֵּ֑ר וְיִפְתַּ֖ח שְׂפָתָ֣יו עִמָּֽךְ׃
6לְךָ֨ ׀ תַּֽעֲלֻמ֣וֹת חָכְמָה֮ כִּֽי־ כִפְלַ֪יִם לְֽת֫וּשִׁיָּ֥ה וְדַ֡ע כִּֽי־ יַשֶּׁ֥ה לְךָ֥ אֱ֝ל֗וֹהַ מֵעֲוֹנֶֽךָ׃
7אֱל֣וֹהַ תִּמְצָ֑א אִ֤ם עַד־ תַּכְלִ֖ית שַׁדַּ֣י תִּמְצָֽא׃
8שָׁ֭מַיִם מַה־ תִּפְעָ֑ל עֲמֻקָּ֥ה מִ֝שְּׁא֗וֹל מַה־ תֵּדָֽע׃
9מֵאֶ֣רֶץ מִדָּ֑הּ וּ֝רְחָבָ֗ה מִנִּי־ יָֽם׃
10יַחֲלֹ֥ף וְיַסְגִּ֑יר וְ֝יַקְהִ֗יל וּמִ֣י יְשִׁיבֶֽנּוּ׃
11ה֭וּא יָדַ֣ע מְתֵי־ שָׁ֑וְא וַיַּרְא־ אָ֝֗וֶן וְלֹ֣א יִתְבּוֹנָֽן׃
12נָ֭בוּב יִלָּבֵ֑ב וְעַ֥יִר פֶּ֝֗רֶא אָדָ֥ם יִוָּלֵֽד׃
13אַ֭תָּ֗ה הֲכִינ֣וֹתָ לִבֶּ֑ךָ וּפָרַשְׂתָּ֖ אֵלָ֣יו כַּפֶּֽךָ׃
14אָ֣וֶן בְּ֭יָדְךָ הַרְחִיקֵ֑הוּ וְאַל־ תַּשְׁכֵּ֖ן בְּאֹהָלֶ֣יךָ עַוְלָֽה׃
15אָ֤ז ׀ תִּשָּׂ֣א פָנֶ֣יךָ מִמּ֑וּם וְהָיִ֥יתָ מֻ֝צָ֗ק וְלֹ֣א תִירָֽא׃
16אַ֭תָּה עָמָ֣ל תִּשְׁכָּ֑ח כְּמַ֖יִם עָבְר֣וּ תִזְכֹּֽר׃
17יָק֣וּם חָ֑לֶד תָּ֝עֻ֗פָה כַּבֹּ֥קֶר תִּהְיֶֽה׃
18כִּי־ יֵ֣שׁ תִּקְוָ֑ה וְ֝חָפַרְתָּ֗ לָבֶ֥טַח תִּשְׁכָּֽב׃
19וְאֵ֣ין מַחֲרִ֑יד וְחִלּ֖וּ פָנֶ֣יךָ רַבִּֽים׃
