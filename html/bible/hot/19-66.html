1לַ֭מְנַצֵּחַ שִׁ֣יר מִזְמ֑וֹר הָרִ֥יעוּ לֵ֝אלֹהִים כָּל־ הָאָֽרֶץ׃
2כְבֽוֹד־ שְׁמ֑וֹ שִׂ֥ימוּ כָ֝ב֗וֹד תְּהִלָּתֽוֹ׃
3לֵ֭אלֹהִים מַה־ נּוֹרָ֣א מַעֲשֶׂ֑יךָ בְּרֹ֥ב עֻ֝זְּךָ֗ יְֽכַחֲשׁ֖וּ לְךָ֣ אֹיְבֶֽיךָ׃
4הָאָ֤רֶץ ׀ יִשְׁתַּחֲו֣וּ לְ֭ךָ וִֽיזַמְּרוּ־ לָ֑ךְ יְזַמְּר֖וּ שִׁמְךָ֣ סֶֽלָה׃
5וּ֭רְאוּ מִפְעֲל֣וֹת אֱלֹהִ֑ים נוֹרָ֥א עֲ֝לִילָ֗ה עַל־ בְּנֵ֥י אָדָֽם׃
6יָ֨ם ׀ לְֽיַבָּשָׁ֗ה בַּ֭נָּהָר יַֽעַבְר֣וּ בְרָ֑גֶל שָׁ֝֗ם נִשְׂמְחָה־ בּֽוֹ׃
7בִּגְבוּרָת֨וֹ ׀ עוֹלָ֗ם עֵ֭ינָיו בַּגּוֹיִ֣ם תִּצְפֶּ֑ינָה הַסּוֹרְרִ֓ים ׀ אַל־ [יָרִימוּ כ] (יָר֖וּמוּ ק) לָ֣מוֹ סֶֽלָה׃
8עַמִּ֥ים ׀ אֱלֹהֵ֑ינוּ וְ֝הַשְׁמִ֗יעוּ ק֣וֹל תְּהִלָּתֽוֹ׃
9נַ֭פְשֵׁנוּ בַּֽחַיִּ֑ים וְלֹֽא־ נָתַ֖ן לַמּ֣וֹט רַגְלֵֽנוּ׃
10בְחַנְתָּ֥נוּ אֱלֹהִ֑ים צְ֝רַפְתָּ֗נוּ כִּצְרָף־ כָּֽסֶף׃
11בַמְּצוּדָ֑ה שַׂ֖מְתָּ מוּעָקָ֣ה בְמָתְנֵֽינוּ׃
12אֱנ֗וֹשׁ לְרֹ֫אשֵׁ֥נוּ בָּֽאנוּ־ בָאֵ֥שׁ וּבַמַּ֑יִם וַ֝תּוֹצִיאֵ֗נוּ לָֽרְוָיָֽה׃
13בֵיתְךָ֣ בְעוֹל֑וֹת אֲשַׁלֵּ֖ם לְךָ֣ נְדָרָֽי׃
14פָּצ֥וּ שְׂפָתָ֑י וְדִבֶּר־ פִּ֝֗י בַּצַּר־ לִֽי׃
15מֵחִ֣ים אַעֲלֶה־ לָּ֭ךְ עִם־ קְטֹ֣רֶת אֵילִ֑ים אֶ֥עֱשֶֽׂה בָקָ֖ר עִם־ עַתּוּדִ֣ים סֶֽלָה׃
16שִׁמְע֣וּ וַ֭אֲסַפְּרָה כָּל־ יִרְאֵ֣י אֱלֹהִ֑ים אֲשֶׁ֖ר עָשָׂ֣ה לְנַפְשִֽׁי׃
17פִּֽי־ קָרָ֑אתִי וְ֝רוֹמַ֗ם תַּ֣חַת לְשׁוֹנִֽי׃
18אִם־ רָאִ֣יתִי בְלִבִּ֑י לֹ֖א יִשְׁמַ֣ע ׀ אֲדֹנָֽי׃
19שָׁמַ֣ע אֱלֹהִ֑ים הִ֝קְשִׁ֗יב בְּק֣וֹל תְּפִלָּתִֽי׃
