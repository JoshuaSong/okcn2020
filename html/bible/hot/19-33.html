1רַנְּנ֣וּ צַ֭דִּיקִים בַּֽיהוָ֑ה לַ֝יְשָׁרִ֗ים נָאוָ֥ה תְהִלָּֽה׃
2לַיהוָ֣ה בְּכִנּ֑וֹר בְּנֵ֥בֶל עָ֝שׂ֗וֹר זַמְּרוּ־ לֽוֹ׃
3ל֭וֹ שִׁ֣יר חָדָ֑שׁ הֵיטִ֥יבוּ נַ֝גֵּ֗ן בִּתְרוּעָֽה׃
4יָשָׁ֥ר דְּבַר־ יְהוָ֑ה וְכָל־ מַ֝עֲשֵׂ֗הוּ בֶּאֱמוּנָֽה׃
5צְדָקָ֣ה וּמִשְׁפָּ֑ט חֶ֥סֶד יְ֝הוָ֗ה מָלְאָ֥ה הָאָֽרֶץ׃
6יְ֭הוָה שָׁמַ֣יִם נַעֲשׂ֑וּ וּבְר֥וּחַ פִּ֝֗יו כָּל־ צְבָאָֽם׃
7כַּ֭נֵּד מֵ֣י הַיָּ֑ם נֹתֵ֖ן בְּאֹצָר֣וֹת תְּהוֹמֽוֹת׃
8מֵ֭יְהוָה כָּל־ הָאָ֑רֶץ מִמֶּ֥נּוּ יָ֝ג֗וּרוּ כָּל־ יֹשְׁבֵ֥י תֵבֵֽל׃
9ה֣וּא אָמַ֣ר וַיֶּ֑הִי הֽוּא־ צִ֝וָּ֗ה וַֽיַּעֲמֹֽד׃
10הֵפִ֥יר עֲצַת־ גּוֹיִ֑ם הֵ֝נִ֗יא מַחְשְׁב֥וֹת עַמִּֽים׃
11יְ֭הוָה לְעוֹלָ֣ם תַּעֲמֹ֑ד מַחְשְׁב֥וֹת לִ֝בּ֗וֹ לְדֹ֣ר וָדֹֽר׃
12הַ֭גּוֹי אֲשֶׁר־ יְהוָ֣ה אֱלֹהָ֑יו הָעָ֓ם ׀ בָּחַ֖ר לְנַחֲלָ֣ה לֽוֹ׃
13הִבִּ֣יט יְהוָ֑ה רָ֝אָ֗ה אֶֽת־ כָּל־ בְּנֵ֥י הָאָדָֽם׃
14שִׁבְתּ֥וֹ הִשְׁגִּ֑יחַ אֶ֖ל כָּל־ יֹשְׁבֵ֣י הָאָֽרֶץ׃
15יַ֣חַד לִבָּ֑ם הַ֝מֵּבִ֗ין אֶל־ כָּל־ מַעֲשֵׂיהֶֽם׃
16הַ֭מֶּלֶךְ נוֹשָׁ֣ע בְּרָב־ חָ֑יִל גִּ֝בּ֗וֹר לֹֽא־ יִנָּצֵ֥ל בְּרָב־ כֹּֽחַ׃
17הַ֭סּוּס לִתְשׁוּעָ֑ה וּבְרֹ֥ב חֵ֝יל֗וֹ לֹ֣א יְמַלֵּֽט׃
18עֵ֣ין יְ֭הוָה אֶל־ יְרֵאָ֑יו לַֽמְיַחֲלִ֥ים לְחַסְדּֽוֹ׃
19מִמָּ֣וֶת נַפְשָׁ֑ם וּ֝לְחַיּוֹתָ֗ם בָּרָעָֽב׃
20חִכְּתָ֣ה לַֽיהוָ֑ה עֶזְרֵ֖נוּ וּמָגִנֵּ֣נוּ הֽוּא׃
21ב֭וֹ יִשְׂמַ֣ח לִבֵּ֑נוּ כִּ֤י בְשֵׁ֖ם קָדְשׁ֣וֹ בָטָֽחְנוּ׃
