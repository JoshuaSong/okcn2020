1יֵ֣שׁ רָעָ֔ה אֲשֶׁ֥ר רָאִ֖יתִי תַּ֣חַת הַשָּׁ֑מֶשׁ וְרַבָּ֥ה הִ֖יא עַל־ הָאָדָֽם׃
2אֲשֶׁ֣ר יִתֶּן־ ל֣וֹ הָאֱלֹהִ֡ים עֹשֶׁר֩ וּנְכָסִ֨ים וְכָב֜וֹד וְֽאֵינֶ֨נּוּ חָסֵ֥ר לְנַפְשׁ֣וֹ ׀ מִכֹּ֣ל אֲשֶׁר־ יִתְאַוֶּ֗ה וְלֹֽא־ יַשְׁלִיטֶ֤נּוּ הָֽאֱלֹהִים֙ לֶאֱכֹ֣ל מִמֶּ֔נּוּ כִּ֛י אִ֥ישׁ נָכְרִ֖י יֹֽאכֲלֶ֑נּוּ זֶ֥ה הֶ֛בֶל וָחֳלִ֥י רָ֖ע הֽוּא׃
3יוֹלִ֣יד אִ֣ישׁ מֵאָ֡ה וְשָׁנִים֩ רַבּ֨וֹת יִֽחְיֶ֜ה וְרַ֣ב ׀ שֶׁיִּהְי֣וּ יְמֵֽי־ שָׁנָ֗יו וְנַפְשׁוֹ֙ לֹא־ תִשְׂבַּ֣ע מִן־ הַטּוֹבָ֔ה וְגַם־ קְבוּרָ֖ה לֹא־ הָ֣יְתָה לּ֑וֹ אָמַ֕רְתִּי ט֥וֹב מִמֶּ֖נּוּ הַנָּֽפֶל׃
4בַהֶ֥בֶל בָּ֖א וּבַחֹ֣שֶׁךְ יֵלֵ֑ךְ וּבַחֹ֖שֶׁךְ שְׁמ֥וֹ יְכֻסֶּֽה׃
5שֶׁ֥מֶשׁ לֹא־ רָאָ֖ה וְלֹ֣א יָדָ֑ע נַ֥חַת לָזֶ֖ה מִזֶּֽה׃
6חָיָ֗ה אֶ֤לֶף שָׁנִים֙ פַּעֲמַ֔יִם וְטוֹבָ֖ה לֹ֣א רָאָ֑ה הֲלֹ֛א אֶל־ מָק֥וֹם אֶחָ֖ד הַכֹּ֥ל הוֹלֵֽךְ׃
7עֲמַ֥ל הָאָדָ֖ם לְפִ֑יהוּ וְגַם־ הַנֶּ֖פֶשׁ לֹ֥א תִמָּלֵֽא׃
8מַה־ יּוֹתֵ֥ר לֶחָכָ֖ם מִֽן־ הַכְּסִ֑יל מַה־ לֶּעָנִ֣י יוֹדֵ֔עַ לַהֲלֹ֖ךְ נֶ֥גֶד הַחַיִּֽים׃
9מַרְאֵ֥ה עֵינַ֖יִם מֵֽהֲלָךְ־ נָ֑פֶשׁ גַּם־ זֶ֥ה הֶ֖בֶל וּרְע֥וּת רֽוּחַ׃
10שֶּֽׁהָיָ֗ה כְּבָר֙ נִקְרָ֣א שְׁמ֔וֹ וְנוֹדָ֖ע אֲשֶׁר־ ה֣וּא אָדָ֑ם וְלֹא־ יוּכַ֣ל לָדִ֔ין עִ֥ם [שֶׁהַתְקִיף כ] (שֶׁתַּקִּ֖יף ק) מִמֶּֽנּוּ׃
11יֵשׁ־ דְּבָרִ֥ים הַרְבֵּ֖ה מַרְבִּ֣ים הָ֑בֶל מַה־ יֹּתֵ֖ר לָאָדָֽם׃
