1הַ֥לְלוּ יָ֨הּ ׀ הַֽלְל֣וּ אֶת־ יְ֭הוָה מִן־ הַשָּׁמַ֑יִם הַֽ֝לְל֗וּהוּ בַּמְּרוֹמִֽים׃
2כָל־ מַלְאָכָ֑יו הַֽ֝לְל֗וּהוּ כָּל־ [צְבָאֹו כ] (צְבָאָֽיו׃ ק)
3שֶׁ֣מֶשׁ וְיָרֵ֑חַ הַ֝לְל֗וּהוּ כָּל־ כּ֥וֹכְבֵי אֽוֹר׃
4שְׁמֵ֣י הַשָּׁמָ֑יִם וְ֝הַמַּ֗יִם אֲשֶׁ֤ר ׀ מֵעַ֬ל הַשָּׁמָֽיִם׃
5אֶת־ שֵׁ֣ם יְהוָ֑ה כִּ֤י ה֭וּא צִוָּ֣ה וְנִבְרָֽאוּ׃
6לָעַ֣ד לְעוֹלָ֑ם חָק־ נָ֝תַ֗ן וְלֹ֣א יַעֲבֽוֹר׃
7אֶת־ יְ֭הוָה מִן־ הָאָ֑רֶץ תַּ֝נִּינִ֗ים וְכָל־ תְּהֹמֽוֹת׃
8וּ֭בָרָד שֶׁ֣לֶג וְקִיט֑וֹר ר֥וּחַ סְ֝עָרָ֗ה עֹשָׂ֥ה דְבָרֽוֹ׃
9וְכָל־ גְּבָע֑וֹת עֵ֥ץ פְּ֝רִ֗י וְכָל־ אֲרָזִֽים׃
10וְכָל־ בְּהֵמָ֑ה רֶ֝֗מֶשׂ וְצִפּ֥וֹר כָּנָֽף׃
11אֶ֭רֶץ וְכָל־ לְאֻמִּ֑ים שָׂ֝רִ֗ים וְכָל־ שֹׁ֥פְטֵי אָֽרֶץ׃
12וְגַם־ בְּתוּל֑וֹת זְ֝קֵנִ֗ים עִם־ נְעָרִֽים׃
13אֶת־ שֵׁ֬ם יְהוָ֗ה כִּֽי־ נִשְׂגָּ֣ב שְׁמ֣וֹ לְבַדּ֑וֹ ה֝וֹד֗וֹ עַל־ אֶ֥רֶץ וְשָׁמָֽיִם׃
