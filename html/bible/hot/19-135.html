1הַ֥לְלוּ יָ֨הּ ׀ הַֽ֭לְלוּ אֶת־ שֵׁ֣ם יְהוָ֑ה הַֽ֝לְלוּ עַבְדֵ֥י יְהוָֽה׃
2בְּבֵ֣ית יְהוָ֑ה בְּ֝חַצְר֗וֹת בֵּ֣ית אֱלֹהֵֽינוּ׃
3יָהּ כִּי־ ט֣וֹב יְהוָ֑ה זַמְּר֥וּ לִ֝שְׁמ֗וֹ כִּ֣י נָעִֽים׃
4יַעֲקֹ֗ב בָּחַ֣ר ל֣וֹ יָ֑הּ יִ֝שְׂרָאֵ֗ל לִסְגֻלָּתֽוֹ׃
5אֲנִ֣י יָ֭דַעְתִּי כִּי־ גָד֣וֹל יְהוָ֑ה וַ֝אֲדֹנֵ֗ינוּ מִכָּל־ אֱלֹהִֽים׃
6אֲשֶׁר־ חָפֵ֥ץ יְהוָ֗ה עָ֫שָׂ֥ה בַּשָּׁמַ֥יִם וּבָאָ֑רֶץ בַּ֝יַּמִּ֗ים וְכָל־ תְּהוֹמֽוֹת׃
7נְשִׂאִים֮ מִקְצֵ֪ה הָ֫אָ֥רֶץ בְּרָקִ֣ים לַמָּטָ֣ר עָשָׂ֑ה מֽוֹצֵא־ ר֝וּחַ מֵאֽוֹצְרוֹתָֽיו׃
8בְּכוֹרֵ֣י מִצְרָ֑יִם מֵ֝אָדָ֗ם עַד־ בְּהֵמָֽה׃
9אֹת֣וֹת וּ֭מֹפְתִים בְּתוֹכֵ֣כִי מִצְרָ֑יִם בְּ֝פַרְעֹ֗ה וּבְכָל־ עֲבָדָֽיו׃
10גּוֹיִ֣ם רַבִּ֑ים וְ֝הָרַ֗ג מְלָכִ֥ים עֲצוּמִֽים׃
11מֶ֤לֶךְ הָאֱמֹרִ֗י וּ֭לְעוֹג מֶ֣לֶךְ הַבָּשָׁ֑ן וּ֝לְכֹ֗ל מַמְלְכ֥וֹת כְּנָֽעַן׃
12אַרְצָ֣ם נַחֲלָ֑ה נַ֝חֲלָ֗ה לְיִשְׂרָאֵ֥ל עַמּֽוֹ׃
13שִׁמְךָ֣ לְעוֹלָ֑ם יְ֝הוָ֗ה זִכְרְךָ֥ לְדֹר־ וָדֹֽר׃
14יָדִ֣ין יְהוָ֣ה עַמּ֑וֹ וְעַל־ עֲ֝בָדָ֗יו יִתְנֶחָֽם׃
15הַ֭גּוֹיִם כֶּ֣סֶף וְזָהָ֑ב מַ֝עֲשֵׂ֗ה יְדֵ֣י אָדָֽם׃
16לָ֭הֶם וְלֹ֣א יְדַבֵּ֑רוּ עֵינַ֥יִם לָ֝הֶ֗ם וְלֹ֣א יִרְאֽוּ׃
17לָ֭הֶם וְלֹ֣א יַאֲזִ֑ינוּ אַ֝֗ף אֵין־ יֶשׁ־ ר֥וּחַ בְּפִיהֶֽם׃
18יִהְי֣וּ עֹשֵׂיהֶ֑ם כֹּ֭ל אֲשֶׁר־ בֹּטֵ֣חַ בָּהֶֽם׃
19יִ֭שְׂרָאֵל בָּרֲכ֣וּ אֶת־ יְהוָ֑ה בֵּ֥ית אַ֝הֲרֹ֗ן בָּרֲכ֥וּ אֶת־ יְהוָֽה׃
20הַ֭לֵּוִי בָּרֲכ֣וּ אֶת־ יְהוָ֑ה יִֽרְאֵ֥י יְ֝הוָ֗ה בָּרֲכ֥וּ אֶת־ יְהוָֽה׃
