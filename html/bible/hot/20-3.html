1בְּ֭נִי תּוֹרָתִ֣י אַל־ תִּשְׁכָּ֑ח וּ֝מִצְוֹתַ֗י יִצֹּ֥ר לִבֶּֽךָ׃
2אֹ֣רֶךְ יָ֭מִים וּשְׁנ֣וֹת חַיִּ֑ים וְ֝שָׁל֗וֹם יוֹסִ֥יפוּ לָֽךְ׃
3וֶאֱמֶ֗ת אַֽל־ יַעַ֫זְבֻ֥ךָ קָשְׁרֵ֥ם עַל־ גַּרְגְּרוֹתֶ֑יךָ כָּ֝תְבֵ֗ם עַל־ ל֥וּחַ לִבֶּֽךָ׃
4חֵ֖ן וְשֵֽׂכֶל־ ט֑וֹב בְּעֵינֵ֖י אֱלֹהִ֣ים וְאָדָֽם׃ פ
5אֶל־ יְ֭הוָה בְּכָל־ לִבֶּ֑ךָ וְאֶל־ בִּֽ֝ינָתְךָ֗ אַל־ תִּשָּׁעֵֽן׃
6דְּרָכֶ֥יךָ דָעֵ֑הוּ וְ֝ה֗וּא יְיַשֵּׁ֥ר אֹֽרְחֹתֶֽיךָ׃
7תְּהִ֣י חָכָ֣ם בְּעֵינֶ֑יךָ יְרָ֥א אֶת־ יְ֝הוָ֗ה וְס֣וּר מֵרָֽע׃
8תְּהִ֣י לְשָׁרֶּ֑ךָ וְ֝שִׁקּ֗וּי לְעַצְמוֹתֶֽיךָ׃
9אֶת־ יְ֭הוָה מֵהוֹנֶ֑ךָ וּ֝מֵרֵאשִׁ֗ית כָּל־ תְּבוּאָתֶֽךָ׃
10אֲסָמֶ֣יךָ שָׂבָ֑ע וְ֝תִיר֗וֹשׁ יְקָבֶ֥יךָ יִפְרֹֽצוּ׃ פ
11יְ֭הוָה בְּנִ֣י אַל־ תִּמְאָ֑ס וְאַל־ תָּ֝קֹ֗ץ בְּתוֹכַחְתּֽוֹ׃
12אֶ֥ת אֲשֶׁ֣ר יֶאֱהַ֣ב יְהוָ֣ה יוֹכִ֑יחַ וּ֝כְאָ֗ב אֶת־ בֵּ֥ן יִרְצֶֽה׃
13אָ֭דָם מָצָ֣א חָכְמָ֑ה וְ֝אָדָ֗ם יָפִ֥יק תְּבוּנָֽה׃
14ט֣וֹב סַ֭חְרָהּ מִסְּחַר־ כָּ֑סֶף וּ֝מֵחָר֗וּץ תְּבוּאָתָֽהּ׃
15הִ֭יא [מִפְּנִיִּים כ] (מִפְּנִינִ֑ים ק) וְכָל־ חֲ֝פָצֶ֗יךָ לֹ֣א יִֽשְׁווּ־ בָֽהּ׃
16יָ֭מִים בִּֽימִינָ֑הּ בִּ֝שְׂמֹאולָ֗הּ עֹ֣שֶׁר וְכָבֽוֹד׃
17דַרְכֵי־ נֹ֑עַם וְֽכָל־ נְתִ֖יבוֹתֶ֣יהָ שָׁלֽוֹם׃
18חַיִּ֣ים הִ֭יא לַמַּחֲזִיקִ֣ים בָּ֑הּ וְֽתֹמְכֶ֥יהָ מְאֻשָּֽׁר׃ פ
19בְּחָכְמָ֥ה יָֽסַד־ אָ֑רֶץ כּוֹנֵ֥ן שָׁ֝מַ֗יִם בִּתְבוּנָֽה׃
20תְּהוֹמ֣וֹת נִבְקָ֑עוּ וּ֝שְׁחָקִ֗ים יִרְעֲפוּ־ טָֽל׃
21אַל־ יָלֻ֣זוּ מֵעֵינֶ֑יךָ נְצֹ֥ר תֻּ֝שִׁיָּ֗ה וּמְזִמָּֽה׃
22חַיִּ֣ים לְנַפְשֶׁ֑ךָ וְ֝חֵ֗ן לְגַרְגְּרֹתֶֽיךָ׃
23תֵּלֵ֣ךְ לָבֶ֣טַח דַּרְכֶּ֑ךָ וְ֝רַגְלְךָ֗ לֹ֣א תִגּֽוֹף׃
24תִּשְׁכַּ֥ב לֹֽא־ תִפְחָ֑ד וְ֝שָׁכַבְתָּ֗ וְֽעָרְבָ֥ה שְׁנָתֶֽךָ׃
25תִּ֭ירָא מִפַּ֣חַד פִּתְאֹ֑ם וּמִשֹּׁאַ֥ת רְ֝שָׁעִ֗ים כִּ֣י תָבֹֽא׃
26יְ֭הוָה יִהְיֶ֣ה בְכִסְלֶ֑ךָ וְשָׁמַ֖ר רַגְלְךָ֣ מִלָּֽכֶד׃ אַל־ תִּמְנַע־
27מִבְּעָלָ֑יו בִּהְי֨וֹת לְאֵ֖ל [יָדֶיךָ כ] (יָדְךָ֣ ק) לַעֲשֽׂוֹת׃
28תֹּ֘אמַ֤ר [לְרֵעֶיךָ כ] (לְרֵֽעֲךָ֨ ק) לֵ֣ךְ וָ֭שׁוּב וּמָחָ֥ר אֶתֵּ֗ן וְיֵ֣שׁ אִתָּֽךְ׃
29תַּחֲרֹ֣שׁ עַל־ רֵעֲךָ֣ רָעָ֑ה וְהֽוּא־ יוֹשֵׁ֖ב לָבֶ֣טַח אִתָּֽךְ׃
30[תָּרֹוב כ] (תָּרִ֣יב ק) עִם־ אָדָ֣ם חִנָּ֑ם אִם־ לֹ֖א גְמָלְךָ֣ רָעָֽה׃
31תְּ֭קַנֵּא בְּאִ֣ישׁ חָמָ֑ס וְאַל־ תִּ֝בְחַ֗ר בְּכָל־ דְּרָכָֽיו׃
32תוֹעֲבַ֣ת יְהוָ֣ה נָל֑וֹז וְֽאֶת־ יְשָׁרִ֥ים סוֹדֽוֹ׃
33יְ֭הוָה בְּבֵ֣ית רָשָׁ֑ע וּנְוֵ֖ה צַדִּיקִ֣ים יְבָרֵֽךְ׃
34לַלֵּצִ֥ים הֽוּא־ יָלִ֑יץ [וְלַעֲנִיִּים כ] (וְ֝לַעֲנָוִ֗ים ק) יִתֶּן־ חֵֽן׃
