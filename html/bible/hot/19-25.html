1לְדָוִ֡ד אֵלֶ֥יךָ יְ֝הוָ֗ה נַפְשִׁ֥י אֶשָּֽׂא׃
2בְּךָ֣ בָ֭טַחְתִּי אַל־ אֵב֑וֹשָׁה אַל־ יַֽעַלְצ֖וּ אֹיְבַ֣י לִֽי׃
3כָּל־ קֹ֭וֶיךָ לֹ֣א יֵבֹ֑שׁוּ יֵ֝בֹ֗שׁוּ הַבּוֹגְדִ֥ים רֵיקָֽם׃
4יְ֭הוָה הוֹדִיעֵ֑נִי אֹ֖רְחוֹתֶ֣יךָ לַמְּדֵֽנִי׃
5בַאֲמִתֶּ֨ךָ ׀ וְֽלַמְּדֵ֗נִי כִּֽי־ אַ֭תָּה אֱלֹהֵ֣י יִשְׁעִ֑י אוֹתְךָ֥ קִ֝וִּ֗יתִי כָּל־ הַיּֽוֹם׃
6רַחֲמֶ֣יךָ יְ֭הוָה וַחֲסָדֶ֑יךָ כִּ֖י מֵעוֹלָ֣ם הֵֽמָּה׃
7נְעוּרַ֨י ׀ וּפְשָׁעַ֗י אַל־ תִּ֫זְכֹּ֥ר כְּחַסְדְּךָ֥ זְכָר־ לִי־ אַ֑תָּה לְמַ֖עַן טוּבְךָ֣ יְהוָֽה׃
8וְיָשָׁ֥ר יְהוָ֑ה עַל־ כֵּ֤ן יוֹרֶ֖ה חַטָּאִ֣ים בַּדָּֽרֶךְ׃
9עֲ֭נָוִים בַּמִּשְׁפָּ֑ט וִֽילַמֵּ֖ד עֲנָוִ֣ים דַּרְכּֽוֹ׃
10אָרְח֣וֹת יְ֭הוָה חֶ֣סֶד וֶאֱמֶ֑ת לְנֹצְרֵ֥י בְ֝רִית֗וֹ וְעֵדֹתָֽיו׃
11שִׁמְךָ֥ יְהוָ֑ה וְֽסָלַחְתָּ֥ לַ֝עֲוֹנִ֗י כִּ֣י רַב־ הֽוּא׃
12זֶ֣ה הָ֭אִישׁ יְרֵ֣א יְהוָ֑ה י֝וֹרֶ֗נּוּ בְּדֶ֣רֶךְ יִבְחָֽר׃
13בְּט֣וֹב תָּלִ֑ין וְ֝זַרְע֗וֹ יִ֣ירַשׁ אָֽרֶץ׃
14יְ֭הוָה לִירֵאָ֑יו וּ֝בְרִית֗וֹ לְהוֹדִיעָֽם׃
15תָּ֭מִיד אֶל־ יְהוָ֑ה כִּ֤י הֽוּא־ יוֹצִ֖יא מֵרֶ֣שֶׁת רַגְלָֽי׃
16אֵלַ֥י וְחָנֵּ֑נִי כִּֽי־ יָחִ֖יד וְעָנִ֣י אָֽנִי׃
17לְבָבִ֣י הִרְחִ֑יבוּ מִ֝מְּצֽוּקוֹתַ֗י הוֹצִיאֵֽנִי׃
18עָ֭נְיִי וַעֲמָלִ֑י וְ֝שָׂ֗א לְכָל־ חַטֹּאותָֽי׃
19אוֹיְבַ֥י כִּי־ רָ֑בּוּ וְשִׂנְאַ֖ת חָמָ֣ס שְׂנֵאֽוּנִי׃
20נַ֭פְשִׁי וְהַצִּילֵ֑נִי אַל־ אֵ֝ב֗וֹשׁ כִּֽי־ חָסִ֥יתִי בָֽךְ׃
21וָיֹ֥שֶׁר יִצְּר֑וּנִי כִּ֝֗י קִוִּיתִֽיךָ׃
