1דִּבְרֵ֤י ׀ אָג֥וּר בִּן־ יָקֶ֗ה הַמַּ֫שָּׂ֥א נְאֻ֣ם הַ֭גֶּבֶר לְאִֽיתִיאֵ֑ל לְאִ֖יתִיאֵ֣ל וְאֻכָֽל׃
2בַ֣עַר אָנֹכִ֣י מֵאִ֑ישׁ וְלֹֽא־ בִינַ֖ת אָדָ֣ם לִֽי׃
3לָמַ֥דְתִּי חָכְמָ֑ה וְדַ֖עַת קְדֹשִׁ֣ים אֵדָֽע׃
4עָלָֽה־ שָׁמַ֨יִם ׀ וַיֵּרַ֡ד מִ֤י אָֽסַף־ ר֨וּחַ ׀ בְּחָפְנָ֡יו מִ֤י צָֽרַר־ מַ֨יִם ׀ בַּשִּׂמְלָ֗ה מִ֭י הֵקִ֣ים כָּל־ אַפְסֵי־ אָ֑רֶץ מַה־ שְּׁמ֥וֹ וּמַֽה־ שֶּׁם־ בְּ֝נ֗וֹ כִּ֣י תֵדָֽע׃
5אִמְרַ֣ת אֱל֣וֹהַּ צְרוּפָ֑ה מָגֵ֥ן ה֝֗וּא לַֽחֹסִ֥ים בּֽוֹ׃
6תּ֥וֹסְףְּ עַל־ דְּבָרָ֑יו פֶּן־ יוֹכִ֖יחַ בְּךָ֣ וְנִכְזָֽבְתָּ׃ פ
7שָׁאַ֣לְתִּי מֵאִתָּ֑ךְ אַל־ תִּמְנַ֥ע מִ֝מֶּ֗נִּי בְּטֶ֣רֶם אָמֽוּת׃
8וּֽדְבַר־ כָּזָ֡ב הַרְחֵ֬ק מִמֶּ֗נִּי רֵ֣אשׁ וָ֭עֹשֶׁר אַל־ תִּֽתֶּן־ לִ֑י הַ֝טְרִיפֵ֗נִי לֶ֣חֶם חֻקִּֽי׃
9אֶשְׂבַּ֨ע ׀ וְכִחַשְׁתִּי֮ וְאָמַ֗רְתִּי מִ֥י יְה֫וָ֥ה וּפֶֽן־ אִוָּרֵ֥שׁ וְגָנַ֑בְתִּי וְ֝תָפַ֗שְׂתִּי שֵׁ֣ם אֱלֹהָֽי׃ פ
10תַּלְשֵׁ֣ן עֶ֭בֶד אֶל־ [אֲדֹנֹו כ] (אֲדֹנָ֑יו ק) פֶּֽן־ יְקַלֶּלְךָ֥ וְאָשָֽׁמְתָּ׃
11אָבִ֣יו יְקַלֵּ֑ל וְאֶת־ אִ֝מּ֗וֹ לֹ֣א יְבָרֵֽךְ׃
12טָה֣וֹר בְּעֵינָ֑יו וּ֝מִצֹּאָת֗וֹ לֹ֣א רֻחָֽץ׃
13מָה־ רָמ֣וּ עֵינָ֑יו וְ֝עַפְעַפָּ֗יו יִנָּשֵֽׂאוּ׃
14חֲרָב֣וֹת שִׁנָּיו֮ וּֽמַאֲכָל֪וֹת מְֽתַלְּעֹ֫תָ֥יו לֶאֱכֹ֣ל עֲנִיִּ֣ים מֵאֶ֑רֶץ וְ֝אֶבְיוֹנִ֗ים מֵאָדָֽם׃ פ
15שְׁתֵּ֥י בָנוֹת֮ הַ֤ב ׀ הַ֥ב שָׁל֣וֹשׁ הֵ֭נָּה לֹ֣א תִשְׂבַּ֑עְנָה אַ֝רְבַּ֗ע לֹא־ אָ֥מְרוּ הֽוֹן׃
16וְעֹ֪צֶ֫ר רָ֥חַם אֶ֭רֶץ לֹא־ שָׂ֣בְעָה מַּ֑יִם וְ֝אֵ֗שׁ לֹא־ אָ֥מְרָה הֽוֹן׃
17תִּֽלְעַ֣ג לְאָב֮ וְתָב֪וּז לִֽיקֲּהַ֫ת־ אֵ֥ם יִקְּר֥וּהָ עֹרְבֵי־ נַ֑חַל וְֽיֹאכְל֥וּהָ בְנֵי־ נָֽשֶׁר׃ פ
18הֵ֭מָּה נִפְלְא֣וּ מִמֶּ֑נִּי [וְאַרְבָּע כ] (וְ֝אַרְבָּעָ֗ה ק) לֹ֣א יְדַעְתִּֽים׃
19הַנֶּ֨שֶׁר ׀ בַּשָּׁמַיִם֮ דֶּ֥רֶךְ נָחָ֗שׁ עֲלֵ֫י צ֥וּר דֶּֽרֶךְ־ אֳנִיָּ֥ה בְלֶב־ יָ֑ם וְדֶ֖רֶךְ גֶּ֣בֶר בְּעַלְמָֽה׃
20דֶּ֥רֶךְ אִשָּׁ֗ה מְנָ֫אָ֥פֶת אָ֭כְלָה וּמָ֣חֲתָה פִ֑יהָ וְ֝אָמְרָ֗ה לֹֽא־ פָעַ֥לְתִּי אָֽוֶן׃ פ
21שָׁ֭לוֹשׁ רָ֣גְזָה אֶ֑רֶץ וְתַ֥חַת אַ֝רְבַּ֗ע לֹא־ תוּכַ֥ל שְׂאֵֽת׃
22עֶ֭בֶד כִּ֣י יִמְל֑וֹךְ וְ֝נָבָ֗ל כִּ֣י יִֽשְׂבַּֽע־ לָֽחֶם׃
23שְׂ֭נוּאָה כִּ֣י תִבָּעֵ֑ל וְ֝שִׁפְחָ֗ה כִּֽי־ תִירַ֥שׁ גְּבִרְתָּֽהּ׃ פ
24הֵ֭ם קְטַנֵּי־ אָ֑רֶץ וְ֝הֵ֗מָּה חֲכָמִ֥ים מְחֻכָּמִֽים׃
25עַ֣ם לֹא־ עָ֑ז וַיָּכִ֖ינוּ בַקַּ֣יִץ לַחְמָֽם׃
26עַ֣ם לֹא־ עָצ֑וּם וַיָּשִׂ֖ימוּ בַסֶּ֣לַע בֵּיתָֽם׃
27אֵ֣ין לָאַרְבֶּ֑ה וַיֵּצֵ֖א חֹצֵ֣ץ כֻּלּֽוֹ׃
28בְּיָדַ֣יִם תְּתַפֵּ֑שׂ וְ֝הִ֗יא בְּהֵ֣יכְלֵי מֶֽלֶךְ׃ פ
29הֵ֭מָּה מֵיטִ֣יבֵי צָ֑עַד וְ֝אַרְבָּעָ֗ה מֵיטִ֥בֵי לָֽכֶת׃
30גִּבּ֣וֹר בַּבְּהֵמָ֑ה וְלֹא־ יָ֝שׁ֗וּב מִפְּנֵי־ כֹֽל׃
31מָתְנַ֣יִם אוֹ־ תָ֑יִשׁ וּ֝מֶ֗לֶךְ אַלְק֥וּם עִמּֽוֹ׃
32נָבַ֥לְתָּ בְהִתְנַשֵּׂ֑א וְאִם־ זַ֝מּ֗וֹתָ יָ֣ד לְפֶֽה׃
