1הַדָּבָ֞ר אֲשֶׁר־ הָיָ֤ה עַֽל־ יִרְמְיָ֙הוּ֙ עַל־ כָּל־ עַ֣ם יְהוּדָ֔ה בַּשָּׁנָה֙ הָֽרְבִעִ֔ית לִיהוֹיָקִ֥ים בֶּן־ יֹאשִׁיָּ֖הוּ מֶ֣לֶךְ יְהוּדָ֑ה הִ֗יא הַשָּׁנָה֙ הָרִ֣אשֹׁנִ֔ית לִנְבֽוּכַדְרֶאצַּ֖ר מֶ֥לֶךְ בָּבֶֽל׃
2דִּבֶּ֜ר יִרְמְיָ֤הוּ הַנָּבִיא֙ עַל־ כָּל־ עַ֣ם יְהוּדָ֔ה וְאֶ֛ל כָּל־ יֹשְׁבֵ֥י יְרוּשָׁלִַ֖ם לֵאמֹֽר׃
3שְׁלֹ֣שׁ עֶשְׂרֵ֣ה שָׁנָ֡ה לְיֹאשִׁיָּ֣הוּ בֶן־ אָמוֹן֩ מֶ֨לֶךְ יְהוּדָ֜ה וְעַ֣ד ׀ הַיּ֣וֹם הַזֶּ֗ה זֶ֚ה שָׁלֹ֤שׁ וְעֶשְׂרִים֙ שָׁנָ֔ה הָיָ֥ה דְבַר־ יְהוָ֖ה אֵלָ֑י וָאֲדַבֵּ֧ר אֲלֵיכֶ֛ם אַשְׁכֵּ֥ים וְדַבֵּ֖ר וְלֹ֥א שְׁמַעְתֶּֽם׃
4יְהוָ֨ה אֲלֵיכֶ֜ם אֶֽת־ כָּל־ עֲבָדָ֧יו הַנְּבִאִ֛ים הַשְׁכֵּ֥ם וְשָׁלֹ֖חַ וְלֹ֣א שְׁמַעְתֶּ֑ם וְלֹֽא־ הִטִּיתֶ֥ם אֶֽת־ אָזְנְכֶ֖ם לִשְׁמֹֽעַ׃
5שֽׁוּבוּ־ נָ֞א אִ֣ישׁ מִדַּרְכּ֤וֹ הָֽרָעָה֙ וּמֵרֹ֣עַ מַעַלְלֵיכֶ֔ם וּשְׁבוּ֙ עַל־ הָ֣אֲדָמָ֔ה אֲשֶׁ֨ר נָתַ֧ן יְהוָ֛ה לָכֶ֖ם וְלַאֲבֽוֹתֵיכֶ֑ם לְמִן־ עוֹלָ֖ם וְעַד־ עוֹלָֽם׃
6תֵּלְכ֗וּ אַֽחֲרֵי֙ אֱלֹהִ֣ים אֲחֵרִ֔ים לְעָבְדָ֖ם וּלְהִשְׁתַּחֲוֹ֣ת לָהֶ֑ם וְלֹֽא־ תַכְעִ֤יסוּ אוֹתִי֙ בְּמַעֲשֵׂ֣ה יְדֵיכֶ֔ם וְלֹ֥א אָרַ֖ע לָכֶֽם׃
7שְׁמַעְתֶּ֥ם אֵלַ֖י נְאֻם־ יְהוָ֑ה לְמַ֧עַן [הִכְעִסוּנִי כ] (הַכְעִיסֵ֛נִי ק) בְּמַעֲשֵׂ֥ה יְדֵיכֶ֖ם לְרַ֥ע לָכֶֽם׃ ס
8כֹּ֥ה אָמַ֖ר יְהוָ֣ה צְבָא֑וֹת יַ֕עַן אֲשֶׁ֥ר לֹֽא־ שְׁמַעְתֶּ֖ם אֶת־ דְּבָרָֽי׃
9שֹׁלֵ֡חַ וְלָקַחְתִּי֩ אֶת־ כָּל־ מִשְׁפְּח֨וֹת צָפ֜וֹן נְאֻם־ יְהוָ֗ה וְאֶל־ נְבֽוּכַדְרֶאצַּ֣ר מֶֽלֶךְ־ בָּבֶל֮ עַבְדִּי֒ וַהֲבִ֨אֹתִ֜ים עַל־ הָאָ֤רֶץ הַזֹּאת֙ וְעַל־ יֹ֣שְׁבֶ֔יהָ וְעַ֛ל כָּל־ הַגּוֹיִ֥ם הָאֵ֖לֶּה סָבִ֑יב וְהַ֣חֲרַמְתִּ֔ים וְשַׂמְתִּים֙ לְשַׁמָּ֣ה וְלִשְׁרֵקָ֔ה וּלְחָרְב֖וֹת עוֹלָֽם׃
10מֵהֶ֗ם ק֤וֹל שָׂשׂוֹן֙ וְק֣וֹל שִׂמְחָ֔ה ק֥וֹל חָתָ֖ן וְק֣וֹל כַּלָּ֑ה ק֥וֹל רֵחַ֖יִם וְא֥וֹר נֵֽר׃
11כָּל־ הָאָ֣רֶץ הַזֹּ֔את לְחָרְבָּ֖ה לְשַׁמָּ֑ה וְעָ֨בְד֜וּ הַגּוֹיִ֥ם הָאֵ֛לֶּה אֶת־ מֶ֥לֶךְ בָּבֶ֖ל שִׁבְעִ֥ים שָׁנָֽה׃
12כִמְלֹ֣אות שִׁבְעִ֣ים שָׁנָ֡ה אֶפְקֹ֣ד עַל־ מֶֽלֶךְ־ בָּבֶל֩ וְעַל־ הַגּ֨וֹי הַה֧וּא נְאֻם־ יְהוָ֛ה אֶת־ עֲוֹנָ֖ם וְעַל־ אֶ֣רֶץ כַּשְׂדִּ֑ים וְשַׂמְתִּ֥י אֹת֖וֹ לְשִֽׁמְמ֥וֹת עוֹלָֽם׃
13כ] (וְהֵֽבֵאתִי֙ ק) עַל־ הָאָ֣רֶץ הַהִ֔יא אֶת־ כָּל־ דְּבָרַ֖י אֲשֶׁר־ דִּבַּ֣רְתִּי עָלֶ֑יהָ אֵ֤ת כָּל־ הַכָּתוּב֙ בַּסֵּ֣פֶר הַזֶּ֔ה אֲשֶׁר־ נִבָּ֥א יִרְמְיָ֖הוּ עַל־ כָּל־ הַגּוֹיִֽם׃
14עָֽבְדוּ־ בָ֤ם גַּם־ הֵ֙מָּה֙ גּוֹיִ֣ם רַבִּ֔ים וּמְלָכִ֖ים גְּדוֹלִ֑ים וְשִׁלַּמְתִּ֥י לָהֶ֛ם כְּפָעֳלָ֖ם וּכְמַעֲשֵׂ֥ה יְדֵיהֶֽם׃ ס
15כֹה֩ אָמַ֨ר יְהוָ֜ה אֱלֹהֵ֤י יִשְׂרָאֵל֙ אֵלַ֔י קַ֠ח אֶת־ כּ֨וֹס הַיַּ֧יִן הַחֵמָ֛ה הַזֹּ֖את מִיָּדִ֑י וְהִשְׁקִיתָ֤ה אֹתוֹ֙ אֶת־ כָּל־ הַגּוֹיִ֔ם אֲשֶׁ֧ר אָנֹכִ֛י שֹׁלֵ֥חַ אוֹתְךָ֖ אֲלֵיהֶֽם׃
16וְהִֽתְגֹּֽעֲשׁ֖וּ וְהִתְהֹלָ֑לוּ מִפְּנֵ֣י הַחֶ֔רֶב אֲשֶׁ֛ר אָנֹכִ֥י שֹׁלֵ֖חַ בֵּינֹתָֽם׃
17אֶת־ הַכּ֖וֹס מִיַּ֣ד יְהוָ֑ה וָֽאַשְׁקֶה֙ אֶת־ כָּל־ הַגּוֹיִ֔ם אֲשֶׁר־ שְׁלָחַ֥נִי יְהוָ֖ה אֲלֵיהֶֽם׃
18יְרוּשָׁלִַ֙ם֙ וְאֶת־ עָרֵ֣י יְהוּדָ֔ה וְאֶת־ מְלָכֶ֖יהָ אֶת־ שָׂרֶ֑יהָ לָתֵ֨ת אֹתָ֜ם לְחָרְבָּ֧ה לְשַׁמָּ֛ה לִשְׁרֵקָ֥ה וְלִקְלָלָ֖ה כַּיּ֥וֹם הַזֶּֽה׃
19פַּרְעֹ֧ה מֶֽלֶךְ־ מִצְרַ֛יִם וְאֶת־ עֲבָדָ֥יו וְאֶת־ שָׂרָ֖יו וְאֶת־ כָּל־ עַמּֽוֹ׃
20כָּל־ הָעֶ֔רֶב וְאֵ֕ת כָּל־ מַלְכֵ֖י אֶ֣רֶץ הָע֑וּץ וְאֵ֗ת כָּל־ מַלְכֵי֙ אֶ֣רֶץ פְּלִשְׁתִּ֔ים וְאֶת־ אַשְׁקְל֤וֹן וְאֶת־ עַזָּה֙ וְאֶת־ עֶקְר֔וֹן וְאֵ֖ת שְׁאֵרִ֥ית אַשְׁדּֽוֹד׃
21אֱד֥וֹם וְאֶת־ מוֹאָ֖ב וְאֶת־ בְּנֵ֥י עַמּֽוֹן׃
22כָּל־ מַלְכֵי־ צֹ֔ר וְאֵ֖ת כָּל־ מַלְכֵ֣י צִיד֑וֹן וְאֵת֙ מַלְכֵ֣י הָאִ֔י אֲשֶׁ֖ר בְּעֵ֥בֶר הַיָּֽם׃
23דְּדָ֤ן וְאֶת־ תֵּימָא֙ וְאֶת־ בּ֔וּז וְאֵ֖ת כָּל־ קְצוּצֵ֥י פֵאָֽה׃
24כָּל־ מַלְכֵ֣י עֲרָ֑ב וְאֵת֙ כָּל־ מַלְכֵ֣י הָעֶ֔רֶב הַשֹּׁכְנִ֖ים בַּמִּדְבָּֽר׃
25כָּל־ מַלְכֵ֣י זִמְרִ֗י וְאֵת֙ כָּל־ מַלְכֵ֣י עֵילָ֔ם וְאֵ֖ת כָּל־ מַלְכֵ֥י מָדָֽי׃
26כָּל־ מַלְכֵ֣י הַצָּפ֗וֹן הַקְּרֹבִ֤ים וְהָֽרְחֹקִים֙ אִ֣ישׁ אֶל־ אָחִ֔יו וְאֵת֙ כָּל־ הַמַּמְלְכ֣וֹת הָאָ֔רֶץ אֲשֶׁ֖ר עַל־ פְּנֵ֣י הָאֲדָמָ֑ה וּמֶ֥לֶךְ שֵׁשַׁ֖ךְ יִשְׁתֶּ֥ה אַחֲרֵיהֶֽם׃
27אֲלֵיהֶ֡ם ס כֹּֽה־ אָמַר֩ יְהוָ֨ה צְבָא֜וֹת אֱלֹהֵ֣י יִשְׂרָאֵ֗ל שְׁת֤וּ וְשִׁכְרוּ֙ וּקְי֔וּ וְנִפְל֖וּ וְלֹ֣א תָק֑וּמוּ מִפְּנֵ֣י הַחֶ֔רֶב אֲשֶׁ֛ר אָנֹכִ֥י שֹׁלֵ֖חַ בֵּינֵיכֶֽם׃
28כִּ֧י יְמָאֲנ֛וּ לָקַֽחַת־ הַכּ֥וֹס מִיָּדְךָ֖ לִשְׁתּ֑וֹת וְאָמַרְתָּ֣ אֲלֵיהֶ֗ם כֹּ֥ה אָמַ֛ר יְהוָ֥ה צְבָא֖וֹת שָׁת֥וֹ תִשְׁתּֽוּ׃
29הִנֵּ֨ה בָעִ֜יר אֲשֶׁ֧ר נִֽקְרָא־ שְׁמִ֣י עָלֶ֗יהָ אָֽנֹכִי֙ מֵחֵ֣ל לְהָרַ֔ע וְאַתֶּ֖ם הִנָּקֵ֣ה תִנָּק֑וּ לֹ֣א תִנָּק֔וּ כִּ֣י חֶ֗רֶב אֲנִ֤י קֹרֵא֙ עַל־ כָּל־ יֹשְׁבֵ֣י הָאָ֔רֶץ נְאֻ֖ם יְהוָ֥ה צְבָאֽוֹת׃
30תִּנָּבֵ֣א אֲלֵיהֶ֔ם אֵ֥ת כָּל־ הַדְּבָרִ֖ים הָאֵ֑לֶּה וְאָמַרְתָּ֣ אֲלֵיהֶ֗ם יְהוָ֞ה מִמָּר֤וֹם יִשְׁאָג֙ וּמִמְּע֤וֹן קָדְשׁוֹ֙ יִתֵּ֣ן קוֹל֔וֹ שָׁאֹ֤ג יִשְׁאַג֙ עַל־ נָוֵ֔הוּ הֵידָד֙ כְּדֹרְכִ֣ים יַֽעֲנֶ֔ה אֶ֥ל כָּל־ יֹשְׁבֵ֖י הָאָֽרֶץ׃
31שָׁאוֹן֙ עַד־ קְצֵ֣ה הָאָ֔רֶץ כִּ֣י רִ֤יב לַֽיהוָה֙ בַּגּוֹיִ֔ם נִשְׁפָּ֥ט ה֖וּא לְכָל־ בָּשָׂ֑ר הָרְשָׁעִ֛ים נְתָנָ֥ם לַחֶ֖רֶב נְאֻם־ יְהוָֽה׃ ס
32אָמַר֙ יְהוָ֣ה צְבָא֔וֹת הִנֵּ֥ה רָעָ֛ה יֹצֵ֖את מִגּ֣וֹי אֶל־ גּ֑וֹי וְסַ֣עַר גָּד֔וֹל יֵע֖וֹר מִיַּרְכְּתֵי־ אָֽרֶץ׃
33חַֽלְלֵ֤י יְהוָה֙ בַּיּ֣וֹם הַה֔וּא מִקְצֵ֥ה הָאָ֖רֶץ וְעַד־ קְצֵ֣ה הָאָ֑רֶץ לֹ֣א יִסָּפְד֗וּ וְלֹ֤א יֵאָֽסְפוּ֙ וְלֹ֣א יִקָּבֵ֔רוּ לְדֹ֛מֶן עַל־ פְּנֵ֥י הָאֲדָמָ֖ה יִֽהְיֽוּ׃
34הָרֹעִ֜ים וְזַעֲק֗וּ וְהִֽתְפַּלְּשׁוּ֙ אַדִּירֵ֣י הַצֹּ֔אן כִּֽי־ מָלְא֥וּ יְמֵיכֶ֖ם לִטְב֑וֹחַ וּתְפוֹצ֣וֹתִיכֶ֔ם וּנְפַלְתֶּ֖ם כִּכְלִ֥י חֶמְדָּֽה׃
35מָנ֖וֹס מִן־ הָֽרֹעִ֑ים וּפְלֵיטָ֖ה מֵאַדִּירֵ֥י הַצֹּֽאן׃
36צַעֲקַ֣ת הָֽרֹעִ֔ים וִֽילְלַ֖ת אַדִּירֵ֣י הַצֹּ֑אן כִּֽי־ שֹׁדֵ֥ד יְהוָ֖ה אֶת־ מַרְעִיתָֽם׃
37נְא֣וֹת הַשָּׁל֑וֹם מִפְּנֵ֖י חֲר֥וֹן אַף־ יְהוָֽה׃
