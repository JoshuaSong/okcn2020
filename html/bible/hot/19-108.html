1שִׁ֖יר מִזְמ֣וֹר לְדָוִֽד׃ נָכ֣וֹן לִבִּ֣י אֱלֹהִ֑ים אָשִׁ֥ירָה וַ֝אֲזַמְּרָ֗ה אַף־ כְּבוֹדִֽי׃
2הַנֵּ֥בֶל וְכִנּ֗וֹר אָעִ֥ירָה שָּֽׁחַר׃
3בָעַמִּ֥ים ׀ יְהוָ֑ה וַ֝אֲזַמֶּרְךָ֗ בַּל־ אֻמִּֽים׃
4גָד֣וֹל מֵֽעַל־ שָׁמַ֣יִם חַסְדֶּ֑ךָ וְֽעַד־ שְׁחָקִ֥ים אֲמִתֶּֽךָ׃
5עַל־ שָׁמַ֣יִם אֱלֹהִ֑ים וְעַ֖ל כָּל־ הָאָ֣רֶץ כְּבוֹדֶֽךָ׃
6יֵחָלְצ֣וּן יְדִידֶ֑יךָ הוֹשִׁ֖יעָה יְמִֽינְךָ֣ וַעֲנֵֽנִי׃
7דִּבֶּ֥ר בְּקָדְשׁ֗וֹ אֶעְלֹ֥זָה אֲחַלְּקָ֥ה שְׁכֶ֑ם וְעֵ֖מֶק סֻכּ֣וֹת אֲמַדֵּֽד׃
8גִלְעָ֨ד ׀ לִ֤י מְנַשֶּׁ֗ה וְ֭אֶפְרַיִם מָע֣וֹז רֹאשִׁ֑י יְ֝הוּדָ֗ה מְחֹקְקִֽי׃
9סִ֬יר רַחְצִ֗י עַל־ אֱ֭דוֹם אַשְׁלִ֣יךְ נַעֲלִ֑י עֲלֵֽי־ פְ֝לֶ֗שֶׁת אֶתְרוֹעָֽע׃
10יֹ֭בִלֵנִי עִ֣יר מִבְצָ֑ר מִ֖י נָחַ֣נִי עַד־ אֱדֽוֹם׃
11אֱלֹהִ֥ים זְנַחְתָּ֑נוּ וְֽלֹא־ תֵצֵ֥א אֱ֝לֹהִ֗ים בְּצִבְאֹתֵֽינוּ׃
12לָּ֣נוּ עֶזְרָ֣ת מִצָּ֑ר וְ֝שָׁ֗וְא תְּשׁוּעַ֥ת אָדָֽם׃
