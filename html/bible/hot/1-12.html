1וַיֹּ֤אמֶר יְהוָה֙ אֶל־ אַבְרָ֔ם לֶךְ־ לְךָ֛ מֵאַרְצְךָ֥ וּמִמּֽוֹלַדְתְּךָ֖ וּמִבֵּ֣ית אָבִ֑יךָ אֶל־ הָאָ֖רֶץ אֲשֶׁ֥ר אַרְאֶֽךָּ׃
2לְג֣וֹי גָּד֔וֹל וַאֲבָ֣רֶכְךָ֔ וַאֲגַדְּלָ֖ה שְׁמֶ֑ךָ וֶהְיֵ֖ה בְּרָכָֽה׃
3מְבָ֣רְכֶ֔יךָ וּמְקַלֶּלְךָ֖ אָאֹ֑ר וְנִבְרְכ֣וּ בְךָ֔ כֹּ֖ל מִשְׁפְּחֹ֥ת הָאֲדָמָֽה׃
4אַבְרָ֗ם כַּאֲשֶׁ֨ר דִּבֶּ֤ר אֵלָיו֙ יְהוָ֔ה וַיֵּ֥לֶךְ אִתּ֖וֹ ל֑וֹט וְאַבְרָ֗ם בֶּן־ חָמֵ֤שׁ שָׁנִים֙ וְשִׁבְעִ֣ים שָׁנָ֔ה בְּצֵאת֖וֹ מֵחָרָֽן׃
5אַבְרָם֩ אֶת־ שָׂרַ֨י אִשְׁתּ֜וֹ וְאֶת־ ל֣וֹט בֶּן־ אָחִ֗יו וְאֶת־ כָּל־ רְכוּשָׁם֙ אֲשֶׁ֣ר רָכָ֔שׁוּ וְאֶת־ הַנֶּ֖פֶשׁ אֲשֶׁר־ עָשׂ֣וּ בְחָרָ֑ן וַיֵּצְא֗וּ לָלֶ֙כֶת֙ אַ֣רְצָה כְּנַ֔עַן וַיָּבֹ֖אוּ אַ֥רְצָה כְּנָֽעַן׃
6אַבְרָם֙ בָּאָ֔רֶץ עַ֚ד מְק֣וֹם שְׁכֶ֔ם עַ֖ד אֵל֣וֹן מוֹרֶ֑ה וְהַֽכְּנַעֲנִ֖י אָ֥ז בָּאָֽרֶץ׃
7יְהוָה֙ אֶל־ אַבְרָ֔ם וַיֹּ֕אמֶר לְזַ֨רְעֲךָ֔ אֶתֵּ֖ן אֶת־ הָאָ֣רֶץ הַזֹּ֑את וַיִּ֤בֶן שָׁם֙ מִזְבֵּ֔חַ לַיהוָ֖ה הַנִּרְאֶ֥ה אֵלָֽיו׃
8מִשָּׁ֜ם הָהָ֗רָה מִקֶּ֛דֶם לְבֵֽית־ אֵ֖ל וַיֵּ֣ט אָהֳלֹ֑ה בֵּֽית־ אֵ֤ל מִיָּם֙ וְהָעַ֣י מִקֶּ֔דֶם וַיִּֽבֶן־ שָׁ֤ם מִזְבֵּ֙חַ֙ לַֽיהוָ֔ה וַיִּקְרָ֖א בְּשֵׁ֥ם יְהוָֽה׃
9אַבְרָ֔ם הָל֥וֹךְ וְנָס֖וֹעַ הַנֶּֽגְבָּה׃ פ
10רָעָ֖ב בָּאָ֑רֶץ וַיֵּ֨רֶד אַבְרָ֤ם מִצְרַ֙יְמָה֙ לָג֣וּר שָׁ֔ם כִּֽי־ כָבֵ֥ד הָרָעָ֖ב בָּאָֽרֶץ׃
11כַּאֲשֶׁ֥ר הִקְרִ֖יב לָב֣וֹא מִצְרָ֑יְמָה וַיֹּ֙אמֶר֙ אֶל־ שָׂרַ֣י אִשְׁתּ֔וֹ הִנֵּה־ נָ֣א יָדַ֔עְתִּי כִּ֛י אִשָּׁ֥ה יְפַת־ מַרְאֶ֖ה אָֽתְּ׃
12כִּֽי־ יִרְא֤וּ אֹתָךְ֙ הַמִּצְרִ֔ים וְאָמְר֖וּ אִשְׁתּ֣וֹ זֹ֑את וְהָרְג֥וּ אֹתִ֖י וְאֹתָ֥ךְ יְחַיּֽוּ׃
13נָ֖א אֲחֹ֣תִי אָ֑תְּ לְמַ֙עַן֙ יִֽיטַב־ לִ֣י בַעֲבוּרֵ֔ךְ וְחָיְתָ֥ה נַפְשִׁ֖י בִּגְלָלֵֽךְ׃
14כְּב֥וֹא אַבְרָ֖ם מִצְרָ֑יְמָה וַיִּרְא֤וּ הַמִּצְרִים֙ אֶת־ הָ֣אִשָּׁ֔ה כִּֽי־ יָפָ֥ה הִ֖וא מְאֹֽד׃
15אֹתָהּ֙ שָׂרֵ֣י פַרְעֹ֔ה וַיְהַֽלְל֥וּ אֹתָ֖הּ אֶל־ פַּרְעֹ֑ה וַתֻּקַּ֥ח הָאִשָּׁ֖ה בֵּ֥ית פַּרְעֹֽה׃
16הֵיטִ֖יב בַּעֲבוּרָ֑הּ וַֽיְהִי־ ל֤וֹ צֹאן־ וּבָקָר֙ וַחֲמֹרִ֔ים וַעֲבָדִים֙ וּשְׁפָחֹ֔ת וַאֲתֹנֹ֖ת וּגְמַלִּֽים׃
17יְהוָ֧ה ׀ אֶת־ פַּרְעֹ֛ה נְגָעִ֥ים גְּדֹלִ֖ים וְאֶת־ בֵּית֑וֹ עַל־ דְּבַ֥ר שָׂרַ֖י אֵ֥שֶׁת אַבְרָֽם׃
18פַרְעֹה֙ לְאַבְרָ֔ם וַיֹּ֕אמֶר מַה־ זֹּ֖את עָשִׂ֣יתָ לִּ֑י לָ֚מָּה לֹא־ הִגַּ֣דְתָּ לִּ֔י כִּ֥י אִשְׁתְּךָ֖ הִֽוא׃
19אָמַ֙רְתָּ֙ אֲחֹ֣תִי הִ֔וא וָאֶקַּ֥ח אֹתָ֛הּ לִ֖י לְאִשָּׁ֑ה וְעַתָּ֕ה הִנֵּ֥ה אִשְׁתְּךָ֖ קַ֥ח וָלֵֽךְ׃
