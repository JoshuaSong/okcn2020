1וַיָּבֹ֣א רְחַבְעָם֮ יְרוּשָׁלִַם֒ וַיַּקְהֵל֩ אֶת־ בֵּ֨ית יְהוּדָ֜ה וּבִנְיָמִ֗ן מֵאָ֨ה וּשְׁמוֹנִ֥ים אֶ֛לֶף בָּח֖וּר עֹשֵׂ֣ה מִלְחָמָ֑ה לְהִלָּחֵם֙ עִם־ יִשְׂרָאֵ֔ל לְהָשִׁ֥יב אֶת־ הַמַּמְלָכָ֖ה לִרְחַבְעָֽם׃ פ
2דְּבַר־ יְהוָ֔ה אֶל־ שְׁמַֽעְיָ֥הוּ אִישׁ־ הָאֱלֹהִ֖ים לֵאמֹֽר׃
3אֶל־ רְחַבְעָ֥ם בֶּן־ שְׁלֹמֹ֖ה מֶ֣לֶךְ יְהוּדָ֑ה וְאֶל֙ כָּל־ יִשְׂרָאֵ֔ל בִּיהוּדָ֥ה וּבִנְיָמִ֖ן לֵאמֹֽר׃
4אָמַ֣ר יְהוָ֡ה לֹא־ תַעֲלוּ֩ וְלֹא־ תִלָּ֨חֲמ֜וּ עִם־ אֲחֵיכֶ֗ם שׁ֚וּבוּ אִ֣ישׁ לְבֵית֔וֹ כִּ֧י מֵֽאִתִּ֛י נִהְיָ֖ה הַדָּבָ֣ר הַזֶּ֑ה וַֽיִּשְׁמְעוּ֙ אֶת־ דִּבְרֵ֣י יְהוָ֔ה וַיָּשֻׁ֖בוּ מִלֶּ֥כֶת אֶל־ יָרָבְעָֽם׃ פ
5רְחַבְעָ֖ם בִּירוּשָׁלִָ֑ם וַיִּ֧בֶן עָרִ֛ים לְמָצ֖וֹר בִּיהוּדָֽה׃
6אֶת־ בֵּֽית־ לֶ֛חֶם וְאֶת־ עֵיטָ֖ם וְאֶת־ תְּקֽוֹעַ׃
7בֵּֽית־ צ֥וּר וְאֶת־ שׂוֹכ֖וֹ וְאֶת־ עֲדֻלָּֽם׃
8גַּ֥ת וְאֶת־ מָרֵשָׁ֖ה וְאֶת־ זִֽיף׃
9אֲדוֹרַ֥יִם וְאֶת־ לָכִ֖ישׁ וְאֶת־ עֲזֵקָֽה׃
10צָרְעָה֙ וְאֶת־ אַיָּל֔וֹן וְאֶת־ חֶבְר֔וֹן אֲשֶׁ֥ר בִּיהוּדָ֖ה וּבְבִנְיָמִ֑ן עָרֵ֖י מְצֻרֽוֹת׃
11אֶת־ הַמְּצֻר֑וֹת וַיִּתֵּ֤ן בָּהֶם֙ נְגִידִ֔ים וְאֹצְר֥וֹת מַאֲכָ֖ל וְשֶׁ֥מֶן וָיָֽיִן׃
12עִ֤יר וָעִיר֙ צִנּ֣וֹת וּרְמָחִ֔ים וַֽיְחַזְּקֵ֖ם לְהַרְבֵּ֣ה מְאֹ֑ד וַיְהִי־ ל֖וֹ יְהוּדָ֥ה וּבִנְיָמִֽן׃ ס
13וְהַלְוִיִּ֔ם אֲשֶׁ֖ר בְּכָל־ יִשְׂרָאֵ֑ל הִֽתְיַצְּב֥וּ עָלָ֖יו מִכָּל־ גְּבוּלָֽם׃
14עָזְב֣וּ הַלְוִיִּ֗ם אֶת־ מִגְרְשֵׁיהֶם֙ וַאֲחֻזָּתָ֔ם וַיֵּלְכ֥וּ לִיהוּדָ֖ה וְלִֽירוּשָׁלִָ֑ם כִּֽי־ הִזְנִיחָ֤ם יָֽרָבְעָם֙ וּבָנָ֔יו מִכַּהֵ֖ן לַיהוָֽה׃
15לוֹ֙ כֹּֽהֲנִ֔ים לַבָּמ֖וֹת וְלַשְּׂעִירִ֑ים וְלָעֲגָלִ֖ים אֲשֶׁ֥ר עָשָֽׂה׃
16מִכֹּל֙ שִׁבְטֵ֣י יִשְׂרָאֵ֔ל הַנֹּֽתְנִים֙ אֶת־ לְבָבָ֔ם לְבַקֵּ֕שׁ אֶת־ יְהוָ֖ה אֱלֹהֵ֣י יִשְׂרָאֵ֑ל בָּ֚אוּ יְר֣וּשָׁלִַ֔ם לִזְבּ֕וֹחַ לַיהוָ֖ה אֱלֹהֵ֥י אֲבוֹתֵיהֶֽם׃
17אֶת־ מַלְכ֣וּת יְהוּדָ֔ה וַֽיְאַמְּצ֛וּ אֶת־ רְחַבְעָ֥ם בֶּן־ שְׁלֹמֹ֖ה לְשָׁנִ֣ים שָׁל֑וֹשׁ כִּ֣י הָֽלְכ֗וּ בְּדֶ֧רֶךְ דָּוִ֛יד וּשְׁלֹמֹ֖ה לְשָׁנִ֥ים שָׁלֽוֹשׁ׃
18ל֤וֹ רְחַבְעָם֙ אִשָּׁ֔ה אֶת־ מָ֣חֲלַ֔ת [בֶּן כ] (בַּת־ יְרִימ֖וֹת ק) בֶּן־ דָּוִ֑יד אֲבִיהַ֕יִל בַּת־ אֱלִיאָ֖ב בֶּן־ יִשָֽׁי׃
19ל֖וֹ בָּנִ֑ים אֶת־ יְע֥וּשׁ וְאֶת־ שְׁמַרְיָ֖ה וְאֶת־ זָֽהַם׃
20לָקַ֔ח אֶֽת־ מַעֲכָ֖ה בַּת־ אַבְשָׁל֑וֹם וַתֵּ֣לֶד ל֗וֹ אֶת־ אֲבִיָּה֙ וְאֶת־ עַתַּ֔י וְאֶת־ זִיזָ֖א וְאֶת־ שְׁלֹמִֽית׃
21רְחַבְעָ֜ם אֶת־ מַעֲכָ֣ה בַת־ אַבְשָׁל֗וֹם מִכָּל־ נָשָׁיו֙ וּפִ֣ילַגְשָׁ֔יו כִּ֠י נָשִׁ֤ים שְׁמוֹנֶֽה־ עֶשְׂרֵה֙ נָשָׂ֔א וּפִֽילַגְשִׁ֖ים שִׁשִּׁ֑ים וַיּ֗וֹלֶד עֶשְׂרִ֧ים וּשְׁמוֹנָ֛ה בָּנִ֖ים וְשִׁשִּׁ֥ים בָּנֽוֹת׃
22לָרֹ֧אשׁ רְחַבְעָ֛ם אֶת־ אֲבִיָּ֥ה בֶֽן־ מַעֲכָ֖ה לְנָגִ֣יד בְּאֶחָ֑יו כִּ֖י לְהַמְלִיכֽוֹ׃
