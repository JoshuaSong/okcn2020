1וַיִּשְׁמַ֤ע פַּשְׁחוּר֙ בֶּן־ אִמֵּ֣ר הַכֹּהֵ֔ן וְהֽוּא־ פָקִ֥יד נָגִ֖יד בְּבֵ֣ית יְהוָ֑ה אֶֽת־ יִרְמְיָ֔הוּ נִבָּ֖א אֶת־ הַדְּבָרִ֥ים הָאֵֽלֶּה׃
2פַשְׁח֔וּר אֵ֖ת יִרְמְיָ֣הוּ הַנָּבִ֑יא וַיִּתֵּ֨ן אֹת֜וֹ עַל־ הַמַּהְפֶּ֗כֶת אֲשֶׁ֨ר בְּשַׁ֤עַר בִּנְיָמִן֙ הָֽעֶלְי֔וֹן אֲשֶׁ֖ר בְּבֵ֥ית יְהוָֽה׃
3מִֽמָּחֳרָ֔ת וַיֹּצֵ֥א פַשְׁח֛וּר אֶֽת־ יִרְמְיָ֖הוּ מִן־ הַמַּהְפָּ֑כֶת וַיֹּ֨אמֶר אֵלָ֜יו יִרְמְיָ֗הוּ לֹ֤א פַשְׁחוּר֙ קָרָ֤א יְהוָה֙ שְׁמֶ֔ךָ כִּ֖י אִם־ מָג֥וֹר מִסָּבִֽיב׃ פ
4כֹ֣ה אָמַ֣ר יְהוָ֡ה הִנְנִי֩ נֹתֶנְךָ֨ לְמָג֜וֹר לְךָ֣ וּלְכָל־ אֹהֲבֶ֗יךָ וְנָֽפְל֛וּ בְּחֶ֥רֶב אֹיְבֵיהֶ֖ם וְעֵינֶ֣יךָ רֹא֑וֹת וְאֶת־ כָּל־ יְהוּדָ֗ה אֶתֵּן֙ בְּיַ֣ד מֶֽלֶךְ־ בָּבֶ֔ל וְהִגְלָ֥ם בָּבֶ֖לָה וְהִכָּ֥ם בֶּחָֽרֶב׃
5אֶת־ כָּל־ חֹ֙סֶן֙ הָעִ֣יר הַזֹּ֔את וְאֶת־ כָּל־ יְגִיעָ֖הּ וְאֶת־ כָּל־ יְקָרָ֑הּ וְאֵ֨ת כָּל־ אוֹצְר֜וֹת מַלְכֵ֣י יְהוּדָ֗ה אֶתֵּן֙ בְּיַ֣ד אֹֽיְבֵיהֶ֔ם וּבְזָזוּם֙ וּלְקָח֔וּם וֶהֱבִיא֖וּם בָּבֶֽלָה׃
6פַשְׁח֗וּר וְכֹל֙ יֹשְׁבֵ֣י בֵיתֶ֔ךָ תֵּלְכ֖וּ בַּשֶּׁ֑בִי וּבָבֶ֣ל תָּב֗וֹא וְשָׁ֤ם תָּמוּת֙ וְשָׁ֣ם תִּקָּבֵ֔ר אַתָּה֙ וְכָל־ אֹ֣הֲבֶ֔יךָ אֲשֶׁר־ נִבֵּ֥אתָ לָהֶ֖ם בַּשָּֽׁקֶר׃ ס
7יְהוָה֙ וָֽאֶפָּ֔ת חֲזַקְתַּ֖נִי וַתּוּכָ֑ל הָיִ֤יתִי לִשְׂחוֹק֙ כָּל־ הַיּ֔וֹם כֻּלֹּ֖ה לֹעֵ֥ג לִֽי׃
8מִדֵּ֤י אֲדַבֵּר֙ אֶזְעָ֔ק חָמָ֥ס וָשֹׁ֖ד אֶקְרָ֑א כִּֽי־ הָיָ֨ה דְבַר־ יְהוָ֥ה לִ֛י לְחֶרְפָּ֥ה וּלְקֶ֖לֶס כָּל־ הַיּֽוֹם׃
9לֹֽא־ אֶזְכְּרֶ֗נּוּ וְלֹֽא־ אֲדַבֵּ֥ר עוֹד֙ בִּשְׁמ֔וֹ וְהָיָ֤ה בְלִבִּי֙ כְּאֵ֣שׁ בֹּעֶ֔רֶת עָצֻ֖ר בְּעַצְמֹתָ֑י וְנִלְאֵ֥יתִי כַּֽלְכֵ֖ל וְלֹ֥א אוּכָֽל׃
10שָׁמַ֜עְתִּי דִּבַּ֣ת רַבִּים֮ מָג֣וֹר מִסָּבִיב֒ הַגִּ֙ידוּ֙ וְנַגִּידֶ֔נּוּ כֹּ֚ל אֱנ֣וֹשׁ שְׁלוֹמִ֔י שֹׁמְרֵ֖י צַלְעִ֑י אוּלַ֤י יְפֻתֶּה֙ וְנ֣וּכְלָה ל֔וֹ וְנִקְחָ֥ה נִקְמָתֵ֖נוּ מִמֶּֽנּוּ׃
11אוֹתִי֙ כְּגִבּ֣וֹר עָרִ֔יץ עַל־ כֵּ֛ן רֹדְפַ֥י יִכָּשְׁל֖וּ וְלֹ֣א יֻכָ֑לוּ בֹּ֤שׁוּ מְאֹד֙ כִּֽי־ לֹ֣א הִשְׂכִּ֔ילוּ כְּלִמַּ֥ת עוֹלָ֖ם לֹ֥א תִשָּׁכֵֽחַ׃
12צְבָאוֹת֙ בֹּחֵ֣ן צַדִּ֔יק רֹאֶ֥ה כְלָי֖וֹת וָלֵ֑ב אֶרְאֶ֤ה נִקְמָֽתְךָ֙ מֵהֶ֔ם כִּ֥י אֵלֶ֖יךָ גִּלִּ֥יתִי אֶת־ רִיבִֽי׃ ס
13לַֽיהוָ֔ה הַֽלְל֖וּ אֶת־ יְהוָ֑ה כִּ֥י הִצִּ֛יל אֶת־ נֶ֥פֶשׁ אֶבְי֖וֹן מִיַּ֥ד מְרֵעִֽים׃ ס
14הַיּ֔וֹם אֲשֶׁ֥ר יֻלַּ֖דְתִּי בּ֑וֹ י֛וֹם אֲשֶׁר־ יְלָדַ֥תְנִי אִמִּ֖י אַל־ יְהִ֥י בָרֽוּךְ׃
15הָאִ֗ישׁ אֲשֶׁ֨ר בִּשַּׂ֤ר אֶת־ אָבִי֙ לֵאמֹ֔ר יֻֽלַּד־ לְךָ֖ בֵּ֣ן זָכָ֑ר שַׂמֵּ֖חַ שִׂמֳּחָֽהוּ׃
16הָאִ֣ישׁ הַה֔וּא כֶּֽעָרִ֛ים אֲשֶׁר־ הָפַ֥ךְ יְהוָ֖ה וְלֹ֣א נִחָ֑ם וְשָׁמַ֤ע זְעָקָה֙ בַּבֹּ֔קֶר וּתְרוּעָ֖ה בְּעֵ֥ת צָהֳרָֽיִם׃
17לֹא־ מוֹתְתַ֖נִי מֵרָ֑חֶם וַתְּהִי־ לִ֤י אִמִּי֙ קִבְרִ֔י וְרַחְמָ֖הֿ הֲרַ֥ת עוֹלָֽם׃
