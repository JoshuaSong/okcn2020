1כֹּ֚ה אָמַ֣ר יְהוָ֔ה הִנְנִי֙ מֵעִ֣יר עַל־ בָּבֶ֔ל וְאֶל־ יֹשְׁבֵ֖י לֵ֣ב קָמָ֑י ר֖וּחַ מַשְׁחִֽית׃
2לְבָבֶ֤ל ׀ זָרִים֙ וְזֵר֔וּהָ וִיבֹקְק֖וּ אֶת־ אַרְצָ֑הּ כִּֽי־ הָי֥וּ עָלֶ֛יהָ מִסָּבִ֖יב בְּי֥וֹם רָעָֽה׃
3יִדְרֹ֤ךְ [יִדְרֹךְ כ] הַדֹּרֵךְ֙ קַשְׁתּ֔וֹ וְאֶל־ יִתְעַ֖ל בְּסִרְיֹנ֑וֹ וְאַֽל־ תַּחְמְלוּ֙ אֶל־ בַּ֣חֻרֶ֔יהָ הַחֲרִ֖ימוּ כָּל־ צְבָאָֽהּ׃
4חֲלָלִ֖ים בְּאֶ֣רֶץ כַּשְׂדִּ֑ים וּמְדֻקָּרִ֖ים בְּחוּצוֹתֶֽיהָ׃
5לֹֽא־ אַלְמָ֨ן יִשְׂרָאֵ֤ל וִֽיהוּדָה֙ מֵֽאֱלֹהָ֔יו מֵֽיְהוָ֖ה צְבָא֑וֹת כִּ֤י אַרְצָם֙ מָלְאָ֣ה אָשָׁ֔ם מִקְּד֖וֹשׁ יִשְׂרָאֵֽל׃
6מִתּ֣וֹךְ בָּבֶ֗ל וּמַלְּטוּ֙ אִ֣ישׁ נַפְשׁ֔וֹ אַל־ תִּדַּ֖מּוּ בַּעֲוֹנָ֑הּ כִּי֩ עֵ֨ת נְקָמָ֥ה הִיא֙ לַֽיהוָ֔ה גְּמ֕וּל ה֥וּא מְשַׁלֵּ֖ם לָֽהּ׃
7זָהָ֤ב בָּבֶל֙ בְּיַד־ יְהוָ֔ה מְשַׁכֶּ֖רֶת כָּל־ הָאָ֑רֶץ מִיֵּינָהּ֙ שָׁת֣וּ גוֹיִ֔ם עַל־ כֵּ֖ן יִתְהֹלְל֥וּ גוֹיִֽם׃
8נָפְלָ֥ה בָבֶ֖ל וַתִּשָּׁבֵ֑ר הֵילִ֣ילוּ עָלֶ֗יהָ קְח֤וּ צֳרִי֙ לְמַכְאוֹבָ֔הּ אוּלַ֖י תֵּרָפֵֽא׃
9כ] (רִפִּ֣ינוּ ק) אֶת־ בָּבֶל֙ וְלֹ֣א נִרְפָּ֔תָה עִזְב֕וּהָ וְנֵלֵ֖ךְ אִ֣ישׁ לְאַרְצ֑וֹ כִּֽי־ נָגַ֤ע אֶל־ הַשָּׁמַ֙יִם֙ מִשְׁפָּטָ֔הּ וְנִשָּׂ֖א עַד־ שְׁחָקִֽים׃
10יְהוָ֖ה אֶת־ צִדְקֹתֵ֑ינוּ בֹּ֚אוּ וּנְסַפְּרָ֣ה בְצִיּ֔וֹן אֶֽת־ מַעֲשֵׂ֖ה יְהוָ֥ה אֱלֹהֵֽינוּ׃
11הַחִצִּים֮ מִלְא֣וּ הַשְּׁלָטִים֒ הֵעִ֣יר יְהוָ֗ה אֶת־ ר֙וּחַ֙ מַלְכֵ֣י מָדַ֔י כִּֽי־ עַל־ בָּבֶ֥ל מְזִמָּת֖וֹ לְהַשְׁחִיתָ֑הּ כִּֽי־ נִקְמַ֤ת יְהוָה֙ הִ֔יא נִקְמַ֖ת הֵיכָלֽוֹ׃
12חוֹמֹ֨ת בָּבֶ֜ל שְׂאוּ־ נֵ֗ס הַחֲזִ֙יקוּ֙ הַמִּשְׁמָ֔ר הָקִ֙ימוּ֙ שֹֽׁמְרִ֔ים הָכִ֖ינוּ הָאֹֽרְבִ֑ים כִּ֚י גַּם־ זָמַ֣ם יְהוָ֔ה גַּם־ עָשָׂ֕ה אֵ֥ת אֲשֶׁר־ דִּבֶּ֖ר אֶל־ יֹשְׁבֵ֥י בָבֶֽל׃
13כ] (שֹׁכַנְתְּ֙ ק) עַל־ מַ֣יִם רַבִּ֔ים רַבַּ֖ת אֽוֹצָרֹ֑ת בָּ֥א קִצֵּ֖ךְ אַמַּ֥ת בִּצְעֵֽךְ׃
14יְהוָ֥ה צְבָא֖וֹת בְּנַפְשׁ֑וֹ כִּ֣י אִם־ מִלֵּאתִ֤יךְ אָדָם֙ כַּיֶּ֔לֶק וְעָנ֥וּ עָלַ֖יִךְ הֵידָֽד׃ ס
15אֶ֙רֶץ֙ בְּכֹח֔וֹ מֵכִ֥ין תֵּבֵ֖ל בְּחָכְמָת֑וֹ וּבִתְבוּנָת֖וֹ נָטָ֥ה שָׁמָֽיִם׃
16תִּתּ֜וֹ הֲמ֥וֹן מַ֙יִם֙ בַּשָּׁמַ֔יִם וַיַּ֥עַל נְשִׂאִ֖ים מִקְצֵה־ אָ֑רֶץ בְּרָקִ֤ים לַמָּטָר֙ עָשָׂ֔ה וַיֹּ֥צֵא ר֖וּחַ מֵאֹצְרֹתָֽיו׃
17כָּל־ אָדָם֙ מִדַּ֔עַת הֹבִ֥ישׁ כָּל־ צֹרֵ֖ף מִפָּ֑סֶל כִּ֛י שֶׁ֥קֶר נִסְכּ֖וֹ וְלֹא־ ר֥וּחַ בָּֽם׃
18הֵ֔מָּה מַעֲשֵׂ֖ה תַּעְתֻּעִ֑ים בְּעֵ֥ת פְּקֻדָּתָ֖ם יֹאבֵֽדוּ׃
19כְאֵ֜לֶּה חֵ֣לֶק יַעֲק֗וֹב כִּֽי־ יוֹצֵ֤ר הַכֹּל֙ ה֔וּא וְשֵׁ֖בֶט נַחֲלָת֑וֹ יְהוָ֥ה צְבָא֖וֹת שְׁמֽוֹ׃ ס
20אַתָּ֣ה לִ֔י כְּלֵ֖י מִלְחָמָ֑ה וְנִפַּצְתִּ֤י בְךָ֙ גּוֹיִ֔ם וְהִשְׁחַתִּ֥י בְךָ֖ מַמְלָכֽוֹת׃
21בְךָ֔ ס֖וּס וְרֹֽכְב֑וֹ וְנִפַּצְתִּ֣י בְךָ֔ רֶ֖כֶב וְרֹכְבֽוֹ׃
22בְךָ֙ אִ֣ישׁ וְאִשָּׁ֔ה וְנִפַּצְתִּ֥י בְךָ֖ זָקֵ֣ן וָנָ֑עַר וְנִפַּצְתִּ֣י בְךָ֔ בָּח֖וּר וּבְתוּלָֽה׃
23בְךָ֙ רֹעֶ֣ה וְעֶדְר֔וֹ וְנִפַּצְתִּ֥י בְךָ֖ אִכָּ֣ר וְצִמְדּ֑וֹ וְנִפַּצְתִּ֣י בְךָ֔ פַּח֖וֹת וּסְגָנִֽים׃
24לְבָבֶ֜ל וּלְכֹ֣ל ׀ יוֹשְׁבֵ֣י כַשְׂדִּ֗ים אֵ֧ת כָּל־ רָעָתָ֛ם אֲשֶׁר־ עָשׂ֥וּ בְצִיּ֖וֹן לְעֵֽינֵיכֶ֑ם נְאֻ֖ם יְהוָֽה׃ ס
25אֵלֶ֜יךָ הַ֤ר הַמַּשְׁחִית֙ נְאֻם־ יְהוָ֔ה הַמַּשְׁחִ֖ית אֶת־ כָּל־ הָאָ֑רֶץ וְנָטִ֨יתִי אֶת־ יָדִ֜י עָלֶ֗יךָ וְגִלְגַּלְתִּ֙יךָ֙ מִן־ הַסְּלָעִ֔ים וּנְתַתִּ֖יךָ לְהַ֥ר שְׂרֵפָֽה׃
26יִקְח֤וּ מִמְּךָ֙ אֶ֣בֶן לְפִנָּ֔ה וְאֶ֖בֶן לְמֽוֹסָד֑וֹת כִּֽי־ שִׁמְמ֥וֹת עוֹלָ֛ם תִּֽהְיֶ֖ה נְאֻם־ יְהוָֽה׃
27נֵ֣ס בָּאָ֗רֶץ תִּקְע֨וּ שׁוֹפָ֤ר בַּגּוֹיִם֙ קַדְּשׁ֤וּ עָלֶ֙יהָ֙ גּוֹיִ֔ם הַשְׁמִ֧יעוּ עָלֶ֛יהָ מַמְלְכ֥וֹת אֲרָרַ֖ט מִנִּ֣י וְאַשְׁכְּנָ֑ז פִּקְד֤וּ עָלֶ֙יהָ֙ טִפְסָ֔ר הַֽעֲלוּ־ ס֖וּס כְּיֶ֥לֶק סָמָֽר׃
28עָלֶ֤יהָ גוֹיִם֙ אֶת־ מַלְכֵ֣י מָדַ֔י אֶת־ פַּחוֹתֶ֖יהָ וְאֶת־ כָּל־ סְגָנֶ֑יהָ וְאֵ֖ת כָּל־ אֶ֥רֶץ מֶמְשַׁלְתּֽוֹ׃
29הָאָ֖רֶץ וַתָּחֹ֑ל כִּ֣י קָ֤מָה עַל־ בָּבֶל֙ מַחְשְׁב֣וֹת יְהוָ֔ה לָשׂ֞וּם אֶת־ אֶ֧רֶץ בָּבֶ֛ל לְשַׁמָּ֖ה מֵאֵ֥ין יוֹשֵֽׁב׃
30גִבּוֹרֵ֨י בָבֶ֜ל לְהִלָּחֵ֗ם יָֽשְׁבוּ֙ בַּמְּצָד֔וֹת נָשְׁתָ֥ה גְבוּרָתָ֖ם הָי֣וּ לְנָשִׁ֑ים הִצִּ֥יתוּ מִשְׁכְּנֹתֶ֖יהָ נִשְׁבְּר֥וּ בְרִיחֶֽיהָ׃
31לִקְרַאת־ רָץ֙ יָר֔וּץ וּמַגִּ֖יד לִקְרַ֣את מַגִּ֑יד לְהַגִּיד֙ לְמֶ֣לֶךְ בָּבֶ֔ל כִּֽי־ נִלְכְּדָ֥ה עִיר֖וֹ מִקָּצֶֽה׃
32נִתְפָּ֔שׂוּ וְאֶת־ הָאֲגַמִּ֖ים שָׂרְפ֣וּ בָאֵ֑שׁ וְאַנְשֵׁ֥י הַמִּלְחָמָ֖ה נִבְהָֽלוּ׃ ס
33כֹ֨ה אָמַ֜ר יְהוָ֤ה צְבָאוֹת֙ אֱלֹהֵ֣י יִשְׂרָאֵ֔ל בַּת־ בָּבֶ֕ל כְּגֹ֖רֶן עֵ֣ת הִדְרִיכָ֑הּ ע֣וֹד מְעַ֔ט וּבָ֥אָה עֵֽת־ הַקָּצִ֖יר לָֽהּ׃
34כ] (אֲכָלַ֣נִי ק) [הֲמָמָנוּ כ] (הֲמָמַ֗נִי ק) נְבוּכַדְרֶאצַּר֮ מֶ֣לֶךְ בָּבֶל֒ [הִצִּיגָנוּ כ] (הִצִּיגַ֙נִי֙ ק) כְּלִ֣י רִ֔יק [בְּלָעָנוּ כ] (בְּלָעַ֙נִי֙ ק) כַּתַּנִּ֔ין מִלָּ֥א כְרֵשׂ֖וֹ מֵֽעֲדָנָ֑י [הֱדִיחָנוּ כ] (הֱדִיחָֽנִי׃ ק)
35וּשְׁאֵרִי֙ עַל־ בָּבֶ֔ל תֹּאמַ֖ר יֹשֶׁ֣בֶת צִיּ֑וֹן וְדָמִי֙ אֶל־ יֹשְׁבֵ֣י כַשְׂדִּ֔ים תֹּאמַ֖ר יְרוּשָׁלִָֽם׃ ס
36כֹּ֚ה אָמַ֣ר יְהוָ֔ה הִנְנִי־ רָב֙ אֶת־ רִיבֵ֔ךְ וְנִקַּמְתִּ֖י אֶת־ נִקְמָתֵ֑ךְ וְהַחֲרַבְתִּי֙ אֶת־ יַמָּ֔הּ וְהֹבַשְׁתִּ֖י אֶת־ מְקוֹרָֽהּ׃
37בָבֶ֨ל ׀ לְגַלִּ֧ים ׀ מְעוֹן־ תַּנִּ֛ים שַׁמָּ֥ה וּשְׁרֵקָ֖ה מֵאֵ֥ין יוֹשֵֽׁב׃
38כַּכְּפִרִ֣ים יִשְׁאָ֑גוּ נָעֲר֖וּ כְּגוֹרֵ֥י אֲרָיֽוֹת׃
39אָשִׁ֣ית אֶת־ מִשְׁתֵּיהֶ֗ם וְהִשְׁכַּרְתִּים֙ לְמַ֣עַן יַעֲלֹ֔זוּ וְיָשְׁנ֥וּ שְׁנַת־ עוֹלָ֖ם וְלֹ֣א יָקִ֑יצוּ נְאֻ֖ם יְהוָֽה׃
40כְּכָרִ֣ים לִטְב֑וֹחַ כְּאֵילִ֖ים עִם־ עַתּוּדִֽים׃
41נִלְכְּדָ֣ה שֵׁשַׁ֔ךְ וַתִּתָּפֵ֖שׂ תְּהִלַּ֣ת כָּל־ הָאָ֑רֶץ אֵ֣יךְ הָיְתָ֧ה לְשַׁמָּ֛ה בָּבֶ֖ל בַּגּוֹיִֽם׃
42עַל־ בָּבֶ֖ל הַיָּ֑ם בַּהֲמ֥וֹן גַּלָּ֖יו נִכְסָֽתָה׃
43עָרֶ֙יהָ֙ לְשַׁמָּ֔ה אֶ֖רֶץ צִיָּ֣ה וַעֲרָבָ֑ה אֶ֗רֶץ לֹֽא־ יֵשֵׁ֤ב בָּהֵן֙ כָּל־ אִ֔ישׁ וְלֹֽא־ יַעֲבֹ֥ר בָּהֵ֖ן בֶּן־ אָדָֽם׃
44עַל־ בֵּ֜ל בְּבָבֶ֗ל וְהֹצֵאתִ֤י אֶת־ בִּלְעוֹ֙ מִפִּ֔יו וְלֹֽא־ יִנְהֲר֥וּ אֵלָ֛יו ע֖וֹד גּוֹיִ֑ם גַּם־ חוֹמַ֥ת בָּבֶ֖ל נָפָֽלָה׃
45מִתּוֹכָהּ֙ עַמִּ֔י וּמַלְּט֖וּ אִ֣ישׁ אֶת־ נַפְשׁ֑וֹ מֵחֲר֖וֹן אַף־ יְהוָֽה׃
46יֵרַ֤ךְ לְבַבְכֶם֙ וְתִֽירְא֔וּ בַּשְּׁמוּעָ֖ה הַנִּשְׁמַ֣עַת בָּאָ֑רֶץ וּבָ֧א בַשָּׁנָ֣ה הַשְּׁמוּעָ֗ה וְאַחֲרָ֤יו בַּשָּׁנָה֙ הַשְּׁמוּעָ֔ה וְחָמָ֣ס בָּאָ֔רֶץ וּמֹשֵׁ֖ל עַל־ מֹשֵֽׁל׃
47הִנֵּ֣ה יָמִ֣ים בָּאִ֔ים וּפָקַדְתִּי֙ עַל־ פְּסִילֵ֣י בָבֶ֔ל וְכָל־ אַרְצָ֖הּ תֵּב֑וֹשׁ וְכָל־ חֲלָלֶ֖יהָ יִפְּל֥וּ בְתוֹכָֽהּ׃
48עַל־ בָּבֶל֙ שָׁמַ֣יִם וָאָ֔רֶץ וְכֹ֖ל אֲשֶׁ֣ר בָּהֶ֑ם כִּ֧י מִצָּפ֛וֹן יָבוֹא־ לָ֥הּ הַשּׁוֹדְדִ֖ים נְאֻם־ יְהוָֽה׃
49בָּבֶ֕ל לִנְפֹּ֖ל חַֽלְלֵ֣י יִשְׂרָאֵ֑ל גַּם־ לְבָבֶ֥ל נָפְל֖וּ חַֽלְלֵ֥י כָל־ הָאָֽרֶץ׃
50מֵחֶ֔רֶב הִלְכ֖וּ אַֽל־ תַּעֲמֹ֑דוּ זִכְר֤וּ מֵֽרָחוֹק֙ אֶת־ יְהוָ֔ה וִירֽוּשָׁלִַ֖ם תַּעֲלֶ֥ה עַל־ לְבַבְכֶֽם׃
51כִּֽי־ שָׁמַ֣עְנוּ חֶרְפָּ֔ה כִּסְּתָ֥ה כְלִמָּ֖ה פָּנֵ֑ינוּ כִּ֚י בָּ֣אוּ זָרִ֔ים עַֽל־ מִקְדְּשֵׁ֖י בֵּ֥ית יְהוָֽה׃ ס
52הִנֵּֽה־ יָמִ֤ים בָּאִים֙ נְאֻם־ יְהוָ֔ה וּפָקַדְתִּ֖י עַל־ פְּסִילֶ֑יהָ וּבְכָל־ אַרְצָ֖הּ יֶאֱנֹ֥ק חָלָֽל׃
53תַעֲלֶ֤ה בָבֶל֙ הַשָּׁמַ֔יִם וְכִ֥י תְבַצֵּ֖ר מְר֣וֹם עֻזָּ֑הּ מֵאִתִּ֗י יָבֹ֧אוּ שֹׁדְדִ֛ים לָ֖הּ נְאֻם־ יְהוָֽה׃ ס
54זְעָקָ֖ה מִבָּבֶ֑ל וְשֶׁ֥בֶר גָּד֖וֹל מֵאֶ֥רֶץ כַּשְׂדִּֽים׃
55שֹׁדֵ֤ד יְהוָה֙ אֶת־ בָּבֶ֔ל וְאִבַּ֥ד מִמֶּ֖נָּה ק֣וֹל גָּד֑וֹל וְהָמ֤וּ גַלֵּיהֶם֙ כְּמַ֣יִם רַבִּ֔ים נִתַּ֥ן שְׁא֖וֹן קוֹלָֽם׃
56בָ֨א עָלֶ֤יהָ עַל־ בָּבֶל֙ שׁוֹדֵ֔ד וְנִלְכְּדוּ֙ גִּבּוֹרֶ֔יהָ חִתְּתָ֖ה קַשְּׁתוֹתָ֑ם כִּ֣י אֵ֧ל גְּמֻל֛וֹת יְהוָ֖ה שַׁלֵּ֥ם יְשַׁלֵּֽם׃
57שָׂרֶ֨יהָ וַחֲכָמֶ֜יהָ פַּחוֹתֶ֤יהָ וּסְגָנֶ֙יהָ֙ וְגִבּוֹרֶ֔יהָ וְיָשְׁנ֥וּ שְׁנַת־ עוֹלָ֖ם וְלֹ֣א יָקִ֑יצוּ נְאֻ֨ם־ הַמֶּ֔לֶךְ יְהוָ֥ה צְבָא֖וֹת שְׁמֽוֹ׃ ס
58אָמַ֞ר יְהוָ֣ה צְבָא֗וֹת חֹ֠מוֹת בָּבֶ֤ל הָֽרְחָבָה֙ עַרְעֵ֣ר תִּתְעַרְעָ֔ר וּשְׁעָרֶ֥יהָ הַגְּבֹהִ֖ים בָּאֵ֣שׁ יִצַּ֑תּוּ וְיִֽגְע֨וּ עַמִּ֧ים בְּדֵי־ רִ֛יק וּלְאֻמִּ֥ים בְּדֵי־ אֵ֖שׁ וְיָעֵֽפוּ׃ ס
59אֲשֶׁר־ צִוָּ֣ה ׀ יִרְמְיָ֣הוּ הַנָּבִ֗יא אֶת־ שְׂרָיָ֣ה בֶן־ נֵרִיָּה֮ בֶּן־ מַחְסֵיָה֒ בְּלֶכְתּ֞וֹ אֶת־ צִדְקִיָּ֤הוּ מֶֽלֶךְ־ יְהוּדָה֙ בָּבֶ֔ל בִּשְׁנַ֥ת הָרְבִעִ֖ית לְמָלְכ֑וֹ וּשְׂרָיָ֖ה שַׂ֥ר מְנוּחָֽה׃
60יִרְמְיָ֗הוּ אֵ֧ת כָּל־ הָרָעָ֛ה אֲשֶׁר־ תָּב֥וֹא אֶל־ בָּבֶ֖ל אֶל־ סֵ֣פֶר אֶחָ֑ד אֵ֚ת כָּל־ הַדְּבָרִ֣ים הָאֵ֔לֶּה הַכְּתֻבִ֖ים אֶל־ בָּבֶֽל׃
61יִרְמְיָ֖הוּ אֶל־ שְׂרָיָ֑ה כְּבֹאֲךָ֣ בָבֶ֔ל וְֽרָאִ֔יתָ וְֽקָרָ֔אתָ אֵ֥ת כָּל־ הַדְּבָרִ֖ים הָאֵֽלֶּה׃
62יְהוָה֙ אַתָּ֨ה דִבַּ֜רְתָּ אֶל־ הַמָּק֤וֹם הַזֶּה֙ לְהַכְרִית֔וֹ לְבִלְתִּ֤י הֱיֽוֹת־ בּוֹ֙ יוֹשֵׁ֔ב לְמֵאָדָ֖ם וְעַד־ בְּהֵמָ֑ה כִּֽי־ שִׁמְמ֥וֹת עוֹלָ֖ם תִּֽהְיֶֽה׃
63כְּכַלֹּ֣תְךָ֔ לִקְרֹ֖א אֶת־ הַסֵּ֣פֶר הַזֶּ֑ה תִּקְשֹׁ֤ר עָלָיו֙ אֶ֔בֶן וְהִשְׁלַכְתּ֖וֹ אֶל־ תּ֥וֹךְ פְּרָֽת׃
