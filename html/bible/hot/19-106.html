1הַֽלְלוּיָ֨הּ ׀ הוֹד֣וּ לַיהוָ֣ה כִּי־ ט֑וֹב כִּ֖י לְעוֹלָ֣ם חַסְדּֽוֹ׃
2יְ֭מַלֵּל גְּבוּר֣וֹת יְהוָ֑ה יַ֝שְׁמִ֗יעַ כָּל־ תְּהִלָּתֽוֹ׃
3שֹׁמְרֵ֣י מִשְׁפָּ֑ט עֹשֵׂ֖ה צְדָקָ֣ה בְכָל־ עֵֽת׃
4יְ֭הוָה בִּרְצ֣וֹן עַמֶּ֑ךָ פָּ֝קְדֵ֗נִי בִּישׁוּעָתֶֽךָ׃
5בְּט֘וֹבַ֤ת בְּחִירֶ֗יךָ לִ֭שְׂמֹחַ בְּשִׂמְחַ֣ת גּוֹיֶ֑ךָ לְ֝הִתְהַלֵּ֗ל עִם־ נַחֲלָתֶֽךָ׃
6עִם־ אֲבוֹתֵ֗ינוּ הֶעֱוִ֥ינוּ הִרְשָֽׁעְנוּ׃
7בְמִצְרַ֨יִם ׀ לֹא־ הִשְׂכִּ֬ילוּ נִפְלְאוֹתֶ֗יךָ לֹ֣א זָ֭כְרוּ אֶת־ רֹ֣ב חֲסָדֶ֑יךָ וַיַּמְר֖וּ עַל־ יָ֣ם בְּיַם־ סֽוּף׃
8לְמַ֣עַן שְׁמ֑וֹ לְ֝הוֹדִ֗יעַ אֶת־ גְּבוּרָתֽוֹ׃
9בְּיַם־ ס֭וּף וַֽיֶּחֱרָ֑ב וַיּוֹלִיכֵ֥ם בַּ֝תְּהֹמ֗וֹת כַּמִּדְבָּֽר׃
10מִיַּ֣ד שׂוֹנֵ֑א וַ֝יִּגְאָלֵ֗ם מִיַּ֥ד אוֹיֵֽב׃
11מַ֥יִם צָרֵיהֶ֑ם אֶחָ֥ד מֵ֝הֶ֗ם לֹ֣א נוֹתָֽר׃
12בִדְבָרָ֑יו יָ֝שִׁ֗ירוּ תְּהִלָּתֽוֹ׃
13שָׁכְח֣וּ מַעֲשָׂ֑יו לֹֽא־ חִ֝כּ֗וּ לַעֲצָתֽוֹ׃
14תַ֭אֲוָה בַּמִּדְבָּ֑ר וַיְנַסּוּ־ אֵ֝֗ל בִּֽישִׁימֽוֹן׃
15לָ֭הֶם שֶׁאֱלָתָ֑ם וַיְשַׁלַּ֖ח רָז֣וֹן בְּנַפְשָֽׁם׃
16לְ֭מֹשֶׁה בַּֽמַּחֲנֶ֑ה לְ֝אַהֲרֹ֗ן קְד֣וֹשׁ יְהוָֽה׃
17אֶ֭רֶץ וַתִּבְלַ֣ע דָּתָ֑ן וַ֝תְּכַ֗ס עַל־ עֲדַ֥ת אֲבִירָֽם׃
18אֵ֥שׁ בַּעֲדָתָ֑ם לֶ֝הָבָ֗ה תְּלַהֵ֥ט רְשָׁעִֽים׃
19עֵ֥גֶל בְּחֹרֵ֑ב וַ֝יִּשְׁתַּחֲו֗וּ לְמַסֵּכָֽה׃
20אֶת־ כְּבוֹדָ֑ם בְּתַבְנִ֥ית שׁ֝֗וֹר אֹכֵ֥ל עֵֽשֶׂב׃
21אֵ֣ל מוֹשִׁיעָ֑ם עֹשֶׂ֖ה גְדֹל֣וֹת בְּמִצְרָֽיִם׃
22בְּאֶ֣רֶץ חָ֑ם נ֝וֹרָא֗וֹת עַל־ יַם־ סֽוּף׃
23לְֽהַשְׁמִ֫ידָ֥ם לוּלֵ֡י מֹ֘שֶׁ֤ה בְחִיר֗וֹ עָמַ֣ד בַּפֶּ֣רֶץ לְפָנָ֑יו לְהָשִׁ֥יב חֲ֝מָת֗וֹ מֵֽהַשְׁחִֽית׃
24בְּאֶ֣רֶץ חֶמְדָּ֑ה לֹֽא־ הֶ֝אֱמִ֗ינוּ לִדְבָרֽוֹ׃
25בְאָהֳלֵיהֶ֑ם לֹ֥א שָׁ֝מְע֗וּ בְּק֣וֹל יְהוָֽה׃
26יָד֣וֹ לָהֶ֑ם לְהַפִּ֥יל א֝וֹתָ֗ם בַּמִּדְבָּֽר׃
27זַ֭רְעָם בַּגּוֹיִ֑ם וּ֝לְזָרוֹתָ֗ם בָּאֲרָצֽוֹת׃
28לְבַ֣עַל פְּע֑וֹר וַ֝יֹּאכְל֗וּ זִבְחֵ֥י מֵתִֽים׃
29בְּמַֽעַלְלֵיהֶ֑ם וַתִּפְרָץ־ בָּ֝֗ם מַגֵּפָֽה׃
30פִּֽ֭ינְחָס וַיְפַלֵּ֑ל וַ֝תֵּעָצַ֗ר הַמַּגֵּפָֽה׃
31ל֭וֹ לִצְדָקָ֑ה לְדֹ֥ר וָ֝דֹ֗ר עַד־ עוֹלָֽם׃
32עַל־ מֵ֥י מְרִיבָ֑ה וַיֵּ֥רַע לְ֝מֹשֶׁ֗ה בַּעֲבוּרָֽם׃
33הִמְר֥וּ אֶת־ רוּח֑וֹ וַ֝יְבַטֵּ֗א בִּשְׂפָתָֽיו׃
34הִ֭שְׁמִידוּ אֶת־ הָֽעַמִּ֑ים אֲשֶׁ֤ר אָמַ֖ר יְהוָ֣ה לָהֶֽם׃
35בַגּוֹיִ֑ם וַֽ֝יִּלְמְד֗וּ מַֽעֲשֵׂיהֶֽם׃
36אֶת־ עֲצַבֵּיהֶ֑ם וַיִּהְי֖וּ לָהֶ֣ם לְמוֹקֵֽשׁ׃
37אֶת־ בְּ֭נֵיהֶם וְאֶת־ בְּנֽוֹתֵיהֶ֗ם לַשֵּֽׁדִים׃
38דָ֪ם נָקִ֡י דַּם־ בְּנֵ֘יהֶ֤ם וּֽבְנוֹתֵיהֶ֗ם אֲשֶׁ֣ר זִ֭בְּחוּ לַעֲצַבֵּ֣י כְנָ֑עַן וַתֶּחֱנַ֥ף הָ֝אָ֗רֶץ בַּדָּמִֽים׃
39בְמַעֲשֵׂיהֶ֑ם וַ֝יִּזְ֗נוּ בְּמַֽעַלְלֵיהֶֽם׃
40אַ֣ף יְהוָ֣ה בְּעַמּ֑וֹ וַ֝יְתָעֵ֗ב אֶת־ נַחֲלָתֽוֹ׃
41בְּיַד־ גּוֹיִ֑ם וַֽיִּמְשְׁל֥וּ בָ֝הֶ֗ם שֹׂנְאֵיהֶֽם׃
42אוֹיְבֵיהֶ֑ם וַ֝יִּכָּנְע֗וּ תַּ֣חַת יָדָֽם׃
43רַבּ֗וֹת יַצִּ֫ילֵ֥ם וְ֭הֵמָּה יַמְר֣וּ בַעֲצָתָ֑ם וַ֝יָּמֹ֗כּוּ בַּעֲוֹנָֽם׃
44בַּצַּ֣ר לָהֶ֑ם בְּ֝שָׁמְע֗וֹ אֶת־ רִנָּתָֽם׃
45לָהֶ֣ם בְּרִית֑וֹ וַ֝יִּנָּחֵ֗ם כְּרֹ֣ב [חַסְדֹּו כ] (חֲסָדָֽיו׃ ק)
46אוֹתָ֣ם לְרַחֲמִ֑ים לִ֝פְנֵ֛י כָּל־ שׁוֹבֵיהֶֽם׃
47יְה֘וָ֤ה אֱלֹהֵ֗ינוּ וְקַבְּצֵנוּ֮ מִֽן־ הַגּ֫וֹיִ֥ם לְ֭הֹדוֹת לְשֵׁ֣ם קָדְשֶׁ֑ךָ לְ֝הִשְׁתַּבֵּ֗חַ בִּתְהִלָּתֶֽךָ׃
