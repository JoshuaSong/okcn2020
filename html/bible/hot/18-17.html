1רוּחִ֣י חֻ֭בָּלָה יָמַ֥י נִזְעָ֗כוּ קְבָרִ֥ים לִֽי׃
2לֹ֣א הֲ֭תֻלִים עִמָּדִ֑י וּ֝בְהַמְּרוֹתָ֗ם תָּלַ֥ן עֵינִֽי׃
3נָּ֭א עָרְבֵ֣נִי עִמָּ֑ךְ מִֽי ה֝֗וּא לְיָדִ֥י יִתָּקֵֽעַ׃
4לִ֭בָּם צָפַ֣נְתָּ מִּשָּׂ֑כֶל עַל־ כֵּ֝֗ן לֹ֣א תְרֹמֵֽם׃
5יַגִּ֣יד רֵעִ֑ים וְעֵינֵ֖י בָנָ֣יו תִּכְלֶֽנָה׃
6לִמְשֹׁ֣ל עַמִּ֑ים וְתֹ֖פֶת לְפָנִ֣ים אֶֽהְיֶֽה׃
7מִכַּ֣עַשׂ עֵינִ֑י וִֽיצֻרַ֖י כַּצֵּ֣ל כֻּלָּֽם׃
8יְשָׁרִ֣ים עַל־ זֹ֑את וְ֝נָקִ֗י עַל־ חָנֵ֥ף יִתְעֹרָֽר׃
9צַדִּ֣יק דַּרְכּ֑וֹ וּֽטֳהָר־ יָ֝דַ֗יִם יֹסִ֥יף אֹֽמֶץ׃
10כֻּלָּ֣ם תָּ֭שֻׁבוּ וּבֹ֣אוּ נָ֑א וְלֹֽא־ אֶמְצָ֖א בָכֶ֣ם חָכָֽם׃
11עָ֭בְרוּ זִמֹּתַ֣י נִתְּק֑וּ מ֖וֹרָשֵׁ֣י לְבָבִֽי׃
12לְי֣וֹם יָשִׂ֑ימוּ א֝֗וֹר קָר֥וֹב מִפְּנֵי־ חֹֽשֶׁךְ׃
13אֲ֭קַוֶּה שְׁא֣וֹל בֵּיתִ֑י בַּ֝חֹ֗שֶׁךְ רִפַּ֥דְתִּי יְצוּעָֽי׃
14קָ֭רָאתִי אָ֣בִי אָ֑תָּה אִמִּ֥י וַ֝אֲחֹתִ֗י לָֽרִמָּֽה׃
15אֵפ֣וֹ תִקְוָתִ֑י וְ֝תִקְוָתִ֗י מִ֣י יְשׁוּרֶֽנָּה׃
