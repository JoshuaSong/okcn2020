1בַּחֹ֙דֶשׁ֙ הַשְּׁלִישִׁ֔י לְצֵ֥את בְּנֵי־ יִשְׂרָאֵ֖ל מֵאֶ֣רֶץ מִצְרָ֑יִם בַּיּ֣וֹם הַזֶּ֔ה בָּ֖אוּ מִדְבַּ֥ר סִינָֽי׃
2מֵרְפִידִ֗ים וַיָּבֹ֙אוּ֙ מִדְבַּ֣ר סִינַ֔י וַֽיַּחֲנ֖וּ בַּמִּדְבָּ֑ר וַיִּֽחַן־ שָׁ֥ם יִשְׂרָאֵ֖ל נֶ֥גֶד הָהָֽר׃
3עָלָ֖ה אֶל־ הָאֱלֹהִ֑ים וַיִּקְרָ֨א אֵלָ֤יו יְהוָה֙ מִן־ הָהָ֣ר לֵאמֹ֔ר כֹּ֤ה תֹאמַר֙ לְבֵ֣ית יַעֲקֹ֔ב וְתַגֵּ֖יד לִבְנֵ֥י יִשְׂרָאֵֽל׃
4רְאִיתֶ֔ם אֲשֶׁ֥ר עָשִׂ֖יתִי לְמִצְרָ֑יִם וָאֶשָּׂ֤א אֶתְכֶם֙ עַל־ כַּנְפֵ֣י נְשָׁרִ֔ים וָאָבִ֥א אֶתְכֶ֖ם אֵלָֽי׃
5אִם־ שָׁמ֤וֹעַ תִּשְׁמְעוּ֙ בְּקֹלִ֔י וּשְׁמַרְתֶּ֖ם אֶת־ בְּרִיתִ֑י וִהְיִ֨יתֶם לִ֤י סְגֻלָּה֙ מִכָּל־ הָ֣עַמִּ֔ים כִּי־ לִ֖י כָּל־ הָאָֽרֶץ׃
6תִּהְיוּ־ לִ֛י מַמְלֶ֥כֶת כֹּהֲנִ֖ים וְג֣וֹי קָד֑וֹשׁ אֵ֚לֶּה הַדְּבָרִ֔ים אֲשֶׁ֥ר תְּדַבֵּ֖ר אֶל־ בְּנֵ֥י יִשְׂרָאֵֽל׃
7מֹשֶׁ֔ה וַיִּקְרָ֖א לְזִקְנֵ֣י הָעָ֑ם וַיָּ֣שֶׂם לִפְנֵיהֶ֗ם אֵ֚ת כָּל־ הַדְּבָרִ֣ים הָאֵ֔לֶּה אֲשֶׁ֥ר צִוָּ֖הוּ יְהוָֽה׃
8כָל־ הָעָ֤ם יַחְדָּו֙ וַיֹּ֣אמְר֔וּ כֹּ֛ל אֲשֶׁר־ דִּבֶּ֥ר יְהוָ֖ה נַעֲשֶׂ֑ה וַיָּ֧שֶׁב מֹשֶׁ֛ה אֶת־ דִּבְרֵ֥י הָעָ֖ם אֶל־ יְהוָֽה׃
9יְהוָ֜ה אֶל־ מֹשֶׁ֗ה הִנֵּ֨ה אָנֹכִ֜י בָּ֣א אֵלֶיךָ֮ בְּעַ֣ב הֶֽעָנָן֒ בַּעֲב֞וּר יִשְׁמַ֤ע הָעָם֙ בְּדַבְּרִ֣י עִמָּ֔ךְ וְגַם־ בְּךָ֖ יַאֲמִ֣ינוּ לְעוֹלָ֑ם וַיַּגֵּ֥ד מֹשֶׁ֛ה אֶת־ דִּבְרֵ֥י הָעָ֖ם אֶל־ יְהוָֽה׃
10יְהוָ֤ה אֶל־ מֹשֶׁה֙ לֵ֣ךְ אֶל־ הָעָ֔ם וְקִדַּשְׁתָּ֥ם הַיּ֖וֹם וּמָחָ֑ר וְכִבְּס֖וּ שִׂמְלֹתָֽם׃
11נְכֹנִ֖ים לַיּ֣וֹם הַשְּׁלִישִׁ֑י כִּ֣י ׀ בַּיּ֣וֹם הַשְּׁלִישִׁ֗י יֵרֵ֧ד יְהוָ֛ה לְעֵינֵ֥י כָל־ הָעָ֖ם עַל־ הַ֥ר סִינָֽי׃
12אֶת־ הָעָם֙ סָבִ֣יב לֵאמֹ֔ר הִשָּׁמְר֥וּ לָכֶ֛ם עֲל֥וֹת בָּהָ֖ר וּנְגֹ֣עַ בְּקָצֵ֑הוּ כָּל־ הַנֹּגֵ֥עַ בָּהָ֖ר מ֥וֹת יוּמָֽת׃
13תִגַּ֨ע בּ֜וֹ יָ֗ד כִּֽי־ סָק֤וֹל יִסָּקֵל֙ אוֹ־ יָרֹ֣ה יִיָּרֶ֔ה אִם־ בְּהֵמָ֥ה אִם־ אִ֖ישׁ לֹ֣א יִחְיֶ֑ה בִּמְשֹׁךְ֙ הַיֹּבֵ֔ל הֵ֖מָּה יַעֲל֥וּ בָהָֽר׃
14מֹשֶׁ֛ה מִן־ הָהָ֖ר אֶל־ הָעָ֑ם וַיְקַדֵּשׁ֙ אֶת־ הָעָ֔ם וַֽיְכַבְּס֖וּ שִׂמְלֹתָֽם׃
15אֶל־ הָעָ֔ם הֱי֥וּ נְכֹנִ֖ים לִשְׁלֹ֣שֶׁת יָמִ֑ים אַֽל־ תִּגְּשׁ֖וּ אֶל־ אִשָּֽׁה׃
16בַיּ֨וֹם הַשְּׁלִישִׁ֜י בִּֽהְיֹ֣ת הַבֹּ֗קֶר וַיְהִי֩ קֹלֹ֨ת וּבְרָקִ֜ים וְעָנָ֤ן כָּבֵד֙ עַל־ הָהָ֔ר וְקֹ֥ל שֹׁפָ֖ר חָזָ֣ק מְאֹ֑ד וַיֶּחֱרַ֥ד כָּל־ הָעָ֖ם אֲשֶׁ֥ר בַּֽמַּחֲנֶֽה׃
17מֹשֶׁ֧ה אֶת־ הָעָ֛ם לִקְרַ֥את הָֽאֱלֹהִ֖ים מִן־ הַֽמַּחֲנֶ֑ה וַיִּֽתְיַצְּב֖וּ בְּתַחְתִּ֥ית הָהָֽר׃
18סִינַי֙ עָשַׁ֣ן כֻּלּ֔וֹ מִ֠פְּנֵי אֲשֶׁ֨ר יָרַ֥ד עָלָ֛יו יְהוָ֖ה בָּאֵ֑שׁ וַיַּ֤עַל עֲשָׁנוֹ֙ כְּעֶ֣שֶׁן הַכִּבְשָׁ֔ן וַיֶּחֱרַ֥ד כָּל־ הָהָ֖ר מְאֹֽד׃
19ק֣וֹל הַשּׁוֹפָ֔ר הוֹלֵ֖ךְ וְחָזֵ֣ק מְאֹ֑ד מֹשֶׁ֣ה יְדַבֵּ֔ר וְהָאֱלֹהִ֖ים יַעֲנֶ֥נּוּ בְקֽוֹל׃
20יְהוָ֛ה עַל־ הַ֥ר סִינַ֖י אֶל־ רֹ֣אשׁ הָהָ֑ר וַיִּקְרָ֨א יְהוָ֧ה לְמֹשֶׁ֛ה אֶל־ רֹ֥אשׁ הָהָ֖ר וַיַּ֥עַל מֹשֶֽׁה׃
21יְהוָה֙ אֶל־ מֹשֶׁ֔ה רֵ֖ד הָעֵ֣ד בָּעָ֑ם פֶּן־ יֶהֶרְס֤וּ אֶל־ יְהוָה֙ לִרְא֔וֹת וְנָפַ֥ל מִמֶּ֖נּוּ רָֽב׃
22הַכֹּהֲנִ֛ים הַנִּגָּשִׁ֥ים אֶל־ יְהוָ֖ה יִתְקַדָּ֑שׁוּ פֶּן־ יִפְרֹ֥ץ בָּהֶ֖ם יְהוָֽה׃
23מֹשֶׁה֙ אֶל־ יְהוָ֔ה לֹא־ יוּכַ֣ל הָעָ֔ם לַעֲלֹ֖ת אֶל־ הַ֣ר סִינָ֑י כִּֽי־ אַתָּ֞ה הַעֵדֹ֤תָה בָּ֙נוּ֙ לֵאמֹ֔ר הַגְבֵּ֥ל אֶת־ הָהָ֖ר וְקִדַּשְׁתּֽוֹ׃
24אֵלָ֤יו יְהוָה֙ לֶךְ־ רֵ֔ד וְעָלִ֥יתָ אַתָּ֖ה וְאַהֲרֹ֣ן עִמָּ֑ךְ וְהַכֹּהֲנִ֣ים וְהָעָ֗ם אַל־ יֶֽהֶרְס֛וּ לַעֲלֹ֥ת אֶל־ יְהוָ֖ה פֶּן־ יִפְרָץ־ בָּֽם׃
