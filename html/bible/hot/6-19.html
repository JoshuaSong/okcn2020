1וַיֵּצֵ֞א הַגּוֹרָ֤ל הַשֵּׁנִי֙ לְשִׁמְע֔וֹן לְמַטֵּ֥ה בְנֵֽי־ שִׁמְע֖וֹן לְמִשְׁפְּחוֹתָ֑ם וַֽיְהִי֙ נַֽחֲלָתָ֔ם בְּת֖וֹךְ נַחֲלַ֥ת בְּנֵֽי־ יְהוּדָֽה׃
2לָהֶ֖ם בְּנַֽחֲלָתָ֑ם בְּאֵֽר־ שֶׁ֥בַע וְשֶׁ֖בַע וּמוֹלָדָֽה׃
3שׁוּעָ֛ל וּבָלָ֖ה וָעָֽצֶם׃
4וּבְת֖וּל וְחָרְמָֽה׃
5וּבֵית־ הַמַּרְכָּב֖וֹת וַחֲצַ֥ר סוּסָֽה׃
6לְבָא֖וֹת וְשָֽׁרוּחֶ֑ן עָרִ֥ים שְׁלֹשׁ־ עֶשְׂרֵ֖ה וְחַצְרֵיהֶֽן׃
7רִמּ֖וֹן וָעֶ֣תֶר וְעָשָׁ֑ן עָרִ֥ים אַרְבַּ֖ע וְחַצְרֵיהֶֽן׃
8הַֽחֲצֵרִ֗ים אֲשֶׁ֤ר סְבִיבוֹת֙ הֶֽעָרִ֣ים הָאֵ֔לֶּה עַד־ בַּֽעֲלַ֥ת בְּאֵ֖ר רָ֣אמַת נֶ֑גֶב זֹ֗את נַחֲלַ֛ת מַטֵּ֥ה בְנֵֽי־ שִׁמְע֖וֹן לְמִשְׁפְּחֹתָֽם׃
9בְּנֵ֣י יְהוּדָ֔ה נַחֲלַ֖ת בְּנֵ֣י שִׁמְע֑וֹן כִּֽי־ הָיָ֞ה חֵ֤לֶק בְּנֵֽי־ יְהוּדָה֙ רַ֣ב מֵהֶ֔ם וַיִּנְחֲל֥וּ בְנֵֽי־ שִׁמְע֖וֹן בְּת֥וֹךְ נַחֲלָתָֽם׃ פ
10הַגּוֹרָ֣ל הַשְּׁלִישִׁ֔י לִבְנֵ֥י זְבוּלֻ֖ן לְמִשְׁפְּחֹתָ֑ם וַיְהִ֛י גְּב֥וּל נַחֲלָתָ֖ם עַד־ שָׂרִֽיד׃
11גְבוּלָ֧ם ׀ לַיָּ֛מָּה וּמַרְעֲלָ֖ה וּפָגַ֣ע בְּדַבָּ֑שֶׁת וּפָגַע֙ אֶל־ הַנַּ֔חַל אֲשֶׁ֖ר עַל־ פְּנֵ֥י יָקְנְעָֽם׃
12מִשָּׂרִ֗יד קֵ֚דְמָה מִזְרַ֣ח הַשֶּׁ֔מֶשׁ עַל־ גְּב֥וּל כִּסְלֹ֖ת תָּבֹ֑ר וְיָצָ֥א אֶל־ הַדָּֽבְרַ֖ת וְעָלָ֥ה יָפִֽיעַ׃
13עָבַר֙ קֵ֣דְמָה מִזְרָ֔חָה גִּתָּ֥ה חֵ֖פֶר עִתָּ֣ה קָצִ֑ין וְיָצָ֛א רִמּ֥וֹן הַמְּתֹאָ֖ר הַנֵּעָֽה׃
14אֹתוֹ֙ הַגְּב֔וּל מִצְּפ֖וֹן חַנָּתֹ֑ן וְהָיוּ֙ תֹּֽצְאֹתָ֔יו גֵּ֖י יִפְתַּח־ אֵֽל׃
15וְנַֽהֲלָל֙ וְשִׁמְר֔וֹן וְיִדְאֲלָ֖ה וּבֵ֣ית לָ֑חֶם עָרִ֥ים שְׁתֵּים־ עֶשְׂרֵ֖ה וְחַצְרֵיהֶֽן׃
16נַחֲלַ֥ת בְּנֵֽי־ זְבוּלֻ֖ן לְמִשְׁפְּחוֹתָ֑ם הֶֽעָרִ֥ים הָאֵ֖לֶּה וְחַצְרֵיהֶֽן׃ פ
17יָצָ֖א הַגּוֹרָ֣ל הָֽרְבִיעִ֑י לִבְנֵ֥י יִשָּׂשכָ֖ר לְמִשְׁפְּחוֹתָֽם׃
18גְּבוּלָ֑ם יִזְרְעֶ֥אלָה וְהַכְּסוּלֹ֖ת וְשׁוּנֵֽם׃
19וְשִׁיאֹ֖ן וַאֲנָחֲרַֽת׃
20וְקִשְׁי֖וֹן וָאָֽבֶץ׃
21וְעֵין־ גַּנִּ֛ים וְעֵ֥ין חַדָּ֖ה וּבֵ֥ית פַּצֵּֽץ׃
22הַגְּב֨וּל בְּתָב֤וֹר [וְשַׁחֲצוּמָה כ] (וְשַׁחֲצִ֙ימָה֙ ק) וּבֵ֣ית שֶׁ֔מֶשׁ וְהָי֛וּ תֹּצְא֥וֹת גְּבוּלָ֖ם הַיַּרְדֵּ֑ן עָרִ֥ים שֵׁשׁ־ עֶשְׂרֵ֖ה וְחַצְרֵיהֶֽן׃
23נַחֲלַ֛ת מַטֵּ֥ה בְנֵֽי־ יִשָּׂשכָ֖ר לְמִשְׁפְּחֹתָ֑ם הֶעָרִ֖ים וְחַצְרֵיהֶֽן׃ פ
24הַגּוֹרָ֣ל הַֽחֲמִישִׁ֔י לְמַטֵּ֥ה בְנֵֽי־ אָשֵׁ֖ר לְמִשְׁפְּחוֹתָֽם׃
25גְּבוּלָ֑ם חֶלְקַ֥ת וַחֲלִ֖י וָבֶ֥טֶן וְאַכְשָֽׁף׃
26וְעַמְעָ֖ד וּמִשְׁאָ֑ל וּפָגַ֤ע בְּכַרְמֶל֙ הַיָּ֔מָּה וּבְשִׁיח֖וֹר לִבְנָֽת׃
27מִזְרַ֣ח הַשֶּׁמֶשׁ֮ בֵּ֣ית דָּגֹן֒ וּפָגַ֣ע בִּ֠זְבֻלוּן וּבְגֵ֨י יִפְתַּח־ אֵ֥ל צָפ֛וֹנָה בֵּ֥ית הָעֵ֖מֶק וּנְעִיאֵ֑ל וְיָצָ֥א אֶל־ כָּב֖וּל מִשְּׂמֹֽאל׃
28וּרְחֹ֖ב וְחַמּ֣וֹן וְקָנָ֑ה עַ֖ד צִיד֥וֹן רַבָּֽה׃
29הַגְּבוּל֙ הָֽרָמָ֔ה וְעַד־ עִ֖יר מִבְצַר־ צֹ֑ר וְשָׁ֤ב הַגְּבוּל֙ חֹסָ֔ה [וְיִהְיוּ כ] (וְהָי֧וּ ק) תֹצְאֹתָ֛יו הַיָּ֖מָּה מֵחֶ֥בֶל אַכְזִֽיבָה׃
30וַאֲפֵ֖ק וּרְחֹ֑ב עָרִ֛ים עֶשְׂרִ֥ים וּשְׁתַּ֖יִם וְחַצְרֵיהֶֽן׃
31נַחֲלַ֛ת מַטֵּ֥ה בְנֵֽי־ אָשֵׁ֖ר לְמִשְׁפְּחֹתָ֑ם הֶעָרִ֥ים הָאֵ֖לֶּה וְחַצְרֵיהֶֽן׃ פ
32נַפְתָּלִ֔י יָצָ֖א הַגּוֹרָ֣ל הַשִּׁשִּׁ֑י לִבְנֵ֥י נַפְתָּלִ֖י לְמִשְׁפְּחֹתָֽם׃
33גְבוּלָ֗ם מֵחֵ֨לֶף מֵֽאֵל֜וֹן בְּצַעֲנַנִּ֗ים וַאֲדָמִ֥י הַנֶּ֛קֶב וְיַבְנְאֵ֖ל עַד־ לַקּ֑וּם וַיְהִ֥י תֹצְאֹתָ֖יו הַיַּרְדֵּֽן׃
34הַגְּב֥וּל יָ֙מָּה֙ אַזְנ֣וֹת תָּב֔וֹר וְיָצָ֥א מִשָּׁ֖ם חוּקֹ֑קָה וּפָגַ֨ע בִּזְבֻל֜וּן מִנֶּ֗גֶב וּבְאָשֵׁר֙ פָּגַ֣ע מִיָּ֔ם וּבִ֣יהוּדָ֔ה הַיַּרְדֵּ֖ן מִזְרַ֥ח הַשָּֽׁמֶשׁ׃
35מִבְצָ֑ר הַצִּדִּ֣ים צֵ֔ר וְחַמַּ֖ת רַקַּ֥ת וְכִנָּֽרֶת׃
36וְהָרָמָ֖ה וְחָצֽוֹר׃
37וְאֶדְרֶ֖עִי וְעֵ֥ין חָצֽוֹר׃
38וּמִגְדַּל־ אֵ֔ל חֳרֵ֥ם וּבֵית־ עֲנָ֖ת וּבֵ֣ית שָׁ֑מֶשׁ עָרִ֥ים תְּשַֽׁע־ עֶשְׂרֵ֖ה וְחַצְרֵיהֶֽן׃
39נַחֲלַ֛ת מַטֵּ֥ה בְנֵֽי־ נַפְתָּלִ֖י לְמִשְׁפְּחֹתָ֑ם הֶעָרִ֖ים וְחַצְרֵיהֶֽן׃ פ
40בְנֵי־ דָ֖ן לְמִשְׁפְּחֹתָ֑ם יָצָ֖א הַגּוֹרָ֥ל הַשְּׁבִיעִֽי׃
41גְּב֣וּל נַחֲלָתָ֑ם צָרְעָ֥ה וְאֶשְׁתָּא֖וֹל וְעִ֥יר שָֽׁמֶשׁ׃
42וְאַיָּל֖וֹן וְיִתְלָֽה׃
43וְתִמְנָ֖תָה וְעֶקְרֽוֹן׃
44וְגִבְּת֖וֹן וּבַעֲלָֽת׃
45וּבְנֵֽי־ בְרַ֖ק וְגַת־ רִמּֽוֹן׃
46הַיַּרְק֖וֹן וְהָֽרַקּ֑וֹן עִֽם־ הַגְּב֖וּל מ֥וּל יָפֽוֹ׃
47גְבוּל־ בְּנֵי־ דָ֖ן מֵהֶ֑ם וַיַּעֲל֣וּ בְנֵֽי־ דָ֠ן וַיִּלָּחֲמ֨וּ עִם־ לֶ֜שֶׁם וַיִּלְכְּד֥וּ אוֹתָ֣הּ ׀ וַיַּכּ֧וּ אוֹתָ֣הּ לְפִי־ חֶ֗רֶב וַיִּֽרְשׁ֤וּ אוֹתָהּ֙ וַיֵּ֣שְׁבוּ בָ֔הּ וַיִּקְרְא֤וּ לְלֶ֙שֶׁם֙ דָּ֔ן כְּשֵׁ֖ם דָּ֥ן אֲבִיהֶֽם׃
48נַחֲלַ֛ת מַטֵּ֥ה בְנֵי־ דָ֖ן לְמִשְׁפְּחֹתָ֑ם הֶֽעָרִ֥ים הָאֵ֖לֶּה וְחַצְרֵיהֶֽן׃ פ
49לִנְחֹל־ אֶת־ הָאָ֖רֶץ לִגְבֽוּלֹתֶ֑יהָ וַיִּתְּנ֨וּ בְנֵי־ יִשְׂרָאֵ֧ל נַחֲלָ֛ה לִיהוֹשֻׁ֥עַ בִּן־ נ֖וּן בְּתוֹכָֽם׃
50פִּ֨י יְהוָ֜ה נָ֣תְנוּ ל֗וֹ אֶת־ הָעִיר֙ אֲשֶׁ֣ר שָׁאָ֔ל אֶת־ תִּמְנַת־ סֶ֖רַח בְּהַ֣ר אֶפְרָ֑יִם וַיִּבְנֶ֥ה אֶת־ הָעִ֖יר וַיֵּ֥שֶׁב בָּֽהּ׃
