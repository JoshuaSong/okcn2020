1מַדּ֗וּעַ מִ֭שַּׁדַּי לֹא־ נִצְפְּנ֣וּ עִתִּ֑ים [וְיֹדְעֹו כ] (וְ֝יֹדְעָ֗יו ק) לֹא־ חָ֥זוּ יָמָֽיו׃
2יַשִּׂ֑יגוּ עֵ֥דֶר גָּ֝זְל֗וּ וַיִּרְעֽוּ׃
3יְתוֹמִ֣ים יִנְהָ֑גוּ יַ֝חְבְּל֗וּ שׁ֣וֹר אַלְמָנָֽה׃
4אֶבְיוֹנִ֣ים מִדָּ֑רֶךְ יַ֥חַד חֻ֝בְּא֗וּ עֲנִיֵּי־ אָֽרֶץ׃
5פְּרָאִ֨ים ׀ בַּֽמִּדְבָּ֗ר יָצְא֣וּ בְּ֭פָעֳלָם מְשַׁחֲרֵ֣י לַטָּ֑רֶף עֲרָבָ֥ה ל֥וֹ לֶ֝֗חֶם לַנְּעָרִֽים׃
6בְּלִיל֣וֹ [יַקְצִירוּ כ] (יִקְצֹ֑ורוּ ק) וְכֶ֖רֶם רָשָׁ֣ע יְלַקֵּֽשׁוּ׃
7יָ֭לִינוּ מִבְּלִ֣י לְב֑וּשׁ וְאֵ֥ין כְּ֝ס֗וּת בַּקָּרָֽה׃
8הָרִ֣ים יִרְטָ֑בוּ וּֽמִבְּלִ֥י מַ֝חְסֶ֗ה חִבְּקוּ־ צֽוּר׃
9מִשֹּׁ֣ד יָת֑וֹם וְֽעַל־ עָנִ֥י יַחְבֹּֽלוּ׃
10הִ֭לְּכוּ בְּלִ֣י לְב֑וּשׁ וּ֝רְעֵבִ֗ים נָ֣שְׂאוּ עֹֽמֶר׃
11שׁוּרֹתָ֥ם יַצְהִ֑ירוּ יְקָבִ֥ים דָּ֝רְכ֗וּ וַיִּצְמָֽאוּ׃
12מְתִ֨ים ׀ יִנְאָ֗קוּ וְנֶֽפֶשׁ־ חֲלָלִ֥ים תְּשַׁוֵּ֑עַ וֶ֝אֱל֗וֹהַּ לֹא־ יָשִׂ֥ים תִּפְלָֽה׃
13הָיוּ֮ בְּֽמֹרְדֵ֫י־ א֥וֹר לֹֽא־ הִכִּ֥ירוּ דְרָכָ֑יו וְלֹ֥א יָ֝שְׁב֗וּ בִּנְתִיבֹתָֽיו׃
14יָ֘ק֤וּם רוֹצֵ֗חַ יִֽקְטָל־ עָנִ֥י וְאֶבְי֑וֹן וּ֝בַלַּ֗יְלָה יְהִ֣י כַגַּנָּֽב׃
15נֹאֵ֨ף ׀ שָׁ֤מְרָֽה נֶ֣שֶׁף לֵ֭אמֹר לֹא־ תְשׁוּרֵ֣נִי עָ֑יִן וְסֵ֖תֶר פָּנִ֣ים יָשִֽׂים׃
16בַּחֹ֗שֶׁךְ בָּ֫תִּ֥ים יוֹמָ֥ם חִתְּמוּ־ לָ֗מוֹ לֹא־ יָ֥דְעוּ אֽוֹר׃
17יַחְדָּ֨ו ׀ בֹּ֣קֶר לָ֣מוֹ צַלְמָ֑וֶת כִּֽי־ יַ֝כִּ֗יר בַּלְה֥וֹת צַלְמָֽוֶת׃
18ה֤וּא ׀ עַל־ פְּנֵי־ מַ֗יִם תְּקֻלַּ֣ל חֶלְקָתָ֣ם בָּאָ֑רֶץ לֹֽא־ יִ֝פְנֶה דֶּ֣רֶךְ כְּרָמִֽים׃
19גַם־ חֹ֗ם יִגְזְל֥וּ מֵֽימֵי־ שֶׁ֗לֶג שְׁא֣וֹל חָטָֽאוּ׃
20רֶ֨חֶם ׀ מְתָ֘ק֤וֹ רִמָּ֗ה ע֥וֹד לֹֽא־ יִזָּכֵ֑ר וַתִּשָּׁבֵ֖ר כָּעֵ֣ץ עַוְלָֽה׃
21עֲ֭קָרָה לֹ֣א תֵלֵ֑ד וְ֝אַלְמָנָ֗ה לֹ֣א יְיֵטִֽיב׃
22אַבִּירִ֣ים בְּכֹח֑וֹ יָ֝ק֗וּם וְֽלֹא־ יַאֲמִ֥ין בַּֽחַיִּֽין׃
23ל֣וֹ לָ֭בֶטַח וְיִשָּׁעֵ֑ן וְ֝עֵינֵ֗יהוּ עַל־ דַּרְכֵיהֶֽם׃
24מְּעַ֨ט ׀ וְֽאֵינֶ֗נּוּ וְֽהֻמְּכ֗וּ כַּכֹּ֥ל יִקָּפְצ֑וּן וּכְרֹ֖אשׁ שִׁבֹּ֣לֶת יִמָּֽלוּ׃
