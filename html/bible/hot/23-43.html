1וְעַתָּ֞ה כֹּֽה־ אָמַ֤ר יְהוָה֙ בֹּרַאֲךָ֣ יַעֲקֹ֔ב וְיֹצֶרְךָ֖ יִשְׂרָאֵ֑ל אַל־ תִּירָא֙ כִּ֣י גְאַלְתִּ֔יךָ קָרָ֥אתִי בְשִׁמְךָ֖ לִי־ אָֽתָּה׃
2תַעֲבֹ֤ר בַּמַּ֙יִם֙ אִתְּךָ־ אָ֔נִי וּבַנְּהָר֖וֹת לֹ֣א יִשְׁטְפ֑וּךָ כִּֽי־ תֵלֵ֤ךְ בְּמוֹ־ אֵשׁ֙ לֹ֣א תִכָּוֶ֔ה וְלֶהָבָ֖ה לֹ֥א תִבְעַר־ בָּֽךְ׃
3אֲנִי֙ יְהוָ֣ה אֱלֹהֶ֔יךָ קְד֥וֹשׁ יִשְׂרָאֵ֖ל מוֹשִׁיעֶ֑ךָ נָתַ֤תִּי כָפְרְךָ֙ מִצְרַ֔יִם כּ֥וּשׁ וּסְבָ֖א תַּחְתֶּֽיךָ׃
4יָקַ֧רְתָּ בְעֵינַ֛י נִכְבַּ֖דְתָּ וַאֲנִ֣י אֲהַבְתִּ֑יךָ וְאֶתֵּ֤ן אָדָם֙ תַּחְתֶּ֔יךָ וּלְאֻמִּ֖ים תַּ֥חַת נַפְשֶֽׁךָ׃
5תִּירָ֖א כִּ֣י אִתְּךָ־ אָ֑נִי מִמִּזְרָח֙ אָבִ֣יא זַרְעֶ֔ךָ וּמִֽמַּעֲרָ֖ב אֲקַבְּצֶֽךָּ׃
6לַצָּפוֹן֙ תֵּ֔נִי וּלְתֵימָ֖ן אַל־ תִּכְלָ֑אִי הָבִ֤יאִי בָנַי֙ מֵרָח֔וֹק וּבְנוֹתַ֖י מִקְצֵ֥ה הָאָֽרֶץ׃
7הַנִּקְרָ֣א בִשְׁמִ֔י וְלִכְבוֹדִ֖י בְּרָאתִ֑יו יְצַרְתִּ֖יו אַף־ עֲשִׂיתִֽיו׃
8עַם־ עִוֵּ֖ר וְעֵינַ֣יִם יֵ֑שׁ וְחֵרְשִׁ֖ים וְאָזְנַ֥יִם לָֽמוֹ׃
9הַגּוֹיִ֞ם נִקְבְּצ֣וּ יַחְדָּ֗ו וְיֵאָֽסְפוּ֙ לְאֻמִּ֔ים מִ֤י בָהֶם֙ יַגִּ֣יד זֹ֔את וְרִֽאשֹׁנ֖וֹת יַשְׁמִיעֻ֑נוּ יִתְּנ֤וּ עֵֽדֵיהֶם֙ וְיִצְדָּ֔קוּ וְיִשְׁמְע֖וּ וְיֹאמְר֥וּ אֱמֶֽת׃
10עֵדַי֙ נְאֻם־ יְהוָ֔ה וְעַבְדִּ֖י אֲשֶׁ֣ר בָּחָ֑רְתִּי לְמַ֣עַן תֵּ֠דְעוּ וְתַאֲמִ֨ינוּ לִ֤י וְתָבִ֙ינוּ֙ כִּֽי־ אֲנִ֣י ה֔וּא לְפָנַי֙ לֹא־ נ֣וֹצַר אֵ֔ל וְאַחֲרַ֖י לֹ֥א יִהְיֶֽה׃ ס
11אָנֹכִ֖י יְהוָ֑ה וְאֵ֥ין מִבַּלְעָדַ֖י מוֹשִֽׁיעַ׃
12הִגַּ֤דְתִּי וְהוֹשַׁ֙עְתִּי֙ וְהִשְׁמַ֔עְתִּי וְאֵ֥ין בָּכֶ֖ם זָ֑ר וְאַתֶּ֥ם עֵדַ֛י נְאֻם־ יְהוָ֖ה וַֽאֲנִי־ אֵֽל׃
13מִיּוֹם֙ אֲנִ֣י ה֔וּא וְאֵ֥ין מִיָּדִ֖י מַצִּ֑יל אֶפְעַ֖ל וּמִ֥י יְשִׁיבֶֽנָּה׃ ס
14אָמַ֧ר יְהוָ֛ה גֹּאַלְכֶ֖ם קְד֣וֹשׁ יִשְׂרָאֵ֑ל לְמַעַנְכֶ֞ם שִׁלַּ֣חְתִּי בָבֶ֗לָה וְהוֹרַדְתִּ֤י בָֽרִיחִים֙ כֻּלָּ֔ם וְכַשְׂדִּ֖ים בָּאֳנִיּ֥וֹת רִנָּתָֽם׃
15יְהוָ֖ה קְדֽוֹשְׁכֶ֑ם בּוֹרֵ֥א יִשְׂרָאֵ֖ל מַלְכְּכֶֽם׃ ס
16אָמַ֣ר יְהוָ֔ה הַנּוֹתֵ֥ן בַּיָּ֖ם דָּ֑רֶךְ וּבְמַ֥יִם עַזִּ֖ים נְתִיבָֽה׃
17רֶֽכֶב־ וָס֖וּס חַ֣יִל וְעִזּ֑וּז יַחְדָּ֤ו יִשְׁכְּבוּ֙ בַּל־ יָק֔וּמוּ דָּעֲכ֖וּ כַּפִּשְׁתָּ֥ה כָבֽוּ׃
18תִּזְכְּר֖וּ רִֽאשֹׁנ֑וֹת וְקַדְמֹנִיּ֖וֹת אַל־ תִּתְבֹּנָֽנוּ׃
19עֹשֶׂ֤ה חֲדָשָׁה֙ עַתָּ֣ה תִצְמָ֔ח הֲל֖וֹא תֵֽדָע֑וּהָ אַ֣ף אָשִׂ֤ים בַּמִּדְבָּר֙ דֶּ֔רֶךְ בִּֽישִׁמ֖וֹן נְהָרֽוֹת׃
20חַיַּ֣ת הַשָּׂדֶ֔ה תַּנִּ֖ים וּבְנ֣וֹת יַֽעֲנָ֑ה כִּֽי־ נָתַ֨תִּי בַמִּדְבָּ֜ר מַ֗יִם נְהָרוֹת֙ בִּֽישִׁימֹ֔ן לְהַשְׁק֖וֹת עַמִּ֥י בְחִירִֽי׃
21זוּ֙ יָצַ֣רְתִּי לִ֔י תְּהִלָּתִ֖י יְסַפֵּֽרוּ׃ ס
22אֹתִ֥י קָרָ֖אתָ יַֽעֲקֹ֑ב כִּֽי־ יָגַ֥עְתָּ בִּ֖י יִשְׂרָאֵֽל׃
23הֵבֵ֤יאתָ לִּי֙ שֵׂ֣ה עֹלֹתֶ֔יךָ וּזְבָחֶ֖יךָ לֹ֣א כִבַּדְתָּ֑נִי לֹ֤א הֶעֱבַדְתִּ֙יךָ֙ בְּמִנְחָ֔ה וְלֹ֥א הוֹגַעְתִּ֖יךָ בִּלְבוֹנָֽה׃
24קָנִ֨יתָ לִּ֤י בַכֶּ֙סֶף֙ קָנֶ֔ה וְחֵ֥לֶב זְבָחֶ֖יךָ לֹ֣א הִרְוִיתָ֑נִי אַ֗ךְ הֶעֱבַדְתַּ֙נִי֙ בְּחַטֹּאותֶ֔יךָ הוֹגַעְתַּ֖נִי בַּעֲוֹנֹתֶֽיךָ׃ ס
25אָנֹכִ֥י ה֛וּא מֹחֶ֥ה פְשָׁעֶ֖יךָ לְמַעֲנִ֑י וְחַטֹּאתֶ֖יךָ לֹ֥א אֶזְכֹּֽר׃
26נִשָּׁפְטָ֖ה יָ֑חַד סַפֵּ֥ר אַתָּ֖ה לְמַ֥עַן תִּצְדָּֽק׃
27הָרִאשׁ֖וֹן חָטָ֑א וּמְלִיצֶ֖יךָ פָּ֥שְׁעוּ בִֽי׃
