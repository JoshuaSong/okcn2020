1חָ֭כְמוֹת בָּנְתָ֣ה בֵיתָ֑הּ חָצְבָ֖ה עַמּוּדֶ֣יהָ שִׁבְעָֽה׃
2טִ֭בְחָהּ מָסְכָ֣ה יֵינָ֑הּ אַ֝֗ף עָֽרְכָ֥ה שֻׁלְחָנָֽהּ׃
3נַעֲרֹתֶ֣יהָ תִקְרָ֑א עַל־ גַּ֝פֵּ֗י מְרֹ֣מֵי קָֽרֶת׃
4פֶ֭תִי יָסֻ֣ר הֵ֑נָּה חֲסַר־ לֵ֝֗ב אָ֣מְרָה לּֽוֹ׃
5לַחֲמ֣וּ בְֽלַחֲמִ֑י וּ֝שְׁת֗וּ בְּיַ֣יִן מָסָֽכְתִּי׃
6פְתָאיִ֣ם וִֽחְי֑וּ וְ֝אִשְׁר֗וּ בְּדֶ֣רֶךְ בִּינָֽה׃
7לֵ֗ץ לֹקֵ֣חַֽ ל֣וֹ קָל֑וֹן וּמוֹכִ֖יחַ לְרָשָׁ֣ע מוּמֽוֹ׃
8תּ֣וֹכַח לֵ֭ץ פֶּן־ יִשְׂנָאֶ֑ךָּ הוֹכַ֥ח לְ֝חָכָ֗ם וְיֶאֱהָבֶֽךָּ׃
9לְ֭חָכָם וְיֶחְכַּם־ ע֑וֹד הוֹדַ֥ע לְ֝צַדִּ֗יק וְי֣וֹסֶף לֶֽקַח׃ פ
10חָ֭כְמָה יִרְאַ֣ת יְהוָ֑ה וְדַ֖עַת קְדֹשִׁ֣ים בִּינָֽה׃
11בִ֭י יִרְבּ֣וּ יָמֶ֑יךָ וְיוֹסִ֥יפוּ לְּ֝ךָ֗ שְׁנ֣וֹת חַיִּֽים׃
12חָ֭כַמְתָּ חָכַ֣מְתָּ לָּ֑ךְ וְ֝לַ֗צְתָּ לְֽבַדְּךָ֥ תִשָּֽׂא׃
13כְּ֭סִילוּת הֹֽמִיָּ֑ה פְּ֝תַיּ֗וּת וּבַל־ יָ֥דְעָה מָּֽה׃
14לְפֶ֣תַח בֵּיתָ֑הּ עַל־ כִּ֝סֵּ֗א מְרֹ֣מֵי קָֽרֶת׃
15לְעֹֽבְרֵי־ דָ֑רֶךְ הַֽ֝מְיַשְּׁרִ֗ים אֹֽרְחוֹתָֽם׃
16פֶ֭תִי יָסֻ֣ר הֵ֑נָּה וַחֲסַר־ לֵ֝֗ב וְאָ֣מְרָה לּֽוֹ׃
17גְּנוּבִ֥ים יִמְתָּ֑קוּ וְלֶ֖חֶם סְתָרִ֣ים יִנְעָֽם׃
