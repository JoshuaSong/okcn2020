1 לַמְנַצֵּחַ מַשְׂכִּיל לְדָוִד׃ בְּבֹוא דֹּואֵג הָאֲדֹמִי וַיַּגֵּד לְשָׁאוּל וַיֹּאמֶר לֹו בָּא דָוִד אֶל־בֵּית אֲחִימֶלֶךְ׃ מַה־תִּתְהַלֵּל בְּרָעָה הַגִּבֹּור חֶסֶד אֵל כָּל־הַיֹּום׃

2 הַוֹּות תַּחְשֹׁב לְשֹׁונֶךָ כְּתַעַר מְלֻטָּשׁ עֹשֵׂה רְמִיָּה׃

3 אָהַבְתָּ רָּע מִטֹּוב שֶׁקֶר מִדַּבֵּר צֶדֶק סֶלָה׃

4 אָהַבְתָּ כָל־דִּבְרֵי־בָלַע לְשֹׁון מִרְמָה׃

5 גַּם־אֵל יִתָּצְךָ לָנֶצַח יַחְתְּךָ וְיִסָּחֲךָ מֵאֹהֶל וְשֵׁרֶשְׁךָ מֵאֶרֶץ חַיִּים סֶלָה׃

6 וְיִרְאוּ צַדִּיקִים וְיִירָאוּ וְעָלָיו יִשְׂחָקוּ׃

7 הִנֵּה הַגֶּבֶר לֹא יָשִׂים אֱלֹהִים מָעוּזֹּו וַיִּבְטַח בְּרֹב עָשְׁרֹו יָעֹז בְּהַוָּתֹו׃

8 וַאֲנִי כְּזַיִת רַעֲןָן בְּבֵית אֱלֹהִים בָּטַחְתִּי בְחֶסֶד־אֱלֹהִים עֹולָם וָעֶד׃

9 אֹודְךָ לְעֹולָם כִּי עָשִׂיתָ וַאֲקַוֶּה שִׁמְךָ כִי־טֹוב נֶגֶד חֲסִידֶיךָ׃
