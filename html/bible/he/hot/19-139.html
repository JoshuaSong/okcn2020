1 לַמְנַצֵּחַ לְדָוִד מִזְמֹור יְהוָה חֲקַרְתַּנִי וַתֵּדָע׃

2 אַתָּה יָדַעְתָּ שִׁבְתִּי וְקוּמִי בַּנְתָּה לְרֵעִי מֵרָחֹוק׃

3 אָרְחִי וְרִבְעִי זֵרִיתָ וְכָל־דְּרָכַי הִסְכַּנְתָּה׃

4 כִּי אֵין מִלָּה בִּלְשֹׁונִי הֵן יְהוָה יָדַעְתָּ כֻלָּהּ׃

5 אָחֹור וָקֶדֶם צַרְתָּנִי וַתָּשֶׁת עָלַי כַּפֶּכָה׃

6 [כ= פִּלְאִיָּה] [ק= פְּלִיאָה] דַעַת מִמֶּנִּי נִשְׂגְּבָה לֹא־אוּכַל לָהּ׃

7 אָנָה אֵלֵךְ מֵרוּחֶךָ וְאָנָה מִפָּנֶיךָ אֶבְרָח׃

8 אִם־אֶסַּק שָׁמַיִם שָׁם אָתָּה וְאַצִּיעָה שְּׁאֹול הִנֶּךָּ׃

9 אֶשָּׂא כַנְפֵי־שָׁחַר אֶשְׁכְּנָה בְּאַחֲרִית יָם׃

10 גַּם־שָׁם יָדְךָ תַנְחֵנִי וְתֹאחֲזֵנִי יְמִינֶךָ׃

11 וָאֹמַר אַךְ־חֹשֶׁךְ יְשׁוּפֵנִי וְלַיְלָה אֹור בַּעֲדֵנִי׃

12 גַּם־חֹשֶׁךְ לֹא־יַחְשִׁיךְ מִמֶּךָ וְלַיְלָה כַּיֹּום יָאִיר כַּחֲשֵׁיכָה כָּאֹורָה׃

13 כִּי־אַתָּה קָנִיתָ כִלְיֹתָי תְּסֻכֵּנִי בְּבֶטֶן אִמִּי׃

14 אֹודְךָ עַל כִּי נֹורָאֹות נִפְלֵיתִי נִפְלָאִים מַעֲשֶׂיךָ וְנַפְשִׁי יֹדַעַת מְאֹד׃

15 לֹא־נִכְחַד עָצְמִי מִמֶּךָּ אֲשֶׁר־עֻשֵּׂיתִי בַסֵּתֶר רֻקַּמְתִּי בְּתַחְתִּיֹּות אָרֶץ׃

16 גָּלְמִי רָאוּ עֵינֶיךָ וְעַל־סִפְרְךָ כֻּלָּם יִכָּתֵבוּ יָמִים יֻצָּרוּ [כ= וְלֹא] [ק= וְלֹו] אֶחָד בָּהֶם׃

17 וְלִי מַה־יָּקְרוּ רֵעֶיךָ אֵל מֶה עָצְמוּ רָאשֵׁיהֶם׃

18 אֶסְפְּרֵם מֵחֹול יִרְבּוּן הֱקִיצֹתִי וְעֹודִי עִמָּךְ׃

19 אִם־תִּקְטֹל אֱלֹוהַּ רָשָׁע וְאַנְשֵׁי דָמִים סוּרוּ מֶנִּי׃

20 אֲשֶׁר יֹאמְרֻךָ לִמְזִמָּה נָשֻׂא לַשָּׁוְא עָרֶיךָ׃

21 הֲלֹוא־מְשַׂנְאֶיךָ יְהוָה אֶשְׂנָא וּבִתְקֹומְמֶיךָ אֶתְקֹוטָט׃

22 תַּכְלִית שִׂנְאָה שְׂנֵאתִים לְאֹויְבִים הָיוּ לִי׃

23 חָקְרֵנִי אֵל וְדַע לְבָבִי בְּחָנֵנִי וְדַע שַׂרְעַפָּי׃

24 וּרְאֵה אִם־דֶּרֶךְ־עֹצֶב בִּי וּנְחֵנִי בְּדֶרֶךְ עֹולָם׃
